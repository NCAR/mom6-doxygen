96subroutine initialize_amsua_netcdf(filename, verbose)
97
98character(len=*), intent(in) :: filename
99integer, intent(in) :: verbose
100

148!------------------------------------------------------------------------------
149!> Convert character representation of desired channels to the channel index
150!> The desired input can be either an alphanumeric or numeric representation.
151!>
152!> from the documentation
153!> A2-1, A2-2 | A1-1, A1-2, A1-3, A1-4, A1-5 A1-6, A1-7 ... A1-13
154!> 23.8, 31.4 | 50.3, 52.8, 53.596, 54.4, 54.940, 55.5, 57.29034, 89
155
156subroutine channel_list_to_indices(channel_list,use_channels)
157
158character(len=*), intent(in) :: channel_list(:)
159logical, intent(out) :: use_channels(:)
160
161! from the dump of the converted netCDF file (chs 9-14 ?)
162! center_freq = 23.8, 31.4, 50.3, 52.8, 53.596, ...
163! 54.4, 54.94, 55.5, 57.29034, 57.29034, ...
164! 57.29034, 57.29034, 57.29034, 57.29034, 89 ;
165
166integer :: ichannel
167
168if ( .not. module_initialized ) call initialize_module
169
170use_channels = .false.
171
172channels: do ichannel = 1,size(channel_list)
173 select case (channel_list(ichannel))
174 case('1','A2-1','23.8')
175 use_channels(1) = .true.
176 case('2','A2-2','31.4')
177 use_channels(2) = .true.
178 case('3','A1-1','50.3')
179 use_channels(3) = .true.
180 case('4','A1-2','52.8')
181 use_channels(4) = .true.
182 case('5','A1-3','53.596')
183 use_channels(5) = .true.
184 case('6','A1-4','54.4')
185 use_channels(6) = .true.
186 case('7','A1-5','54.94')
187 use_channels(7) = .true.
188 case('8','A1-6','55.5')
189 use_channels(8) = .true.
190 case('9','A1-7')
191 use_channels(9) = .true.
192 case('10','A1-8')
193 use_channels(10) = .true.
194 case('11','A1-9')
195 use_channels(11) = .true.
196 case('12','A1-10')
197 use_channels(12) = .true.
198 case('13','A1-11')
199 use_channels(13) = .true.
200 case('14','A1-12')
201 use_channels(14) = .true.
202 case('15','A1-13','89')
203 use_channels(15) = .true.
204 case ('null')
205 exit channels
206 case default
207 write(string1,*)'<'//trim(channel_list(ichannel))//'> is not a valid channel declaration'
208 call error_handler(e_err,'channel_list_to_indices','unknown channel input', &
209 source,text2=trim(channel_list(ichannel)))
210 end select
211enddo channels
212
213if (all(use_channels .eqv. .false.)) then
214 call error_handler(e_err,'channel_list_to_indices','no valid input channels',source)
215endif
216
217! FIXME ... print a summary of the channels and frequences being used
218
219end subroutine channel_list_to_indices

669!> routine to extract required parts of an AMSUA granule needed for a

671
672subroutine add_granule_observations(seq, lon1, lon2, lat1, lat2, use_channels, &
673 scan_thin, xtrack_thin, obs, prev_obs, is_first_obs, &
674 pre_time, swath)
675
676type(obs_sequence_type), intent(inout) :: seq
677real(r8), intent(in) :: lon1, lon2, lat1, lat2
678logical, intent(in) :: use_channels(:)
679integer, intent(in) :: scan_thin, xtrack_thin
680type(obs_type), intent(inout) :: obs, prev_obs
681logical, intent(inout) :: is_first_obs
682type(time_type), intent(inout) :: pre_time
683type(amsua_bt_granule), intent(in) :: swath
684
685integer :: ipix, iscan, ichan
686integer :: days, seconds
687integer :: obs_num, key
688integer :: which_vert
689
690real(r8) :: olon, olat, vloc
691real(r8) :: obs_value, obs_err
692real(r8) :: rqc
693
694real(r8) :: lam1, lam2, phi1, phi2, x, y
695
696real(r8) :: sat_az, sat_ze
697integer :: platform_id, sat_id, sensor_id
698real(r8) :: mag_field, cosbk
699real(r8) :: fastem_p1
700real(r8) :: fastem_p2
701real(r8) :: fastem_p3
702real(r8) :: fastem_p4
703real(r8) :: fastem_p5
704
705type(time_type) :: obs_time, time_base
706character(len=*), parameter :: routine = 'add_granule_observations:'
707
708integer :: robstype
709
710real(digits12) :: time_offset(size(swath%Time,1),size(swath%Time,2))
711
712! things known (and constant) about the input data and rttov
713! consistent with values in rttov_sensor_db.csv
714
715platform_id = 9 ! EOS
716sat_id = 2 ! satellite
717sensor_id = 3 ! AMSUA
718
719!>@todo FIXME not happy with the time - seems like the swath%start_Time
720! and the time from the data are off by 10 seconds. leaving this
721! code commented out for more exploration later.
722
723!TJH print*,'swath%start_Time ',swath%start_Time
724!TJH print*,'min from data ',minval(swath%Time)
725!TJH print*,'max from data ',maxval(swath%Time)
726!TJH print*,'swath%end_Time ',swath%end_Time
727!TJH
728!TJH obs_time = set_date(1993, 1, 1) + set_time(int(swath%start_Time))
729!TJH call print_date(obs_time,'start date from 1993')
730!TJH call print_time(obs_time,'start time from 1993')
731!TJH obs_time = set_date(1993, 1, 1) + set_time(int(swath%end_Time))
732!TJH call print_date(obs_time,'end date from 1993')
733!TJH call print_time(obs_time,'end time from 1993')
734
735! The global metadata has start year.month.day... and start_Time
736! The data (Time) have the same base as start_Time.
737! So I can convert the data time to an offset to get the number of
738! seconds since the start_Time
739
740time_base = set_date(swath%start_year, swath%start_month, swath%start_day, &
741 swath%start_hour, swath%start_minute, int(swath%start_sec))
742
743time_offset = swath%Time - swath%start_Time
744
745!TJH call print_date(time_base,'start date from global metadata')
746!TJH call print_time(time_base,'start time from global metadata')
747!TJH obs_time = time_base + set_time(int(minval(time_offset)), 0)
748!TJH call print_date(obs_time,'minimum date ')
749!TJH call print_time(obs_time,'minimum time ')
750!TJH obs_time = time_base + set_time(int(maxval(time_offset)), 0)
751!TJH call print_date(obs_time,'maximum date ')
752!TJH call print_time(obs_time,'maxiumu time ')
753
754!------------------------------------------------------------------------------
755! loop over all observations within the file
756
757obs_num = 0
758which_vert = vertisundef
759
760! assign each observation the correct observation type
761robstype = get_index_for_type_of_obs('EOS_2_AMSUA_TB')
762if (robstype < 1) then
763 string1 = 'unknown observation type EOS_2_AMSUA_TB'
764 call error_handler(e_err,routine,string1,source)
765endif
766
767! Latitude( AMSUA_BT_GEOXTRACK, AMSUA_BT_GEOTRACK)
768! Longitude( AMSUA_BT_GEOXTRACK, AMSUA_BT_GEOTRACK)
769! brightness_temp(AMSUA_BT_CHANNEL, AMSUA_BT_GEOXTRACK, AMSUA_BT_GEOTRACK)
770! rows are along-track, stepping in the direction the satellite is moving
771
772scanloop: do iscan=1,size(swath%Latitude,2) ! AKA GEOTRACK
773
774 ! if we're going to subset rows, we will cycle here
775 if (scan_thin > 0) then
776 if (modulo(iscan, scan_thin) /= 0) cycle scanloop
777 endif
778
779 ! Reject bad scans here, if desired.
780 ! "Bit field for each scanline (bit 0 set if sun glint in scanline;
781 ! bit 1 set if costal crossing in scanline, bit 2 set if some
782 ! channels had excessive NeDT estimated), dimension (45)"
783
784 rqc = swath%qa_scanline(iscan)
785 ! if (rqc /= 0) cycle scanloop ! REJECT IF DESIRED
786
787 ! columns are across-track, varying faster than rows.
788 pixloop: do ipix=1,size(swath%Latitude,1) ! AKA GEOXTRACK
789
790 ! if we're going to subset columns, ditto
791 if (xtrack_thin > 0) then
792 if (modulo(ipix, xtrack_thin) /= 0) cycle pixloop
793 endif
794
795 ! observation lat, lon:
796 olat = swath%Latitude (ipix,iscan) ! valid range [ -90.00, 90.00]
797 olon = swath%Longitude(ipix,iscan) ! valid range [-180.00, 180.00]
798
799 ! verify the location is not outside valid limits. AIRS uses -180/180
800 if((olon > 180.0_r8) .or. (olon < -180.0_r8) .or. &
801 (olat > 90.0_r8) .or. (olat < -90.0_r8)) then
802 write(*,*)'WARNING : invalid location. col,row,lon,lat = ', ipix,iscan,olon,olat
803 cycle pixloop
804 endif
805
806 ! make sure lon is between 0 and 360
807 if (olon < 0.0_r8) olon = olon + 360.0_r8
808
809 ! reject observations outside the bounding box (allowing wrapping)
810 if(( olat < lat1) .or. ( olat > lat2 ) .or. &
811 (.not. is_longitude_between(olon, lon1, lon2))) cycle pixloop
812
813 ! set the zenith angle (aka earth incidence angle)
814 sat_ze = swath%satzen(ipix,iscan)
815
816 lam1 = deg2rad*swath%longitude(ipix,iscan)
817 lam2 = deg2rad*swath%sat_lon(iscan)
818 phi1 = deg2rad*swath%latitude(ipix,iscan)
819 phi2 = deg2rad*swath%sat_lat(iscan)
820
821 ! calculate the bearing between the obs lat/lon and the SClat/lon
822 y = sin(lam2-lam1)*cos(phi2)
823 x = cos(phi1)*sin(phi2)-sin(phi1)*cos(phi2)*cos(lam2-lam1)
824 sat_az = rad2deg*atan2(y,x)
825
826 ! convert observation time to DART format
827 obs_time = time_base + set_time(int(time_offset(ipix,iscan)), 0)
828 call get_time(obs_time, seconds, days)
829
830 channel_loop: do ichan=1, size(use_channels)
831
832 if (.not. use_channels(ichan)) cycle channel_loop
833
834 ! create the radiance obs for this observation, add to sequence
835
836 ! apparently -9999 is missing data, outside of qc mechanism
837 ! FIXME ... there is a '_FillValue'
838 obs_value = swath%brightness_temp(ichan,ipix,iscan)
839 if (obs_value < 0.0_r8) cycle channel_loop
840
841 obs_err = swath%brightness_temp_err(ichan,ipix,iscan)
842
843 ! column integrated value, so no vertical location
844 vloc = 0.0_r8
845
846 if (get_rttov_option_logical('use_zeeman')) then
847
848 ! From RTTOV v12 Users Guide Section 8.12: "For microwave sensors
849 ! that have high peaking weighting functions in the mesosphere ...,
850 ! channels close to lines of molecular oxygen may be significantly
851 ! affected by the redistribution of line intensity through Zeeman
852 ! splitting as described in the RTTOV v10 Science and Validation Report.
853 ! The absorption for the affected channels will depend on the strength
854 ! and orientation of the magnetic field. You must specify two input
855 ! variables for the geomagnetic field in the rttov_profiles structure,
856 ! these being the magnitude, Be, of the field and the cosine, cosbk,
857 ! of the angle between the field vector and the viewing path considered.
858 ! ...
859 ! For AMSU-A, only channel 14 is affected. This channel ... impact is
860 ! therefore much smaller (~0.5K). If a Zeeman coefficient file is used,
861 ! then a small set of additional predictors will be included. These will
862 ! contribute for channel 14 but will be nullified for the other channels
863 ! by zero coefficients."
864 !
865 ! From RTTOV v12 Users Guide Table 12 (p 46):
866 ! profiles(i)%Be is the Earth magnetic field strength in Gauss
867 ! profiles(i)%cosbk is the cosine of the angle between the Earth magnetic
868 ! field and wave propagation direction
869
870 write(string1,*) 'Compensating for the Zeeman effect is not supported for AMSU-A observations.'
871 write(string2,*) 'See https://github.com/NCAR/DART/issues/99#'
872 write(string3,*) 'Also Section 8.12 of the RTTOV v12 user guide.'
873 call error_handler(e_err, routine, string1, &
874 source, text2=string2, text3=string3)
875 else
876 mag_field = missing_r8
877 cosbk = missing_r8
878 end if
879
880 ! FIXME: consider adding an atlas for looking up FASTEM parameters
881 ! From the RTTOV User guide:
882 ! Surface type
883 ! Typical RTTOV default for land: 3.0, 5.0, 15.0, 0.1, 0.3
884 ! Summer land surface:
885 ! Forest: 1.7, 1.0, 163.0, 0.0, 0.5
886 ! Open grass: 2.2, 1.3, 138.0, 0.0, 0.42
887 ! Bare soil: 2.3, 1.9, 21.8, 0.0, 0.5
888 ! Winter surface type:
889 ! Forest and snow: 2.9, 3.4, 27.0, 0.0, 0.0
890 ! Deep dry snow: 3.0, 24.0, 60.0, 0.1, 0.15
891 ! Frozen soil: 117.8, 2.0, 0.19, 0.2 ,0.35
892 ! Sea ice
893 ! Grease ice: 23.7, 7.7, 17.3, 0.0, 0.15
894 ! Baltic nilas: 1.6, 3.3, 2.2, 0.0, 0.0
895 ! New ice (no snow): 2.9, 3.4, 27.0, 0.0, 0.0
896 ! New ice (snow): 2.2, 3.7, 122.0, 0.0, 0.15
897 ! Brash ice: 3.0, 5.5, 183.0, 0.0, 0.0
898 ! Compact pack ice: 2.0, 1700000.0, 49000000.0, 0.0, 0.0
899 ! Fast ice: 1.5, 77.8, 703.0, 0.1, 0.35
900 ! Lake ice + snow: 1.8, 67.1, 534.0, 0.1, 0.15
901 ! Multi-year ice: 1.5, 85000.0, 4700000.0, 0.0, 0.0
902 fastem_p1 = 3.0d0
903 fastem_p2 = 5.0d0
904 fastem_p3 = 15.0d0
905 fastem_p4 = 0.1d0
906 fastem_p5 = 0.3d0
907
908 call set_mw_metadata(key, sat_az, sat_ze, platform_id, sat_id, sensor_id, &
909 ichan, mag_field, cosbk, &
910 fastem_p1, fastem_p2, fastem_p3, fastem_p4, fastem_p5)
911
912 call create_3d_obs(olat, olon, vloc, which_vert, obs_value, robstype, &
913 obs_err, days, seconds, rqc, obs, key)
914
915 call add_obs_to_seq(seq, obs, obs_time, prev_obs, pre_time, is_first_obs)
916
917 obs_num = obs_num + 1
918
919 enddo channel_loop
920 enddo pixloop
921enddo scanloop
922
923! Print a little summary
924if (verbosity > 2) call print_obs_seq_summary(seq)
925
926write(string1,*) 'Converted ',obs_num,' obs for ',trim(swath%instrument), &
927 '; total obs = ',key
928call error_handler(e_msg, routine, string1, source)
929
930end subroutine add_granule_observations
