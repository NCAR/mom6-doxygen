42!
43! invert_altimeter - function that computes the surface pressure
44! given an altimeter setting and the station
45! elevation.
46!
47! altimeter_setting - altimeter setting (hPa)
48! elevation - elevation of station (m)
49! invert_altimeter - surface pressure value (hPa)
50!
51! created Dec. 2007 Ryan Torn, NCAR/MMM
52!
53!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
54function invert_altimeter(altimeter_setting, elevation)
55
56use types_mod, only : r8
57
58implicit none
59
60real(r8), parameter :: k1 = 0.190284_r8
61real(r8), parameter :: k2 = 8.4228807e-5_r8
62
63real(r8), intent(in) :: altimeter_setting, elevation
64
65real(r8) :: invert_altimeter ! (hPa)
66
67invert_altimeter = (altimeter_setting ** k1 - k2 * elevation) ** (1 / k1) + 0.3_r8
68
69return
70end function invert_altimeter

109real(r8) :: sat_vapor_pressure
110! Clausius-Clapeyron w/ constant Lv in Pa

117!
118! specific_humidity - function that computes the specific humidity
119! given the vapor pressure and atmospheric dry

122! vapor_pres - vapor pressure (Pa)
123! pres - atmospheric pressure (Pa)

128function specific_humidity(vapor_pres, pres)

130real(r8), intent(in) :: vapor_pres, pres
131
132real(r8) :: specific_humidity
133
134! 11/2009 changed to Emanuel (1994) eqn 4.1.4

187!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
188!
189! sat_vapor_press_bolton - function that uses Bolton's approximation to
190! compute saturation vapor pressure given
191! temperature.
192!
193! reference: Bolton 1980, MWR, 1046-1053
194!
195! sat_vapor_press_bolton - saturation vapor pressure (Pa)
196! tmpk - temperature (K)
197!
198! created Dec. 2008 David Dowell, NCAR/MMM
199!
200!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
201function sat_vapor_press_bolton(tmpk)
202
203real(r8) :: sat_vapor_press_bolton
204real(r8), intent(in) :: tmpk
205
206real(r8) :: tmpc ! temperature (Celsius)
207
208tmpc = tmpk - tfrez
209if ( tmpc <= -200.0_r8 ) then
210 print*,'sat_vapor_press_bolton: tmpc too low ',tmpc
211 stop
212end if
213sat_vapor_press_bolton = es0c * exp( 17.67_r8 * tmpc / (tmpc + 243.5_r8) )
214
215return
216end function sat_vapor_press_bolton

230function temp_and_dewpoint_to_rh(tmpk, dptk)
231
232real(r8) :: temp_and_dewpoint_to_rh
233real(r8), intent(in) :: tmpk
234real(r8), intent(in) :: dptk
235
236real(r8) :: e ! vapor pressure (Pa)
237real(r8) :: es ! saturation vapor pressure (Pa)
238
239e = sat_vapor_press_bolton(dptk)

242temp_and_dewpoint_to_rh = e / es

245 print*,'rh = ', temp_and_dewpoint_to_rh, ', resetting to 1.00'

253!
254! rh_and_temp_to_dewpoint - function that computes the dewpoint
255! given relative humidity and temperature
256!
257! rh_and_temp_to_dewpoint - dewpoint (Kelvin)
258! rh - relative humidity (0.00 - 1.00)
259! tmpk - temperature (Kelvin)
260!
261! created Dec. 2008 David Dowell, NCAR/MMM
262!
263!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
264function rh_and_temp_to_dewpoint(rh, tmpk)
265
266real(r8) :: rh_and_temp_to_dewpoint
267real(r8), intent(in) :: rh
268real(r8), intent(in) :: tmpk
269
270real(r8) :: e ! vapor pressure (Pa)
271real(r8) :: es ! saturation vapor pressure (Pa)
272real(r8) :: dptc ! dptc (Celsius)
273
274if ( ( rh <= 0.00_r8 ) .or. ( rh > 1.00_r8 ) ) then
275 print*,'rh_and_temp_to_dewpoint: bad rh ',rh
276 stop
277end if
278if ( rh <= 0.01_r8 ) then
279 print*,'rh_and_temp_to_dewpoint: low rh ',rh
280end if
281
282es = sat_vapor_press_bolton(tmpk)
283e = rh * es
284
285dptc = 243.5_r8 / (17.67_r8 / log(e/es0c) - 1.0_r8)
286
287rh_and_temp_to_dewpoint = dptc + tfrez
288
289return
290end function rh_and_temp_to_dewpoint
