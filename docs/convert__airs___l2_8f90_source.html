95! when this routine returns, the l2_files variable will have

97filecount = set_filename_list(l2_files, l2_file_list, "convert_airs_l2")
98
99! used to estimate the max size of the output sequence
100thin_factor = compute_thin_factor(along_track_thin, cross_track_thin)
101
102! initialize an empty obs_seq to start
103seq = initialize_obs_sequence(filecount, thin_factor)
104
105! for each input file
106do index=1, filecount
107
108 ! read from HDF file into a derived type that holds all the information
109 call airs_ret_rdr(l2_files(index), granule, version)
110
111 ! convert derived type information to DART sequence

113 min_mmr_threshold, top_pressure_level, &

115 use_ncep_errs, version)
116
117enddo

121

124
125call error_handler(e_msg, source, 'Finished successfully.',source,revision,revdate)
126call finalize_utilities()
