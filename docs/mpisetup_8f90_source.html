44subroutine mpi_initialize()
45integer :: ierr
46call mpi_init(ierr)
47! nproc is process number, numproc is total number of processes.
48call mpi_comm_rank(mpi_comm_world,nproc,ierr)
49call mpi_comm_size(mpi_comm_world,numproc,ierr)
50
51if (nproc == 0) print *,'running on ',numproc,' processors ...'
52if (r_kind == r_single) then
53 mpi_realkind = mpi_real4
54 if (nproc == 0) print *,'mpi_realkind set to mpi_real4',mpi_real4
55else if (r_kind == r_double) then
56 mpi_realkind = mpi_real8
57 if (nproc == 0) print *,'mpi_realkind set to mpi_real8',mpi_real8
58else
59 print *,'ERROR: illegal r_kind (must be single or double)'
60 call mpi_cleanup()
61endif
62
63end subroutine mpi_initialize
64
65subroutine mpi_cleanup()
66integer :: ierr
67flush(6,err=10)
68flush(0,err=10)
6910 continue
70call mpi_barrier(mpi_comm_world,ierr)
71if (nproc == 0) write(6,*) 'all done!'
72call mpi_finalize(ierr)
73if (ierr /= 0) then
74 print *, 'MPI_Finalize error status = ',ierr
75end if
76end subroutine mpi_cleanup
