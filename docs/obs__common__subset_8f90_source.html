6!> This program takes as input one or more sets of N obs_seq filenames.
7!> N is usually 2, but can be any number of files to compare together.
8!> It opens the obs_seq files N at a time and does an obs-by-obs comparison.
9!> It is intended to pull out a common subset of obs which were successfully
10!> assimilated in all experiments so the subsequent diagnostics can give
11!> an apples-to-apples comparison of the relative error or fit to obs
12!> between different experiments.
13!>
14!> This tool does not search in the input obs_seq files to try to match obs
15!> in different orders or to skip over obs present in only one input file,
16!> so all experiments should be done with identical input obs_seq files.
17!> The obs to be assimilated or evaluated can be selected by namelist
18!> at run time if they are different, and different experiment settings
19!> can be selected by other namelist values.
20!>
21!> If the next input obs don't match exactly (identical types, times, locations,
22!> and DART QC) they are discarded. If they match, and if the DART QC indicates
23!> they have been successfully assimilated in all experiments, then each of the
24!> N input obs are copied through to the N output obs_seq files. At the end of
25!> running this tool it creates sets of N output obs_seq files which contain a
26!> consistent set of assimilated obs across the set.
27!>
28!> The outputs are suitable for input to the obs_diag or obs_seq_to_netcdf tools
29!> and there are matlab scripts for comparing the results of N experiments.
30!> Differences in the output diagnostics will now be because of the experiment
31!> differences, not differences in the number of obs assimilated.
32!>
33!> Running this program creates sets of N new output files corresponding to each
34!> of N input files. The output names are the input filenames with a suffix appended.
35!> It can take a list of obs_seq files, either given directly in the namelist or
36!> a list of N filenames where each file contains one input filename per line.
37!> If the lists contain more than 1 filename, this program will loop over each
38!> set of filenames in the lists until the end of the file.If more than N files
39!>
40!> There is an option to allow observations which have been either successfully
41!> assimilated or evaluated to match each other. The default requires them to
42!> have been treated identically in all experiments.
43

85type(obs_type) :: obs_out(maxcomp), prev_obs_out(maxcomp)
86logical :: is_this_last(maxcomp)

89integer :: size_seq_out, num_inserted, iunit, io, i, j, k, nextfile

109

111character(len=256) :: filename_seq_list(maxcomp) = ''
112character(len=32) :: filename_out_suffix = '.common'

115logical :: print_only = .false.
116
117! not expected to be changed often, but its here if needed.
118integer :: print_every = 10000
119integer :: dart_qc_threshold = 3 ! ok if DART QC <= this

125 num_to_compare_at_once, &

128 print_every, dart_qc_threshold, &

131!----------------------------------------------------------------
132! Start of the program:
133!
134! Process each N input observation sequence file in turn,
135! selecting observations to insert into the output sequence files.
136!----------------------------------------------------------------
137
138call setup()
139
140! Read the namelist entry
141call find_namelist_in_file("input.nml", "obs_common_subset_nml", iunit)
142read(iunit, nml = obs_common_subset_nml, iostat = io)
143call check_namelist_read(iunit, io, "obs_common_subset_nml")
144
145! Record the namelist values used for the run ...
146if (do_nml_file()) write(nmlfileunit, nml=obs_common_subset_nml)
147if (do_nml_term()) write( * , nml=obs_common_subset_nml)
148
149! the logic here is slightly different than the obs_sequence_tool.
150! the user gives us a list of obs_seq files; either in the namelist
151! or as the name of a file which contains the list, one per line.
152! the files are compared N at a time, until the list is exhausted.
153! as in the other tools, it is an error to specify both an explicit
154! list and the name of file for input.
155
156write(msgstring, '(A,I3,A)') "Asking to compare ", num_to_compare_at_once, " obs_seq files at a time"
157call error_handler(e_msg, "obs_common_subset", msgstring, source)
158
159if ((num_to_compare_at_once < 2) .or. (num_to_compare_at_once > maxcomp)) then
160 write(msgstring, *) "num_to_compare_at_once must be >= 2 and <= ", maxcomp
161 call error_handler(e_err, "obs_common_subset", msgstring, source)
162endif
163
164! if there is only a single file from each experiment to compare, the order
165! doesn't matter. if each experiment has a list of input obs_seq.final files from
166! a multi-step assimilation, list all the files from experiment 1 in list-file 1,
167! all files from experiment 2 in list-file 2, etc.
168call handle_filenames(num_to_compare_at_once, maxcomp, filename_seq, filename_seq_list, &
169 num_input_sets)
170
171if (num_input_sets > 1) then
172 write(msgstring, '(A,I3,A)') "Will loop ", num_input_sets, " times to process the entire input filelist"
173 call error_handler(e_msg, "obs_common_subset", msgstring, source)
174endif
175
176! because i am a lazy typist
177atonce = num_to_compare_at_once
178nsets = num_input_sets
179
180! the default is a gregorian calendar. if you are using a different type
181! set it in the namelist. this only controls how it prints out the first
182! and last timestamps in the obs_seq files.
183call set_calendar_type(calendar)
184
185! set a logial to see if we have a calendar or not
186cal = (get_calendar_type() /= no_calendar)
187
188! end of namelist processing and setup
189
190
191! process the files in sets, making sure the metadata matches. keep matching
192! obs which share the same QC values, as long as the QCs are <= threshold.
193! since this tool has very little utility without the DART QC values, fail
194! if they aren't present.
195
196! count of input sets was set in the handle_filenames() routine above.
197nextfile = 1
198numsets: do j = 1, nsets
199
200 ! if num_to_compare_at_once = 3 and num_input_sets = 100, then this
201 ! assumes the input file list is organized in this order:
202 ! a1,b1,c1, a2,b2,c2, ..., a100,b100,c100
203 ! where a, b, c are the 3 separate experiments to compare,
204 ! and 1, 2, ... 100 are the obs_seq.final output files from 100 steps
205 ! of a multi-step assimilation. handle_filenames() arranges this
206 ! order if list files were used.
207
208 ! fill in the names of the current set of obs_seq files
209 do i = 1, atonce
210 filename_in(i) = filename_seq(nextfile)
211 nextfile = nextfile + 1
212 enddo
213
214 ! now filename_in(:) has all the names for this set. read the headers in.
215 do i = 1, atonce
216
217 call read_obs_seq_header(filename_in(i), num_copies_in(i), num_qc_in(i), &
218 size_seq_in(i), max_num_obs(i), file_id, read_format, pre_i_format, &
219 close_the_file = .true.)
220
221 if (size_seq_in(i) == 0) then
222 write(msgstring1, *) 'This tool cannot process empty obs_seq files'
223 write(msgstring,*) 'No obs found in input sequence file ', trim(filename_in(i))
224 call error_handler(e_err,'obs_common_subset',msgstring, &
225 source, text2=msgstring1)
226 endif
227
228 enddo
229
230 do i = 2, atonce
231 if (size_seq_in(1) /= size_seq_in(i)) then
232 write(msgstring,*) 'Input obs_seq files cannot have different observation counts'
233 write(msgstring2,*) 'Counts for file 1 and file ', i,' are ', size_seq_in(1), &
234 ' obs and ', size_seq_in(i), ' obs'
235 call error_handler(e_err,'obs_common_subset',msgstring, &
236 source, text2=msgstring2)
237 endif
238 enddo
239
240 write( *, *) 'Starting to process input sequence files: '
241 write(logfileunit, *) 'Starting to process input sequence files: '
242
243 do i = 1, atonce
244 write( * , *) trim(filename_in(i))
245 write(logfileunit, *) trim(filename_in(i))
246
247 call read_obs_seq(filename_in(i), 0, 0, 0, seq_in(i))
248 enddo
249
250 ! make sure the files have compatible metadata. this errors out here if not.
251 call compare_metadata(atonce, seq_in, filename_in)
252
253 ! sanity check - ensure the linked list times are in increasing time order
254 do i = 1, atonce
255 call validate_obs_seq_time(seq_in(i), filename_in(i))
256 enddo
257
258 ! the output can have no more than the number of input obs, and we already
259 ! tested that all sequences have the same obs count.
260 size_seq_out = size_seq_in(1)
261
262 ! find which DART QC copy has the DART quality control. this version of
263 ! the tool fails if one is not found. there seems very little utility in
264 ! this tool if it tries to process files without a DART QC.
265 qc_index = -1
266 qcloop: do i=1, get_num_qc(seq_in(1))
267 if(index(get_qc_meta_data(seq_in(1),i), dart_qc_meta_data) > 0) then
268 qc_index = i
269 exit qcloop
270 endif
271 enddo qcloop
272
273 ! if we do decide there is utility in running these on files without a
274 ! DART QC, remove this error exit here. nsc 24 oct 2013
275 if (qc_index < 0) then
276 write(msgstring,*) 'DART QC was not found in the input obs_seq files.'
277 write(msgstring2,*) 'This tool only works on output files from an assimilation.'
278 call error_handler(e_err,'obs_common_subset',msgstring, &
279 source, text2=msgstring2)
280 endif
281
282
283 ! blank line, start of actually creating output file
284 call error_handler(e_msg,' ',' ')
285
286 do i = 1, atonce
287 ! Initialize individual observation variables
288 call init_obs( obs_in(i), num_copies_in(1), num_qc_in(1))
289 call init_obs( next_obs_in(i), num_copies_in(1), num_qc_in(1))
290 call init_obs( obs_out(i), num_copies_in(1), num_qc_in(1))
291 call init_obs(prev_obs_out(i), num_copies_in(1), num_qc_in(1))
292
293 ! create the output sequences
294 call init_obs_sequence(seq_out(i), num_copies_in(1), num_qc_in(1), size_seq_out)
295
296 do k=1, num_copies_in(1)
297 meta_data = get_copy_meta_data(seq_in(1), k)
298 call set_copy_meta_data(seq_out(i), k, meta_data)
299 enddo
300 do k=1, num_qc_in(1)
301 meta_data = get_qc_meta_data(seq_in(1), k)
302 call set_qc_meta_data(seq_out(i), k, meta_data)
303 enddo
304
305 enddo
306
307 ! Start to insert obs from sequence_in into sequence_out
308 ! NOTE: insert_obs_in_seq CHANGES the obs passed in.
309 ! Must pass a copy of incoming obs to insert_obs_in_seq.
310
311 num_inserted = 0
312 num_rejected_badqc = 0
313 num_rejected_diffqc = 0
314 num_rejected_other = 0
315 num_mismatch_time = 0
316 num_mismatch_type = 0
317 num_mismatch_loc = 0
318
319 ! get the first obs from each file. since we have already checked
320 ! for empty sequence files, this is not expected to fail at this point.
321 do i = 1, atonce
322 if ( .not. get_first_obs(seq_in(i), obs_in(i))) then
323 write(msgstring, *)'no first observation in ',trim(filename_in(i))
324 call error_handler(e_err,'obs_common_subset',msgstring,source)
325 endif
326 is_this_last(i) = .false.
327 next_obs_in(i) = obs_in(i)
328 enddo
329
330
331 ! since we've checked to be sure that all the input obs_seq files have
332 ! the same obs counts, i can test the result from the first seq only.
333 obsloop : do while ( .not. is_this_last(1) )
334
335 ! move the next obs into position to be worked on
336 do i = 1, atonce
337 obs_in(i) = next_obs_in(i)
338 enddo
339
340 ! see if we want to keep it or not
341 wanted = all_good(obs_in, atonce, qc_index, dart_qc_threshold)
342
343 if ( wanted ) then
344 do i = 1, atonce
345 obs_out(i) = obs_in(i)
346 enddo
347
348 ! Since the stride through the observation sequence file is always
349 ! guaranteed to be in temporally-ascending order, we can use the
350 ! 'previous' observation as the starting point to search for the
351 ! correct insertion point. This speeds up the insert code a lot.
352
353 if (num_inserted == 0) then
354 do i = 1, atonce
355 call insert_obs_in_seq(seq_out(i), obs_out(i))
356 enddo
357 else
358 do i = 1, atonce
359 call insert_obs_in_seq(seq_out(i), obs_out(i), prev_obs_out(i))
360 enddo
361 endif
362
363 ! update position in seq for next insert
364 do i = 1, atonce
365 prev_obs_out(i) = obs_out(i)
366 enddo
367
368 num_inserted = num_inserted + 1
369
370 if (print_every > 0) then
371 if (mod(num_inserted,print_every) == 0) then
372 print*, 'inserted obs number ',num_inserted,' of ',size_seq_out
373 endif
374 endif
375
376 endif
377
378 do i = 1, atonce
379 call get_next_obs(seq_in(i), obs_in(i), next_obs_in(i), is_this_last(i))
380 enddo
381
382 enddo obsloop
383
384
385 ! finished creating output sequences. print info and write out files
386
387 write(msgstring, *) 'Number of obs in each output seq file :', get_num_key_range(seq_out(1))
388 call error_handler(e_msg,'obs_common_subset',msgstring)
389
390 do i = 1, atonce
391 filename_out(i) = trim(filename_in(i))//trim(filename_out_suffix)
392 call print_obs_seq(seq_out(i), filename_out(i))
393 enddo
394
395 if (print_only) then
396 write(msgstring,*) 'Output sequence files not created; print_only in namelist is .true.'
397 call error_handler(e_msg,'', msgstring)
398 else
399
400 print*, '--------- Obs seq file set # : ', j
401 print*, 'Number of obs in each input : ', size_seq_in(1)
402 print*, 'Number of obs rejected diff qc : ', num_rejected_diffqc
403 print*, 'Number of obs rejected bad qc : ', num_rejected_badqc
404 print*, 'Number of obs rejected other : ', num_rejected_other
405 if (num_rejected_other > 0) then
406 print*, ' Number of mismatches in time : ', num_mismatch_time
407 print*, ' Number of mismatches in type : ', num_mismatch_type
408 print*, ' Number of mismatches in loc : ', num_mismatch_loc
409 endif
410 print*, 'Number of obs copied to output : ', num_inserted
411 print*, '---------------------------------------------------------'
412
413 write(msgstring, *) 'Starting to write output sequence files'
414 call error_handler(e_msg,'obs_common_subset',msgstring)
415
416 do i = 1, atonce
417 call write_obs_seq(seq_out(i), filename_out(i))
418 enddo
419 endif
420
421 ! clean up - prev_obs_out is a copy of obs_out, so don't call destroy on it
422 ! because it's already been deleted.
423
424 do i = 1, atonce
425 call destroy_obs_sequence(seq_in(i))
426 call destroy_obs_sequence(seq_out(i))
427 call destroy_obs(obs_in(i))
428 call destroy_obs(obs_out(i))
429 call destroy_obs(next_obs_in(i))
430 enddo
431
432enddo numsets
433
434call shutdown()
435

437! end of main program.

464subroutine handle_filenames(setsize, maxsets, filename_seq, filename_seq_list, &
465 num_input_sets)
466
467! sort out the input lists, set the length as a return in num_input_sets,
468! fill in filename_seq if list was given, and make sure what's specified is consistent.
469
470integer, intent(in) :: setsize
471integer, intent(in) :: maxsets
472character(len=*), intent(inout) :: filename_seq(:)
473character(len=*), intent(in) :: filename_seq_list(:)
474integer, intent(out) :: num_input_sets
475
476integer :: indx, num_input_files, i, ifiles, fcount, this_set_filecount
477integer :: files_per_set, nlists, iset, src, dst
478logical :: from_file
479character(len=32) :: nname
480
481! if the user specifies neither filename_seq nor filename_seq_list, error
482! if the user specifies both, error.
483! if list length isn't a whole multiple of num_to_compare_at_once, error.
484! if user gives a filelist name, make sure the length is not more than
485! maxfiles and read it into the explicit list and continue.
486! set num_input_sets to the count of sets in the list
487
488if (filename_seq(1) == '' .and. filename_seq_list(1) == '') then
489 msgstring2='One of filename_seq or filename_seq_list must be specified in namelist'
490 call error_handler(e_err,'handle_filenames', &
491 'no filenames specified as input', source, text2=msgstring2)
492endif
493
494! make sure the namelist specifies one or the other but not both
495if (filename_seq(1) /= '' .and. filename_seq_list(1) /= '') then
496 call error_handler(e_err,'handle_filenames', &
497 'cannot specify both filename_seq and filename_seq_list in namelist', source)
498endif
499
500! count of all filenames, either explicitly in the namelist
501! or specified by a list of files which contain lists of names.
502num_input_files = -1
503nlists = -1
504
505if (filename_seq(1) /= '') then
506
507 ! if the files are given in the namelist, just need to make sure
508 ! there are at least N given (where N == num_to_compare_at_once).
509
510 nname = 'filename_seq'
511 from_file = .false.
512
513 fileloop: do indx = 1, max_num_input_files
514 ! an empty name ends the list
515 if (filename_seq(indx) == '') then
516 if (indx < num_to_compare_at_once) then
517 write(msgstring, '(A,I5,A)') trim(nname)//' must contain at least ', &
518 num_to_compare_at_once, ' obs_seq filenames'
519 call error_handler(e_err,'handle_filenames', msgstring, source)
520 endif
521 num_input_files = indx - 1
522 exit fileloop
523 endif
524 enddo fileloop
525
526 ! make sure the list is an even multiple of setsize
527 if (modulo(num_input_files, setsize) /= 0) then
528 write(msgstring, *) 'number of input files must be an even multiple of ', setsize
529 write(msgstring1, *) 'found ', num_input_files, ' input files'
530 call error_handler(e_err,'handle_filenames', msgstring, source, text2=msgstring1)
531 endif
532
533 num_input_sets = num_input_files / setsize
534else
535
536 ! if they have specified a file which contains a list, read it into
537 ! a temp array, set the counts, and do some initial error checks.
538
539 nname = 'filename_seq_list'
540 from_file = .true.
541
542 ! this is the total running count of files across all lists
543 fcount = 1
544
545 exploop: do iset = 1, maxsets
546
547 ! an empty name ends the list of lists
548 if (filename_seq_list(iset) == '') then
549 num_input_files = fcount - 1
550 nlists = iset - 1
551 if (nlists /= setsize) then
552 write(msgstring1, '(A)') 'number of input file lists in "'//trim(nname)// &
553 '" must equal "num_to_compare_at_once"'
554 write(msgstring2, '(2(A,I5))') trim(nname)//' has ', nlists, &
555 ' items while num_to_compare_at_once is ', setsize
556
557 call error_handler(e_err,'handle_filenames', msgstring1, &
558 source, text2=msgstring2)
559 endif
560 exit exploop
561 endif
562
563 listloop: do ifiles=1, max_num_input_files
564 if (fcount >= max_num_input_files) then
565 write(msgstring, *) 'cannot specify more than ',max_num_input_files,' total files'
566 write(msgstring1, *) 'for more, change "max_num_input_files" in the source and recompile'
567 call error_handler(e_err,'handle_filenames', msgstring, source, &
568 text2=msgstring1)
569 endif
570
571 temp_filelist(fcount) = get_next_filename(filename_seq_list(iset), ifiles)
572
573 ! an empty name ends the list
574 if (temp_filelist(fcount) == '') then
575 if (ifiles == 1) then
576 call error_handler(e_err,'handle_filenames', &
577 trim(filename_seq_list(iset))//' contains no input obs_seq filenames', &
578 source)
579 endif
580 this_set_filecount = ifiles-1
581 exit listloop
582 endif
583 fcount = fcount + 1
584 enddo listloop
585
586 ! make sure each file in the list contains the same number
587 ! of filenames to be compared. set the target the first time
588 ! through and then compare subsequent counts to be sure they match.
589 if (iset == 1) then
590 files_per_set = this_set_filecount
591 else

834subroutine validate_obs_seq_time(seq, filename)
835
836! this eventually belongs in the obs_seq_mod code, but for now
837! try it out here. we just fixed a hole in the interactive create
838! routine which would silently let you create out-of-time-order
839! linked lists, which gave no errors but didn't assimilate the
840! right obs at the right time when running filter. this runs
841! through the times in the entire sequence, ensuring they are
842! monotonically increasing in time. this should help catch any
843! bad files which were created with older versions of code.
844
845type(obs_sequence_type), intent(in) :: seq
846character(len=*), intent(in) :: filename
847
848type(obs_type) :: obs, next_obs
849type(obs_def_type) :: this_obs_def
850logical :: is_there_one, is_this_last
851integer :: size_seq, obs_count
852integer :: key
853type(time_type) :: last_time, this_time
854
855
856! make sure there are obs left to process before going on.
857size_seq = get_num_obs(seq)
858if (size_seq == 0) then
859 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
860 call error_handler(e_msg,'validate_obs_seq_time',msgstring)
861 return
862endif
863
864! Initialize individual observation variables
865call init_obs( obs, get_num_copies(seq), get_num_qc(seq))
866call init_obs(next_obs, get_num_copies(seq), get_num_qc(seq))
867
868obs_count = 0
869
870
871! Start to process obs from seq
872
873is_there_one = get_first_obs(seq, obs)
874
875! we already tested for 0 obs above, so there should be a first obs here.
876if ( .not. is_there_one ) then
877 write(msgstring,*)'no first obs in sequence ' // trim(filename)
878 call error_handler(e_err,'validate_obs_seq_time', &
879 msgstring, source)
880 return
881endif
882
883is_this_last = .false.
884last_time = set_time(0, 0)
885obsloop : do while ( .not. is_this_last)
886
887 call get_obs_def(obs, this_obs_def)
888 this_time = get_obs_def_time(this_obs_def)
889
890 if (last_time > this_time) then
891 ! bad time order of observations in linked list
892 call print_time(last_time, ' previous timestamp: ')
893 if (cal) call print_date(last_time, ' calendar date: ')
894 call print_time(this_time, ' next timestamp: ')
895 if (cal) call print_date(this_time, ' calendar date: ')
896
897 key = get_obs_key(obs)
898 write(msgstring1,*)'obs number ', key, ' has earlier time than previous obs'
899 write(msgstring2,*)'observations must be in increasing time order, file ' // trim(filename)
900 call error_handler(e_err,'validate_obs_seq_time', msgstring2, &
901 source, text2=msgstring1)
902 endif
903
904 last_time = this_time
905 obs_count = obs_count + 1
906
907 call get_next_obs(seq, obs, next_obs, is_this_last)
908 if (.not. is_this_last) obs = next_obs
909
910enddo obsloop
911
912! clean up
913call destroy_obs( obs)
914call destroy_obs(next_obs)
915
916! technically not a time validation, but easy to check. obs_count should never
917! be larger than size_seq - that's a fatal error. obs_count < size_seq would
918! suggest there are obs in the file that aren't part of the linked list.
919! this does not necessarily indicate a fatal error but it's not a common
920! situation and might indicate someone should check on the file.
921if (obs_count /= size_seq) then
922 write(msgstring,*) 'input sequence ', trim(filename)
923 call error_handler(e_msg,'validate_obs_seq_time', msgstring)
924
925 write(msgstring,*) 'total obs in file: ', size_seq, ' obs in linked list: ', obs_count
926 if (obs_count > size_seq) then
927 ! this is a fatal error
928 write(msgstring1,*) 'linked list obs_count > total size_seq, should not happen'
929 call error_handler(e_err,'validate_obs_seq_time', msgstring, &
930 source, text2=msgstring1)
931 else
932 ! just warning msg
933 write(msgstring1,*) 'only observations in linked list will be processed'
934 call error_handler(e_msg,'validate_obs_seq_time', msgstring, &
935 source, text2=msgstring1)
936 endif
937endif
938
939end subroutine validate_obs_seq_time

981function all_good(obs_in, count, indx, threshold)
982
983! compare location, time, type, QC, plus make sure QC <= threshold
984
985type(obs_type), intent(in) :: obs_in(:)
986integer, intent(in) :: count, indx, threshold
987logical :: all_good
988
989type(obs_def_type) :: test1_obs_def, testn_obs_def
990integer :: test1_obs_type, testn_obs_type
991type(time_type) :: test1_obs_time, testn_obs_time
992type(location_type) :: test1_obs_loc, testn_obs_loc
993integer :: test1_qc, testn_qc
994real(r8) :: temp(1)
995integer :: i, key
996
997! assume failure so we can return as soon as we know they don't match.
998all_good = .false.
999
1000! get the info we need from sequence 1
1001call get_obs_def(obs_in(1), test1_obs_def)
1002test1_obs_loc = get_obs_def_location(test1_obs_def)
1003test1_obs_time = get_obs_def_time(test1_obs_def)
1004test1_obs_type = get_obs_def_type_of_obs(test1_obs_def)
1005call get_qc(obs_in(1), temp, indx)
1006test1_qc = nint(temp(1))
1007
1008
1009! compare the sets against the entries in sequence 1
1010do i = 2, count
1011 call get_obs_def(obs_in(i), testn_obs_def)
1012 testn_obs_loc = get_obs_def_location(testn_obs_def)
1013 testn_obs_time = get_obs_def_time(testn_obs_def)
1014 testn_obs_type = get_obs_def_type_of_obs(testn_obs_def)
1015 call get_qc(obs_in(i), temp, indx)
1016 testn_qc = nint(temp(1))
1017
1018 ! we expect the types, locations, and times to match. test them
1019 ! first so we can report those errors if they don't, since that is
1020 ! not expected.
1021
1022 if ((test1_obs_type /= testn_obs_type) .or. &
1023 (test1_obs_time /= testn_obs_time) .or. &
1024 (test1_obs_loc /= testn_obs_loc)) then
1025
1026 ! count of rejected obs.
1027 num_rejected_other = num_rejected_other + 1
1028
1029 key = get_obs_key(obs_in(1))
1030 write(msgstring,*) 'obs number ', key, ' does not match one or more of type/time/location'
1031 call error_handler(e_msg, '', msgstring)
1032
1033 ! more than one reason can be true for a single rejected ob
1034 if (test1_obs_type /= testn_obs_type) &
1035 num_mismatch_type = num_mismatch_type + 1
1036 if (test1_obs_time /= testn_obs_time) &
1037 num_mismatch_time = num_mismatch_time + 1
1038 if (test1_obs_loc /= testn_obs_loc ) &
1039 num_mismatch_loc = num_mismatch_loc + 1
1040
1041 return
1042 endif
1043
1044 ! check first to see if either of the DART QCs indicate
1045 ! that this obs was rejected for some reason. this code
1046 ! is testing the first QC multiple times if N > 2 but
1047 ! it makes the logic simpler so just repeat the test.
1048 if (test1_qc > threshold .or. testn_qc > threshold) then
1049 num_rejected_badqc = num_rejected_badqc + 1
1050 return
1051 endif
1052
1053 ! this is the test we expect to be the reason most obs fail:
1054 ! if they've been assimilated in one experiment but not another.
1055 if (.not. match_qc(test1_qc, testn_qc)) then
1056 num_rejected_diffqc = num_rejected_diffqc + 1
1057 return
1058 endif
1059
1060enddo
1061
1062! all match - good return.
1063all_good = .true.
1064
1065end function all_good

1069function match_qc(qc1, qc2)
1070
1071! normal behavior is to return true if the values are equal,
1072! and false if not. but if the "it's ok for evaluated and
1073! assimilated obs to match", then allow DART QC value 0 to
1074! match 1, and 2 to match 3, but no other combinations are valid.
1075
1076integer, intent(in) :: qc1
1077integer, intent(in) :: qc2
1078logical :: match_qc
1079
1080! if the combination of values is ok, return early
1081match_qc = .true.
1082
1083if (qc1 == qc2) return
1084
1085if (eval_and_assim_can_match) then
1086 if (qc1 == 0 .and. qc2 == 1) return
1087 if (qc1 == 1 .and. qc2 == 0) return
1088
1089 if (qc1 == 2 .and. qc2 == 3) return
1090 if (qc1 == 3 .and. qc2 == 2) return
1091endif
1092
1093! not one of the allowed matches. these qcs really
1094! are not compatible and the match failed.
1095
1096match_qc = .false.
1097
1098end function match_qc
