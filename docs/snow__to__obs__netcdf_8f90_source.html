10!
11! snow_to_obs_netcdf - input is a snow-coverage file that has been
12! converted from HDF to netCDF with an automated tool. this
13! program then takes the unsigned byte/integer(1) data and
14! converts it into a snow coverage obs_seq file.
15!
16! created 5 jul 2012 nancy collins NCAR/IMAGe
17! based on a previous text-based version done in mar 2010
18!

44integer :: n, i, j, oday, osec, rcio, iunit, otype, io
45integer :: num_copies, num_qc, max_obs, iacc, ialo, ncid, varid
46integer :: along, across, coarse_along, coarse_across, per_along, per_across
47integer :: along_base, across_base
48integer, allocatable :: qc_array(:,:), snowcover_type(:,:)
49
50integer(1), allocatable :: sn_byte(:,:)
51character(len=128) :: varname
52
53logical :: file_exist, first_obs
54
55real(r8) :: temp, terr, qc, wdir, wspeed, werr
56real(r8) :: vert, uwnd, uerr, vwnd, verr
57real(r8) :: dlon, dlat, thislon, thislat
58real(r8), allocatable :: lat(:,:), lon(:,:)

60
61type(obs_sequence_type) :: obs_seq
62type(obs_type) :: obs, prev_obs
63type(time_type) :: comp_day0, time_obs, prev_time

65integer :: year = 2000
66integer :: doy = 1
67character(len=128) :: snow_input_file = 'snowdata.input'
68logical :: debug = .false. ! set to .true. to print info

70
71namelist /snow_to_obs_nc_nml/ year, doy, &
72 snow_input_file, debug
73
74! ------------------------
75! start of executable code
76! ------------------------
77
78call initialize_utilities('snow_to_obs_netcdf')
79
80call find_namelist_in_file('input.nml', 'snow_to_obs_nc_nml', iunit)
81read(iunit, nml = snow_to_obs_nc_nml, iostat = io)
82call check_namelist_read(iunit, io, 'snow_to_obs_nc_nml')
83
84! Record the namelist values used for the run
85if (do_nml_file()) write(nmlfileunit, nml=snow_to_obs_nc_nml)
86if (do_nml_term()) write( * , nml=snow_to_obs_nc_nml)
87
88! open netcdf file here.
89call nc_check( nf90_open(snow_input_file, nf90_nowrite, ncid), &
90 'snow_to_obs_netcdf', 'opening file '//trim(snow_input_file))
91
92! get dims along the swath path, and across the swath path. the rest of
93! the data arrays use these for their dimensions
94call getdimlen(ncid, 'Along_swath_lines_500m_MOD_Swath_Snow', along)
95call getdimlen(ncid, 'Cross_swath_pixels_500m_MOD_Swath_Snow', across)
96call getdimlen(ncid, 'Coarse_swath_lines_5km_MOD_Swath_Snow', coarse_along)
97call getdimlen(ncid, 'Coarse_swath_pixels_5km_MOD_Swath_Snow', coarse_across)
98
99! remember that when you ncdump the netcdf file, the dimensions are
100! listed in C order. when you allocate them for fortran, reverse the order.
101allocate(sn_byte(across, along))
102allocate(fsnowcover(across, along))
103allocate(snowcover_type(across, along))
104allocate(qc_array(across, along))
105allocate(lon(coarse_across, coarse_along), lat(coarse_across, coarse_along))
106
107! apparently the lat/lon info is every Nth data value in each dim
108! FIXME: it seems like there are 2 less across values than there
109! should be to have 10 obs / across swath??
110per_along = along / coarse_along
111per_across = (across+2) / coarse_across
112
113! read the data for the requested arrays.
114varname = 'Fractional_Snow_Cover'
115call nc_check( nf90_inq_varid(ncid, varname, varid), &
116 'snow_to_obs_netcdf', 'inquire var '// trim(varname))
117call nc_check( nf90_get_var(ncid, varid, sn_byte), &
118 'snow_to_obs_netcdf', 'getting var '// trim(varname))
119
120! snowcover fraction is stored as unsigned bytes in the netcdf file.
121! read into a 1 byte int array, convert to unsigned and real
122! values from 0 to 100 are valid percentages, anything from 101 to 256
123! is a flag (see the netcdf header for details) and will be ignored.
124
125fsnowcover = sn_byte
126where (fsnowcover < 0.0_r8) fsnowcover = fsnowcover + 256.0_r8
127
128varname = 'Snow_Cover'
129call nc_check( nf90_inq_varid(ncid, varname, varid), &
130 'snow_to_obs_netcdf', 'inquire var '// trim(varname))
131call nc_check( nf90_get_var(ncid, varid, sn_byte), &
132 'snow_to_obs_netcdf', 'getting var '// trim(varname))
133
134varname = 'Snow_Cover_Pixel_QA'
135call nc_check( nf90_inq_varid(ncid, varname, varid), &
136 'snow_to_obs_netcdf', 'inquire var '// trim(varname))
137call nc_check( nf90_get_var(ncid, varid, sn_byte), &
138 'snow_to_obs_netcdf', 'getting var '// trim(varname))
139
140qc_array = sn_byte
141where (qc_array < 0) qc_array = qc_array + 256
142
143varname = 'Longitude'
144call nc_check( nf90_inq_varid(ncid, varname, varid), &
145 'snow_to_obs_netcdf', 'inquire var '// trim(varname))
146call nc_check( nf90_get_var(ncid, varid, lon), &
147 'snow_to_obs_netcdf', 'getting var '// trim(varname))
148
149! convert -180/180 to 0/360
150where (lon < 0.0_r8) lon = lon + 180.0_r8
151
152varname = 'Latitude'
153call nc_check( nf90_inq_varid(ncid, varname, varid), &
154 'snow_to_obs_netcdf', 'inquire var '// trim(varname))
155call nc_check( nf90_get_var(ncid, varid, lat), &
156 'snow_to_obs_netcdf', 'getting var '// trim(varname))
157
158! not sure if we need this one but read it in just in case
159! this seems to be some kind of cell type (night, no snow, lake, ocean,
160! cloud, lake ice snow, detector saturated, missing data, fill)
161varname = 'Snow_Cover'
162call nc_check( nf90_inq_varid(ncid, varname, varid), &
163 'snow_to_obs_netcdf', 'inquire var '// trim(varname))
164call nc_check( nf90_get_var(ncid, varid, sn_byte), &
165 'snow_to_obs_netcdf', 'getting var '// trim(varname))
166
167! snowcover is stored as unsigned bytes in the netcdf file.
168! read into a character array and convert to integer.
169
170snowcover_type = sn_byte
171where (snowcover_type < 0) snowcover_type = snowcover_type + 256
172
173
174! time setup
175call set_calendar_type(gregorian)
176
177!! all obs in a single file are the same time.
178comp_day0 = set_date(year, 1, 1, 0, 0, 0)
179time_obs = comp_day0 + set_time(0, doy)
180
181! extract time of observation into gregorian day, sec.
182call get_time(time_obs, osec, oday)
183
184! surface obs. normally we set vert to the actual surface elevation,
185! we do not have it here, so set to 0 for now.
186vert = 0.0_r8
187
188! -------------------------
189
190! each observation in this series will have a single observation value
191! and a quality control flag. the max possible number of obs needs to
192! be specified but it will only write out the actual number created.
193max_obs = across * along
194num_copies = 1
195num_qc = 1
196
197! call the initialization code, and initialize two empty observation types
198call static_init_obs_sequence()
199call init_obs(obs, num_copies, num_qc)
200call init_obs(prev_obs, num_copies, num_qc)
201first_obs = .true.
202
203! create a new, empty obs_seq file. you must give a max limit
204! on number of obs. increase the size if too small.
205call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
206
207! the first one needs to contain the string 'observation' and the
208! second needs the string 'QC'.
209call set_copy_meta_data(obs_seq, 1, 'observation')
210call set_qc_meta_data(obs_seq, 1, 'Data QC')
211
212! if you want to append to existing files (e.g. you have a lot of
213! small text files you want to combine), you can do it this way,
214! or you can use the obs_sequence_tool to merge a list of files
215! once they are in DART obs_seq format.
216
217! ! existing file found, append to it
218! inquire(file=obs_out_file, exist=file_exist)
219! if ( file_exist ) then
220! call read_obs_seq(obs_out_file, 0, 0, max_obs, obs_seq)
221! endif
222
223! we have to pick an error range. since this is a snow cover fraction
224! observation, the valid values should go from 0 to 1.0, so pick 0.1 for now.
225terr = 0.1_r8 ! FIXME - wild guess
226qc = 0.0_r8 ! we will reject anything with a bad qc
227
228
229alongloop: do ialo = 1, coarse_along
230
231 acrossloop: do iacc = 1, coarse_across
232
233if (debug) print *, 'start of main loop, ', iacc, ialo
234
235 !! check the lat/lon values to see if they are ok
236 if ( lat(iacc, ialo) > 90.0_r8 .or. lat(iacc, ialo) < -90.0_r8 ) cycle alongloop
237 if ( lon(iacc, ialo) < 0.0_r8 .or. lon(iacc, ialo) > 360.0_r8 ) cycle alongloop
238
239 ! the actual data values are denser, so inner loop here
240 along_base = (ialo-1) * per_along + 1
241 inner_along: do j = along_base, along_base + per_along
242
243 across_base = (iacc-1) * per_across + 1
244 inner_across: do i = across_base, across_base + per_across
245
246 if (i > across) cycle inner_across
247 if (j > along) cycle inner_across
248 if (qc_array(i,j) /= 0) cycle inner_across
249 if (fsnowcover(i, j) > 101) cycle inner_across
250
251 ! compute the lat/lon for this obs FIXME: this isn't right
252
253 thislat = lat(iacc, ialo)
254 thislon = lon(iacc, ialo)
255
256 ! make an obs derived type, and then add it to the sequence
257 call create_3d_obs(thislat, thislon, vert, vertissurface, fsnowcover(i,j)/100.0_r8, &
258 modis_snowcover_frac, terr, oday, osec, qc, obs)
259 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
260
261 if (debug) print *, 'added snow obs to output seq'
262
263 end do inner_across
264 end do inner_along
265 end do acrossloop
266end do alongloop
267
268! if we added any obs to the sequence, write it out to a file now.
269if ( get_num_obs(obs_seq) > 0 ) then
270 if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
271 call write_obs_seq(obs_seq, obs_out_file)
272endif
273
274! end of main program
275call finalize_utilities()
276
277end program snow_to_obs_netcdf
