7program test_quad_irreg_interp
8
9! intended to show how the state structure and quad code can be used
10! together. start with a simple regional grid and work out from there.
11
12! Modules that are absolutely required for use are listed
13use types_mod, only : r8, i8, missing_r8, deg2rad, rad2deg
14use utilities_mod, only : error_handler, initialize_utilities, finalize_utilities
15use random_seq_mod, only : init_random_seq, random_seq_type, &
16 random_uniform, random_gaussian
17
18use quad_utils_mod, only : quad_interp_handle, init_quad_interp, finalize_quad_interp, set_quad_coords, &
19 quad_lon_lat_locate, quad_lon_lat_evaluate, grid_quad_fully_regular, &
20 grid_quad_irreg_spaced_regular, grid_quad_fully_irregular, grid_quad_unknown_type, &
21 quad_located_unknown, quad_located_cell_centers, quad_located_lon_edges, &
22 quad_located_lat_edges, quad_located_cell_corners

25implicit none
26
27integer :: debug = 0
28
29type(quad_interp_handle) :: h
30
31! data grid size
32integer, parameter :: nx = 9
33integer, parameter :: ny = 5
34
35! locations of data grid corners
36real(r8) :: data_lons(nx, ny) = missing_r8
37real(r8) :: data_lats(nx, ny) = missing_r8
38
39! extents of the data grid (these mimic a regional model's grid)
40real(r8) :: start_lon = 100.0_r8
41real(r8) :: end_lon = 150.5_r8
42real(r8) :: start_lat = -11.4_r8
43real(r8) :: end_lat = 34.1_r8
44
45! angle to rotate data grid in degrees
46! positive is counterclockwise; will rotate
47! around lower left grid point (start lon/lat).
48 real(r8) :: angle = 10.0_r8
49!real(r8) :: angle = 45.0_r8
50!real(r8) :: angle = 30.0_r8
51!real(r8) :: angle = 90.0_r8
52!real(r8) :: angle = -30.0_r8
53!real(r8) :: angle = -10.0_r8
54!real(r8) :: angle = 0.0_r8
55
56! deform grid by this fraction of the deltas
57real(r8) :: lon_def = 0.25_r8
58real(r8) :: lat_def = 0.25_r8
59
60! data values on the grid
61real(r8) :: grid_data(nx, ny) = missing_r8
62integer :: data_choice = 0 ! see code for selection values
63
64! percent of data values that should be marked 'missing data'
65!real(r8) :: miss_percent = 0.0_r8 ! none
66 real(r8) :: miss_percent = 3.0_r8 ! 3%
67!real(r8) :: miss_percent = 100.0_r8 ! all
68
69! sampling grid size
70integer, parameter :: nrx = 210
71integer, parameter :: nry = 150
72
73! locations of sampling grid
74real(r8) :: sample_lons(nrx) = missing_r8
75real(r8) :: sample_lats(nry) = missing_r8
76
77! extents of the sampling grid
78real(r8) :: sample_start_lon = 110.0_r8
79real(r8) :: sample_end_lon = 140.0_r8
80real(r8) :: sample_start_lat = -20.0_r8
81real(r8) :: sample_end_lat = 30.0_r8
82
83! where interpolated values are stored on reg grid
84real(r8) :: interp_data(nrx, nry) = missing_r8
85
86
87type(random_seq_type) :: ran
88
89integer :: i, j, k
90real(r8) :: lon_del, lat_del, sample_lon_del, sample_lat_del
91integer :: lon_bot, lat_bot, lon_top, lat_top
92integer :: four_lons(4), four_lats(4)
93integer :: istatus
94real(r8) :: invals(4), outval
95integer :: iunit_orig, iunit_interp
96
97call initialize_utilities('test_quad_irreg_interp')
98call init_random_seq(ran)
99
100lon_del = (end_lon - start_lon) / (nx-1)
101lat_del = (end_lat - start_lat) / (ny-1)
102
103! "data grid" corners and data vals
104do i=1, nx
105 do j=1, ny
106 ! generate locations of the corners of all the quads
107 data_lons(i, j) = start_lon + (i-1)*lon_del + deform(lon_del, lon_def, ran)
108 data_lats(i, j) = start_lat + (j-1)*lat_del + deform(lat_del, lat_def, ran)
109 if (angle /= 0.0_r8) &
110 call rotate(data_lons(i, j), data_lats(i, j), angle, start_lon, start_lat)
111
112 ! generate the data values on the corners. pick one:
113 select case (data_choice)
114 case (1)
115 ! increasing monotonically
116 grid_data(i, j) = (j-1)*nx + i
117 case (2)
118 ! constant by row
119 grid_data(i, j) = j
120 case (3)
121 ! constant by column
122 grid_data(i, j) = i
123 case (4)
124 ! based on lon only
125 grid_data(i, j) = data_lons(i, j)
126 case (5)
127 ! based on lat only
128 grid_data(i, j) = data_lats(i, j)
129 case (6)
130 ! random between (0-10)
131 grid_data(i, j) = random_uniform(ran) * 10.0_r8
132 case default
133 ! gaussian with mean 0 and stddev 1
134 grid_data(i, j) = random_gaussian(ran, 0.0_r8, 1.0_r8)
135 end select
136
137 if (miss_percent > 0.0_r8) then
138 if (random_uniform(ran) * 100.0_r8 < miss_percent) grid_data(i, j) = missing_r8
139 endif
140
141 enddo
142enddo
143
144sample_lon_del = (sample_end_lon - sample_start_lon) / nrx
145sample_lat_del = (sample_end_lat - sample_start_lat) / nry
146
147! "sampled grid" spacing along each axis
148do i=1, nrx
149 sample_lons(i) = sample_start_lon + (i-1)*sample_lon_del
150enddo
151do j=1, nry
152 sample_lats(j) = sample_start_lat + (j-1)*sample_lat_del
153enddo
154
155! end of data setup - now call interp routines
156
157call init_quad_interp(grid_quad_fully_irregular, nx, ny, quad_located_cell_centers, .false., .false., .false., h)
158call set_quad_coords(h, data_lons, data_lats)
159
160! for each location in the sampling grid, interpolate a data value
161do i=1, nrx
162 do j=1, nry
163 call quad_lon_lat_locate(h, sample_lons(i), sample_lats(j), four_lons, four_lats, istatus)
164 if (istatus /= 0) then
165 !print *, 'location outside of grid: ', sample_lons(i), sample_lats(j)
166 interp_data(i, j) = missing_r8
167 cycle
168 endif
169 if (debug > 0) print *, i, j, four_lons(1), four_lons(3), four_lats(1), four_lats(3), sample_lons(i), sample_lats(j)
170
171 ! get values of data at lon/lat bot/top indices, counterclockwise around quad
172 do k=1, 4
173 invals(k) = grid_data(four_lons(k), four_lats(k))
174 enddo
175
176 call quad_lon_lat_evaluate(h, sample_lons(i), sample_lats(j), four_lons, four_lats, &
177 invals, outval, istatus)
178
179 interp_data(i, j) = outval
180
181 enddo
182enddo
183
184! this program doesn't currently have any missing locations - but i'll test that next.
185if (debug > 0) print *, 'number of missing values in input data: ', count(grid_data(:,:) == missing_r8)
186if (debug > 0) print *, 'number of missing values in output data: ', count(interp_data(:,:) == missing_r8)
187
188call writeit_2d('data_lons_2d_irreg_test.txt', nx, ny, data_lons)
189call writeit_2d('data_lats_2d_irreg_test.txt', nx, ny, data_lats)
190call writeit_2d('data_data_2d_irreg_test.txt', nx, ny, grid_data)
191
192call writeit_1d('sample_lons_1d_irreg_test.txt', nrx, sample_lons)
193call writeit_1d('sample_lats_1d_irreg_test.txt', nry, sample_lats)
194call writeit_2d('sample_data_2d_irreg_test.txt', nrx, nry, interp_data)
195
196call finalize_quad_interp(h)
197call finalize_utilities('test_quad_irreg_interp')
198
199if (debug > 0) print *, 'closed files and finalized interp handle'
200
201contains

203!------------------------------------------------------------
204! rotate vector a counterclockwise by angle theta, relative
205! to the given origin point.
206
207subroutine rotate(x, y, theta, x0, y0)
208 real(r8), intent(inout) :: x, y
209 real(r8), intent(in) :: theta
210 real(r8), intent(in) :: x0, y0
211
212real(r8) :: a(2), b(2)
213real(r8) :: r(2,2)
214real(r8) :: rads
215
216a(1) = x - x0
217a(2) = y - y0
218
219rads = theta * deg2rad
220
221r(1,1) = cos(rads)
222r(1,2) = sin(rads)
223r(2,1) = sin(-rads)
224r(2,2) = cos(rads)
225
226b(1) = r(1,1)*a(1) + r(1,2)*a(2)
227b(2) = r(2,1)*a(1) + r(2,2)*a(2)
228
229x = b(1) + x0
230y = b(2) + y0
231
232end subroutine rotate

234!------------------------------------------------------------
235! compute +/- a random value based on a width and percentage
236! of that width
237
238function deform(width, fraction, seq)
239
240use random_seq_mod
241
242 real(r8), intent(in) :: width
243 real(r8), intent(in) :: fraction
244 type(random_seq_type), intent(inout) :: seq
245 real(r8) :: deform
246
247real(r8) :: val
248
249! random val between -1 and 1
250val = (random_uniform(seq) * 2.0_r8) - 1.0_r8
251
252deform = val * width * fraction
253
254end function deform
