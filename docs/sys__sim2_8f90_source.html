5program sys_sim2
6
7! This one is described in notes from 14 Dec. or thereabout, 2001. Sample the
8! uncertainty in the correlation coefficient due to sample size and approximate
9! the resulting distribution in terms of a Gaussian (mean and variance). Probably
10! want to evaluate how close to Gaussian this is, too.
11
12! See notes from first two weeks of December, 2001. This program begins attempts
13! to analyze the value of particular observations. Here, we begin by trying to
14! determine the value of observations with a given correlation to a state variable
15! using an N member ensemble to compute the correlations.
16
17use random_seq_mod, only : random_seq_type, init_random_seq, random_gaussian, &
18 twod_gaussians, random_uniform
19
20implicit none
21
22type (random_seq_type) :: r
23double precision, allocatable :: rnum(:, :)
24double precision :: c(2, 2), sum_err_var, sigma_y_p, sigma_y_o, sigma_x_p
25double precision :: y_truth, y_o, sample_correl, reg_coef, sample_reg_coef
26double precision :: sigma_y_u, y_u, sigma_x_u, x_u, sample_x_u, x_error
27double precision :: x_error_var, total_err_var, mean(2), correl, est_err_var
28double precision :: temp_sum, pe_2_sum, xx, s_xx, s_xx_2, xx_mean, xx_var
29double precision, allocatable :: x_mean(:), x_var(:)
30integer :: n, i, j, n_samples, index
31
32! Initialize repeatable random sequence
33call init_random_seq(r)
34
35! Initialize error accumulation for averaging
36sum_err_var = 0.0
37mean = 0.0; temp_sum = 0.0; pe_2_sum = 0.0;
38
39! For now have a set of free parameters that may be too large
40write(*, *) 'Input prior variance of observation variable'
41read(*, *) sigma_y_p
42write(*, *) 'Input variance of observing instrument'
43read(*, *) sigma_y_o
44write(*, *) 'Input prior variance of state variable'
45read(*, *) sigma_x_p
46write(*, *) 'Input ensemble size'
47read(*, *) n
48write(*, *) 'input an expected correlation'
49read(*, *) correl
50
51! Set up covariance matrix for computing the sample correlation
52c(1, 1) = 1.0; c(2, 2) = 1.0; c(1, 2) = correl; c(2, 1) = correl
53
54! Allocate storage for doing the correlation
55allocate(rnum(2, n))
56
57write(*, *) 'Input number of samples for statistics'
58read(*, *) n_samples
59allocate(x_mean(n_samples), x_var(n_samples))
60
61! Loop through the number of samples
62do i = 1, n_samples
63
64! Produce a sample of an observation yo by getting truth from prior
65! and adding on sample from observatoinal error distribution
66 y_truth = random_gaussian(r, dble(0.0), sqrt(sigma_y_p))
67 y_o = y_truth + random_gaussian(r, dble(0.0), sqrt(sigma_y_o))
68
69
70
71y_o = 0.4
72
73
74
75! Use random sample to generate erroneous sample correlation
76! Generate n pairs of numbers from this distribution
77 do j = 1, n
78 call twod_gaussians(r, mean, c, rnum(:, j))
79 end do
80
81! Compute the sample correlation
82 call comp_correl(rnum, n, sample_correl)
83
84! Compute regression coefficient, erroneous regression coefficient
85 reg_coef = correl * sqrt(sigma_x_p) / sqrt(sigma_y_p)
86 sample_reg_coef = sample_correl * sqrt(sigma_x_p) / sqrt(sigma_y_p)
87
88! Compute the updated covariance and mean for y given this observation
89 sigma_y_u = 1.0 / (1.0 / sigma_y_p + 1.0 / sigma_y_o)
90 y_u = sigma_y_u * (0.0 + y_o / sigma_y_o)
91
92! Compute updated covariance for x
93! See notes from 11 Dec. for sigma_x_u computation
94 sigma_x_u = sigma_x_p - correl**2 * sigma_x_p * sigma_y_p &
95 / (sigma_y_o + sigma_y_p)
96
97! Compute correct x_u and sample x_u
98 x_u = reg_coef * y_u
99 sample_x_u = sample_reg_coef * y_u
100
101! Compute the mean and variance for this particular update
102 x_mean(i) = sample_x_u
103 x_var(i) = sigma_x_p - sample_correl**2 * sigma_y_p &
104 / (sigma_y_o + sigma_y_p)
105
106! Compute temporary sum to evaluate partial analytic solution
107 temp_sum = temp_sum + (sample_correl - correl)**2 * y_o**2
108 pe_2_sum = pe_2_sum + (sample_correl - correl)**2
109
110! Compute error in x
111 x_error = sample_x_u - x_u
112 x_error_var = x_error **2
113
114! Total expected variance for this one is x_error_var plus sigma_x_u
115 total_err_var = x_error_var + sigma_x_u
116
117! Accumulate this for averaging
118 sum_err_var = sum_err_var + total_err_var
119
120end do
121
122
123! Output the expected value of the mean x
124write(*, *) 'Expected updated mean is ', sum(x_mean) / n_samples
125
126! Now compute the expected value of the updated variance by doing sampling
127! The distribution is a sum of Gaussians with means and variance given
128! by x_mean and x_var; each is equally likely. So, do a sample, randomly
129! select a 'kernel', then randomly select a value from it.
130s_xx = 0.0
131s_xx_2 = 0.0
132do i = 1, 10 * n_samples
133 index = random_uniform(r) * n_samples + 1
134 if(index < 1 .or. index > n_samples) then
135 write(*, *) 'bad index ', i, index
136 stop
137 end if
138 if(i / 500000 * 500000 == i) then
139 write(*, *) 'mean, var ', i, index, real(x_mean(index)), real(x_var(index))
140 endif
141 xx = random_gaussian(r, x_mean(index), sqrt(x_var(index)))
142 s_xx = s_xx + xx
143 s_xx_2 = s_xx_2 + xx**2
144end do
145
146xx_mean = s_xx / (10.0 * n_samples)
147write(*, *) 'Sampled mean is ', xx_mean
148write(*, *) 'Expected mean is ', x_u
149xx_var = (s_xx_2 - (10.0 * n_samples) * xx_mean**2) / &
150 (10.0 * n_samples - 1.0)
151write(*, *) 'Sampled variance is ', xx_var
152write(*, *) 'Expected raw var is ', sigma_x_u
153write(*, *) 'Mean variance is ', s_xx_2 / (10.0 * n_samples)
154
155
156
157
158if(1 == 1) stop
159
160! Output the mean total_err_var
161write(*, *) '----------'
162write(*, *) 'Large N Mean total_err_var is ', sigma_x_u
163write(*, *) 'Expected sample x variance is ', sum_err_var / n_samples
164write(*, *) '----------'
165
166! Using sample for product of correlation and y_0_2
167write(*, *) 'Expected error variance is ', &
168 (temp_sum / n_samples )*(sigma_x_p / sigma_y_p) * (sigma_y_u / sigma_y_o)**2
169
170! Estimate using sample only for correlation
171write(*, *) 'Expected error variance, only sample correl ', &
172 sigma_x_p * sigma_y_p * (pe_2_sum / n_samples) / (sigma_y_p + sigma_y_o)
173
174est_err_var = ((1.0 - correl**2)**2 / (n - 1.)) * sigma_x_p * sigma_y_p / &
175 (sigma_y_p + sigma_y_o)
176write(*, *) 'Estimated correl expected variance is ', est_err_var
177write(*, *) '----------'
178
179write(*, *) 'pe_2_mean ', pe_2_sum / n_samples
180write(*, *) 'expected value of pe_2_mean ', (1.0 - correl**2)**2 / (n - 1.)
181
182write(*, *) '----------'
183write(*, *) 'net improvement ', (sigma_x_p - sum_err_var / n_samples) / &
184 (sigma_x_p - sigma_x_u)
185write(*, *) 'Computed net improvement ', 1. - (pe_2_sum / n_samples) / correl**2
186
187end program sys_sim2

191subroutine comp_correl(ens, n, correl)
192
193implicit none
194
195integer, intent(in) :: n
196double precision, intent(in) :: ens(2, n)
197double precision, intent(out) :: correl
198double precision :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
199
200
201sum_x = sum(ens(2, :))
202sum_y = sum(ens(1, :))
203sum_xy = sum(ens(2, :) * ens(1, :))
204sum_x2 = sum(ens(2, :) * ens(2, :))
205
206! Computation of correlation
207sum_y2 = sum(ens(1, :) * ens(1, :))
208
209correl = (n * sum_xy - sum_x * sum_y) / &
210 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
211
212end subroutine comp_correl
