861SUBROUTINE read_obserr (filein, platform, sensor_name, err, nlevels)
862!------------------------------------------------------------------------------!
863!
864! Read observational error on pressure levels (in hPa)
865!
866!------------------------------------------------------------------------------!
867 IMPLICIT NONE
868!------------------------------------------------------------------------------!
869
870 CHARACTER (LEN=80), intent(in) :: filein != 'obserr.txt'
871 CHARACTER (LEN=80), intent(in) :: platform != 'RAOBS'
872 CHARACTER (LEN=80), intent(in) :: sensor_name != 'WIND SENSOR ERRORS'
873 INTEGER, intent(in) :: nlevels
874
875 REAL(r8), intent(out) :: err(nlevels)
876
877 character(len=129) :: msgstring
878 INTEGER :: io_error, i, iunit, ncol
879 CHARACTER (LEN=80) :: fmt_err != '(5(1X,F5.1))'
880 CHARACTER (LEN=80) :: line1, line2, line3, line4
881 LOGICAL :: sensor_found, found
882
883!------------------------------------------------------------------------------!
884
885! 1. OPEN INPUT FILE
886! ===================
887
888 iunit = get_unit()
889 OPEN (unit = iunit , file = filein , form = 'FORMATTED' , &
890 action = 'READ' , status = 'OLD', iostat = io_error)
891
892 IF (io_error /= 0) THEN
893 write(msgstring, *) 'Unable to open input observational error file ',trim(filein)
894 call error_handler(e_err,'READ_OBSERR', &
895 msgstring, source, revision, revdate)
896 ENDIF
897
898
899! 2. READ DATA
900! =============
901
902! Read file until sensor_name is found
903 found = .false.
904 sensor_found = .false.
905 io_error= 0
906
907 DO WHILE (io_error == 0.)
908
909 READ (unit = iunit, iostat = io_error, fmt = '(A)') line1
910 ! Exit when error or at end of file
911
912 IF (io_error /= 0 .OR. (line1(1:2) == '*.')) EXIT
913
914 IF (trim(sensor(line1)) == 'BOGUS') THEN
915
916 IF (trim(obstype(line1)) == trim(sensor_name)) sensor_found = .true.
917
918 ENDIF
919
920 IF (sensor_found .and. (trim(sensor(line1)) == trim(platform))) THEN
921
922 found = .true.
923 READ (unit = iunit, iostat = io_error, fmt = '(A)') line2
924 READ (unit = iunit, iostat = io_error, fmt = '(A)') line3
925 IF (trim(sensor_name) == 'WIND SENSOR ERRORS') &
926 READ (unit = iunit, iostat = io_error, fmt = '(A)') line4
927 EXIT
928
929 ELSE
930
931 ! If obstype is not found, keep on reading
932
933 cycle
934
935 ENDIF
936
937 ENDDO
938
939 IF (found) THEN
940
941 ! Sensor_name has been found, Error at mandatory pressure levels follow
942 ! Break down data upon obs type
943
944 SELECT CASE (trim(sensor_name))
945 CASE ('WIND SENSOR ERRORS')
946 ncol = 7
947 fmt_err = '(7(1X,F5.1))'
948 CASE ('RH SENSOR ERRORS')
949 ncol = 5
950 fmt_err = '(5(1X,F5.2))'
951 CASE DEFAULT
952 ncol = 5
953 fmt_err = '(5(1X,F5.1))'
954 END SELECT
955
956 READ (line1, fmt_err) (err(i), i = 1, ncol)
957 READ (line2, fmt_err) (err(i), i = ncol+1, 2*ncol)
958 READ (line3, fmt_err) (err(i), i = 2*ncol+1, 3*ncol)
959 ! Winds are given over 4 lines, any other data are given over 3 lines
960 IF (trim(sensor_name) == 'WIND SENSOR ERRORS') &
961 READ (line4, fmt_err) (err(i), i = 3*ncol+1, min(4*ncol,nlevels))
962
963 ELSE
964
965 WRITE (*, fmt = '(/,A,A,A,/)') trim(sensor_name), &
966 ' were not found in file ', trim(filein)
967
968 ENDIF

970
971! 3. CLOSE INPUT FILE
972! ====================
973
974 CLOSE (unit = iunit)

976END SUBROUTINE read_obserr
