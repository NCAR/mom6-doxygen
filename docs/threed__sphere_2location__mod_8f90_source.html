5!> Implements location interfaces for a three dimensional spherical shell
6!> with a choice of vertical coordinates.
7!> Horizontal coordinates are always latitude and longitude.
8!> Vertical coordinate choices include pressure, height, model level,
9!> scale height, surface, and non-specific (column-integrated values, or
10!> with no logically defined vertical location, e.g. hurricane vortex center)
11!> The internal representation of the location is stored as
12!> radians from 0 to 2 PI for longitude and -PI/2 to PI/2 for latitude to
13!> minimize computational cost for distances. However, the external
14!> representation is longitude in degrees from 0 to 360 and latitude
15!> from -90 to 90 for consistency with most applications in the field.
16!>
17!> This version supports multiple cutoff distances in an efficient manner.
18!> Smaller cutoff values will do less searching than larger ones. (This was
19!> not true in earlier implementations of this code.)
20!>

252module_initialized = .true.
253
254! give these initial values before reading them from the namelist.

261
262! Read the namelist entry
263call find_namelist_in_file("input.nml", "location_nml", iunit)

268

273 source)

277 write(msgstring, '(A,I5,A,I5)') 'from the namelist, nlon is ', nlon, '; nlat is ', nlat
278 write(msgstring2, '(A)') 'both must be >= 1, and nlon must be odd'

283if((nlon / 2) * 2 == nlon) then
284 write(msgstring, '(A,I5,A)') 'nlon is ', nlon, '. Must be odd'
285 call error_handler(e_err, 'location_mod', msgstring, source)
286endif
287
288! Copy the normalization factors in the vertical into an array
289vert_normalization(vertislevel) = vert_normalization_level
290vert_normalization(vertispressure) = vert_normalization_pressure
291vert_normalization(vertisheight) = vert_normalization_height
292vert_normalization(vertisscaleheight) = vert_normalization_scale_height
293
294! If the user has set a namelist to make different vertical normalization factors
295! when computing the localization distances, allocate and set that up here.
296! It overrides the defaults above.
297
298if (special_vert_normalization_obs_types(1) /= 'null' .or. &
299 special_vert_normalization_pressures(1) /= missing_r8 .or. &
300 special_vert_normalization_heights(1) /= missing_r8 .or. &
301 special_vert_normalization_levels(1) /= missing_r8 .or. &
302 special_vert_normalization_scale_heights(1) /= missing_r8) then
303

308 allocate(per_type_vert_norm(vert_type_count, typecount))

313 per_type_vert_norm(vertisheight, :) = vert_normalization_height
314 per_type_vert_norm(vertisscaleheight, :) = vert_normalization_scale_height

331 call error_handler(e_err,'location_mod:', msgstring, source, &

343 write(msgstring, *) 'one or more special vertical normalization values is uninitialized.'

345 'special vert normalization value namelist requires all 4 values per type', &
346 source, text2=trim(msgstring))
347 endif
348
349endif
350
351! if the namelist control says we are only basing the localization on
352! distances in the horizontal, log that in the dart log file.
353if (horiz_dist_only) then
354 call error_handler(e_msg,'location_mod:', &
355 'Ignoring vertical separation when computing distances; horizontal distances only', &
356 source)
357else
358 call error_handler(e_msg,'location_mod:', &
359 'Including vertical separation when computing distances:', source)

361 call error_handler(e_msg,'location_mod:',msgstring,source)

364 write(msgstring,'(A,f17.5)') ' # model levels ~ 1 horiz radian: ', vert_normalization_level

367 call error_handler(e_msg,'location_mod:',msgstring,source)

374 (per_type_vert_norm(vertisheight, i) /= vert_normalization_height) .or. &
375 (per_type_vert_norm(vertisscaleheight, i) /= vert_normalization_scale_height)) then

377 write(msgstring,'(2A)') 'Altering default vertical normalization for type ', trim(get_name_for_type_of_obs(i))

383 endif

386 per_type_vert_norm(vertisheight, i)
387 call error_handler(e_msg,'location_mod:',msgstring,source)

401 endif
402endif

408 do i = -sincos_limit, sincos_limit

423function get_dist(loc1, loc2, type1, kind2, no_vert)
424
425! returns the distance between 2 locations in units of radians.
426! Distance depends on only horizontal distance or both horizontal and
427! vertical distance.
428! The choice is determined by horiz_dist_only and the which_vert of loc1.
429! May want to allow return of some joint distance in the long run?
430! Or just a distance that is a function of all 3 things.
431! The namelist controls whether default computations use just horizontal distance.
432! However, this behavior can be over-ridden by the no_vert optional argument.
433! If set to false, this will always do full 3d distance if possible. If set to
434! true it will never do the full 3d distance. At present asking to do a vertical
435! distance computation for incompatible vertical location units results
436! in a fatal error unless one of the vertical units is UNDEFINED.
437
438! CHANGE from previous versions: the 3rd argument is now a specific type
439! (e.g. RADIOSONDE_TEMPERATURE, AIRCRAFT_SPECIFIC_HUMIDITY) associated
440! with loc1, while the 4th argument is a generic kind (QTY_TEMPERATURE,
441! QTY_U_WIND_COMPONENT) associated with loc2.
442! The type and kind are part of the interface in case user-code wants to do
443! a more sophisticated distance calculation based on the base type or target
444! kind. In the usual case this code still doesn't use the kind/type, but
445! it does require at least type1 if using per-type vertical normalization.
446
447type(location_type), intent(in) :: loc1, loc2
448integer, optional, intent(in) :: type1, kind2
449logical, optional, intent(in) :: no_vert
450real(r8) :: get_dist
451
452real(r8) :: lon_dif, vert_dist, rtemp
453integer :: lat1_ind, lat2_ind, lon_ind, temp ! indexes into lookup tables
454logical :: comp_h_only
455
456
457if ( .not. module_initialized ) call initialize_module()
458
459! Begin with the horizontal distance
460! Compute great circle path shortest route between two points
461lon_dif = loc1%lon - loc2%lon
462
463if(approximate_distance) then
464 ! Option 1: Use table lookup; faster but less accurate
465 lat1_ind = int(loc1%lat*sincos_delta)
466 lat2_ind = int(loc2%lat*sincos_delta)
467 lon_ind = int(lon_dif *sincos_delta)
468 temp = int(acos_delta * (my_sin(lat2_ind) * my_sin(lat1_ind) + &
469 my_cos(lat2_ind) * my_cos(lat1_ind) * my_cos(lon_ind)))
470 get_dist = my_acos(temp)
471else
472 ! Option 2: Use pre-defined trig functions: accurate but slow
473 ! First 2 ifs avoids round-off error that can kill acos;
474 if(abs(loc1%lat) >= pi/2.0_r8 .or. abs(loc2%lat) >= pi/2.0_r8 .or. &
475 lon_dif == 0.0_r8) then
476 get_dist = abs(loc2%lat - loc1%lat)
477 else
478 ! This test is for apparent roundoff error which may be a result of
479 ! running r8 == r4.
480 rtemp = sin(loc2%lat) * sin(loc1%lat) + &
481 cos(loc2%lat) * cos(loc1%lat) * cos(lon_dif)
482 if (rtemp < -1.0_r8) then
483 get_dist = pi
484 else if (rtemp > 1.0_r8) then
485 get_dist = 0.0_r8
486 else
487 get_dist = acos(rtemp)
488 endif
489 endif
490endif
491
492! Now compute a vertical distance if requested. Highest priority is
493! the optional no_vert argument, so test it first.
494if(present(no_vert)) then
495 comp_h_only = no_vert
496! Namelist horizontal only has second highest priority
497else
498 comp_h_only = horiz_dist_only
499endif
500
501! If which_vert has no vertical definition for either location do only horizontal
502if(loc1%which_vert == vertisundef .or. loc2%which_vert == vertisundef) comp_h_only = .true.
503! If both verts are surface, do only horizontal
504if(loc1%which_vert == vertissurface .and. loc2%which_vert == vertissurface) comp_h_only = .true.
505
506! Add in vertical component if required
507if(.not. comp_h_only) then
508 ! Vert distance can only be done for like vertical locations units
509 if(loc1%which_vert /= loc2%which_vert) then
510 write(msgstring,*)'loc1%which_vert (',loc1%which_vert, &
511 ') /= loc2%which_vert (',loc2%which_vert,')'
512 call error_handler(e_msg, 'get_dist', msgstring, source)
513 call write_location(logfileunit,loc1)
514 call write_location(logfileunit,loc2)
515 call write_location(0,loc1)
516 call write_location(0,loc2)
517 call error_handler(e_err, 'get_dist', &
518 'Dont know how to compute vertical distance for unlike vertical coordinates', &
519 source)
520 endif
521
522 ! Compute the difference and divide by the appropriate normalization factor
523 ! Normalization factor computes relative distance in vertical compared to one radian
524 ! This is new - if per-type localization distances given, use the specific type of loc1
525 ! to determine the vertical mapping distance. it defaults to the 4 standard ones,
526 ! but can be specified separately if desired.
527
528 ! note that per-type vertical conversion factors are a global here, and were set
529 ! by namelist values. they apply to all calls to get_dist() based on the obs type.
530 if (allocated(per_type_vert_norm)) then

541 get_dist = sqrt(get_dist**2 + vert_dist**2)
542endif

737subroutine write_location(locfile, loc, fform, charstring)
738
739! Writes a location to a file.
740! most recent change: adding the optional charstring option. if present,
741! locfile is ignored, and a pretty-print formatting is done into charstring.
742! the locations are converted to lat/lon, and the vert is put into more
743! common units (e.g. hPa, km, etc)
744
745integer, intent(in) :: locfile
746type(location_type), intent(in) :: loc
747character(len = *), intent(in), optional :: fform
748character(len = *), intent(out), optional :: charstring
749
750integer :: charlength
751logical :: writebuf
752character(len=129) :: string1
753
754! 10 format(1x,3(f22.14,1x),i4) ! old
75510 format(1x,3(g25.16,1x),i2)
756
757if ( .not. module_initialized ) call initialize_module()
758
759! writing to a file (normal use) or to a character buffer?
760writebuf = present(charstring)
761
762! output file; test for ascii or binary, write what's asked, and return
763if (.not. writebuf) then
764 if (ascii_file_format(fform)) then
765 write(locfile, '(''loc3d'')' )
766 write(locfile, 10) loc%lon, loc%lat, loc%vloc, loc%which_vert
767 else
768 write(locfile) loc%lon, loc%lat, loc%vloc, loc%which_vert
769 endif
770 return
771endif
772
773! you only get here if you're writing to a buffer and not
774! to a file, and you can't have binary format set.

777 'Cannot use string buffer with binary format', source)
778endif
779
780! format the location to be more human-friendly; meaning
781! degrees instead of radians, and kilometers for height,
782! hectopascals instead of pascals for pressure, etc.
783
784! this must be the sum of the longest of the formats below.
785charlength = 72

787if (len(charstring) < charlength) then

789 call error_handler(e_err, 'write_location', msgstring, source)

796! then pretty up the vertical choices, trying to get them to line up in
797! case the caller is listing out locations with different vert units.
798! concatinate the vertical on the end of the horizontal and put it all
799! into the return string.
800select case (loc%which_vert)
801 case (vertisundef)
802 write(charstring, '(A,A)') trim(string1), ' Undefined'
803 case (vertissurface)
804 write(charstring, '(A,F13.5,A)') trim(string1), loc%vloc, ' surface (m)'
805 case (vertislevel)
806 write(charstring, '(A,F13.6,A)') trim(string1), loc%vloc, ' level'
807 case (vertispressure)
808 write(charstring, '(A,F13.7,A)') trim(string1), loc%vloc / 100.0_r8, ' hPa'
809 case (vertisheight)
810 write(charstring, '(A,F13.7,A)') trim(string1), loc%vloc / 1000.0_r8, ' km'
811 case (vertisscaleheight)
812 write(charstring, '(A,F13.7,A)') trim(string1), loc%vloc, ' scale ht'
813 case default
814 write(msgstring, *) 'unrecognized key for vertical type: ', loc%which_vert
815 call error_handler(e_err, 'write_location', msgstring, source)
816end select

819end subroutine write_location

1195else
1196 ! no per-type settings, everyone uses same distance
1197 gc%gtt(1)%maxdist = maxdist
1198endif
1199
1200if (present(maxdist_list)) deallocate(distlist) ! temp storage
1201
1202! Allocate the storage for the grid dependent boxes
1203do i=1, gc%nt
1204 allocate(gc%gtt(i)%count(nlon, nlat), gc%gtt(i)%start(nlon, nlat))
1205 allocate(gc%gtt(i)%lon_offset(nlat, nlat))
1206 gc%gtt(i)%lon_offset = -1
1207 gc%gtt(i)%count = -1
1208 gc%gtt(i)%start = -1
1209enddo
1210
1211! store the location counts in all derived types
1212do i=1, gc%nt
1213 gc%gtt(i)%num = num
1214enddo
1215
1216! If there are no locs to operate on, no point in going any further.
1217if (num == 0) return
1218
1219do i=1, gc%nt
1220 ! Allocate storage for locs number dependent part
1221 allocate(gc%gtt(i)%loc_box(num))
1222 gc%gtt(i)%loc_box(:) = -1
1223enddo
1224
1225do n=1, gc%nt
1226 ! Determine where the boxes should be for this set of locs and maxdist
1227 call find_box_ranges(gc%gtt(n), num, locs)
1228
1229 ! Figure out which boxes are close to a box on a given latitude circle
1230 ! MIGHT AVOID DOING THIS WITH A COPY ROUTINE: HAVE SAME BOXES OFTEN
1231 do blat_ind = 1, nlat
1232 ! Search from east side of base block
1233 ! Start searching out, have to look for closest point in box being checked
1234 ! Only have to search latitude boxes that are within maximum distance
1235 bot_tlat_ind = blat_ind - floor(gc%gtt(n)%maxdist / gc%gtt(n)%lat_width) - 1
1236 if(bot_tlat_ind < 1) bot_tlat_ind = 1
1237 top_tlat_ind = blat_ind + floor(gc%gtt(n)%maxdist / gc%gtt(n)%lat_width) + 1
1238 if(top_tlat_ind > nlat) top_tlat_ind = nlat
1239 do tlat_ind = bot_tlat_ind, top_tlat_ind
1240 ! Spherical geometry can be tricky
1241 ! We want to find the MINIMUM distance between two lat/lon boxes
1242 ! This distance is known to be corners of the boxes. It is also known
1243 ! to be between the corners such that the longitude difference between
1244 ! the corners is a minimum. HOWEVER, determining whether it is between
1245 ! the closest latitudes or not is a non-trivial computation. Hence,
1246 ! since this isn't done much, we just check all four possible combinations
1247 ! of latitude and pick the one that gives the closest distance.
1248 do j = 1, 2
1249 base_lat(j) = gc%gtt(n)%bot_lat + (blat_ind - 2 + j) * gc%gtt(n)%lat_width
1250 target_lat(j) = gc%gtt(n)%bot_lat + (tlat_ind - 2 + j) * gc%gtt(n)%lat_width
1251 end do
1252
1253 ! If the max distance > PI, then everybody is close.
1254 ! Do a test for something slightly less than PI to avoid round-off error.
1255 ! Set max_del_lon to something much larger than PI since it doesn't matter.
1256 if(gc%gtt(n)%maxdist > pi - 0.0001_r8) then
1257 max_del_lon = 2.0 * pi
1258 else
1259 ! Find the maximum longitude offset for the different possible latitudes edges
1260 max_del_lon = 0.0_r8
1261 do tj = 1, 2
1262 do bj = 1, 2
1263 ! Compute the lon offset directly by inverting distance
1264 cos_del_lon = (cos(gc%gtt(n)%maxdist) - sin(base_lat(bj)) * sin(target_lat(tj))) / &
1265 (cos(base_lat(bj)) * cos(target_lat(tj)))
1266 if(cos_del_lon < -1.0_r8) then
1267 del_lon = pi
1268 else if(cos_del_lon > 1.0_r8) then
1269 del_lon = 0.0_r8
1270 else
1271 del_lon = acos(cos_del_lon)
1272 endif
1273 if(del_lon > max_del_lon) max_del_lon = del_lon
1274
1275 end do
1276 end do
1277 endif
1278
1279 ! Compute the number of boxes to search in longitude for maximum del_lon
1280 gc%gtt(n)%lon_offset(blat_ind, tlat_ind) = floor(max_del_lon / gc%gtt(n)%lon_width) + 1
1281 ! Watch for roundoff leading to a search of more offsets than exist
1282 if(gc%gtt(n)%lon_offset(blat_ind, tlat_ind) > nlon / 2) &
1283 gc%gtt(n)%lon_offset(blat_ind, tlat_ind) = nlon / 2
1284
1285 end do
1286 end do
1287
1288 ! Begin by computing the number of locations in each box in lat/lon
1289 gc%gtt(n)%count = 0
1290 do i = 1, num
1291 lon_box(i) = get_lon_box(gc%gtt(n), locs(i)%lon)
1292 if(lon_box(i) < 0 .or. lon_box(i) > nlon) then
1293 write(msgstring, *) 'Contact Dart Developers: this error should not happen'
1294 call error_handler(e_msg, 'get_close_init', msgstring, source)
1295 write(msgstring, *) 'location outside grid boxes, index value:', lon_box(i)
1296 call error_handler(e_err, 'get_close_init', msgstring, source)
1297 endif
1298
1299 lat_box(i) = floor((locs(i)%lat - gc%gtt(n)%bot_lat) / gc%gtt(n)%lat_width) + 1
1300 if(lat_box(i) > nlat) lat_box(i) = nlat
1301 if(lat_box(i) < 1) lat_box(i) = 1
1302
1303 gc%gtt(n)%count(lon_box(i), lat_box(i)) = gc%gtt(n)%count(lon_box(i), lat_box(i)) + 1
1304 end do
1305
1306 ! Figure out where storage for each boxes members should begin
1307 cum_start = 1
1308 do i = 1, nlon
1309 do j = 1, nlat
1310 gc%gtt(n)%start(i, j) = cum_start
1311 cum_start = cum_start + gc%gtt(n)%count(i, j)
1312 end do
1313 end do
1314
1315 ! Now we know how many are in each box, get a list of which are in each box
1316 tstart = gc%gtt(n)%start
1317 do i = 1, num
1318 gc%gtt(n)%loc_box(tstart(lon_box(i), lat_box(i))) = i
1319 tstart(lon_box(i), lat_box(i)) = tstart(lon_box(i), lat_box(i)) + 1
1320 end do
1321
1322end do
1323
1324! info on how well the boxes are working. by default print nothing.
1325! set print_box_level to higher values to get more and more detail.
1326! user info should be level 1; 2 and 3 should be for debug only.
1327! special for grid-decomposition debugging; set print level to -8.
1328if (output_box_info) then
1329 ! if this task normally prints, call the print routine.
1330 ! if print level > 2, set all tasks to print and call print.
1331 ! then reset the status to off again.
1332 if (do_output()) then
1333 call print_get_close_type(gc, 1, print_box_level)
1334 else if (print_box_level >= 2 .or. print_box_level < 0) then
1335 ! print status was false, but turn on temporarily
1336 ! to output box info from all tasks.
1337 call set_output(.true.)
1338 call print_get_close_type(gc, 1, print_box_level)
1339 call set_output(.false.)
1340 endif
1341endif
1342
1343end subroutine get_close_init

1409subroutine get_close(gc, base_loc, base_type, locs, loc_qtys, &
1410 num_close, close_ind, dist, ens_handle)
1411
1412! The specific type of the base observation, plus the generic kinds list
1413! for either the state or obs lists are available if a more sophisticated
1414! distance computation is needed.
1415
1416type(get_close_type), intent(in) :: gc
1417type(location_type), intent(inout) :: base_loc, locs(:)
1418integer, intent(in) :: base_type, loc_qtys(:)
1419integer, intent(out) :: num_close, close_ind(:)
1420real(r8), optional, intent(out) :: dist(:)
1421type(ensemble_type), optional, intent(in) :: ens_handle
1422
1423! If dist is NOT present, just find everybody in a box, put them in the list,
1424! but don't compute any distances
1425
1426integer :: lon_box, lat_box, i, j, k, n_lon, lon_ind, n_in_box, st, t_ind, bt
1427real(r8) :: this_dist, this_maxdist
1428
1429! Variables needed for comparing against correct case
1430integer :: cnum_close, cclose_ind(size(locs))
1431real(r8) :: cdist(size(locs))
1432
1433
1434! First, set the intent out arguments to a missing value
1435num_close = 0
1436close_ind = -99
1437if(present(dist)) dist = -99.0_r8
1438this_dist = 999999.0_r8 ! something big.
1439
1440! handle identity obs correctly -- only support them if there are no
1441! per-obs-type cutoffs. identity obs don't have a specific type, they
1442! only have a generic kind based on what index into the state vector is
1443! specified. if this is absolutely needed by someone, i can rejigger the
1444! code to try to use the default cutoff for identity obs, but it's tricky
1445! to identify which obs type is using the default. in theory it's possible
1446! to specify a default cutoff and then specify a per-type cutoff for every
1447! other type in the system. in that case i don't have a map entry for the
1448! default, and it's a pain to construct one. so i'm punting for now.
1449if (base_type < 0) then
1450 if (gc%nt > 1) then
1451 write(msgstring, '(A)') 'no support for identity obs if per-obs-type cutoffs are specified'
1452 write(msgstring1, '(A)') 'contact dart support if you have a need for this combination'
1453 call error_handler(e_err, 'get_close', msgstring, source, text2=msgstring1)
1454 endif
1455 bt = 1
1456else
1457 ! map from type index to gtt index
1458 if (base_type < 1 .or. base_type > size(gc%type_to_cutoff_map)) then
1459 write(msgstring,'(A,I8)')'base_type out of range, is ', base_type
1460 write(msgstring1,'(A,2I8)')'must be between ', 1, size(gc%type_to_cutoff_map)
1461 call write_location (0, base_loc, charstring=msgstring2)
1462 call error_handler(e_err, 'get_close', msgstring, source, &
1463 text2=msgstring1, text3=msgstring2)
1464 endif
1465 bt = gc%type_to_cutoff_map(base_type)
1466 if (bt < 1 .or. bt > gc%nt) then
1467 write(msgstring,'(A,I8)')'mapped type index out of range, is ', bt
1468 write(msgstring1,'(A,2I8)')'must be between ', 1, gc%nt
1469 write(msgstring2, '(A)')'internal error, should not happen. Contact DART Support'
1470 call error_handler(e_err, 'get_close', msgstring, source, &
1471 text2=msgstring1, text3=msgstring2)
1472 endif
1473endif
1474
1475! the list of locations in the locs() argument must be the same
1476! as the list of locations passed into get_close_init(), so
1477! gc%gtt(bt)%num and size(locs) better be the same. if the list
1478! changes you have to destroy the old gc and init a new one.
1479if (size(locs) /= gc%gtt(bt)%num) then
1480 write(msgstring,'(A)')'locs() array must match one passed to get_close_init()'
1481 write(msgstring1,'(A,2I8)')'init size, current list size: ', gc%gtt(bt)%num, size(locs)
1482 write(msgstring2,'(A,I8)')'bt = ', bt
1483 call error_handler(e_err, 'get_close', msgstring, source, &
1484 text2=msgstring1, text3=msgstring2)
1485endif
1486
1487! If num == 0, no point in going any further.
1488if (gc%gtt(bt)%num == 0) return
1489
1490
1491! local variable for what the maxdist is in this particular case.
1492this_maxdist = gc%gtt(bt)%maxdist
1493
1494!--------------------------------------------------------------
1495! For validation, it is useful to be able to compare against exact
1496! exhaustive search
1497if(compare_to_correct) then
1498 cnum_close = 0
1499 do i = 1, gc%gtt(bt)%num
1500 if (locs(i)%which_vert /= base_loc%which_vert) then
1501 this_dist = get_dist(base_loc, locs(i), base_type, loc_qtys(i), &
1502 no_vert = .true.)
1503 else
1504 this_dist = get_dist(base_loc, locs(i), base_type, loc_qtys(i))
1505 endif
1506 if(this_dist <= this_maxdist) then
1507 ! Add this location to correct list
1508 cnum_close = cnum_close + 1
1509 cclose_ind(cnum_close) = i
1510 cdist(cnum_close) = this_dist
1511 endif
1512 end do
1513endif
1514
1515!--------------------------------------------------------------
1516
1517! Begin by figuring out which box the base_ob is in.
1518! Note that the boxes will not cover the entire sphere for sets of locations
1519! that cover less than a hemisphere. In that case you expect to get out of range
1520! box numbers if the location is outside the covered region. But you do have to
1521! handle the case of a location which is exactly on the top latitude boundary.
1522! Include it in the top box if it is in the next box but still within N*epsilon
1523! of the lower edge.
1524! FIXME: the longitude tests don't try to be smart about this, probably
1525! because of the complication of wrapping around the greenwich line.
1526! it just bins all values in box nlon+1 into box nlon. we could do
1527! the same thing here.
1528lon_box = get_lon_box(gc%gtt(bt), base_loc%lon)
1529lat_box = floor((base_loc%lat - gc%gtt(bt)%bot_lat) / gc%gtt(bt)%lat_width) + 1
1530! FIXME: cheaper to test now or later?
1531!if ((lat_box == nlat+1) .and. &
1532! (base_loc%lat <= gc%gtt(bt)%top_lat + EDGE_TOLERANCE)) then
1533!!DEBUG write(*,'(A,4(G25.16,1X))') 'add to top lat_box', base_loc%lat, gc%gtt(bt)%top_lat, &
1534!!DEBUG epsilon(0.0_r8), (base_loc%lat - gc%gtt(bt)%top_lat)/epsilon(0.0_r8)
1535! lat_box = nlat
1536!endif
1537! consistent with the current longitude code in get_lon_box():
1538if (lat_box == nlat+1) lat_box = nlat
1539
1540! If it is not in any box, then it is more than the maxdist away from everybody
1541if(lat_box > nlat .or. lat_box < 1 .or. lon_box < 0) return
1542
1543! Next, loop through to find each box that is close to this box
1544do j = 1, nlat
1545 n_lon = gc%gtt(bt)%lon_offset(lat_box, j)
1546 if(n_lon >= 0) then
1547 lon_offset: do i = -1 * n_lon, n_lon
1548 lon_ind = lon_box + i
1549 ! Search a box at this latitude j offset in longitude by i
1550 ! If domain is cyclic, need to wrap around
1551 if(gc%gtt(bt)%lon_cyclic) then
1552 if(lon_ind > nlon) lon_ind = lon_ind - nlon
1553 if(lon_ind < 1) lon_ind = lon_ind + nlon
1554 else
1555 ! Domain is not cyclic, don't search if outside of range
1556 if(lon_ind > nlon .or. lon_ind < 1) cycle lon_offset
1557 endif
1558 ! Box to search is lon_ind, j
1559 n_in_box = gc%gtt(bt)%count(lon_ind, j)
1560 st = gc%gtt(bt)%start(lon_ind, j)
1561 ! Loop to check how close all locs in the box are; add those that are close
1562 do k = 1, n_in_box
1563
1564 ! SHOULD ADD IN OPTIONAL ARGUMENT FOR DOING THIS!!!
1565 ! Could avoid adding any that have nums lower than base_ob???
1566 t_ind = gc%gtt(bt)%loc_box(st - 1 + k)
1567
1568 if(.not. present(dist)) then
1569 ! Dist isn't present; add this ob to list without computing distance
1570 num_close = num_close + 1
1571 close_ind(num_close) = t_ind
1572 else
1573 if(base_loc%which_vert == locs(t_ind)%which_vert) then
1574 ! Can compute total distance here if verts are the same
1575 this_dist = get_dist(base_loc, locs(t_ind), base_type, loc_qtys(t_ind))
1576 else
1577 ! Otherwise can just get horizontal distance
1578 this_dist = get_dist(base_loc, locs(t_ind), base_type, loc_qtys(t_ind), &
1579 no_vert = .true.)
1580 endif
1581
1582 ! If this locations distance is less than cutoff, add it to the list
1583 if(this_dist <= this_maxdist) then
1584 num_close = num_close + 1
1585 close_ind(num_close) = t_ind
1586 dist(num_close) = this_dist
1587 endif
1588 endif
1589 end do
1590 end do lon_offset
1591 endif
1592end do
1593
1594!------------------------ Verify by comparing to exhaustive search --------------
1595if(compare_to_correct) then
1596 ! Do comparisons against full search
1597 if((num_close /= cnum_close) .and. present(dist)) then
1598 write(msgstring, *) 'get_close (', num_close, ') should equal exhaustive search (', cnum_close, ')'
1599 call error_handler(e_err, 'get_close', msgstring, source, &
1600 text2='optional arg "dist" is present; we are computing exact distances', &
1601 text3='the exhaustive search should find an identical number of locations')
1602 else if (num_close < cnum_close) then
1603 write(msgstring, *) 'get_close (', num_close, ') should not be smaller than exhaustive search (', cnum_close, ')'
1604 call error_handler(e_err, 'get_close', msgstring, source, &
1605 text2='optional arg "dist" not present; we are returning a superset of close locations', &
1606 text3='the exhaustive search should find an equal or lesser number of locations')
1607 endif
1608endif
1609!--------------------End of verify by comparing to exhaustive search --------------
1610
1611end subroutine get_close

1615subroutine find_box_ranges(gtt, num, locs)
1616
1617! Finds boundaries for boxes in N/S direction. If data is localized in N/S
1618! tries to find boxes that only span the range of the data.
1619
1620type(get_close_type_by_type), intent(inout) :: gtt
1621integer, intent(in) :: num
1622type(location_type), intent(in) :: locs(num)
1623
1624real(r8) :: min_lat, max_lat, beg_box_lon, end_box_lon, first_loc_lon, last_loc_lon
1625real(r8) :: longitude_range, degrees, lon_dist
1626integer :: i, indx, gap_start, gap_end, gap_length
1627logical :: lon_box_full(360)
1628
1629! Initialize boxes used to see where locations are.
1630! Assume regional until we prove that we have to use the
1631! entire 360 in longitude.
1632lon_box_full = .false.
1633gtt%lon_cyclic = .false.
1634
1635! Figure out domain over which an additional locs MIGHT be close to one in this set
1636! If any points within maxdist of the poles, our boxes have to cover all 360 of
1637! longitude - no point in trying to restrict boxes to a region of the globe.
1638min_lat = minval(locs(:)%lat) - gtt%maxdist
1639max_lat = maxval(locs(:)%lat) + gtt%maxdist
1640if(min_lat <= -pi / 2.0_r8) then
1641 min_lat = -pi / 2.0_r8
1642 gtt%lon_cyclic = .true.
1643endif
1644if(max_lat >= pi / 2.0_r8) then
1645 max_lat = pi / 2.0_r8
1646 gtt%lon_cyclic = .true.
1647endif
1648
1649! Put this into storage for this get_close_type
1650gtt%bot_lat = min_lat
1651gtt%top_lat = max_lat
1652gtt%lat_width = (max_lat - min_lat) / nlat
1653! don't have to do all this work if we already know it has to be cyclic
1654if (.not. gtt%lon_cyclic) then
1655
1656 ! Finding the longitude range is tricky because of cyclic nature
1657 ! Want to find minimum range spanned by locs even if they wrap-around Greenwich
1658 ! Would like to do this without sorting if possible at low-cost
1659 ! First, partition into 360 1-degree boxes and find the biggest gap
1660 do i = 1, num
1661 degrees = locs(i)%lon * 180.0_r8 / pi
1662 ! If the value of the longitude is very close to an integer number of degrees
1663 ! a roundoff can occur that leads to an assignment in the wrong box. We avoid this
1664 ! by first testing to see if this is possible and then setting both boxes to full.
1665 ! If this is not the case, then we fill the box the location is in.
1666 if (abs(degrees - nint(degrees)) < 0.00001_r8) then
1667 indx = nint(degrees)
1668 if(indx < 1) indx = 360
1669 lon_box_full(indx) = .true.
1670
1671 indx = nint(degrees) + 1
1672 if(indx > 360) indx = 1
1673 lon_box_full(indx) = .true.
1674 else
1675 indx = floor(degrees) + 1
1676 lon_box_full(indx) = .true.
1677 endif
1678 end do
1679
1680 ! Find the longest sequence of empty boxes
1681 call find_longest_gap(lon_box_full, 360, gap_start, gap_end, gap_length)
1682
1683 if (gap_length > 0) then
1684
1685 ! There is a gap; figure out locs that are closest to ends of non-gap
1686 beg_box_lon = (gap_end / 180.0_r8) * pi
1687 end_box_lon = ((gap_start -1) / 180.0_r8) * pi
1688 first_loc_lon = find_closest_to_start(beg_box_lon, num, locs)
1689 last_loc_lon = find_closest_to_end(end_box_lon, num, locs)
1690 ! Determine the final longitude range
1691 longitude_range = last_loc_lon - first_loc_lon
1692 if(longitude_range <= 0.0_r8) longitude_range = longitude_range + 2.0_r8 * pi
1693
1694 ! Add on the extra distance needed for the boxes
1695 ! To avoid any hard thinking about wraparound with sub-domain boxes
1696 ! Must span less than 180 degrees to get smaller boxes
1697 ! If addition of halos for close location fills more than half of space
1698 ! things go 0 to 2PI
1699
1700 ! other places we are computing in radians. here we are computing in
1701 ! lat/lon, and you can't just add maxdist to the edges - that doesn't
1702 ! take into account the great-circle distance. the separation in longitude
1703 ! varies with latitude. compute the delta longitude based on the most
1704 ! poleward latitude and add that onto the edges of both boxes. that
1705 ! overestimates for points closer to the equator, but that's better
1706 ! than underestimating and excluding points that are within maxdist.
1707 lon_dist = find_del_lon(minval(locs(:)%lat), maxval(locs(:)%lat), gtt%maxdist)
1708
1709 if(longitude_range + 2.0_r8 * lon_dist > pi) then
1710 gtt%lon_cyclic = .true.
1711 else
1712 first_loc_lon = first_loc_lon - lon_dist
1713 if(first_loc_lon < 0.0_r8 ) first_loc_lon = first_loc_lon + 2.0_r8 * pi
1714 last_loc_lon = last_loc_lon + lon_dist
1715 if(last_loc_lon > 2.0_r8 * pi) last_loc_lon = last_loc_lon - 2.0_r8 * pi
1716 gtt%lon_cyclic = .false.
1717 endif
1718 else
1719 ! No gap was found: all 360 boxes had an location in them
1720 gtt%lon_cyclic = .true.
1721 endif
1722endif
1723
1724if (gtt%lon_cyclic) then
1725 first_loc_lon = 0.0_r8
1726 last_loc_lon = 2.0_r8 * pi
1727endif
1728
1729! Put in storage for structure
1730gtt%bot_lon = first_loc_lon
1731gtt%top_lon = last_loc_lon
1732longitude_range = last_loc_lon - first_loc_lon
1733if(longitude_range <= 0.0_r8) longitude_range = longitude_range + 2.0_r8 * pi
1734gtt%lon_width = longitude_range / nlon
1735
1736end subroutine find_box_ranges

1740subroutine find_longest_gap(lon_box_full, num_boxes, gap_start, gap_end, gap_length)
1741
1742! Find the longest gap in the boxes (take the first one if there's a tie)
1743
1744integer, intent(in) :: num_boxes
1745integer, intent(out) :: gap_start, gap_end, gap_length
1746logical, intent(in) :: lon_box_full(num_boxes)
1747
1748integer :: g_start, g_end, g_length, next_box, i, full_count
1749logical :: all_done
1750
1751! Initialize these to known values.
1752gap_start = -1
1753gap_end = -1
1754gap_length = -1
1755
1756! If more than half of the boxes are full, then assume that there is no
1757! meaningful gap and just use the whole domain for get_close
1758full_count = 0
1759do i = 1, num_boxes
1760 if(lon_box_full(i)) full_count = full_count + 1
1761end do
1762if(full_count >= num_boxes / 2) return
1763
1764! More than half of the boxes were empty, try to hone in on potentially
1765! local locations
1766next_box = 1
1767all_done = .false.
1768! Loop long enough to be sure we go around
1769do i = 1, num_boxes
1770 call find_next_gap(next_box, lon_box_full, num_boxes, g_start, g_end, g_length)
1771 next_box = g_end
1772 ! Easy way to terminate at cost of some small additional computation
1773 if(g_start == gap_start .and. g_end == gap_end) all_done = .true.
1774 if(g_length > gap_length) then
1775 gap_start = g_start
1776 gap_end = g_end
1777 gap_length = g_length
1778 endif
1779 ! Exit if longest has been found twice
1780 if(all_done) return
1781end do
1782
1783end subroutine find_longest_gap

1787subroutine find_next_gap(start_box, lon_box_full, num_boxes, gap_start, gap_end, gap_length)

1790logical, intent(in) :: lon_box_full(num_boxes)
1791
1792integer :: next_full
1793
1794! This never gets called unless at least half of the boxes are empty
1795! No need to error check for all boxes empty (means num is 0) or for
1796! all boxes full.
1797
1798! Finds the next gap of empty boxes in the cyclic set
1799! First, find the next full box from the start
1800next_full = next_full_box(start_box, num_boxes, lon_box_full)
1801! Find the next empty after that, make it the start of the gap
1802gap_start = next_empty_box(next_full, num_boxes, lon_box_full)
1803! Find the next full, box before that is the end of the gap
1804gap_end = next_full_box(gap_start, num_boxes, lon_box_full) - 1
1805if(gap_end < 1) gap_end = gap_end + num_boxes
1806! Carefully compute gap length
1807if(gap_end >= gap_start) then
1808 gap_length = gap_end - gap_start + 1
1809else
1810 gap_length = gap_end - gap_start + 360 + 1
1811endif

1960function find_del_lon(minlat, maxlat, maxdist)
1961
1962! for the given latitudes, find the furthest longitude that is still
1963! within maxdist away. this will be at a different latitude at any
1964! location other than the equator. all values specified in radians.
1965! distance returned in radians. if either lat is closer to the pole
1966! than maxdist, it returns 2*PI.
1967
1968real(r8), intent(in) :: minlat, maxlat, maxdist
1969real(r8) :: find_del_lon
1970
1971real(r8) :: a, b, c
1972real(r8) :: latval, poleward_lat
1973
1974! find the most poleward of the two latitudes
1975poleward_lat = max(abs(minlat), abs(maxlat))
1976
1977! if either latitude is within maxdist of either pole, return 2 PI
1978! because you are now covering all possible longitudes.
1979if (poleward_lat + maxdist > (pi / 2.0_r8)) then
1980 find_del_lon = 2.0_r8 * pi
1981 return
1982endif
1983
1984! compute some values we will reuse a couple times
1985a = cos(maxdist)
1986b = sin(poleward_lat)
1987c = cos(poleward_lat)
1988
1989! lat at which max offset is found
1990latval = asin(b/a)
1991
1992! distance to furthest lon, at latval
1993find_del_lon = acos((a - (b*sin(latval))) / (c*cos(latval)))
1994
1995end function find_del_lon
