248 subroutine get_expected_iasi_co(state_handle, ens_size, location, key, val, istatus)
249!---------------------------------------------------------------------
250type(ensemble_type), intent(in) :: state_handle
251integer, intent(in) :: ens_size
252type(location_type), intent(in) :: location
253integer, intent(in) :: key
254real(r8), intent(out) :: val(ens_size)
255integer, intent(out) :: istatus(ens_size)
256integer :: i,j
257integer :: num_levs, lev
258type(location_type) :: loc1,loc2,loc3,loc3p,loc3m,locs
259real(r8) :: mloc(3), mloc1(3), mloc2(3)
260real(r8) :: obs_val(ens_size), obs_val_int(ens_size)
261real(r8) :: top_pres(ens_size), bot_pres(ens_size), coef(ens_size), mop_layer_wght(ens_size)
262real(r8) :: i_top_pres(ens_size), i_bot_pres(ens_size), i_pres
263real(r8) :: p_col(ens_size, max_model_levs)
264real(r8) :: mopitt_pres_local(ens_size, mopitt_dim)
265integer :: nlevels, start_i, end_i
266integer, allocatable :: dim_sizes(:)
267integer :: p_col_istatus(ens_size), obs_val_int_istatus(ens_size)
268logical :: return_now
269integer :: imem
270
271if ( .not. module_initialized ) call initialize_module
272val = 0.0_r8
273
274mloc = get_location(location)
275if (mloc(2)>90.0_r8) then
276 mloc(2)=90.0_r8
277elseif (mloc(2)<-90.0_r8) then
278 mloc(2)=-90.0_r8
279endif
280
281do imem = 1, ens_size
282 mopitt_pres_local(imem, :) = mopitt_pressure
283enddo
284
285nlevels = mopitt_nlevels(key)
286! Modify AFAJ (072513)
287start_i = mopitt_dim-nlevels+1
288!mopitt_pres_local(start_i)=mopitt_psurf(key)
289! KDR Redefining one element of a global array probably wasn't a good idea.
290end_i = mopitt_dim
291
292! Find the number of model levels and the pressures on them.
293istatus = 0
294p_col = missing_r8
295lev = 1
296model_levels: do
297 locs = set_location(mloc(1),mloc(2),real(lev,r8),vertislevel)
298 call interpolate(state_handle, ens_size, locs, qty_pressure, p_col(:, lev), p_col_istatus)
299 if (any(p_col_istatus /= 0)) then
300 p_col(:, lev) = missing_r8
301 num_levs = lev - 1
302 exit model_levels
303 endif
304 lev = lev + 1
305enddo model_levels
306
307! KDR: see comments in the same place in get_expected_mopitt_co.
308
309! Reset istatus to zero, so that it can be used properly by track_status
310istatus = 0
311
312!Barre'; here p_col is the pressure levels at the grid points, we will need to have in
313!the future the pressure values at the mid-points (need to create a new function
314!plevs_cam in model_mod using the the mid-levels hybrid coefs)
315locs = set_location(mloc(1),mloc(2),0.0_r8,vertissurface)
316call interpolate(state_handle, ens_size, locs, qty_surface_pressure, p_col(:,num_levs), p_col_istatus)
317call track_status(ens_size, p_col_istatus, p_col(:,num_levs), istatus, return_now)
318if (return_now) return
319
320mopitt_pres_local(:, 1) = p_col(:, num_levs)
321! KDR Should we really be redefining this global array element?
322!> @todo this is not a global array (anymore?) so it's ok
323
324do i=start_i, end_i
325 obs_val=0.0_r8
326
327 bot_pres(:)=mopitt_pres_local(:, i)
328 if (i == mopitt_dim) then
329 top_pres(:)=mopitt_pres_local(:, i)/2.0_r8
330 else
331 top_pres(:)=mopitt_pres_local(:, i+1)
332 endif
333! This is used in all coefs, below.
334 mop_layer_wght = 1.0_r8/abs(bot_pres-top_pres)
335
336 do j=1,num_levs
337 i_bot_pres(:)=p_col(:, j)
338 if (j == 1) then
339 i_top_pres(:)=0.0_r8
340 else
341 i_top_pres(:)=p_col(:, j-1)
342 endif
343
344 coef=0.0_r8
345 obs_val_int=0.0_r8
346
347!> @todo can this happen? and if so, is it ok to call
348!> interpolate anyway for all ensemble members?
349
350 if ( any(bot_pres > top_pres) ) then
351 loc3 = set_location(mloc(1),mloc(2),real(j,r8), vertislevel)
352 call interpolate(state_handle, ens_size, loc3, qty_co, obs_val_int, obs_val_int_istatus)
353 call track_status(ens_size, obs_val_int_istatus, obs_val_int, istatus, return_now)
354 if (return_now) return
355
356 where (i_bot_pres <= bot_pres .and. i_bot_pres > top_pres)
357 where (i_top_pres <= top_pres)
358 coef=abs(i_bot_pres-top_pres) * mop_layer_wght
359 else where
360 coef=abs(i_bot_pres-i_top_pres) * mop_layer_wght
361 end where
362 else where
363 where (i_top_pres <= top_pres)
364 coef=abs(bot_pres-top_pres) * mop_layer_wght
365 else where
366 coef=abs(bot_pres-i_top_pres) * mop_layer_wght
367 end where
368 end where
369 endif
370
371 where (istatus == 0) obs_val = obs_val + obs_val_int * coef
372 !>@todo FIXME JH need to figure out how to enclose write statements
373 ! where (coef<0 .or. coef>1) write(*,*) 'coef', coef, i, j, mloc(1),mloc(2)
374 ! where (coef<0 .or. coef>1) write(*,*) 'pres', i_bot_pres,i_top_pres,bot_pres,top_pres
375 !if (obs_val_int<0) write(*,*) 'inerp neg', obs_val_int, i, j,mloc(1),mloc(2)
376
377 enddo
378 where (obs_val > 0.0_r8) val = val + avg_kernel(key,i) * (obs_val)
379enddo
380
381where (istatus ==0) val = val + mopitt_prior(key)
382! KDR Should this now be MISSING_R8?
383! if (val < 0.0_r8) val=-777.0_r8
384!> @todo check with the cam/chem guys - do they absolutely
385!> have to have -777 as some magic value for this failed case?
386where (val < 0.0_r8)
387 val=missing_r8
388 istatus = 7
389end where
390
391 end subroutine get_expected_iasi_co

393 subroutine get_expected_mopitt_co(state_handle, ens_size, location, key, val, istatus)
394!----------------------------------------------------------------------
395type(ensemble_type), intent(in) :: state_handle
396integer, intent(in) :: ens_size
397type(location_type), intent(in) :: location
398integer, intent(in) :: key
399real(r8), intent(out) :: val(ens_size)
400integer, intent(out) :: istatus(ens_size)
401
402integer :: i,j
403type(location_type) :: loc1,loc2,loc3,loc3p,loc3m,locs
404real(r8) :: mloc(3), mloc1(3), mloc2(3)
405real(r8) :: obs_val(ens_size), obs_val_int(ens_size)
406
407integer :: nlevels, start_i, end_i
408
409real(r8) :: top_pres(ens_size), bot_pres(ens_size), coef(ens_size), mop_layer_wght(ens_size)
410real(r8) :: i_top_pres(ens_size), i_bot_pres(ens_size), i_pres
411integer :: num_levs, lev
412real(r8) :: p_col(ens_size, max_model_levs)
413real(r8) :: mopitt_pres_local(ens_size, mopitt_dim)
414integer, allocatable :: dim_sizes(:)
415integer :: p_col_istatus(ens_size), obs_val_int_istatus(ens_size)
416logical :: return_now
417integer :: imem
418
419if ( .not. module_initialized ) call initialize_module
420mloc = get_location(location)
421
422! Apply MOPITT Column Averaging kernel vector a_t and MOPITT Prior xa- a_t xa
423! x = a_t xm + xa- a_t xa , where x is a 10 element vector
424! xm is the 10-element partial columnis at MOPITT grid
425! also here, the averaging kernel is operated in log VMR.
426! i.e. the state vector CO is in log VMR
427! edit dart_to_cam and cam_to_dart for the transformation
428! also, it turns our xa - a_txa is a scalar quantity
429! KDR Why not initialize to mopitt_prior(key), instead of adding it after
430! the end of the i-loop?
431val = 0.0_r8
432
433if (mloc(2)>90.0_r8) then
434 mloc(2)=90.0_r8
435elseif (mloc(2)<-90.0_r8) then
436 mloc(2)=-90.0_r8
437endif
438
439do imem = 1, ens_size
440 mopitt_pres_local(imem, :) = mopitt_pressure
441enddo
442
443nlevels = mopitt_nlevels(key)
444! Modify AFAJ (072513)
445! KDR start_i >= 1
446start_i = mopitt_dim-nlevels+1
447! KDR Ack! redefining one element of a global array.
448! Various elements will be redefined during various calls to this subroutine,
449! because the array is initialized in the specification statement.
450mopitt_pres_local(:, start_i)=mopitt_psurf(key)
451end_i = mopitt_dim
452
453!JB: do partial columns aproximations
454! Find the number of model levels and the pressures on them.
455istatus = 0
456p_col = missing_r8
457lev = 1
458model_levels: do
459 locs = set_location(mloc(1),mloc(2),real(lev,r8),vertislevel)
460 call interpolate(state_handle, ens_size, locs, qty_pressure, p_col(:, lev), p_col_istatus)
461 if (any(p_col_istatus /= 0)) then
462 p_col(:, lev) = missing_r8
463 num_levs = lev - 1
464 exit model_levels
465 endif
466 lev = lev + 1
467enddo model_levels
468
469!Barre: here p_col is the pressure levels at the grid points, we will need to have in
470!the future the pressure values at the mid-points (need to create a new function
471!plevs_cam in model_mod using the the mid-levels hybrid coefs)
472! KDR This comment looks confused. plevs_cam returns pressures on what CESM refers to
473! as layer midpoints. Maybe MOPITT needs pressures on the model interfaces,
474! in which case a new subroutine using hyai and hybi would replace hyam and hybm.
475! Anyway, p_col(30) is being redefined as the CAM surface pressure,
476! but the rest of the p_col is not redefined.
477! This all may be fine, since val is an accumulation of contributions
478! from pressure sub-layers.
479locs = set_location(mloc(1),mloc(2),0.0_r8,vertissurface)
480call interpolate(state_handle, ens_size, locs, qty_surface_pressure, p_col(:, num_levs), p_col_istatus)
481call track_status(ens_size, p_col_istatus, p_col(:,num_levs), istatus, return_now)
482if (return_now) return
483
484! KDR Ack! redefining one element of a global array.
485!> @todo not global array
486! This may be the 2nd redefinition of (1), or the first, if start_i /= 1.
487mopitt_pres_local(:,1) = p_col(:,num_levs)
488
489! KDR; Algorithm;
490! Work through the MOPITT pressure level layers. (i loop)
491! Find CAM levels that are in each layer. (j loop)
492! Calculate CO from CAM state using level as the vertical location.
493! Accumulate contributions of CO from the CAM layers,
494! weighted by the the thickness of the CAM layer that lies within the MOPITT layer.
495! Work through the MOPITT pressure level layers. (i loop)
496! Loop from some number to 10; the mopitt pressure layers above the ground at this location.
497do i=start_i, end_i
498 obs_val=0.0_r8
499
500 bot_pres(:)=mopitt_pres_local(:, i)
501 if (i == mopitt_dim) then
502 top_pres(:)=mopitt_pres_local(:, i)/2.0_r8
503 else
504 top_pres(:)=mopitt_pres_local(:, i+1)
505 endif
506! This is used in all 'coef's, below.
507 mop_layer_wght = 1.0_r8/abs(bot_pres-top_pres)
508
509! KDR Search through CAM layers for the ones which overlap the current mopitt layer.
510 do j=1,num_levs
511 i_bot_pres(:)=p_col(:,j)
512 if (j == 1) then
513 i_top_pres(:)=0.0_r8
514 else
515 i_top_pres(:)=p_col(:,j-1)
516 endif
517
518 coef=0.0_r8
519 obs_val_int=0.0_r8
520!> @todo same comment as previous subroutine
521 if ( any(bot_pres > top_pres) ) then
522 loc3 = set_location(mloc(1),mloc(2),real(j,r8), vertislevel)
523 call interpolate(state_handle, ens_size, loc3, qty_co, obs_val_int, obs_val_int_istatus)

556 !if (obs_val <= 0.0) then
557 !write(*,*) 'neg!', obs_val, bot_pres, i_bot_pres, i_top_pres
558 !endif
559 ! KDR different from get_expected_iasi_co line
560 where (obs_val > 0.0_r8 .and. avg_kernel(key,i) > -700) val = val + avg_kernel(key,i) * log10(obs_val)
561enddo
562!val = val + 10.0_r8**mopitt_prior(key)

568!write(*,*) 'after 10^', val
