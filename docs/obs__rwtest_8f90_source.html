190subroutine print_obs_seq(seq_in, filename)
191
192! you can get more info by running the obs_diag program, but this
193! prints out a quick table of obs types and counts, overall start and
194! stop times, and metadata strings and counts.
195
196type(obs_sequence_type), intent(in) :: seq_in
197character(len=*), intent(in) :: filename
198
199type(obs_type) :: obs, next_obs
200type(obs_def_type) :: this_obs_def
201logical :: is_there_one, is_this_last
202integer :: size_seq_in
203integer :: i
204integer :: this_obs_kind
205! max_defined_types_of_obs is a public from obs_kind_mod.f90
206integer :: type_count(max_defined_types_of_obs), identity_count
207
208
209! Initialize input obs_types
210do i = 1, max_defined_types_of_obs
211 type_count(i) = 0
212enddo
213identity_count = 0
214
215! make sure there are obs left to process before going on.
216! num_obs should be ok since we just constructed this seq so it should
217! have no unlinked obs. if it might for some reason, use this instead:
218! size_seq_in = get_num_key_range(seq_in) !current size of seq_in
219
220size_seq_in = get_num_obs(seq_in)
221if (size_seq_in == 0) then
222 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
223 call error_handler(e_msg,'obs_rwtest',msgstring)
224 return
225endif
226
227! Initialize individual observation variables
228call init_obs( obs, get_num_copies(seq_in), get_num_qc(seq_in))
229call init_obs(next_obs, get_num_copies(seq_in), get_num_qc(seq_in))
230
231! blank line
232call error_handler(e_msg,'',' ')
233
234write(msgstring,*) 'Processing sequence file ', trim(filename)
235call error_handler(e_msg,'',msgstring)
236
237call print_metadata(seq_in, filename)
238
239!-------------------------------------------------------------
240! Start to process obs from seq_in
241!--------------------------------------------------------------
242is_there_one = get_first_obs(seq_in, obs)
243
244if ( .not. is_there_one ) then
245 write(msgstring,*)'no first observation in ',trim(filename)
246 call error_handler(e_msg,'obs_rwtest', msgstring)
247endif
248
249! process it here
250is_this_last = .false.
251
252call get_obs_def(obs, this_obs_def)
253call print_time(get_obs_def_time(this_obs_def), ' First timestamp: ')
254! does not work with NO_CALENDAR
255if (cal) call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
256
257obsloop : do while ( .not. is_this_last)
258
259 call get_obs_def(obs, this_obs_def)
260 this_obs_kind = get_obs_def_type_of_obs(this_obs_def)
261 if (this_obs_kind < 0) then
262 identity_count = identity_count + 1
263 else
264 type_count(this_obs_kind) = type_count(this_obs_kind) + 1
265 endif
266! print *, 'obs kind index = ', this_obs_kind
267! if(this_obs_kind > 0)print *, 'obs name = ', get_name_for_type_of_obs(this_obs_kind)
268
269 call get_next_obs(seq_in, obs, next_obs, is_this_last)
270 if (.not. is_this_last) then
271 obs = next_obs
272 else
273 call print_time(get_obs_def_time(this_obs_def), ' Last timestamp: ')
274 if (cal) call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
275 endif
276
277enddo obsloop
278
279
280write(msgstring, *) 'Number of obs processed : ', size_seq_in
281call error_handler(e_msg, '', msgstring)
282write(msgstring, *) '---------------------------------------------------------'
283call error_handler(e_msg, '', msgstring)
284do i = 1, max_defined_types_of_obs
285 if (type_count(i) > 0) then
286 write(msgstring, '(a32,i8,a)') trim(get_name_for_type_of_obs(i)), &
287 type_count(i), ' obs'
288 call error_handler(e_msg, '', msgstring)
289 endif
290enddo
291if (identity_count > 0) then
292 write(msgstring, '(a32,i8,a)') 'Identity observations', &
293 identity_count, ' obs'
294 call error_handler(e_msg, '', msgstring)
295endif
296
297! another blank line
298call error_handler(e_msg, '', ' ')
299
300! Time to clean up
301
302call destroy_obs( obs)
303call destroy_obs(next_obs)
304
305end subroutine print_obs_seq

309subroutine validate_obs_seq_time(seq, filename)
310
311! this eventually belongs in the obs_seq_mod code, but for now
312! try it out here. we just fixed a hole in the interactive create
313! routine which would silently let you create out-of-time-order
314! linked lists, which gave no errors but didn't assimilate the
315! right obs at the right time when running filter. this runs
316! through the times in the entire sequence, ensuring they are
317! monotonically increasing in time. this should help catch any
318! bad files which were created with older versions of code.
319
320type(obs_sequence_type), intent(in) :: seq
321character(len=*), intent(in) :: filename
322
323type(obs_type) :: obs, next_obs
324type(obs_def_type) :: this_obs_def
325logical :: is_there_one, is_this_last
326integer :: size_seq, obs_count
327integer :: key
328type(time_type) :: last_time, this_time
329
330
331! make sure there are obs left to process before going on.
332size_seq = get_num_obs(seq)
333if (size_seq == 0) then
334 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
335 call error_handler(e_msg,'obs_rwtest:validate',msgstring)
336 return
337endif
338
339! Initialize individual observation variables
340call init_obs( obs, get_num_copies(seq), get_num_qc(seq))
341call init_obs(next_obs, get_num_copies(seq), get_num_qc(seq))
342
343obs_count = 0
344
345!-------------------------------------------------------------
346! Start to process obs from seq
347!--------------------------------------------------------------
348is_there_one = get_first_obs(seq, obs)
349
350! we already tested for 0 obs above, so there should be a first obs here.
351if ( .not. is_there_one ) then
352 write(msgstring,*)'no first obs in sequence ' // trim(filename)
353 call error_handler(e_err,'obs_rwtest:validate', &
354 msgstring, source, revision, revdate)
355 return
356endif
357
358is_this_last = .false.
359last_time = set_time(0, 0)
360obsloop : do while ( .not. is_this_last)
361
362 call get_obs_def(obs, this_obs_def)
363 this_time = get_obs_def_time(this_obs_def)
364
365 if (last_time > this_time) then
366 ! bad time order of observations in linked list
367 call print_time(last_time, ' previous timestamp: ')
368 if (cal) call print_date(last_time, ' calendar date: ')
369 call print_time(this_time, ' next timestamp: ')
370 if (cal) call print_date(this_time, ' calendar date: ')
371
372 key = get_obs_key(obs)
373 write(msgstring1,*)'obs number ', key, ' has earlier time than previous obs'
374 write(msgstring2,*)'observations must be in increasing time order, file ' // trim(filename)
375 call error_handler(e_err,'obs_rwtest:validate', msgstring2, &
376 source, revision, revdate, &
377 text2=msgstring1)
378 endif
379
380 last_time = this_time
381 obs_count = obs_count + 1
382
383 call get_next_obs(seq, obs, next_obs, is_this_last)
384 if (.not. is_this_last) obs = next_obs
385
386enddo obsloop
387
388! clean up
389call destroy_obs( obs)
390call destroy_obs(next_obs)
391
392! technically not a time validation, but easy to check. obs_count should never
393! be larger than size_seq - that's a fatal error. obs_count < size_seq would
394! suggest there are obs in the file that aren't part of the linked list.
395! this does not necessarily indicate a fatal error but it's not a common
396! situation and might indicate someone should check on the file.
397if (obs_count /= size_seq) then
398 write(msgstring,*) 'input sequence ', trim(filename)
399 call error_handler(e_msg,'obs_rwtest:validate', msgstring)
400
401 write(msgstring,*) 'total obs in file: ', size_seq, ' obs in linked list: ', obs_count
402 if (obs_count > size_seq) then
403 ! this is a fatal error
404 write(msgstring1,*) 'linked list obs_count > total size_seq, should not happen'
405 call error_handler(e_err,'obs_rwtest:validate', msgstring, &
406 source, revision, revdate, &
407 text2=msgstring1)
408 else
409 ! just warning msg
410 write(msgstring1,*) 'only observations in linked list will be processed'
411 call error_handler(e_msg,'obs_rwtest:validate', msgstring, &
412 source, revision, revdate, text2=msgstring1)
413 endif
414endif
415
416end subroutine validate_obs_seq_time

461! a fortran read seems to stop at a slash if it's a free format (*) read.
462! unfortunate choice for reading filenames with directories. but i'd like
463! to support running the program like this: echo a b | ./obs_rwtest
464! as well as just ./obs_rwtest and having it prompt and possibly getting
465! one filename per input line. the latter isn't hard, but for the first
466! i need to parse 2 strings separated by spaces. but (A) format will
467! read the entire string into a single variable. bof.
468
469subroutine get_filenames(filename_in, filename_out)
470
471character(len=*), intent(out) :: filename_in
472character(len=*), intent(out) :: filename_out
473
474character(512) :: oneline
475integer :: part2, lastchar
476
477! get the entire line
478read(*, '(A)') oneline
479
480! find index of last real character
481lastchar = len_trim(oneline)
482
483! find the separating space, if there is one, starting from the back
484part2 = index(oneline(1:lastchar), ' ', .true.)
485if (part2 == 0) then
486 ! no spaces
487 filename_in = oneline
488
489 ! read another line to get the output filename
490 read(*, '(A)') oneline
491 filename_out = oneline
492
493else if (part2 == 1) then
494 ! no text?
495 print *, 'part2 is 1'
496else if (part2 == lastchar) then
497 ! no text?
498 print *, 'part2 is lastchar'
499else
500 ! what we expect if both filenames are on one line
501 filename_in = oneline(1:part2-1)
502 filename_out = oneline(part2:lastchar)
503endif
504
505end subroutine get_filenames
