70subroutine bnrh_cdf(x, ens_size, bounded_below, bounded_above, lower_bound, upper_bound, &
71 sort_x, quantiles, &
72 tail_amp_left, tail_mean_left, tail_sd_left, do_uniform_tail_left, &
73 tail_amp_right, tail_mean_right, tail_sd_right, do_uniform_tail_right)
74
75integer, intent(in) :: ens_size
76real(r8), intent(in) :: x(ens_size)
77logical, intent(in) :: bounded_below, bounded_above
78real(r8), intent(in) :: lower_bound, upper_bound
79real(r8), intent(out) :: sort_x(ens_size)
80real(r8), intent(out) :: quantiles(ens_size)
81real(r8), intent(out) :: tail_amp_left, tail_mean_left, tail_sd_left
82real(r8), intent(out) :: tail_amp_right, tail_mean_right, tail_sd_right
83logical, intent(out) :: do_uniform_tail_left, do_uniform_tail_right
84
85real(r8) :: q(ens_size)
86real(r8) :: del_q, mean, bound_quantile
87integer :: sort_index(ens_size), indx, i
88
89! Computes all information about a rank histogram cdf given the ensemble and bounds
90
91! Get ensemble mean and sd
92call normal_mean_sd(x, ens_size, mean, tail_sd_left)
93tail_sd_right = tail_sd_left
94
95! Don't know what to do if sd is 0; tail_sds returned are illegal value to indicate this
96if(tail_sd_left <= 0.0_r8) then
97 tail_sd_left = -99_r8
98 tail_sd_right = -99_r8
99 return
100endif
101
102! Sort. For now, don't worry about efficiency, but may need to somehow pass previous
103! sorting indexes and use a sort that is faster for nearly sorted data. Profiling can guide the need
104call index_sort(x, sort_index, ens_size)
105sort_x = x(sort_index)
106
107! Fail if lower bound is larger than smallest ensemble member
108if(bounded_below) then
109 ! Do in two ifs in case the bound is not defined
110 if(sort_x(1) < lower_bound) then
111 write(errstring, *) 'Smallest ensemble member less than lower bound', &
112 sort_x(1), lower_bound
113 call error_handler(e_err, 'bnrh_cdf', errstring, source)
114 endif
115endif
116
117! Fail if upper bound is smaller than the largest ensemble member
118if(bounded_above) then
119 if(sort_x(ens_size) > upper_bound) then
120 write(errstring, *) 'Largest ensemble member greater than upper bound', &
121 sort_x(ens_size), upper_bound
122 call error_handler(e_err, 'bnrh_cdf', errstring, source)
123 endif
124endif
125
126! The ensemble size array q contains the sorted quantiles corresponding to the sorted ensemble sort_x
127call ens_quantiles(sort_x, ens_size, &
128 bounded_below, bounded_above, lower_bound, upper_bound, q)
129! The quantiles array has the unsorted quantiles corresponding to the unsorted input ensemble, x
130do i = 1, ens_size
131 indx = sort_index(i)
132 quantiles(indx) = q(i)
133end do
134
135! Compute the characteristics of tails
136
137! For unit normal, find distance from mean to where cdf is 1/(ens_size+1) (del_q_.
138! Saved to avoid redundant computation for repeated calls with same ensemble size
139del_q = 1.0_r8 / (ens_size + 1.0_r8)
140
141if(saved_ens_size /= ens_size) then
142 dist_for_unit_sd = inv_std_normal_cdf(del_q)
143 ! This will be negative, want it to be a distance so make it positive
144 dist_for_unit_sd = -1.0_r8 * dist_for_unit_sd
145 ! Keep a record of the ensemble size used to compute dist_for_unit_sd
146 saved_ens_size = ens_size
147endif
148
149! Find a mean so that 1 / (ens_size + 1) probability is in outer regions
150tail_mean_left = sort_x(1) + dist_for_unit_sd * tail_sd_left
151tail_mean_right = sort_x(ens_size) - dist_for_unit_sd * tail_sd_right
152
153! If the distribution is bounded, still want 1 / (ens_size + 1) (del_q) in outer regions
154! Put an amplitude term (greater than 1) in front of the tail normals
155! Amplitude is 1 if there are no bounds, so start with that
156tail_amp_left = 1.0_r8
157tail_amp_right = 1.0_r8
158
159! Switch to uniform for cases where bound and outermost ensemble have close quantiles
160! Default: not close
161do_uniform_tail_left = .false.
162if(bounded_below) then
163 ! Compute the CDF at the bounds
164 bound_quantile = normal_cdf(lower_bound, tail_mean_left, tail_sd_left)
165 ! Note that due to roundoff it is possible for del_q - quantile to be slightly negative
166 if((del_q - bound_quantile) / del_q < uniform_threshold) then
167 ! If bound and ensemble member are too close, do uniform approximation
168 do_uniform_tail_left = .true.
169 else
170 ! Compute the left tail amplitude
171 tail_amp_left = del_q / (del_q - bound_quantile);
172 endif
173endif
174
175! Default: not close
176do_uniform_tail_right = .false.
177if(bounded_above) then
178 ! Compute the CDF at the bounds
179 bound_quantile = normal_cdf(upper_bound, tail_mean_right, tail_sd_right)
180 ! Note that due to roundoff it is possible for the numerator to be slightly negative
181 if((bound_quantile - (1.0_r8 - del_q)) / del_q < uniform_threshold) then
182 ! If bound and ensemble member are too close, do uniform approximation
183 do_uniform_tail_right = .true.
184 else
185 ! Compute the right tail amplitude
186 tail_amp_right = del_q / (del_q - (1.0_r8 - bound_quantile))
187 endif
188endif
189
190end subroutine bnrh_cdf

194subroutine bnrh_cdf_initialized_vector(x, num, p, quantiles)
195
196integer, intent(in) :: num
197real(r8), intent(in) :: x(num)
198type(distribution_params_type), intent(in) :: p
199real(r8), intent(out) :: quantiles(num)
200
201real(r8) :: tail_amp_left, tail_mean_left, tail_sd_left
202real(r8) :: tail_amp_right, tail_mean_right, tail_sd_right
203logical :: do_uniform_tail_left, do_uniform_tail_right
204
205! Given the sorted ensemble (sort_ens) that defines a bnrh CDF and all the corresponding
206! information about that distribution, computes the value of the CDF for a vector of num
207! elsements (x) and returns those quantiles.
208
209! In the default filter usage, this is only used for doing the probit transform for the
210! posterior observation ensemble. In this case, the size of vector x is the same as the
211! ensemble size. For hybrid filter applications, the ensemble size defining the BNRH CDF
212! might be different from the ensemble that needs updating, so x could have a different size.
213
214real(r8) :: q(p%ens_size)
215integer :: i
216
217! Extract the required information from the distribution_params_type
218call unpack_bnrh_params(p, do_uniform_tail_left, do_uniform_tail_right, &
219 tail_amp_left, tail_amp_right, tail_mean_left, tail_mean_right, tail_sd_left, tail_sd_right)
220
221! Compute the quantiles of each of the sorted ensemble members that define the BNRH distribution.
222! This was all computed when the distribution was originally set up, could choose to cache that
223! in the params structure for efficiency. This is only used for the single observation posterior
224! so one could only save in that case removing any storage concerns.
225call ens_quantiles(p%ens, p%ens_size, p%bounded_below, p%bounded_above, p%lower_bound, p%upper_bound, q)
226
227! Loop through the values in the x vector to compute the CDF at each one.
228! This can be done vastly more efficiently with either binary searches or by first sorting the
229! vector of values (x) for which the CDF needs to be computed
230do i = 1, p%ens_size
231 ! Figure out which bin it is in
232 call bnrh_cdf_initialized(x(i), p%ens_size, p%ens, p%bounded_below, p%bounded_above, p%lower_bound, p%upper_bound, &
233 tail_amp_left, tail_mean_left, tail_sd_left, do_uniform_tail_left, &
234 tail_amp_right, tail_mean_right, tail_sd_right, do_uniform_tail_right, q, quantiles(i))
235end do
236
237end subroutine bnrh_cdf_initialized_vector

241subroutine bnrh_cdf_initialized(x, ens_size, sort_ens, bounded_below, bounded_above, &
242 lower_bound, upper_bound, &
243 tail_amp_left, tail_mean_left, tail_sd_left, do_uniform_tail_left, &
244 tail_amp_right, tail_mean_right, tail_sd_right, do_uniform_tail_right,&
245 q, quantile)
246
247real(r8), intent(in) :: x
248integer, intent(in) :: ens_size
249real(r8), intent(in) :: sort_ens(ens_size)
250logical, intent(in) :: bounded_below, bounded_above
251real(r8), intent(in) :: lower_bound, upper_bound
252real(r8), intent(in) :: tail_amp_left, tail_mean_left, tail_sd_left
253real(r8), intent(in) :: tail_amp_right, tail_mean_right, tail_sd_right
254logical, intent(in) :: do_uniform_tail_left, do_uniform_tail_right
255real(r8), intent(in) :: q(ens_size)
256real(r8), intent(out) :: quantile
257
258real(r8) :: upper_q, fract, del_q, q_at_largest_ens
259integer :: j
260
261! Quantile increment between ensemble members for bnrh
262del_q = 1.0_r8 / (ens_size + 1.0_r8)
263
264if(x < sort_ens(1)) then
265 ! In the left tail
266 ! Do an error check to make sure ensemble member isn't outside bounds, may be redundant
267 if(bounded_below .and. x < lower_bound) then
268 write(errstring, *) 'Ensemble member less than lower bound', x, lower_bound
269 call error_handler(e_err, 'bnrh_cdf_initialized', errstring, source)
270 ! This error can occur due to roundoff in increment generation from BNRHF
271 ! See discussion in function fix_bounds.
272 endif
273
274 if(do_uniform_tail_left) then
275 ! Uniform approximation for left tail; Note that denominator cannot be 0 but could be small
276 quantile = (x - lower_bound) / (sort_ens(1) - lower_bound) * del_q
277 else
278 ! It's a normal tail
279 if(bounded_below) then
280 quantile = tail_amp_left * (normal_cdf(x, tail_mean_left, tail_sd_left) - &
281 normal_cdf(lower_bound, tail_mean_left, tail_sd_left))
282 else ! Unbounded, tail normal goes all the way down to quantile 0, amplitude is 1
283 quantile = (normal_cdf(x, tail_mean_left, tail_sd_left) / &
284 normal_cdf(sort_ens(1), tail_mean_left, tail_sd_left)) * del_q
285 endif
286 ! Make sure it doesn't sneak past the quantile of the smallest ensemble member due to round-off
287 quantile = min(quantile, q(1))
288 endif
289elseif(x == sort_ens(1)) then
290 ! This takes care of cases where there are multiple bnrh values at the bdry or at first ensemble
291 quantile = q(1)
292elseif(x > sort_ens(ens_size)) then
293 ! In the right tail
294 ! Do an error check to make sure ensemble member isn't outside bounds, may be redundant
295 if(bounded_above .and. x > upper_bound) then
296 write(errstring, *) 'Ensemble member greater than upper bound first check(see code)', x, upper_bound
297 call error_handler(e_err, 'bnrh_cdf_initialized', errstring, source)
298 ! This error can occur due to roundoff in increment generation from bounded BNRHF
299 ! See discussion in function fix_bounds
300 endif
301
302 if(do_uniform_tail_right) then
303 ! Uniform approximation for right tail
304 ! The division here could be a concern. However, if sort_ens(ens_size) == upper_bound, then
305 ! x cannot be > sort_ens(ens_size).
306 quantile = ens_size *del_q + &
307 (x - sort_ens(ens_size)) / (upper_bound - sort_ens(ens_size)) * del_q
308 else
309 ! It's a normal tail
310 q_at_largest_ens = normal_cdf(sort_ens(ens_size), tail_mean_right, tail_sd_right)
311 ! Want to avoid quantiles exceeding 1 due to numerical issues. Do fraction of the normal part
312 if(bounded_above) then
313 upper_q = tail_amp_right * normal_cdf(upper_bound, tail_mean_right, tail_sd_right)
314 fract = (tail_amp_right * normal_cdf(x, tail_mean_right, tail_sd_right) - &
315 tail_amp_right * q_at_largest_ens) / (upper_q - tail_amp_right * q_at_largest_ens)
316 else
317 ! Normal goes all the way to infinity, amplitude is 1, q at infinity is 1
318 fract = (normal_cdf(x, tail_mean_right, tail_sd_right) - q_at_largest_ens) / &
319 (1.0_r8 - q_at_largest_ens)
320 endif
321
322 quantile = ens_size * del_q + fract * del_q
323 quantile = min(quantile, 1.0_r8)
324 endif
325
326else
327 ! In an interior bin
328 do j = 1, ens_size - 1
329 if(x < sort_ens(j+1)) then
330 ! The division here could be a concern.
331 ! However, sort_ens(j)< x < sort_ens(j+1) so the two cannot be equal
332 quantile = j * del_q + &
333 ((x - sort_ens(j)) / (sort_ens(j+1) - sort_ens(j))) * del_q
334 exit
335 elseif(x == sort_ens(j+1)) then
336 quantile = q(j+1)
337 exit
338 endif
339 enddo
340endif
341
342end subroutine bnrh_cdf_initialized

368subroutine inv_bnrh_cdf(quantiles, ens_size, sort_ens, &
369 bounded_below, bounded_above, lower_bound, upper_bound, &
370 tail_amp_left, tail_mean_left, tail_sd_left, do_uniform_tail_left, &
371 tail_amp_right, tail_mean_right, tail_sd_right, do_uniform_tail_right, x)
372
373integer, intent(in) :: ens_size
374real(r8), intent(in) :: quantiles(ens_size)
375real(r8), intent(in) :: sort_ens(ens_size)
376logical, intent(in) :: bounded_below, bounded_above
377real(r8), intent(in) :: lower_bound, upper_bound
378real(r8), intent(in) :: tail_amp_left, tail_mean_left, tail_sd_left
379real(r8), intent(in) :: tail_amp_right, tail_mean_right, tail_sd_right
380logical, intent(in) :: do_uniform_tail_left, do_uniform_tail_right
381real(r8), intent(out) :: x(ens_size)
382
383integer :: region, i, j
384real(r8) :: lower_state, upper_state, lower_mass, upper_mass, target_mass
385real(r8) :: q(ens_size), curr_q, lower_q, upper_q, del_q, fract
386
387! Quantile increment between ensemble members for bnrh
388del_q = 1.0_r8 / (ens_size + 1.0_r8)
389
390do i = 1, ens_size
391 q(i) = i * del_q
392end do
393
394! Loop through each ensemble member to find posterior state
395do i = 1, ens_size
396 curr_q = quantiles(i)
397 ! Which region is this quantile in?
398 ! BNRH quantiles are uniform; finding region for this quantile is trivial
399 region = floor(curr_q * (ens_size + 1.0_r8))
400 ! Careful about numerical issues moving outside of region [0 ens_size]
401 if(region < 0) region = 0
402 if(region > ens_size) region = ens_size
403
404 if(region == 0) then
405 ! Lower tail
406 if(bounded_below .and. do_uniform_tail_left) then
407 ! Lower tail uniform
408 upper_state = sort_ens(1)
409 x(i) = lower_bound + (curr_q / q(1)) * (upper_state - lower_bound)
410 else
411 ! Find the mass at the lower bound (which could be unbounded)
412 if(bounded_below) then
413 lower_mass = tail_amp_left * &
414 normal_cdf(lower_bound, tail_mean_left, tail_sd_left)
415 else
416 lower_mass = 0.0_r8
417 endif
418 ! Find the mass at the upper bound (ensemble member 1)
419 upper_mass = tail_amp_left * &
420 normal_cdf(sort_ens(1), tail_mean_left, tail_sd_left)
421 ! What fraction of this mass difference should we go?
422 fract = curr_q / q(1)
423 target_mass = lower_mass + fract * (upper_mass - lower_mass)
424 x(i) = inv_weighted_normal_cdf(tail_amp_left, tail_mean_left, &
425 tail_sd_left, target_mass)
426 endif
427
428 elseif(region == ens_size) then
429 ! Upper tail
430 if(bounded_above .and. do_uniform_tail_right) then
431 ! Upper tail is uniform
432 lower_state = sort_ens(ens_size)
433 upper_state = upper_bound
434 x(i) = lower_state + (curr_q - q(ens_size)) * &
435 (upper_state - lower_state) / (1.0_r8 - q(ens_size))
436 else
437 ! Upper tail is (bounded) normal
438 ! Find the mass at the upper bound (which could be unbounded)
439 if(bounded_above) then
440 upper_mass = tail_amp_right * &
441 normal_cdf(upper_bound, tail_mean_right, tail_sd_right)
442 else
443 upper_mass = 1.0_r8
444 endif
445 ! Find the mass at the lower edge of the region (ensemble member n)
446 lower_mass = tail_amp_right * &
447 normal_cdf(sort_ens(ens_size), tail_mean_right, tail_sd_right)
448 ! What fraction of the last interval do we need to move
449 fract = (curr_q - q(ens_size)) / (1.0_r8 - q(ens_size))
450 target_mass = lower_mass + fract * (upper_mass - lower_mass)
451 x(i) = inv_weighted_normal_cdf(tail_amp_right, tail_mean_right, &
452 tail_sd_right, target_mass)
453 endif
454
455 else
456 ! Interior region; get the quantiles of the region boundary
457 lower_q = q(region)
458 upper_q = q(region + 1)
459 x(i) = sort_ens(region) + ((curr_q - lower_q) / (upper_q - lower_q)) * &
460 (sort_ens(region + 1) - sort_ens(region))
461 endif
462
463 ! Imprecision can lead to x being slightly out of bounds, fix it to bounds
464 call check_bounds(x(i), curr_q, bounded_below, lower_bound, &
465 bounded_above, upper_bound, 'inf_bnrh_cdf')
466enddo
467
468end subroutine inv_bnrh_cdf

473subroutine inv_bnrh_cdf_like(quantiles, ens_size, sort_ens, &
474 bounded_below, bounded_above, lower_bound, upper_bound, &
475 tail_amp_left, tail_mean_left, tail_sd_left, do_uniform_tail_left, &
476 tail_amp_right, tail_mean_right, tail_sd_right, do_uniform_tail_right, x, &
477 like)
478
479integer, intent(in) :: ens_size
480real(r8), intent(in) :: quantiles(ens_size)
481real(r8), intent(in) :: sort_ens(ens_size)
482logical, intent(in) :: bounded_below, bounded_above
483real(r8), intent(in) :: lower_bound, upper_bound
484real(r8), intent(in) :: tail_amp_left, tail_mean_left, tail_sd_left
485real(r8), intent(in) :: tail_amp_right, tail_mean_right, tail_sd_right
486logical, intent(in) :: do_uniform_tail_left, do_uniform_tail_right
487real(r8), intent(out) :: x(ens_size)
488real(r8), intent(inout) :: like(ens_size)
489
490! This inverts the cdf which is optionally multiplied by a likelihood.
491
492integer :: region, i, j
493real(r8) :: lower_state, upper_state, lower_mass, upper_mass, target_mass
494real(r8) :: q(ens_size), curr_q, amp_adj, lower_q, upper_q, del_q, fract
495
496! Quantile increment between ensemble members for bnrh
497del_q = 1.0_r8 / (ens_size + 1.0_r8)
498
499! Normalize the likelihood to have a sum of 1
500like = like / (sum(like) + like(1) / 2.0_r8 + like(ens_size) / 2.0_r8)
501
502! Go from left to right adjusting the quantiles through the x's
503! Assume that the quantiles of the original ensemble for the BNRH are uniform
504q(1) = like(1)
505do i = 2, ens_size
506 q(i) = q(i - 1) + (like(i-1) + like(i)) / 2.0_r8
507end do
508
509! Temporary test to confirm posterior is a pdf
510if(abs(q(ens_size) + like(ens_size) - 1.0_r8) > 1.0e-12) then
511 write(*, *) 'final q ', q(ens_size) + like(ens_size)
512 stop
513endif
514
515! Loop through each ensemble member to find posterior state
516do i = 1, ens_size
517 curr_q = quantiles(i)
518 ! Which region is this quantile in?
519 ! Find which region this quantile is in
520 ! Need to make this more efficient once it is working
521 ! Default is that region is the highest one; quantile(i) >= largest q
522 region = ens_size
523 do j = 1, ens_size
524 if(curr_q < q(j)) then
525 region = j - 1
526 exit
527 endif
528 end do
529
530 if(region == 0) then
531 ! Lower tail
532 if(bounded_below .and. do_uniform_tail_left) then
533 ! Lower tail uniform
534 upper_state = sort_ens(1)
535 x(i) = lower_bound + (curr_q / q(1)) * (upper_state - lower_bound)
536 else
537 ! Find the mass at the lower bound (which could be unbounded)
538 ! The amplitude is changed if there is a non-uniform likelihood
539 amp_adj = q(1) / del_q
540 if(bounded_below) then
541 lower_mass = amp_adj * tail_amp_left * &
542 normal_cdf(lower_bound, tail_mean_left, tail_sd_left)
543 else
544 lower_mass = 0.0_r8
545 endif
546 ! Find the mass at the upper bound (ensemble member 1)
547 upper_mass = amp_adj * tail_amp_left * &
548 normal_cdf(sort_ens(1), tail_mean_left, tail_sd_left)
549 ! What fraction of this mass difference should we go?
550 fract = curr_q / q(1)
551 target_mass = lower_mass + fract * (upper_mass - lower_mass)
552 x(i) = inv_weighted_normal_cdf(amp_adj*tail_amp_left, tail_mean_left, &
553 tail_sd_left, target_mass)
554 endif
555
556 elseif(region == ens_size) then
557 ! Upper tail
558 if(bounded_above .and. do_uniform_tail_right) then
559 ! Upper tail is uniform
560 lower_state = sort_ens(ens_size)
561 upper_state = upper_bound
562 x(i) = lower_state + (curr_q - q(ens_size)) * &
563 (upper_state - lower_state) / (1.0_r8 - q(ens_size))
564 else
565 ! Upper tail is (bounded) normal
566 ! Find the mass at the upper bound (which could be unbounded)
567 amp_adj = (1.0_r8 - q(ens_size)) / del_q
568 if(bounded_above) then
569 upper_mass = amp_adj * tail_amp_right * &
570 normal_cdf(upper_bound, tail_mean_right, tail_sd_right)
571 else
572 upper_mass = amp_adj * 1.0_r8
573 endif
574 ! Find the mass at the lower edge of the region (ensemble member n)
575 lower_mass = amp_adj * tail_amp_right * &
576 normal_cdf(sort_ens(ens_size), tail_mean_right, tail_sd_right)
577 ! What fraction of the last interval do we need to move
578 fract = (curr_q - q(ens_size)) / (1.0_r8 - q(ens_size))
579 target_mass = lower_mass + fract * (upper_mass - lower_mass)
580 x(i) = inv_weighted_normal_cdf(amp_adj * tail_amp_right, tail_mean_right, &
581 tail_sd_right, target_mass)
582 endif
583
584 else
585 ! Interior region; get the quantiles of the region boundary
586 lower_q = q(region)
587 upper_q = q(region + 1)
588 x(i) = sort_ens(region) + ((curr_q - lower_q) / (upper_q - lower_q)) * &
589 (sort_ens(region + 1) - sort_ens(region))
590 endif
591
592 ! Imprecision can lead to x being slightly out of bounds, fix it to bounds
593 call check_bounds(x(i), curr_q, bounded_below, lower_bound, &
594 bounded_above, upper_bound, 'inf_bnrh_cdf_like')
595enddo
596
597end subroutine inv_bnrh_cdf_like

601subroutine ens_quantiles(sorted_ens, ens_size, bounded_below, bounded_above, &
602 lower_bound, upper_bound, q)
603
604! Given sorted ensemble which may have members identical to the bounds or may contain
605! duplicates, compute the quantiles for each member in an bounded normal rh distribution
606
607integer, intent(in) :: ens_size
608real(r8), intent(in) :: sorted_ens(ens_size)
609logical, intent(in) :: bounded_below, bounded_above
610real(r8), intent(in) :: lower_bound
611real(r8), intent(in) :: upper_bound
612real(r8), intent(out) :: q(ens_size)
613
614integer :: i, j, lower_dups, upper_dups, d_start, d_end, series_num
615integer :: series_start(ens_size), series_end(ens_size), series_length(ens_size)
616
617! Get number of ensemble members that are duplicates of the lower bound
618lower_dups = 0
619if(bounded_below) then
620 do i = 1, ens_size
621 if(sorted_ens(i) == lower_bound) then
622 lower_dups = lower_dups + 1
623 else
624 exit
625 endif
626 end do
627endif
628
629! Get number of ensemble members that are duplicates of the upper bound
630upper_dups = 0
631if(bounded_above) then
632 do i = ens_size, 1, -1
633 if(sorted_ens(i) == upper_bound) then
634 upper_dups = upper_dups + 1
635 else
636 exit
637 endif
638 end do
639endif
640
641! If there are duplicate ensemble members away from the boundaries need to revise quantiles
642! Make sure not to count duplicates already handled at the boundaries
643! Outer loop determines if a series of duplicates starts at sorted index i
644d_start = lower_dups + 1
645d_end = ens_size - upper_dups
646
647! Get start, length, and end of each series of duplicates away from the bounds
648series_num = 1
649series_start(series_num) = d_start
650series_length(series_num) = 1
651do i = d_start + 1, d_end
652 if(sorted_ens(i) == sorted_ens(i - 1)) then
653 series_length(series_num) = series_length(series_num) + 1
654 else
655 series_end(series_num) = i-1
656 series_num = series_num + 1
657 series_start(series_num) = i
658 series_length(series_num) = 1
659 endif
660end do
661
662! Off the end, finish up the last series
663series_end(series_num) = d_end
664
665! Now get the value of the quantile for the exact ensemble members
666! Start with the lower bound duplicates
667do i = 1, lower_dups
668 q(i) = lower_dups / (2.0_r8 * (ens_size + 1.0_r8))
669end do
670
671! Top bound duplicates next
672do i = ens_size - upper_dups + 1, ens_size
673 q(i) = 1.0_r8 - upper_dups / (2.0_r8 * (ens_size + 1.0_r8))
674end do
675
676! Do the interior series
677do i = 1, series_num
678 do j = series_start(i), series_end(i)
679 q(j) = series_start(i) / (ens_size + 1.0_r8) + (series_length(i) - 1.0_r8) / (2.0_r8 * (ens_size + 1.0_r8))
680 end do
681end do
682
683end subroutine ens_quantiles
