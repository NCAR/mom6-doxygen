47integer, parameter :: num_copies = 1, & ! number of copies in sequence
48 num_qc = 1 ! number of QC entries
49
50character(len=256) :: next_infile
51character(len=512) :: string1, string2
52
53integer :: ncid, num_files, io, iunit, filenum
54integer :: day, seconds
55integer :: num_new_obs, obs_num, num_obs_in_1_file
56integer :: i, j, k, num_latitudes, num_longitudes
57
58logical :: first_obs, from_list = .false.
59
60real(r8) :: oerr, qc, lat, lon, vertvalue
61real(r8) :: smx_fill, smxe_fill
62
63type(obs_sequence_type) :: obs_seq
64type(obs_type) :: obs, prev_obs
65type(time_type) :: obs_time
66
67real(r8), allocatable :: latitude(:), longitude(:)
68real(r8), allocatable :: soil_moisture_x(:,:)
69real(r8), allocatable :: sm_x_error(:,:)
70
71!------------------------------------------------------------------------
72! Declare namelist parameters
73!------------------------------------------------------------------------
74
75character(len=256) :: input_file = ''
76character(len=256) :: input_file_list = ''
77character(len=256) :: output_file = 'obs_seq.lprm'
78logical :: debug = .false.
79real(r8) :: lon_bounds(2) = (/ 0.0_r8, 360.0_r8 /)
80real(r8) :: lat_bounds(2) = (/ -90.0_r8, 90.0_r8 /)
81
82! representation error, which depends on model/data grid size,
83real(r8) :: lprm_rep_error = 0.1 ! representativeness error variance
84
85namelist /lprm_l3_to_obs_nml/ input_file, &
86 input_file_list, output_file, &
87 debug, lprm_rep_error, &
88 lon_bounds, lat_bounds
89
90! start of executable code
91
92call set_calendar_type(gregorian)
93
94! read the necessary parameters from input.nml
95call initialize_utilities()
96
97call find_namelist_in_file('input.nml', 'LPRM_L3_to_obs_nml', iunit)
98read(iunit, nml = lprm_l3_to_obs_nml, iostat = io)
99call check_namelist_read(iunit, io, 'LPRM_L3_to_obs_nml')
100
101! Record the namelist values used for the run
102if (do_nml_file()) write(nmlfileunit, nml=lprm_l3_to_obs_nml)
103if (do_nml_term()) write( * , nml=lprm_l3_to_obs_nml)
104
105! cannot have both a single filename and a list;
106! the namelist must shut one off.
107
108if (input_file /= '' .and. input_file_list /= '') then
109 call error_handler(e_err, 'LPRM_L3_to_obs', &
110 'One of input_file or filelist must be NULL', &
111 source)
112endif
113if (input_file_list /= '') from_list = .true.
114
115!-----------------------------------------------------------------------
116! Get number of observations in a single file
117
118if (from_list) then
119 next_infile = get_next_filename(input_file_list, 1)
120 call find_textfile_dims(input_file_list, num_files)
121else
122 next_infile = input_file
123 num_files = 1
124endif
125
126ncid = nc_open_file_readonly(next_infile, 'LPRM_L3_to_obs')
127num_latitudes = nc_get_dimension_size(ncid,'Latitude')
128num_longitudes = nc_get_dimension_size(ncid,'Longitude')
129call nc_close_file(ncid)
130
131num_obs_in_1_file = num_latitudes*num_longitudes
132
133num_new_obs = num_obs_in_1_file * num_files
134
135call static_init_obs_sequence()
136call init_obs(obs, num_copies, num_qc)
137call init_obs(prev_obs, num_copies, num_qc)
138call init_obs_sequence(obs_seq, num_copies, num_qc, num_new_obs)
139
140do k = 1, num_copies
141 call set_copy_meta_data(obs_seq, k, 'observation')
142enddo
143
144do k = 1, num_qc
145 call set_qc_meta_data(obs_seq, k, 'original QC')
146enddo
147
148obs_num = 1
149first_obs = .true.
150
151! main loop that does either a single file or a list of files
152
153filenum = 1
154fileloop: do ! until out of files
155
156 ! get the single name, or the next name from a list
157 if (from_list) then
158 next_infile = get_next_filename(input_file_list, filenum)
159 else
160 next_infile = input_file
161 if (filenum > 1) next_infile = ''
162 endif
163 if (next_infile == '') exit fileloop
164
165 ncid = nc_open_file_readonly(next_infile, 'LPRM_L3_to_obs')
166 num_latitudes = nc_get_dimension_size(ncid,'Latitude')
167 num_longitudes = nc_get_dimension_size(ncid,'Longitude')
168
169 allocate(latitude( num_latitudes))
170 allocate(longitude( num_longitudes))
171 allocate(soil_moisture_x(num_latitudes, num_longitudes))
172 allocate(sm_x_error( num_latitudes, num_longitudes))
173
174 call getvar_real(ncid, 'Longitude', longitude)
175 call getvar_real(ncid, 'Latitude', latitude)
176 call get_2dshort_as_r8(ncid, 'soil_moisture_x', soil_moisture_x, smx_fill)
177 call get_2dshort_as_r8(ncid, 'sm_x_error', sm_x_error, smxe_fill)
178 call nc_close_file(ncid)
179
180 ! Daniel wants to convert from
181 ! soil_moisture_x:long_name = "Volumetric Soil Moisture from X-band" ;
182 ! soil_moisture_x:units = "percent" ;
183
184 where( soil_moisture_x /= smx_fill ) soil_moisture_x = soil_moisture_x/100.0_r8
185
186 ! time is determined from the file name
187
188 obs_time = get_time_from_name(next_infile)
189
190 call get_time(obs_time, seconds, day)
191
192 ! ensure longitudes are [0,360]
193 where(longitude < 0.0_r8) longitude = longitude + 360.0_r8
194
195 lonloop: do i = 1, num_longitudes
196 lon = longitude(i)
197 if ( lon < lon_bounds(1) ) cycle lonloop
198 if ( lon > lon_bounds(2) ) cycle lonloop
199
200 latloop: do j = 1, num_latitudes
201
202 lat = latitude( j)
203
204 if ( lat < lat_bounds(1) ) cycle latloop
205 if ( lat > lat_bounds(2) ) cycle latloop
206
207 qc = 0.0
208
209 if ( lat < lat_bounds(1) ) cycle latloop
210
211 !------------------------------------------------------------------
212 ! observation error is instrument plus representativeness
213
214 oerr = sm_x_error(j,i) + lprm_rep_error
215 vertvalue = 0.005_r8 ! 5mm depth for X band
216
217 if (soil_moisture_x(j,i) == smx_fill) cycle latloop
218 if ( sm_x_error(j,i) == smxe_fill) cycle latloop
219
220 ! If you want to geographically subset, do it here.
221
222 call create_3d_obs(lat, lon, vertvalue, vertisheight, &
223 soil_moisture_x(j,i), lprm_soil_moisture, &
224 oerr, day, seconds, qc, obs)
225
226 ! first one, insert with no prev. otherwise, since all times are the
227 ! same for this column, insert with the prev obs as the starting point.
228 ! (the first insert with no prev means it will search for the right
229 ! time ordered starting point.)
230 if (first_obs) then
231 call insert_obs_in_seq(obs_seq, obs)
232 first_obs = .false.
233 else
234 call insert_obs_in_seq(obs_seq, obs, prev_obs)
235 endif
236 obs_num = obs_num+1
237 prev_obs = obs
238
239 if (obs_num > num_new_obs) then
240 write(string1,*)'reached maximum number of observations while'
241 write(string2,*)'reading file # ',filenum,' "'//trim(next_infile)//'"'
242 call error_handler(e_err, 'LPRM_L3_to_obs', string1, &
243 source, text2=string2)
244 endif
245
246 enddo latloop
247 enddo lonloop
248
249 filenum = filenum + 1
250
251 deallocate(latitude)
252 deallocate(longitude)
253 deallocate(soil_moisture_x)
254 deallocate(sm_x_error)
255
256end do fileloop
257
258! done with main loop. if we added any obs to the sequence, write it out.
259if (obs_num > 0) then
260 print *, 'ready to write, nobs = ', get_num_obs(obs_seq)
261 if (get_num_obs(obs_seq) > 0) &
262 call write_obs_seq(obs_seq, output_file)
263
264 call destroy_obs(obs)
265 if (get_num_obs(obs_seq) > 0) call destroy_obs_sequence(obs_seq)
266else
267 call error_handler(e_err,'LPRM_L3_to_obs','no obs converted', &
268 source)
269endif
270
271call error_handler(e_msg,'LPRM_L3_to_obs','Finished successfully.')
272call finalize_utilities()
273
274
275contains

283function get_time_from_name(fname)
284character(len=*), intent(in) :: fname
285type(time_type) :: get_time_from_name
286
287integer :: indx, indx1, indx2
288integer :: year, month, day, hours, minutes, seconds
289
290indx = index(fname,'.nc') ! may not be universally applicable
291
292if (indx < 1) then
293 call error_handler(e_err,'get_time_from_name','cannot determine where to start', &
294 source, text2=fname)
295endif
296
297indx1 = indx - 8
298indx2 = indx
299
300write(*,*)'subset of time string is ',fname(indx1:indx2)
301
302read(fname(indx1:indx2),'(i4,2(i2))')year,month,day
303
304! According to Daniel
305hours = 1
306minutes = 30
307seconds = 0
308
309get_time_from_name = set_date(year,month,day,hours,minutes,seconds)
310
311end function get_time_from_name
