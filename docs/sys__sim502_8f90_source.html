7! Looks at how to deal with correlation errors given null assumption about
8! a uniform distribution for the underlying true correlation distirubtions.
9! Generates a huge sample of [-1, 1) uniform correlations, then samples these,
10! and keeps track of the sample correlation / true correlation pairs. Then
11! it bins these together over small sample correlation ranges and looks at
12! the value of alpha that would minimize errors. NOTE: this gives much less
13! info than group filter because group filter draws from a much more
14! meaningful prior (i.e. from what the model and assim have generated).
15
16use types_mod, only : r8
17use random_seq_mod, only : random_seq_type, init_random_seq, random_gaussian, &
18 twod_gaussians, random_uniform
19
20implicit none
21
22integer, parameter :: num_bins = 50
23real(r8) :: sum(num_bins) = 0.0, sum2(num_bins)
24integer :: num(num_bins) = 0.0
25type (random_seq_type) :: r
26real(r8) :: correl, cov(2, 2), zero_2(2) = 0.0, correl_mean, correl_var, sd
27real(r8) :: q, alpha, alpha2, mean_abs_correl, sample_correl, real_correl
28real(r8), allocatable :: pairs(:, :)
29real(r8) :: bottom_cor, top_cor, sum_sample2, sum_sample_real, sum_real
30real(r8) :: q_factor
31integer :: ens_size, n_samples, i, j, k, sample_size, bin
32
33! Generate a bazillion cases where the background correlation is assumed to
34! be U(-1, 1)
35! For each case sampled from this, create a SAMPLE correlation
36! and store the sample correlation and the actual correlation in a huge
37! array.
38
39! Initialize repeatable random sequence
40call init_random_seq(r)
41
42! For now have a set of free parameters that may be too large
43write(*, *) 'Input ensemble size'
44read(*, *) ens_size
45write(*, *) 'Input number of samples for statistics'
46read(*, *) n_samples
47
48! Allocate storage for sample from this correlation
49allocate(pairs(2, ens_size))
50
51! Loop through a set of sample draws from uniform (-1, 1)
52do k = 1, n_samples
53 111 real_correl = -1.0 + 2.0 * random_uniform(r)
54 ! Generate the covariance matrix for this correlation
55 cov(1, 1) = 1.0; cov(2, 2) = 1.0
56 cov(1, 2) = real_correl; cov(2, 1) = real_correl
57
58 ! Loop to generate an ensemble size sample from this correl
59 ! Generate a random sample of size ens_size from something with this correlation
60 do i = 1, ens_size
61 call twod_gaussians(r, zero_2, cov, pairs(:, i))
62 end do
63
64 ! Compute the sample correlation
65 call comp_correl(pairs, ens_size, sample_correl)
66
67 !!! if(sample_correl > 0.1 .or. sample_correl < -0.1) goto 111
68
69 !!!write(*, *) k, real_correl, sample_correl
70
71 ! Now adjust the information for the appropriate sample_correl bin
72 bin = (sample_correl + 1.0) / 2.0 * (num_bins) + 1
73
74 num(bin) = num(bin) + 1
75 sum(bin) = sum(bin) + real_correl
76 sum2(bin) = sum2(bin) + real_correl*real_correl
77
78end do
79
80do i = 1, num_bins
81 ! Output sample and mean real for this bin
82 sample_correl = ((i - 0.5) / num_bins) * 2.0 - 1.0
83 real_correl = sum(i) / num(i)
84 sd = sqrt((sum2(i) - num(i) * real_correl**2) / (num(i) - 1.0))
85 q = sd / abs(real_correl)
86 q_factor = (num(i) - q**2) / ((num(i) - 1) * q + num(i))
87 write(*, 11) i, num(i), sample_correl, real_correl, &
88 real_correl / sample_correl, sd, q_factor
89 11 format(i4, 1x, i8, 1x, 5(e10.4, 1x))
90end do
91
92if(1 == 1) stop
93
94!!! write(*, *) 'optimal alpha is ', sum_sample_real / sum_sample2, sample_size
95!!! write(*, *) 'mean real_correl is ', sum_real / sample_size
96
97end program sys_sim502
