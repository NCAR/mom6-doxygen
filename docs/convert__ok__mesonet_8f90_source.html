10!
11! convert_ok_mesonet - program to convert Oklahoma Mesonet MDF files
12! into DART observation sequence files.
13!
14! The observation files can be obtained from the Oklahoma Mesonet
15! archive using urls of the format:
16! http://www.mesonet.org/index.php/dataMdfMts/dataController/getFile/YYYYMMDDHHMM/mdf/TEXT/
17! where YYYYMMDDHHMM is the date and time of the desired set of observations.
18! files are available every 5 minutes.
19!
20! Static fields are drawn from the station description file provided by
21! the OK Mesonet. Update the local file from:
22! http://www.mesonet.org/index.php/api/siteinfo/from_all_active_with_geo_fields/format/csv/
23!
24! NOTE: you may want to consider using METAR surface ob errors
25! with Oklahoma Mesonet surface obs. See flag below.
26!
27! Written by G. Romine using the madis converters as a template, Aug. 2013
28!

281 end if
282
283 end if

287 dptk = rh_and_temp_to_dewpoint(relh/100.0_r8, tmpk) ! Kelvin temp, rh 0.0-1.0

312subroutine get_geo(search_stid,nlat,nlon,elevf)
313
314use utilities_mod, only : get_unit, find_namelist_in_file, check_namelist_read, &
315 do_nml_file, do_nml_term, logfileunit, nmlfileunit
316use types_mod, only : r8, missing_r8
317
318implicit none
319
320character(len=4), intent(in) :: search_stid
321real(r8), intent(out) :: nlat, nlon, elevf
322
323! Given a station id, find the lat, lon and elevation from the geo file
324! provided by the Oklahoma Mesonent website
325!
326! the geoinfo.csv file has data in the following order:
327!
328! stnm - station number
329! stid - CHAR len=4
330! name - CHAR of variable size
331! city - CHAR of variable size
332! rang - distance of station from city, float
333! cdir - CHAR of variable size
334! cnty - CHAR of variable size
335! nlat - latitude, float
336! nlon - longitude, float
337! elev - elevation, float
338! + other stuff we don't need
339!
340
341! temp vars
342integer :: stnm, iunit, elev
343character(len=4) :: stid
344character(len=15) :: name, city, cdir, cnty
345real :: rang
346character(len=330) :: line
347integer :: n1, n2, n3, n4, n5, n6, n7, n8, n9, n10, nend
348logical :: stn_found
349
350
351 stn_found = .false.
352 iunit = get_unit()
353 open(unit=iunit, file='geoinfo.csv', status='old')
354
355 read(iunit,*) ! header
356
357 do while (1 .eq. 1)
358! read through each line and look for a match
359 read(iunit,'(A330)',END=200) line
360
361! Ugh... a csv file with mixed var types is a
362! pain. Find the locations of delimiters and
363! assign vars as whatever is found between them.
364! Getting out to the 10th item gives us all we
365! need.
366! Determine locations of the first 10 delimiters
367n1 = index(line, ',')
368nend = len_trim(line)
369n2 = n1 + index(line(n1+1:nend), ',')
370n3 = n2 + index(line(n2+1:nend), ',')
371n4 = n3 + index(line(n3+1:nend), ',')
372n5 = n4 + index(line(n4+1:nend), ',')
373n6 = n5 + index(line(n5+1:nend), ',')
374n7 = n6 + index(line(n6+1:nend), ',')
375n8 = n7 + index(line(n7+1:nend), ',')
376n9 = n8 + index(line(n8+1:nend), ',')
377n10 = n9 + index(line(n9+1:nend), ',')
378
379!
380 read (line(1:n1-1),'(I3)') stnm
381 read (line(n1+1:n2-1),'(A4)') stid
382 read (line(n2+1:n3-1),'(A15)') name
383 read (line(n3+1:n4-1),'(A15)') city
384 read (line(n4+1:n5-1),'(f6.3)') rang
385 read (line(n5+1:n6-1),'(A3)') cdir
386 read (line(n6+1:n7-1),'(A15)') cnty
387 read (line(n7+1:n8-1),'(f10.7)') nlat
388 read (line(n8+1:n9-1),'(f10.7)') nlon
389 read (line(n9+1:n10-1),'(I4)') elev
390 elevf = 1.0_r8 * elev
391 if (search_stid .eq. stid) then
392 stn_found = .true.
393 goto 200
394 end if
395
396 end do
397200 continue
398 if (.not. stn_found) then
399 write(*,*) ' station ',stid,' not found. Update the geo file.'
400 end if
401 close(iunit)
402end subroutine get_geo
