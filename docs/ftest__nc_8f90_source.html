5program ftest_nc

7! fortran program which uses netcdf (and NOT mpi). netcdf 3.6.x and beyond
8! comes with a fortran 90 module to define netcdf interfaces and
9! parameters. typically the module is found by giving the compiler
10! a flag pointing to the netcdf include directory, but sometimes
11! it can be the lib dir.
12
13! if you get an error about the netcdf or typesizes module not
14! being found, you need to set NETCDF in your mkmf.template file
15! to the right directory, *OR* you can comment NETCDF out of
16! your template file and set it in your environment:
17! ksh & related: export NETCDF=xxx
18! csh & related: setenv NETCDF xxx
19! if your system uses the 'module' command, be sure you have a
20! compatible netCDF module loaded; typically the netCDF installation
21! needs to be compiled with the same Fortran compiler as you are using.
22
23use netcdf
24use typesizes
25
26implicit none
27
28character(len=32) :: filename = "ftestdata.nc"
29integer :: ncfileid, istat, i
30integer :: test1_length = 5
31integer :: test1dimid, dataid
32
33 print *, "program start"
34
35
36! Typical sequence:
37! NF90_OPEN ! create netCDF dataset: enter define mode
38! NF90_def_dim ! define dimensions: from name and length
39! NF90_def_var ! define variables: from name, type, and dims
40! NF90_put_att ! assign attribute values
41! NF90_ENDDEF ! end definitions: leave define mode
42! NF90_put_var ! provide values for variable
43! NF90_CLOSE ! close: save updated netCDF dataset
44
45 ! netcdf test to be sure this system can create all necessary kinds
46 if(.not. bytesizesok()) then
47 print *, 'Compiler does not support required kinds of variables.'
48 stop
49 end if
50
51!-------------------------------------------------------------------------------
52! Open/Create file
53!-------------------------------------------------------------------------------
54
55 istat = nf90_create(path = trim(filename), cmode = nf90_share, &
56 ncid = ncfileid)
57 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
58
59 print *, 'successfully opened ' // trim(filename)
60
61!-------------------------------------------------------------------------------
62! Define dimension(s)
63!-------------------------------------------------------------------------------
64
65 istat = nf90_def_dim(ncid=ncfileid, name="test1", &
66 len = test1_length, dimid = test1dimid)
67 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
68
69!-------------------------------------------------------------------------------
70! Write global attributes
71!-------------------------------------------------------------------------------
72
73 istat = nf90_put_att(ncfileid, nf90_global, "title", "netcdf test File")
74 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
75
76!-------------------------------------------------------------------------------
77! Create variables and attributes.
78!-------------------------------------------------------------------------------
79
80 istat = nf90_def_var(ncid=ncfileid, name="data", xtype=nf90_int, &
81 dimids=test1dimid, varid=dataid)
82 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
83
84 istat = nf90_put_att(ncfileid, dataid, "long_name", "test data array")
85 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
86
87!-------------------------------------------------------------------------------
88! Leave define mode so we can fill
89!-------------------------------------------------------------------------------
90 istat = nf90_enddef(ncfileid)
91 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
92
93!-------------------------------------------------------------------------------
94! Fill the coordinate variables.
95! The time variable is filled as time progresses.
96!-------------------------------------------------------------------------------
97
98 istat = nf90_put_var(ncfileid, dataid, (/ (i,i=1,5) /) )
99 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
100
101!-------------------------------------------------------------------------------
102! Close
103!-------------------------------------------------------------------------------
104
105 istat = nf90_close(ncfileid)
106 if (istat /= nf90_noerr) call netcdf_error_exit(istat)
107
108 print *, "netcdf file successfully closed"
109
110 print *, "program end"
111
112contains
113

124end subroutine netcdf_error_exit
125
126end program ftest_nc
