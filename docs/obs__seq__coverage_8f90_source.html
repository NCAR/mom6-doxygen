561function find_voxel_location(ObsType, ObsLocation) result(voxel_id)
562! Simply try to find a matching lat/lon for an observation type
563! The lons/lats get yanked around "a lot" - being converted from ASCII radians
564! to r8 degrees to r8 radians to r8 degrees and then checked for "equality".
565! So - we're actually just checking to see if the lat/lon is within something
566! like 500 cm either direction. Seems like a reasonable definition of 'match'.
567!
568! By this point the ObsLocation has one of the mandatory vertical levels.
569
570integer, intent(in) :: obstype
571type(location_type), intent(in) :: obslocation
572integer :: voxel_id
573
574integer :: i
575real(r8), dimension(3) :: obslocarray, stnlocarray
576real(r8) :: londiff, latdiff, hgtdiff
577
578voxel_id = 0
579
580if (num_voxels == 0) return
581
582obslocarray = get_location(obslocation)
583
584findloop : do i = 1,num_voxels
585
586 if (obstype /= voxels(i)%obs_type) cycle findloop
587
588 stnlocarray = get_location(voxels(i)%location)
589
590 londiff = abs(obslocarray(1) - stnlocarray(1))
591 latdiff = abs(obslocarray(2) - stnlocarray(2))
592
593 if (is_vertical(obslocation, "PRESSURE") ) then
594 hgtdiff = abs(obslocarray(3) - stnlocarray(3))
595 else
596 ! if the vertical is undefined or surface, then
597 ! the vertical will always match.
598 hgtdiff = 0.0_r8
599 endif
600
601 if ( (londiff <= half_meter) .and. &
602 (latdiff <= half_meter) .and. &
603 (hgtdiff < onepa) ) then ! within 1 Pa is close enough
604 voxel_id = i
605 exit findloop
606 endif
607
608enddo findloop
609
610end function find_voxel_location

619! enough space, must copy the info to a temporary list, deallocate/reallocate
620! copy the info back, and deallocate the temporary list. Ugh.
621
622integer, intent(in) :: obstype
623type(location_type), intent(in) :: obslocation
624integer, intent(in) :: ilevel
625integer :: voxel_id
626
627type(voxel_type), allocatable, dimension(:) :: templist
628integer :: i
629
630if ( num_voxels >= max_voxels ) then ! need to make room
631
632 if (verbose) write(*,*)'Doubling number of possible voxels from ', &
633 & num_voxels,' to ',2*max_voxels
634
635 ! Allocate temporary space; Copy.
636 ! Deallocate/nullify existing space.
637 ! Double the size of the existing space.
638 ! Copy the information back.
639 ! Deallocate/nullify the temporary space.
640 ! Actually add the new voxel information
641
642 ! Allocate the temporary space.
643 ! We'll worry about the number of time steps later.
644 call initialize_voxels(num_voxels, templist)
645
646 ! Copy the information to the temporary space.
647 duploop1 : do i = 1,num_voxels
648
649 templist(i)%obs_type = voxels(i)%obs_type
650 templist(i)%location = voxels(i)%location
651 templist(i)%levelindx = voxels(i)%levelindx
652 templist(i)%first_time = voxels(i)%first_time
653 templist(i)%last_time = voxels(i)%last_time
654 templist(i)%ntimes = voxels(i)%ntimes
655
656 ! Make sure the time array is the right size, then copy.
657 if (associated(templist(i)%times)) then
658 deallocate( templist(i)%times )
659 nullify( templist(i)%times )
660 endif
661 allocate( templist(i)%times( size(voxels(i)%times) ) )
662 templist(i)%times = voxels(i)%times
663
664 enddo duploop1
665
666 ! Deallocate the voxels, double the array length,
667 ! allocate the new space.
668 call destroy_voxels(voxels)
669 max_voxels = 2 * max_voxels
670 call initialize_voxels(max_voxels, voxels)
671
672 ! Copy the information BACK to the new space.
673 duploop2 : do i = 1,num_voxels
674
675 voxels(i)%obs_type = templist(i)%obs_type
676 voxels(i)%location = templist(i)%location
677 voxels(i)%levelindx = templist(i)%levelindx
678 voxels(i)%first_time = templist(i)%first_time
679 voxels(i)%last_time = templist(i)%last_time
680 voxels(i)%ntimes = templist(i)%ntimes
681
682 if (associated(voxels(i)%times)) then
683 deallocate( voxels(i)%times )
684 nullify( voxels(i)%times )
685 endif
686 allocate( voxels(i)%times( size(templist(i)%times) ) )
687 voxels(i)%times = templist(i)%times
688
689 enddo duploop2
690
691 ! Remove the temporary space.
692 call destroy_voxels(templist)
693
694endif
695
696! Add the new voxel information.
697! Create voxel with nominal (mandatory) vertical value.
698! The vertically_desired() routine replaces the ob vertical value with
699! the closest mandatory value, so we're good.
700
701num_voxels = num_voxels + 1
702voxel_id = num_voxels
703
704voxels(voxel_id)%obs_type = obstype
705voxels(voxel_id)%location = obslocation
706voxels(voxel_id)%levelindx = ilevel
707
708if (debug) then
709 call write_location(0,obslocation,'ascii',string1)
710 call write_location(0,voxels(voxel_id)%location,'ascii',string2)
711 write(*,*)
712 write(*,*)'Added voxel ',voxel_id,' for type ',obstype
713 write(*,*)'observation location', trim(string1)
714 write(*,*)'voxel location', trim(string2)
715 write(*,*)'voxel level', voxels(voxel_id)%levelindx
716 write(*,*)
717endif
718
719end function add_new_voxel

726
727! The voxel has a list of the observation times closest to the
728! verification times. Determine if the observation time is closer to
729! the verification time than what we already have.
730
731type(time_type), intent(in) :: obstime
732integer, intent(in) :: voxelid
733integer, intent(out) :: timeindex
734logical :: time_is_wanted
735
736type(time_type) :: stndelta, obdelta
737integer :: i
738
739timeindex = 0
740time_is_wanted = .false.
741
742! the time_minus function always returns a positive difference
743
744timeloop : do i = 1,num_verification_times
745
746 obdelta = obstime - all_verif_times(i)
747
748 ! If observation is not within half a verification step,
749 ! try the next verification time.
750 if (obdelta >= half_stride) cycle timeloop
751
752 ! we must be close now ...
753 stndelta = voxels(voxelid)%times(i) - all_verif_times(i)
754
755 ! Check to see if the observation is closer to the verification time
756 ! than the one we have.
757 if (obdelta < stndelta) then
758 if (debug) call print_time(voxels(voxelid)%times(i),'replacing ')
759 if (debug) call print_time(obstime,'with this observation time')
760 timeindex = i
761 time_is_wanted = .true.
762 exit timeloop
763 endif
764
765enddo timeloop
766
767end function time_is_wanted

1449subroutine set_required_times(analysis1, analysisN, flen_days, &
1450 flen_seconds, v_int_seconds, coverage_pcnt )
1451!-------------------------------------------------------------------------------
1452! Setting the required times for the verification consists of knowing
1453! the time of the first analysis through the time of the last analysis plus
1454! the forecast length (from the last analysis) AND the frequency of the
1455! verification_interval_seconds. An example with three analysis and
1456! 4 verify times per forecast. Results in 7 total times needed to verify.
1457!
1458! analysis1 o o o o
1459! analysis2 o o o o
1460! analysis3 o o o o
1461! |
1462! |__________________________________________________________________|
1463! T1 TN
1464! V1 V2 V3 V4 V5 V6 V7
1465!
1466! Global variables modified/set by this routine:
1467!
1468! num_verify_per_fcst the number of verification times per forecast
1469! num_analyses the number of forecast experiments
1470! num_verification_times the total number of verification times for ALL experiments
1471! nT_minimum the minimum number of verif times we can live with
1472! verification_times a matrix of times: rows are forecast runs,
1473! columns are verification times
1474! experiment_Tr8 same as verification_times ... just in digits12 format
1475
1476
1477integer, dimension(:), intent(in) :: analysis1 ! time of first analysis
1478integer, dimension(:), intent(in) :: analysisN ! time of last analysis
1479integer, intent(in) :: flen_days ! forecast length (days)
1480integer, intent(in) :: flen_seconds ! forecast length (seconds)
1481integer, intent(in) :: v_int_seconds ! verification interval (seconds)
1482real(r8), intent(in) :: coverage_pcnt ! temporal coverage percentage
1483
1484! declare some local variables
1485
1486type(time_type) :: flen ! forecast length
1487type(time_type) :: T1, TN, TimeMax, thistime, nexttime
1488
1489integer :: i, j, nsteps, seconds, days
1490
1491verification_stride = set_time(v_int_seconds,0) ! global variable
1492half_stride = verification_stride / 2
1493flen = set_time(flen_seconds, flen_days) ! set the forecast length
1494
1495t1 = set_date(analysis1(1), analysis1(2), analysis1(3), &
1496 analysis1(4), analysis1(5), analysis1(6) )
1497
1498tn = set_date(analysisn(1), analysisn(2), analysisn(3), &
1499 analysisn(4), analysisn(5), analysisn(6) )
1500
1501if ( tn < t1 ) then
1502 write(string1,*)'namelist: last_analysis must be >= first analysis'
1503 call error_handler(e_err,'set_required_times',string1,source)
1504endif
1505
1506! Check to make sure the forecast length is a multiple
1507! of the verification interval.
1508
1509nsteps = flen / verification_stride ! time_manager_mod:time_divide
1510nexttime = verification_stride * nsteps ! time_manager_mod:time_scalar_mult
1511
1512if (nexttime /= flen) then
1513 call print_time(verification_stride,'verification stride')
1514 call print_time(flen, 'original forecast length')
1515 call print_time(nexttime,'implied forecast length')
1516 write(string1,*)'namelist: forecast length is not a multiple of the verification interval'
1517 write(string2,*)'check forecast_length_[days,seconds] and verification_interval_seconds'
1518 call error_handler(e_err, 'set_required_times', string1, source, text2=string2)
1519endif
1520
1521num_verify_per_fcst = nsteps+1 ! SET GLOBAL VALUE
1522
1523if (verbose) then
1524 write(*,*) ! a little whitespace
1525 write(*,*)'There are ',num_verify_per_fcst,' verification times per forecast.'
1526endif
1527
1528! Check to make sure the last analysis is a multiple number
1529! of verification intervals from the first analysis.
1530! If we were to launch a forecast at every possible verification
1531! time (between analysis1 and analsyisN), how many would that be?
1532! Generates a list of potential analysis times.
1533! FIXME - pathological cases ... 3 hour verification, but
1534! FIXME - analysis times separated by 1 hour ???
1535
1536if (tn /= t1) then
1537
1538 thistime = tn - t1
1539 nsteps = thistime / verification_stride
1540 nexttime = verification_stride * nsteps
1541
1542 if (nexttime /= thistime) then
1543 call print_time(verification_stride,'verification stride')
1544 call print_time(thistime,'original analysis duration')
1545 call print_time(nexttime,'implied analysis duration')
1546
1547 ! FIXME - offer a last analysis time ... T1 + nint(steps)*verification_stride
1548
1549 write(string1,*)'namelist: last analysis time is not a multiple of the verification interval'
1550 write(string2,*)'check [first,last]_analysis and verification_interval_seconds'
1551 call error_handler(e_err, 'set_required_times', string1, source, text2=string2)
1552 endif
1553
1554 num_analyses = nsteps+1 ! SET GLOBAL VALUE
1555else
1556 num_analyses = 1 ! SET GLOBAL VALUE
1557endif
1558
1559if (verbose) write(*,*)'There are ',num_analyses,' supported forecasts.'
1560
1561! Now that we know the start/end/stride are correct,
1562! figure out the total number of verification times we need.
1563
1564timemax = tn + flen ! The time of the last verification
1565thistime = timemax - t1
1566nsteps = thistime / verification_stride
1567nexttime = verification_stride * nsteps
1568
1569if (nexttime /= thistime) then
1570 call print_time(verification_stride,'verification stride')
1571 call print_time(thistime,'total time interval')
1572 call print_time(nexttime,'implied total time')
1573 write(string1,*)'bad logic on Tims part. Should not be able to get here.'
1574 call error_handler(e_err,'set_required_times',string1,source)
1575endif
1576
1577num_verification_times = nsteps + 1 ! SET GLOBAL VALUE
1578nt_minimum = nint(num_verification_times * coverage_pcnt/100.0_r8) ! SET GLOBAL VALUE
1579
1580if (verbose) then
1581 write(string1,*)'Need ',num_verification_times, &
1582 ' verification times for 100.00% coverage.'
1583 write(string2,'(''at least '',i5,'' verification times for'',&
1584 & f7.2,''% coverage.'')') nt_minimum, coverage_pcnt
1585
1586 call error_handler(e_msg,'set_required_times',string1,text2=string2)
1587endif
1588
1589allocate(all_verif_times(num_verification_times))
1590allocate(verification_times(num_analyses,num_verify_per_fcst))
1591allocate(experiment_tr8(num_verify_per_fcst,num_analyses)) ! TRANSPOSED for netCDF
1592
1593! Fill the desired verification times
1594all_verif_times(1) = t1
1595do i=2,num_verification_times
1596 all_verif_times(i) = all_verif_times(i-1) + verification_stride
1597enddo
1598
1599! Use those desired times to fill the times for each forecast experiment
1600do i = 1,num_analyses ! SET GLOBAL VALUE
1601 verification_times(i, 1:num_verify_per_fcst) = &
1602 all_verif_times(i:(i+num_verify_per_fcst-1))
1603enddo
1604
1605! Use those to create the matching array of real(digits12)
1606
1607do j = 1,num_verify_per_fcst
1608do i = 1,num_analyses
1609 call get_time(verification_times(i,j),seconds,days)
1610 experiment_tr8(j,i) = days + seconds/(86400.0_digits12)
1611enddo
1612enddo
1613
1614
1615end subroutine set_required_times
