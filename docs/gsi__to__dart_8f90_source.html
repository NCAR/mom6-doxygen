257subroutine para_range(n1, n2, nprocs, irank, ista, iend)
258
259 use kinds, only: i_kind
260 implicit none
261
262 integer(i_kind), intent(in) :: n1, n2, nprocs, irank
263 integer(i_kind), intent(out) :: ista, iend
264
265 integer(i_kind) :: iwork1, iwork2
266
267 iwork1 = (n2 - n1 + 1) / nprocs
268 iwork2 = mod(n2 - n1 + 1, nprocs)
269 ista = irank * iwork1 + n1 + min(irank, iwork2)
270 iend = ista + iwork1 - 1
271 if (iwork2 > irank) iend = iend + 1
272 return
273end subroutine para_range
