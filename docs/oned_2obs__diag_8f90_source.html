930function innovzscore(obsval, prmean, prspred, errvar, qcval, qcmaxval)
931
932! This function tries to get a handle on the magnitude of the innovations.
933! If the ratio of the observation to the prior mean is 'big', it is an outlier.
934! If the prior mean cannot be calculated (i.e. is missing) we put it in the
935! last 'bin' of the crude histogram. This is pretty much a 'z' score in the
936! statistical sense.
937
938real(r8) :: innovzscore
939real(r8), intent(in) :: obsval, prmean, prspred, errvar
940integer, intent(in) :: qcval, qcmaxval
941
942real(r8) :: numer, denom
943
944if ( qcval <= qcmaxval ) then ! QC indicates a valid obs
945 numer = abs(prmean - obsval)
946 denom = sqrt( prspred**2 + errvar )
947 innovzscore = numer / denom
948else
949 innovzscore = real(maxsigmabins,r8)
950endif
951
952end function innovzscore

1211subroutine setschedule(bin1time, num_epochs, fullwidth, halfwidth, &
1212 bin_center, bin_edges, epoch_center, epoch_edges)
1213
1214type(time_type), intent(in) :: bin1time
1215integer, intent(in) :: num_epochs
1216type(time_type), intent(in) :: fullwidth
1217type(time_type), intent(in) :: halfwidth

1221real(digits12), dimension(:,:), intent(out) :: epoch_edges

1224! There is a chance the first bin center is at T=0, which is nigh
1225! impossible to subtract half a bin width from ...

1227! end of the bin MUST be a positive time, so it really pertains
1228! only to the first bin leading edge.
1229
1230iepoch = 1

1232if (bin1time <= halfwidth ) then
1233 bin_edges(1,iepoch) = set_time(0,0)
1234else
1235 bin_edges(1,iepoch) = bin1time - halfwidth + set_time(1,0)
1236endif
1237bin_edges(2,iepoch) = bin1time + halfwidth
1238
1239call get_time(bin_center(iepoch),seconds,days)
1240epoch_center(iepoch) = days + seconds/86400.0_digits12
1241
1242call get_time(bin_edges(1,iepoch),seconds,days)
1243epoch_edges(1,iepoch) = days + seconds/86400.0_digits12
1244
1245call get_time(bin_edges(2,iepoch),seconds,days)
1246epoch_edges(2,iepoch) = days + seconds/86400.0_digits12
1247
1248binloop : do iepoch = 2,num_epochs
1249
1250 bin_center( iepoch) = bin_center(iepoch-1) + fullwidth
1251 bin_edges(1,iepoch) = bin_center(iepoch) - halfwidth + set_time(1,0)
1252 bin_edges(2,iepoch) = bin_center(iepoch) + halfwidth
1253
1254 call get_time(bin_center(iepoch),seconds,days)
1255 epoch_center(iepoch) = days + seconds/86400.0_digits12
1256
1257 call get_time(bin_edges(1,iepoch),seconds,days)
1258 epoch_edges(1,iepoch) = days + seconds/86400.0_digits12
1259
1260 call get_time(bin_edges(2,iepoch),seconds,days)
1261 epoch_edges(2,iepoch) = days + seconds/86400.0_digits12
1262
1263enddo binloop
1264
1265if ( verbose ) then
1266 do iepoch = 1,num_epochs
1267 write(logfileunit,*)
1268 write( * ,*)
1269 write(string1,'(''epoch '',i6,'' start'')')iepoch
1270 write(string2,'(''epoch '',i6,'' center'')')iepoch
1271 write(string3,'(''epoch '',i6,'' end'')')iepoch
1272 call print_time(bin_edges(1,iepoch),string1,logfileunit)
1273 call print_time(bin_center( iepoch),string2,logfileunit)
1274 call print_time(bin_edges(2,iepoch),string3,logfileunit)
1275 call print_time(bin_edges(1,iepoch),string1)
1276 call print_time(bin_center( iepoch),string2)
1277 call print_time(bin_edges(2,iepoch),string3)
1278
1279 call print_date(bin_edges(1,iepoch),string1,logfileunit)
1280 call print_date(bin_center( iepoch),string2,logfileunit)
1281 call print_date(bin_edges(2,iepoch),string3,logfileunit)
1282 call print_date(bin_edges(1,iepoch),string1)
1283 call print_date(bin_center( iepoch),string2)
1284 call print_date(bin_edges(2,iepoch),string3)
1285 enddo
1286 write(logfileunit,*)
1287 write( * ,*)
1288endif

1487function rank_histogram(copyvalues, observation_index, error_variance ) result(rank)
1488
1489! Calculates the bin/rank
1490! We don't care about the QC value. If the ob wasn't assimilated
1491! the bin is meaningless.
1492
1493real(r8),dimension(:), intent(in) :: copyvalues
1494integer, intent(in) :: observation_index
1495real(r8), intent(in) :: error_variance
1496integer :: rank
1497
1498! Local Variables
1499
1500real(r8) :: obsvalue, mean, stddev
1501real(r8), dimension(ens_size) :: ensemble_values
1502real(r8), dimension(ens_size) :: sampling_noise
1503
1504! Grab the observation value from the myriad copy values.
1505
1506obsvalue = copyvalues(observation_index)
1507mean = 0.0_r8
1508stddev = sqrt(error_variance)
1509
1510! Grab the ensemble member values from the myriad copy values.
1511! It should be noted that the ensemble values here WILL INCLUDE the
1512! application of any prior inflation value ...
1513
1514do i = 1,ens_size
1515 ensemble_values(i) = copyvalues(ens_copy_index(i))
1516enddo
1517
1518! Create an array of 'observation error' ~ N(0,sigma)
1519! ran_seq is 'global' and has been initialized already
1520
1521call several_random_gaussians(ran_seq, mean, stddev, ens_size, sampling_noise)
1522
1523! Add the 'sampling noise' to the ensemble values,
1524! sort, and find the rank of the observation. Simple.
1525
1526ensemble_values = sort(ensemble_values + sampling_noise)
1527 ! ensemble_values = sort(ensemble_values) ! DEBUG - U-shaped valley, here we come.
1528
1529rank = 0
1530rankloop : do i = 1,ens_size
1531
1532 if ( obsvalue <= ensemble_values(i) ) then
1533 rank = i
1534 exit rankloop
1535 endif
1536
1537enddo rankloop
1538
1539if (rank == 0) then ! ob is larger than largest ensemble member.
1540 rank = ens_size + 1
1541endif
1542
1543
1544if ( .false. ) then ! DEBUG block
1545 write(*,*)'observation error variance is ',error_variance
1546 write(*,*)'observation value is ',obsvalue
1547 write(*,*)'observation rank is ',rank
1548 write(*,*)'noisy ensemble values are '
1549 write(*,*)ensemble_values
1550 write(*,*)
1551endif
1552
1553end function rank_histogram

1571!======================================================================
1572!> This function simply accumulates the appropriate sums.
1573!> The normalization occurs after all the data has been read, naturally.
1574!> ... spread is computed via sqrt(ensemble_spread**2 + observation_error**2).
1575
1576 subroutine bin3d(iqc, iepoch, iregion, flavor, trusted, &
1577 obsval, obserrvar, prmean, prsprd, pomean, posprd, rank)
1578
1579 ! If you are verifying the ensemble against imperfect (real) observations,
1580 ! it is necessary to account for the observation error when computing the
1581 ! spread. Since the observation error is not included as output from
1582 ! obs_diag, it is not possible to compute this quantity now.
1583
1584 integer, intent(in) :: iqc, iepoch, iregion, flavor
1585 logical, intent(in) :: trusted
1586 real(r8), intent(in) :: obsval, obserrvar, prmean, prsprd, pomean, posprd
1587 integer, intent(in) :: rank
1588
1589 real(r8) :: priorsqerr ! PRIOR Squared Error
1590 real(r8) :: priorbias ! PRIOR simple bias
1591 real(r8) :: priorspred ! PRIOR (spread,variance)
1592 real(r8) :: priorspredplus ! PRIOR (spread,variance**)
1593 real(r8) :: postsqerr ! POSTERIOR Squared Error
1594 real(r8) :: postbias ! POSTERIOR simple bias
1595 real(r8) :: postspred ! POSTERIOR (spread,variance)
1596 real(r8) :: postspredplus ! POSTERIOR (spread,variance**)
1597
1598 real(r8) :: priormean, postmean, obsmean
1599
1600 priorsqerr = (prmean - obsval)**2
1601 postsqerr = (pomean - obsval)**2
1602 priorbias = prmean - obsval
1603 postbias = pomean - obsval
1604 priorspred = prsprd**2
1605 postspred = posprd**2
1606 priorspredplus = prsprd**2 + obserrvar
1607 postspredplus = posprd**2 + obserrvar
1608 obsmean = obsval
1609 priormean = prmean
1610 postmean = pomean
1611
1612 !----------------------------------------------------------------------
1613 ! Track the number of possible observations
1614 !----------------------------------------------------------------------
1615
1616 call ipe(prior%Nposs(iepoch,iregion,flavor), 1)
1617 if (has_posteriors) &
1618 call ipe(poste%Nposs(iepoch,iregion,flavor), 1)
1619
1620 !----------------------------------------------------------------------
1621 ! Enforce the use of trusted observations.
1622 !----------------------------------------------------------------------
1623
1624 if ( trusted ) then
1625
1626 call ipe(prior%Ntrusted(iepoch,iregion,flavor), 1)
1627 if (has_posteriors) &
1628 call ipe(poste%Ntrusted(iepoch,iregion,flavor), 1)
1629
1630 ! Accrue the PRIOR quantities
1631 if ( any(iqc == trusted_prior_qcs) ) then
1632 call ipe(prior%Nused( iepoch,iregion,flavor), 1 )
1633 call rpe(prior%observation(iepoch,iregion,flavor), obsmean )
1634 call rpe(prior%ens_mean( iepoch,iregion,flavor), priormean )
1635 call rpe(prior%bias( iepoch,iregion,flavor), priorbias )
1636 call rpe(prior%rmse( iepoch,iregion,flavor), priorsqerr)
1637 call rpe(prior%spread( iepoch,iregion,flavor), priorspred)
1638 call rpe(prior%totspread( iepoch,iregion,flavor), priorspredplus)
1639 else
1640 call ipe(prior%NbadDartQC(iepoch,iregion,flavor), 1 )
1641 endif
1642
1643 ! Accrue the POSTERIOR quantities
1644 if (has_posteriors) then
1645 if ( any(iqc == trusted_poste_qcs) ) then
1646 call ipe(poste%Nused( iepoch,iregion,flavor), 1 )
1647 call rpe(poste%observation(iepoch,iregion,flavor), obsmean )
1648 call rpe(poste%ens_mean( iepoch,iregion,flavor), postmean )
1649 call rpe(poste%bias( iepoch,iregion,flavor), postbias )
1650 call rpe(poste%rmse( iepoch,iregion,flavor), postsqerr)
1651 call rpe(poste%spread( iepoch,iregion,flavor), postspred)
1652 call rpe(poste%totspread( iepoch,iregion,flavor), postspredplus)
1653 else
1654 call ipe(poste%NbadDartQC(iepoch,iregion,flavor), 1 )
1655 endif
1656 endif
1657
1658 return ! EXIT THE BINNING ROUTINE
1659 endif
1660
1661 !----------------------------------------------------------------------
1662 ! Proceed 'as normal'.
1663 !----------------------------------------------------------------------
1664
1665 if ( iqc > qc_max_prior ) then ! prior and posterior failed
1666
1667 call ipe(prior%NbadDartQC(iepoch,iregion,flavor), 1 )
1668 if (has_posteriors) &
1669 call ipe(poste%NbadDartQC(iepoch,iregion,flavor), 1 )
1670
1671 else if ( iqc > qc_max_posterior ) then
1672
1673 ! Then at least the prior (A.K.A. prior) is good
1674 call ipe(prior%Nused( iepoch,iregion,flavor), 1 )
1675 call rpe(prior%observation(iepoch,iregion,flavor), obsmean )
1676 call rpe(prior%ens_mean( iepoch,iregion,flavor), priormean )
1677 call rpe(prior%bias( iepoch,iregion,flavor), priorbias )
1678 call rpe(prior%rmse( iepoch,iregion,flavor), priorsqerr)
1679 call rpe(prior%spread( iepoch,iregion,flavor), priorspred)
1680 call rpe(prior%totspread( iepoch,iregion,flavor), priorspredplus)
1681
1682 ! However, the posterior is bad
1683 if (has_posteriors) &
1684 call ipe(poste%NbadDartQC(iepoch,iregion,flavor), 1 )
1685
1686 else
1687
1688 ! The prior is good
1689 call ipe(prior%Nused( iepoch,iregion,flavor), 1 )
1690 call rpe(prior%observation(iepoch,iregion,flavor), obsmean )
1691 call rpe(prior%ens_mean( iepoch,iregion,flavor), priormean )
1692 call rpe(prior%bias( iepoch,iregion,flavor), priorbias )
1693 call rpe(prior%rmse( iepoch,iregion,flavor), priorsqerr)
1694 call rpe(prior%spread( iepoch,iregion,flavor), priorspred)
1695 call rpe(prior%totspread( iepoch,iregion,flavor), priorspredplus)
1696
1697 ! The posterior is good
1698 if (has_posteriors) then
1699 call ipe(poste%Nused( iepoch,iregion,flavor), 1 )
1700 call rpe(poste%observation(iepoch,iregion,flavor), obsmean )
1701 call rpe(poste%ens_mean( iepoch,iregion,flavor), postmean )
1702 call rpe(poste%bias( iepoch,iregion,flavor), postbias )
1703 call rpe(poste%rmse( iepoch,iregion,flavor), postsqerr)
1704 call rpe(poste%spread( iepoch,iregion,flavor), postspred)
1705 call rpe(poste%totspread( iepoch,iregion,flavor), postspredplus)
1706 endif
1707
1708 endif
1709
1710 ! The rank histogram binning is a bit of a peculiar situation.
1711 ! Only the prior is of interest ... so DART QCs of 0 1 2 3 are 'good'.
1712 ! There is some debate about whether we should be considering the
1713 ! 'outlier' observations (DART QC == 7), so that is namelist controlled.
1714
1715 if ( (rank > 0) .and. create_rank_histogram ) then
1716 if ( any(iqc == hist_qcs(1:numqcvals) ) ) then
1717 call ipe(prior%hist_bin(iepoch,iregion,flavor,rank), 1)
1718 endif
1719 endif
1720
1721 end subroutine bin3d
