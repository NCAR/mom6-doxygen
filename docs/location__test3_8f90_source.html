112 enddo

123 endif
124 enddo
125 endif
126

134 cycle
135 endif
136 print *, 'nearest location index = ', near1index
137 call write_location(0, loc1(near1index), charstring=buf)
138 write(*,*) 'near loc ', trim(buf)
139
140enddo
141
142close(iunit)
143
144! Now read them back in and compute the distances from loc1
145open(iunit, file = 'location_test_file')
146
147do i = 1, nl
148 loc1(i) = read_location(iunit)
149enddo
150do i = 1, nl
151 loc2 = read_location(iunit)
152 write(*, *) 'distance ', i, ' is ', get_dist(loc1(i), loc2)
153enddo
154
155close(iunit)
156
157call finalize_utilities()
158
159end program location_test3
