6!> This program removes duplicate observations.
7
8!> This file contains 1 module and 1 program. the module code has
9!> to come first, so page down for the main program.
10!> The module is a custom sort routine needed to compare 2 observations,
11!> and then the program uses the sort to match duplicate observations
12!> in an obs_seq file.
13
14
15!> Module:
16!> Routine to sort observations at the same time into a consistent order.
17!> Obs sequence files are guarenteed to be traversed in time order;
18!> running them through the standard obs_sequence_tool will physically
19!> order them in the file in the same way a linked-list traversal would.
20!>
21!> But for multiple obs at the same time there is no way to indicate or
22!> control how to order them relative to each other. This tool sorts
23!> same-time obs based on location, then kind, then variance. If there
24!> are duplicate obs in the same file this helps get them together and
25!> pass through only one copy of the observation.
26

51function obssort(i, j)
52 integer, intent(in) :: i, j
53 integer :: obssort
54
55! this is requesting a compare of obs_this_bin(i) and obs_this_bin(j)
56! they should have identical times, so the compare needs to be by
57! location, type, value, etc. return -1 if i < j ; 0 if == ; 1 if i > j
58
59type(obs_def_type) :: this_obs_def1, this_obs_def2
60integer :: this_kind1, this_kind2
61type(location_type) :: this_loc1, this_loc2
62type(time_type) :: this_time1, this_time2
63real(r8) :: this_var1, this_var2
64real(r8) :: loc1(locationdims), loc2(locationdims) ! try for general?
65integer :: ndim
66character(len=129) :: locstring1, locstring2
67logical :: local_debug
68
69local_debug = .false.
70
71call get_obs_def(obs_this_bin(i), this_obs_def1)
72call get_obs_def(obs_this_bin(j), this_obs_def2)
73
74this_time1 = get_obs_def_time(this_obs_def1)
75this_time2 = get_obs_def_time(this_obs_def2)
76
77this_loc1 = get_obs_def_location(this_obs_def1)
78this_loc2 = get_obs_def_location(this_obs_def2)
79
80loc1 = get_location(this_loc1)
81loc2 = get_location(this_loc2)
82
83this_kind1 = get_obs_def_type_of_obs(this_obs_def1)
84this_kind2 = get_obs_def_type_of_obs(this_obs_def2)
85
86this_var1 = get_obs_def_error_variance(this_obs_def1)
87this_var2 = get_obs_def_error_variance(this_obs_def2)
88
89if (this_time1 /= this_time2) then
90 print *, 'error, times not the same'
91 print *, 'comparing items ', i, j
92 call print_time(this_time1, 'time1')
93 call print_time(this_time2, 'time2')
94 stop
95endif
96
97if (local_debug) then
98 print *, 'comparing items ', i, j
99 call print_time(this_time1, 'time: ')
100 print *, 'kinds ', this_kind1, this_kind2
101 print *, 'vars ', this_var1, this_var2
102 call write_location(0, this_loc1, charstring=locstring1)
103 call write_location(0, this_loc2, charstring=locstring2)
104 print *, 'locs: '
105 print *, trim(locstring1)
106 print *, trim(locstring2)
107 print *, ''
108endif
109
110! try for a general location solution
111do ndim=1, locationdims
112
113 if (loc1(ndim) > loc2(ndim)) then
114 obssort = 1
115 return
116 else if (loc1(ndim) < loc2(ndim)) then
117 obssort = -1
118 return
119 endif
120
121enddo
122
123! locations the same, so try types
124if (this_kind1 > this_kind2) then
125 obssort = 1
126 return
127else if (this_kind1 < this_kind2) then
128 obssort = -1
129 return
130endif
131
132! same up to now, so try errors (variance)
133if (this_var1 > this_var2) then
134 obssort = 1
135 return
136else if (this_var1 < this_var2) then
137 obssort = -1
138 return
139endif
140
141! ok, i give up. they're the same.
142! or enough for us to not try to resort them.
143obssort = 0
144
145if (local_debug) then
146 print *, 'decided items ', i, j, ' are same'
147 call print_time(this_time1, 'time: ')
148 print *, 'kinds ', this_kind1, this_kind2
149 print *, 'vars ', this_var1, this_var2
150 call write_location(0, this_loc1, charstring=locstring1)
151 call write_location(0, this_loc2, charstring=locstring2)
152 print *, 'locs: '
153 print *, trim(locstring1)
154 print *, trim(locstring2)
155 print *, ''
156endif
157
158end function obssort
159

551subroutine print_obs_seq(seq_in, filename)
552
553! you can get more info by running the obs_diag program, but this
554! prints out a quick table of obs types and counts, overall start and
555! stop times, and metadata strings and counts.
556
557type(obs_sequence_type), intent(in) :: seq_in
558character(len=*), intent(in) :: filename
559

565integer :: this_obs_type

567
568
569! Initialize input obs_types
570type_count(:) = 0
571identity_count = 0
572
573! make sure there are obs left to process before going on.
574! num_obs should be ok since we just constructed this seq so it should
575! have no unlinked obs. if it might for some reason, use this instead:
576! size_seq_in = get_num_key_range(seq_in) !current size of seq_in
577
578size_seq_in = get_num_obs(seq_in)
579if (size_seq_in == 0) then
580 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
581 call error_handler(e_msg,'obs_remove_dups',msgstring)
582 return
583endif
584
585! Initialize individual observation variables
586call init_obs( obs, get_num_copies(seq_in), get_num_qc(seq_in))
587call init_obs(next_obs, get_num_copies(seq_in), get_num_qc(seq_in))
588
589! blank line
590call error_handler(e_msg,'',' ')
591
592write(msgstring,*) 'Processing sequence file ', trim(filename)
593call error_handler(e_msg,'',msgstring)
594
595call print_metadata(seq_in, filename)
596
597!-------------------------------------------------------------
598! Start to process obs from seq_in
599!--------------------------------------------------------------
600is_there_one = get_first_obs(seq_in, obs)
601
602if ( .not. is_there_one ) then
603 write(msgstring,*)'no first observation in ',trim(filename)
604 call error_handler(e_msg,'obs_remove_dups', msgstring)
605endif
606
607! process it here
608is_this_last = .false.
609
610call get_obs_def(obs, this_obs_def)
611call print_time(get_obs_def_time(this_obs_def), ' First timestamp: ')
612! does not work with NO_CALENDAR
613if (cal) call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
614
615obsloop : do while ( .not. is_this_last)
616
617 call get_obs_def(obs, this_obs_def)
618 this_obs_type = get_obs_def_type_of_obs(this_obs_def)
619 if (this_obs_type < 0) then
620 identity_count = identity_count + 1
621 else
622 type_count(this_obs_type) = type_count(this_obs_type) + 1
623 endif
624! print *, 'obs type index = ', this_obs_type
625! if(this_obs_type > 0)print *, 'obs name = ', get_name_for_type_of_obs(this_obs_type)
626
627 call get_next_obs(seq_in, obs, next_obs, is_this_last)
628 if (.not. is_this_last) then
629 obs = next_obs
630 else
631 call print_time(get_obs_def_time(this_obs_def), ' Last timestamp: ')
632 if (cal) call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
633 endif
634
635enddo obsloop
636
637
638write(msgstring, *) 'Number of obs processed : ', size_seq_in
639call error_handler(e_msg, '', msgstring)
640write(msgstring, *) '---------------------------------------------------------'
641call error_handler(e_msg, '', msgstring)

667subroutine validate_obs_seq_time(seq, filename)
668
669! this eventually belongs in the obs_seq_mod code, but for now
670! try it out here. we just fixed a hole in the interactive create
671! routine which would silently let you create out-of-time-order
672! linked lists, which gave no errors but didn't assimilate the
673! right obs at the right time when running filter. this runs
674! through the times in the entire sequence, ensuring they are
675! monotonically increasing in time. this should help catch any
676! bad files which were created with older versions of code.
677
678type(obs_sequence_type), intent(in) :: seq
679character(len=*), intent(in) :: filename
680
681type(obs_type) :: obs, next_obs
682type(obs_def_type) :: this_obs_def
683logical :: is_there_one, is_this_last
684integer :: size_seq, obs_count
685integer :: key
686type(time_type) :: last_time, this_time
687
688
689! make sure there are obs left to process before going on.
690size_seq = get_num_obs(seq)
691if (size_seq == 0) then
692 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
693 call error_handler(e_msg,'obs_remove_dups:validate',msgstring)
694 return
695endif
696
697! Initialize individual observation variables
698call init_obs( obs, get_num_copies(seq), get_num_qc(seq))
699call init_obs(next_obs, get_num_copies(seq), get_num_qc(seq))
700
701obs_count = 0
702
703!-------------------------------------------------------------
704! Start to process obs from seq
705!--------------------------------------------------------------
706is_there_one = get_first_obs(seq, obs)
707
708! we already tested for 0 obs above, so there should be a first obs here.
709if ( .not. is_there_one ) then
710 write(msgstring,*)'no first obs in sequence ' // trim(filename)
711 call error_handler(e_err,'obs_remove_dups:validate', msgstring, source)
712 return
713endif
714
715is_this_last = .false.
716last_time = set_time(0, 0)
717obsloop : do while ( .not. is_this_last)
718
719 call get_obs_def(obs, this_obs_def)
720 this_time = get_obs_def_time(this_obs_def)
721
722 if (last_time > this_time) then
723 ! bad time order of observations in linked list
724 call print_time(last_time, ' previous timestamp: ')
725 if (cal) call print_date(last_time, ' calendar date: ')
726 call print_time(this_time, ' next timestamp: ')
727 if (cal) call print_date(this_time, ' calendar date: ')
728
729 key = get_obs_key(obs)
730 write(msgstring1,*)'obs number ', key, ' has earlier time than previous obs'
731 write(msgstring2,*)'observations must be in increasing time order, file ' // trim(filename)
732 call error_handler(e_err,'obs_remove_dups:validate', msgstring2, &
733 source, text2=msgstring1)
734 endif
735
736 last_time = this_time
737 obs_count = obs_count + 1
738
739 call get_next_obs(seq, obs, next_obs, is_this_last)
740 if (.not. is_this_last) obs = next_obs
741
742enddo obsloop
743
744! clean up
745call destroy_obs( obs)
746call destroy_obs(next_obs)
747
748! technically not a time validation, but easy to check. obs_count should never
749! be larger than size_seq - that's a fatal error. obs_count < size_seq would
750! suggest there are obs in the file that aren't part of the linked list.
751! this does not necessarily indicate a fatal error but it's not a common
752! situation and might indicate someone should check on the file.
753if (obs_count /= size_seq) then
754 write(msgstring,*) 'input sequence ', trim(filename)
755 call error_handler(e_msg,'obs_remove_dups:validate', msgstring)
756
757 write(msgstring,*) 'total obs in file: ', size_seq, ' obs in linked list: ', obs_count
758 if (obs_count > size_seq) then
759 ! this is a fatal error
760 write(msgstring1,*) 'linked list obs_count > total size_seq, should not happen'
761 call error_handler(e_err,'obs_remove_dups:validate', msgstring, &
762 source, text2=msgstring1)
763 else
764 ! just warning msg
765 write(msgstring1,*) 'only observations in linked list will be processed'
766 call error_handler(e_msg,'obs_remove_dups:validate', msgstring, &
767 source, text2=msgstring1)
768 endif
769endif
770
771end subroutine validate_obs_seq_time
