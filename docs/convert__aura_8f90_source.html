136 do k = 1, num_copies
137 call set_copy_meta_data(obs_seq, k, 'observation')

143end if

148
149filenum = 1

152
153 ! get the single name, or the next name from a list
154 if (from_list) then
155 next_infile = get_next_filename(aura_netcdf_filelist, filenum)
156 else
157 next_infile = aura_netcdf_file
158 if (filenum > 1) next_infile = ''
159 endif

162 ! open the file
163 call nc_check( nf90_open(next_infile, nf90_nowrite, ncid), 'file open', next_infile)

165 ! get the number of events and altitude:
166 call nc_check( nf90_inq_dimid(ncid,"levels",varid), 'inq dimid altitude')
167 call nc_check( nf90_inquire_dimension(ncid,varid,name,nalt), 'inq dim altitude')
168
169 call nc_check( nf90_inq_dimid(ncid,"times",varid), 'inq dimid event')
170 call nc_check( nf90_inquire_dimension(ncid,varid,name,nevent), 'inq dim event')
171 print *, 'nalt=',nalt
172 allocate( time(nevent) )
173 allocate( lat(nevent) )
174 allocate( lon(nevent) )
175 allocate( pres(nalt) )
176 allocate( temp(nalt,nevent) )
177 allocate( qual(nevent) )
178 allocate( conv(nevent) )
179 allocate( localsolartime(nevent) )
180
181 ! read in the latitude array
182 call nc_check( nf90_inq_varid(ncid, "Latitude", varid) ,'inq varid Lat')
183 call nc_check( nf90_get_var(ncid, varid, lat) ,'get var Lat')
184
185 ! read the longitude array
186 call nc_check( nf90_inq_varid(ncid, "Longitude", varid) ,'inq varid Lon')
187 call nc_check( nf90_get_var(ncid, varid, lon) ,'get var Lon')
188
189 ! read the altitude array
190 call nc_check( nf90_inq_varid(ncid, "Pressure", varid) ,'inq varid pressure')
191 call nc_check( nf90_get_var(ncid, varid, pres) ,'get_var pressure')
192
193 ! read the temperature
194 call nc_check( nf90_inq_varid(ncid, "Temperature", varid) ,'inq varid ktemp')
195 call nc_check( nf90_get_var(ncid, varid, temp) ,'get_var ktemp')
196
197 ! quality
198 call nc_check( nf90_inq_varid(ncid, "Quality", varid) ,'inq varid quality')
199 call nc_check( nf90_get_var(ncid, varid, qual) ,'get_var quality')
200 ! convergence
201 call nc_check( nf90_inq_varid(ncid, "Convergence", varid) ,'inq varid conv')
202 call nc_check( nf90_get_var(ncid, varid, conv) ,'get_var conv')
203 ! LST
204 call nc_check( nf90_inq_varid(ncid, "LocalSolarTime", varid) ,'inq varid lst')
205 call nc_check( nf90_get_var(ncid, varid, localsolartime) ,'get_var lst')
206
207 ! read the date/time
208 call nc_check( nf90_inq_varid(ncid, "Time", varid) ,'inq varid time')
209 call nc_check( nf90_get_var(ncid, varid, time) ,'get_var time')
210
211 ! finished reading the data
212 call nc_check( nf90_close(ncid), 'close file')
213
214 ! now loop through each event / altitude and add to obs sequence file
215 do ievent=1,nevent
216
217 do ialt=1,nalt
218
219 !calculate seconds from LST
220 isec = int( (localsolartime(ievent)-(lon(ievent)/15.))/24.*86400. )
221 if (isec.le.0) isec = isec + 86400
222 if (isec.ge.86400) isec = isec - 86400
223
224 iyear = aura_yr ! yr & day are inputs
225 idoy = aura_doy
226 call convert_day(iyear,idoy,imonth,iday)
227
228 if (temp(ialt,ievent).ne.-999. .and. & ! data is missing
229 qual(ievent).ge.0.65 .and. & !specified quality check
230 conv(ievent).le.1.2 .and. & ! specified convergence
231 pres(ialt).ge.0.001 .and. pres(ialt).le.260 .and. & !good altitude range
232 isec .lt.86400 ) then
233
234
235 ihr = int( isec / 3600 )
236 imin = int( (isec-ihr*3600) / 60 )
237 isec = int( (isec-ihr*3600)-imin*60 )
238 time_obs = set_date(iyear,imonth,iday,ihr,imin,isec)
239 call get_time(time_obs,osec,oday)
240
241 ! need to set the error (for now just take SABER error):
242 call saber_error(pres(ialt),oerr)
243
244 ! convert pressure from mbar(hPa) -> Pa
245 pres_out = pres(ialt)*100.
246
247 if (lon(ievent).le.0) lon(ievent) = lon(ievent)+360.
248
249 ! now creat the observation:
250 call create_3d_obs(lat(ievent),lon(ievent),pres_out, &
251 vertispressure,temp(ialt,ievent), &
252 auramls_temperature,oerr,oday,osec,qc,obs)
253 call add_obs_to_seq(obs_seq,obs,time_obs,prev_obs,prev_time,first_obs)
254
255 if(.not. did_obs) did_obs = .true.
256 else
257
258 end if
259 end do ! end alt loop
260 end do ! end event loop
261
262 ! clean up and loop if there is another input file
263 deallocate( lat,lon,pres,qual,conv,time,temp)
264
265 filenum = filenum + 1
266
267end do fileloop
268
269! done with main loop. if we added any obs to the sequence, write it out:
270if (did_obs) then
271!print *, 'ready to write, nobs = ', get_num_obs(obs_seq)
272 if (get_num_obs(obs_seq) > 0) &
273 call write_obs_seq(obs_seq, aura_outfile)
274
275 ! minor stab at cleanup, in the off chance this will someday get turned
276 ! into a subroutine in a module. probably not all that needs to be done,
277 ! but a start.
278 call destroy_obs(obs)
279! if obs == prev_obs then you can't delete the same obs twice.
280! buf if they differ, then it's a leak. for now, don't delete prev
281! since the program is exiting here anyway.
282 !call destroy_obs(prev_obs)
283 if (get_num_obs(obs_seq) > 0) call destroy_obs_sequence(obs_seq)
284endif

286! END OF MAIN ROUTINE

290subroutine convert_day(iyear,idoy,imonth,iday)
291! NMP - Subroutine to convert from yr,doy to yr,mon,day
292
293
294 integer, intent(in) :: iyear
295 integer, intent(in) :: idoy
296 integer, intent(out) :: imonth
297 integer, intent(out) :: iday
298
299 integer :: t
300
301 t = 0
302 if(modulo(iyear, 4) == 0) t = 1
303
304 if(modulo(iyear, 400) /= 0 .and. modulo(iyear, 100) == 0) t = 0
305
306 iday = idoy
307 if(idoy > 59+t) iday = iday + 2 - t
308 imonth = ((iday+91)*100)/3055
309 iday = (iday+91) - (imonth*3055)/100
310 imonth = imonth - 2
311
312 if(imonth >= 1 .and. imonth <= 12) return
313 write(unit=*,fmt="(a,i11,a)") &
314 "$$CALEND: DAY OF THE YEAR INPUT =",idoy," IS OUT OF RANGE."
315 stop
316
317end subroutine

320subroutine saber_error(pres,err)
321
322real(r8), intent(in) :: pres
323real(r8), intent(out) :: err
324
325! NMP - simple function to calculate the error based on height
326! function loosely based on Remsberg et al (2008JGR)
327
328err = 1.6718_r8 - 0.4281_r8*log(pres) + 0.0977_r8*log(pres)**2
329
