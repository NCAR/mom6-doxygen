35!------------------------------------------------------
36!> parallel read of netcdf restart file using pnetcdf
37subroutine read_ensemble_restart_parallel(ens_handle, file_name, start_copy, end_copy, giant_restart, transpose_giant, init_time)
38
39type(ensemble_type), intent(inout) :: ens_handle
40character(len = *), intent(in) :: file_name
41integer, intent(in) :: start_copy, end_copy
42logical, intent(in) :: giant_restart
43logical, intent(in) :: transpose_giant
44type(time_type), intent(in), optional :: init_time
45
46character(len = 4) :: extension !< for restart file number
47integer :: i !< for loops
48character(len=256) :: this_file_name !< giant restart filename
49real(r8), allocatable :: model_state(:,:) !< to hold transposed copies
50integer :: my_num_vars

52! Parallel netcdf variables
53integer :: ret !< return code for pnetcdf calls
54integer :: ncfile !< ncfile
55integer :: stateId !< Id for state_vector
56integer :: timeId !< Id for time
57integer :: stateDimId !< Id of the dimension of state
58integer :: timeDimId !< Id of the dimension of time
59integer :: varsDimId !< for giant read
60integer :: copiesDimId !< for giant read
61integer(KIND=MPI_OFFSET_KIND) :: start(1) ! state is one dimensional
62integer(KIND=MPI_OFFSET_KIND) :: count(1)
63integer(KIND=MPI_OFFSET_KIND) :: stride(1)
64integer(KIND=MPI_OFFSET_KIND) :: state_length, time_length
65integer(KIND=MPI_OFFSET_KIND) :: num_blocks
66integer(KIND=MPI_OFFSET_KIND) :: num_copies
67
68
69! timing variables
70double precision :: time_at_start
71
72num_blocks = ens_handle%my_num_vars ! Just me
73
74time_at_start = mpi_wtime()
75
76if (giant_restart) then
77
78 this_file_name = 'filter_ic_giant.nc'
79
80 ! Open netcdf file
81 ret = nfmpi_open(mpi_comm_world, this_file_name, nf_nowrite, mpi_info_null, ncfile)
82 call pnet_check(ret, 'read_ensemble_restart', 'open file')
83
84 ! get id for vars_dim - the length of the state
85 ret = nfmpi_inq_dimid(ncfile, 'vars', varsdimid)
86 call pnet_check(ret, 'read_ensemble_restart', 'cannot get vars dim')
87
88 ! get num vars
89 ret = nfmpi_inq_dimlen(ncfile, varsdimid, state_length)
90 call pnet_check(ret, 'read_ensemble_restart', 'get state_length')
91
92 ! get id for copies_dim - the number of copies
93 ret = nfmpi_inq_dimid(ncfile, 'copies', copiesdimid)
94 call pnet_check(ret, 'read_ensemble_restart', 'cannot get state dim')
95
96 ! get num_copies
97 ret = nfmpi_inq_dimlen(ncfile, copiesdimid, num_copies)
98 call pnet_check(ret, 'read_ensemble_restart', 'get num_copies')
99
100 ! get id for time_dim - the length of the time (should be 2)
101 ret = nfmpi_inq_dimid(ncfile, 'time', timedimid)
102 call pnet_check(ret, 'read_ensemble_restart', 'cannot get time dim')
103
104 ! Get status of varible
105 ret = nfmpi_inq_varid(ncfile, 'state_vector', stateid)
106 call pnet_check(ret, 'read_ensemble_restart', 'cannot get state_vector id')
107
108 ! Get status of varible
109 ret = nfmpi_inq_varid(ncfile, 'time', timeid)
110 call pnet_check(ret, 'read_ensemble_restart', 'cannot get time id')
111
112 if (transpose_giant) then
113
114 ! need to read transposed state
115 allocate(model_state(num_blocks, num_copies))
116
117 call aread_state_restart(state_length, stateid, timeid, num_blocks, num_copies, ens_handle%time, model_state, transpose_giant, ncfile)
118
119 ens_handle%copies(1:num_copies,:) = transpose(model_state)
120
121 deallocate(model_state)
122
123 else
124
125 call aread_state_restart(state_length, stateid, timeid, num_blocks, num_copies, ens_handle%time, ens_handle%copies(1:num_copies,:),transpose_giant, ncfile)
126 endif
127
128 ret = nfmpi_close(ncfile)
129 call pnet_check(ret, 'read_ensemble_restart', 'closing file')
130
131 if(present(init_time)) ens_handle%time(1:num_copies) = init_time
132
133 if (my_task_id() == 0) print*, 'giant read time ', mpi_wtime() - time_at_start, 'copies = ', end_copy - start_copy + 1
134else
135
136 ! Loop to read in all ensemble members
137 pnetcdf_restarts: do i = start_copy, end_copy
138
139 ! Get global index for my ith ensemble
140 write(extension, '(i4.4)') i
141 this_file_name = trim(file_name) // '.' // extension // '.nc'
142 ! Open netcdf file
143 ret = nfmpi_open(mpi_comm_world, this_file_name, nf_nowrite, mpi_info_null, ncfile)
144 call pnet_check(ret, 'read_ensemble_restart', 'open file')
145
146 ! get id for state_dim - the length of the state
147 ret = nfmpi_inq_dimid(ncfile, 'state', statedimid)
148 call pnet_check(ret, 'read_ensemble_restart', 'get state_dim')
149
150 ! get id for time_dim - the length of the time (should be 2)
151 ret = nfmpi_inq_dimid(ncfile, 'time', timedimid)
152 call pnet_check(ret, 'read_ensemble_restart', 'get time_dim')
153
154 ! get state length ( a dimension called state )
155 ret = nfmpi_inq_dimlen(ncfile, statedimid, state_length)
156 call pnet_check(ret, 'read_ensemble_restart', 'get state length')
157
158 ! get time length ( a dimension called time ) should be 2
159 ret = nfmpi_inq_dimlen(ncfile, timedimid, time_length)
160 call pnet_check(ret, 'read_ensemble_restart', 'get time length')
161
162 ! I think the model_mod gets the model size (state_length) separately
163
164 ! Get status of varible - Do you need to do this to get the variable id?
165 ret = nfmpi_inq_varid(ncfile, 'state_vector', stateid)
166 call pnet_check(ret, 'read_ensemble_restart', 'get state id')
167
168 ! Get status of varible - Do you need to do this to get the variable id?
169 ret = nfmpi_inq_varid(ncfile, 'time', timeid)
170 call pnet_check(ret, 'read_ensemble_restart', 'get time id')
171
172 ! Read the file directly into storage
173 my_num_vars = ens_handle%my_num_vars
174 call aread_state_restart(stateid, timeid, my_num_vars, ens_handle%time(i), ens_handle%copies(i, :), ncfile)
175 if(present(init_time)) ens_handle%time(i) = init_time
176
177 ! Close the restart file
178 ret = nfmpi_close(ncfile)
179 call pnet_check(ret, 'read_ensemble_restart', 'close restart file')
180
181 end do pnetcdf_restarts
182
183 if (my_task_id() == 0) print*, 'Read time : ', mpi_wtime() - time_at_start, 'average :', (mpi_wtime() - time_at_start) / (end_copy - start_copy + 1)
184
185endif

187end subroutine read_ensemble_restart_parallel

194character(len = *), intent(in) :: file_name

205! pnetcdf variables

209integer :: timeId !< Id for time
210integer :: stateDimId !< Id of the dimension of state

259 endif

261
262 ! Define the variables - state needs to go last so it can be unlimited size

264 ret = nfmpi_def_var(ncfile, "time", nf_int, num_dims, timedim, timeid) ! time
265 call pnet_check(ret, 'write_ensemble_restart', 'defining time var')

268 if (datasize == mpi_real4) then ! single precision state

276 call pnet_check(ret, 'write_ensemble_restart', 'defining state var')

289
290 deallocate(model_state)
291
292 else
293
294 call awrite_state_restart(timeid, stateid, num_blocks, num_copies, ens_handle%time(1), ens_handle%copies(1:num_copies,:), transpose_giant, ncfile)
295
296 endif
297
298 ret = nfmpi_close(ncfile)
299 call pnet_check(ret, 'write_ensemble_restart', 'closing file')
300
301 if (my_task_id() == 0) print*, 'giant write time ', mpi_wtime() - start_at_time, 'copies = ', end_copy - start_copy + 1
302
303else
304
305 pnetcdf_restarts: do i = start_copy, end_copy
306 write(extension, '(i4.4)') i
307 this_file_name = trim(file_name) // '.' // extension // '.nc'
308
309 ! create netcdf file
310 ret = nfmpi_create(mpi_comm_world, this_file_name, nf_clobber, mpi_info_null, ncfile)
311 call pnet_check(ret, 'write_ensemble_restart', 'cannot create netcdf file')
312
313 ! load up state and time in netcdf metadata
314 ! Define the dimensions
315 ret = nfmpi_def_dim(ncfile, 'time', time_length, timedimid) !time
316 call pnet_check(ret, 'write_ensemble_restart', 'defining time dim')
317
318 ret = nfmpi_def_dim(ncfile, 'state', state_length, statedimid) !state
319 call pnet_check(ret, 'write_ensemble_restart', 'defining state dim')
320
321 ! The dimids array is used to pass the IDs of the dimensions of the variables
322 statedim = (/statedimid/)
323 timedim = (/timedimid/)
324
325 ! Define the variables - state needs to go last because it is huge
326
327 num_dims = 1
328 ret = nfmpi_def_var(ncfile, "time", nf_int, num_dims, timedim, timeid) ! time
329 call pnet_check(ret, 'write_ensemble_restart', 'defining time var')
330
331 if (datasize == mpi_real4) then ! single precision state
332
333 ret = nfmpi_def_var(ncfile, "state_vector", nf_float, num_dims, statedim, stateid) ! state
334 call pnet_check(ret, 'write_ensemble_restart', 'defining state var')
335
336 else ! double precision
337
338 ret = nfmpi_def_var(ncfile, "state_vector", nf_double, num_dims, statedim, stateid) ! state
339 call pnet_check(ret, 'write_ensemble_restart', 'defining state var')
340
341 endif
342
343 ret = nfmpi_enddef(ncfile) ! metadata IO occurs in this
344 call pnet_check(ret, 'write_ensemble_restart', 'metadata IO')
345
346 call awrite_state_restart(timeid, stateid, num_blocks, ens_handle%time(i), ens_handle%copies(i, :), state_length,ncfile)
347
348 ret = nfmpi_close(ncfile)
349 call pnet_check(ret, 'write_ensemble_restart', 'closing file')
350
351 end do pnetcdf_restarts
352
353 if (my_task_id() == 0) print*, 'write time: ', mpi_wtime() - start_at_time, 'average :', (mpi_wtime() - start_at_time) / (end_copy - start_copy + 1)
354
355endif
