504subroutine get_expected_obs_from_def_distrib_state(state_handle, ens_size, copy_indices, key, obs_def, obs_kind_ind, &
505 state_time, isprior, assimilate_this_ob, evaluate_this_ob, expected_obs, istatus)
506
507! Compute forward operator for a particular obs_def
508type(ensemble_type), intent(in) :: state_handle
509integer, intent(in) :: ens_size
510integer, intent(in) :: copy_indices(ens_size)
511integer, intent(in) :: key
512type(obs_def_type), intent(in) :: obs_def
513integer, intent(in) :: obs_kind_ind
514type(time_type), intent(in) :: state_time
515logical, intent(in) :: isprior
516integer, intent(out) :: istatus(ens_size)
517logical, intent(out) :: assimilate_this_ob, evaluate_this_ob
518real(r8), intent(out) :: expected_obs(ens_size)
519
520
521type(location_type) :: location
522type(time_type) :: obs_time
523integer :: obs_key
524real(r8) :: error_var
525logical :: use_precomputed_fo
526integer :: copy
527
528character(len=512) :: string1, string2, string3
529
530! Load up the assimilate and evaluate status for this observation kind
531assimilate_this_ob = assimilate_this_type_of_obs(obs_kind_ind)
532evaluate_this_ob = evaluate_this_type_of_obs(obs_kind_ind)
533use_precomputed_fo = use_ext_prior_this_type_of_obs(obs_kind_ind)
534
535! If not being assimilated or evaluated return with missing_r8 and istatus 0
536if(assimilate_this_ob .or. evaluate_this_ob) then
537 ! for speed, access directly instead of using accessor functions
538 location = obs_def%location
539 obs_time = obs_def%time
540 obs_key = obs_def%key
541 error_var = obs_def%error_variance
542
543 ! the decision process here is that if some external source computed
544 ! the prior forward operator values then we can use them or not, but
545 ! if we use them there is no way to compute a consistent posterior.
546 ! so the posteriors are always marked as 'failed forward operator'.
547 if (use_precomputed_fo) then
548 if (obs_def%ens_size < ens_size) then
549 write(string1,'(A,1x,I4,1x,A,1x,I4)')'The number of precomputed forward operators (', &
550 obs_def%ens_size, ') is smaller than the ensemble size of ', ens_size
551 write(string2,*)'observation type '//trim(get_name_for_type_of_obs(obs_def%kind))
552 write(string3,*)'precomputed value(1) ', obs_def%external_FO(1)
553 call error_handler(e_err, 'get_expected_obs_from_def', string1, source, &
554 text2=string2, text3=string3)
555 endif
556
557 if (isprior) then
558 if ( obs_def%has_external_FO ) then
559 do copy = 1, ens_size
560 expected_obs(copy) = obs_def%external_FO(copy_indices(copy))
561 enddo
562 istatus = 0
563 else
564 call error_handler(e_err, 'get_expected_obs_from_def', &
565 'Attempt to access an external FO that is not present in the observation information.', &
566 source, revision, revdate, text2='observation type '//trim(get_name_for_type_of_obs(obs_def%kind)))
567 endif
568 else ! posterior - missing value
569 expected_obs(:) = missing_r8
570 istatus = 1
571 endif
572 else
573 ! Compute the forward operator. In spite of the variable name,
574 ! obs_kind_ind is in fact a 'type' index number. use the function
575 ! get_quantity_for_type_of_obs from the obs_kind_mod if you want to map
576 ! from a specific type to a generic kind. the third argument of
577 ! a call to the 'interpolate()' function must be a kind index and
578 ! not a type. normally the preprocess program does this for you.
579 select case(obs_kind_ind)
580
581 ! arguments available to an obs_def forward operator code are:
582 ! state_handle -- to access the state vector
583 ! ens_size -- the number of ensemble members to do at once (between 1 and ens_size)
584 ! copy_indices -- the indicies the ensemble members (between 1 and ens_size)
585 ! location -- observation location
586 ! obs_kind_ind -- the index of the observation specific type
587 ! obs_time -- the time of the observation
588 ! error_var -- the observation error variance
589 ! isprior -- true for prior eval; false for posterior
590 !
591 ! the routine must return values for:
592 ! expected_obs -- the computed forward operator values for all ensemble members
593 ! istatus -- return code: 0=ok, >0 is error, <0 reserved for system use
594 !
595 ! to call interpolate() directly, the arg list MUST BE:
596 ! interpolate(state_handle, ens_size, location, QTY_xxx, expected_obs, istatus)
597 !
598 ! the preprocess program generates lines like this automatically,
599 ! and this matches the interfaces in each model_mod.f90 file.
600 !
601 ! CASE statements and algorithms for specific observation kinds are
602 ! inserted here by the DART preprocess program.
603
604 ! DART PREPROCESS GET_EXPECTED_OBS_FROM_DEF INSERTED HERE
605
606 ! If the observation kind is not available, it is an error. The DART
607 ! preprocess program should provide code for all available kinds.
608 case DEFAULT
609 call error_handler(e_err, 'get_expected_obs_from_def', &
610 'Attempt to evaluate or assimilate undefined obs_kind type.', &
611 source, revision, revdate)
612 end select
613 endif
614else
615 ! Not computing forward operator for this kind
616 expected_obs(:) = missing_r8
617 istatus = 0
618endif
619
620end subroutine get_expected_obs_from_def_distrib_state

624 subroutine read_obs_def(ifile, obs_def, key, obs_val, fform)
625!----------------------------------------------------------------------------
626! subroutine read_obs_def(ifile, obs_def, key, obs_val, fform)
627!
628! ifile
629! obs_def
630! key
631! obs_val needed if you want to perform operations based on value
632! fform
633!
634! Reads an obs_def from file which is just an integer unit number in the
635! current preliminary implementation.
636
637integer, intent(in) :: ifile
638type(obs_def_type), intent(inout) :: obs_def
639integer, intent(in) :: key
640real(r8), intent(inout) :: obs_val
641character(len=*), optional, intent(in) :: fform
642
643character(len=5) :: header
644integer :: o_index
645logical :: is_ascii
646character(len=32) :: fileformat ! here for backwards compatibility only
647character(len=512) :: errstring
648character(len=11) :: header_external_fo
649integer :: ii, secs,days
650character(len=128) :: string
651logical :: time_set
652integer, save :: counter = 0
653
654if ( .not. module_initialized ) call initialize_module
655
656is_ascii = ascii_file_format(fform)
657
658! here for backwards compatibility only; after the next release,
659! remove this and force people in their own obs_def_xxx_mod.f90 code
660! to set the last arg to read/write to be fform instead of fileformat
661if (is_ascii) then
662 fileformat = 'formatted'
663else
664 fileformat = 'unformatted'
665endif
666
667! Begin by reading five character ascii header, then location, kind, error variance, index
668
669! Need to add additional error checks on read
670if (is_ascii) then
671 read(ifile, '(a5)') header
672 if(header /= 'obdef') then
673 write(errstring, *) 'read "//header//" instead'
674 call error_handler(e_err,'read_obs_def', &
675 'Expected header "obdef" in input file', &
676 source, revision, revdate, text2=errstring)
677 endif
678endif
679
680! Read the location, kind, time and error variance
681obs_def%location = read_location(ifile, fform)
682if (is_ascii) then
683 read(ifile, '(a5)' ) header
684 if(header /= 'kind ') then
685 write(errstring, *) 'read "//header//" instead'
686 call error_handler(e_err,'read_kind', &
687 'Expected kind header "kind " in input file', &
688 source, revision, revdate, text2=errstring)
689 endif
690 read(ifile, *) o_index
691else
692 read(ifile) o_index
693endif
694
695! Negative value is identity obs, doesn't need mapped
696! Positive value must use mapping to get to proper index in obs_kind
697if(o_index < 0) then
698 obs_def%kind = o_index
699else
700 obs_def%kind = map_type_of_obs_table(o_index)
701endif
702
703! This kind may have its own module that needs to read more

757 obs_def%has_external_FO = .false.
