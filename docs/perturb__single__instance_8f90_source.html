52!----------------------------------------------------------------
53! These variables are namelist-controllable.
54!
55integer :: ens_size = 1
56character(len=256) :: input_files(max_files) = ''
57character(len=256) :: output_file_list(max_num_doms) = ''
58character(len=256) :: output_files(max_files) = ''
59real(r8) :: perturbation_amplitude = 0.0
60logical :: single_restart_file_in = .false.
61
62namelist /perturb_single_instance_nml/ &
63 ens_size, &
64 input_files, &
65 output_files, &
66 output_file_list, &
67 perturbation_amplitude, &
68 single_restart_file_in
69
70!----------------------------------------------------------------
71! Additional global variables
72!
73type(ensemble_type) :: ens_handle
74character(len=256), allocatable :: file_array_input(:,:)
75character(len=256), allocatable :: file_array_output(:,:)
76character(len=512) :: msgstring, msgstring1
77character(len=256) :: my_base, my_desc
78integer :: idom, imem, iunit, io, i
79integer :: ndomains
80logical :: interf_provided
81integer(i8) :: model_size
82type(time_type) :: member_time
83type(file_info_type) :: file_info_input, file_info_output
84type(stage_metadata_type) :: input_restart_files
85type(stage_metadata_type) :: output_restart_files
86
87!----------------------------------------------------------------
88! program start
89!----------------------------------------------------------------
90
91call initialize_mpi_utilities('perturb_single_instance')
92
93
94! Read the namelist entry and print it
95call find_namelist_in_file("input.nml", "perturb_single_instance_nml", iunit)
96read(iunit, nml = perturb_single_instance_nml, iostat = io)
97call check_namelist_read(iunit, io, "perturb_single_instance_nml")
98
99if (do_nml_file()) write(nmlfileunit, nml=perturb_single_instance_nml)
100if (do_nml_term()) write( * , nml=perturb_single_instance_nml)
101
102if (single_restart_file_in) then
103 write(msgstring, *) 'single_restart_file_in is not supported.'
104 write(msgstring1, *) 'Please contact DART if you would like to use this capability.'
105 call error_handler(e_err,msgstring,msgstring1)
106endif
107
108!----------------------------------------------------------------------
109! Calling static_init_assim_model() is required, which also calls
110! static_init_model(), so there is no need to explicitly call it.
111!----------------------------------------------------------------------
112
113call static_init_assim_model()
114
115!----------------------------------------------------------------------
116! initialization code, model size
117!----------------------------------------------------------------------
118
119model_size = get_model_size()
120
121!----------------------------------------------------------------------
122! Make space that is ensemble handle
123!----------------------------------------------------------------------
124call init_ensemble_manager(ens_handle, ens_size, model_size)
125
126!----------------------------------------------------------------------
127! Allocate space for file arrays.
128! Contains a matrix of files (ncopies x ndomains)
129! If perturbing from a single instance the number of
130! input files does not have to be ens_size but rather
131! a single file (or multiple files if more than one domain)
132!----------------------------------------------------------------------
133
134ndomains = get_num_domains()
135
136!----------------------------------------------------------------------
137! can be ens_size but rather a single file
138! (or multiple files if more than one domain)
139!----------------------------------------------------------------------
140allocate(file_array_input(ens_size, ndomains))
141
142file_array_input = reshape(input_files, (/1, ndomains/))
143
144!----------------------------------------------------------------------
145! read in a single ensemble member
146!----------------------------------------------------------------------
147call io_filenames_init(file_info_input, &
148 ncopies = 1, &
149 cycling = single_restart_file_in, &
150 single_file = single_restart_file_in, &
151 restart_files = file_array_input)
152
153!----------------------------------------------------------------------
154! Read the template file to get the shape of netCDF file
155! and its variables. It is possible to have multiple domains
156! but only require one member.
157!----------------------------------------------------------------------
158write(my_base,'(A)') 'template'
159write(my_desc,'(A)') 'template file'
160call set_file_metadata(file_info_input, &
161 cnum = 1, &
162 fnames = file_array_input(1,:), &
163 basename = my_base, &
164 desc = my_desc)
165
166call set_io_copy_flag(file_info_input, &
167 cnum = 1, &
168 io_flag = read_copy)
169
170input_restart_files = get_stage_metadata(file_info_input)
171
172imem = 1
173do idom = 1, ndomains
174 write(msgstring1, *) '- Reading File : ', &
175 trim(get_restart_filename(input_restart_files, &
176 copy = imem, &
177 domain = idom))
178 call error_handler(e_msg, 'perturb_single_instance: ', msgstring1, source)
179enddo
180
181!----------------------------------------------------------------------
182! Read the ensemble from files
183!----------------------------------------------------------------------
184member_time = set_time_missing()
185call read_state(ens_handle, file_info_input, read_time_from_file=.true., model_time=member_time)
186
187!----------------------------------------------------------------------
188! Copy from ensemble member 1 to the other copies
189!----------------------------------------------------------------------
190do i = 1, get_my_num_vars(ens_handle)
191 ens_handle%copies(2:ens_size, i) = ens_handle%copies(1, i)
192enddo
193
194call pert_model_copies(ens_handle, ens_size, perturbation_amplitude, interf_provided)
195if (.not. interf_provided) then
196 call error_handler(e_err, 'model_mod::pert_model_copies interface required', source)
197endif
198
199
200!----------------------------------------------------------------------
201! can be ens_size but rather a single file
202! (or multiple files if more than one domain)
203!----------------------------------------------------------------------
204allocate(file_array_output(ens_size, ndomains))
205
206!----------------------------------------------------------------------
207! Given either a vector of in/output_files or a text file containing
208! a list of files, return a vector of files containing the filenames.
209!----------------------------------------------------------------------
210call set_multiple_filename_lists(output_files(:), &
211 output_file_list(:), &
212 ndomains, &
213 ens_size, &
214 'perturb_single_instance', &
215 'output_files', &
216 'output_file_list')
217
218file_array_output = reshape(output_files, (/ens_size, ndomains/))
219
220!----------------------------------------------------------------------
221! output ens_size perturbed restarts
222!----------------------------------------------------------------------
223call io_filenames_init(file_info_output, &
224 ncopies = ens_size, &
225 cycling = single_restart_file_in, &
226 single_file = single_restart_file_in, &
227 restart_files = file_array_output)
228
229do imem = 1, ens_size
230 write(my_base,'(A,I0.2)') 'output_', imem
231 write(my_desc,'(A,I0.2)') 'output ensemble member ', imem
232 call set_file_metadata(file_info_output, &
233 cnum = imem, &
234 fnames = file_array_output(imem,:), &
235 basename = my_base, &
236 desc = my_desc)
237
238 call set_io_copy_flag(file_info_output, &
239 cnum = imem, &
240 io_flag = write_copy)
241enddo
242
243output_restart_files = get_stage_metadata(file_info_output)
244do imem = 1, ens_size
245 do idom = 1, ndomains
246 write(msgstring1, *) '- Writing File : ', imem, idom, &
247 trim(get_restart_filename(output_restart_files, &
248 copy = imem, &
249 domain = idom))
250 call error_handler(e_msg, 'perturb_single_instance: ', msgstring1, source)
251 enddo
252enddo
253
254call write_state(ens_handle, file_info_output)
255
256!----------------------------------------------------------------------
257! clean up allocated memory
258!----------------------------------------------------------------------
259call end_ensemble_manager(ens_handle)
260deallocate(file_array_output, file_array_input)
261
262call finalize_mpi_utilities()
263
264!----------------------------------------------------------------
265!----------------------------------------------------------------
266!----------------------------------------------------------------
267
268end program perturb_single_instance
