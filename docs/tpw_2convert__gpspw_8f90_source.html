7program convert_gpspw
8
9! convert ground-based GPS observations of Total Precipitable Water
10! into DART observation sequence files, that is archived in the
11! SuomiNet website (http://www.suominet.ucar.edu/data/index.html).
12! THIS CODE ASSUMES ALL ELEVATIONS ARE AT 0.0 meters.
13! this is true for all ocean obs. if you have land obs, you will
14! have to get the hght of the surface elevation at the lat/lon and
15! use that instead.
16!
17! assumes two input text files:
18! the first one is the data which match the following read line:
19! read(iunit, '(A4, 1x, i4, 2i2, 1x, 2i2, 5x, f7.1)' &
20! stnid, iyear, imonth, iday, ihour, imin
21!
22! the second one is a list of station locations.
23! read(junit, '(2f11.6,f8.2,'(A4)') &
24! lon, lat, hgt, stnid
25!
26! constructs a input filename based on:
27! ObsBase/InfilePrefix + YYYYMMDD + InfileSuffix
28!
29! constructs a output filename based on:
30! ./OutfilePrefix + YYYYMMDD + OutfileSuffix
31!
32! any of the prefix or suffixes can be '' (blank)
33!
34

143call check_namelist_read(iunit, io, 'convert_gpspw_nml')

145! Record the namelist values used for the run ...
146if (do_nml_file()) write(nmlfileunit, nml=convert_gpspw_nml)
147if (do_nml_term()) write( * , nml=convert_gpspw_nml)

149! some simple error checking - if you specify either of the longitude
150! limits, since it's a cyclic value you have to specify both. the included
151! area will start at min_lon and go east until max_lon. for latitude
152! you can let one default if you only want to set a min or a max.
153if (((min_lon /= missing_r8) .and. (max_lon == missing_r8)) .or. &
154 ((min_lon == missing_r8) .and. (max_lon /= missing_r8))) then
155 write(msgstring, *) 'if you set the min or max limit on longitude you must specify both'
156 call error_handler(e_err,'convert_gpspw', msgstring, source, revision, revdate)
157endif

159! convert a string into an observation type number
160obstype = get_index_for_type_of_obs(observation_name)
161if (obstype < 0) then
162 write(msgstring, *) 'unrecognized observation type ', trim(observation_name)
163 call error_handler(e_err,'convert_gpspw', msgstring, source, revision, revdate, &
164 text2='check the obs_defs in your &preprocess list')
165endif

167call set_calendar_type(gregorian)

173! FIXME: Assume all the input files have the same year (for now).

175write(*,*) 'idate = ',idate

178

191 write(obsdate, '(i4.4, 2i2.2)') start_year, start_month, start_day + idd
192 write(outfile, '(A)') trim(outfileprefix)//trim(obsdate)//trim(outfilesuffix)
193 write(*, '(A)') 'creating file: '//trim(outfile)
194
195 nrec = 0

197 ! read in all observation records

199
200 ! construct the input and output filenames
201 write(indate, '(i4.4, A1, i3.3, A1, i2.2, A3)') start_year, '.', idate + idd, '.', ihr, '.00'
202 write(infile, '(A)') trim(obsbase)//'/'//trim(infileprefix)//trim(indate)//trim(infilesuffix)
203 write(*, '(A)') 'reading file: '//trim(infile)
204
205 iunit = open_file(infile,'formatted',action='read')
206 read(iunit,*)
207 read(iunit,*)
208
209 readloop: do
210
211 read(iunit, '(A4, 1x, i4, 2i2, 1x, 2i2, 5x, f7.1)', iostat=io) &
212 stnid, iyear, imonth, iday, ihour, imin, tpw
213 if (io /= 0) exit readloop
214
215 rewind(junit)
216 stnloop: do
217 read(junit, '(f11.6,f10.6,f8.2,1x,A4)', iostat=io) lon, lat, hght, locid
218 if(locid.eq.stnid) exit stnloop
219 if(io /= 0) exit stnloop
220 enddo stnloop
221 if(locid.ne.stnid) then
222 write(msgstring, *) 'Could not find the station: ',stnid,' vs. ',locid
223 call error_handler(e_err,'convert_gpspw', msgstring, source, revision, revdate)
224 endif
225
226 nrec = nrec + 1
227
228 if(nrec > max_obs) then
229 write(msgstring, *) 'number of input observations larger than "max_obs" limit'
230 call error_handler(e_err,'convert_gpspw', msgstring, source, revision, revdate, &
231 text2='increase parameter "max_obs" in namelist and rerun')
232 endif
233
234 ! adjust longitude ranges to [0,360]
235 if (lon < 360.0_r8) lon = lon + 360.0_r8
236
237 ! optionally select a range of lat/lon.
238 ! now that we have the average lat and lon, test them against the limits.
239 ! we do this test last so the location of the average lat/lon is guarenteed
240 ! to be within the bounds; possibly individual obs being averaged with this
241 ! one may be slightly out of the range depending on the box size.
242 if (min_lat /= missing_r8) then
243 if (lat .lt. min_lat) cycle readloop
244 endif
245 if (max_lat /= missing_r8) then
246 if (lat .gt. max_lat) cycle readloop
247 endif
248 if ((min_lon /= missing_r8) .and. (max_lon /= missing_r8)) then
249 if (.not. is_longitude_between(lon, min_lon, max_lon)) cycle readloop
250 endif
251
252 qc = 0.0_r8 ! quality control - all good.
253 !hght = 0.0_r8 ! This only works over the ocean; over land need elevation
254 oerr = 1.5 ! observation error in cm
255 obs_time = set_date(iyear, imonth, iday, ihour, imin, 0)
256 call get_time(obs_time, osec, oday)
257
258 call create_3d_obs(lat, lon, hght, vertissurface, tpw, &
259 obstype, oerr, oday, osec, qc, obs)
260 call add_obs_to_seq(obs_seq, obs, obs_time, prev_obs, prev_time, first_obs)
261 num_obs = num_obs + 1
262
263 enddo readloop
264
265 write(msgstring, '(A,I12)') 'number of observations in input file is ', nrec
266 call error_handler(e_msg,'', msgstring, source, revision, revdate)
267
268 call close_file(iunit)
269
270 enddo hourloop
271
272
273 if (num_obs > 0) then
274 write(msgstring, '(A,I12)') 'number of observations in output file is ', num_obs
275 call error_handler(e_msg,'', msgstring, source, revision, revdate)
276 call write_obs_seq(obs_seq, outfile)
277 else
278 call error_handler(e_msg,'', 'not creating output file because 0 observations found', &
279 source, revision, revdate)
280 endif
281
282 call destroy_obs(obs)
283
284enddo dayloop
285
286call finalize_utilities()
287
288end program
