469SUBROUTINE insert_gts_obs_gpsref(gtsobs, qc_threshold, &
470 gpsref_type, gpsref_key, gpsref_form, isgps, &
471 which_vert, num_obs, obs, seq)
472!-------------------------------------------------------------------------------
473
474 implicit none
475
476 type(multi_level_type), intent(in) :: gtsobs
477 integer, optional, intent(in) :: qc_threshold
478 integer, intent(in) :: gpsref_type
479 integer, intent(in) :: which_vert, gpsref_form
480 logical, intent(in) :: isgps
481 integer, intent(inout) :: num_obs, gpsref_key
482 type(obs_type), intent(inout) :: obs
483 type(obs_sequence_type), intent(inout) :: seq
484
485 type(obs_def_type) :: obs_def
486 type(location_type) :: location
487 type(time_type) :: time
488
489 character(len=80) :: dummy
490 integer :: year, month, day, hours, minutes, seconds
491 real(r8) :: lat,lon,pressure, height
492
493 character(len=6) :: subset
494 integer :: i, levels
495 real(r8) :: td, nx, ny, nz, ds, htop, rfict
496 real(r8) :: obs_value(1), obs_err, obs_qc(1)
497
498 logical :: valid
499
500
501 ! time info
502 read(gtsobs%info%date_char, '(I4,5(A1,I2))') &
503 year, dummy, month, dummy, day, dummy, hours, dummy, minutes, dummy, seconds
504 time = set_date(year, month, day, hours, minutes, seconds)
505 call set_obs_def_time(obs_def, time)
506
507 ! number of vertical levels
508 levels = gtsobs%info%levels
509
510!------------------
511 do i = 1, levels
512!------------------
513
514 gpsref_key = gpsref_key+1
515
516 !gpsref data stored in dew_point
517
518 valid = .true.
519 if (present(qc_threshold)) then
520 if (qc_threshold .ge. 0 ) &
521 valid = (gtsobs%each(i)%dew_point%qc .ge. 0) &
522 .and. (gtsobs%each(i)%dew_point%qc .le. qc_threshold)
523 endif
524 if (valid) then
525 ! location info
526 lat = gtsobs%each(i)%speed%data
527 lon = gtsobs%each(i)%direction%data
528 ! Dart longitude from 0 to 360
529 if(lon < 0.0_r8) lon = lon + 360.0_r8
530 pressure = gtsobs%each(i)%pressure%data
531 height = gtsobs%each(i)%height%data
532 if (which_vert.eq.vertissurface .or. which_vert.eq.vertisheight) then
533 location = set_location(lon, lat, height, which_vert)
534 else if (which_vert.eq.vertispressure) then
535 location = set_location(lon, lat, pressure, which_vert)
536 end if
537 call set_obs_def_location(obs_def, location)
538
539 if ( gpsref_form .eq. 1 ) then
540 nx = 0.0
541 ny = 0.0
542 nz = 0.0
543 ds = 0.0
544 htop = 0.0
545 rfict = 0.0
546 subset = 'GPSREF'
547 else
548 subset = 'GPSEXC'
549 print*,'Not implemented yet! Return!'
550 endif
551
552 call set_gpsro_ref(gpsref_key, nx, ny, nz, rfict, ds, htop, subset)

594FUNCTION compute_vapor_mixing_ratio(t_k, p_pa, rh_pct, rh_liq_flag) RESULT(wvmr)
595!--------------------------------------------------------------------------------
596
597 ! Computes the water vapor mixing ratio (dimensionless) given
598 ! temperature (K), pressure (Pa), relative humidity (%), and a flag
599 ! that tells the function whether or not the RH being provided is
600 ! with respect to liquid everywhere or with respect to liquid only
601 ! at temperature > Tref_frz. This formulation comes from a paper by
602 ! Buck (1991)
603
604 IMPLICIT NONE
605 REAL(r8) :: t_k
606 REAL(r8) :: p_pa
607 REAL(r8) :: rh_pct
608 LOGICAL :: rh_liq_flag
609 REAL(r8) :: wvmr
610
611 REAL(r8) :: e,es
612 REAL(r8), parameter :: tref_frz = 273.15
613
614 ! Compute saturation vapor pressure using an approx.
615 ! developed by Buck (1981)
616
617 IF ((rh_liq_flag).OR.((.NOT.rh_liq_flag).AND.(t_k.GE.tref_frz)))THEN
618
619 ! Compute saturation vapor pressure wrt liquid
620
621 es = 100.0 * 6.1121 * exp( (17.502*t_k-4780.6713)/(t_k-32.18) )
622
623 ELSE
624
625 ! Compute es wrt ice
626 ! Factor of 100 converts mb to Pa
627 es = 100.0 * 6.1115 * exp( (22.452*t_k-6132.7638)/(t_k-0.60) )
628
629 ENDIF
630
631 ! From McKee (1993 CSU AT650 class notes) the values above
632 ! need to be multiplied by 1.004 at pressures greater than 800 mb
633
634 IF (p_pa.GE.80000.) es = es * 1.004
635
636 ! Compute e from RH
637
638 e = rh_pct*.01 * es
639
640 ! Compute wvmr from e and p_pa
641
642 wvmr = 0.622 * e / (p_pa - e )
643END FUNCTION compute_vapor_mixing_ratio
