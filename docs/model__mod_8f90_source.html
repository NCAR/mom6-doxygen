142!------------------------------------------------------------------
143!
144! Called to do one time initialization of the model. As examples,
145! might define information about the model size or model timestep.
146! In models that require pre-computed static data, for instance
147! spherical harmonic weights, these would also be computed here.
148
149subroutine static_init_model()
150
151integer :: iunit, io
152character(len=vtablenamelength) :: variable_table(max_state_variables, num_state_table_columns)
153
154integer :: state_qty_list(max_state_variables)
155logical :: update_var_list(max_state_variables)
156
157! identifiers for variable_table
158integer, parameter :: var_name_index = 1
159integer, parameter :: var_qty_index = 2
160integer, parameter :: var_update_index = 3
161
162module_initialized = .true.
163
164call find_namelist_in_file("input.nml", "model_nml", iunit)
165read(iunit, nml = model_nml, iostat = io)
166call check_namelist_read(iunit, io, "model_nml")
167
168! Record the namelist values used for the run
169if (do_nml_file()) write(nmlfileunit, nml=model_nml)
170if (do_nml_term()) write( * , nml=model_nml)
171
172call set_calendar_type('gregorian')
173
174! This time is both the minimum time you can ask the model to advance
175! (for models that can be advanced by filter) and it sets the assimilation
176! window. All observations within +/- 1/2 this interval from the current
177! model time will be assimilated. If this is not settable at runtime
178! feel free to hardcode it and remove from the namelist.
179assimilation_time_step = set_time(assimilation_period_seconds, &
180 assimilation_period_days)
181
182! verify that the model_state_variables namelist was filled in correctly.
183! returns variable_table which has variable names, kinds and update strings.
184call verify_state_variables(model_state_variables, nfields, variable_table, state_qty_list, update_var_list)
185
186! Define which variables are in the model state
187dom_id = add_domain(template_file, nfields, &
188 var_names = variable_table(1:nfields, var_name_index), &
189 kind_list = state_qty_list(1:nfields), &
190 update_list = update_var_list(1:nfields))
191
192model_size = get_domain_size(dom_id)
193
194call read_horizontal_grid()
195call setup_interpolation()
196
197call read_num_layers()
198call read_ocean_geometry() ! ocean vs. land and basin depth
199
200end subroutine static_init_model

215!------------------------------------------------------------------
216! Given a state handle, a location, and a state quantity,
217! interpolates the state variable fields to that location and returns
218! the values in expected_obs. The istatus variables should be returned as
219! 0 unless there is some problem in computing the interpolation in
220! which case a positive istatus should be returned.
221
222subroutine model_interpolate(state_handle, ens_size, location, qty, expected_obs, istatus)
223
224
225type(ensemble_type), intent(in) :: state_handle
226integer, intent(in) :: ens_size
227type(location_type), intent(in) :: location
228integer, intent(in) :: qty
229real(r8), intent(out) :: expected_obs(ens_size) !< array of interpolated values
230integer, intent(out) :: istatus(ens_size)
231
232integer :: which_vert, four_ilons(4), four_ilats(4), lev(ens_size,2)
233integer :: locate_status, quad_status
234real(r8) :: lev_fract(ens_size)
235real(r8) :: lon_lat_vert(3)
236real(r8) :: quad_vals(4, ens_size)
237real(r8) :: expected(ens_size, 2) ! level below and above obs
238type(quad_interp_handle) :: interp
239integer :: varid, i, e, thick_id
240integer(i8) :: th_indx, indx(ens_size)
241real(r8) :: depth_at_x(ens_size), thick_at_x(ens_size) ! depth, layer thickness at obs lat lon
242logical :: found(ens_size)
243
244if ( .not. module_initialized ) call static_init_model
245
246expected_obs(:) = missing_r8
247istatus(:) = 1
248
249varid = get_varid_from_kind(dom_id, qty)
250if (varid < 0) then ! not in state
251 istatus = not_in_state
252 return
253endif
254
255thick_id = get_varid_from_kind(dom_id, qty_layer_thickness)
256if (thick_id < 0) then ! thickness not in state
257 istatus = thickness_not_in_state
258 return ! HK else use pseudo depth?
259endif
260
261! find which grid the qty is on
262interp = get_interp_handle(qty)
263
264! unpack the location type into lon, lat, vert, vert_type
265lon_lat_vert = get_location(location)
266which_vert = nint(query_location(location))
267
268! get the indices for the 4 corners of the quad in the horizontal
269call quad_lon_lat_locate(interp, lon_lat_vert(1), lon_lat_vert(2), &
270 four_ilons, four_ilats, locate_status)
271if (locate_status /= 0) then
272 istatus(:) = quad_locate_failed
273 return
274endif
275
276! check if all four corners are in the ocean
277if (on_land(four_ilons, four_ilats)) then
278 istatus(:) = quad_on_land
279 return
280endif
281
282! find which layer the observation is in. Layer thickness is a state variable.
283! HK @todo Do you need to use t_grid interp for thickess four_ilons, four_ilats?
284found(:) = .false.
285depth_at_x(:) = 0
286find_layer: do i = 2, nz
287
288 ! corner1
289 th_indx = get_dart_vector_index(four_ilons(1), four_ilats(1), i, dom_id, thick_id)
290 quad_vals(1, :) = get_state(th_indx, state_handle)
291
292 ! corner2
293 th_indx = get_dart_vector_index(four_ilons(1), four_ilats(2), i, dom_id, thick_id)
294 quad_vals(2, :) = get_state(th_indx, state_handle)
295
296 ! corner3
297 th_indx = get_dart_vector_index(four_ilons(2), four_ilats(1), i, dom_id, thick_id)
298 quad_vals(3, :) = get_state(th_indx, state_handle)
299
300 ! corner4
301 th_indx = get_dart_vector_index(four_ilons(2), four_ilats(2), i, dom_id, thick_id)
302 quad_vals(4, :) = get_state(th_indx, state_handle)
303
304 call quad_lon_lat_evaluate(interp, &
305 lon_lat_vert(1), lon_lat_vert(2), & ! lon, lat of obs
306 four_ilons, four_ilats, &
307 ens_size, &
308 quad_vals, & ! 4 corners x ens_size
309 thick_at_x, &
310 quad_status)
311 if (quad_status /= 0) then
312 istatus(:) = thickness_quad_evalutate_failed
313 return
314 endif
315
316 depth_at_x = depth_at_x + thick_at_x
317
318 do e = 1, ens_size
319 if (lon_lat_vert(3) < depth_at_x(e)) then
320 lev(e,1) = i ! layer_below
321 lev(e,2) = i-1 ! layer_above
322 lev_fract(e) = (depth_at_x(e) - lon_lat_vert(3)) / thick_at_x(e)
323 found(e) = .true.
324 if (all(found)) exit find_layer
325 endif
326 enddo
327
328enddo find_layer
329
330if (any(found .eqv. .false.)) then
331 istatus(:) = obs_too_deep
332 return
333endif
334
335if (on_basin_edge(four_ilons, four_ilats, ens_size, depth_at_x)) then
336 istatus(:) = quad_on_basin_edge
337 return
338endif
339
340do i = 1, 2
341 !HK which corner of the quad is which?
342 ! corner1
343 do e = 1, ens_size
344 indx(e) = get_dart_vector_index(four_ilons(1), four_ilats(1), lev(e, i), dom_id, varid)
345 enddo
346 call get_state_array(quad_vals(1, :), indx, state_handle)
347
348 ! corner2
349 do e = 1, ens_size
350 indx(e) = get_dart_vector_index(four_ilons(1), four_ilats(2), lev(e, i), dom_id, varid)
351 enddo
352 call get_state_array(quad_vals(2, :), indx, state_handle)
353
354 ! corner3
355 do e = 1, ens_size
356 indx(e) = get_dart_vector_index(four_ilons(2), four_ilats(1), lev(e, i), dom_id, varid)
357 enddo
358 call get_state_array(quad_vals(3, :), indx, state_handle)
359
360 ! corner4
361 do e = 1, ens_size
362 indx(e) = get_dart_vector_index(four_ilons(2), four_ilats(2), lev(e, i), dom_id, varid)
363 enddo
364 call get_state_array(quad_vals(4, :), indx, state_handle)
365
366 call quad_lon_lat_evaluate(interp, &
367 lon_lat_vert(1), lon_lat_vert(2), & ! lon, lat of obs
368 four_ilons, four_ilats, &
369 ens_size, &
370 quad_vals, & ! 4 corners x ens_size
371 expected(:,i), &
372 quad_status)
373 if (quad_status /= 0) then
374 istatus(:) = quad_evalutate_failed
375 return
376 else
377 istatus = 0
378 endif
379
380enddo
381
382! Interpolate between levels
383! expected_obs = bot_val + lev_fract * (top_val - bot_val)
384expected_obs = expected(:,1) + lev_fract(:) * (expected(:,2) - expected(:,1))
385
386if (qty == qty_salinity) then ! convert from PSU (model) to MSU (obersvation)
387 expected_obs = expected_obs/1000.0_r8
388endif
389
390
391end subroutine model_interpolate

442subroutine convert_vertical_state(state_handle, num, locs, loc_qtys, loc_indx, &
443which_vert, istatus)
444
445type(ensemble_type), intent(in) :: state_handle
446integer, intent(in) :: num
447type(location_type), intent(inout) :: locs(num) !locations
448integer, intent(in) :: loc_qtys(num) !qty at location
449integer(i8), intent(in) :: loc_indx(num) !state index
450integer, intent(in) :: which_vert
451integer, intent(out) :: istatus
452
453integer :: i,j,k
454integer :: ii, layer ! loop variables
455integer :: thick_id
456integer(i8) :: indx
457real(r8) :: depth(1)
458
459! assert(which_vert == VERTISHEIGHT)
460
461thick_id = get_varid_from_kind(dom_id, qty_layer_thickness)
462if (thick_id < 0) then
463 istatus = thickness_not_in_state
464 return
465endif
466
467do ii = 1, num
468
469 if (loc_qtys(ii) == qty_dry_land) then
470 call set_vertical(locs(ii), 0.0_r8, vertisheight)
471 else
472
473 call get_model_variable_indices(loc_indx(ii), i, j, k)
474
475 depth = 0.0_r8
476 do layer = 1, k
477 indx = get_dart_vector_index(i, j, layer, dom_id, thick_id)
478 depth = depth + get_state(indx, state_handle)
479 enddo
480
481 call set_vertical(locs(ii), depth(1), vertisheight)
482
483 endif
484
485enddo
486
487istatus = 0
488
489end subroutine convert_vertical_state

516subroutine get_close_state(gc, base_loc, base_type, locs, loc_qtys, loc_indx, &
517 num_close, close_ind, dist, ens_handle)
518
519type(get_close_type), intent(in) :: gc ! handle to a get_close structure
520type(location_type), intent(inout) :: base_loc ! location of interest
521integer, intent(in) :: base_type ! observation TYPE
522type(location_type), intent(inout) :: locs(:) ! state locations
523integer, intent(in) :: loc_qtys(:) ! QTYs for state
524integer(i8), intent(in) :: loc_indx(:) ! indices into DART state vector
525integer, intent(out) :: num_close ! how many are close
526integer, intent(out) :: close_ind(:) ! indices into the locs array
527real(r8), optional, intent(out) :: dist(:) ! distances in radians
528type(ensemble_type), optional, intent(in) :: ens_handle
529
530character(len=*), parameter :: routine = 'get_close_state'
531
532integer :: ii ! loop index
533integer :: i, j, k
534real(r8) :: lon_lat_vert(3)
535
536call loc_get_close_state(gc, base_loc, base_type, locs, loc_qtys, loc_indx, &
537 num_close, close_ind, dist, ens_handle)
538
539if (.not. present(dist)) return
540
541! Put any land or sea floor points very far away
542! so they are not updated by assimilation
543do ii = 1, num_close
544
545 if(loc_qtys(close_ind(ii)) == qty_dry_land) dist = 1.0e9_r8
546
547 lon_lat_vert = get_location(locs(close_ind(ii))) ! assuming VERTISHEIGHT
548 call get_model_variable_indices(loc_indx(ii), i, j, k)
549 if ( below_sea_floor(i,j,lon_lat_vert(3)) ) dist = 1.0e9_r8
550
551enddo

553end subroutine get_close_state

592!------------------------------------------------------------
593! Read lon, lat for T,U,V grids from mom6 static file
594subroutine read_horizontal_grid()
595
596integer :: ncid
597integer :: nxy(2) ! (nx,ny)
598
599character(len=*), parameter :: routine = 'read_horizontal_grid'
600
601ncid = nc_open_file_readonly(static_file)
602
603call nc_get_variable_size(ncid, 'geolon', nxy)
604nx = nxy(1)
605ny = nxy(2)
606allocate(geolon(nx,ny), geolat(nx,ny)) ! T grid
607allocate(geolon_u(nx,ny), geolat_u(nx,ny)) ! U grid
608allocate(geolon_v(nx,ny), geolat_v(nx,ny)) ! V grid
609
610call nc_get_variable(ncid, 'geolon', geolon, routine)
611call nc_get_variable(ncid, 'geolon_u', geolon_u, routine)
612call nc_get_variable(ncid, 'geolon_v', geolon_v, routine)
613
614
615! mom6 example file has longitude > 360
616! DART uses [0,360]
617where(geolon > 360.0_r8 ) geolon = geolon - 360.0_r8
618where(geolon_u > 360.0_r8 ) geolon_u = geolon_u - 360.0_r8
619where(geolon_v > 360.0_r8 ) geolon_v = geolon_v - 360.0_r8
620

873!------------------------------------------------------------------
874! Verify that the namelist was filled in correctly, and check
875! that there are valid entries for the dart_kind.
876! Returns a table with columns:
877!
878! netcdf_variable_name ; dart_qty_string ; update_string
879
880subroutine verify_state_variables(state_variables, ngood, table, qty_list, update_var)
881
882character(len=*), intent(inout) :: state_variables(:)
883integer, intent(out) :: ngood
884character(len=*), intent(out) :: table(:,:)
885integer, intent(out) :: qty_list(:) ! kind number
886logical, intent(out) :: update_var(:) ! logical update
887
888integer :: nrows, i
889character(len=NF90_MAX_NAME) :: varname, dartstr, update
890character(len=256) :: string1, string2
891
892if ( .not. module_initialized ) call static_init_model
893
894nrows = size(table,1)
895
896ngood = 0
897
898myloop : do i = 1, nrows
899
900 varname = trim(state_variables(3*i -2))
901 dartstr = trim(state_variables(3*i -1))
902 update = trim(state_variables(3*i ))
903
904 call to_upper(update)
905
906 table(i,1) = trim(varname)
907 table(i,2) = trim(dartstr)
908 table(i,3) = trim(update)
909
910 if ( table(i,1) == ' ' .and. table(i,2) == ' ' .and. table(i,3) == ' ') exit myloop ! Found end of list.
911
912 if ( table(i,1) == ' ' .or. table(i,2) == ' ' .or. table(i,3) == ' ' ) then
913 string1 = 'model_nml:model_state_variables not fully specified'
914 call error_handler(e_err,'verify_state_variables',string1)
915 endif
916
917 ! Make sure DART qty is valid
918
919 qty_list(i) = get_index_for_quantity(dartstr)
920 if( qty_list(i) < 0 ) then
921 write(string1,'(''there is no obs_kind <'',a,''> in obs_kind_mod.f90'')') trim(dartstr)
922 call error_handler(e_err,'verify_state_variables',string1)
923 endif
924
925 ! Make sure the update variable has a valid name
926
927 select case (update)
928 case ('UPDATE')
929 update_var(i) = .true.
930 case ('NO_COPY_BACK')
931 update_var(i) = .false.
932 case default
933 write(string1,'(A)') 'only UPDATE or NO_COPY_BACK supported in model_state_variable namelist'
934 write(string2,'(6A)') 'you provided : ', trim(varname), ', ', trim(dartstr), ', ', trim(update)
935 call error_handler(e_err,'verify_state_variables',string1, text2=string2)
936 end select
937
938 ngood = ngood + 1
939enddo myloop
940
941
942end subroutine verify_state_variables
