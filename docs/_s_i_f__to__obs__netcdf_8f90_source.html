7!-------------------------------------------------------------------------------
8!
9! SIF_to_obs_netcdf - reads monthly 'harmonized' SIF
10! observations and writes a DART obs_seq file.
11!
12! SIF observation description: Harmonized long term SIF data product(SIF005)
13! which combines GOME-2 and SCIAMACHY SIF retrievals, combined with MODIS
14! data to produce single, continuous product (global,monthly, 0.05 degree,
15! 2002-2018; Wen et al., 2020, RSE). Downloaded from:
16! https://climatesciences.jpl.nasa.gov/sif/download-data/level-3

178 ! on number of obs. increase the size if too small.
179 num_new_obs=nlon*nlat
180
181 call init_obs_sequence(obs_seq, num_copies, num_qc, num_new_obs)

202 call print_date(time_obs, str=string1)
203 call print_time(time_obs, str=string2)
204 endif
205
206 latloop: do j=1,nlat
207
208 ! ensure the lat values are in range
209 rlat = lat(j)
210 if ( rlat > 90.0_r8 .or. rlat < -90.0_r8 ) cycle latloop
211
212 lonloop: do i=1,nlon
213
214 ! ensure the lat values are in range
215 rlon = lon(i)
216 if ( rlon > 360.0_r8 .or. rlon < 0.0_r8 ) cycle lonloop
217
218 ! Skip anything marked as such
219 if ( has_fillvalue .and. sif(i,j) == fillvalue ) cycle lonloop
220
221 ! reject obviously bad observations
222 if ( sif(i,j) < 0.0_r8 ) cycle lonloop
223
224 ! reject observations with questionable QC values
225 qc = decipherevi(evi_quality(i,j),i,j)
226 if ( qc >= 50 ) cycle lonloop
227
228 ! Update the metadata array holding the wavelengths.
229 ! Return the position in that array.
230 key = set_sif_wavelength(wavelength)
231
232 ! make an obs derived type, and then add it to the sequence
233 ! VERTISUNDEF declares that the entire column will have the same
234 ! localization distance - only dependent on the horizontal distance.
235 ! VERTISSURFACE is also horizontal distance only.
236
237 call create_3d_obs(lat(j), lon(i), 0.0_r8, vertisundef, sif(i,j), &
238 harmonized_sif, sif_err(i,j), oday, osec, qc, evi_quality(i,j), key, obs)
239
240 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
241
242 enddo lonloop
243 enddo latloop
244
245 ! if we added any obs to the sequence, write it out to a file now.
246 if ( get_num_obs(obs_seq) > 0 ) then
247 if (verbose > 0) then
248 write(string1,*)'"'//trim(output_name)//'" has ', &
249 get_num_obs(obs_seq), ' observations.'
250 call error_handler(e_msg, source, string1)
251 endif
252 call write_obs_seq(obs_seq, output_name)
253 endif
254
255 deallocate(sif)
256 deallocate(sif_err)
257 deallocate(evi_quality)
258 deallocate(lon)
259 deallocate(lat)
260
261 call destroy_obs_sequence(obs_seq)
262
263enddo fileloop
264
265! end of main program
266call finalize_utilities()

273! Read a file containing a list of filenames and checks that they exist
274
275function check_input_files(input_list, output_list)
276
277character(len=*), intent(in) :: input_list
278character(len=*), dimension(:), intent(out) :: output_list
279integer :: check_input_files
280
281character(len=256) :: input_line
282character(len=256) :: ladjusted
283integer, parameter :: maxlines = 500
284integer :: iline
285
286iunit = open_file(trim(input_list), 'formatted', 'read')
287
288check_input_files = 0
289filenameloop: do iline = 1,maxlines ! a lot of lines
290
291 ! read in entire text line into a buffer
292 read(iunit, "(A)", iostat=iocode) input_line
293 if (iocode > 0) then
294 write(string1,*) 'While reading ', trim(input_list)
295 write(string2,*) 'got read code (iostat) = ', iocode,' around line ',iline
296 call error_handler(e_err, 'Check_Input_Files', string1, &
297 source,text2=string2)
298 elseif (iocode < 0) then
299 ! Normal end of file
300 exit filenameloop
301 else
302 check_input_files = check_input_files + 1
303
304 ladjusted = adjustl(input_line)
305 if ( file_exist(trim(ladjusted)) ) then
306 output_list(check_input_files) = trim(ladjusted)
307 else
308 write(string1,*)'file does not exist.'
309 call error_handler(e_err,'Check_Input_Files',&
310 string1,source,text2=trim(ladjusted))
311 endif
312 endif
313
314enddo filenameloop
315
316if (check_input_files >= maxlines-1 ) then
317 write(string1,*)'Too many files to process. Increase MAXLINES and try again.'
318 call error_handler(e_err,'Check_Input_Files',string1,source)
319endif
320
321call close_file(iunit)
322
323end function check_input_files

405!-------------------------------------------------------------------------------
406! Decipher the binary-coded-integer into a DART-compatible error value.
407! The DART-compatible error scheme is based on NCEP-like error codes.
408! 0.0 = best quality
409! 1.0 = lesser
410! ... etc.
411! By not rejecting the observations at this step, the 'input_qc_threshold'
412! namelist value can be used to test whether or not 'suspect' observations
413! improve the result or not.
414
415function decipherevi(evi,ilon,jlat) result (qc)
416
417integer, intent(in) :: evi
418integer, intent(in) :: ilon,jlat
419real(r8) :: qc
420
421character(len=64) :: binary_string
422character(len=6) :: bits543210
423integer :: ndigits
424
425! QC Interpretation
426! https://cornell.app.box.com/s/gkp4moy4grvqsus1q5oz7u5lc30i7o41/file/687233625170
427! https://lpdaac.usgs.gov/documents/103/MOD13_User_Guide_V6.pdf
428! From Table 5:
429! Bits 0-1
430! 00 produced with good quality
431! 01 produced, but check other QA
432! 10 produced, but most likely cloudy
433! 11 not produced, due to other reasons than clouds
434! Bits 2-5
435! 0000 Highest quality
436! 0001 Lower quality
437! 0010 Decreasing quality
438! 0100 Decreasing quality
439! 1000 Decreasing quality
440! 1001 Decreasing quality
441! 1010 Decreasing quality
442! 1100 Lowest quality
443! 1101 Quality so low that it is not useful
444! 1110 L1B data faulty
445! 1111 Not useful for any other reason/not processed
446
447! Determine if least significant bits are right or left
448! write(*, '(''8 is '',B4)')8
449! write(*, '(''7 is '',B4)')7
450! write(*, '(''6 is '',B4)')6
451! write(*, '(''5 is '',B4)')5
452! write(*, '(''4 is '',B4)')4
453! write(*, '(''3 is '',B4)')3
454! write(*, '(''2 is '',B4)')2
455! write(*, '(''1 is '',B4)')1
456! write(*, '(''0 is '',B4)')0
457
458! write(*,*)
459! write(*,'(''1234567890123456'')')
460! write(*, '(I0,1x,B0)') evi,evi
461
462write(binary_string, '(B0)') evi
463
464ndigits = len_trim(binary_string)
465
466! Grab the least significant six digits (as a character string)
467bits543210 = binary_string((ndigits-5):ndigits)
468
469! This interpretation is ENTIRELY SUBJECTIVE ...
470
471select case (bits543210)
472 !543210 ! bits 2-5 | bits 0-1
473 case ('000000') ! 0000 ... Highest Quality | 00 V1 produced with qood quality
474 qc = 0.0_r8
475 case ('000100') ! 0001 ... Lower Quality | 00 V1 produced with good quality
476 qc = 1.0_r8
477 case ('001000') ! 0010 ... Lower Quality | 00 ditto
478 qc = 2.0_r8
479 case ('010000') ! 0100 ... Lower Quality | 00 ditto
480 qc = 3.0_r8
481 case ('100000') ! 1000 ... Lower Quality | 00 ditto
482 qc = 4.0_r8
483 case ('100100') ! 1001 ... Lower Quality | 00 ditto
484 qc = 5.0_r8
485 case ('101000') ! 1010 ... Lower Quality | 00 ditto
486 qc = 6.0_r8
487 case ('110000') ! 1100 ... Lowest Quality | 00 ditto
488 qc = 7.0_r8
489
490
491
492 case ('000001') ! 0000 ... Highest Quality | 01 V1 produced, but check with other QA
493 qc =10.0_r8
494 case ('000101') ! 0001 ... Lower Quality | 01 V1 produced, but check with other QA
495 qc =11.0_r8
496 case ('001001') ! 0010 ... Lower Quality | 01 ditto
497 qc =12.0_r8
498 case ('010001') ! 0100 ... Lower Quality | 01 ditto
499 qc =13.0_r8
500 case ('100001') ! 1000 ... Lower Quality | 01 ditto
501 qc =14.0_r8
502 case ('100101') ! 1001 ... Lower Quality | 01 ditto
503 qc =15.0_r8
504 case ('101001') ! 1010 ... Lower Quality | 01 ditto
505 qc =16.0_r8
506 case ('110001') ! 1100 ... Lowest Quality | 01 ditto
507 qc =17.0_r8
508
509 case default
510 qc = 50.0_r8 ! do not use
511end select
512
513! Large negative values will be thrown out
514
515if (evi < 0) qc=100
516
517if (verbose > 1) &
518 write(*,'(''ilon,jlat,EVI,... '',3(I7,1x),A,F10.2)')ilon, jlat, evi, bits543210, qc
519
520end function decipherevi
