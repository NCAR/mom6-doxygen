192 call get_2dshort_as_r8(ncid, 'soil_moisture_x', soil_moisture_x, smx_fill)

194 call get_2dshort_as_r8(ncid, 'sm_x_error', sm_x_error, smxe_fill)
195 call get_2dshort_as_r8(ncid, 'opt_depth_c', opt_depth_c, odc_fill)
196 call get_2dshort_as_r8(ncid, 'opt_depth_x', opt_depth_x, odx_fill)
197 call nc_close_file(ncid)

199 ! time is determined from the file name
200
201 obs_time = get_time_from_name(next_infile)
202
203 call get_time(obs_time, seconds, day)
204
205 ! ensure longitudes are [0,360]
206 where(longitude < 0.0_r8) longitude = longitude + 360.0_r8
207
208 scanloop: do j = 1, nscan
209 pixloop: do i = 1, npix
210
211 if (rfi_code(i,j) > max_rfi_code ) cycle pixloop
212
213 lon = longitude(i,j)
214 lat = latitude( i,j)
215 qc = rfi_code( i,j)
216
217 !------------------------------------------------------------------
218 ! Create C-band observation
219 ! observation error is instrument plus representativeness
220
221 oerr = sm_c_error(i,j) + amsre_rep_error ! units of standard deviation
222 vertvalue = 0.005_r8 ! 5mm depth for C band
223
224 if ((soil_moisture_c(i,j) /= smc_fill) .and. &
225 (sm_c_error(i,j) /= smce_fill) ) then
226 call create_3d_obs(lat, lon, vertvalue, vertisheight, &

230 ! first one, insert with no prev. otherwise, since all times are the
231 ! same for this column, insert with the prev obs as the starting point.
232 ! (the first insert with no prev means it will search for the right
233 ! time ordered starting point.)
234 if (first_obs) then
235 call insert_obs_in_seq(obs_seq, obs)
236 first_obs = .false.
237 else
238 call insert_obs_in_seq(obs_seq, obs, prev_obs)
239 endif
240 obs_num = obs_num+1
241 prev_obs = obs
242
243 if (obs_num > num_new_obs) then
244 write(string1,*)'reached maximum number of observations while'
245 write(string2,*)'reading file # ',filenum,' "'//trim(next_infile)//'"'
246 call error_handler(e_err, 'AMSR_E_L2_to_obs', string1, &
247 source, text2=string2)
248 endif
249
250 endif
251
252 !------------------------------------------------------------------
253 ! Create X-band observation
254 ! observation error is instrument plus representativeness
255
256 oerr = sm_x_error(i,j) + amsre_rep_error
257 vertvalue = 0.005_r8 ! 5mm depth for X band
258
259 if ((soil_moisture_x(i,j) /= smx_fill) .and. &
260 (sm_x_error(i,j) /= smxe_fill) ) then
261 call create_3d_obs(lat, lon, vertvalue, vertisheight, &
262 soil_moisture_x(i,j), x_observation_kind, &
263 oerr, day, seconds, qc, obs)
264
265 ! first one, insert with no prev. otherwise, since all times are the
266 ! same for this column, insert with the prev obs as the starting point.
267 ! (the first insert with no prev means it will search for the right
268 ! time ordered starting point.)
269 if (first_obs) then
270 call insert_obs_in_seq(obs_seq, obs)
271 first_obs = .false.
272 else
273 call insert_obs_in_seq(obs_seq, obs, prev_obs)
274 endif
275 obs_num = obs_num+1
276 prev_obs = obs
277
278 if (obs_num > num_new_obs) then
279 write(string1,*)'reached maximum number of observations while'
280 write(string2,*)'reading file # ',filenum,' "'//trim(next_infile)//'"'
281 call error_handler(e_err, 'AMSR_E_L2_to_obs', string1, &
282 source, text2=string2)
283 endif
284
285 endif
286
287 enddo pixloop
288 enddo scanloop
289
290 filenum = filenum + 1
291
292 deallocate(latitude)
293 deallocate(longitude)
294 deallocate(rfi_code)
295 deallocate(soil_moisture_c)
296 deallocate(soil_moisture_x)
297 deallocate(sm_c_error)
298 deallocate(sm_x_error)
299 deallocate(opt_depth_c)
300 deallocate(opt_depth_x)
301
302end do fileloop
303
304! done with main loop. if we added any obs to the sequence, write it out.
305if (obs_num > 0) then
306 print *, 'ready to write, nobs = ', get_num_obs(obs_seq)
307 if (get_num_obs(obs_seq) > 0) &
308 call write_obs_seq(obs_seq, output_file)
309
310 call destroy_obs(obs)
311 if (get_num_obs(obs_seq) > 0) call destroy_obs_sequence(obs_seq)
312else
313 call error_handler(e_err,'AMSR_E_L2_to_obs','no obs converted', &
314 source)
315endif
316
317call error_handler(e_msg,'AMSR_E_L2_to_obs','Finished successfully.')
318call finalize_utilities()

323!-----------------------------------------------------------------------
324!> determine the observation kind from the filename. There are separate
325!> kinds for ascending vs. descending orbits.
326!>
327!> LPRM-AMSR_E_L2_A_SOILM2_V002_20030702114630.nc
328!> LPRM-AMSR_E_L2_D_SOILM2_V002_20030702114630.nc

330function asc_desc(fname,band)
331character(len=*), intent(in) :: fname
332character(len=*), intent(in) :: band
333integer :: asc_desc
334
335character(len=64) :: mystring
336
337if (index(fname,'L2_A_') > 0 ) then
338 mystring = 'AMSRE_A_SOIL_MOISTURE_'//trim(band)
339elseif (index(fname,'L2_D_') > 0 ) then
340 mystring = 'AMSRE_D_SOIL_MOISTURE_'//trim(band)
341else
342 call error_handler(e_err,'asc_desc','cannot find asc/desc character', &
343 source, text2=fname)
344endif
345
346SELECT CASE (mystring)
347 CASE ('AMSRE_A_SOIL_MOISTURE_C')
348 asc_desc = amsre_a_soil_moisture_c
349 CASE ('AMSRE_A_SOIL_MOISTURE_X')
350 asc_desc = amsre_a_soil_moisture_x
351 CASE ('AMSRE_D_SOIL_MOISTURE_C')
352 asc_desc = amsre_d_soil_moisture_c
353 CASE ('AMSRE_D_SOIL_MOISTURE_X')
354 asc_desc = amsre_d_soil_moisture_x
355 CASE DEFAULT
356 call error_handler(e_err,'asc_desc','bad string "'//trim(mystring)//'', &
357 source)
358END SELECT
359
360end function asc_desc

363!-----------------------------------------------------------------------
364!> determine the observation kind from the filename. There are separate
365!> kinds for ascending vs. descending orbits.
366!>
367!> LPRM-AMSR_E_L2_D_SOILM2_V002_20030702114630.nc
368!> YYYYMMDDHHmmss

370function get_time_from_name(fname)
371character(len=*), intent(in) :: fname
372type(time_type) :: get_time_from_name
373
374integer :: indx, indx1, indx2
375integer :: year, month, day, hours, minutes, seconds
376
377indx = index(fname,'_V002_')
378
379if (indx < 1) then
380 call error_handler(e_err,'get_time_from_name','cannot determine where to start', &
381 source, text2=fname)
382endif
383
384indx1 = indx + 6 ! need to know where '_V002_' ends
385indx2 = indx1 + 13
386
387read(fname(indx1:indx2),'(i4,5(i2))')year,month,day,hours,minutes,seconds
388
389get_time_from_name = set_date(year,month,day,hours,minutes,seconds)
390
391end function get_time_from_name
