5!> Generate initial inflation files from namelist values.
6!> This way an experiment can always start from a restart file
7!> without having to alter the namelist between cycles 1 and 2.
8!>
9!> an alternative to running this program is to use the nco utilities thus:
10!>
11!> Here is an example using version 4.4.2 or later of the NCO tools:
12!> ncap2 -s "T=1.0;U=1.0;V=1.0" wrfinput_d01 prior_inflation_mean.nc
13!> ncap2 -s "T=0.6;U=0.6;V=0.6" wrfinput_d01 prior_inflation_sd.nc'

181! if users have variables in the dart state which are set to 'no update'
182! for assimilation purposes, they still have to have corresponding fields
183! in the inflation files. (science question is what those values should be.)
184!
185! the normal write routine skips 'no update' fields, so the output of
186! this program would be missing those fields and cause an error when
187! filter tries to read the inflation files. the following code
188! overwrites the update flag values so all variables are always written.
189!
190! also tell users to script their workflows to copy the input inflation
191! file to the output name before running filter, so there is an existing
192! output inflation file for filter to update. it will already include the
193! 'no update' fields, whose inflation values will remain unchanged.
194! this prevents confusing errors at the next execution of filter.
195! [ed note: tim is brilliant.]
196
197do idom = 1, ndomains
198 ivars = get_num_variables(idom)
199 call set_update_list(idom, ivars, (/ ( .true., imem=1,ivars ) /) )

206 single_file = single_file)

210endif
211
212if(write_post_inf) then
213 call fill_inflation_files(post_inf_mean, post_inf_sd, 'post')
214endif
215

224!> fill inflation values in a separate file
225
226subroutine fill_inflation_files(inf_mean, inf_sd, stage)
227real(r8), intent(in) :: inf_mean
228real(r8), intent(in) :: inf_sd
229character(len=*), intent(in) :: stage
230
231if (inf_mean == missing_r8 .or. inf_sd == missing_r8) then
232 write(string1,*) 'you must specify both inflation mean and inflation standard deviation values'
233 write(string2,*) 'you have "',trim(stage),'_inf_mean = ', inf_mean,'" and '
234 write(string3,*) ' "',trim(stage),'_inf_sd = ', inf_sd, '" '
235 call error_handler(e_msg, 'fill_inflation_restart: ', string1, &
236 source, text2=string2, text3=string3)
237 return
238endif
239ens_handle%copies(ss_inflate_index , :) = inf_mean
240ens_handle%copies(ss_inflate_sd_index, :) = inf_sd
241
242write(my_stage,'(3A)') 'input_', stage, 'inf'
243write(my_base, '(A)') 'mean'
244write(my_desc, '(2A)') stage, ' inflation mean'
245call set_file_metadata(file_info_output, &
246 cnum = ss_inflate_index, &
247 stage = my_stage, &
248 basename = my_base, &
249 desc = my_desc)
250
251call set_io_copy_flag(file_info_output, &
252 cnum = 1, &
253 io_flag = write_copy)
254
255write(my_base, '(A)') 'sd'
256write(my_desc, '(2A)') stage, ' inflation sd'
257call set_file_metadata(file_info_output, &
258 cnum = ss_inflate_sd_index, &
259 stage = my_stage, &
260 basename = my_base, &
261 desc = my_desc)
262
263call set_io_copy_flag(file_info_output, &
264 cnum = 2, &
265 io_flag = write_copy)
266
267output_restart_files = get_stage_metadata(file_info_output)
268
269do idom = 1, ndomains
270 ! write inflation mean
271 write(string1, *) '- Writing ',trim(stage), ' Mean Inflation File : "', &
272 trim(get_restart_filename(output_restart_files, &
273 copy = ss_inflate_index, &
274 domain = idom)), '"'
275 write(string2, *) ' with value = ', inf_mean
276 call error_handler(e_msg, 'fill_inflation_restart: ', string1, source, text2=string2)
277
278 ! write inflation stadard deviation
279 write(string1, *) '- Writing ',trim(stage), ' SD Inflation File : "', &
280 trim(get_restart_filename(output_restart_files, &
281 copy = ss_inflate_sd_index, &
282 domain = idom)), '"'
283 write(string2, *) ' with value = ', inf_sd
284 call error_handler(e_msg, 'fill_inflation_restart: ', string1, source, text2=string2)
285enddo
286
287
288call write_state(ens_handle, file_info_output)
289
290end subroutine fill_inflation_files
