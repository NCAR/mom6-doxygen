57subroutine move_ahead(ens_handle, ens_size, seq, last_key_used, window_time, &
58 key_bounds, num_obs_in_set, curr_ens_time, next_ens_time)
59
60! Based on the current ens time and the time of the next available
61! observation, compute whether and how far the ensemble copies of the
62! state need to be advanced. This returns the number of obs in the
63! next assimilation window, and both the current and expected next data
64! times. It NO LONGER advances the model - the calling code must call
65! advance_state() itself in order to do that if next time is /= curr time.
66
67implicit none
68
69type(ensemble_type), intent(inout) :: ens_handle
70integer, intent(in) :: ens_size
71type(obs_sequence_type), intent(in) :: seq
72integer, intent(in) :: last_key_used
73type(time_type), intent(in) :: window_time
74integer, intent(out) :: key_bounds(2), num_obs_in_set
75type(time_type), intent(inout) :: curr_ens_time
76type(time_type), intent(out) :: next_ens_time
77
78type(time_type) :: next_time, time2, start_time, end_time, delta_time, ens_time
79type(obs_type) :: observation
80type(obs_def_type) :: obs_def
81logical :: is_this_last, is_there_one, out_of_range, leaving_early
82
83! Initialize if needed
84if(.not. module_initialized) then
85 call initialize_module
86 module_initialized = .true.
87endif
88
89! Prepare values for 'early' returns
90num_obs_in_set = 0
91key_bounds(1:2) = -99
92leaving_early = .false.
93!curr_ens_time = set_time(0,0)
94next_ens_time = set_time(0,0)
95
96
97! If none of my copies are regular ensemble members no need to advance.
98! This is true either if I have no copies or if the copy #s I have are
99! all larger than the ensemble count. This does assume that ensemble
100! copies are in the first ens_size global copies of ensemble and that
101! global indices are monotonically increasing in each pes local copies
102! (note: Violating the private boundaries on ens_handle by direct access)
103! This was the original code:
104! if((ens_handle%my_num_copies < 1) .or. (ens_handle%my_copies(1) > ens_size)) then
105! but if the compiler tried to evaluate both halves at the same time,
106! (which it's allowed to do), it got an access violation if num copies was 0.
107
108! if this task has no ensembles to advance, return directly from here.
109! the calling code will sort out syncing up in the parallel case.
110if (.not. have_members(ens_handle, ens_size)) return
111
112
113! if you get here, this task has at least one ensemble to try to advance;
114! it is possible we are at the end of the observations and there in fact
115! is no need to advance. if so, can return.
116
117! ens_handle%my_pe 0 does the output.
118! Don't want two pes outputing if task 0 also has a copy
119! FIMXE Commment
120if ( map_task_to_pe(ens_handle, 0) >= ens_handle%num_copies .and. &
121 ens_handle%my_pe == 0 .and. my_task_id() /= 0) then
122 call set_output(.true.)
123endif
124
125
126! Initialize a temporary observation type to use
127! after here, must delete observation before returning.
128call init_obs(observation, get_num_copies(seq), get_num_qc(seq))
129
130! Get the next observation in the sequence that hasn't already been assimilated
131if(last_key_used > 0) then
132 call get_next_obs_from_key(seq, last_key_used, observation, is_this_last)
133 ! If the end of the observation sequence has been passed, return
134 if(is_this_last) leaving_early = .true.
135else
136 is_there_one = get_first_obs(seq, observation)
137 if(.not. is_there_one) leaving_early = .true.
138endif
139
140if (leaving_early) then
141 ! need to destroy obs here before returning
142 call destroy_obs(observation)
143
144 if (ens_handle%my_pe == 0 .and. my_task_id() /= 0) then
145 call set_output(.false.)
146 endif
147
148 return
149endif
150
151! Get the time of this observation
152call get_obs_def(observation, obs_def)
153next_time = get_obs_def_time(obs_def)
154
155! Get the time of the ensemble, assume consistent across all ensemble copies
156call get_ensemble_time(ens_handle, 1, ens_time)
157
158! Compute the model time step and center a window around the closest time
159delta_time = get_model_time_step()
160
161! print out current window, if requested
162if (print_trace_details > 0) then
163 if(delta_time / 2 > ens_time) then
164 start_time = set_time(0, 0)
165 else
166 start_time = ens_time - delta_time / 2 + set_time(1, 0)
167 endif
168 end_time = ens_time + delta_time / 2
169 call timechat(ens_time, 'move_ahead', .false., 'Current model data time is: ')
170 call timechat(start_time, 'move_ahead', .false., 'Current assimilation window starts at: ')
171 call timechat(next_time, 'move_ahead', .false., 'Next available observation time is: ')
172 call timechat(end_time, 'move_ahead', .false., 'Current assimilation window ends at: ')
173 !call timechat(delta_time, 'move_ahead', .false., 'Width of assimilation window is: ')
174endif
175
176! now recompute for the next window, so the code below can remain unchanged.
177! Figure out what time to advance the model to.
178
179! More control over time window use of observations would come in here
180time2 = get_closest_state_time_to(ens_time, next_time)
181
182! WATCH OUT FOR USING BOUNDARY TIME OBS TWICE; add one second to bottom time
183! ALSO, avoid having a negative time for the start (for low-order models in general)
184if(delta_time / 2 > time2) then
185 start_time = set_time(0, 0)
186else
187 start_time = time2 - delta_time / 2 + set_time(1, 0)
188endif
189end_time = time2 + delta_time / 2
190
191! Output very brief current start and end time at the message level
192if (print_trace_details == 0) then
193 call timechat(start_time, 'move_ahead', .false., 'Next assimilation window starts at: ')
194 call timechat(end_time, 'move_ahead', .false., 'Next assimilation window ends at: ')
195endif
196
197! This block of code gets called either if the next obs is not in the current window,
198! or if you're asking for the details of the assimilation window and obs times.
199if(next_time < start_time .or. next_time > end_time .or. print_trace_details > 0) then
200
201 if (time2 /= ens_time) then
202 !call timechat(next_time, 'move_ahead', .false., 'Next available observation time is: ')
203 call timechat(time2, 'move_ahead', .false., 'Next data time should be at: ', &
204 'Not within current window, model will be called to advance state.')
205 call timechat(start_time, 'move_ahead', .false., 'Next assimilation window starts at: ')
206 call timechat(end_time, 'move_ahead', .false., 'Next assimilation window ends at: ')
207 else
208 if (next_time >= start_time .and. next_time <= end_time) then
209 call timechat(next_time, 'move_ahead', .false., 'Next available observation time is: ', &
210 'Within current assimilation window, model does not need advance.')
211 else
212 call timechat(next_time, 'move_ahead', .false., 'Next available observation time is: ', &
213 'Next obs outside current assimilation window.')
214 endif
215 endif
216
217 ! if different mpi tasks have different times, the default is only process 0
218 ! will print messages. in this case we're headed towards a fatal error and
219 ! just trying to give the most info possible before exiting. so make all
220 ! mpi tasks which get into this block output. in the worst case you'll get
221 ! N sets of these messages which is messy, but probably better than having
222 ! the case where process 0 works but some other tasks fail and you get no
223 ! helpful info from them.
224 if (next_time < start_time .or. next_time > end_time) then
225 call set_output(.true.)
226 call error_handler(e_msg, ' ', ' ')
227 call error_handler(e_msg, 'move_ahead', 'Inconsistent model state/observation times. ')
228
229 if (next_time < start_time) then
230 call error_handler(e_msg, 'move_ahead', &
231 'Next observation cannot be earlier than start of new time window')
232 call error_handler(e_msg, ' ', ' ')
233 call timechat(next_time, 'move_ahead', .false., 'Next available observation at: ')
234 call timechat(start_time, 'move_ahead', .false., 'Next assimilation window starts at: ')
235 call timechat(ens_time, 'move_ahead', .false., 'Current model data time is: ')
236 errstring1 = 'If this is the start of the obs_seq file, '
237 errstring2 = 'can use filter namelist to set first obs or initial data time.'
238 else
239 call error_handler(e_msg, 'move_ahead', &
240 'Next observation is later than end of new time window')
241 call error_handler(e_msg, ' ', ' ')
242 call timechat(next_time, 'move_ahead', .false., 'Next available observation at: ')
243 call timechat(end_time, 'move_ahead', .false., 'Next assimilation window ends at: ')
244 call timechat(ens_time, 'move_ahead', .false., 'Current model data time is: ')
245 errstring1 = 'should not happen; code has miscomputed how far to advance'
246 errstring2 = ''
247 endif
248 call error_handler(e_msg, ' ', ' ')
249 call error_handler(e_err, 'move_ahead', &
250 'Inconsistent model state/observation times, cannot continue', &
251 source, text2=errstring1, text3=errstring2)
252 endif
253endif
254
255! WANT BETTER WINDOW CONTROL, TIME INTERPOLATION, TOO.
256! Get all the observations that are in the observation window - do this with the
257! window times, but set the returns to the data times.
258call get_obs_time_range(seq, start_time, end_time, key_bounds, num_obs_in_set, &
259 out_of_range, observation)
260
261! ok, not really a time detail, but if turned off, the output is pretty much
262! the same as the original.
263if (print_trace_details > 0) then
264 write (errstring, '(A,I8,A)') 'Next assimilation window contains up to ', &
265 num_obs_in_set, ' observations'
266 call error_handler(e_msg, 'move_ahead', errstring)
267endif
268
269curr_ens_time = ens_time
270next_ens_time = time2
271
272! Release the storage associated with the observation temp variable
273call destroy_obs(observation)
274
275if (ens_handle%my_pe == 0 .and. my_task_id() /= 0) then
276 call set_output(.false.)
277endif
278
279end subroutine move_ahead

284subroutine advance_state(ens_handle, ens_size, target_time, async, adv_ens_command, &
285 tasks_per_model_advance, file_info_output, file_info_input)
286
287! Advances all ensemble copies of the state to the target_time. Note that
288! there may be more than ens_size copies in the ens_handle storage. Copies other
289! than the ensemble copies need not be advanced in time.
290
291implicit none
292
293! ens_size is the number of ensembles that are model state and need to be advanced
294type(ensemble_type), intent(inout) :: ens_handle
295type(time_type), intent(in) :: target_time
296integer, intent(in) :: ens_size, async
297character(len=*), intent(in) :: adv_ens_command
298integer, intent(in) :: tasks_per_model_advance
299type(file_info_type), intent(inout) :: file_info_output
300type(file_info_type), intent(inout) :: file_info_input
301
302character(len = 129), dimension(ens_handle%num_copies) :: ic_file_name, ud_file_name
303character(len = 129) :: control_file_name
304! WARNING: ARE THESE LENGTHS OKAY?
305character(len = 129) :: system_string
306
307type(time_type) :: time_step, ens_time
308integer :: is1, is2, id1, id2, my_num_state_copies, global_ens_index
309integer :: i, control_unit, rc
310integer :: need_advance, any_need_advance
311logical :: read_time_from_file = .true.
312
313! Initialize if needed
314if(.not. module_initialized) then
315 call initialize_module
316 module_initialized = .true.
317endif
318
319! Tasks without ensemble members do not know the ensemble time, so assume
320! no advance until we can read the time on at least some tasks and confirm
321! the advance is needed (which it should be).
322need_advance = 0
323
324! Determine model time_step
325time_step = get_model_time_step()
326
327! Loop through each model state and advance; Count number of copies that are state ensembles
328my_num_state_copies = 0
329ensemble_members: do i = 1, ens_handle%my_num_copies
330 global_ens_index = ens_handle%my_copies(i)
331
332 ! If global_ens_index is not one of the model state ensembles, don't need to advance
333 ! Assumes that state ensemble copies have smallest global index and are stored
334 ! contiguously on local processor
335 if(global_ens_index > ens_size) exit ensemble_members
336
337 ! No need to advance if already at target time
338 if(ens_handle%time(i) == target_time) exit ensemble_members
339
340 ! Check for time error
341 if(ens_handle%time(i) > target_time) then
342 call get_time(ens_handle%time(i), is1, id1)
343 call get_time(target_time, is2, id2)
344 write(errstring,*)'target time ',is2,id2,' is before model_time ',is1,id1
345 call error_handler(e_err,'advance_state', errstring, source)
346 endif
347
348 ! Ok, this task does need to advance something.
349 need_advance = 1
350
351 ! Increment number of ensemble member copies I have.
352 my_num_state_copies = my_num_state_copies + 1
353
354 !------------- Block for subroutine callable adv_1step interface -----------
355 if(async == 0) then
356
357 do while(ens_handle%time(i) < target_time)
358 call adv_1step(ens_handle%vars(:, i), ens_handle%time(i))
359 ens_handle%time(i) = ens_handle%time(i) + time_step
360 end do
361
362 !-------------- End single subroutine callable adv_1step interface ---------
363 else
364 !-------------- Block for calling shell to advance model -------------------
365
366 ! Can only handle up to 10000 ensemble members
367 if(global_ens_index >= 10000) then
368 write(errstring,*)'Trying to use ', ens_size,' model states -- too many.'
369 call error_handler(e_warn,'advance_state',errstring,source)
370 call error_handler(e_err,'advance_state','Use less than 10000 member ensemble.', &
371 source)
372 endif
373
374 !>@todo FIXME write the netCDF restart names into the control file
375 ! Create file names for input and output state files for this copy
376! write(ic_file_name(i), '("assim_model_state_ic.", i4.4)') global_ens_index
377! write(ud_file_name(i), '("assim_model_state_ud.", i4.4)') global_ens_index
378
379 ! Open a restart file and write state following a target time
380 call write_state(ens_handle, file_info_output)
381
382 endif
383
384 !-------------- End of block for calling shell to advance model -------------
385
386end do ensemble_members
387
388
389! Following is for async options that use shell to advance model
390shell_advance_methods: if(async /= 0) then
391
392 ! If no one needs advance, get out now. This is a global communication routine.
393 call sum_across_tasks(need_advance, any_need_advance)
394 if (any_need_advance == 0) then
395 call error_handler(e_msg, 'advance_state', 'Model time already at requested time')
396 return
397 endif
398
399 ! Get a unique name for the control file; use process id
400 if(my_task_id() >= 10000) call error_handler(e_err, 'advance_state', &
401 'Can only have 10000 processes', source)
402 write(control_file_name, '("filter_control", i5.5)') my_task_id()
403
404 if (async == 2) then
405
406 if (have_members(ens_handle, ens_size)) then
407 ! Everyone writes the ic and ud file names to the control file
408 control_unit = get_unit()
409 open(unit = control_unit, file = trim(control_file_name))
410 ! Write out the file names to a control file
411 ! Also write out the global ensemble member number for the script
412 ! if it needs to create unique filenames for its own use.
413 do i = 1, my_num_state_copies
414 write(control_unit, '(i5)') ens_handle%my_copies(i)
415 write(control_unit, '(a)' ) trim(ic_file_name(i)) !!@todo netCDF
416 write(control_unit, '(a)' ) trim(ud_file_name(i)) !!@todo netCDF
417 end do
418 close(control_unit)
419
420 ! Arguments to advance model script are unique id and number of copies
421 write(system_string, '(i10, 1x, i10)') my_task_id(), my_num_state_copies
422
423 if (debug) write(*,*) 'iam ', my_task_id(), ' ready to execute shell command'
424
425 ! Issue a system command with arguments my_task, my_num_copies, and control_file
426 rc = shell_execute(trim(adv_ens_command)//' '//trim(system_string)//' '//trim(control_file_name))
427 if (debug) write(*,*) 'iam ', my_task_id(), ' shell execute returns ', rc
428 if (debug) write(*,*) 'iam ', my_task_id(), ' checking existance of file ', trim(control_file_name)
429
430 ! if control file is still here, the advance failed
431 if(file_exist(control_file_name)) then
432 write(errstring, "(A)")'If advance script finishes ok it removes '//trim(control_file_name)
433 write(errstring1,"(A)")'It still exists, so 1+ members listed in that file failed to run'
434 write(errstring2,"(A)")'Check the output of the model or script to find the error.'
435 call error_handler(e_err,'advance_state', errstring, source, &
436 text2=errstring1, text3=errstring2)
437 endif
438 endif
439
440 else if (async == 4) then
441
442 ! Only task 0 writes all ens names to file.
443 if (my_task_id() == 0) then
444 ! Write the ic and ud file names to the control file
445 control_unit = get_unit()
446 open(unit = control_unit, file = trim(control_file_name))
447 ! Write out the file names to a control file
448 ! Also write out the global ensemble member number for the script
449 ! if it needs to create unique filenames for its own use.
450 do i = 1, ens_size
451 write(control_unit, '(i5)') i
452 write(control_unit, '("assim_model_state_ic.", i4.4)') i
453 write(control_unit, '("assim_model_state_ud.", i4.4)') i
454 end do
455 close(control_unit)
456 endif
457
458 ! make sure all tasks have finished writing their initial condition files
459 ! (in the previous block above) before letting proc 0 start the model advances.
460 call task_sync()
461
462 ! PE0 tells the run script to all the model advance script here, then sleeps.
463 ! the other tasks just sleep. TODO: this code is too intertwined with the
464 ! run script - it assumes that the script has already created 2 fifo/pipe
465 ! files with specific names: mkfifo filter_to_model.lock model_to_filter.lock
466 ! eventually it should probably be yet another separate executable which
467 ! keeps the code all together in the mpi module.
468
469 ! all tasks now get here, even those which do not have ensemble members,
470 ! so this bookkeeping is not needed, i think.
471 !! tell any waiting tasks that we are indeed doing a model advance
472 !! here and they need to block.
473 !call sum_across_tasks(1, i)
474
475 ! block, not spin, until the model advance is done.
476 ! the code does not resume running until the separate program 'wakeup_filter'
477 ! calls the companion 'resume_task()'.
478 call block_task()
479
480
481 ! if control file is still here, the advance failed
482 if(file_exist(control_file_name)) then
483 write(errstring, "(A)")'If advance script finishes ok it removes '//trim(control_file_name)
484 write(errstring1,"(A)")'It still exists, so 1+ members listed in that file failed to run'
485 write(errstring2,"(A)")'Check the output of the model or script to find the error.'
486 call error_handler(e_err,'advance_state', errstring, source, &
487 text2=errstring1, text3=errstring2)
488 endif
489
490 else
491 ! Unsupported option for async error
492 write(errstring,*)'input.nml - async is ',async,' must be 0, 2, or 4'
493 call error_handler(e_err,'advance_state', errstring, source)
494 endif
495
496 ! All should be done, read in the states and proceed
497 ! NOTE: if async 2 and we add support for tasks_per_model_advance, then we need
498 ! another sync here before trying to read in from a task which might not have been
499 ! the one which advanced it.
500 call read_state(ens_handle, file_info_input, read_time_from_file, ens_time)
501
502end if shell_advance_methods
503
504
505! any process that owns members, check that they advanced ok.
506if (have_members(ens_handle, ens_size)) then
507
508 do i = 1, my_num_state_copies
509
510 ! Get times of all ensemble members, one by one, to catch errors
511 ! when some advanced ok but not all even if they appeared to.
512 call get_ensemble_time(ens_handle, i, ens_time)
513
514 ! error out if model state did not advance to when requested.
515 if (ens_time /= target_time) then
516 ! make sure times print out on whatever task this is (it uses msgs)
517 call set_output(.true.)
518 call timechat(ens_time, 'advance_state', .true., 'Model advance time is now: ')
519 call timechat(target_time, 'advance_state', .false., 'Model advance time NOT what requested: ')
520
521 write(errstring2, *) 'in timestamp on ensemble member ', ens_handle%my_copies(i)
522 call error_handler(e_err, 'advance_state', 'Model advance complete but model time not correct', &
523 source, text2=errstring2)
524 endif
525 enddo
526
527 if (print_trace_details > 0) then
528 call error_handler(e_msg, 'advance_state', 'Model advance complete, model time updated to requested time')
529 endif
530
531endif
532
533
534end subroutine advance_state

542subroutine timechat(a_time, label, blank, string1, string2, string3)
543 type(time_type), intent(in) :: a_time
544 character(len=*), intent(in) :: label, string1
545 logical, intent(in) :: blank
546 character(len=*), intent(in), optional :: string2, string3
547
548! prettyprint a time, with an optional preceeding blank line, a label
549! before the time, and up to 2 additional text lines afterwards.
550

552
553call get_time(a_time, a_sec, a_day)
554write (errstring, '(A,A,I8,A,I6)') string1, ' day=', a_day, ' sec=', a_sec
555
556if (blank) call error_handler(e_msg, ' ', ' ')
557call error_handler(e_msg, label, trim(errstring))
558
559if (present(string2)) then
560 call error_handler(e_msg, label, string2)
561endif
562
563if (present(string3)) then
564 call error_handler(e_msg, label, string3)
565endif

567end subroutine timechat

571subroutine set_obs_model_trace(execution_level, timestamp_level)
572 integer, intent(in) :: execution_level
573 integer, intent(in) :: timestamp_level
574
575! set module local vars from the calling code to indicate how much
576! output we should generate from this code. execution level is
577! intended to make it easier to figure out where in the code a crash
578! is happening; timestamp level is intended to help with gross levels
579! of overall performance profiling. eventually, a level of 1 will

594! return true if this task contains ensemble copies.
595! mean, sd, etc do not count.
596!
597! note this depends on how filter (or the calling code) has laid out
598! the actual ensemble members vs the other state-length arrays
599! carried around in the ensemble handle.
600
601type(ensemble_type), intent(in) :: ens_handle
602integer, intent(in) :: ens_size

605integer :: my_first_copy
606

608if(ens_handle%my_num_copies >= 1) then
609 my_first_copy = ens_handle%my_copies(1)
610else
611 my_first_copy = 0
612endif
613

615! not aux data (mean, inflation, etc), return true.
616have_members = ((ens_handle%my_num_copies >= 1) .and. (my_first_copy <= ens_size))
