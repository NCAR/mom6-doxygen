35subroutine test_normal
36
37! This routine provides limited tests of the numerics in this module. It begins
38! by comparing a handful of cases of the cdf to results from Matlab. It
39! then tests the quality of the inverse cdf for a single mean/sd. Failing
40! these tests suggests a serious problem. Passing them does not indicate that
41! there are acceptable results for all possible inputs.
42
43! Set number of equally spaced trials for the test F-1(F(x)) where F is the CDF.
44integer, parameter :: num_trials = 10000000
45
46integer :: i, j
47real(r8) :: sd, quantile, inv, max_diff(16), max_q(16), max_matlab_diff
48
49! Comparative results for a handful of cases from MATLAB21a
50real(r8) :: cdf_diff(7)
51real(r8) :: mmean(7) = [0.0_r8, 1.0_r8, -1.0_r8, 0.0_r8, 0.0_r8, 0.0_r8, 0.5_r8]
52real(r8) :: msd(7) = [0.5_r8, 1.0_r8, 2.0_r8, 4.0_r8, 5.0_r8, 6.0_r8, 0.25_r8]
53real(r8) :: mx(7) = [0.1_r8, 0.2_r8, 0.3_r8, 0.4_r8, 0.5_r8, 0.6_r8, 0.7_r8]
54! Generated by matlab normcdf(mx, mmean, msd)
55real(r8) :: mcdf(7) = [0.579259709439103_r8, 0.211855398583397_r8, 0.742153889194135_r8, &
56 0.539827837277029_r8, 0.539827837277029_r8, 0.539827837277029_r8, &
57 0.788144601416603_r8]
58! Bounds for quantile inversion differences
59real(r8) :: inv_diff_bound(16) = [1e-10_r8, 1e-10_r8, 1e-10_r8, 1e-10_r8, 1e-10_r8, &
60 1e-9_r8, 1e-8_r8, 1e-7_r8, 1e-7_r8, 1e-6_r8, &
61 1e-5_r8, 1e-4_r8, 1e-3_r8, 1e-2_r8, 1e-1_r8, 1e-0_r8]
62
63! Compare to matlab
64! Absolute value of differences should be less than 1e-15
65do i = 1, 7
66 cdf_diff(i) = normal_cdf(mx(i), mmean(i), msd(i)) - mcdf(i)
67end do
68max_matlab_diff = maxval(abs(cdf_diff))
69if(max_matlab_diff > 1.0e-15_r8) then
70 write(*, *) 'WARNING: Difference from Matlab baseline is too large ', max_matlab_diff
71else
72 write(*, *) 'Agreement with Matlab baseline is okay: max diff is < 1e-15 ', max_matlab_diff
73endif
74
75! Keep track of differences as function of quantile
76max_diff = 0.0_r8
77do j = 1, 16
78 max_q(j) = 1.0_r8 - 0.1**j
79enddo
80
81! Test the inversion of the cdf over +/- 30 standard deviations around mean
82do i = 1, num_trials + 1
83 sd = min_sd + (i - 1.0_r8) * (max_sd - min_sd) / num_trials
84 quantile = normal_cdf(sd, 0.0_r8, 1.0_r8)
85 inv = inv_std_normal_cdf(quantile)
86 do j = 1, 16
87 if(quantile < max_q(j)) then
88 max_diff(j) = max(abs(sd-inv), max_diff(j))
89 endif
90 enddo
91end do
92
93do j = 1, 16
94 if(max_diff(j) > inv_diff_bound(j)) then
95 write(*, *) 'WARNING: Max inversion diff ', max_diff(j), ' > bound ', inv_diff_bound(j), &
96 'for quantiles < ', max_q(j)
97 else
98 write(*, *) 'Max inversion diff ', max_diff(j), ' OK, bound ', inv_diff_bound(j), &
99 'for quantiles < ', max_q(j)
100 endif
101end do
102
103end subroutine test_normal

126function normal_cdf(x_in, mean, sd)
127
128! Approximate cumulative distribution function for normal
129
130real(r8) :: normal_cdf
131real(r8), intent(in) :: x_in
132real(r8), intent(in) :: mean, sd
133
134real(digits12) :: nx
135
136! Convert to a standard normal
137nx = (x_in - mean) / sd
138
139if(nx < 0.0_digits12) then
140 normal_cdf = 0.5_digits12 * erfc(-nx / sqrt(2.0_digits12))
141else
142 normal_cdf = 0.5_digits12 * (1.0_digits12 + erf(nx / sqrt(2.0_digits12)))
143endif
144
145end function normal_cdf

193function approx_inv_normal_cdf(quantile_in) result(x)
194
195real(r8) :: x
196real(r8), intent(in) :: quantile_in
197
198! This is used to get a good first guess for the search in inv_std_normal_cdf
199! The params argument is not needed here but is required for consistency &
200! with other distributions
201
202! normal inverse
203! translate from http://home.online.no/~pjacklam/notes/invnorm
204! a routine written by john herrero
205
206real(r8) :: quantile
207real(r8) :: quantile_low,quantile_high
208real(r8) :: a1,a2,a3,a4,a5,a6
209real(r8) :: b1,b2,b3,b4,b5
210real(r8) :: c1,c2,c3,c4,c5,c6
211real(r8) :: d1,d2,d3,d4
212real(r8) :: r, s
213
214! Truncate out of range quantiles, converts them to smallest positive number or largest number <1
215! This solution is stable, but may lead to underflows being thrown. May want to
216! think of a better solution.
217quantile = quantile_in
218if(quantile <= 0.0_r8) quantile = tiny(quantile_in)
219if(quantile >= 1.0_r8) quantile = nearest(1.0_r8, -1.0_r8)
220
221a1 = -39.69683028665376_digits12
222a2 = 220.9460984245205_digits12
223a3 = -275.9285104469687_digits12
224a4 = 138.357751867269_digits12
225a5 = -30.66479806614716_digits12
226a6 = 2.506628277459239_digits12
227b1 = -54.4760987982241_digits12
228b2 = 161.5858368580409_digits12
229b3 = -155.6989798598866_digits12
230b4 = 66.80131188771972_digits12
231b5 = -13.28068155288572_digits12
232c1 = -0.007784894002430293_digits12
233c2 = -0.3223964580411365_digits12
234c3 = -2.400758277161838_digits12
235c4 = -2.549732539343734_digits12
236c5 = 4.374664141464968_digits12
237c6 = 2.938163982698783_digits12
238d1 = 0.007784695709041462_digits12
239d2 = 0.3224671290700398_digits12
240d3 = 2.445134137142996_digits12
241d4 = 3.754408661907416_digits12
242quantile_low = 0.02425_digits12
243quantile_high = 1_digits12 - quantile_low
244! Split into an inner and two outer regions which have separate fits
245if(quantile < quantile_low) then
246 s = sqrt(-2.0_digits12 * log(quantile))
247 x = (((((c1*s + c2)*s + c3)*s + c4)*s + c5)*s + c6) / &
248 ((((d1*s + d2)*s + d3)*s + d4)*s + 1.0_digits12)
249else if(quantile > quantile_high) then
250 s = sqrt(-2.0_digits12 * log(1.0_digits12 - quantile))
251 x = -(((((c1*s + c2)*s + c3)*s + c4)*s + c5)*s + c6) / &
252 ((((d1*s + d2)*s + d3)*s + d4)*s + 1.0_digits12)
253else
254 s = quantile - 0.5_digits12
255 r = s*s
256 x = (((((a1*r + a2)*r + a3)*r + a4)*r + a5)*r + a6)*s / &
257 (((((b1*r + b2)*r + b3)*r + b4)*r + b5)*r + 1.0_digits12)
258endif
259
260end function approx_inv_normal_cdf

314interface
315 function first_guess(quantile, p)
316 use types_mod, only : r8
317 use distribution_params_mod, only : distribution_params_type
318 real(r8) :: first_guess
319 real(r8), intent(in) :: quantile
320 type(distribution_params_type), intent(in) :: p
321 end function
322end interface
323
324real(r8) :: x
325real(r8), intent(in) :: quantile_in
326type(distribution_params_type), intent(in) :: p
327
328! This naive Newton method is much more accurate than approx_inv_normal_cdf, especially
329! for quantile values less than 0.5.
330
331! Given a quantile q, finds the value of x for which the standard normal cdf
332! has approximately this quantile
333
334! Limit on the total iterations; Increasing this does not change any of the results
335! that do not converge for the test_normal call on gfortran.
336integer, parameter :: max_iterations = 50
337
338! Limit on number of times to halve the increment; No deep thought.
339integer, parameter :: max_half_iterations = 25
340
341real(r8) :: quantile
342real(r8) :: reltol, dq_dx, delta
343real(r8) :: x_guess, q_guess, x_new, q_new, del_x, del_q, del_q_old, q_old
344integer :: iter, j
345
346real(r8) :: lower_bound, upper_bound
347logical :: bounded_below, bounded_above
348
349! Extract the required information from the p type
350bounded_below = p%bounded_below; bounded_above = p%bounded_above
351lower_bound = p%lower_bound; upper_bound = p%upper_bound
352
353quantile = quantile_in
354
355! Do a test for illegal values on the quantile
356if(quantile < 0.0_r8 .or. quantile > 1.0_r8) then
357 ! Need an error message
358 write(errstring, *) 'Illegal Quantile input', quantile
359 call error_handler(e_err, 'inv_cdf', errstring, source)
360endif
361
362! If the distribution is bounded, quantiles at the limits have values at the bounds
363if(bounded_below .and. quantile == 0.0_r8) then
364 x = lower_bound
365 return
366endif
367if(bounded_above .and. quantile == 1.0_r8) then
368 x = upper_bound
369 return
370endif
371
372! If input quantiles are outside the numerically supported range, move them to the extremes
373quantile = min(quantile, max_quantile)
374! code tests stably for many distributions with min_quantile of 0.0, could remove this
375quantile = max(quantile, min_quantile)
376
377! Get first guess from functional approximation
378x_guess = first_guess(quantile, p)
379
380! Evaluate the cdf
381q_guess = cdf(x_guess, p)
382
383del_q = q_guess - quantile
384
385! Iterations of the Newton method to approximate the root
386do iter = 1, max_iterations
387 ! Analytically, the PDF is derivative of CDF but this can be numerically inaccurate for extreme values
388 ! Use numerical derivatives of the CDF to get more accurate inversion
389 ! These values for the delta for the approximation work with Gfortran
390 delta = max(1e-8_r8, 1e-8_r8 * abs(x_guess))
391 dq_dx = (cdf(x_guess + delta, p) - cdf(x_guess - delta, p)) / (2.0_r8 * delta)
392 ! Derivative of 0 means we're not going anywhere else
393 if(dq_dx <= 0.0_r8) then
394 x = x_guess
395 return
396 endif
397
398 ! Linear approximation for how far to move in x
399 del_x = del_q / dq_dx
400 x_new = x_guess - del_x
401
402 ! Look for convergence; If the change in x is smaller than approximate precision
403 reltol = (epsilon(x_guess))**(0.75_r8)
404 if(abs(del_x) <= reltol) then
405 x = x_new
406 return
407 endif
408
409 ! If we've gone too far, the new error will be bigger than the old;
410 ! Repeatedly half the distance until this is rectified
411 del_q_old = del_q
412 q_new = cdf(x_new, p)
413 do j = 1, max_half_iterations
414 del_q = q_new - quantile
415 if (abs(del_q) < abs(del_q_old)) then
416 exit
417 endif
418 q_old = q_new
419 x_new = (x_guess + x_new)/2.0_r8
420 q_new = cdf(x_new, p)
421 ! If q isn't changing, no point in continuing
422 if(q_old == q_new) exit
423
424 end do
425
426 x_guess = x_new
427end do
428
429! For now, have switched a failed convergence to return the latest guess
430! This has implications for stability of probit algorithms that require further study
431! Not currently happening for any of the test cases on gfortran
432x = x_new
433write(errstring, *) 'Failed to converge for quantile ', quantile
434call error_handler(e_msg, 'inv_cdf', errstring, source)
435!!!call error_handler(E_ERR, 'inv_cdf', errstring, source)
436
437end function inv_cdf

497subroutine inv_cdf_quadrature_like(quantiles, ens, likelihood, ens_size, cdf, p, x_out)
498
499interface
500 function cdf(x, p)
501 use types_mod, only : r8
502 use distribution_params_mod, only : distribution_params_type
503 real(r8) :: cdf
504 real(r8), intent(in) :: x
505 type(distribution_params_type), intent(in) :: p
506 end function
507end interface
508
509integer, intent(in) :: ens_size
510real(r8), intent(in) :: quantiles(ens_size)
511real(r8), intent(in) :: ens(ens_size)
512real(r8), intent(in) :: likelihood(ens_size)
513type(distribution_params_type), intent(in) :: p
514real(r8), intent(out) :: x_out(ens_size)
515
516integer :: i
517real(r8) :: quad_like(ens_size + 1), q_ens(ens_size + 1)
518
519! Assume that the quantiles and the corresponding ens are sorted
520
521! Get the likelihood for each of the ens_size + 1 intervals
522do i = 2, ens_size
523 quad_like(i) = (likelihood(i - 1) + likelihood(i)) / 2.0_r8
524end do
525quad_like(1) = likelihood(1)
526quad_like(ens_size + 1) = likelihood(ens_size)
527
528! Compute the quantiles at the ensemble boundaries for the posterior
529q_ens(1) = quad_like(1) * quantiles(1)
530do i = 2, ens_size
531 q_ens(i) = q_ens(i - 1) + quad_like(i) * (quantiles(i) - quantiles(i - 1))
532end do
533q_ens(ens_size + 1) = q_ens(ens_size) + &
534 quad_like(ens_size + 1) * (1.0_r8 - quantiles(ens_size))
535
536! Normalize so that this is a posterior cdf
537q_ens = q_ens / q_ens(ens_size + 1)
538
539end subroutine inv_cdf_quadrature_like
