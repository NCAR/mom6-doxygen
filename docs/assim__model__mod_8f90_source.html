80function get_closest_state_time_to(model_time, given_time)
81!----------------------------------------------------------------------
82!
83! Returns the time closest to the given time that the model can reach
84! with its state. Initial implementation just assumes fixed timestep.
85! Need to describe potentially more general time-stepping capabilities
86! from the underlying model in the long run.
87
88type(time_type) :: get_closest_state_time_to
89type(time_type), intent(in) :: model_time, given_time
90
91type(time_type) :: time_step
92
93! Get the model time step capabilities - this is the
94! shortest amount of time you can/want to ask the model
95! to advance the state
96time_step = shortest_time_between_assimilations()
97
98if(model_time > given_time) then
99 ! If model_time is past start of obs window, don't advance it
100 get_closest_state_time_to = model_time
101 return
102endif
103
104get_closest_state_time_to = model_time
105
106do while((time_step + 2*get_closest_state_time_to) < 2*given_time)
107 get_closest_state_time_to = get_closest_state_time_to + time_step
108enddo
109
110end function get_closest_state_time_to
