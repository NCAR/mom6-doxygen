79subroutine get_expected_absolute_humidity(state_handle, ens_size, location, ah, istatus)
80
81type(ensemble_type), intent(in) :: state_handle
82integer, intent(in) :: ens_size
83type(location_type), intent(in) :: location
84real(r8), intent(out) :: ah(ens_size) ! absolute humidity
85integer, intent(out) :: istatus(ens_size)
86
87real(r8), dimension(ens_size) :: qvap, tmpk, pres, rho_dry, tmpv
88integer, dimension(ens_size) :: qvap_istatus, tmpk_istatus, pres_istatus
89real(r8) :: xyz(3)
90logical :: return_now
91
92if ( .not. module_initialized ) call initialize_module
93
94istatus = 0
95
96! interpolate the mixing ratio to the location
97call interpolate(state_handle, ens_size, location, qty_vapor_mixing_ratio, qvap, qvap_istatus)
98call track_status(ens_size, qvap_istatus, ah, istatus, return_now)
99
100where (istatus == 0 .and. qvap < 0.0_r8) qvap = epsilon(0.0_r8) ! clamping?
101where (istatus /= 0) istatus = 99
102if(return_now) return
103
104! interpolate the temperature to the desired location
105call interpolate(state_handle, ens_size, location, qty_temperature, tmpk, tmpk_istatus)
106call track_status(ens_size, tmpk_istatus, ah, istatus, return_now)
107
108where (tmpk <= 0.0_r8)
109 ah = missing_r8
110 istatus = 99
111endwhere
112
113if(all(istatus /= 0)) return
114
115! interpolate the pressure, if observation location is not pressure
116if ( is_vertical(location, "PRESSURE") ) then
117 xyz = get_location(location)
118 pres = xyz(3)
119else
120 ! pressure comes back in pascals (not hPa or mb)
121 call interpolate(state_handle, ens_size, location, qty_pressure, pres, pres_istatus)
122 call track_status(ens_size, pres_istatus, ah, istatus, return_now)
123
124 where (pres <= 0.0_r8 .or. pres >= 120000.0_r8)
125 ah = missing_r8
126 istatus = 99
127 end where
128 if(all(istatus /= 0)) return
129endif
130
131! Compute the ah
132where (istatus == 0)
133 tmpv = tmpk * (1.0_r8 + qvap * gas_constant_v / gas_constant)
134 rho_dry = pres / (gas_constant * tmpv)
135 ah = qvap * rho_dry
136end where
137
138! Warnings - first time only
139if (first_time_warn_low) then
140 if (any(ah < min_value .and. istatus == 0)) then
141 write(string1, *) 'checking absolute humidity value computed by the forward operator'
142 write(string2,*)'all values lower than ',min_value,' will be set to ',min_value
143 call error_handler(e_msg,'get_expected_absolute_humidity', string1, &
144 text2=string2, text3='this message will only print once')
145 first_time_warn_low = .false.
146 endif
147endif
148if (first_time_warn_high) then
149 if (any(ah > max_value .and. istatus == 0)) then
150 write(string1, *) 'checking absolute humidity value computed by the forward operator'
151 write(string2,*)'all values larger than ',max_value,' will be set to ',max_value
152 call error_handler(e_msg,'get_expected_absolute_humidity', string1, &
153 text2=string2, text3='this message will only print once')
154 first_time_warn_high = .false.
155 endif
156endif
157
158! Actually force the absolute humidity to be between MIN_VALUE and MAX_VALUE
159where(ah < min_value .and. istatus == 0) ah = min_value
160where(ah > max_value .and. istatus == 0) ah = max_value
161
162return
163end subroutine get_expected_absolute_humidity
