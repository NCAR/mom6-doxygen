58function comp_reg_factor(num_groups, regress, obs_time, &
59 obs_index, state_index, obs_state_ind, obs_state_max)
60
61! Computes factor by which to multiply regression coefficients
62! for a given distribution of sample regressions OR computes
63! factor for a single sample of a regression using some other
64! methodology (for instance time mean from previous runs). Could
65! also implement the standard distance dependence method, too.
66
67integer, intent(in) :: num_groups
68real(r8), intent(in) :: regress(num_groups)
69type(time_type), intent(in) :: obs_time
70integer, intent(in) :: obs_index
71integer(i8), intent(in) :: state_index
72integer, intent(in), optional :: obs_state_ind, obs_state_max
73
74real(r8) :: comp_reg_factor
75
76real(r8) :: sum_reg2, sum_reg_reg
77
78integer :: i, j, ii, jj, iunit, io, secs, days
79
80!--------------------------------------------------------
81! Initialize namelist if not already done
82if(.not. namelist_initialized) then
83
84
85 namelist_initialized = .true.
86
87 ! Read the namelist entry
88 call find_namelist_in_file("input.nml", "reg_factor_nml", iunit)
89 read(iunit, nml = reg_factor_nml, iostat = io)
90 call check_namelist_read(iunit, io, "reg_factor_nml")
91
92 ! Record the namelist values used for the run ...
93 if (do_nml_file()) write(nmlfileunit, nml=reg_factor_nml)
94 if (do_nml_term()) write( * , nml=reg_factor_nml)
95
96 ! See if diagnostic output is requested, if so, open file
97 if(save_reg_diagnostics) then
98 diag_unit = open_file(reg_diagnostics_file, action = 'write')
99 endif
100
101endif
102!---------------------------------------------------------
103
104!_____________________________________________________________________
105if(select_regression == 1) then
106
107! Get regression directly from sampling theory
108! If only one group, don't know what else to do
109 if(num_groups == 1) then
110 comp_reg_factor = 1.0_r8
111 else
112
113 sum_reg_reg = 0.0_r8
114 sum_reg2 = sum(regress * regress)
115 do i = 1, num_groups
116 do j = i + 1, num_groups
117 sum_reg_reg = sum_reg_reg + regress(i) * regress(j)
118 end do
119 end do
120 if (sum_reg2 /= 0.0_r8) then
121 comp_reg_factor = 2.0_r8 * sum_reg_reg / (sum_reg2 * (num_groups - 1))
122 else
123 comp_reg_factor = 0.0_r8
124 endif
125
126 if(comp_reg_factor < 0.0_r8) comp_reg_factor = 0.0_r8
127
128 ! Write out diagnostic information
129 if(save_reg_diagnostics) then
130
131! DATA REDUCTION FOR WORKSHOP PURPOSES
132 if(obs_index <= 4 .and. state_index > 0) then
133
134 call get_time(obs_time, secs, days)
135 write(diag_unit, 22) days, secs, obs_index, state_index, comp_reg_factor
136 22 format(4(i7, 1x), e14.4)
137 endif
138 endif
139
140 endif
141
142!___________________________________________________________________
143
144else if(select_regression == 2) then
145
146! Table lookup version for time mean, temporary implementation
147! This only works for a model with a time invariant observation set
148 if(first_call) then
149 first_call = .false.
150! Read in the regression statistics file
151 iunit = get_unit()
152 open(unit = iunit, file = input_reg_file)
153 read(iunit, *) num_obs, model_size
154 allocate(time_mean_reg(num_obs, model_size))
155 do j = 1, num_obs
156 do i = 1, model_size
157 read(iunit, *) jj, ii, time_mean_reg(j, i)
158 end do
159 end do
160 close(iunit)
161 endif
162
163 comp_reg_factor = time_mean_reg(obs_index, state_index)
164
165 if(comp_reg_factor < 0.0_r8) comp_reg_factor = 0.0_r8
166
167!_____________________________________________________________________
168
169else if(select_regression == 3) then
170
171 if(first_call) then
172 first_call = .false.
173 iunit = get_unit()
174 open(unit = iunit, file = 'obs_state_reg_file')
175 allocate(obs_state_reg(obs_state_max))
176 do i = 1, obs_state_max
177 read(iunit, 11) obs_state_reg(i)
178 close(unit = iunit)
17911 format(f5.3)
180 end do
181 end if
182
183 comp_reg_factor = obs_state_reg(obs_state_ind)
184
185!_____________________________________________________________________
186
187else
188 call error_handler(e_err,'comp_reg_factor', &
189 'Illegal value for namelist parameter select_regression',source)
190endif
191
192end function comp_reg_factor
