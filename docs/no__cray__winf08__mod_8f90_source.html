54!-------------------------------------------------------------
55!> For the non-distributed case this is simply a transpose
56!> For the distributed case memory is allocated in this module
57!> then an mpi window is attached to this memory.
58subroutine create_state_window(state_ens_handle, fwd_op_ens_handle, qc_ens_handle)
59
60type(ensemble_type), intent(inout) :: state_ens_handle
61type(ensemble_type), intent(inout), optional :: fwd_op_ens_handle
62type(ensemble_type), intent(inout), optional :: qc_ens_handle
63
64integer :: ierr
65integer :: bytesize !< size in bytes of each element in the window
66integer :: my_num_vars !< my number of vars
67
68! Find out how many copies to put in the window
69! copies_in_window is not necessarily equal to ens_handle%num_copies
70data_count = copies_in_window(state_ens_handle)
71
72if (get_allow_transpose(state_ens_handle)) then
73 call all_copies_to_all_vars(state_ens_handle)
74 if (present(fwd_op_ens_handle)) then
75 call all_copies_to_all_vars(fwd_op_ens_handle)
76 endif
77 if (present(qc_ens_handle)) then
78 call all_copies_to_all_vars(qc_ens_handle)
79 endif
80else
81 ! find how many variables I have
82 my_num_vars = state_ens_handle%my_num_vars
83
84 call mpi_type_size(datasize, bytesize, ierr)
85 window_size = my_num_vars*data_count*bytesize
86
87 allocate(contiguous_fwd(data_count*my_num_vars))
88 contiguous_fwd = reshape(state_ens_handle%copies(1:data_count, :), (/my_num_vars*data_count/))
89
90 ! Expose local memory to RMA operation by other processes in a communicator.
91 call mpi_win_create(contiguous_fwd, window_size, bytesize, mpi_info_null, get_dart_mpi_comm(), state_win, ierr)
92endif
93
94! Set the current window to the state window
95current_win = state_window
96
97data_count = copies_in_window(state_ens_handle)
98
99end subroutine create_state_window

101!-------------------------------------------------------------
102!> Using a mean ensemble handle.
103!>
104subroutine create_mean_window(state_ens_handle, mean_copy, distribute_mean)
105
106type(ensemble_type), intent(in) :: state_ens_handle
107integer, intent(in) :: mean_copy
108logical, intent(in) :: distribute_mean
109
110integer :: ierr
111integer :: bytesize
112integer :: my_num_vars !< number of elements a task owns
113
114! create an ensemble handle of just the mean copy.
115use_distributed_mean = distribute_mean
116
117if (use_distributed_mean) then
118 call init_ensemble_manager(mean_ens_handle, 1, state_ens_handle%num_vars) ! distributed ensemble
119 call set_num_extra_copies(mean_ens_handle, 0)
120 mean_ens_handle%copies(1,:) = state_ens_handle%copies(mean_copy, :)
121 allocate(mean_1d(state_ens_handle%my_num_vars))
122 mean_1d(:) = mean_ens_handle%copies(1,:)
123
124 ! find out how many variables I have
125 my_num_vars = mean_ens_handle%my_num_vars
126 call mpi_type_size(datasize, bytesize, ierr)
127 window_size = my_num_vars*bytesize
128
129 ! Need a simply contiguous piece of memory to pass to mpi_win_create
130 ! Expose local memory to RMA operation by other processes in a communicator.
131 call mpi_win_create(mean_1d, window_size, bytesize, mpi_info_null, get_dart_mpi_comm(), mean_win, ierr)
132else
133 call init_ensemble_manager(mean_ens_handle, 1, state_ens_handle%num_vars, transpose_type_in = 3)
134 call set_num_extra_copies(mean_ens_handle, 0)
135 mean_ens_handle%copies(1,:) = state_ens_handle%copies(mean_copy, :)
136 call all_copies_to_all_vars(mean_ens_handle) ! this is a transpose-duplicate
137endif
138
139! grabbing mean directly, no windows are being used
140current_win = mean_window
141
142data_count = copies_in_window(mean_ens_handle) ! One.
143
144end subroutine create_mean_window

146!-------------------------------------------------------------
147!> End epoch of state access.

149subroutine free_state_window(state_ens_handle, fwd_op_ens_handle, qc_ens_handle)
150
151type(ensemble_type), intent(inout) :: state_ens_handle
152type(ensemble_type), intent(inout), optional :: fwd_op_ens_handle
153type(ensemble_type), intent(inout), optional :: qc_ens_handle
154
155integer :: ierr
156
157if(get_allow_transpose(state_ens_handle)) then ! the forward operators were done var complete
158 !transpose back if present
159 if (present(fwd_op_ens_handle)) &
160 call all_vars_to_all_copies(fwd_op_ens_handle)
161 if (present(qc_ens_handle)) &
162 call all_vars_to_all_copies(qc_ens_handle)
163else
164 ! close mpi window
165 call mpi_win_free(state_win, ierr)
166 deallocate(contiguous_fwd)
167endif
168
169current_win = no_window
170
171end subroutine free_state_window

183 deallocate(mean_1d)
184 call end_ensemble_manager(mean_ens_handle)
185endif
186
187current_win = no_window
