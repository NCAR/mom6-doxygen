47! dummy args for the advance_state call. presumably this
48! executable was invoked by a script that was originally
49! called by advance_state() in filter, so no need to get
50! recursive here. but if you want to use this outside of
51! the dart advance, you could comment these in and add
52! them to the namelist below.
53!character (len=129) :: adv_ens_command = ''
54!integer :: async = 0
55!integer :: tasks_per_model_advance = 1
56
57!----------------------------------------------------------------
58!>@todo either use the namelist or take it out.
59
60! to enable the use of the namelist, change this to .true. and recompile.
61logical :: use_namelist = .false.
62
63! for use outside dart, or to call after dart finishes and
64! writes a restart file (without a target time), set this
65! to .false. and supply a target time with target_days, secs
66logical :: is_model_advance_file = .true.
67
68! for debugging, status
69character(len=256) :: ic_file_name = "temp_ic.nc"
70character(len=256) :: ud_file_name = "temp_ud.nc"
71logical :: trace_execution = .false.
72character(len=512) :: errstring
73
74! to overwrite the target time, set these to something >= 0
75integer :: target_days = -1, target_seconds = -1
76
77! only read in if use_namelist is .true. -- ignored otherwise.
78namelist /integrate_model_nml/ &
79 ic_file_name, ud_file_name, &
80 trace_execution, is_model_advance_file, target_days, target_seconds
81
82!----------------------------------------------------------------
83
84! This program should only be run with a single process
85call initialize_mpi_utilities('integrate_model')
86
87! This version is ok to build with MPI and run with more than a single
88! task, but only task 0 reads the input, advances the model, and writes
89! the output. All the other tasks just hang out and exit when task 0 is done.
90! FIXME: that could be changed to do multiple ens members in parallel.
91! the code does NOT do this now.
92if(task_count() > 1) &
93 call error_handler(e_msg,'integrate_model','Only one process doing the work', source)
94
95
96! this must come AFTER the standard utils are initialized.
97! Read the integrate_model_nml namelist from input.nml if 'use_namelist' true.
98if (use_namelist) then
99 call find_namelist_in_file('input.nml', 'integrate_model_nml', iunit)
100 read(iunit, nml = integrate_model_nml, iostat = rc)
101 call check_namelist_read(iunit, rc, "integrate_model_nml")
102else
103 errstring = ' !must edit integrate_model/integrate_model.f90 to enable this namelist'
104 if (do_nml_file()) write(nmlfileunit, '(A)') trim(errstring)
105 if (do_nml_term()) write( * , '(A)') trim(errstring)
106endif
107
108! Record the namelist values used for the run ...
109if (do_nml_file()) write(nmlfileunit, nml=integrate_model_nml)
110if (do_nml_term()) write( * , nml=integrate_model_nml)
111
112
113
114if (trace_execution) write(*,*) 'inside integrate_model executable'
115
116! Initialize the model class data
117call static_init_assim_model()
118model_size = get_model_size()
119
120! Initialize an ensemble manager type with a single copy
121call init_ensemble_manager(ens_handle, num_copies=1, num_vars=model_size, transpose_type_in = 2)
122
123 !------------------- Read restart from file ----------------------
124
125 call io_filenames_init(input_file_info, 1, cycling=.true., single_file=.true.)
126 call set_file_metadata(input_file_info, 1, (/ic_file_name/), 'temp_ic', 'initial condition')
127 call set_io_copy_flag( input_file_info, 1, read_copy)
128
129 ic_file_name = get_restart_filename(get_stage_metadata(input_file_info),1,1)
130 if (trace_execution) write(*,*) 'ready to open input restart file ', trim(ic_file_name)
131 call read_state(ens_handle, input_file_info, .true., target_time)
132
133 if (trace_execution) write(*,*) 'time of data, advance-to are:'
134 if (trace_execution) call print_time(ens_handle%time(1))
135 if (trace_execution) call print_time(target_time)
136
137 !----------------- Restart read in --------------------------------
138
139 ! Advance this state to the target time
140 ! If the model time is past the obs set time, just need to skip
141 if (trace_execution) write(*,*) 'calling advance_state if needed'
142
143 call io_filenames_init(output_file_info, 1, cycling=.true., single_file=.true.)
144 call set_file_metadata(output_file_info, 1, (/ud_file_name/), 'temp_ud', 'advanced member')
145 call set_io_copy_flag( output_file_info, 1, write_copy)
146
147 if(ens_handle%time(1) < target_time) &
148 call advance_state(ens_handle, 1, target_time, 0, '', 1, input_file_info, output_file_info)
149
150 ! Output the restart file if requested; Force to binary for bitwise reproducing
151 ! use in filter and perfect_model obs with shell advance options
152 ud_file_name = get_restart_filename(get_stage_metadata(output_file_info),1,1)
153 if (trace_execution) write(*,*) 'ready to open output restart file ', trim(ud_file_name)
154
155 call write_state(ens_handle, output_file_info)
156
157if (trace_execution) write(*,*) 'end of integrate_model executable'
158call finalize_mpi_utilities()
159
160end program integrate_model
