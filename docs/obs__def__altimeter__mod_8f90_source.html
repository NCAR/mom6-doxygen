79subroutine get_expected_altimeter(state_handle, ens_size, location, altimeter_setting, istatus)
80
81type(ensemble_type), intent(in) :: state_handle
82integer, intent(in) :: ens_size
83type(location_type), intent(in) :: location
84real(r8), intent(out) :: altimeter_setting(ens_size) ! altimeter (hPa)
85integer, intent(out) :: istatus(ens_size)
86
87real(r8) :: psfc(ens_size) ! surface pressure value (Pa)
88real(r8) :: hsfc(ens_size) ! surface elevation level (m above SL)
89integer :: psfc_istatus(ens_size)
90integer :: hsfc_istatus(ens_size)
91logical :: return_now
92integer :: imem
93
94real(r8), parameter :: min_realistic_altimeter = 880.0_r8 ! (hPa)
95real(r8), parameter :: max_realistic_altimeter = 1100.0_r8 ! (hPa)
96
97if ( .not. module_initialized ) call initialize_module
98
99istatus = 0 ! Need to initialize this to zero for track_status.
100altimeter_setting = missing_r8
101
102! interpolate the surface pressure to the desired location.
103! The implicit units of pressure here are Pascals.
104call interpolate(state_handle, ens_size, location, qty_surface_pressure, psfc, psfc_istatus)
105call track_status(ens_size, psfc_istatus, altimeter_setting, istatus, return_now)
106if (return_now) return
107
108! interpolate the surface elevation to the desired location
109call interpolate(state_handle, ens_size, location, qty_surface_elevation, hsfc, hsfc_istatus)
110call track_status(ens_size, hsfc_istatus, altimeter_setting, istatus, return_now)
111if (return_now) return
112
113! Compute the altimeter setting given surface pressure (hPa) and height (m), altimeter is hPa
114do imem = 1, ens_size
115 if(istatus(imem) == 0) altimeter_setting(imem) = compute_altimeter(psfc(imem) * 0.01_r8, hsfc(imem))
116enddo
117
118where (altimeter_setting < min_realistic_altimeter .or. &
119 altimeter_setting >= max_realistic_altimeter)
120 altimeter_setting = missing_r8
121 istatus = 1
122endwhere
123
124end subroutine get_expected_altimeter

128function compute_altimeter(psfc, hsfc)
129
130real(r8), parameter :: k1 = 0.190284_r8
131real(r8), parameter :: k2 = 8.4228807e-5_r8
132
133real(r8), intent(in) :: psfc ! (hPa)
134real(r8), intent(in) :: hsfc ! (m above MSL)
135
136real(r8) :: compute_altimeter ! (hPa)
137
138compute_altimeter = ((psfc - 0.3_r8) ** k1 + k2 * hsfc) ** (1.0_r8 / k1)
139
140return
141end function compute_altimeter
