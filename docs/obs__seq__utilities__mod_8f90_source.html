72!-----------------------------------------------------------------------
73!> prints out a quick table of obs types and counts, overall start and
74!> stop times, and metadata strings and counts.
75!> you can get more info by running the obs_diag program.
76!>
77!> @param seq observation sequence of interest
78!> @param filename the file that was the origin of the observation sequence.
79!> Used for messages.
80
81subroutine print_obs_seq(seq, filename)
82
83type(obs_sequence_type), intent(in) :: seq
84character(len=*), intent(in) :: filename
85
86type(obs_type) :: obs, next_obs
87type(obs_def_type) :: this_obs_def
88logical :: is_there_one, is_this_last
89integer :: size_seq
90integer :: i
91integer :: this_obs_kind
92! max_obs_kinds is a public from obs_kind_mod.f90 and really is
93! counting the max number of types, not kinds
94integer :: type_count(max_obs_kinds), identity_count
95
96
97! Initialize input obs_types
98do i = 1, max_obs_kinds
99 type_count(i) = 0
100enddo
101identity_count = 0
102
103! make sure there are obs left to process before going on.
104! num_obs should be ok since we just constructed this seq so it should
105! have no unlinked obs. if it might for some reason, use this instead:
106! size_seq = get_num_key_range(seq) !current size of seq
107
108size_seq = get_num_obs(seq)
109if (size_seq == 0) then
110 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
111 call error_handler(e_msg,'obs_sort',msgstring)
112 return
113endif
114
115! Initialize individual observation variables
116call init_obs( obs, get_num_copies(seq), get_num_qc(seq))
117call init_obs(next_obs, get_num_copies(seq), get_num_qc(seq))
118
119! blank line
120call error_handler(e_msg,'',' ')
121
122write(msgstring,*) 'Processing sequence file ', trim(filename)
123call error_handler(e_msg,'',msgstring)
124
125call print_metadata(seq, filename)
126
127! Start to process obs from seq
128
129is_there_one = get_first_obs(seq, obs)
130
131if ( .not. is_there_one ) then
132 write(msgstring,*)'no first observation in ',trim(filename)
133 call error_handler(e_msg,'obs_sort', msgstring)
134endif
135
136! process it here
137is_this_last = .false.
138
139call get_obs_def(obs, this_obs_def)
140call print_time(get_obs_def_time(this_obs_def), ' First timestamp: ')
141call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
142
143obsloop : do while ( .not. is_this_last)
144
145 call get_obs_def(obs, this_obs_def)
146 this_obs_kind = get_obs_kind(this_obs_def)
147 if (this_obs_kind < 0) then
148 identity_count = identity_count + 1
149 else
150 type_count(this_obs_kind) = type_count(this_obs_kind) + 1
151 endif
152
153! print *, 'obs kind index = ', this_obs_kind
154! if(this_obs_kind > 0)print *, 'obs name = ', get_obs_kind_name(this_obs_kind)
155
156 call get_next_obs(seq, obs, next_obs, is_this_last)
157 if (.not. is_this_last) then
158 obs = next_obs
159 else
160 call print_time(get_obs_def_time(this_obs_def), ' Last timestamp: ')
161 call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
162 endif
163
164enddo obsloop
165
166
167write(msgstring, *) 'Number of obs processed : ', size_seq
168call error_handler(e_msg, '', msgstring)
169write(msgstring, *) '---------------------------------------------------------'
170call error_handler(e_msg, '', msgstring)
171do i = 1, max_obs_kinds
172 if (type_count(i) > 0) then
173 write(msgstring, '(a32,i8,a)') trim(get_obs_kind_name(i)), &
174 type_count(i), ' obs'
175 call error_handler(e_msg, '', msgstring)
176 endif
177enddo
178if (identity_count > 0) then
179 write(msgstring, '(a32,i8,a)') 'Identity observations', &
180 identity_count, ' obs'
181 call error_handler(e_msg, '', msgstring)
182endif
183
184! another blank line
185call error_handler(e_msg, '', ' ')
186
187! Time to clean up
188
189call destroy_obs( obs)
190call destroy_obs(next_obs)
191
192end subroutine print_obs_seq

195!-----------------------------------------------------------------------
196!> ensures the observation sequence is temporally ascending when
197!> traversed as a linked list.
198!>
199!> @param seq observation sequence of interest
200!> @param filename the file that was the origin of the observation sequence.
201!> Used for messages.
202
203subroutine validate_obs_seq_time(seq, filename)
204
205! this eventually belongs in the obs_seq_mod code, but for now
206! try it out here. we just fixed a hole in the interactive create
207! routine which would silently let you create out-of-time-order
208! linked lists, which gave no errors but didn't assimilate the
209! right obs at the right time when running filter. this runs
210! through the times in the entire sequence, ensuring they are
211! monotonically increasing in time. this should help catch any
212! bad files which were created with older versions of code.
213
214type(obs_sequence_type), intent(in) :: seq
215character(len=*), intent(in) :: filename
216
217type(obs_type) :: obs, next_obs
218type(obs_def_type) :: this_obs_def
219logical :: is_there_one, is_this_last
220integer :: size_seq, obs_count
221integer :: key
222type(time_type) :: last_time, this_time
223
224
225! make sure there are obs left to process before going on.
226size_seq = get_num_obs(seq)
227if (size_seq == 0) then
228 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
229 call error_handler(e_msg,'obs_sort:validate',msgstring)
230 return
231endif
232
233! Initialize individual observation variables
234call init_obs( obs, get_num_copies(seq), get_num_qc(seq))
235call init_obs(next_obs, get_num_copies(seq), get_num_qc(seq))
236
237obs_count = 0
238
239! Start to process obs from seq
240
241is_there_one = get_first_obs(seq, obs)
242
243! we already tested for 0 obs above, so there should be a first obs here.
244if ( .not. is_there_one ) then
245 write(msgstring,*)'no first obs in sequence ' // trim(filename)
246 call error_handler(e_err,'obs_sort:validate', &
247 msgstring, source, revision, revdate)
248 return
249endif
250
251is_this_last = .false.
252last_time = set_time(0, 0)
253obsloop : do while ( .not. is_this_last)
254
255 call get_obs_def(obs, this_obs_def)
256 this_time = get_obs_def_time(this_obs_def)
257
258 if (last_time > this_time) then
259 ! bad time order of observations in linked list
260 call print_time(last_time, ' previous timestamp: ')
261 call print_date(last_time, ' calendar date: ')
262 call print_time(this_time, ' next timestamp: ')
263 call print_date(this_time, ' calendar date: ')
264
265 key = get_obs_key(obs)
266 write(msgstring1,*)'obs number ', key, ' has earlier time than previous obs'
267 write(msgstring2,*)'observations must be in increasing time order, file ' // trim(filename)
268 call error_handler(e_err,'obs_sort:validate', msgstring2, &
269 source, revision, revdate, &
270 text2=msgstring1)
271 endif
272
273 last_time = this_time
274 obs_count = obs_count + 1
275
276 call get_next_obs(seq, obs, next_obs, is_this_last)
277 if (.not. is_this_last) obs = next_obs
278
279enddo obsloop
280
281! clean up
282call destroy_obs( obs)
283call destroy_obs(next_obs)
284
285! technically not a time validation, but easy to check. obs_count should never
286! be larger than size_seq - that's a fatal error. obs_count < size_seq would
287! suggest there are obs in the file that aren't part of the linked list.
288! this does not necessarily indicate a fatal error but it's not a common
289! situation and might indicate someone should check on the file.
290if (obs_count /= size_seq) then
291 write(msgstring,*) 'input sequence ', trim(filename)
292 call error_handler(e_msg,'obs_sort:validate', msgstring)
293
294 write(msgstring,*) 'total obs in file: ', size_seq, ' obs in linked list: ', obs_count
295 if (obs_count > size_seq) then
296 ! this is a fatal error
297 write(msgstring1,*) 'linked list obs_count > total size_seq, should not happen'
298 call error_handler(e_err,'obs_sort:validate', msgstring, &
299 source, revision, revdate, &
300 text2=msgstring1)
301 else
302 ! just warning msg
303 write(msgstring1,*) 'only observations in linked list will be processed'
304 call error_handler(e_msg,'obs_sort:validate', msgstring, &
305 source, revision, revdate, text2=msgstring1)
306 endif
307endif
308
309end subroutine validate_obs_seq_time
