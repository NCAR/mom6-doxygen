7program test_corr
8
9! test of twod_gaussian distribution, and two different numerical
10! methods of computing the variance and correlation. one is more
11! stable in the face of 8 byte reals being redefined as 4 byte values.
12
13use types_mod, only : r8
14use utilities_mod, only : check_namelist_read, find_namelist_in_file, &
15 register_module, error_handler, e_err, &
16 nmlfileunit, do_nml_file, do_nml_term, &

22! version controlled file description for error handling, do not edit
23character(len=256), parameter :: source = &
24 "$URL$"
25character(len=32 ), parameter :: revision = "$Revision$"
26character(len=128), parameter :: revdate = "$Date$"

28type (random_seq_type) :: r
29real(r8), allocatable :: rnum(:, :)
30real(r8) :: c(2, 2), mean(2)
31integer :: i, j, iunit, io
32
33real(r8) :: mean_correla, vara, meana, meanc_2a, sample_correla
34real(r8) :: mean_correlb, varb, meanb, meanc_2b, sample_correlb
35
36! namelist variables
37integer :: n_pairs = 10
38integer :: n_cycles = 100
39real(r8) :: desired_correl = 0.75_r8
40
41namelist /test_corr_nml/ n_pairs, desired_correl, n_cycles
42
43
44! start of program
45
46call initialize_utilities('test_corr')
47call register_module(source, revision, revdate)
48
49! Read the namelist entry
50call find_namelist_in_file("input.nml", "test_corr_nml", iunit)
51read(iunit, nml = test_corr_nml, iostat = io)
52call check_namelist_read(iunit, io, "test_corr_nml")
53
54! Write the namelist values to the log file
55if (do_nml_file()) write(nmlfileunit, nml=test_corr_nml)
56if (do_nml_term()) write( * , nml=test_corr_nml)
57
58
59if (n_cycles < 2 .or. n_pairs < 2) then
60 call error_handler(e_err, 'test_corr', 'both n_pairs and n_cycles must be >= 2', &
61 source, revision, revdate)
62endif
63
64mean_correla = 0.0
65mean_correlb = 0.0
66vara = 0.0
67varb = 0.0
68
69! Allocate space for the pairs of data
70allocate(rnum(2, n_pairs))
71
72call init_random_seq(r)
73
74! Mean for generated
75mean = 0.0
76meanc_2a = 0.0
77meanc_2b = 0.0
78
79! Generate the covariance matrix
80c(1, 1) = 1.0
81c(2, 2) = 1.0
82c(1, 2) = desired_correl
83c(2, 1) = desired_correl
84
85do j = 1, n_cycles
86
87 do i = 1, n_pairs
88 call twod_gaussians(r, mean, c, rnum(:, i))
89 end do
90
91 ! Compute the correlation of the random numbers two different ways
92 call comp_correla(rnum, n_pairs, sample_correla)
93 call comp_correlb(rnum, n_pairs, sample_correlb)
94
95 mean_correla = mean_correla + sample_correla
96 meanc_2a = meanc_2a + sample_correla**2
97 vara = vara + (sample_correla - desired_correl) ** 2
98
99 mean_correlb = mean_correlb + sample_correlb
100 meanc_2b = meanc_2b + sample_correlb**2
101 varb = varb + (sample_correlb - desired_correl) ** 2
102end do
103
104mean_correla = mean_correla / n_cycles
105mean_correlb = mean_correlb / n_cycles
106
107write(*, *) ''
108write(*, *) 'target mean correl is ', desired_correl
109write(*, *) 'computed mean correl, version A, is ', mean_correla
110write(*, *) 'computed mean correl, version B, is ', mean_correlb
111
112write(*, *) ''
113write(*, *) 'predicted variance is ', (1.0_r8 - desired_correl**2)**2 / (n_pairs - 1._r8)
114write(*, *) 'computed variance, version A, is ', (meanc_2a - n_cycles * mean_correla**2) / (n_cycles - 1._r8)
115write(*, *) 'computed variance, version B, is ', (meanc_2b - n_cycles * mean_correlb**2) / (n_cycles - 1._r8)
116
117call finalize_utilities('test_corr')
118

123subroutine comp_correla(ens, n, correl)
124
125integer, intent(in) :: n
126real(r8), intent(in) :: ens(2, n)
127real(r8), intent(out) :: correl
128
129real(r8) :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
130
131sum_x = sum(ens(2, :))
132sum_y = sum(ens(1, :))
133sum_xy = sum(ens(2, :) * ens(1, :))
134sum_x2 = sum(ens(2, :) * ens(2, :))
135
136! Computation of correlation
137sum_y2 = sum(ens(1, :) * ens(1, :))
138
139! debug:
140!print *, 'x,y,xy,x2,y2: ', sum_x, sum_y, sum_xy, sum_x2, sum_y2
141
142correl = (n * sum_xy - sum_x * sum_y) / &
143 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
144
145end subroutine comp_correla
146

149subroutine comp_correlb(ens, n, correl)
150
151! alternative way to comput covariance and variance - more numerically
152! stable in the face of 4 byte floats:
153
154integer, intent(in) :: n
155real(r8), intent(in) :: ens(2, n)
156real(r8), intent(out) :: correl
157
158real(r8) :: sum_x, sum_y
159real(r8) :: x_mean, y_mean, x_var, y_var, cov
160
161sum_x = sum(ens(2, :))
162sum_y = sum(ens(1, :))
163
164x_mean = sum_x / n
165y_mean = sum_y / n
166x_var = sum((ens(2, :) - x_mean)**2) / (n - 1)
167y_var = sum((ens(1, :) - y_mean)**2) / (n - 1)
168cov = sum((ens(2, :) - x_mean) * (ens(1, :) - y_mean)) / (n - 1)
169correl = cov / sqrt(x_var * y_var)
170
171! debug:
172!print *, x_mean, y_mean, x_var, y_var, cov, abs(x_var * y_var), sqrt(x_var*y_var), correl
173
174end subroutine comp_correlb
175

178end program test_corr
