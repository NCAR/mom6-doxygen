36subroutine dart_obs_seq (datestring, &
37 nobs_conv, nobs_oz, nobs_sat, nobstot, &
38 ens_size, obs_seq_out_filename, nobs_start, nobs_end)
39 implicit none
40
41 character(len=10), intent(in) :: datestring
42 integer, intent(in) :: nobs_conv, nobs_oz, nobs_sat, nobstot
43 integer, intent(in) :: ens_size
44 character(len = 129), intent(in) :: obs_seq_out_filename
45 integer, intent(in) :: nobs_start, nobs_end
46
47 logical :: is_conv, is_sat
48
49 integer :: num_obs, num_obs_last, num_copies, num_qc, max_num_obs=8000000
50 type(obs_sequence_type) :: seq
51 type(obs_type) :: obs, prev_obs
52 type(obs_def_type) :: obs_def
53 type(location_type) :: location
54 type(time_type) :: time_obs, prev_time
55 integer :: i, k, n, which_vert, ie, obskind, obs_quantity
56 character(len = 129) :: copy_meta_data, qc_meta_data
57 logical :: first_obs
58 integer :: year, month, day, hour, days, seconds
59 integer :: gps_key = 0
60 real(r_kind) :: gnx, gny, gnz, ds, htop, rfict
61 real(r_kind) :: lat, lon, vloc, obsv, oerr
62 real(r_kind) :: this_mean, this_stdv, this_variance, test_val
63 real(r_kind) :: qc_val(2)
64 real(r_kind) :: missing_val = -888888.000
65 real(r_kind),allocatable :: ens_copy(:)
66
67 logical :: write_external
68 logical :: has_external = .true.
69
70 ! Number of obs to process
71 num_obs = nobs_end - nobs_start + 1
72
73! add back the unbiascorrected mean that was extracted in enkf
74 do i = 1, ens_size
75 anal_ob(i,:) = anal_ob(i,:) + ensmean_obnobc(:)
76! anal_ob_chunk(i,:) = anal_ob_chunk(i,:) + ensmean_obnobc(nobs_start:nobs_end) ! CSS for scatterv
77 end do
78
79 ! Figure out the number of copies/metadata to output
80 if ( write_prior_copies ) then
81 num_copies = 5 + 2*ens_size! Obs,prior/post mean, prior/post spread, plus ensemble of priors and posteriors
82 allocate(ens_copy(num_copies))
83 num_qc = 2 ! Number of QC--1 for input, the 2nd for DART QC flag, which is bogus, but needed so other DART programs can work
84 else
85 num_copies = 1 ! Just 1 true copy for the observation. Priors are metadata, NOT copies
86 allocate(ens_copy(ens_size+2)) ! Ensemble of priors, plus obs, plus a mean value
87 num_qc = 1 ! Number of QC--just 1, coming from GSI
88 endif
89
90 ! Initialize some values
91 ens_copy(:) = missing_val
92
93! Initialize an obs_sequence structure
94 call init_obs_sequence(seq, num_copies, num_qc, num_obs) ! Last entry can be > num_obs (e.g., max_num_obs) but that wastes memory
95
96! Set QC meta data
97 qc_meta_data = 'GSI Quality Control'
98 call set_qc_meta_data(seq, 1, qc_meta_data)
99
100! Set copy meta data
101 copy_meta_data = 'observation'
102 call set_copy_meta_data(seq, 1, copy_meta_data)
103
104 ! Set more copy and QC meta data if write_prior_copies
105 if ( write_prior_copies ) then
106 copy_meta_data = 'prior ensemble mean' ; call set_copy_meta_data(seq, 2, copy_meta_data)
107 copy_meta_data = 'posterior ensemble mean' ; call set_copy_meta_data(seq, 3, copy_meta_data)
108 copy_meta_data = 'prior ensemble spread' ; call set_copy_meta_data(seq, 4, copy_meta_data)
109 copy_meta_data = 'posterior ensemble spread' ; call set_copy_meta_data(seq, 5, copy_meta_data)
110 do ie=1,ens_size
111 write(copy_meta_data,'(a21,1x,i6)') 'prior ensemble member', ie
112 call set_copy_meta_data(seq,5+ie*2-1,copy_meta_data)
113 write(copy_meta_data,'(a25,1x,i6)') 'posterior ensemble member', ie
114 call set_copy_meta_data(seq,5+ie*2,copy_meta_data)
115 enddo
116 ! Additional QC metadata
117 qc_meta_data = 'DART quality control'
118 call set_qc_meta_data(seq, 2, qc_meta_data)
119 endif
120
121! initialize the obs variable
122 call init_obs(obs, num_copies, num_qc)
123 call init_obs(prev_obs, num_copies, num_qc)
124
125! set observation time type
126 call set_calendar_type(gregorian)
127
128 first_obs = .true.
129 obsloop: do i = nobs_start, nobs_end
130
131 ! Initialize QC values inside the loop, because we might modify these values later if modify_dart_qc_flag_for_big_ob_error = true
132 qc_val(1) = 1.0 ! For Input GSI QC
133 qc_val(2) = 0.0 ! For DART QC, not used if write_prior_copies == .false.
134
135 ! Find out if the ob is a conventional ob or radiance
136 if ( i > nobs_conv + nobs_oz ) then
137 is_sat = .true. ; is_conv = .false.
138 if ( .not. convert_sat ) cycle obsloop
139 else
140 is_sat = .false. ; is_conv = .true.
141 if ( .not. convert_conv ) cycle obsloop
142 endif
143
144 ! here, obskind is really obs_type, obs_quantity is the generic quantity
145 if ( is_conv ) then
146 call prepbufr_to_dart_obs_kind(obtype(i),stattype(i),obskind,which_vert,obs_quantity)
147 else if ( is_sat .and. convert_sat ) then
148 call radiance_to_dart_obs_kind(obtype(i),nuchan(indxsat(i-nobs_conv-nobs_oz)),obskind,which_vert,obs_quantity) ! indxsat has size "nobs_sat"
149 end if
150
151 ! skip if unsupported by prepbufr_to_dart_obs_kind or radiance_to_dart_obs_kind
152 if ( obskind < 0 ) cycle obsloop
153
154 ! time info
155 read(datestring, '(i4,3i2)') year, month, day, hour
156 if ( obtime(i) >= 0.0 ) then
157 call get_time(increment_time(set_date(year, month, day, hour), &
158 nint(obtime(i))*3600, 0), &
159 seconds, days)
160 else if ( obtime(i) < 0.0 ) then
161 call get_time(decrement_time(set_date(year, month, day, hour), &
162 nint(abs(obtime(i)))*3600, 0), &
163 seconds, days)
164 end if
165 time_obs = set_time(seconds, days)
166
167 if ( lie_about_ob_times ) then
168 !To make DART assimilate observations in order obs are provided, rather
169 !than in time-order, need to lie about the observation times.
170 ! When writing out more than 1 obs_seq file, they need to be combined in proper order, so
171 ! make obs time dependent on which processor we're on ("nproc")
172 call get_time(increment_time(set_date(year, month, day, hour), &
173 nproc, 0), &
174 seconds, days) !
175 time_obs = set_time(seconds, days)
176 endif
177
178 !transfer to r_kind (r8) variables
179 vloc = obpress(i)*100.0 !Convert to Pa
180 lon = obloclon(i)
181 lat = obloclat(i)
182 oerr = oberrvar(i) ! Already ob error variances
183
184 ens_copy(1) = ob(i) ! Observation value
185
186 !Recenter radiance or surface pressure prior perturbations (which don't
187 !include bias-corrections) about the ensemble mean
188 !prior H(x_bar), which does include BC.
189 ! Really only need to do for obs with bias-correction, like surface
190 ! pressure or radiance, but doesn't hurt to do for all obs
191 if ( recenter_about_mean_prior ) then ! Default is true
192 anal_ob(1:ens_size,i) = anal_ob(1:ens_size,i) + (ensmean_ob(i)-ensmean_obnobc(i))
193 !anal_ob_chunk(1:ens_size,i) = anal_ob_chunk(1:ens_size,i) + (ensmean_ob(i)-ensmean_obnobc(i)) ! CSS scatterv
194 endif
195
196 ! modify observations and observation errors if necessary
197 if( obs_quantity == qty_surface_pressure) then
198 ens_copy(1) = ens_copy(1)*100.0 ! Get into Pa
199 oerr = oerr * 10000.0 ! Get into Pa^2--this is variance
200 anal_ob(1:ens_size,i) = anal_ob(1:ens_size,i) * 100. ! Get into Pa
201 ensmean_ob(i) = ensmean_ob(i) * 100. ! Get into Pa
202 !anal_ob_chunk(1:ens_size,i) = anal_ob_chunk(1:ens_size,i) * 100. ! CSS scatterv
203 endif
204
205 ! Fill array with priors
206 if ( write_prior_copies ) then
207 ens_copy(6:num_copies:2) = anal_ob(1:ens_size,i) ! Fill the prior elements
208! ens_copy(6:num_copies:2) = anal_ob_chunk(1:ens_size,i) ! CSS scatterv
209 call compute_mean_std(ens_size,ens_copy(6:num_copies:2),ens_copy(2),ens_copy(4)) ! Calculate prior mean/stddev
210 else
211 ens_copy(2:ens_size+1) = anal_ob(1:ens_size,i)
212! ens_copy(2:ens_size+1) = anal_ob_chunk(1:ens_size,i) ! CSS scatterv
213 ens_copy(ens_size+2) = ensmean_ob(i)
214 endif
215
216 ! modify QC flag if really big observation error
217 ! note that obs_sprd_prior(i) also has spread information but it is safer to compute the spread
218 ! from exactly what is going into the obs_seq file
219 modify_dart_qc_flag_for_big_ob_error = .false. ! for now this doesn't work properly. so force to false.
220 if ( modify_dart_qc_flag_for_big_ob_error ) then
221 if ( write_prior_copies ) then
222 call compute_mean_std(ens_size,ens_copy(6:num_copies:2),this_mean,this_stdv) ! Calculate prior mean/stddev
223 else
224 call compute_mean_std(ens_size,ens_copy(2:ens_size+1),this_mean,this_stdv)
225 endif
226 this_variance = this_stdv * this_stdv ! get into variance
227 test_val = variance_coef * this_variance
228 if ( oerr .gt. test_val ) then
229 ! print *, sqrt(oerr), this_stdv, vloc*0.01
230 qc_val(1) = 10.0
231 if ( write_prior_copies ) qc_val(2) = 7.0 ! tell DART/obs_diag this observation failed the outlier check so obs_diag won't process
232 endif
233 endif
234
235 location = set_location(lon, lat, vloc, which_vert)
236
237 call set_obs_def_location(obs_def, location)
238 call set_obs_def_type_of_obs(obs_def, obskind)
239 call set_obs_def_time(obs_def, time_obs)
240 if ( obskind == gpsro_refractivity ) then
241 gnx = 0.0
242 gny = 0.0
243 gnz = 0.0
244 ds = 0.0
245 htop = 0.0
246 rfict = 0.0
247 call set_gpsro_ref(gps_key, gnx, gny, gnz, rfict, ds, htop, 'GPSREF')
248 call set_obs_def_key(obs_def, gps_key)
249 end if
250
251 ! Metadata stuff
252 if ( .not. write_prior_copies ) then ! No need for metadata if we're writing prior copies
253 write_external = write_this_ob_type_external_fo(obskind)
254 call set_obs_def_external_fo(obs_def, has_external, write_external,i,ens_size,real(ens_copy(2:ens_size+2),r_kind)) ! CSS sets obs_def%has_exteral_FO
255 !call set_obs_def_external_FO(obs_def, has_external, write_external,i,ens_size,real(anal_ob_chunk(1:ens_size,i),r_kind)) ! CSS scatterv
256 endif
257
258 ! If not using precomputed obs, GPSRO_REFRACTIVITY must have height
259 ! coordinate, but that is unavailable from GSI diag files. So don't
260 ! include GPSRO obs in the obs_sequence file--make user get it from
261 ! elsewhere, I guess.
262 !if ( .not. write_external ) then
263 ! if ( obskind == GPSRO_REFRACTIVITY ) cycle obsloop
264 !endif
265
266 call set_obs_def_error_variance(obs_def, oerr) ! oerr is the ob error variance
267 !call set_obs_def_key(obs_def, key)
268 call set_obs_def(obs, obs_def)
269 call set_obs_values(obs, ens_copy(1:num_copies))
270 call set_qc(obs, qc_val(1:num_qc))
271 call add_obs_to_seq(seq, obs, time_obs, prev_obs, prev_time, first_obs)
272
273 end do obsloop
274
275 call write_obs_seq(seq, trim(adjustl(obs_seq_out_filename)) )
276
277 call destroy_obs_def(obs_def) ! CSS added, is this needed?
278 call destroy_obs(obs)
279 call destroy_obs_sequence(seq)
280
281 if (allocated(ens_copy)) deallocate(ens_copy)
282
283end subroutine dart_obs_seq

295subroutine prepbufr_to_dart_obs_kind (obtype, obstype, obs_kind, which_vert, obs_quantity)
296
297! based on DART/observations/NCEP/ascii_to_obs/real_obs_mod.f90
298
299 implicit none
300 character(len=20), intent(in) :: obtype !'ps','u','v','oz','t','amsua_n15', etc.
301 integer, intent(in) :: obstype !prepbufr report type
302 integer, intent(out) :: obs_kind
303 integer, intent(out) :: which_vert
304 integer, intent(out) :: obs_quantity
305
306! assign each observation the correct observation type
307!------------------------------------------------------------------------------
308
309 ! make sure we do not fall through the code below without setting
310 ! a valid obs kind (e.g. the obstype is one not listed)
311 obs_kind = -1
312 obs_quantity = -1
313
314 if(obtype(1:3) == 'gps') then
315 obs_quantity = qty_gpsro
316 obs_kind = gpsro_refractivity
317 endif
318
319 if(obtype(1:3) == ' pw') then
320 obs_quantity = qty_precipitable_water
321 if(obstype == 153 ) obs_kind = gps_precipitable_water
322 endif
323
324 if(obtype(1:3) == ' t') then
325 obs_quantity = qty_temperature
326 if(obstype == 120 .or. obstype == 132) obs_kind = radiosonde_temperature
327 if(obstype == 130 .or. obstype == 131) obs_kind = aircraft_temperature
328 if(obstype == 133 .or. obstype == 134) obs_kind = acars_temperature ! CSS HRRRE calls this AMDAR_TEMPERATURE
329 if(obstype == 135 ) obs_kind = amdar_temperature
330 if(obstype == 161 .or. obstype == 163) obs_kind = atov_temperature
331 if(obstype == 171 .or. obstype == 173) obs_kind = atov_temperature
332 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_temperature
333 if(obstype == 181 .or. obstype == 183 .or. obstype == 187 ) obs_kind = land_sfc_temperature
334 if(obstype == 188 ) obs_kind = mesonet_temperature
335 endif
336
337 if(obtype(1:3) == ' q') then
338 obs_quantity = qty_specific_humidity
339 if(obstype == 120 .or. obstype == 132) obs_kind = radiosonde_relative_humidity ! RADIOSONDE_SPECIFIC_HUMIDITY
340 if(obstype == 130 .or. obstype == 131) obs_kind = aircraft_relative_humidity ! AIRCRAFT_SPECIFIC_HUMIDITY
341 if(obstype == 133 .or. obstype == 134) obs_kind = acars_relative_humidity ! ACARS_SPECIFIC_HUMIDITY ! Really should be AMDAR per HRRRE changes, but that requires some obs_def/preprocess changes.
342 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_relative_humidity ! MARINE_SFC_SPECIFIC_HUMIDITY
343 if(obstype == 181 .or. obstype == 183 .or. obstype == 187 ) obs_kind = land_sfc_relative_humidity ! LAND_SFC_SPECIFIC_HUMIDITY
344 if(obstype == 188 ) obs_kind = mesonet_relative_humidity
345 endif
346
347 if(obtype(1:3) == ' ps') then
348 obs_quantity = qty_surface_pressure
349 ! what to use: PRESSURE or ALTIMETER ?
350 if(obstype == 120 ) obs_kind = radiosonde_surface_pressure ! RADIOSONDE_SURFACE_ALTIMETER
351 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_pressure ! MARINE_SFC_ALTIMETER
352 if(obstype == 181 .or. obstype == 187) obs_kind = land_sfc_pressure ! LAND_SFC_ALTIMETER
353 if(obstype == 188 ) obs_kind = mesonet_surface_pressure
354 endif
355
356 if(obtype(1:3) == ' u') then
357 obs_quantity = qty_u_wind_component
358 if(obstype == 220 .or. obstype == 232) obs_kind = radiosonde_u_wind_component
359 if(obstype == 221 ) obs_kind = radiosonde_u_wind_component
360 if(obstype == 223 .or. obstype == 227) obs_kind = profiler_u_wind_component
361 if(obstype == 224 ) obs_kind = vadwnd_u_wind_component ! CSS added new type ! PILOT_U_WIND_COMPONENT !VADWND actually
362 if(obstype == 229 ) obs_kind = pilot_u_wind_component
363 if(obstype == 230 .or. obstype == 231) obs_kind = aircraft_u_wind_component
364 if(obstype == 233 .or. obstype == 234) obs_kind = acars_u_wind_component
365 if(obstype == 235 ) obs_kind = amdar_u_wind_component
366 if(obstype == 280 .or. obstype == 282) obs_kind = marine_sfc_u_wind_component
367 if(obstype == 281 .or. obstype == 284 .or. obstype == 287 ) obs_kind = land_sfc_u_wind_component
368 if(obstype == 288 ) obs_kind = mesonet_u_wind_component
369 if(obstype >= 240 .and. obstype <= 259) obs_kind = sat_u_wind_component
370 ! 285 is QSCAT
371 if(obstype == 285 ) obs_kind = qkswnd_u_wind_component
372 ! 289 is WINDSAT, 290 is ASCAT, 291 is OSCAT
373 if(obstype >= 289 .and. obstype <= 291) obs_kind = qkswnd_u_wind_component
374 endif
375
376 if(obtype(1:3) == ' v') then
377 obs_quantity = qty_v_wind_component
378 if(obstype == 220 .or. obstype == 232) obs_kind = radiosonde_v_wind_component
379 if(obstype == 221 ) obs_kind = radiosonde_v_wind_component
380 if(obstype == 223 .or. obstype == 227) obs_kind = profiler_v_wind_component
381 if(obstype == 224 ) obs_kind = vadwnd_v_wind_component ! CSS added new type !PILOT_V_WIND_COMPONENT !VADWND actually
382 if(obstype == 229 ) obs_kind = pilot_v_wind_component
383 if(obstype == 230 .or. obstype == 231) obs_kind = aircraft_v_wind_component
384 if(obstype == 233 .or. obstype == 234) obs_kind = acars_v_wind_component
385 if(obstype == 235 ) obs_kind = amdar_v_wind_component
386 if(obstype == 280 .or. obstype == 282) obs_kind = marine_sfc_v_wind_component
387 if(obstype == 281 .or. obstype == 284 .or. obstype == 287 ) obs_kind = land_sfc_v_wind_component
388 if(obstype == 288 ) obs_kind = mesonet_v_wind_component ! CSS added
389 if(obstype >= 240 .and. obstype <= 259) obs_kind = sat_v_wind_component
390 ! 285 is QSCAT
391 if(obstype == 285 ) obs_kind = qkswnd_v_wind_component
392 ! 289 is WINDSAT, 290 is ASCAT, 291 is OSCAT
393 if(obstype >= 289 .and. obstype <= 291) obs_kind = qkswnd_v_wind_component
394 endif
395
396 if (debug .and. obs_kind < 0) then
397 ! the "real" fix if the record type is not found might actually be to
398 ! accept all record types within valid ranges, and depend on the first
399 ! preprocessing steps (in the prepbufr converter) to remove obs record
400 ! types which are not desired. for now, avoid giving them the wrong type
401 ! and quietly loop. (the looping happens just after this function returns)
402 write(string1,*) 'unsupported obtype,report type combination: ', obtype, obstype
403 call error_handler(e_msg,'prepbufr_to_dart_obs_kind',string1,'dart_obs_seq_mod.f90')
404 endif
405
406 which_vert = vertispressure
407! if ( obstype >= 180 .and. obstype <= 191 ) which_vert = VERTISSURFACE ! CSS maybe keep as pressure
408! if ( obstype >= 280 .and. obstype <= 291 ) which_vert = VERTISSURFACE
409end subroutine prepbufr_to_dart_obs_kind

411subroutine radiance_to_dart_obs_kind(obtype, channel, obs_kind, which_vert, obs_quantity)
412 character(len=*), intent(in) :: obtype !'ps','u','v','oz','t','amsua_n15', etc.
413 integer, intent(in) :: channel ! radiance channel
414 integer, intent(out) :: obs_kind, which_vert
415 integer, intent(out) :: obs_quantity
416
417 character(len=256) :: str_channel, this_string
418
419 ! Make a string containing satellite,sensor,and channel to match obs_def_radiance_mod.f90 entries
420 write(str_channel,fmt='(i8)') channel
421 this_string = trim(adjustl(obtype))//'_ch'//trim(adjustl(str_channel))
422
423 ! Make uppercase to match the case in obs_def_radiance_mod.f90
424 call to_upper(this_string)
425
426 ! Bad things will happen in DART preprocess "obs_def" files if there are hyphens
427 ! in names. Replace with underscores
428 call replace_hyphen(this_string)
429
430 ! Be careful about upper/lower case. Make sure this matches the obs_def
431 obs_kind = get_index_for_type_of_obs(this_string) ! from obs_kind_mod
432 obs_quantity = qty_temperature ! QTY_RADIANCE
433 which_vert = vertispressure
434
435end subroutine radiance_to_dart_obs_kind

437subroutine replace_hyphen(string)
438 character(*), intent(inout) :: string

440 do j = 1,len_trim(string)
441 if ( string(j:j) == '-') string(j:j) = '_'
442 enddo
443end subroutine replace_hyphen

445function write_this_ob_type_external_fo(ob_type)
446 integer, intent(in) :: ob_type
447 logical :: write_this_ob_type_external_fo
448
449 integer :: i
450 logical :: is_all
451 character(len=obstypelength) :: ob_type_string
452
453 ! Initialize to false
454 write_this_ob_type_external_fo = .false.
455 is_all = .false.
456
457 ! Get the ob_type string corresponding to the ob_type integer
458 ob_type_string = get_name_for_type_of_obs(ob_type) ! from obs_kind_mod
459
460 ! Determine whether we want to write out this ob type
461 ! ntypes_to_compute_FO_max,write_FO_for_these_obs_types from params
462 do i=1,ntypes_to_compute_fo_max
463 if( trim(adjustl(write_fo_for_these_obs_types(i))) == trim(adjustl(ob_type_string)) .or. &
464 trim(adjustl(write_fo_for_these_obs_types(i))) == 'all' ) then ! If any element is 'all', it gets output
465 write_this_ob_type_external_fo = .true.
466 is_all = trim(adjustl(write_fo_for_these_obs_types(i))) == 'all'
467 endif
468 enddo
469
470 if ( is_all ) then
471 do i=1,ntypes_to_compute_fo_max
472 if( trim(adjustl(exclude_these_obs_types(i))) == trim(adjustl(ob_type_string)) ) then
473 write_this_ob_type_external_fo = .false.
474 endif
475 enddo
476 endif
477
478end function write_this_ob_type_external_fo
