7program create_obs_radar_sequence
8
9! This program creates a group of observations as would be returned from a
10! radar observation system. The radar location, sweep angle, elevations,
11! etc are user inputs. The output are observation locations and types
12! without time information or data. Either radial velocity, reflectivity,
13! or both types of observations can be generated.
14!
15! To add time, run 'create_fixed_network_seq'.
16! To add data, run 'perfect_model_obs'.
17!
18! This program creates an observation at each possible location, whether there
19! is reflectivity there or not. To cull clear-air observations, some other
20! tool would have to be run after 'perfect_model_obs' has added observation
21! data values.
22!
23! If more complex timestepping than 'create_fixed_network_seq' can add is
24! wanted, look in the code below for the string 'TIME:'. There are comments
25! there about how to add this function. It will require adding new code
26! into this program and recompiling it.
27
28

108function simulate_radar_sequence()
109
110! Main routine that loops over radar locations, generating a series of
111! observation locations based on user input.
112
113type(obs_sequence_type) :: simulate_radar_sequence
114type(obs_type) :: obs
115type(obs_def_type) :: obs_def
116type(location_type) :: radar_location, obs_location
117
118real(r8) :: first_az, last_az, delta_az, beam_az
119real(r8) :: first_gate, last_gate, delta_gate, beam_gate
120real(r8) :: beam_elev, ceiling
121real(r8), allocatable :: elev(:), azim(:), gate(:)
122real(r8) :: radial_vel_variance, reflectivity_variance
123
124integer :: i_elev, num_elev, i_az, num_az
125integer :: i_gate, num_gate, i_radar, num_radar
126integer :: i, max_num_obs, select_obstype
127integer :: num_copies, num_qc, defkey, elev_type, days, secs
128integer :: which_vert, rc
129logical :: qc
130
131
132! This program only generates the observation locations, sets the variance,
133! and sets the obs types. It adds no quality control or data values, by design,
134! since separate programs already exist to add them. (See top of this file for
135! more info.)
136num_copies = 0
137num_qc = 0
138
139! TIME: This program does not attach any time information to the observations.
140! It sets all times to (0,0) and depends on a subsequent program (generally
141! 'create_fixed_network_seq') to set the calendar date/times on the obs.
142! However if more complex time stepping (e.g. by pairs of observations if
143! generating both types, or by scan elevation, or all obs for one cone at the
144! same time) then set the days and seconds as desired someplace in this code
145! inside the appropriate loops but before the two calls to fillin_obsdef().
146! The code will also probably need to prompt the user for initial and final
147! times, or delta times, and probably for the actual calendar day of the
148! observation, since at the time perfect_model_obs is run, real date
149! information is required. This will require setting a default calendar type.
150days = 0
151secs = 0
152
153! Max number of obs to allocate space for, total of all obs.
154write(*, *) 'Input upper bound on number of observations in sequence (e.g. 1000000)'
155read(*, *) max_num_obs
156
157! Initialize an obs_sequence structure
158call init_obs_sequence(simulate_radar_sequence, num_copies, num_qc, max_num_obs)
159
160! Initialize an obs variable
161call init_obs(obs, num_copies, num_qc)
162
163! The following data applies to all radar observations generated.
164! The actual location of the radar will be the only difference between
165! each separate group of observations.
166write(*, *) 'The following information will apply to all radar sweeps'
167write(*, *) 'generated. Multiple radar locations can be specified but they'
168write(*, *) 'all share the other characteristics.'
169
170! Ask user which of the two (or both) observation types to generate at
171! each location as the radar sweeps around.
172select_obstype = 0
173do while (select_obstype < 1 .or. select_obstype > 3)
174 write(*,*) 'What type of observations do you want to generate?'
175 write(*,*) ' 1=Doppler radial velocity, 2=reflectivity, 3=both'
176 read(*,*) select_obstype
177end do
178
179! beam elevations
180elev_type = -1
181do while (elev_type < 1 .or. elev_type > 3)
182 write(*, *) 'Input 1 for 9 pre-defined WSR-88D elevations (clear mode)'
183 write(*, *) 'Input 2 for 14 pre-defined WSR-88D elevations (storm mode)'
184 write(*, *) 'Input 3 to choose number and elevation angles by hand'
185 read(*, *) elev_type
186end do
187
188! figure out how many
189selectcase(elev_type)
190 case(1)
191 num_elev = n_elev_clear
192 case(2)
193 num_elev = n_elev_storm
194 case(3)
195 num_elev = 0
196 do while(num_elev < 1)
197 write(*, *) 'Input number of elevations'
198 read(*, *) num_elev
199 end do
200end select
201
202! make space
203allocate(elev(num_elev), stat=rc)
204if (rc /= 0) then
205 call error_handler(e_err,'simulate_radar_sequence', &
206 'Allocation error on elev() array', &
207 source, revision, revdate)
208endif
209
210! fill in values
211selectcase(elev_type)
212 case(1)
213 elev(:) = elev_clear(:)
214 case(2)
215 elev(:) = elev_storm(:)
216 case(3)
217 do i = 1, num_elev
218 write(*, "(a,i3)") 'Input elevation angle (degrees) # ',i
219 read(*, *) elev(i)
220 end do
221end select
222
223! convert to radians
224elev = deg2rad*elev
225
226! beam azimuth limits
227write(*, *) 'Input first azimuth angle (degrees): example value = 0.'
228read(*, *) first_az
229first_az = deg2rad*first_az
230
231write(*, *) 'Input last azimuth angle (degrees): example value = 359.'
232read(*, *) last_az
233last_az = deg2rad*last_az
234
235write(*, *) 'Input azimuth angle increment (degrees): example value = 1.'
236read(*, *) delta_az
237delta_az = deg2rad*delta_az
238
239num_az = int((last_az - first_az)/delta_az) + 1
240
241allocate(azim(num_az), stat=rc)
242if (rc /= 0) call error_handler(e_err,'simulate_radar_sequence', &
243 'Allocation error on azim() array', &
244 source, revision, revdate)
245do i_az = 1, num_az
246 azim(i_az) = first_az + (i_az-1)*delta_az
247enddo
248
249! beam gate limits
250write(*, *) 'Input closest gate (m): example value = 3000.'
251read(*, *) first_gate
252
253write(*, *) 'Input farthest gate (m): example value = 100000.'
254read(*, *) last_gate
255
256write(*, *) 'Input gate length (m): example value = 1000.'
257read(*, *) delta_gate
258
259num_gate = int((last_gate - first_gate)/delta_gate) + 1
260
261allocate(gate(num_gate), stat=rc)
262if (rc /= 0) call error_handler(e_err,'simulate_radar_sequence', &
263 'Allocation error on gate() array', &
264 source, revision, revdate)
265do i_gate = 1, num_gate
266 gate(i_gate) = first_gate + (i_gate-1)*delta_gate
267enddo
268
269! do not generate observations above this value even if beam sweeps
270! up higher than this value.
271write(*, *) 'Input highest allowed observation height (m): example value 8500.'
272read(*, *) ceiling
273
274num_gate = int((last_gate - first_gate)/delta_gate) + 1
275
276! error variances, per obs type
277if (select_obstype == 1 .or. select_obstype == 3) then
278 write(*, *) 'Input error variance for Doppler velocity (m/s)^2: example value = 4.'
279 read(*, *) radial_vel_variance
280endif
281
282if (select_obstype == 2 .or. select_obstype == 3) then
283 write(*, *) 'Input error variance for reflectivity (dBZ)^2: example value = 4.'
284 read(*, *) reflectivity_variance
285endif
286
287
288! All radar locations share the above values
289write(*,*) 'How many radar locations do you want to specify?'
290read(*,*) num_radar
291
292write(*, *) 'When setting the location of the radar, the vertical type'
293write(*, *) '_must_ be specified in height (meters).'
294
295radars : do i_radar = 1, num_radar
296 write(*, *) ''
297 write(*, "(a,i3)") 'Input radar location for radar number ', i_radar
298
299 call interactive_location(radar_location)
300
301 which_vert = nint(query_location(radar_location,'which_vert'))
302 if(which_vert /= vertisheight) then
303 call error_handler(e_err,'simulate_radar_sequence', &
304 'Vertical coordinate of the radar location must be height', &
305 source, revision, revdate)
306 endif
307
308
309 elevs : do i_elev = 1, num_elev
310 beam_elev = elev(i_elev)
311 azimuths : do i_az = 1, num_az
312 beam_az = azim(i_az)
313 gates : do i_gate = 1, num_gate
314 beam_gate = gate(i_gate)
315
316 ! Compute the 3d location of this observation
317 call compute_location(radar_location, &
318 beam_gate, beam_az, beam_elev, &
319 ceiling, select_obstype, defkey, &
320 obs_location, qc)
321
322 ! Bad quality control; most likely above ceiling setting.
323 if(.not. qc) cycle gates
324
325 ! Generate a radial velocity obs
326 if (select_obstype == 1 .or. select_obstype == 3) then
327 call fillin_obsdef(doppler_radial_velocity, obs_location, &
328 days, secs, radial_vel_variance, &
329 defkey, obs_def)
330
331 call set_obs_def(obs, obs_def)
332
333 call append_obs_to_seq(simulate_radar_sequence, obs)
334
335 endif
336 ! Generate a reflectivity obs
337 if (select_obstype == 2 .or. select_obstype == 3) then
338 call fillin_obsdef(radar_reflectivity, obs_location, &
339 days, secs, reflectivity_variance, &
340 0, obs_def)
341
342 call set_obs_def(obs, obs_def)
343
344 call append_obs_to_seq(simulate_radar_sequence, obs)
345
346 endif
347
348 end do gates
349 end do azimuths
350 end do elevs
351end do radars
352
353deallocate(gate, azim, elev)
354call destroy_obs(obs)
355
356end function simulate_radar_sequence

362subroutine compute_location(radar_location, beam_gate, beam_az, beam_elev, &
363 ceiling, select_obstype, defkey, obs_location, qc)
364
365! Compute an actual observation location, given all the input arguments.
366
367type(location_type), intent(in) :: radar_location
368real(r8), intent(in) :: beam_gate, beam_az, beam_elev, ceiling
369integer, intent(in) :: select_obstype
370integer, intent(inout) :: defkey
371type(location_type), intent(out) :: obs_location
372logical, intent(out) :: qc ! quality control: 0=good
373
374real(r8) :: h, spath, x, y, radar_lon, radar_lat, obs_lon, obs_lat, vloc
375real(r8) :: nyquist_velocity, beam_direction(3), elev_obs
376real(r8) :: effect_radius_m, actual_radius_m
377integer :: which_vert
378
379
380! Doviak & Zrnic, 1993: Doppler radar and weather observations, eq. 2.28b-c
381
382! Effective earth radius (4/3 times actual radius), converted to meters.
383effect_radius_m = 4.0_r8 * earth_radius / 3.0_r8 * 1000.0_r8
384
385h = sqrt(beam_gate*beam_gate + effect_radius_m*effect_radius_m &
386 + 2.0_r8*beam_gate*effect_radius_m*sin(beam_elev)) - effect_radius_m
387spath = effect_radius_m * asin(beam_gate*cos(beam_elev)/(effect_radius_m + h))
388
389x = spath*sin(beam_az)
390y = spath*cos(beam_az)
391
392! The query routines return locations in radians.
393radar_lon = query_location(radar_location, 'lon')
394radar_lat = query_location(radar_location, 'lat')
395
396! Actual earth radius in meters
397actual_radius_m = earth_radius * 1000.0_r8
398
399obs_lat = y/actual_radius_m + radar_lat
400obs_lon = x/(actual_radius_m*cos(radar_lat + y/(2.0_r8*actual_radius_m))) + &
401 radar_lon
402
403! Already verified to be height in meters.
404vloc = query_location(radar_location, 'vloc')
405
406! Add the height increment and check for 'too high'.
407vloc = vloc + h
408
409if(vloc.gt.ceiling) then ! max. height for radar obs = 8.5km
410 qc = .false. ! bad data
411 return
412endif
413
414! Convert back to degrees (required by set_location() call).
415obs_lon = obs_lon*rad2deg
416obs_lat = obs_lat*rad2deg
417which_vert = nint(query_location(radar_location,'which_vert'))
418
419obs_location = set_location(obs_lon, obs_lat, vloc, which_vert)
420
421
422! The elevation angle at the observation location is based on p. 23 of
423! Battan's radar meteorology book (equation 3.18), assuming an effective
424! radius of the earth of 4a/3.
425
426elev_obs = sqrt(1.5_r8 * h / actual_radius_m + beam_elev*beam_elev)
427
428! The radial velocity obs have additional metadata associated with them
429! that must be set here.
430if(select_obstype .eq. 1 .or. select_obstype .eq. 3) then
431 beam_direction(1) = sin(beam_az)*cos(elev_obs)
432 beam_direction(2) = cos(beam_az)*cos(elev_obs)
433 beam_direction(3) = sin(elev_obs)
434
435 nyquist_velocity = missing_r8
436
437 call set_radial_vel(defkey, radar_location, beam_direction, nyquist_velocity)
438endif
439
440qc = .true. ! all good
441
442end subroutine compute_location

446subroutine fillin_obsdef(obskind, obsloc, days, secs, variance, defkey, obs_def)
447
448! Set the values in a DART obs_def_type derived type.
449
450integer, intent(in) :: obskind
451type(location_type), intent(in) :: obsloc
452integer, intent(in) :: days
453integer, intent(in) :: secs
454real(r8), intent(in) :: variance
455integer, intent(in) :: defkey
456type(obs_def_type), intent(inout) :: obs_def
457
458! set each part of the obs def
459call set_obs_def_type_of_obs(obs_def, obskind)
460
461call set_obs_def_location(obs_def, obsloc)
462
463! Note that set_time wants seconds first, days second.
464call set_obs_def_time(obs_def, set_time(secs, days) )
465
466call set_obs_def_error_variance(obs_def, variance)
467
468! The key is required in the radial velocity obs; it is unused in the
469! reflectivity obs. (The key is needed only for obs types which have
470! auxilliary data associated with each obs.)
471call set_obs_def_key(obs_def, defkey)
472
473end subroutine fillin_obsdef
