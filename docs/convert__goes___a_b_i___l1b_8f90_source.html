56logical :: reject_dqf_1 = .true.
57logical :: verbose = .false.

59real(r8) :: obs_err = missing_r8 ! the fixed obs error (std dev, in radiance units)
60 ! TODO: make this more sophisticated

63
64namelist /convert_goes_abi_l1b_nml/ l1_files, l1_file_list, &

73! ----------------------------------------------------------------------
74

88call check_namelist_read(iunit, io, 'convert_goes_ABI_L1b_nml')
89
90! Record the namelist values used for the run ...
91if (do_nml_file()) write(nmlfileunit, nml=convert_goes_abi_l1b_nml)
92if (do_nml_term()) write( * , nml=convert_goes_abi_l1b_nml)
93
94if (obs_err == missing_r8) then
95 call error_handler(e_err, source, 'An obs_err value was not specified in the namelist, but is required.',source,revision,revdate)
96end if
97
98if (vloc_pres_hpa < 0.0_r8) then
99 call error_handler(e_msg, source, 'No vertical pressure level defined, using VERTISUNDEF.',source,revision,revdate)
100end if
101
102! when this routine returns, the l1_files variable will have
103! all the filenames, regardless of which way they were specified.
104filecount = set_filename_list(l1_files, l1_file_list, "convert_goes_ABI_L1b")
105
106! for each input file
107do ifile=1, filecount
108 ! read from netCDF file into a derived type that holds all the information
109 call goes_load_abi_map(l1_files(ifile), map)
110
111 ! convert derived type information to DART sequence
112 call make_obs_sequence(seq, map, lon1, lon2, lat1, lat2, &

115
116 ! write the sequence to a disk file
117 call write_obs_seq(seq, outputfile)
118
119 if (verbose) call print_obs_seq_summary(seq,l1_files(ifile))
120
121 ! release the sequence memory
122 call destroy_obs_sequence(seq)
123enddo
124
125call error_handler(e_msg, source, 'Finished successfully.',source,revision,revdate)
126call finalize_utilities()
127
128end program convert_goes_abi_l1b
