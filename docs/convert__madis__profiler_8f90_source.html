7program convert_madis_profiler
8
9!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
10!
11! convert_madis_profiler - program that reads a MADIS netCDF profiler
12! wind observation file and writes a DART
13! obs_seq file using the DART library routines.
14!
15! created Dec. 2007 Ryan Torn, NCAR/MMM
16! modified Dec. 2008 Soyoung Ha and David Dowell, NCAR/MMM
17! modified to include QC_flag check (Soyoung Ha, NCAR/MMM, 08-04-2009)
18!
19! modified to use a common set of utilities, better netcdf error checks,
20! able to insert obs with any time correctly (not only monotonically
21! increasing times) nancy collins, ncar/image 11 march 2010
22!
23! keep original obs times, make source for all converters as similar
24! as possible. nancy collins, ncar/image 26 march 2010
25!
26! adapted June 2011, nancy collins, ncar/image, Glen Romine ncar/mmm
27! - split from the satwind version
28!
29!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
30
31use types_mod, only : r8, missing_r8
32use utilities_mod, only : initialize_utilities, finalize_utilities
33use netcdf_utilities_mod, only : nc_open_file_readonly, nc_close_file
34use time_manager_mod, only : time_type, set_calendar_type, set_date, &
35 increment_time, get_time, operator(-), gregorian
36use location_mod, only : vertisheight
37use obs_sequence_mod, only : obs_sequence_type, obs_type, read_obs_seq, &
38 static_init_obs_sequence, init_obs, write_obs_seq, &
39 init_obs_sequence, get_num_obs, &
40 set_copy_meta_data, set_qc_meta_data
41use obs_err_mod, only : prof_wind_error
42use obs_kind_mod, only : profiler_u_wind_component, profiler_v_wind_component
43use sort_mod, only : index_sort
44use obs_utilities_mod, only : getvar_real, get_or_fill_qc, add_obs_to_seq, &
45 create_3d_obs, getvar_int, getdimlen, getvar_real_2d, &
46 getvar_int_2d, query_varname
47
48use netcdf
49
50implicit none
51
52character(len=20), parameter :: profiler_netcdf_file = 'profiler_input.nc'
53character(len=129), parameter :: profiler_out_file = 'obs_seq.profiler'
54
55logical, parameter :: use_input_qc = .true.
56
57integer, parameter :: num_copies = 1, & ! number of copies in sequence
58 num_qc = 1 ! number of QC entries
59
60
61integer :: ncid, nsta, nlev, n, i, oday, osec, nused, k, index
62logical :: file_exist, first_obs
63real(r8) :: uwnd_miss, vwnd_miss
64
65real(r8) :: oerr, qc
66
67real(r8), allocatable :: lat(:), lon(:), latu(:), lonu(:), &
68 levs(:,:), elev(:), tobs(:), tobu(:), &
69 uwnd(:,:), vwnd(:,:), levu(:)
70integer, allocatable :: qc_uwnd(:,:), qc_vwnd(:,:)
71
72type(obs_sequence_type) :: obs_seq
73type(obs_type) :: obs, prev_obs
74type(time_type) :: comp_day0, time_obs, prev_time
75
76character(len=NF90_MAX_NAME) :: namelist(5)
77
78!------------
79! start of executable code
80!------------
81
82call initialize_utilities('convert_madis_profiler')
83
84
85! put the reference date into DART format
86call set_calendar_type(gregorian)
87comp_day0 = set_date(1970, 1, 1, 0, 0, 0)
88
89first_obs = .true.
90
91
92ncid = nc_open_file_readonly(profiler_netcdf_file, 'convert_madis_profiler')
93
94call getdimlen(ncid, "recNum", nsta)
95call getdimlen(ncid, "level" , nlev)
96
97allocate( lat(nsta)) ; allocate( lon(nsta))
98allocate(latu(nsta*nlev)) ; allocate(lonu(nsta*nlev))
99allocate(tobs(nsta)) ; allocate(tobu(nsta*nlev))
100allocate(elev(nsta)) ; allocate(levu(nsta*nlev))
101! 2d variables - levels at each station
102allocate(uwnd(nlev,nsta)) ; allocate(vwnd(nlev,nsta))
103allocate(qc_uwnd(nlev,nsta)) ; allocate(qc_vwnd(nlev,nsta))
104allocate(levs(nlev,nsta))
105
106! read in the data arrays
107
108! we have profiler data files which have different names for the
109! lat/lon/elev/obs arrays in the netcdf file. there doesn't seem
110! to be a global attr to say which one is in use, so for now try
111! both options.
112
113namelist(1) = 'staLat'
114namelist(2) = 'latitude'
115call query_varname(ncid, 2, namelist, index, force=.true.)
116call getvar_real(ncid, namelist(index), lat ) ! station latitude
117
118namelist(1) = 'staLon'
119namelist(2) = 'longitude'
120call query_varname(ncid, 2, namelist, index, force=.true.)
121call getvar_real(ncid, namelist(index), lon ) ! station longitude
122
123namelist(1) = 'staElev'
124namelist(2) = 'elevation'
125call query_varname(ncid, 2, namelist, index, force=.true.)
126call getvar_real(ncid, namelist(index), elev ) ! station elevation
127
128namelist(1) = 'timeObs'
129namelist(2) = 'observationTime'
130call query_varname(ncid, 2, namelist, index, force=.true.)
131call getvar_real(ncid, namelist(index), tobs ) ! observation time
132
133call getvar_real_2d(ncid, "levels", levs ) ! height above station in meters
134call getvar_real_2d(ncid, "uComponent", uwnd, uwnd_miss) ! e-w component
135call getvar_real_2d(ncid, "vComponent", vwnd, vwnd_miss) ! n-s component
136
137! if user says to use them, read in QCs if present
138if (use_input_qc) then
139 call getvar_int_2d(ncid, "uComponentQCR", qc_uwnd ) ! wind direction qc
140 call getvar_int_2d(ncid, "vComponentQCR", qc_vwnd ) ! wind speed qc
141else
142 qc_uwnd = 0 ; qc_vwnd = 0
143endif
144
145! levels is height above station. we want actual elevation which means
146! adding on the station elevation. add it in here so when we use levs
147! below it is the actual height above MSL.
148do n = 1, nsta
149 levs(:,n) = levs(:,n) + elev(n)
150enddo
151
152! either read existing obs_seq or create a new one
153call static_init_obs_sequence()
154call init_obs(obs, num_copies, num_qc)
155call init_obs(prev_obs, num_copies, num_qc)
156
157inquire(file=profiler_out_file, exist=file_exist)
158
159if ( file_exist ) then
160
161 ! existing file found, append to it
162 call read_obs_seq(profiler_out_file, 0, 0, 2*nsta*nlev, obs_seq)
163
164else
165
166 ! create a new one
167 call init_obs_sequence(obs_seq, num_copies, num_qc, 2*nsta*nlev)
168 do i = 1, num_copies
169 call set_copy_meta_data(obs_seq, i, 'MADIS observation')
170 end do
171 do i = 1, num_qc
172 call set_qc_meta_data(obs_seq, i, 'Data QC')
173 end do
174
175endif
176
177! Set the DART data quality control. Be consistent with NCEP codes;
178! 0 is 'must use', 1 is good, no reason not to use it.
179qc = 1.0_r8
180
181nused = 0
182staloop: do n = 1, nsta
183
184 ! compute time of observation
185 time_obs = increment_time(comp_day0, nint(tobs(n)))
186
187 ! check the lat/lon values to see if they are ok
188 if ( lat(n) > 90.0_r8 .or. lat(n) < -90.0_r8 ) cycle staloop
189 if ( lon(n) > 180.0_r8 .or. lon(n) < -180.0_r8 ) cycle staloop
190
191 if ( lon(n) < 0.0_r8 ) lon(n) = lon(n) + 360.0_r8
192 ! extract actual time of observation in file into oday, osec.
193 call get_time(time_obs, osec, oday)
194
195levloop: do k = 1, nlev
196 ! Check for duplicate observations
197 do i = 1, nused
198 if ( lon(n) == lonu(i) .and. &
199 lat(n) == latu(i) .and. &
200 levs(k,n) == levu(i) .and. &
201 tobs(n) == tobu(i) ) cycle levloop
202 end do
203
204
205 ! add wind component data to obs_seq
206 if ( uwnd(k,n) /= uwnd_miss .and. qc_uwnd(k,n) == 0 .and. &
207 vwnd(k,n) /= vwnd_miss .and. qc_vwnd(k,n) == 0 ) then
208
209 ! FIXME: we do not have pressure in these files, only height
210 ! need pres(n) = convert_std_atm(height(n))
211 !oerr = prof_wind_error(pres(n)) ! pressure based table
212 oerr = 2.000_r8 ! this works mostly for surface to 100mb
213
214 ! perform sanity checks on observation errors and values
215 if ( oerr == missing_r8 .or. &
216 abs(uwnd(k,n)) > 150.0_r8 .or. &
217 abs(vwnd(k,n)) > 150.0_r8 ) cycle levloop
218
219 call create_3d_obs(lat(n), lon(n), levs(k,n), vertisheight, uwnd(k,n), &
220 profiler_u_wind_component, oerr, oday, osec, qc, obs)
221 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
222
223 call create_3d_obs(lat(n), lon(n), levs(k,n), vertisheight, vwnd(k,n), &
224 profiler_v_wind_component, oerr, oday, osec, qc, obs)
225 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
226
227 endif
228
229 nused = nused + 1
230 latu(nused) = lat(n)
231 lonu(nused) = lon(n)
232 levu(nused) = levs(k,n)
233 tobu(nused) = tobs(n)
234
235end do levloop
236end do staloop
237
238! need to wait to close file because in the loop it queries the
239! report types.
240call nc_close_file(ncid, 'convert_madis_profiler')
241
242! if we added any obs to the sequence, write it now.
243if ( get_num_obs(obs_seq) > 0 ) call write_obs_seq(obs_seq, profiler_out_file)
244
245! end of main program
246call finalize_utilities()
247
248
249end program
