136
137type(obs_sequence_type) :: seq
138type(obs_type) :: obs
139type(obs_def_type) :: obs_def
140type(random_seq_type) :: random_seq

145integer, allocatable :: keys(:)
146integer :: i, j, iunit, time_step_number, obs_seq_file_id
147integer :: cnum_copies, cnum_qc, cnum_obs, cnum_max
148integer :: additional_qc, additional_copies, forward_unit
149integer :: io, num_obs_in_set, nth_obs
150integer :: key_bounds(2), num_qc, last_key_used
151integer :: seed
152integer(i8) :: model_size
153
154integer :: istatus(1)
155
156real(r8) :: true_obs(1), obs_value(1), qc(1)
157

167integer :: secs, days

171
172type(file_info_type) :: file_info_input
173type(file_info_type) :: file_info_output
174type(file_info_type) :: file_info_true

184! Initialize all modules used that require it
185call perfect_initialize_modules_used()
186
187! Read the namelist entry
188call find_namelist_in_file("input.nml", "perfect_model_obs_nml", iunit)
189read(iunit, nml = perfect_model_obs_nml, iostat = io)
190call check_namelist_read(iunit, io, "perfect_model_obs_nml")
191
192! Record the namelist values used for the run ...
193if (do_nml_file()) write(nmlfileunit, nml=perfect_model_obs_nml)
194if (do_nml_term()) write( * , nml=perfect_model_obs_nml)
195
196! set the level of output
197call set_trace(trace_execution, output_timestamps, silence)
198
199call trace_message('Perfect_model start')
200call timestamp_message('Perfect_model start')
201
202! Default to printing nothing
203nth_obs = -1
204
205call trace_message('Before setting up space for observations')
206call timestamp_message('Before setting up space for observations')
207
208!>@todo FIXME: copies and qc should be set using the meta data strings not hard
209! coded.
210
211! Find out how many data copies are in the obs_sequence
212call read_obs_seq_header(obs_seq_in_file_name, cnum_copies, cnum_qc, cnum_obs, cnum_max, &
213 obs_seq_file_id, obs_seq_read_format, pre_i_format, close_the_file = .true.)
214
215! First two copies of output will be truth and observation;
216! Will overwrite first two existing copies in file if there are any
217additional_copies = 2 - cnum_copies
218if(additional_copies < 0) additional_copies = 0
219
220! Want to have a qc field available in case forward op wont work
221if(cnum_qc == 0) then
222 additional_qc = 1
223else
224 additional_qc = 0
225endif
226
227! Read in definition part of obs sequence; expand to include observation and truth field
228call read_obs_seq(obs_seq_in_file_name, additional_copies, additional_qc, 0, seq)
229
230! Initialize an obs type variable
231call init_obs(obs, cnum_copies + additional_copies, cnum_qc + additional_qc)
232
233! Need metadata for added qc field
234if(additional_qc == 1) then
235 qc_meta_data = 'Quality Control'
236 call set_qc_meta_data(seq, 1, qc_meta_data)
237endif
238
239! Need space to put in the obs_values in the sequence;
240copy_meta_data(1) = 'observations'
241copy_meta_data(2) = 'truth'
242call set_copy_meta_data(seq, 1, copy_meta_data(1))
243call set_copy_meta_data(seq, 2, copy_meta_data(2))
244
245call timestamp_message('After setting up space for observations')
246call trace_message('After setting up space for observations')
247
248! Initialize the model now that obs_sequence is all set up
249model_size = get_model_size()
250write(msgstring,*)'Model size = ',model_size
251call error_handler(e_msg,'perfect_main',msgstring)
252
253! Set up the ensemble storage and read in the restart file
254call trace_message('Before reading in ensemble restart file')
255call init_ensemble_manager(ens_handle, ens_size, model_size)
256
257call set_num_extra_copies(ens_handle, 0)
258
259! for now, make the default to allow cycling if we're writing all data
260! to a single file, and false if not.
261has_cycling = single_file_out
262
263! Initialize file names:
264
265! this routine allocates the second argument to be the correct length
266call parse_filenames(input_state_files, input_filelist, nfilesin)
267call parse_filenames(output_state_files, output_filelist, nfilesout)
268
269!> @todo FIXME if nfilesout == 0 and write_output_state_to_file is .false.
270!> that shouldn't be an error. if nfilesin == 0 and read_input_state_from_file
271!> is false, that also shouldn't be an error. (unless you're writing the mean
272!> and sd, and then maybe we should have a different name for output of input values.)
273if (nfilesin == 0 .or. nfilesout == 0 ) then
274 msgstring = 'must specify both "input_state_files" and "output_state_files" in the namelist'
275 call error_handler(e_err,'perfect_main',msgstring,source)
276endif
277
278allocate(true_state_filelist(nfilesout))
279
280! mutiple domains ( this is very unlikely to be the case, but in order to
281! set_file_metadata we need to have a file list that has the same number
282! of files as domains.
283if (nfilesout > 1) then
284 do i = 1, nfilesout
285 write(true_state_filelist(i),'(a,i0.2)') 'true_state_d',i
286 enddo
287else
288 true_state_filelist(1) = 'true_state.nc'
289endif
290
291call io_filenames_init(file_info_input, 1, cycling=has_cycling, single_file=single_file_in)
292call set_file_metadata(file_info_input, 1, input_filelist, 'perfect_input', 'pmo initial condition')
293call set_io_copy_flag( file_info_input, 1, read_copy)
294
295! True State
296call io_filenames_init(file_info_true, 1, cycling=has_cycling, single_file=single_file_out)
297call set_file_metadata(file_info_true, 1, true_state_filelist, 'true_state', 'true state')
298call set_io_copy_flag( file_info_true, 1, 1, write_copy, num_output_ens=1)
299
300! Perfect Restart
301call io_filenames_init(file_info_output, 1, cycling=has_cycling, single_file=single_file_out)
302call set_file_metadata(file_info_output, 1, output_filelist, 'perfect_restart', 'pmo restart')
303call set_io_copy_flag( file_info_output, 1, 1, write_copy, num_output_ens=1)
304
305! Set a time type for initial time if namelist inputs are not negative
306call filter_set_initial_time(init_time_days, init_time_seconds, time1, read_time_from_file)
307
308if (read_input_state_from_file) then
309
310 call error_handler(e_msg,'perfect_read_restart:', 'reading input state from file')
311 call read_state(ens_handle, file_info_input, read_time_from_file, time1)
312
313else ! model spin up
314
315 call error_handler(e_msg,'perfect_read_restart:', &
316 'Using code in model_mod to initialize ensemble')
317 call allocate_vars(ens_handle)
318 if(ens_handle%my_pe == 0) call get_initial_condition(ens_handle%time(1), ens_handle%vars(:, 1))
319 call all_vars_to_all_copies(ens_handle)
320
321endif
322
323! Temporary print of initial model time
324if (my_task_id() == 0) then
325 call get_time(ens_handle%time(1),secs,days)
326 write(msgstring, *) 'initial model time of perfect_model member (days,seconds) ',days,secs
327 call error_handler(e_dbg,'perfect_read_restart',msgstring,source)
328endif
329
330call trace_message('After reading in ensemble restart file')
331
332!>@todo FIXME this block must be supported in the single file loop with time dimension
333call trace_message('Before initializing output diagnostic file')
334state_meta(1) = 'true state'
335call trace_message('After initializing output diagnostic file')
336
337! Get the time of the first observation in the sequence
338write(msgstring, *) 'total number of obs in sequence is ', get_num_obs(seq)
339call error_handler(e_msg,'perfect_main',msgstring)
340
341num_qc = get_num_qc(seq)
342write(msgstring, *) 'number of qc values is ',num_qc
343call error_handler(e_msg,'perfect_main',msgstring)
344
345call trace_message('Before trimming obs seq if start/stop time specified')
346
347! Need to find first obs with appropriate time, delete all earlier ones
348if(first_obs_seconds > 0 .or. first_obs_days > 0) then
349 first_obs_time = set_time(first_obs_seconds, first_obs_days)
350 call delete_seq_head(first_obs_time, seq, all_gone)
351 if(all_gone) then
352 msgstring = 'All obs in sequence are before first_obs_days:first_obs_seconds'
353 call error_handler(e_err,'perfect_main',msgstring,source)
354 endif
355endif
356
357last_key_used = -99
358
359! Also get rid of observations past the last_obs_time if requested
360if(last_obs_seconds >= 0 .or. last_obs_days >= 0) then
361 last_obs_time = set_time(last_obs_seconds, last_obs_days)
362 call delete_seq_tail(last_obs_time, seq, all_gone)
363 if(all_gone) then
364 msgstring = 'All obs in sequence are after last_obs_days:last_obs_seconds'
365 call error_handler(e_err,'perfect_main',msgstring,source)
366 endif
367endif
368
369! Do verbose forward operator output if requested
370write(task_str, '(i10)') ens_handle%my_pe
371if(output_forward_op_errors) forward_unit = open_file('forward_op_errors' // trim(adjustl(task_str)), 'formatted', 'append')
372
373call trace_message('After trimming obs seq if start/stop time specified')
374
375! Time step number is used to do periodic diagnostic output
376time_step_number = -1
377window_time = set_time(0,0)
378curr_ens_time = set_time(0, 0)
379next_ens_time = set_time(0, 0)
380
381! Advance model to the closest time to the next available observations
382advancetime: do
383 time_step_number = time_step_number + 1
384
385 write(msgstring , '(A,I5)') 'Main evaluation loop, starting iteration', time_step_number
386 call trace_message(' ', ' ', -1)
387 call trace_message(msgstring, 'perfect_model_obs: ', -1)
388
389 call trace_message('Before move_ahead checks time of data and next obs')
390
391 ! Only processes with an ensemble copy know to exit;
392 ! For now, let process 0 broadcast its value of key_bounds
393 ! This will synch the loop here and allow everybody to exit
394 ! Need to clean up and have a broadcast that just sends a single integer???
395 ! PAR For now, can only broadcast real arrays
396
397 ! Get the model to a good time to use a next set of observations
398 call move_ahead(ens_handle, 1, seq, last_key_used, window_time, &
399 key_bounds, num_obs_in_set, curr_ens_time, next_ens_time)
400
401 call filter_sync_keys_time(ens_handle, key_bounds, num_obs_in_set, curr_ens_time, next_ens_time)
402
403 if(key_bounds(1) < 0) then
404 call trace_message('No more obs to evaluate, exiting main loop', 'perfect_model_obs:', -1)
405 exit advancetime
406 endif
407
408 call trace_message('After move_ahead checks time of data and next obs')
409
410 if (curr_ens_time /= next_ens_time) then
411 call trace_message('Ready to run model to advance data ahead in time', 'perfect_model_obs:', -1)
412 call print_ens_time(ens_handle, 'Ensemble data time before advance')
413
414 ! we are going to advance the model - make sure we're doing single file output
415 if (.not. has_cycling) then
416 call error_handler(e_err,'filter:', &
417 'advancing the model inside PMO and multiple file output not currently supported', &
418 source, text2='support will be added in subsequent releases', &
419 text3='set "single_file_out=.true" for PMO to advance the model, or advance the model outside PMO')
420 endif
421
422 call trace_message('Before running model')
423 call timestamp_message('Before running model', sync=.true.)
424
425 call allocate_vars(ens_handle)
426 call all_copies_to_all_vars(ens_handle)
427
428 if (ens_handle%my_pe == 0) call advance_state(ens_handle, 1, next_ens_time, async, &
429 adv_ens_command, tasks_per_model_advance, file_info_true, file_info_input)
430
431 call all_vars_to_all_copies(ens_handle)
432
433 ! update so curr time is accurate.
434 curr_ens_time = next_ens_time
435 ens_handle%current_time = curr_ens_time
436
437 call timestamp_message('After running model', sync=.true.)
438 call trace_message('After running model')
439 call print_ens_time(ens_handle, 'Ensemble data time after advance')
440 else
441 call trace_message('Model does not need to run; data already at required time', 'perfect_model_obs:', -1)
442 endif
443
444 ! Initialize a repeatable random sequence for perturbations
445 seed = generate_seed(next_ens_time)
446 call init_random_seq(random_seq,seed)
447
448 call trace_message('Before setup for next group of observations')
449 write(msgstring, '(A,I7)') 'Number of observations to be evaluated', &
450 num_obs_in_set
451 call trace_message(msgstring)
452 call print_obs_time(seq, key_bounds(1), 'Time of first observation in window')
453 call print_obs_time(seq, key_bounds(2), 'Time of last observation in window')
454
455 ! for multi-core runs, each core needs to store the forward operator and the qc value
456 call init_ensemble_manager(fwd_op_ens_handle, ens_size, int(num_obs_in_set,i8), 1, transpose_type_in = 2)
457 call init_ensemble_manager( qc_ens_handle, ens_size, int(num_obs_in_set,i8), 1, transpose_type_in = 2)
458
459 ! Allocate storage for observation keys for this part of sequence
460 allocate(keys(num_obs_in_set))
461
462 ! Get all the keys associated with this set of observations
463 call get_time_range_keys(seq, key_bounds, num_obs_in_set, keys)
464
465 call trace_message('After setup for next group of observations')
466
467 ! Output the true state to the netcdf file
468 if((output_interval > 0) .and. &
469 (time_step_number / output_interval * output_interval == time_step_number)) then
470
471 call trace_message('Before updating true state')
472 if(has_cycling) then
473 call write_state(ens_handle, file_info_true)
474 endif
475 call trace_message('After updating true state')
476
477 endif
478
479 write(msgstring, '(A,I8,A)') 'Ready to evaluate up to', size(keys), ' observations'
480 call trace_message(msgstring, 'perfect_model_obs:', -1)
481
482 ! Set up access to the state
483 call create_state_window(ens_handle)
484
485 ! Compute the forward observation operator for each observation in set
486 do j = 1, fwd_op_ens_handle%my_num_vars
487
488 ! Some compilers do not like mod by 0, so test first.
489 if (print_every_nth_obs > 0) nth_obs = mod(j, print_every_nth_obs)
490
491 ! If requested, print out a message every Nth observation
492 ! to indicate progress is being made and to allow estimates
493 ! of how long the assim will take.
494 if (nth_obs == 0) then
495 write(msgstring, '(A,1x,I8,1x,A,I8)') 'Processing observation ', j, &
496 ' of ', num_obs_in_set
497 call trace_message(msgstring, 'perfect_model_obs:', -1)
498 ! or if you want timestamps:
499 ! call timestamp(msgstring, pos="debug")
500 endif
501
502 ! Compute the observations from the state
503 global_obs_num = fwd_op_ens_handle%my_vars(j)
504 call get_expected_obs_distrib_state(seq, keys(global_obs_num:global_obs_num), &
505 curr_ens_time, .true., &
506 istatus, assimilate_this_ob, evaluate_this_ob, &
507 ens_handle, ens_size, copy_indices, true_obs)
508
509 fwd_op_ens_handle%copies(1, j) = true_obs(1)
510
511 qc_ens_handle%copies(1, j) = set_input_qc(istatus(1), assimilate_this_ob, evaluate_this_ob)
512
513 ! FIXME: we could set different qc codes, like 1004, 1005, to
514 ! indicate why this obs isn't being processed - separate failed
515 ! forward operators from those types not on the assim or eval lists.
516 ! the values 4, 5, etc could match the dart QC values + 1000.
517
518 ! if failed forward op logging requested, make sure we're
519 ! only writing out obs with real errors and not those that
520 ! end up in this code section because their type isn't in the namelist.
521 if(output_forward_op_errors) then
522 if ((istatus(1) /= 0) .and. (assimilate_this_ob .or. evaluate_this_ob)) &
523 write(forward_unit, *) keys(global_obs_num), istatus
524 endif
525
526 end do
527
528 ! collect on task 0 and load up the obs_sequence
529 call all_copies_to_all_vars(fwd_op_ens_handle)
530 call all_copies_to_all_vars( qc_ens_handle)
531
532 ! Task 0 loads up the obs_sequence
533 if(my_task_id() == 0) then
534
535 do i = 1, fwd_op_ens_handle%num_vars
536
537 ! true obs is the forward operator
538 ! I'm doing the random noise on task 0 so we remain bitwise the trunk
539 true_obs(1) = fwd_op_ens_handle%vars(i, 1)
540
541 ! Get the observational error covariance (diagonal at present)
542 ! Generate the synthetic observations by adding in error samples
543 call get_obs_from_key(seq, keys(i), obs)
544 call get_obs_def(obs, obs_def)
545
546 ! If observation is not being evaluated or assimilated, skip it
547 ! Ends up setting a 1000 qc field so observation is not used again.
548 if( qc_ens_handle%vars(i, 1) == 0 ) then
549
550 ! Get the information for generating error sample for this observation
551 call obs_error_info(obs_def, error_variance, &
552 bounded_below, bounded_above, lower_bound, upper_bound)
553
554 ! Capability to do a bounded normal error
555 if(bounded_below .and. bounded_above) then
556 ! Bounds on both sides
557 obs_value(1) = lower_bound - 1.0_r8
558 do while(obs_value(1) < lower_bound .or. obs_value(1) > upper_bound)
559 obs_value(1) = random_gaussian(random_seq, true_obs(1), &
560 sqrt(error_variance))
561 end do
562 elseif(bounded_below .and. .not. bounded_above) then
563 ! Bound on lower side
564 obs_value(1) = lower_bound - 1.0_r8
565 do while(obs_value(1) < lower_bound)
566 obs_value(1) = random_gaussian(random_seq, true_obs(1), &
567 sqrt(error_variance))
568 end do
569 elseif(.not. bounded_below .and. bounded_above) then
570 ! Bound on upper side
571 obs_value(1) = upper_bound + 1.0_r8
572 do while(obs_value(1) > upper_bound)
573 obs_value(1) = random_gaussian(random_seq, true_obs(1), &
574 sqrt(error_variance))
575 end do
576 else
577 ! No bounds, regular old normal distribution
578 obs_value(1) = random_gaussian(random_seq, true_obs(1), &
579 sqrt(error_variance))
580 endif
581
582 ! FIX ME SPINT: if the foward operater passed can we directly set the
583 ! qc status?
584
585 ! Set qc to 0 if none existed before
586 if(cnum_qc == 0) then
587 qc(1) = qc_ens_handle%vars(i, 1)
588 call set_qc(obs, qc, 1)
589 endif
590 else
591
592 obs_value(1) = true_obs(1)
593 qc(1) = qc_ens_handle%vars(i, 1)
594 call set_qc(obs, qc, 1)
595
596 endif
597
598 ! true obs is fowrard op
599 ! obs value is forward op plus noise
600 call set_obs_values(obs, obs_value, 1)
601 call set_obs_values(obs, true_obs, 2)
602
603 ! Insert the observations into the sequence first copy
604 call set_obs(seq, obs, keys(i))
605
606 enddo
607
608 endif
609
610 ! End access to the state
611 call free_state_window(ens_handle)
612
613 ! Deallocate the keys storage
614 deallocate(keys)
615
616 ! The last key used is updated to move forward in the observation sequence
617 last_key_used = key_bounds(2)
618
619end do advancetime
620
621! if logging errors, close unit
622if(output_forward_op_errors) call close_file(forward_unit)
623
624call trace_message('End of main evaluation loop, starting cleanup', 'perfect_model_obs:', -1)
625
626! Write out the sequence
627call trace_message('Before writing output sequence file')
628if (ens_handle%my_pe == 0) call write_obs_seq(seq, obs_seq_out_file_name)
629call end_ensemble_manager(fwd_op_ens_handle)
630call end_ensemble_manager(qc_ens_handle)
631call trace_message('After writing output sequence file')
632
633! Output a restart file if requested
634call trace_message('Before writing state restart file if requested')
635if(write_output_state_to_file) then
636 call write_state(ens_handle, file_info_output)
637 if (single_file_out) &
638 call finalize_single_file_io(file_info_output)
639endif
640
641! Write the last time step to the true state file
642if (single_file_out) then
643 call write_state(ens_handle, file_info_true)
644 call finalize_single_file_io(file_info_true)
645endif
646
647call trace_message('After writing state restart file if requested')
648call trace_message('Before ensemble and obs memory cleanup')
649
650! Release storage for ensemble
651call end_ensemble_manager(ens_handle)
652
653! Free up the observation kind and obs sequence
654call destroy_obs(obs)
655call destroy_obs_sequence(seq)
656call trace_message('After ensemble and obs memory cleanup')
657
658call end_algorithm_info_mod()
659
660call trace_message('Perfect_model done')
661call timestamp_message('Perfect_model done')
662
663!call error_handler(E_MSG,'perfect_main','FINISHED',source)
664
665! closes the log file.
666call finalize_mpi_utilities()
667
668end subroutine perfect_main
