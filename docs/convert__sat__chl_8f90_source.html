79! Observation uncertainty: MAny studies indicate that
80! obs error sd is 35% [e.g., Moore et al. 2009]
81! Discussed this with Ariane Verdy from UCSD,
82! chl = log(data) to go from lognormal to normal pdf
83! if in the lognormal space sd is 0.5 the data value,
84! then in the trasnformed space obs_err_sd is simply 0.35
85real(r8) :: chl_error_sd = 0.35_r8
86
87! If we need to mask other sea areas
88! For instance, we are only interested in the Red Sea
89! and not the Persian/Arabian Gulf
90real(r8) :: lon_mask = 47.5_r8
91real(r8) :: lat_mask = 26.5_r8

97character(len=256) :: file_out = 'obs_seq.chl'
98real(r8) :: chl_thresh = 0.03_r8
99integer :: subsample_intv = 1
100logical :: special_mask = .false.
101logical :: debug = .false.
102
103namelist /convert_sat_chl_nml/ file_in, &
104 file_out, &
105 chl_thresh, &
106 subsample_intv, &
107 special_mask, &
108 debug

134call nc_get_variable(ncid, 'lon' , lon, source)

138
139base_time = set_base_time(ncid)
140

143allocate(chl(nlon,nlat))
144
145call static_init_obs_sequence()
146
147io = nf90_inq_varid(ncid, 'chlor_a', varid)
148call nc_check(io, source, context='getting chl variable ID', ncid=ncid)
149call nc_get_attribute_from_variable(ncid, 'chlor_a', '_FillValue', missing_value, source)

151timeloop: do itime = 1,ndays
152
153 ! convert to integer days and seconds, and add on to reference time.
154 iday = time(itime)
155 isec = (time(itime) - iday) * 86400
156 delta_time = set_time(isec, iday)
157 obs_time = base_time + delta_time
158 call get_time(obs_time, osec, oday)
159
160 call print_time(obs_time, str='obs time is ')
161 call print_date(obs_time, str='obs date is ')

163 call get_date(obs_time, year, month, day, hour, minutes, seconds)
164
165 seconds = seconds + (hour*60 + minutes)*60
166
167 write(string1,'(i4,''-'',i2.2,''-'',i2.2,''-'',i5.5)') year,month,day,seconds
168 write(output_file,'(A)') trim(file_out)//'.'//trim(string1)
169 write(*,*)'output file is ',trim(output_file)
170
171 io = nf90_get_var(ncid, varid, chl, start=(/1,1,itime/), count=(/nlon,nlat,1/))
172 call nc_check(io, source, context='get_var chl', ncid=ncid)
173
174 first_obs = .true.
175 nmissing = 0
176 call init_obs( obs, num_copies, num_qc)
177 call init_obs(prev_obs, num_copies, num_qc)
178 call init_obs_sequence(obs_seq, num_copies, num_qc, num_new_obs)
179 call set_copy_meta_data(obs_seq, 1, 'CHL observation')
180 call set_qc_meta_data( obs_seq, 1, 'CHL QC')
181
182 latloop: do j = 1, nlat, subsample_intv
183 lonloop: do i = 1, nlon, subsample_intv
184
185 if (special_mask) then
186 ! Upper right corner of the Red Sea domain (Persian/Arabian Gulf)
187 if (lon(i) >= lon_mask .and. lat(j) >= lat_mask .and. chl(i, j) /= missing_value) then
188 ! Make sure it's masked
189 chl(i,j) = missing_value
190 endif
191 endif
192
193 if (chl(i,j) == missing_value) then
194 nmissing = nmissing + 1
195 cycle lonloop
196 endif
197
198 ! log-transform of the data
199 ! Assimilate the logarithm of the data
200 log_chl = log10(max(chl_thresh, chl(i, j)))
201
202 call create_3d_obs(lat(j), lon(i), 0.0_r8, vertissurface, log_chl, &
203 ocean_color, chl_error_sd, oday, osec, qc, obs)
204 call add_obs_to_seq(obs_seq, obs, obs_time, prev_obs, prev_time, first_obs)
205
206 enddo lonloop
207 enddo latloop
208
209 ! if we added any obs to the sequence, write it out to a file now.
210 if ( get_num_obs(obs_seq) > 0 ) then
211 if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
212 if (debug) print *, ' skipping = ', nmissing
213 call write_obs_seq(obs_seq, output_file)
214 else
215 write(string1,*)'no observations for output file'
216 write(string2,*)'"'//trim(output_file)//'"'
217 call error_handler(e_msg, source, string1, text2=string2)
218 endif
219
220 call destroy_obs_sequence(obs_seq)
221
222end do timeloop
223
224call nc_close_file(ncid, source)
225
226call error_handler(e_msg, source, 'Finished successfully.')
227call finalize_utilities()
