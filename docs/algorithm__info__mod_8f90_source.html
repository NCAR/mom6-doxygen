110subroutine init_algorithm_info_mod()
111
112! Gets number of lines/QTYs in the QCF table, allocates space for the table data
113
114
115integer :: fileid
116integer :: io, iunit
117
118integer :: numrows
119integer :: nlines
120
121if (module_initialized) return
122module_initialized = .true.
123
124! Read the namelist entry
125call find_namelist_in_file("input.nml", "algorithm_info_nml", iunit)
126read(iunit, nml = algorithm_info_nml, iostat = io)
127call check_namelist_read(iunit, io, "algorithm_info_nml")
128
129if (do_nml_file()) write(nmlfileunit, nml=algorithm_info_nml)
130if (do_nml_term()) write( * , nml=algorithm_info_nml)
131
132
133if (qceff_table_filename == '') then
134 write(errstring,*) 'No QCF table file listed in namelist, using default values for all QTYs'
135 call error_handler(e_msg, 'init_algorithm_info_mod:', errstring, source)
136 return
137endif
138
139use_qty_defaults = .false.
140fileid = open_file(trim(qceff_table_filename), 'formatted', 'read')
141
142! Do loop to get number of rows (or QTYs) in the table
143nlines = 0
144do
145 read(fileid,*,iostat=io)
146 if(io/=0) exit
147 nlines = nlines + 1
148end do
149
150call close_file(fileid)
151
152numrows = nlines - header_lines
153
154allocate(specified_qtys(numrows))
155allocate(qceff_table_data(numrows))
156allocate(dist_type_string_probit_inflation(numrows))
157allocate(dist_type_string_probit_state(numrows))
158allocate(dist_type_string_probit_extended_state(numrows))
159allocate(filter_kind_string(numrows))
160
161call read_qceff_table(qceff_table_filename)
162call assert_qceff_table_version()
163call verify_qceff_table_data()
164call log_qceff_table_data()
165
166end subroutine init_algorithm_info_mod

171subroutine read_qceff_table(qceff_table_filename)
172
173! Reads in the QCEFF input options from tabular data file
174
175character(len=129), intent(in) :: qceff_table_filename
176
177integer :: fileid
178integer :: row
179character(len=obstypelength) :: qty_string
180
181if (.not. module_initialized) call init_algorithm_info_mod()
182
183fileid = open_file(trim(qceff_table_filename), 'formatted', 'read')
184
185! skip the headers
186read(fileid, *) header1
187read(fileid, *) header2
188
189! read in table values directly to qceff_table_data type
190do row = 1, size(qceff_table_data)
191 read(fileid, *) qty_string, qceff_table_data(row)%obs_error_info%bounded_below, qceff_table_data(row)%obs_error_info%bounded_above, &
192 qceff_table_data(row)%obs_error_info%lower_bound, qceff_table_data(row)%obs_error_info%upper_bound, dist_type_string_probit_inflation(row), &
193 qceff_table_data(row)%probit_inflation%bounded_below, qceff_table_data(row)%probit_inflation%bounded_above, &
194 qceff_table_data(row)%probit_inflation%lower_bound, qceff_table_data(row)%probit_inflation%upper_bound, dist_type_string_probit_state(row), &
195 qceff_table_data(row)%probit_state%bounded_below, qceff_table_data(row)%probit_state%bounded_above, &
196 qceff_table_data(row)%probit_state%lower_bound, qceff_table_data(row)%probit_state%upper_bound, dist_type_string_probit_extended_state(row), &
197 qceff_table_data(row)%probit_extended_state%bounded_below, qceff_table_data(row)%probit_extended_state%bounded_above, &
198 qceff_table_data(row)%probit_extended_state%lower_bound, qceff_table_data(row)%probit_extended_state%upper_bound, &
199 filter_kind_string(row), qceff_table_data(row)%obs_inc_info%bounded_below, qceff_table_data(row)%obs_inc_info%bounded_above, &
200 qceff_table_data(row)%obs_inc_info%lower_bound, qceff_table_data(row)%obs_inc_info%upper_bound
201
202 call to_upper(qty_string)
203 specified_qtys(row) = get_index_for_quantity(qty_string)
204
205 if(specified_qtys(row) == -1) then
206 write(errstring,*) trim(qty_string), ' is not a valid DART QTY'
207 call error_handler(e_err, 'read_qceff_table:', errstring, source)
208 endif
209
210 ! Converting the distribution types (read in from table as a string) to its corresponding int value
211 call to_upper(dist_type_string_probit_inflation(row))
212
213 select case (trim(dist_type_string_probit_inflation(row)))
214 case ('NORMAL_DISTRIBUTION')
215 qceff_table_data(row)%probit_inflation%dist_type = normal_distribution
216 case ('BOUNDED_NORMAL_RH_DISTRIBUTION')
217 qceff_table_data(row)%probit_inflation%dist_type = bounded_normal_rh_distribution
218 case ('GAMMA_DISTRIBUTION')
219 qceff_table_data(row)%probit_inflation%dist_type = gamma_distribution
220 case ('BETA_DISTRIBUTION')
221 qceff_table_data(row)%probit_inflation%dist_type = beta_distribution
222 case ('LOG_NORMAL_DISTRIBUTION')
223 qceff_table_data(row)%probit_inflation%dist_type = log_normal_distribution
224 case ('UNIFORM_DISTRIBUTION')
225 qceff_table_data(row)%probit_inflation%dist_type = uniform_distribution
226 !!!case ('PARTICLE_FILTER_DISTRIBUTION')
227 !!!qceff_table_data(row)%probit_inflation%dist_type = PARTICLE_FILTER_DISTRIBUTION
228 case default
229 write(errstring, *) 'Invalid distribution type for probit inflation: ', trim(dist_type_string_probit_inflation(row))
230 call error_handler(e_err, 'read_qceff_table:', errstring, source)
231 end select
232
233
234 call to_upper(dist_type_string_probit_state(row))
235
236 select case (trim(dist_type_string_probit_state(row)))
237 case ('NORMAL_DISTRIBUTION')
238 qceff_table_data(row)%probit_state%dist_type = normal_distribution
239 case ('BOUNDED_NORMAL_RH_DISTRIBUTION')
240 qceff_table_data(row)%probit_state%dist_type = bounded_normal_rh_distribution
241 case ('GAMMA_DISTRIBUTION')
242 qceff_table_data(row)%probit_state%dist_type = gamma_distribution
243 case ('BETA_DISTRIBUTION')
244 qceff_table_data(row)%probit_state%dist_type = beta_distribution
245 case ('LOG_NORMAL_DISTRIBUTION')
246 qceff_table_data(row)%probit_state%dist_type = log_normal_distribution
247 case ('UNIFORM_DISTRIBUTION')
248 qceff_table_data(row)%probit_state%dist_type = uniform_distribution
249 !!!case ('PARTICLE_FILTER_DISTRIBUTION')
250 !!!qceff_table_data(row)%probit_state%dist_type = PARTICLE_FILTER_DISTRIBUTION
251 case default
252 write(errstring, *) 'Invalid distribution type for probit state: ', trim(dist_type_string_probit_state(row))
253 call error_handler(e_err, 'read_qceff_table:', errstring, source)
254 end select
255
256 call to_upper(dist_type_string_probit_extended_state(row))
257
258 select case (trim(dist_type_string_probit_extended_state(row)))
259 case ('NORMAL_DISTRIBUTION')
260 qceff_table_data(row)%probit_extended_state%dist_type = normal_distribution
261 case ('BOUNDED_NORMAL_RH_DISTRIBUTION')
262 qceff_table_data(row)%probit_extended_state%dist_type = bounded_normal_rh_distribution
263 case ('GAMMA_DISTRIBUTION')
264 qceff_table_data(row)%probit_extended_state%dist_type = gamma_distribution
265 case ('BETA_DISTRIBUTION')
266 qceff_table_data(row)%probit_extended_state%dist_type = beta_distribution
267 case ('LOG_NORMAL_DISTRIBUTION')
268 qceff_table_data(row)%probit_extended_state%dist_type = log_normal_distribution
269 case ('UNIFORM_DISTRIBUTION')
270 qceff_table_data(row)%probit_extended_state%dist_type = uniform_distribution
271 !!!case ('PARTICLE_FILTER_DISTRIBUTION')
272 !!!qceff_table_data(row)%probit_extended_state%dist_type = PARTICLE_FILTER_DISTRIBUTION
273 case default
274 write(errstring, *) 'Invalid distribution type for probit extended state: ', trim(dist_type_string_probit_extended_state(row))
275 call error_handler(e_err, 'read_qceff_table:', errstring, source)
276 end select
277
278
279 ! Converting the filter kind (read in from table as a string) to its corresponding int value
280 call to_upper(filter_kind_string(row))
281
282 select case (trim(filter_kind_string(row)))
283 case ('EAKF')
284 qceff_table_data(row)%obs_inc_info%filter_kind = eakf
285 case ('ENKF')
286 qceff_table_data(row)%obs_inc_info%filter_kind = enkf
287 case ('UNBOUNDED_RHF')
288 qceff_table_data(row)%obs_inc_info%filter_kind = unbounded_rhf
289 case ('GAMMA_FILTER')
290 qceff_table_data(row)%obs_inc_info%filter_kind = gamma_filter
291 case ('BOUNDED_NORMAL_RHF')
292 qceff_table_data(row)%obs_inc_info%filter_kind = bounded_normal_rhf
293 case default
294 write(errstring, *) 'Invalid filter kind: ', trim(filter_kind_string(row))
295 call error_handler(e_err, 'read_qceff_table:', errstring, source)
296 end select
297
298end do
299
300call close_file(fileid)
301
302end subroutine read_qceff_table

307subroutine obs_error_info(obs_def, error_variance, &
308 bounded_below, bounded_above, lower_bound, upper_bound)
309
310! Computes information needed to compute error sample for this observation
311! This is called by perfect_model_obs when generating noisy obs
312type(obs_def_type), intent(in) :: obs_def
313real(r8), intent(out) :: error_variance
314logical, intent(out) :: bounded_below, bounded_above
315real(r8), intent(out) :: lower_bound, upper_bound
316
317integer :: obs_type, obs_qty
318integer(i8) :: state_var_index
319type(location_type) :: temp_loc
320
321integer :: qty_loc(1)
322
323if (.not. module_initialized) call init_algorithm_info_mod()
324
325! Get the type of the observation
326obs_type = get_obs_def_type_of_obs(obs_def)
327! If it is negative, it is an identity obs
328if(obs_type < 0) then
329 state_var_index = -1 * obs_type
330 call get_state_meta_data(state_var_index, temp_loc, obs_qty)
331else
332 obs_qty = get_quantity_for_type_of_obs(obs_type)
333endif
334
335! Get the default error variance
336error_variance = get_obs_def_error_variance(obs_def)
337
338!use default values if qceff_table_filename is not in namelist
339if (use_qty_defaults) then
340 bounded_below = .false.; bounded_above = .false.
341 lower_bound = missing_r8; upper_bound = missing_r8
342 return
343endif
344
345!find location of QTY in qceff_table_data structure
346qty_loc = findloc(specified_qtys, obs_qty)
347
348if (qty_loc(1) == 0) then
349 !use default values if QTY is not in table
350 bounded_below = .false.; bounded_above = .false.
351 lower_bound = missing_r8; upper_bound = missing_r8
352
353 else
354 bounded_below = qceff_table_data(qty_loc(1))%obs_error_info%bounded_below
355 bounded_above = qceff_table_data(qty_loc(1))%obs_error_info%bounded_above
356 lower_bound = qceff_table_data(qty_loc(1))%obs_error_info%lower_bound
357 upper_bound = qceff_table_data(qty_loc(1))%obs_error_info%upper_bound
358
359endif
360
361end subroutine obs_error_info

366subroutine probit_dist_info(qty, is_state, is_inflation, dist_type, &
367 bounded_below, bounded_above, lower_bound, upper_bound)
368
369integer, intent(in) :: qty
370logical, intent(in) :: is_state ! True for state variable, false for obs
371logical, intent(in) :: is_inflation ! True for inflation transform
372integer, intent(out) :: dist_type
373logical, intent(out) :: bounded_below, bounded_above
374real(r8), intent(out) :: lower_bound, upper_bound
375
376integer :: qty_loc(1)
377
378! Have input information about the qty of the state or observation being transformed
379! along with additional logical info that indicates whether this is an observation
380! or state variable and about whether the transformation is being done for inflation
381! or for regress.
382
383! Selects the appropriate transform, which is specified in the QCF input table per QTY.
384! At present, the options are NORMAL_DISTRIBUTION, BOUNDED_NORMAL_RH_DISTRIBUTION,
385! GAMMA_DISTRIBUTION, BETA_DISTRIBUTION, LOG_NORMAL_DISTRIBUTION,
386! UNIFORM_DISTRIBUTION
387! If the BNRH is selected then information about the bounds must also be set.
388! For example, if qty corresponds to a sea ice fraction then an appropriate choice
389! would be:
390! bounded_below = .true.; bounded_above = .true.
391! lower_bound = 0.0_r8; upper_bounds = 1.0_r8
392
393! In the long run, may not have to have separate controls for each of the input possibilities
394! However, for now these are things that need to be explored for science understanding
395
396if (.not. module_initialized) call init_algorithm_info_mod()
397
398!use default values if qceff_table_filename is not in namelist
399if (use_qty_defaults) then
400 dist_type = normal_distribution
401 bounded_below = .false.; bounded_above = .false.
402 lower_bound = missing_r8; upper_bound = missing_r8
403 return
404endif
405
406qty_loc = findloc(specified_qtys, qty)
407
408if (qty_loc(1) == 0) then
409 !use default values if QTY is not in table
410 dist_type = normal_distribution
411 bounded_below = .false.; bounded_above = .false.
412 lower_bound = missing_r8; upper_bound = missing_r8
413
414 elseif(is_inflation) then
415 ! Case for inflation transformation
416
417 dist_type = qceff_table_data(qty_loc(1))%probit_inflation%dist_type
418 bounded_below = qceff_table_data(qty_loc(1))%probit_inflation%bounded_below
419 bounded_above = qceff_table_data(qty_loc(1))%probit_inflation%bounded_above
420 lower_bound = qceff_table_data(qty_loc(1))%probit_inflation%lower_bound
421 upper_bound = qceff_table_data(qty_loc(1))%probit_inflation%upper_bound
422
423 elseif(is_state) then
424 ! Case for state variable priors
425
426 dist_type = qceff_table_data(qty_loc(1))%probit_state%dist_type
427 bounded_below = qceff_table_data(qty_loc(1))%probit_state%bounded_below
428 bounded_above = qceff_table_data(qty_loc(1))%probit_state%bounded_above
429 lower_bound = qceff_table_data(qty_loc(1))%probit_state%lower_bound
430 upper_bound = qceff_table_data(qty_loc(1))%probit_state%upper_bound
431
432 else
433 ! This case is for observation (extended state) priors
434
435 dist_type = qceff_table_data(qty_loc(1))%probit_extended_state%dist_type
436 bounded_below = qceff_table_data(qty_loc(1))%probit_extended_state%bounded_below
437 bounded_above = qceff_table_data(qty_loc(1))%probit_extended_state%bounded_above
438 lower_bound = qceff_table_data(qty_loc(1))%probit_extended_state%lower_bound
439 upper_bound = qceff_table_data(qty_loc(1))%probit_extended_state%upper_bound
440
441endif
442
443end subroutine probit_dist_info

508subroutine verify_qceff_table_data()
509
510! Subroutine to ensure that the data in the QCF table is valid
511
512integer :: row
513
514if (use_qty_defaults) return
515
516!Checks that all bounds are valid; currently checks that the lower bound in less than the upper
517!Here we could add more specific checks if we have known limits on the bounds
518do row = 1, size(qceff_table_data)
519
520 if (qceff_table_data(row)%obs_error_info%bounded_below .and. qceff_table_data(row)%obs_error_info%bounded_above) then
521 if(qceff_table_data(row)%obs_error_info%lower_bound > qceff_table_data(row)%obs_error_info%upper_bound) then
522 write(errstring,*) 'Invalid bounds in obs_error_info'
523 call error_handler(e_err, 'verify_qceff_table_data:', errstring, source)
524 endif
525 endif
526 if (qceff_table_data(row)%probit_inflation%bounded_below .and. qceff_table_data(row)%probit_inflation%bounded_above) then
527 if(qceff_table_data(row)%probit_inflation%lower_bound > qceff_table_data(row)%probit_inflation%upper_bound) then
528 write(errstring,*) 'Invalid bounds in probit_inflation'
529 call error_handler(e_err, 'verify_qceff_table_data:', errstring, source)
530 endif
531 endif
532 if(qceff_table_data(row)%probit_state%bounded_below .and. qceff_table_data(row)%probit_state%bounded_above) then
533 if(qceff_table_data(row)%probit_state%lower_bound > qceff_table_data(row)%probit_state%upper_bound) then
534 write(errstring,*) 'Invalid bounds in probit_state'
535 call error_handler(e_err, 'verify_qceff_table_data:', errstring, source)
536 endif
537 endif
538 if(qceff_table_data(row)%probit_extended_state%bounded_below .and. qceff_table_data(row)%probit_extended_state%bounded_above) then
539 if(qceff_table_data(row)%probit_extended_state%lower_bound > qceff_table_data(row)%probit_extended_state%upper_bound) then
540 write(errstring,*) 'Invalid bounds in probit_extended_state'
541 call error_handler(e_err, 'verify_qceff_table_data:', errstring, source)
542 endif
543 endif
