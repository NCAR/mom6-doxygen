87!-------------------------------------------------------------------------------
88!> extract the temperature and humidity observations from a granule
89!> and convert to DART observation format. allow caller to specify
90!> a bounding box and only extract data within that region.
91
92subroutine make_obs_sequence ( seq, granule, lon1, lon2, lat1, lat2, &
93 min_MMR_threshold, top_pressure_level, &
94 row_thin, col_thin, use_NCEP_errs, version)
95
96type(obs_sequence_type), intent(inout) :: seq
97type(airs_granule_type), intent(in) :: granule
98real(r8), intent(in) :: lon1, lon2, lat1, lat2
99real(r8), intent(in) :: min_mmr_threshold, top_pressure_level
100integer, intent(in) :: row_thin, col_thin
101logical, intent(in) :: use_ncep_errs
102integer, intent(in) :: version
103
104type(obs_type) :: obs, prev_obs
105
106integer :: i, irow, icol, ivert, istart
107integer :: days, seconds
108integer :: obs_num, temperature_top_index, humidity_top_index
109integer :: which_vert, tobstype, qobstype
110
111real(r8) :: olon, olat, vloc
112real(r8) :: obs_value, obs_var
113real(r8) :: tqc, qqc
114real(r8) :: midpres, log_lower, log_upper
115
116logical :: is_first_obs
117
118type(time_type) :: obs_time, base_time, pre_time, time
119
120if ( .not. module_initialized ) call initialize_module
121
122! Initialize the obs variables
123call init_obs( obs, 1, 1)
124call init_obs(prev_obs, 1, 1)
125
126! assign each observation the correct observation type
127tobstype = get_index_for_type_of_obs('AIRS_TEMPERATURE')
128qobstype = get_index_for_type_of_obs('AIRS_SPECIFIC_HUMIDITY')
129if ((tobstype < 1) .or. (qobstype < 1)) then
130 msgstring = 'unknown observation type [AIRS_TEMPERATURE, AIRS_SPECIFIC_HUMIDITY]'
131 call error_handler(e_err,'make_obs_sequence',msgstring,source,revision,revdate)
132endif
133
134base_time = set_date(1993, 1, 1, 0, 0, 0) ! reference date: jan 1st, 1993
135
136if (debug) call debug_print_size_check(granule)
137
138! The file contains 'Retrieved Water Vapor Mass Mixing Ratio'. Convert
139! to specific humidity here. Original units: gm/kg - scale to kg/kg first.
140
141where (granule%H2OMMRStd > min_mmr_threshold)
142 q = (granule%H2OMMRStd / 1000.0_r8) / ( 1.0_r8 + (granule%H2OMMRStd/1000.0_r8))
143elsewhere
144 q = 0.0_r8
145endwhere
146
147where (granule%H2OMMRStdErr > 0.0_r8)
148 q_err = (granule%H2OMMRStdErr / 1000.0_r8) / ( 1.0_r8 + (granule%H2OMMRStdErr/1000.0_r8))
149elsewhere
150 q_err = 0.0_r8 ! is this really ok?
151endwhere
152
153! see what the top pressure allowed will be, and compute the index number
154! that is still below it in each profile. that will be become the loop end.
155! default to doing the whole column
156temperature_top_index = size(granule%pressStd)
157tloop: do i = 1, size(granule%pressStd)
158 if (granule%pressStd(i) <= top_pressure_level) then
159 temperature_top_index = i
160 exit tloop
161 endif
162enddo tloop
163if (debug) print *, 'temp_top_index = ', temperature_top_index
164
165! temperature obs are on pressure levels. moisture obs are the mean
166! of the layer bounded by the given pressure level below and the next
167! higher layer above, so make sure we are always one level below the
168! top. the loop further down will use i and i+1 as the bounding
169! levels for the moisture obs.
170humidity_top_index = size(granule%pressH2O) - 1
171mloop: do i = 1, size(granule%pressH2O)
172 if (granule%pressH2O(i) <= top_pressure_level) then
173 humidity_top_index = i - 1
174 exit mloop
175 endif
176enddo mloop
177if (debug) print *, 'humd_top_index = ', humidity_top_index
178
179! loop over all observations within the file
180!------------------------------------------------------------------------------
181
182is_first_obs = .true.
183obs_num = 1
184which_vert = vertispressure
185
186! rows are along-track, stepping in the direction the satellite is moving
187rowloop: do irow=1,airs_ret_geotrack
188
189 ! if we're going to subset rows, we will cycle here
190 if (row_thin > 0) then
191 if (modulo(irow, row_thin) /= 1) cycle rowloop
192 endif
193
194 ! columns are across-track, varying faster than rows.
195 colloop: do icol=1,airs_ret_geoxtrack
196
197 ! if we're going to subset columns, ditto
198 if (col_thin > 0) then
199 if (modulo(icol, col_thin) /= 1) cycle colloop
200 endif
201
202 ! observation lat, lon:
203 olat = granule%Latitude (icol,irow) ! valid range [ -90.00, 90.00]
204 olon = granule%Longitude(icol,irow) ! valid range [-180.00, 180.00]
205
206 ! verify the location is not outside valid limits. AIRS uses -180/180
207 if((olon > 180.0_r8) .or. (olon < -180.0_r8) .or. &
208 (olat > 90.0_r8) .or. (olat < -90.0_r8)) then
209 write(*,*)'WARNING : invalid location. col,row,lon,lat = ', icol,irow,olon,olat
210 cycle colloop
211 endif
212
213 ! reject observations outside the bounding box (allowing wrapping)
214 if(( olat < lat1) .or. ( olat > lat2 ) .or. &
215 (.not. is_longitude_between(olon, lon1, lon2))) cycle colloop
216
217 ! make sure lon is between 0 and 360
218 if (olon < 0) olon = olon + 360.0_r8
219
220 obs_time = base_time + set_time(int(granule%Time(icol, irow)))
221 call get_time(obs_time, seconds, days)
222
223 ! avoid -9999 in the starting vertical level
224 istart = granule%nBestStd(icol, irow)
225 if (istart < 1) goto 100 ! skip vert_T_loop
226
227 vert_t_loop: do ivert=istart, temperature_top_index
228
229 tqc = 0 ! if we get here, the quality control is 'best' == 0
230
231 ! create the temperature obs for this observation, add to sequence
232
233 ! apparently -9999 is missing data, outside of qc mechanism
234 obs_value = granule%TAirStd(ivert, icol, irow)
235 if (obs_value == -9999.0_r8) cycle vert_t_loop
236
237 ! temperature values are located directly at the pressure levels
238 vloc = granule%pressStd(ivert) * mb_to_pa
239
240 if (use_ncep_errs) then
241 obs_var = max(rawin_temp_error(vloc / mb_to_pa), & ! back to mb for this call
242 granule%TAirStdErr(ivert, icol, irow))
243 else
244 obs_var = granule%TAirStdErr(ivert, icol, irow)
245 endif
246 obs_var = obs_var * obs_var
247
248 if (debug) print *, 'obs value, var = ', obs_value, obs_var
249
250 call create_3d_obs(olat, olon, vloc, which_vert, obs_value, airs_temperature, obs_var, &
251 days, seconds, tqc, obs)
252
253 call add_obs_to_seq(seq, obs, time, prev_obs, pre_time, is_first_obs)
254
255 if ( debug ) then
256 write(*,*)irow, icol, ivert, olat, olon
257 endif
258
259 obs_num = obs_num + 1
260
261 enddo vert_t_loop
262
263100 continue
264
265 ! avoid -9999 in the starting vertical layer
266 istart = granule%nSurfStd(icol, irow)
267 if (istart < 1) goto 200 !skip vert_Q_loop
268
269 vert_q_loop: do ivert=istart,humidity_top_index
270
271 if ((version.eq.6 .or. version.eq.7) .and. (granule%H2OMMRStd_QC(ivert, icol, irow) > 0)) cycle vert_q_loop
272
273 qqc = 0 ! if we get here, the quality control is 'Best' == 0
274
275 ! create the moisture obs for this observation, add to sequence
276
277 ! if original MMR data was -9999, that is apparently a missing val
278 if (granule%H2OMMRStd(ivert, icol, irow) == -9999.0_r8) cycle vert_q_loop
279
280 obs_value = q(ivert, icol, irow)
281
282 ! moisture obs are the mean of the layer with the bottom at
283 ! the given pressure. compute the midpoint (in log space)
284 ! between this level and the level above it, and use that for
285 ! the moisture obs location. see AIRS.html for more info on
286 ! layers vs levels.
287 log_lower = log(granule%pressH2O(ivert))
288 log_upper = log(granule%pressH2O(ivert+1))
289 midpres = exp((log_lower + log_upper) / 2.0_r8)
290 vloc = midpres * mb_to_pa
291
292 if (use_ncep_errs) then
293 obs_var = max(q_err(ivert, icol, irow), 0.2_r8) ! ncep routine needs temp & rh but always returns 0.2
294 else
295 obs_var = q_err(ivert, icol, irow)
296 endif
297 obs_var = obs_var * obs_var
298
299
300 call create_3d_obs(olat, olon, vloc, which_vert, obs_value, airs_specific_humidity, obs_var, &
301 days, seconds, qqc, obs)
302
303 call add_obs_to_seq(seq, obs, time, prev_obs, pre_time, is_first_obs)
304
305 if ( debug ) then
306 write(*,*)irow, icol, ivert, olat, olon
307 endif
308
309 obs_num = obs_num + 1
310
311 enddo vert_q_loop
312
313200 continue
314
315 enddo colloop
316enddo rowloop
317
318! Print a little summary
319write(msgstring,*) 'obs used = ', obs_num - 1
320call error_handler(e_msg, ' ', msgstring)
321
322call print_obs_seq_summary(seq)
323
324end subroutine make_obs_sequence

365function initialize_obs_sequence(filecount, sample_factor)
366
367integer, intent(in) :: filecount
368integer, intent(in) :: sample_factor
369type(obs_sequence_type) :: initialize_obs_sequence
370
371type(obs_sequence_type) :: seq
372
373! max possible obs from this one granule.
374integer :: max_num, num_copies, num_qc
375
376max_num = (airs_ret_stdpressurelay * airs_ret_geoxtrack * airs_ret_geotrack + &
377 airs_ret_h2opressurelay * airs_ret_geoxtrack * airs_ret_geotrack ) / sample_factor
378
379num_copies = 1
380num_qc = 1
381
382! Initialize an obs_sequence
383call init_obs_sequence(seq, num_copies, num_qc, max_num * filecount)
384
385! set meta data of obs_seq
386call set_copy_meta_data(seq, 1, 'observation')
387call set_qc_meta_data(seq, 1, 'AIRS QC')
388
389initialize_obs_sequence = seq
390
391end function initialize_obs_sequence
