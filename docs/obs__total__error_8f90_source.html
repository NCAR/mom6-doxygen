186subroutine print_obs_seq_info(seq_in, filename)
187
188! you can get more info by running the obs_diag program, but this
189! prints out a quick table of obs types and counts, overall start and
190! stop times, and metadata strings and counts.
191
192! and this one counts up, if there is a 'posterior ensemble mean' copy,
193! how many are missing_r8 and how many are not. it could also count
194! up the 'DART quality control' settings? start with the latter for now.
195
196type(obs_sequence_type), intent(in) :: seq_in
197character(len=*), intent(in) :: filename
198
199type(obs_type) :: obs, next_obs
200type(obs_def_type) :: this_obs_def
201logical :: is_there_one, is_this_last
202integer :: size_seq_in
203integer :: i
204integer :: this_obs_type
205integer :: type_count(max_defined_types_of_obs), identity_count, qc_count(0:7), qcindex
206integer :: trueindex, meanindex, spreadindex, erritems
207real(r8) :: qcval(1), copyval(1), truth, mean, spread
208real(r8) :: tminusm, spreadsq
209logical :: can_do_error
210
211
212! Initialize counters and sums
213type_count(:) = 0
214identity_count = 0
215qc_count(:) = 0
216tminusm = 0.0_r8
217spreadsq = 0.0_r8
218erritems = 0
219
220size_seq_in = get_num_obs(seq_in)
221if (size_seq_in == 0) then
222 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
223 call error_handler(e_msg,'obs_total_error',msgstring)
224 return
225endif
226
227! Initialize individual observation variables
228call init_obs( obs, get_num_copies(seq_in), get_num_qc(seq_in))
229call init_obs(next_obs, get_num_copies(seq_in), get_num_qc(seq_in))
230
231! find the dart qc copy, if there is one
232qcindex = get_dartqc_index(seq_in, filename)
233
234! find the true_state, ensemble_mean, ensemble_spread if they are there
235trueindex = get_dartcopy_index(seq_in, filename, 'truth')
236meanindex = get_dartcopy_index(seq_in, filename, 'posterior ensemble mean')
237spreadindex = get_dartcopy_index(seq_in, filename, 'posterior ensemble spread')
238
239can_do_error = all( (/trueindex, meanindex, spreadindex/) >= 0 )
240
241! blank line
242call error_handler(e_msg,'',' ')
243
244write(msgstring,*) 'Processing sequence file ', trim(filename)
245call error_handler(e_msg,'',msgstring)
246
247call print_metadata(seq_in, filename)
248
249!-------------------------------------------------------------
250! Start to process obs from seq_in
251!--------------------------------------------------------------
252is_there_one = get_first_obs(seq_in, obs)
253
254if ( .not. is_there_one ) then
255 write(msgstring,*)'no first observation in ',trim(filename)
256 call error_handler(e_msg,'obs_total_error', msgstring)
257endif
258
259! process it here
260is_this_last = .false.
261
262! blank line
263call error_handler(e_msg, '', ' ')
264
265call get_obs_def(obs, this_obs_def)
266call print_time(get_obs_def_time(this_obs_def), ' First timestamp: ')
267! does not work with NO_CALENDAR
268if (cal) call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
269
270obsloop : do while ( .not. is_this_last)
271
272 call get_obs_def(obs, this_obs_def)
273 this_obs_type = get_obs_def_type_of_obs(this_obs_def)
274 if (this_obs_type < 0) then
275 identity_count = identity_count + 1
276 else
277 type_count(this_obs_type) = type_count(this_obs_type) + 1
278 endif
279! print *, 'obs type index = ', this_obs_type
280! if(this_obs_type > 0)print *, 'obs name = ', get_name_for_type_of_obs(this_obs_type)
281 if (qcindex > 0) then
282 call get_qc(obs, qcval, qcindex)
283 qc_count(nint(qcval(1))) = qc_count(nint(qcval(1))) + 1
284 endif
285
286 if (can_do_error) then
287 call get_obs_values(obs, copyval, trueindex)
288 truth = copyval(1)
289 call get_obs_values(obs, copyval, meanindex)
290 mean = copyval(1)
291 call get_obs_values(obs, copyval, spreadindex)
292 spread = copyval(1)
293
294 tminusm = tminusm + (truth - mean)**2.0
295 spreadsq = spreadsq + spread**2.0
296 erritems = erritems + 1
297 endif
298
299 call get_next_obs(seq_in, obs, next_obs, is_this_last)
300 if (.not. is_this_last) then
301 obs = next_obs
302 else
303 call print_time(get_obs_def_time(this_obs_def), ' Last timestamp: ')
304 if (cal) call print_date(get_obs_def_time(this_obs_def), ' calendar Date: ')
305 endif
306
307enddo obsloop
308
309
310write(msgstring, *) 'Number of obs processed : ', size_seq_in
311call error_handler(e_msg, '', msgstring)
312write(msgstring, *) '---------------------------------------------------------'
313call error_handler(e_msg, '', msgstring)
314do i = 1, max_defined_types_of_obs
315 if (type_count(i) > 0) then
316 write(msgstring, '(a32,i8,a)') trim(get_name_for_type_of_obs(i)), &
317 type_count(i), ' obs'
318 call error_handler(e_msg, '', msgstring)
319 endif
320enddo
321if (identity_count > 0) then
322 write(msgstring, '(a32,i8,a)') 'Identity observations', &
323 identity_count, ' obs'
324 call error_handler(e_msg, '', msgstring)
325endif
326if (qcindex > 0) then
327 call error_handler(e_msg, '', ' ')
328 write(msgstring, *) 'DART QC results: '
329 call error_handler(e_msg, '', msgstring)
330 do i=0, 7
331 if (qc_count(i) > 0) then
332 write(msgstring, '(a16,2(i8))') 'DART QC value', i, &
333 qc_count(i)
334 call error_handler(e_msg, '', msgstring)
335 endif
336 enddo
337 write(msgstring, *) 'Total obs: ', sum(qc_count(:))
338 call error_handler(e_msg, '', msgstring)
339endif
340
341call error_handler(e_msg, '', ' ')
342if (can_do_error) then
343 write(msgstring, *) 'Error results: '
344 call error_handler(e_msg, '', msgstring)
345 write(msgstring, *) 'Total error: ', sqrt(tminusm)
346 call error_handler(e_msg, '', msgstring)
347 write(msgstring, *) 'Total spread: ', sqrt(spreadsq)
348 call error_handler(e_msg, '', msgstring)
349else
350 write(msgstring, *) 'No error results printed because one or more of "truth",'
351 call error_handler(e_msg, '', msgstring)
352 write(msgstring, *) '"posterior ensemble mean", and/or "posterior ensemble spread" are missing'
353 call error_handler(e_msg, '', msgstring)
354endif
355
356! another blank line
357call error_handler(e_msg, '', ' ')
358
359! Time to clean up
360
361call destroy_obs( obs)
362call destroy_obs(next_obs)
363
364end subroutine print_obs_seq_info

368subroutine validate_obs_seq_time(seq, filename)
369
370! this eventually belongs in the obs_seq_mod code, but for now
371! try it out here. we just fixed a hole in the interactive create
372! routine which would silently let you create out-of-time-order
373! linked lists, which gave no errors but didn't assimilate the
374! right obs at the right time when running filter. this runs
375! through the times in the entire sequence, ensuring they are
376! monotonically increasing in time. this should help catch any
377! bad files which were created with older versions of code.
378
379type(obs_sequence_type), intent(in) :: seq
380character(len=*), intent(in) :: filename
381
382type(obs_type) :: obs, next_obs
383type(obs_def_type) :: this_obs_def
384logical :: is_there_one, is_this_last
385integer :: size_seq, obs_count
386integer :: key
387type(time_type) :: last_time, this_time
388
389
390! make sure there are obs left to process before going on.
391size_seq = get_num_obs(seq)
392if (size_seq == 0) then
393 msgstring = 'Obs_seq file '//trim(filename)//' is empty.'
394 call error_handler(e_msg,'obs_total_error:validate',msgstring)
395 return
396endif
397
398! Initialize individual observation variables
399call init_obs( obs, get_num_copies(seq), get_num_qc(seq))
400call init_obs(next_obs, get_num_copies(seq), get_num_qc(seq))
401
402obs_count = 0
403
404!-------------------------------------------------------------
405! Start to process obs from seq
406!--------------------------------------------------------------
407is_there_one = get_first_obs(seq, obs)
408
409! we already tested for 0 obs above, so there should be a first obs here.
410if ( .not. is_there_one ) then
411 write(msgstring,*)'no first obs in sequence ' // trim(filename)
412 call error_handler(e_err,'obs_total_error:validate', msgstring, source)
413 return
414endif
415
416is_this_last = .false.
417last_time = set_time(0, 0)
418obsloop : do while ( .not. is_this_last)
419
420 call get_obs_def(obs, this_obs_def)
421 this_time = get_obs_def_time(this_obs_def)
422
423 if (last_time > this_time) then
424 ! bad time order of observations in linked list
425 call print_time(last_time, ' previous timestamp: ')
426 if (cal) call print_date(last_time, ' calendar date: ')
427 call print_time(this_time, ' next timestamp: ')
428 if (cal) call print_date(this_time, ' calendar date: ')
429
430 key = get_obs_key(obs)
431 write(msgstring1,*)'obs number ', key, ' has earlier time than previous obs'
432 write(msgstring2,*)'observations must be in increasing time order, file ' // trim(filename)
433 call error_handler(e_err,'obs_total_error:validate', msgstring2, &
434 source, text2=msgstring1)
435 endif
436
437 last_time = this_time
438 obs_count = obs_count + 1
439
440 call get_next_obs(seq, obs, next_obs, is_this_last)
441 if (.not. is_this_last) obs = next_obs
442
443enddo obsloop
444
445! clean up
446call destroy_obs( obs)
447call destroy_obs(next_obs)
448
449! technically not a time validation, but easy to check. obs_count should never
450! be larger than size_seq - that's a fatal error. obs_count < size_seq would
451! suggest there are obs in the file that aren't part of the linked list.
452! this does not necessarily indicate a fatal error but it's not a common
453! situation and might indicate someone should check on the file.
454if (obs_count /= size_seq) then
455 write(msgstring,*) 'input sequence ', trim(filename)
456 call error_handler(e_msg,'obs_total_error:validate', msgstring)
457
458 write(msgstring,*) 'total obs in file: ', size_seq, ' obs in linked list: ', obs_count
459 if (obs_count > size_seq) then
460 ! this is a fatal error
461 write(msgstring1,*) 'linked list obs_count > total size_seq, should not happen'
462 call error_handler(e_err,'obs_total_error:validate', msgstring, &
463 source, text2=msgstring1)
464 else
465 ! just warning msg
466 write(msgstring1,*) 'only observations in linked list will be processed'
467 call error_handler(e_msg,'obs_total_error:validate', msgstring, &
468 source, text2=msgstring1)
469 endif
470endif
471
472end subroutine validate_obs_seq_time
