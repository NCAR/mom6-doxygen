6c====================================================================
7c Filename: read_qscat2b.f
8c
9c Usage:
10c
11c To run this program, use the following command:
12c
13c your_computer% read_qscat2b <filename>
14c
15c where "<filename>" is the name of the QuikSCAT Level 2B
16c input file
17c
18c Description:
19c
20c This file contains 3 subroutines in order to read the
21c QuikSCAT Level 2B data in Hierarchical Data Format (HDF).
22c The subroutines are as follows.
23c
24c 1. read_attrib_byname(): a subroutine to read the name
25c and value(s) of a global attribute
26c referenced by its name.
27c
28c 2. read_timetags(): a subroutine to read the timetag info
29c contained in the HDF VDATA
30c
31c 3. extract_sds(): a subroutine to read the contents of an
32c SDS from an HDF file
33c
34c NOTES:
35c 1. Please refer all questions concerning this program and
36c QuikSCAT data obtained from the JPL PO.DAAC to
37c qscat@podaac.jpl.nasa.gov.
38c
39c 2. The HDF library must be installed before this program will
40c work properly. The HDF library and further information
41c about HDF may be obtained from the National Center for
42c Supercomputing Applications (NCSA) at http://hdf.ncsa.uiuc.edu.
43c
44c 3. The L2B data are read in their entirety. Examples of reading
45c the QuikSCAT data by slabs can be found in read_qscat1b.f
46c read_qscat2a.f.
47c
48c 7/1/1999 R.S. Dunbar, K.L. Perry
49c
50c Modifications:
51c
52c 11/18/1999 Added capability to read new SDSs,
53c wind_speed_selection and wind_dir_selection.
54c K.L. Perry
55c
56c 1/12/2000 Added capability to read new SDSs,
57c mp_rain_probability and nof_rain_index. K.L. Perry
58c
59c 3/10/2000 Corrected handling of 8-bit unsigned integers
60c in extract_sds subroutine. K.L. Perry
61c
62c 5/15/2000 Corrected handling of 32-bit unsigned integers
63c in extract_sds subroutine. This change should
64c not affect the Level 2A or 2B software. K.L. Perry
65c
66c 7/13/2006 sdrad_rain_rate and adapted code for 12.5 km product reading
67C Ted Lungu

235c====================================================================
236c READ_ATTRIB_BYNAME: a subroutine to read the name and
237c value(s) of a global attribute
238c referenced by its name.
239c
240c 5/14/1998 R.S. Dunbar

242
243 subroutine read_attrib_byname(sd_id,in_attr_name,
244 $ num_type,n_values,fvalues)
245
246 integer MAX_NC_NAME
247 parameter(max_nc_name=256)
248
249 integer sd_id,num_type,n_values
250 integer attr_index,count,retn,n,oldn
251 integer sffattr,sfgainfo,sfrattr
252 character*(*) in_attr_name
253 character*(*) fvalues(*)
254 character attr_name*(MAX_NC_NAME),attr_data*512
255 character*(MAX_NC_NAME) values(20)
256 character cr
257
258c Find the attribute assigned to in_attr_name
259 attr_index = sffattr(sd_id,in_attr_name)
260
261c Get information about the file attribute
262 retn = sfgainfo(sd_id,attr_index,attr_name,num_type,count)
263
264c Read the attribute data
265 retn = sfrattr(sd_id,attr_index,attr_data)
266
267 cr = char(10)
268 ival = 0
269 oldn = 1
270 5 continue
271
272c QuikSCAT attributes have atleast three lines:
273c metadata type, array size and metadata contents
274c Use "blank spaces" to identify the end of a line
275
276 n = index(attr_data(oldn:(count-1)),cr)
277
278c Read all of the metadata lines
279 if (n .eq. 0) then
280 ival=ival+1
281 values(ival) = attr_data(oldn:(count-1))
282 goto 99
283 else
284 ival=ival+1
285 values(ival) = attr_data(oldn:(oldn+n-2))
286 endif
287 oldn=n+oldn
288 goto 5
289
290 99 continue
291 n_values = ival - 2
292 do i=1,n_values
293 fvalues(i) = values(i+2)
294 enddo
295 return
296 end

299c READ_TIMETAGS: a subroutine to read the timetag info

308 subroutine read_timetags(filename,timetags)
309
310 character*80 filename
311 character*21 timetags(*)
312 character*60 fields
313 character vdata_name*30
314 integer file_id,vdata_ref,vdata_id
315 integer n_records,interlace,vdata_size
316 integer hopen,vsfgid,vsfatch,vsfinq,vsfread,vfsdtch,hclose
317
318 integer DFACC_RDONLY,FULL_INTERLACE
319 parameter(dfacc_rdonly=1)
320 parameter(full_interlace=0)
321
322c Open the HDF file
323 file_id = hopen(filename,dfacc_rdonly,0)
324
325c Initialize the VS interface
326 call vfstart(file_id)
327
328c Get the reference number for the first vdata in the file
329 vdata_ref = -1
330 vdata_ref = vsfgid(file_id,vdata_ref)
331
332c Attach to the vdata for reading if it is found, otherwise
333c exit the program.
334 if (vdata_ref.eq.0) then
335 print *,'No Timetags were found in the HDF VDATA'
336 print *,'*** Aborting program ***'
337 stop
338 endif
339
340 vdata_id = vsfatch(file_id,vdata_ref,'r')
341
342c Get n_records
343 retn=vsfinq(vdata_id,n_records,interlace,fields,
344 & vdata_size,vdata_name)
345
346c Read the timetags
347 retn = vsfread(vdata_id,timetags,n_records,full_interlace)
348
349c Terminate access to the vdata and to the VS interface,
350c then close the HDF file.
351
352 retn = vsfdtch(vdata_id)
353 call vfend(file_id)
354 retn = hclose(file_id)
355
356 return
357 end

359c====================================================================
360c EXTRACT_SDS: a subroutine to read the contents of an
361c SDS from an HDF file
362c
363c 5/12/1998 R.S. Dunbar
364c
365c Revisions:
366c 7/1999 Code adapted to read input in bytes as well as ints
367c and floats. Comments were also added. K.L. Perry
368c
369c 3/2000 Corrected code for 8-bit unsigned integers. "buffer"
370c was used instead of "buffer2". K.L. Perry
371c
372c 5/2000 Changed MAX_BUF_SIZE from 1000000 to 10000000.
373c Corrected code for 32-bit unsigned integers. Created
374c "buffer3" array of int*4 to correctly read in uint32.
375c K.L. Perry
376c
377c====================================================================
378 subroutine extract_sds(sd_id,in_var,irec,slab_size,out_var)
379
380 integer MAX_BUF_SIZE
381 parameter(max_buf_size=10000000)
382 integer sd_id,sds_index,sds_id,retn
383 integer rank,dim_sizes(3),data_type,nattrs,num_type
384 integer edge(3),stride(3),start(3),irec,slab_size
385 double precision cal,cal_err,off,off_err
386 integer iprod,i,itmp
387
388 character*(*) in_var
389 character name*256
390 integer sfn2index,sfselect,sfginfo,sfrdata,sfgcal,sfendacc
391
392 integer*2 buffer(MAX_BUF_SIZE)
393 byte buffer2(max_buf_size)
394 integer*4 buffer3(MAX_BUF_SIZE)
395 real out_var(MAX_BUF_SIZE)
396
397c Search for the index of "in_var"
398 sds_index = sfn2index(sd_id, in_var)
399
400c Select data set corresponding to the returned index
401 sds_id = sfselect(sd_id,sds_index)
402 retn = sfginfo(sds_id,name,rank,dim_sizes,data_type,nattrs)
403
404 do i=1,rank
405 edge(i)=dim_sizes(i)
406 start(i)=0
407 stride(i)=1
408 enddo
409 edge(rank)=slab_size
410 start(rank)=irec-1
411
412 iprod=1
413 do i=1,rank
414 iprod=iprod*edge(i)
415 enddo
416
417c Get the calibration and offset values of input
418 retn = sfgcal(sds_id,cal,cal_err,off,off_err,num_type)
419
420c Read Arrays which are not float32 or int8 or uint8 or uint32
421 if ((data_type.ne.5).and.(data_type.ne.20).and.
422 & (data_type.ne.21).and.(data_type.ne.25)) then
423
424c Read the data set into the "buffer" array
425 retn=sfrdata(sds_id,start,stride,edge,buffer)
426
427c Calibrate the output
428 do i=1,iprod
429c Correct for 16-bit unsigned integers
430 if ((data_type.eq.23).and.(buffer(i).lt.0)) then
431 out_var(i)=buffer(i)+65536.0
432
433c No correction needed for signed or positive unsigned integers
434 else
435 out_var(i)=buffer(i)
436 endif
437
438 out_var(i)=out_var(i)*cal
439 enddo
440
441c Read uint32 arrays.
442 else if (data_type.eq.25) then
443
444c Read the data set into the "buffer3" uint32 array
445 retn=sfrdata(sds_id,start,stride,edge,buffer3)
446
447c Calibrate the output
448 do i=1,iprod
449c Correct for 32-bit unsigned integers
450 if ((data_type.eq.25).and.(buffer3(i).lt.0)) then
451 out_var(i)=buffer3(i)+4294967296.0
452 else
453 out_var(i)=buffer3(i)
454 endif
455 out_var(i)=out_var(i)*cal
456 enddo
457
458c Read int8 and uint8 arrays.
459 else if ((data_type.eq.20).or.(data_type.eq.21)) then
460
461c Read the data set into the "buffer2" byte-array
462 retn=sfrdata(sds_id,start,stride,edge,buffer2)
463
464c Calibrate the output
465 do i=1,iprod
466
467c Correct for 8-bit unsigned integers
468 itmp=buffer2(i)
469 if ((data_type.eq.21).and.(buffer2(i).lt.0)) then
470 itmp=itmp+256
471 out_var(i)=itmp
472
473c No correction needed for signed or positive unsigned integers
474 else
475 out_var(i)=itmp
476 endif
477
478 out_var(i)=out_var(i)*cal
479 enddo
480
481 else
482c Read float32 arrays directly into the "out_var" array
483 retn=sfrdata(sds_id,start,stride,edge,out_var)
484
485c Calibrate the output
486 do i=1,iprod
487 out_var(i)=out_var(i)*cal
488 enddo
489 endif
490
491c Terminate access to the array data set.
492 retn = sfendacc(sds_id)
493 end
