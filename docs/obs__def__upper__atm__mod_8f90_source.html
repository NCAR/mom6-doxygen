179!-----------------------------------------------------------------------------
180!>@todo Test RMA.
181! Given DART state vector and a location,
182! it computes thermospheric neutral density [Kg/m3]
183! The istatus variable should be returned as 0 unless there is a problem
184
185subroutine get_expected_upper_atm_density(state_handle, ens_size, location, obs_val, istatus)
186
187type(ensemble_type), intent(in) :: state_handle
188integer, intent(in) :: ens_size
189type(location_type), intent(in) :: location
190real(r8), intent(out) :: obs_val(ens_size)
191integer, intent(out) :: istatus(ens_size)
192
193real(r8) :: mmro1(ens_size), mmro2(ens_size) ! mass mixing ratio
194real(r8) :: mass_reciprocal(ens_size), pressure(ens_size), temperature(ens_size)
195integer :: this_istatus(ens_size)
196logical :: return_now
197
198if ( .not. module_initialized ) call initialize_module
199
200istatus = 0
201
202! Some models (i.e. GITM) have density as part of the state.
203! If it is available, just return it. If density is not state,
204! then we need to create it from its constituents.
205
206call interpolate(state_handle, ens_size, location, qty_density, obs_val, istatus)
207if(any(istatus == 0)) return ! density is part of the state
208
209
210! This part was implemented for TIEGCM. Check the units for use with
211! other models.
212istatus(:) = 0
213call interpolate(state_handle, ens_size, location, qty_atomic_oxygen_mixing_ratio, mmro1, this_istatus)
214call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
215if (return_now) return
216
217call interpolate(state_handle, ens_size, location, qty_molec_oxygen_mixing_ratio, mmro2, this_istatus)
218call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
219if (return_now) return
220
221call interpolate(state_handle, ens_size, location, qty_pressure, pressure, this_istatus)
222call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
223if (return_now) return
224
225call interpolate(state_handle, ens_size, location, qty_temperature, temperature, this_istatus)
226call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
227if (return_now) return
228
229
230! density [Kg/m3] = pressure [N/m2] * M [g/mol] / temperature [K] / R [N*m/K/kmol]
231! where M is the mean molar mass
232! 1/M = sum(wi/Mi) where wi are mass mixing fractions and Mi are individual molar masses
233
234where (istatus == 0)
235 mass_reciprocal = mmro1/o_molar_mass + mmro2/o2_molar_mass + &
236 (1.0_r8-mmro1-mmro2)/n2_molar_mass
237
238 obs_val = pressure / mass_reciprocal / temperature / universal_gas_constant
239endwhere
240
241end subroutine get_expected_upper_atm_density

243!-----------------------------------------------------------------------------
244
245! Given DART state vector and a location,
246! it computes ground GPS vertical total electron content
247! The istatus variable should be returned as 0 unless there is a problem
248!>@todo Is the logic correct in this code on the Trunk
249!> Should you return from the subroutine instead of exiting
250!> the loop at exit LEVELS
251subroutine get_expected_gnd_gps_vtec(state_handle, ens_size, location, obs_val, istatus)
252
253type(ensemble_type), intent(in) :: state_handle
254integer, intent(in) :: ens_size
255type(location_type), intent(in) :: location
256real(r8), intent(out) :: obs_val(ens_size)
257integer, intent(out) :: istatus(ens_size)
258
259! Given a location and the state vector from one of the ensemble members,
260! compute the model-predicted total electron content that would be in the
261! integrated column from an instrument looking straight down at the tangent point.
262! 'istatus' is the return code. 0 is success; any positive value signals an
263! error (different values can be used to indicate different error types).
264! Negative istatus values are reserved for internal use only by DART.
265
266integer :: nAlts, iAlt, this_istatus(ens_size)
267real(r8), dimension(ens_size, MAXLEVELS) :: ALT, IDensityS_ie ! num_ens by num levels
268real(r8) :: loc_vals(3)
269real(r8) :: tec(ens_size)
270type(location_type) :: probe
271logical :: return_now
272integer :: i
273
274if ( .not. module_initialized ) call initialize_module
275
276istatus = 0 ! must be 0 to use track_status()
277
278loc_vals = get_location(location)
279
280nalts = 0
281levels: do ialt=1, size(alt,2)+1
282 ! loop over levels. if we get to one more than the allocated array size,
283 ! this model must have more levels than we expected. increase array sizes,
284 ! recompile, and try again.
285
286 if (ialt > size(alt,2)) then
287 write(string1,'(''more than '',i4,'' levels in the model.'')') maxlevels
288 string2='increase MAXLEVELS in obs_def_upper_atm_mod.f90, rerun preprocess and recompile.'
289 call error_handler(e_err, 'get_expected_gnd_gps_vtec', string1, &
290 source, revision, revdate, text2=string2)
291 endif
292
293 ! At each altitude interpolate the 2D IDensityS_ie to the lon-lat where data
294 ! point is located. After this loop we will have a column centered at the data
295 ! point's lon-lat and at all model altitudes.
296 probe = set_location(loc_vals(1), loc_vals(2), real(ialt, r8), vertislevel) !probe is where we have data
297
298 call interpolate(state_handle, ens_size, probe, qty_density_ion_e, idensitys_ie(:, ialt), this_istatus)
299 call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
300 if (any(istatus /= 0)) exit levels
301
302 call interpolate(state_handle, ens_size, probe, qty_geometric_height, alt(:, ialt), this_istatus)
303
304 call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
305
306 if (any(istatus /= 0)) exit levels
307
308 nalts = nalts+1
309enddo levels
310
311! failed first time through loop - no values to return.
312if (nalts == 0) then
313 obs_val(:) = missing_r8
314 return
315endif
316
317istatus(:) = 0
318
319do i=1,ens_size
320 if (any(idensitys_ie(i,1:nalts) == missing_r8) .or. any(alt(i,1:nalts) == missing_r8)) then
321 ! mark the ensemble member as having failed
322 istatus(i) = 1
323 end if
324end do
325
326! clear the error from the last level and start again?
327tec=0.0_r8 !start with zero for the summation
328
329do ialt = 1, nalts-1 !approximate the integral over the altitude as a sum of trapezoids
330 !area of a trapezoid: A = (h2-h1) * (f2+f1)/2
331 where (istatus == 0) &
332 tec = tec + ( alt(:, ialt+1)-alt(:, ialt) ) * ( idensitys_ie(:, ialt+1)+idensitys_ie(:, ialt) ) /2.0_r8
333enddo
334
335where (istatus == 0)
336 obs_val = tec * 10.0**(-16) !units of TEC are "10^16" #electron/m^2 instead of just "1" #electron/m^2
337elsewhere
338 obs_val = missing_r8
339end where
340
341end subroutine get_expected_gnd_gps_vtec

344
345! First, find the number of levels in the model.
346! Then, loop down through the levels to create a top-down vertical profile.
347! As we do that, we accumulate the amount of N2 and O, stopping when
348! the N2 reaches 10^21 M^-2. This will probably mean only using part
349! of the 'last' layer.
350
351subroutine get_expected_o_n2_ratio(state_handle, ens_size, location, obs_val, istatus)
352
353type(ensemble_type), intent(in) :: state_handle
354integer, intent(in) :: ens_size
355type(location_type), intent(in) :: location
356real(r8), intent(out) :: obs_val(ens_size)
357integer, intent(out) :: istatus(ens_size)
358
359real(r8) :: loc_array(3)
360real(r8) :: loc_lon, loc_lat
361type(location_type) :: loc
362logical :: return_now
363
364real(r8), parameter :: Max_N2_column_density = 1.0e21_r8
365
366real(r8) :: N2_total(ens_size)
367real(r8) :: O_total(ens_size)
368
369real(r8) :: O_mmr(ens_size, MAXLEVELS)
370real(r8) :: O2_mmr(ens_size, MAXLEVELS)
371real(r8) :: pressure(ens_size, MAXLEVELS)
372real(r8) :: temperature(ens_size, MAXLEVELS)
373real(r8) :: heights(ens_size, MAXLEVELS)
374real(r8) :: thickness(ens_size, MAXLEVELS)
375real(r8) :: O_integrated(ens_size)
376real(r8) :: N2_integrated(ens_size)
377
378real(r8), allocatable :: N2_mmr(:, :)
379real(r8), allocatable :: mbar(:, :)
380real(r8), allocatable :: N2_number_density(:, :)
381real(r8), allocatable :: total_number_density(:, :)
382real(r8), allocatable :: O_number_density(:, :)
383
384real(r8), parameter :: k_constant = 1.381e-23_r8 ! m^2 * kg / s^2 / K
385integer :: ilayer, nlevels, nilevels
386integer :: this_istatus(ens_size)
387real(r8) :: layerfraction(ens_size)
388
389if ( .not. module_initialized ) call initialize_module
390
391istatus = 0
392
393call error_handler(e_err, 'get_expected_O_N2_ratio', 'routine not tested', &
394 source, revision, revdate, &
395 text2='routine in obs_def/obs_def_upper_atm_mod.f90', &
396 text3='test and inform the DART development team. Thanks -- Tim.')
397
398if ( .not. module_initialized ) call initialize_module
399
400loc_array = get_location(location) ! loc is in DEGREES
401loc_lon = loc_array(1)
402loc_lat = loc_array(2)
403
404! some variables are defined on interface layers
405
406nilevels = 0
407heights = 0.0_r8
408
409!>@todo FIXME: this is setting the same location for VERTISLEVEL
410!> as the loop below, so one can *not* be filling the interfaces and
411!> the other filling the midpoints.
412
413fillinterfaces : do ilayer = 1,maxlevels
414
415 loc = set_location(loc_lon, loc_lat, real(ilayer,r8), vertislevel)
416
417 call interpolate(state_handle, ens_size, loc, qty_geometric_height, heights(:, ilayer), istatus)
418 if (any(istatus /= 0)) exit fillinterfaces
419
420 nilevels = nilevels + 1
421
422enddo fillinterfaces
423
424
425if (nilevels == 0) then
426 obs_val(:) = missing_r8
427 return
428endif
429
430istatus(:) = 0
431thickness = 0.0_r8
432thickness(:, 1:nilevels-1) = heights(:, 2:nilevels) - heights(:, 1:nilevels-1)
433
434! Some variables are defined on midpoints of the layers
435
436nlevels = 0
437
438!>@todo FIXME: don't we know how many layers there are now?
439fillmidpoints : do ilayer = 1, maxlevels
440
441 loc = set_location(loc_lon, loc_lat, real(ilayer,r8), vertislevel)
442
443 call interpolate(state_handle, ens_size, loc, qty_pressure, pressure(:, ilayer), this_istatus)
444 call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
445 if (any(istatus /= 0)) exit fillmidpoints
446
447 call interpolate(state_handle, ens_size, loc, qty_temperature, temperature(:, ilayer), this_istatus)
448 call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
449 if (return_now) return
450
451 call interpolate(state_handle, ens_size, loc, qty_atomic_oxygen_mixing_ratio, o_mmr(:, ilayer), this_istatus)
452 call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
453 if (return_now) return
454
455 call interpolate(state_handle, ens_size, loc, qty_molec_oxygen_mixing_ratio, o2_mmr(:, ilayer), this_istatus)
456 call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
457 if (return_now) return
458
459 nlevels = nlevels + 1
460
461enddo fillmidpoints
462
463if (nlevels == 0) then
464 obs_val(:) = missing_r8
465 return
466endif
467
468! Check to make sure we have more interfaces than layers.
469!>@todo should this be an error instead of a message?
470
471if (nilevels /= (nlevels+1)) then
472 write(string1,*)'Require there to be 1 more interfaces than midpoints.'
473 write(string2,*)'Found ',nilevels,' interface layers.'
474 write(string3,*)'Found ',nlevels,' midpoint layers.'
475 call error_handler(e_msg,'get_expected_O_N2_ratio', string1, &
476 source, revision, revdate, text2=string2,text3=string3)
477 where (istatus == 0) obs_val = missing_r8
478 where (istatus == 0) istatus = 11
479 return
480endif
481
482! calculate what we can using array notation
483
484allocate(n2_mmr(ens_size, nlevels), mbar(ens_size, nlevels), total_number_density(ens_size, nlevels), &
485 o_number_density(ens_size, nlevels), n2_number_density(ens_size, nlevels))
486
487n2_mmr = 1.0_r8 - o_mmr(:, 1:nlevels) - o2_mmr(:, 1:nlevels)
488 mbar = 1.0_r8/( o2_mmr(:, 1:nlevels)/o2_molar_mass + &
489 o_mmr(:, 1:nlevels)/ o_molar_mass + &
490 n2_mmr(:, 1:nlevels)/n2_molar_mass )
491
492! O_mmr and N2_mmr defined at midpoints, heights defined at interfaces, so the
493! calculated thicknesses apply directly to the O and N2 densities.
494
495total_number_density = pressure(:, 1:nlevels) / (k_constant * temperature(:, 1:nlevels))
496
497 o_number_density = o_mmr(:, 1:nlevels) * mbar / o_molar_mass * total_number_density
498n2_number_density = n2_mmr(:, 1:nlevels) * mbar / n2_molar_mass * total_number_density
499
500if ( 1 == 2 ) then ! DEBUG BLOCK NOT IN USE
501 write(*,*)
502 do ilayer = nlevels,1,-1
503 write(*,*)'DEBUG level, thickness, ens member 1: ',ilayer, thickness(1, ilayer), &
504 o_number_density(1, ilayer), n2_number_density(1, ilayer), &
505 temperature(1, ilayer), total_number_density(1, ilayer), &
506 o2_mmr(1, ilayer), o_mmr(1, ilayer), n2_mmr(1, ilayer), mbar(1, ilayer)
507 enddo
508 write(*,*)
509endif
510
511n2_total = 0.0_r8
512 o_total = 0.0_r8
513
514topdown : do ilayer = nlevels,1,-1
515
516 if (ilayer == 1) then
517 write(string1,*)'Integrated all the way down to the surface.'
518 write(string2,*)'Still do not have ',max_n2_column_density,' nitrogen molecules per m^2'
519 call error_handler(e_msg,'get_expected_O_N2_ratio', string1, &
520 source, revision, revdate, text2=string2)
521 where (istatus == 0) istatus = 2
522 return
523 endif

525 ! integrate over layer thickness
526 o_integrated = o_number_density(:, ilayer) * thickness(:, ilayer)
527 n2_integrated = n2_number_density(:, ilayer) * thickness(:, ilayer)
528
529 where ((n2_total+n2_integrated) >= max_n2_column_density)
530 ! only store part of the final layer so as not to overshoot 10^21 m^-2
531 ! Let y2 == N2_total, y = Max_N2_column_density, y1 = N2_total + N2_integrated
532 ! the layer fraction is (y - y2)/(y1-y2)
533 ! (Max_N2_column_density - N2_total)/(N2_total + N2_integrated - N2_total)
534 layerfraction = (max_n2_column_density - n2_total) / n2_integrated
535 n2_total = n2_total + n2_integrated*layerfraction
536 o_total = o_total + o_integrated*layerfraction
537 elsewhere
538 n2_total = n2_total + n2_integrated
539 o_total = o_total + o_integrated
540 endwhere
541
542 if (any((n2_total+n2_integrated) >= max_n2_column_density)) exit topdown
543
544enddo topdown
545
546where (istatus == 0) obs_val = o_total / n2_total
547
548deallocate(n2_mmr, mbar, total_number_density, o_number_density, n2_number_density)

550end subroutine get_expected_o_n2_ratio

574!-----------------------------------------------------------------------------
575!> Given DART state vector and a location, it computes O+ density [1/cm^3].
576!> The istatus variable should be returned as 0 unless there is a problem.
577!> This function was implemented for WACCM-X.
578!> Check the units for use with other models.
579
580subroutine get_expected_oxygen_ion_density(state_handle, ens_size, location, obs_val, istatus)
581type(ensemble_type), intent(in) :: state_handle
582integer, intent(in) :: ens_size
583type(location_type), intent(in) :: location
584integer, intent(out) :: istatus(ens_size)
585real(r8), intent(out) :: obs_val(ens_size)
586
587real(r8), dimension(ens_size) :: mmr_o1, mmr_o2, mmr_n2, mmr_h1, mmr_op ! mass mixing ratio
588real(r8), dimension(ens_size) :: mbar, pressure, temperature
589integer, dimension(ens_size) :: this_istatus
590real(r8), dimension(3) :: loc_vals
591logical :: return_now
592
593istatus = 0 ! Need to have istatus = 0 for track_status()

595! cam-fv returns volume mixing ratio, not mass mixing ratio. undo for computation below.
596call interpolate(state_handle, ens_size, location, qty_atomic_oxygen_mixing_ratio, mmr_o1, this_istatus)
597call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
598if (return_now) return
599mmr_o1 = mmr_o1 / (molar_mass_dry_air/o_molar_mass)
600
601call interpolate(state_handle, ens_size, location, qty_molec_oxygen_mixing_ratio, mmr_o2, this_istatus)
602call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
603if (return_now) return
604mmr_o2 = mmr_o2 / (molar_mass_dry_air/o2_molar_mass)
605
606call interpolate(state_handle, ens_size, location, qty_atomic_h_mixing_ratio, mmr_h1, this_istatus)
607call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
608if (return_now) return
609mmr_h1 = mmr_h1 / (molar_mass_dry_air/h_molar_mass)
610
611call interpolate(state_handle, ens_size, location, qty_ion_o_mixing_ratio, mmr_op, this_istatus)
612call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
613if (return_now) return
614
615call interpolate(state_handle, ens_size, location, qty_pressure, pressure, this_istatus)
616call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
617if (return_now) return
618
619call interpolate(state_handle, ens_size, location, qty_temperature, temperature, this_istatus)
620call track_status(ens_size, this_istatus, obs_val, istatus, return_now)
621if (return_now) return
622
623!------------------------------------------------------------------------------------------------------
624! Need to get number density (cgs units) from mass mixing ratio (kg/kg).
625! mbar is g/mole, same as rMass units
626! kg/kg * (g/mole)/(g/mole) * (Pa = N/m^2)/((Joules/K = N*m/K) * (K)) = m-3 * 1E-06 = cm-3
627!------------------------------------------------------------------------------------------------------
628! WACCM-X .i file pressure unit is Pa
629
630loc_vals = get_location(location)
631
632where (istatus == 0)
633 mmr_n2 = 1.0_r8 - (mmr_o1 + mmr_o2 + mmr_h1)
634 mbar = 1.0_r8/( mmr_o1/o_molar_mass &
635 + mmr_o2/o2_molar_mass &
636 + mmr_h1/h_molar_mass &
637 + mmr_n2/n2_molar_mass)
638 obs_val = mmr_op * mbar/o_molar_mass * pressure/(kboltz * temperature) * 1.e-06_r8
639end where
640
641end subroutine get_expected_oxygen_ion_density
