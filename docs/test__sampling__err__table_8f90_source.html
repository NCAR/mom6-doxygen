10!> read the entry for a single ensemble_size and print out the
11!> two values, true_correl_mean and alpha. mostly as a test for
12!> accuracy, but also example code for assim_tools to use.
13
14program test_sampling_err_table
15
16use types_mod, only : r8
17use utilities_mod, only : error_handler, e_err, &
18 initialize_utilities, finalize_utilities
19
20use sampling_error_correction_mod, only : get_sampling_error_table_size, &
21 read_sampling_error_correction
22
23implicit none
24
25real(r8), allocatable :: true_correl_mean(:), alpha(:)
26
27integer :: i, requested_ens_size, table_size
28
29!
30! start of executable code
31!
32
33call initialize_utilities('test_sampling_err_table')
34
35print *, 'Enter ensemble size: '
36read *, requested_ens_size
37
38table_size = get_sampling_error_table_size()
39
40allocate(true_correl_mean(table_size), alpha(table_size))
41
42print *, 'Reading sampling error correction factors for ensemble size of ',requested_ens_size
43call read_sampling_error_correction(requested_ens_size, true_correl_mean, alpha)
44
45print *, 'bin, true correlation, and alphas read from database: '
46do i=1, table_size
47 print *, i, true_correl_mean(i), alpha(i)
48enddo
49
50deallocate(true_correl_mean, alpha)
51
52call finalize_utilities()
53
54end program test_sampling_err_table
