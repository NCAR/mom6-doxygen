60!> Create and add attributes to a 'location' dimension and variable.
61
62!>@todo FIXME does the last arg need to be an optional actual dim?
63!>or an additional dim? check obs_seq_verify for usage
64
65subroutine nc_write_location_atts(ncFileID, dimlen, use_dimID, fname)
66
67integer, intent(in) :: ncfileid ! handle to the netcdf file
68integer, intent(in) :: dimlen ! number of locations to be created
69integer, optional, intent(in) :: use_dimid ! if other than locations dim, use this
70character(len=*), optional, intent(in) :: fname ! file name (for error printing purposes)
71
72integer :: locdimid, ldimid, varid
73integer :: rc
74character(len=*), parameter :: context = 'location_io_mod:nc_write_location_atts'
75
76! get an id for the location dimension:
77! if the user passes us in a dimension id,
78! assume it is already created and use it.
79! if they pass in a valid length, create a 'location' dimension.
80! otherwise, use the unlimited dimension
81
82if (present(use_dimid)) then
83 locdimid = use_dimid
84
85else if (dimlen /= 0) then
86 rc = nf90_def_dim(ncid=ncfileid, name='location', len=dimlen, dimid=locdimid)
87 call nc_check(rc, context, 'def_dim:location', fname)
88
89else
90 rc = nf90_inquire(ncfileid, unlimiteddimid=locdimid)
91 call nc_check(rc, context, 'inquire:unlimited_dim', fname)
92endif
93
94! if there is more than a single number associated with each location
95! define the dimension "across" the locations. e.g. len 3 for 3d dimensions.
96if (locationdims > 1) then
97 rc = nf90_def_dim(ncid=ncfileid, name='locdim', len=locationdims, dimid=ldimid)
98 call nc_check(rc, context, 'def_dim:locdim', fname)
99endif
100
101! Define the location variable and attributes
102
103if (locationdims > 1) then
104 rc = nf90_def_var(ncfileid, 'location', xtype=nf90_double, &
105 dimids=(/ ldimid, locdimid /), varid=varid)
106 call nc_check(rc, context, 'def_var2:location', fname)
107else
108 rc = nf90_def_var(ncfileid, 'location', xtype=nf90_double, &
109 dimids=(/ locdimid /), varid=varid)
110 call nc_check(rc, context, 'def_var4:location', fname)
111endif
112
113call nc_add_location_atts(ncfileid, 'location', fname)
114
115end subroutine nc_write_location_atts

121subroutine nc_add_location_atts(ncFileID, varname, fname)
122
123integer, intent(in) :: ncfileid ! handle to the netcdf file
124character(len=*), intent(in) :: varname ! name of location variable
125character(len=*), optional, intent(in) :: fname ! file name (for error printing purposes)
126
127integer :: varid
128integer :: rc, ndims
129character(len=*), parameter :: context = 'location_io_mod:nc_add_location_atts'
130
131integer :: dimids(nf90_max_var_dims)
132
133! find the id of the given variable name
134rc = nf90_inq_varid(ncfileid, varname, varid=varid)
135call nc_check(rc, context, 'inq_varid:'//trim(varname), fname)
136
137rc = nf90_put_att(ncfileid, varid, 'description', 'location coordinates')
138call nc_check(rc, context, 'put_att:description', fname)
139
140rc = nf90_put_att(ncfileid, varid, 'location_type', trim(locationname))
141call nc_check(rc, context, 'put_att:location_type', fname)
142
143rc = nf90_put_att(ncfileid, varid, 'long_name', trim(locationlname))
144call nc_check(rc, context, 'put_att:long_name', fname)
145
146rc = nf90_put_att(ncfileid, varid, 'storage_order', trim(locationstorageorder))
147call nc_check(rc, context, 'put_att:storage_order', fname)
148
149rc = nf90_put_att(ncfileid, varid, 'units', trim(locationunits))
150call nc_check(rc, context, 'put_att:units', fname)
151
152! Some of the locations types need another variable to define
153! the vertical coordinate system being used. If you are writing
154! an array of locations, you also need to write an array of the vertical
155! coordinate system. That array must be the same length as the location
156! array being written.
157
158if (has_vertical_choice()) then
159
160 rc = nf90_inquire_variable(ncfileid, varid, ndims=ndims, dimids=dimids)
161 call nc_check(rc, context, 'inquire_variable:'//trim(varname), fname)
162 call nc_write_location_vert(ncfileid, dimids(ndims), fname)
163
164endif

256subroutine nc_write_single_location(ncFileID, location, locindex, do_vert, fname)
257
258integer, intent(in) :: ncFileID
259type(location_type), intent(in) :: location
260integer, intent(in) :: locindex
261logical, optional, intent(in) :: do_vert
262character(len=*), optional, intent(in) :: fname ! file name (for error printing purposes)
263
264integer :: LocationVarID
265integer :: WhichVertVarID
266real(r8), dimension(LocationDims) :: location_array
267integer, dimension(1) :: intval
268logical :: write_vert
269integer :: rc
270character(len=32) :: context = 'nc_write_single_location'
271
272! only let the user choose to write the vertical coordinate info
273! if the location module has a vertical choice
274write_vert = .false.
275if (present(do_vert)) then
276 if (has_vertical_choice()) write_vert = do_vert
277endif
278
279rc = nf90_inq_varid(ncfileid, 'location', varid=locationvarid)
280call nc_check(rc, context, 'location', fname)
281
282location_array = get_location( location )
283
284if (locationdims > 1) then
285 rc = nf90_put_var(ncfileid, locationvarid, location_array, &
286 start=(/ 1, locindex /), count=(/ locationdims, 1 /) )
287else
288 rc = nf90_put_var(ncfileid, locationvarid, location_array, &
289 start=(/ locindex /), count=(/ 1 /) )
290endif
291call nc_check(rc, context, 'put_var:location', fname)
292
293if (write_vert) then
294 rc = nf90_inq_varid(ncfileid, 'which_vert', varid=whichvertvarid)
295 call nc_check(rc, context, 'inq_varid:which_vert', fname)
296
297 intval = query_location(location, 'WHICH_VERT')
298 rc = nf90_put_var(ncfileid, whichvertvarid, intval, &
299 start=(/ locindex /), count=(/ 1 /) )
300 call nc_check(rc, context, 'put_var:which_vert', fname)
301endif
302
303end subroutine nc_write_single_location

310subroutine nc_write_multiple_locations(ncFileID, locations, loccount, startlocindex, do_vert, fname)
311
312integer, intent(in) :: ncFileID
313type(location_type), intent(in) :: locations(:)
314integer, intent(in) :: loccount
315integer, optional, intent(in) :: startlocindex
316logical, optional, intent(in) :: do_vert
317character(len=*), optional, intent(in) :: fname ! file name (for error printing purposes)
318
319integer :: LocationVarID
320integer :: WhichVertVarID
321real(r8), allocatable :: location_array(:,:)
322integer, allocatable :: intvals(:)
323logical :: write_vert
324integer :: rc, i, starthere
325character(len=32) :: context = 'nc_write_single_location'
326
327! only let the user choose to write the vertical coordinate info
328! if the location module has a vertical choice
329write_vert = .false.
330if (present(do_vert)) then
331 if (has_vertical_choice()) write_vert = do_vert
332endif
333
334starthere = 1
335if (present(startlocindex)) starthere = startlocindex
336
337rc = nf90_inq_varid(ncfileid, 'location', varid=locationvarid)
338call nc_check(rc, context, 'location', fname)
339
340allocate(location_array(locationdims,loccount))
341if (write_vert) allocate(intvals(loccount))
342
343do i=1, loccount
344 location_array(:,i) = get_location( locations(i) )
345 if (write_vert) intvals(i) = query_location(locations(i), 'WHICH_VERT')
346enddo
347
348if (locationdims > 1) then
349 rc = nf90_put_var(ncfileid, locationvarid, location_array, &
350 start=(/ 1, starthere /), count=(/ locationdims, loccount /) )
351else
352 rc = nf90_put_var(ncfileid, locationvarid, location_array, &
353 start=(/ starthere /), count=(/ loccount /) )
354endif
355call nc_check(rc, context, 'put_var:location', fname)
356
357if (write_vert) then
358 rc = nf90_inq_varid(ncfileid, 'which_vert', varid=whichvertvarid)
359 call nc_check(rc, context, 'inq_varid:which_vert', fname)
360
361 rc = nf90_put_var(ncfileid, whichvertvarid, intvals, &
362 start=(/ starthere /), count=(/ loccount /) )
363 call nc_check(rc, context, 'put_var:which_vert', fname)
364endif
365
366deallocate(location_array)
367if (write_vert) deallocate(intvals)
368
369end subroutine nc_write_multiple_locations
