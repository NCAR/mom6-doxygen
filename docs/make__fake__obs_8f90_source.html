11! Lots of goals for this routine.

165write(*,*)'height interfaces = ',hlevel_edges(1:nhlevels+1)

174month = first_bin_center(2)

177minute = first_bin_center(5)
178second = first_bin_center(6)

182
183call static_init_obs_sequence()
184call init_obs(obs, num_copies, num_qc)
185call init_obs(prev_obs, num_copies, num_qc)
186call init_obs(next_obs, num_copies, num_qc)
187first_obs = .true.
188prev_time = time_obs
189
190! create a new, empty obs_seq file. you must give a max limit
191! on number of obs. increase the size if too small.
192call init_obs_sequence(seq, num_copies, num_qc, max_obs)
193call generate_copy_meta_data(ens_size, seq)
194call generate_qc_meta_data( seq )
195
196! order: LEVEL HEIGHT SCALEHEIGHT SURFACE PRESSURE PRESSURE UNDEF
197coords : do verttype = 1,maxvertcoords
198levels : do ilevel = 1,numlevels(verttype)
199!regions : do iregion = 1,MaxRegions
200regions : do iregion = 1,1
201obsloop : do iobs = 1,maxobsperregion

203 call completeobservation(vertlevels(verttype), obs_type_list(verttype), ilevel, iregion, iobs)

205enddo obsloop
206enddo regions
207enddo levels
208enddo coords
209
210if ( get_num_obs(seq) > 0 ) then
211 print *, 'writing obs_seq, obs_count = ', get_num_obs(seq)
212 call write_obs_seq(seq, 'make_fake_obs.final')
213endif
214
215call finalize_utilities()

256Subroutine completeobservation( vertType, obsType, levelindx, regionindx, obsindx )
257
258integer, intent(in) :: vertType ! [VERTISUNDEF ... VERTISPRESSURE]
259integer, intent(in) :: obsType ! [PRESSURE ... EFGWORO]
260integer, intent(in) :: levelindx ! [1 .... N]
261integer, intent(in) :: regionindx ! [1 .... 4]
262integer, intent(in) :: obsindx ! [1 .... MaxObsPerRegion]
263
264integer :: i, copycount
265
266real(r8), dimension(1) :: obs_value, true_value, dartqc
267real(r8), dimension(1) :: prior, prior_mean, prior_spread
268real(r8), dimension(1) :: poste, poste_mean, poste_spread
269
270call filllocation (regionindx, obsindx, lon, lat )
271
272! just put the ob right in the middle ... or on one edge ... or ...
273if ( verttype == vertispressure ) then
274! vert = plevel(levelindx)
275 vert = plevel_edges(levelindx)
276elseif (verttype == vertislevel ) then
277! vert = mlevel(levelindx)
278 vert = mlevel_edges(levelindx)
279elseif (verttype == vertisheight ) then
280! vert = hlevel(levelindx)
281 vert = hlevel_edges(levelindx)
282elseif (verttype == vertisscaleheight ) then
283! vert = hlevel(levelindx)
284 vert = hlevel_edges(levelindx)
285else
286 vert = 3.0_r8
287endif
288
289obsval = 1.0_r8
290oberr = 2.0_r8
291obsqc = 0.0_r8
292
293call create_3d_obs(lat, lon, vert, verttype, obsval, &
294 obstype, oberr, obday, obsec, obsqc, obs)
295
296obs_value(1) = 10.0
297true_value(1) = 9.0
298prior_mean(1) = 8.0
299poste_mean(1) = 8.5
300prior_spread(1) = 1.0
301poste_spread(1) = 0.5
302
303call set_obs_values(obs, obs_value, 1)
304call set_obs_values(obs, true_value, 2)
305call set_obs_values(obs, prior_mean, 3)
306call set_obs_values(obs, poste_mean, 4)
307call set_obs_values(obs, prior_spread, 5)
308call set_obs_values(obs, poste_spread, 6)
309
310copycount = 6
311
312! Set up obs ensemble members as requested
313do i = 1, ens_size
314 prior(1) = real(i,r8)
315 poste(1) = real(i,r8)
316 copycount = copycount + 1
317 call set_obs_values(obs, prior, copycount)
318 copycount = copycount + 1
319 call set_obs_values(obs, poste, copycount)
320end do
321
322! Lets make at least one ob have a bad DART value.
323if (obsindx == 5) then
324 dartqc(1) = 7.0_r8
325else
326 dartqc(1) = 0.0_r8
327endif
328call set_qc(obs, dartqc, index_qc_dart)
329
330call add_obs_to_seq(seq, obs, time_obs, prev_obs, prev_time, first_obs)
331
332end Subroutine completeobservation

348Subroutine filllocation (regionindx, obsindx, lon, lat )
349integer, intent(in) :: regionindx, obsindx
350real(r8), intent(out) :: lon, lat
351
352real(r8) :: dlat, dlon
353
354dlat = (latlim2(regionindx) - latlim1(regionindx))/10.0_r8
355dlon = (lonlim2(regionindx) - lonlim1(regionindx))/10.0_r8
356
357if ( obsindx == 1) then ! Upper left corner of region
358 lat = latlim2(regionindx) - dlat
359 lon = lonlim1(regionindx) + dlon
360elseif (obsindx == 2) then ! Upper right corner of region
361 lat = latlim2(regionindx) - dlat
362 lon = lonlim2(regionindx) - dlon
363elseif (obsindx == 2) then ! lower right corner of region
364 lat = latlim1(regionindx) + dlat
365 lon = lonlim2(regionindx) - dlon
366elseif (obsindx == 4) then ! lower left corner of region
367 lat = latlim1(regionindx) + dlat
368 lon = lonlim1(regionindx) + dlon
369else ! dead center
370 lat = (latlim1(regionindx) + latlim2(regionindx)) / 2.0_r8
371 lon = (lonlim1(regionindx) + lonlim2(regionindx)) / 2.0_r8
372endif
373end Subroutine filllocation

377Subroutine setpressurelevels(layerMiddles, layerEdges, nLayers)
378
379! This is a utility routine ... presumes user input is minimalist
380! and correct. Decreasing, no duplicate values ... etc.
381
382real(r8), dimension(:), intent(inout) :: layerMiddles
383real(r8), dimension(:), intent(inout) :: layerEdges
384integer, intent( out) :: nLayers
385
386integer :: i, n
387
388! Count number of layer midpoints specified in namelist
389n = 0
390levelloop : do i = 1,maxlevels
391 if (layermiddles(i) == missing_r8 ) exit levelloop
392 n = n + 1
393enddo levelloop
394
395if ((n == 0) .or. (n == maxlevels) ) then
396 write(string1,*)n,' is too many plevel values - max is ',maxlevels,' (MaxLevels)'
397 call error_handler(e_err,'SetPressureLevels', string1,source,revision,revdate)
398endif
399
400! Set the layer edges ...
401 layeredges( 1) = layermiddles(1) - (layermiddles(2) - layermiddles( 1 ))/2.0_r8
402 layeredges(n+1) = layermiddles(n) + (layermiddles(n) - layermiddles(n-1))/2.0_r8
403do i = 2,n
404 layeredges( i ) = layermiddles(i-1) - (layermiddles(i) - layermiddles(i-1))/2.0_r8
405enddo
406
407layeredges( 1 ) = min(1025.0_r8, layeredges( 1 ))
408layeredges(n+1) = max( 0.0_r8, layeredges(n+1))
409
410nlayers = n
411
412end Subroutine setpressurelevels
