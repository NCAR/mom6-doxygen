453!----------------------------------------------------------------------
454!> configure the tests to run
455
456!> if they used the existing 'test1thru', use that to select tests 1-N.
457!> otherwise:
458!> run_tests(:) is initialized to all -1. if the user didn't set
459!> test1thru and if run_tests() is still all -1, turn everything on.
460!> otherwise if the first entry isn't -1, they did specify
461!> something in that namelist entry and we turn on just those tests.
462
463subroutine setup_run_array()
464
465integer :: i
466
467tests_to_run(:) = .false.
468
469! be backwards compatible - set this to -1 to disable.
470
471if (test1thru > 0) then
472 if (test1thru > max_tests) then
473 write(string1, *) 'test1thru must be between 1 and ', max_tests, '; found value ', test1thru
474 call error_handler(e_err, 'model_mod_check: setup_run_array', string1, source)
475 endif
476
477 tests_to_run(1:test1thru) = .true.
478 return
479endif
480
481! Enable specific tests.
482! the first -1 ends the list.
483
484do i=1, max_tests
485 if (run_tests(i) <= 0) exit
486
487 if (run_tests(i) > max_tests) then
488 write(string1, *) 'test numbers must be between 1 and ', max_tests, '; found value ', run_tests(i)
489 call error_handler(e_err, 'model_mod_check: setup_run_array', string1, source)
490 endif
491
492 tests_to_run(run_tests(i)) = .true.
493enddo
494
495! Make sure they are running something
496
497if (run_tests(1) == -1) then
498 write(string1, *) 'No tests selected from the namelist.'
499 write(string2, *) 'Either specify "test1thru" to be a positive number - or -'
500 write(string3, *) 'specify a list of tests to run in "run_tests".'
501 call error_handler(e_err, 'model_mod_check: setup_run_array', string1, source, text2=string2, text3=string3)
502endif
503
504! enforce and report on unfulfilled dependencies
505
506if ((tests_to_run(4) .or. tests_to_run(5)) .and. .not. tests_to_run(2)) then
507 write(string1, *) 'The interpolation tests (Test 4, Test 5) need a model state,'
508 write(string2, *) 'so Test 2 must be run.'
509 call error_handler(e_msg, 'model_mod_check: setup_run_array', string1, &
510 text2=string2)
511 tests_to_run(2) = .true.
512endif
513
514end subroutine setup_run_array

537!------------------------------------------------------------------
538!> report the metadata for every element in the state. This can be
539!> very slow but is sometimes useful to determine the exact location
540!> for specific indices and to confirm the packing order.
541
542subroutine check_all_meta_data()
543
544integer(i8) :: iloc
545type(location_type) :: loc
546integer :: ix, iy, iz, dom_id, qty_index, var_type, fid
547character(len=256) :: qty_string, metadata_qty_string
548
549fid = open_file(all_metadata_file)
550
551do iloc = 1,model_size
552
553 call get_model_variable_indices(iloc, ix, iy, iz, &
554 dom_id=dom_id, &
555 kind_index=qty_index, &
556 kind_string=qty_string)
557
558 ! CLM has (potentially many) columns and needs i7 ish precision
559! write(string1,'(i11,1x,''i,j,k'',3(1x,i7),'' domain '',i2)') &
560! iloc, ix, iy, iz, dom_id
561 ! EL: integer to short for the new I/O method
562 ! Change to long int to avoid problems
563 write(string1,'(i21,1x,''i,j,k'',3(1x,i21),'' domain '',i2)') &
564 iloc, ix, iy, iz, dom_id
565 call get_state_meta_data(iloc, loc, var_type)
566 metadata_qty_string = trim(get_name_for_quantity(var_type))
567
568 if (trim(qty_string) /= trim(metadata_qty_string) ) then
569 write(string2,*)' expected quantity of "'//trim(qty_string)//'"'
570 write(string3,*)' got quantity of "'//trim(metadata_qty_string)//'"'
571 call error_handler(e_err, 'check_all_meta_data', string1, source, &
572 text2=string2, text3=string3)
573 endif
574
575 call write_location(0,loc,charstring=string2)
576
577 write(string3,'(A,1x,I4,1x,A33,1x,A)') &
578 trim(string1), var_type, trim(qty_string), trim(string2)
579
580 if ( do_output() ) write(fid,'(A)') trim(string3)
581 if ( do_output() .and. mod(iloc,int(100000,i8)) == 0) write( * ,'(A)') trim(string3)
582
583enddo
584
585call close_file(fid)
586
587end subroutine check_all_meta_data

591subroutine do_read_test(ens_handle)
592
593type(ensemble_type), intent(inout) :: ens_handle
594
595! Allocate space for file arrays. contains a matrix of files (num_ens x num_domains)
596! If perturbing from a single instance the number of input files does not have to
597! be ens_size but rather a single file (or multiple files if more than one domain)
598
599allocate(file_array_input( num_ens, num_domains))
600file_array_input = reshape(input_state_files, (/num_ens, num_domains/))
601
602! Test the read portion.
603call io_filenames_init(file_info_input, &
604 ncopies = num_ens, &
605 cycling = single_file, &
606 single_file = single_file, &
607 restart_files = file_array_input)
608
609do imem = 1, num_ens
610 write(my_base,'(A,I2)') 'inens_', imem
611 write(my_desc,'(A,I2)') 'input ens', imem
612 call set_file_metadata(file_info_input, &
613 cnum = imem, &
614 fnames = file_array_input(imem,:), &
615 basename = my_base, &
616 desc = my_desc)
617
618 call set_io_copy_flag(file_info_input, &
619 cnum = imem, &
620 io_flag = read_copy)
621enddo
622
623input_restart_files = get_stage_metadata(file_info_input)
624
625do idom = 1, num_domains
626 do imem = 1, num_ens
627 write(string1, *) 'Reading File : ', trim(get_restart_filename(input_restart_files, imem, domain=idom))
628 call print_info_message(string1)
629 enddo
630enddo
631
632call read_state(ens_handle, file_info_input, read_time_from_file, model_time)
633
634deallocate(file_array_input)
635
636end subroutine do_read_test

746

802 if (present(msg2)) write(*,'(2A)') ' -- ', trim(msg2)
