215 subroutine read_cosmos_metadata(key, obsID, ifile, fform)
216!----------------------------------------------------------------------
217!subroutine read_cosmos_metadata(obs_def%key, key, ifile, fform)
218!
219! This routine reads the metadata for neutron intensity observations.
220!
221integer, intent(out) :: key ! index into local metadata
222integer, intent(in) :: obsid
223integer, intent(in) :: ifile
224character(len=*), intent(in), optional :: fform
225
226! temp variables
227logical :: is_asciifile
228integer :: ierr
229character(len=6) :: header
230integer :: oldkey
231real(r8) :: bd, lattwat, n, alpha, l1, l2, l3, l4
232
233if ( .not. module_initialized ) call initialize_module
234
235is_asciifile = ascii_file_format(fform)
236
237write(string2,*)'observation #',obsid
238
239if ( is_asciifile ) then
240 read(ifile, *, iostat=ierr) header
241 call check_iostat(ierr,'read_cosmos_metadata','header',string2)
242 if (trim(header) /= trim(cosmosstring)) then
243 write(string1,*)"Expected neutron_intensity header ["//cosmosstring//"] in input file, got ["//header//"]"
244 call error_handler(e_err, 'read_cosmos_metadata', string1, source, revision, revdate, text2=string2)
245 endif
246 read(ifile, *, iostat=ierr) bd, lattwat, n, alpha
247 call check_iostat(ierr,'read_cosmos_metadata','bd -> alpha',string2)
248 read(ifile, *, iostat=ierr) l1, l2, l3, l4
249 call check_iostat(ierr,'read_cosmos_metadata','L1 -> L4',string2)
250 read(ifile, *, iostat=ierr) oldkey
251 call check_iostat(ierr,'read_cosmos_metadata','oldkey',string2)
252else
253 read(ifile, iostat=ierr) header
254 call check_iostat(ierr,'read_cosmos_metadata','header',string2)
255 if (trim(header) /= trim(cosmosstring)) then
256 write(string1,*)"Expected neutron_intensity header ["//cosmosstring//"] in input file, got ["//header//"]"
257 call error_handler(e_err, 'read_cosmos_metadata', string1, source, revision, revdate, text2=string2)
258 endif
259 read(ifile, iostat=ierr) bd, lattwat, n, alpha
260 call check_iostat(ierr,'read_cosmos_metadata','bd -> alpha',string2)
261 read(ifile, iostat=ierr) l1, l2, l3, l4
262 call check_iostat(ierr,'read_cosmos_metadata','L1 -> L4',string2)
263 read(ifile, iostat=ierr) oldkey
264 call check_iostat(ierr,'read_cosmos_metadata','oldkey',string2)
265endif
266
267! The oldkey is thrown away.
268
269! Store the metadata in module storage and record the new length of the metadata arrays.
270call set_cosmos_metadata(key, bd, lattwat, n, alpha, l1, l2, l3, l4)
271
272! The new 'key' is returned.
273
274end subroutine read_cosmos_metadata

343function interactive(str1,minvalue,maxvalue)
344real(r8) :: interactive
345character(len=*), intent(in) :: str1
346real(r8), optional, intent(in) :: minvalue
347real(r8), optional, intent(in) :: maxvalue
348
349integer :: i
350
351interactive = missing_r8
352
353! Prompt with a minimum amount of error checking
354
355if (present(minvalue) .and. present(maxvalue)) then
356
357 interactive = minvalue - 1.0_r8
358 minmaxloop : do i = 1,10
359 if ((interactive >= minvalue) .and. (interactive <= maxvalue)) exit minmaxloop
360 write(*, *) 'Enter '//str1
361 read( *, *) interactive
362 end do minmaxloop
363
364elseif (present(minvalue)) then
365
366 interactive = minvalue - 1.0_r8
367 minloop : do i=1,10
368 if (interactive >= minvalue) exit minloop
369 write(*, *) 'Enter '//str1
370 read( *, *) interactive
371 end do minloop
372
373elseif (present(maxvalue)) then
374
375 interactive = maxvalue + 1.0_r8
376 maxloop : do i=1,10
377 if (interactive <= maxvalue) exit maxloop
378 write(*, *) 'Enter '//str1
379 read( *, *) interactive
380 end do maxloop
381
382else ! anything goes ... cannot check
383 write(*, *) 'Enter '//str1
384 read( *, *) interactive
385endif
386

391 subroutine get_expected_neutron_intensity(state_handle, ens_size, location, key, val, istatus)
392!----------------------------------------------------------------------
393! Uses a weighting function calculated by COSMIC (COsmic-ray Soil
394! Moisture Interaction Code)
395
396type(ensemble_type), intent(in) :: state_handle
397integer, intent(in) :: ens_size
398type(location_type), intent(in) :: location ! location of obs
399integer, intent(in) :: key ! key into module metadata
400real(r8), intent(out) :: val(ens_size) ! value of obs
401integer, intent(out) :: istatus(ens_size) ! status of the calculation
402
403!========================================================================================
404! COsmic-ray Soil Moisture Interaction Code (COSMIC) - Version 1.5
405!
406! W. James Shuttleworth and Rafael Rosolem - January/2012
407! Fortran code developed by Rafael Rosolem
408!========================================================================================
409! Updates:
410! 01/20/2012 - Version 1.0: * Original version based on SPAM
411! 01/28/2012 - Version 1.1: * Some parameters are re-defined for better physical realism
412! 02/17/2012 - Version 1.2: * After contribution from all angles are taken, need to
413! multiply fastflux by 2/PI
414! * Angle increments can be specified here (ideg)
415! resolution
416! 02/29/2012 - Version 1.3: * Reduced number of parameters based on relationship of ns
417! and nw (now given as alpha = ns/nw)
418! 04/03/2012 - Version 1.4: * Soil thickness (i.e., input.dat file) needs to be specified
419! at finer resolution (i.e., 0.1 cm)
420! 04/04/2012 - Version 1.5 * Now the contributions to soil and water densities/mass are
421! taken to be at the center of a given soil layer
422!=================================================================================
423! COSMIC: Variables list
424!=================================================================================
425
426!rr: Use 1 mm layer increments (down to 3 meters) to compute the weighting function
427!rr: (as originally done for COSMIC), and then compute the cumulative weights for
428!rr: individual soil layers
429
430integer, parameter :: nlyr=3000 ! Total number of soil layers - each 1mm for 3m
431
432real(r8) :: bd = 0.0_r8 ! Dry soil bulk density (g/m3)
433real(r8) :: vwclat = 0.0_r8 ! Volumetric "lattice" water content (m3/m3)
434real(r8) :: n = 0.0_r8 ! High energy neutron flux (-)
435real(r8) :: alpha = 0.0_r8 ! Ratio of Fast Neutron Creation Factor (Soil to Water), alpha (-)
436real(r8) :: l1 = 0.0_r8 ! High Energy Soil Attenuation Length (g/cm2)
437real(r8) :: l2 = 0.0_r8 ! High Energy Water Attenuation Length (g/cm2)
438real(r8) :: l3 = 0.0_r8 ! Fast Neutron Soil Attenuation Length (g/cm2)
439real(r8) :: l4 = 0.0_r8 ! Fast Neutron Water Attenuation Length (g/cm2)
440real(r8) :: zdeg
441real(r8) :: zrad
442real(r8) :: ideg
443real(r8) :: costheta
444real(r8) :: dtheta
445real(r8), dimension(ens_size) :: totflux ! Total flux of above-ground fast neutrons
446
447real(r8), dimension(:,:), allocatable :: dz ! Soil layers (cm)
448real(r8), dimension(:,:), allocatable :: zthick ! Soil layer thickness (cm)
449real(r8), dimension(:,:), allocatable :: vwc ! Volumetric Water Content (m3/m3)
450real(r8), dimension(:,:), allocatable :: isoimass ! Integrated dry soil mass above layer (g)
451real(r8), dimension(:,:), allocatable :: iwatmass ! Integrated water mass above layer (g)
452real(r8), dimension(:,:), allocatable :: hiflux ! High energy neutron flux
453real(r8), dimension(:,:), allocatable :: fastpot ! Fast neutron source strength of layer
454real(r8), dimension(:,:), allocatable :: h2oeffdens ! "Effective" density of water in layer (g/cm3)
455real(r8), dimension(:,:), allocatable :: idegrad ! Integrated neutron degradation factor (-)
456real(r8), dimension(:,:), allocatable :: fastflux ! Contribution to above-ground neutron flux
457
458!rr: Not needed for DART
459!rr: real(r8), dimension(:), allocatable :: normfast ! Normalized contribution to neutron flux (-) [weighting factors]
460
461real(r8), parameter :: h2odens = 1000.0_r8 ! Density of water (g/cm3)
462
463integer, parameter :: maxlayers = 1000 ! more than the maximum # of model soil layers
464real(r8), allocatable :: layerz(:,:) ! original soil layer depths
465real(r8), allocatable :: soil_moisture(:,:) ! original soil layer moistures
466
467integer :: angle, angledz, maxangle ! loop indices for an integration interval
468integer :: i, zi, nlevels
469real(r8) :: loc_array(3)
470real(r8) :: loc_lon, loc_lat
471real(r8) :: loc_value(ens_size)
472type(location_type) :: loc
473integer :: imem
474integer :: loc_value_istatus(ens_size), loc_istatus(ens_size), layerz_istatus(ens_size)
475logical :: return_now
476
477!=================================================================================
478
479if ( .not. module_initialized ) call initialize_module
480
481val = 0.0_r8 ! set return value early
482
483!=================================================================================
484! COSMIC: Site specific-parameters come from the observation metadata
485!=================================================================================
486
487! Make sure the desired key is within the length of the metadata arrays.
488call key_within_range(key,'get_expected_neutron_intensity')
489
490bd = observation_metadata(key)%bd
491vwclat = observation_metadata(key)%lattwat
492n = observation_metadata(key)%N
493alpha = observation_metadata(key)%alpha
494l1 = observation_metadata(key)%L1
495l2 = observation_metadata(key)%L2
496l3 = observation_metadata(key)%L3
497l4 = observation_metadata(key)%L4
498
499!=================================================================================
500! Determine the number of soil layers and their depths
501! using only the standard DART interfaces to model
502! set the locations for each of the model levels
503!=================================================================================
504
505loc_array = get_location(location) ! loc is in DEGREES
506loc_lon = loc_array(1)
507loc_lat = loc_array(2)
508
509nlevels = 0
510countlevels : do i = 1,maxlayers
511 loc = set_location(loc_lon, loc_lat, real(i,r8), vertislevel)
512 call interpolate(state_handle, ens_size, loc, qty_geopotential_height, loc_value, loc_istatus)
513 if ( any(loc_istatus /= 0 ) ) exit countlevels
514 nlevels = nlevels + 1
515enddo countlevels
516
517if ((nlevels == maxlayers) .or. (nlevels == 0)) then
518 write(string1,*) 'FAILED to determine number of soil layers in model.'
519 if (debug) call error_handler(e_msg,'get_expected_neutron_intensity',string1,source,revision,revdate)
520 istatus = 1
521 val = missing_r8
522 return
523else
524! write(*,*)'get_expected_neutron_intensity: we have ',nlevels,' soil layers'
525endif
526
527! Now actually find the depths at each level
528! While we're at it, might as well get the soil moisture there, too.
529
530allocate(layerz(ens_size, nlevels),&
531 soil_moisture(ens_size, nlevels))
532
533! Set all of the istatuses back to zero for track_status
534istatus = 0
535
536findlevels : do i = 1,nlevels
537 loc = set_location(loc_lon, loc_lat, real(i,r8), vertislevel)
538 call interpolate(state_handle, ens_size, loc, qty_geopotential_height, layerz(:,i), layerz_istatus)
539 call track_status(ens_size, layerz_istatus, val, istatus, return_now)
540 if (return_now) return
541
542 loc = set_location(loc_lon, loc_lat, layerz(1,i), vertisheight)
543 call interpolate(state_handle, ens_size, loc, qty_soil_moisture, loc_value, loc_value_istatus)
544 call track_status(ens_size, loc_value_istatus, val, istatus, return_now)
545 if ( any(loc_value_istatus /=0) ) then
546 write(string1,*) 'FAILED to determine soil moisture for layer',i
547 where (loc_value_istatus /= 0) istatus = 2
548 if (return_now) return
549 endif
550
551 soil_moisture(:,i) = loc_value
552
553enddo findlevels
554
555!rr: DART soil layers are negative and given in meters while COSMIC needs them to be
556!rr: positive and in centimeters
557
558if ( all(layerz < 0.0_r8) ) then
559 layerz = -1.0_r8 * 100.0_r8 * layerz
560elseif ( any(layerz < 0.0_r8) ) then
561 write(string1,*) 'unusual values of soil layers in model.'
562 call error_handler(e_err,'get_expected_neutron_intensity',string1,source,revision,revdate)
563endif
564
565!=================================================================================
566! COSMIC: Allocate arrays and initialize variables
567!=================================================================================
568
569allocate( dz(ens_size,nlyr), &
570 zthick(ens_size,nlyr), &
571 vwc(ens_size,nlyr), &
572 hiflux(ens_size,nlyr), &
573 fastpot(ens_size,nlyr), &
574 h2oeffdens(ens_size,nlyr), &
575 idegrad(ens_size,nlyr), &
576 fastflux(ens_size,nlyr), &
577 isoimass(ens_size,nlyr), &
578 iwatmass(ens_size,nlyr) )
579
580totflux(:) = 0.0_r8
581
582do i = 1,nlyr
583
584 dz(:,i) = real(i,r8)/10.0_r8 ! 0.1 cm intervals ... for now.
585 zthick(:,i) = 0.0_r8
586 h2oeffdens(:,i) = 0.0_r8
587 vwc(:,i) = 0.0_r8
588 isoimass(:,i) = 0.0_r8
589 iwatmass(:,i) = 0.0_r8
590 hiflux(:,i) = 0.0_r8
591 fastpot(:,i) = 0.0_r8
592 idegrad(:,i) = 0.0_r8
593 fastflux(:,i) = 0.0_r8
594
595 !rr: Not needed for DART
596 !rr: normfast(i) = 0.0_r8
597
598enddo
599
600!=================================================================================
601! Get soil moisture from individual model layers and assign them to
602! 1 mm intervals (down to 3 meters)
603
604do i = 1,nlyr
605 !>@todo JH can we some how fix this logic to not include an imem
606 !> loop? can not use WHERE () when there is an exit.
607 do imem = 1,ens_size
608 soil : do zi = 1,nlevels
609 if (dz(imem,i) >= layerz(imem, nlevels)) then
610 vwc(imem, i) = soil_moisture(imem, nlevels)
611 exit soil
612 elseif (dz(imem, i) <= layerz(imem, zi)) then
613 vwc(imem, i) = soil_moisture(imem, zi) ! soil moisture (m3/m3)
614 exit soil
615 endif
616 enddo soil
617 enddo
618enddo
619
620deallocate(layerz ,soil_moisture)
621
622!=================================================================================
623! COSMIC: Neutron flux calculation
624!=================================================================================
625
626! At some point, you might want to tinker around with non-uniform
627! soil layer thicknesses.
628zthick(:, 1) = dz(:, 1) - 0.0_r8 ! Surface layer
629do i = 2,nlyr
630 zthick(:, i) = dz(:, i) - dz(:, i-1) ! Remaining layers
631enddo
632
633! Angle distribution parameters (HARDWIRED)
634!rr: Using 0.5 deg angle intervals appears to be sufficient
635!rr: (smaller angles increase the computing time for COSMIC)
636ideg = 0.5_r8 ! ideg ultimately controls the number of trips through
637angledz = nint(ideg*10.0_r8) ! the ANGLE loop. Make sure the 10.0_r8 is enough
638maxangle = 900 - angledz ! to create integers with no remainder
639dtheta = ideg*(pi/180.0_r8)
640
641if ( real(angledz,r8) /= ideg*10.0_r8 ) then
642 write(string1,*) 'ideg*10.0 must result in an integer - it results in ',ideg*10.0_r8
643 call error_handler(e_err,'get_expected_neutron_intensity',string1,source,revision,revdate)
644endif
645
646do i = 1,nlyr
647
648 !> @todo put these computations in a routine and call it
649 !> only if istatus is ok.
650
651 ! High energy neutron downward flux
652 ! The integration is now performed at the node of each layer (i.e., center of the layer)
653 h2oeffdens(:,i) = ((vwc(:,i)+vwclat)*h2odens)/1000.0_r8
654
655 if(i > 1) then
656 ! Assuming an area of 1 cm2
657 isoimass(:,i) = isoimass(:,i-1) + bd*(0.5_r8*zthick(:,i-1))*1.0_r8 + &
658 bd*(0.5_r8*zthick(:,i ))*1.0_r8
659 ! Assuming an area of 1 cm2
660 iwatmass(:,i) = iwatmass(:,i-1) + h2oeffdens(:, i-1)*(0.5_r8*zthick(:, i-1))*1.0_r8 + &
661 h2oeffdens(:, i )*(0.5_r8*zthick(:, i ))*1.0_r8
662 else
663 isoimass(:, i) = bd*(0.5_r8*zthick(:, i))*1.0_r8 ! Assuming an area of 1 cm2
664 iwatmass(:, i) = h2oeffdens(:, i)*(0.5_r8*zthick(:, i))*1.0_r8 ! Assuming an area of 1 cm2
665 endif
666
667 !>@todo FIXME JH: probably will not do pointwise exponentiation
668 hiflux( :, i) = n*exp(-(isoimass(:, i)/l1 + iwatmass(:, i)/l2) )
669 fastpot(:, i) = zthick(:, i)*hiflux(:, i)*(alpha*bd + h2oeffdens(:, i))
670
671 ! This second loop needs to be done for the distribution of angles for fast neutron release
672 ! the intent is to loop from 0 to 89.5 by 0.5 degrees - or similar.
673 ! Because Fortran loop indices are integers, we have to divide the indices by 10 - you get the idea.
674
675 do angle=0,maxangle,angledz
676 zdeg = real(angle,r8)/10.0_r8 ! 0.0 0.5 1.0 1.5 ...
677 zrad = (zdeg*pi)/180.0_r8
678 costheta = cos(zrad)
679
680 ! Angle-dependent low energy (fast) neutron upward flux
681 !>@todo FIXME JH: probably will not do pointwise exponentiation
682 fastflux(:, i) = fastflux(:, i) + fastpot(:, i)*exp(-(isoimass(:, i)/l3 + iwatmass(:, i)/l4)/costheta)*dtheta
683 enddo
684
685 ! After contribution from all directions are taken into account,
686 ! need to multiply fastflux by 2/PI
687
688 fastflux(:, i) = (2.0_r8/pi)*fastflux(:, i)
689
690 ! Low energy (fast) neutron upward flux
691 totflux(:) = totflux(:) + fastflux(:, i)
692enddo
693
694deallocate(dz, zthick, vwc, hiflux, fastpot, &
695 h2oeffdens, idegrad, fastflux, isoimass, iwatmass)
696
697!=================================================================================
698! ... and finally set the return the neutron intensity (i.e. totflux)
699
700where (istatus == 0) val = totflux
701where (istatus /= 0) val = missing_r8
702
703return
704end subroutine get_expected_neutron_intensity
