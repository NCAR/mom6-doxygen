44!----------------------------------------------------------------------
45! function comp_cov_factor(z_in, c)
46!
47! Computes a covariance cutoff function from Gaspari and Cohn
48! QJRMS, 125, 723-757. (their eqn. 4.10)
49!
50! z_in is the distance while c is the cutoff distance.
51! For distances greater than 2c, the cov_factor returned goes to 0.
52
53! Other ramping shapes are also available and can be selected by a namelist
54! parameter. At present, these include a boxcar with the given halfwidth
55! and a ramp in which the weight is set at 1.0 up to the half-width
56! distance and then decreases linearly to 0 at twice the half-width
57! distance.
58
59! Additional information is passed in about the location and specific type of the
60! observation and the location and generic kind of the variable being targeted for
61! increments. These can be used for more refined algorithms that want to
62! make the cutoff a function of these additional arguments.
63
64implicit none
65
66real(r8), intent(in) :: z_in, c
67type(location_type), optional, intent(in) :: obs_loc, target_loc
68integer, optional, intent(in) :: obs_type, target_kind
69integer, optional, intent(in) :: localization_override
70real(r8) :: comp_cov_factor
71
72real(r8) :: z, r
73integer :: iunit, io
74integer :: localization_type
75
76!--------------------------------------------------------
77! Initialize namelist if not already done
78if(.not. namelist_initialized) then
79
80
81 namelist_initialized = .true.
82
83 ! Read the namelist entry
84 call find_namelist_in_file("input.nml", "cov_cutoff_nml", iunit)
85 read(iunit, nml = cov_cutoff_nml, iostat = io)
86 call check_namelist_read(iunit, io, "cov_cutoff_nml")
87
88 if (do_nml_file()) write(nmlfileunit,nml=cov_cutoff_nml)
89 if (do_nml_term()) write( * ,nml=cov_cutoff_nml)
90
91
92 if (do_output()) then
93 select case (select_localization)
94 case (1)
95 call error_handler(e_msg,'comp_cov_factor:', &
96 'Standard Gaspari Cohn localization selected')
97 case (2)
98 call error_handler(e_msg,'comp_cov_factor:', &
99 'Boxcar localization selected')
100 case (3)
101 call error_handler(e_msg,'comp_cov_factor:', &
102 'Ramped localization selected')
103 case default
104 call error_handler(e_err,'comp_cov_factor', &
105 'Illegal value of "select_localization" in cov_cutoff_mod namelist', source)
106 end select
107 endif
108
109endif
110!---------------------------------------------------------
111
112if(present(localization_override)) then
113 localization_type = localization_override
114else
115 localization_type = select_localization
116endif
117
118z = abs(z_in)
119
120!----------------------------------------------------------
121
122if(localization_type == 1) then ! Standard Gaspari Cohn localization
123
124 if( z >= c*2.0_r8 ) then
125
126 comp_cov_factor = 0.0_r8
127
128 else if( z <= c ) then
129 r = z / c
130 comp_cov_factor = &
131 ( ( ( -0.25_r8*r +0.5_r8 )*r +0.625_r8 )*r -5.0_r8/3.0_r8 )*r**2 + 1.0_r8
132!!$ r**5 * (-0.25_r8 ) + &
133!!$ r**4 / 2.0_r8 + &
134!!$ r**3 * 5.0_r8/8.0_r8 - &
135!!$ r**2 * 5.0_r8/3.0_r8 + 1.0_r8
136 else
137
138 r = z / c
139 comp_cov_factor = &
140 ( ( ( ( r/12.0_r8 -0.5_r8 )*r +0.625_r8 )*r +5.0_r8/3.0_r8 )*r -5.0_r8 )*r &
141!!$ r**5 / 12.0_r8 - &
142!!$ r**4 / 2.0_r8 + &
143!!$ r**3 * 5.0_r8 / 8.0_r8 + &
144!!$ r**2 * 5.0_r8 / 3.0_r8 - 5.0_r8*r &
145 + 4.0_r8 - 2.0_r8 / (3.0_r8 * r)
146 endif
147
148else if(localization_type == 2) then ! BOXCAR localization
149
150 if(z < 2.0_r8 * c) then
151 comp_cov_factor = 1.0_r8
152 else
153 comp_cov_factor = 0.0_r8
154 endif
155
156else if(localization_type == 3) then ! Ramped localization
157
158 if(z >= 2.0_r8 * c) then
159 comp_cov_factor = 0.0_r8
160 else if(z >= c .and. z < 2.0_r8 * c) then
161 comp_cov_factor = (2.0_r8 * c - z) / c
162 else
163 comp_cov_factor = 1.0_r8
164 endif
165
166else ! Otherwise namelist parameter is illegal; this is an error
167
168 call error_handler(e_err,'comp_cov_factor', &
169 'Illegal value of "localization" in cov_cutoff_mod namelist', source)
170
171endif
172
173end function comp_cov_factor
