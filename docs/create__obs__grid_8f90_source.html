5!> program intended to create a set of observations located on

57function create_grid()
58 type(obs_sequence_type) :: create_grid
59
60type(obs_type) :: obs, prev_obs
61type(obs_def_type) :: obs_def
62type(time_type) :: obs_time, prev_time
63type(location_type) :: loc
64integer :: max_num_grids, num_copies, num_qc, end_it_all, max_num_obs
65integer :: num_dim, n(3), i, j, k, l
66
67! these things aren't prompted for - they're fixed in the code
68num_copies = 1
69num_qc = 0
70max_num_obs = 1000000 ! FIXME: made up
71
72write(*, *) 'Input upper bound on number of grids of observations in sequence'
73read(*, *) max_num_grids
74
75! Initialize an obs_sequence structure
76call init_obs_sequence(create_grid, num_copies, num_qc, max_num_obs)
77
78do i = 1, num_copies
79 call set_copy_meta_data(create_grid, i, 'observations')
80end do
81
82! Initialize the obs variable
83call init_obs(obs, num_copies, num_qc)
84call init_obs(prev_obs, num_copies, num_qc)
85
86! Loop to initialize each observation in turn; terminate by -1
87do l = 1, max_num_grids
88 write(*, *) 'input a -1 if there are no more grids'
89
90
91 read(*, *) end_it_all
92 if(end_it_all == -1) exit
93
94 ! FIXME: this is the corner of a grid, need to prompt for
95 ! extents in each dim (2d or 3d) and number of points in same.
96 ! then loop below using same type and error, just bumping
97 ! location each time.
98
99 write(*, *) 'the location of the next observation defines the corner of a box'
100
101 ! Need to have key available for specialized observation modules
102 call interactive_obs(num_copies, num_qc, obs, i)
103
104 write(*, *) 'enter the location of the opposite corner of the box'
105 call interactive_location(loc)
106
107 num_dim = -1
108 do while (num_dim < 1 .and. num_dim > 3)
109 write(*, *) 'input 1, 2 or 3 for 1d, 2d or 3d grid'
110 read(*, *) num_dim
111 enddo
112
113 n = 0
114 do i=1, num_dim
115 write(*,*) 'input nitems for dimension ', i
116 read(*,*) n(i)
117 enddo
118
119 do k=1, n(3)
120 do j=1, n(2)
121 do i=1, n(1)
122 ! set an obs based on the corner one
123 ! compute new location and set it into an obs
124! fixme here, too
125
126 if(i == 1) then
127 call insert_obs_in_seq(create_grid, obs)
128 else
129 ! if this is not the first obs, make sure the time is larger
130 ! than the previous observation. if so, we can start the
131 ! linked list search at the location of the previous obs.
132 ! otherwise, we have to start at the beginning of the entire
133 ! sequence to be sure the obs are ordered correctly in
134 ! monotonically increasing times.
135 call get_obs_def(obs, obs_def)
136 obs_time = get_obs_def_time(obs_def)
137 call get_obs_def(prev_obs, obs_def)
138 prev_time = get_obs_def_time(obs_def)
139 if(prev_time > obs_time) then
140 call insert_obs_in_seq(create_grid, obs)
141 else
142 call insert_obs_in_seq(create_grid, obs, prev_obs)
143 endif
144 endif
145 prev_obs = obs
146 end do ! i
147 end do ! j
148 end do ! k
149end do ! max_grids
150
151call destroy_obs(obs)
152call destroy_obs(prev_obs)
153
154end function create_grid
155
