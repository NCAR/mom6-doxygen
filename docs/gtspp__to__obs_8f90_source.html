75integer, parameter :: nmaxdepths = 5000 ! max number of observation depths
76real(r8) :: obs_depth(nmaxdepths) = -1.0_r8
77real(r8) :: temperature(nmaxdepths) = missing_r8

87character(len=128) :: gtspp_out_file = 'obs_seq.gtspp'

100! time stored relative to jan 1st, 1900.
101call set_calendar_type(gregorian)
102base_time = set_date(1900, 1, 1, 0, 0, 0)
103
104! read the necessary parameters from input.nml
105call initialize_utilities()
106call find_namelist_in_file("input.nml", "gtspp_to_obs_nml", iunit)
107read(iunit, nml = gtspp_to_obs_nml, iostat = io)
108
109! Record the namelist values used for the run
110if (do_output()) write(nmlfileunit, nml=gtspp_to_obs_nml)
111
112! any needed namelist checks for sanity:
113
114! cannot have both a single filename and a list; the namelist must
115! shut one off.
116if (gtspp_netcdf_file /= '' .and. gtspp_netcdf_filelist /= '') then
117 call error_handler(e_err, 'gtspp_to_obs', &
118 'One of gtspp_netcdf_file or filelist must be NULL', &
119 source, revision, revdate)
120endif
121if (gtspp_netcdf_filelist /= '') from_list = .true.
122
123! need to know a reasonable max number of obs that could be added here.
124if (from_list) then
125 call find_textfile_dims(gtspp_netcdf_filelist, nfiles, dummy)
126 num_new_obs = avg_obs_per_file * nfiles
127else
128 num_new_obs = avg_obs_per_file
129endif
130
131! FIXME: is this a good idea? to append to an existing file?
132! either read existing obs_seq or create a new one
133call static_init_obs_sequence()
134call init_obs(obs, num_copies, num_qc)
135call init_obs(prev_obs, num_copies, num_qc)
136inquire(file=gtspp_out_file, exist=file_exist)
137! for now, disable this. too easy to make files with duplicate obs in them.
138!if ( file_exist ) then
139if (.false.) then
140
141print *, "found existing obs_seq file, appending to ", trim(gtspp_out_file)
142 call read_obs_seq(gtspp_out_file, 0, 0, num_new_obs, obs_seq)
143
144else
145
146print *, "no existing obs_seq file, creating ", trim(gtspp_out_file)
147print *, "max entries = ", num_new_obs
148 call init_obs_sequence(obs_seq, num_copies, num_qc, num_new_obs)
149 do k = 1, num_copies
150 call set_copy_meta_data(obs_seq, k, 'GTSPP observation')
151 end do
152 do k = 1, num_qc
153 call set_qc_meta_data(obs_seq, k, 'GTSPP QC')
154 end do
155
156end if
157
158did_obs = .false.
159
160! main loop that does either a single file or a list of files
161
162filenum = 1
163fileloop: do ! until out of files
164
165 ! get the single name, or the next name from a list
166 if (from_list) then
167 next_infile = get_next_filename(gtspp_netcdf_filelist, filenum)
168 else
169 next_infile = gtspp_netcdf_file
170 if (filenum > 1) next_infile = ''
171 endif
172 if (next_infile == '') exit fileloop
173
174 have_temp = .false.
175 have_salt = .false.
176
177 ! open the next profile file
178 call nc_check( nf90_open(next_infile, nf90_nowrite, ncid), 'file open', next_infile)
179
180 ! time is stored in the file 2 ways: as real(double) days since 1900/1/1,
181 ! and as 4 and 2 digit strings for year/mon/day/hr/min
182 ! both of these are variables, not attributes
183
184 ! start out with converting the real time.
185 call nc_check( nf90_inq_varid(ncid, "time", varid) ,'inq varid time')
186 call nc_check( nf90_get_var(ncid, varid, dtime) ,'get var time')
187
188 ! convert to integer days and seconds, and add on to reference time.
189 iday = int(dtime)
190 isec = int((dtime - iday) * 86400)
191 delta_time = set_time(isec, iday)
192 obs_time = base_time + delta_time
193 call get_time(obs_time, osec, oday)
194
195 ! get the number of depths
196 call nc_check( nf90_inq_dimid(ncid, "depth", varid), 'inq dimid depth')
197 call nc_check( nf90_inquire_dimension(ncid, varid, len=ndepths, name=dimname), 'inq dim depth')
198
199 ! and read in the depth array
200 call nc_check( nf90_inq_varid(ncid, "depth", varid),'inq varid depth')
201 call nc_check( nf90_get_var(ncid, varid, obs_depth, &
202 start=(/1/), count=(/ndepths/)),'get var depth')
203
204 ! get the single lat/lon values
205 call nc_check( nf90_inq_varid(ncid,"longitude",varid) ,'inq varid longitude')
206 call nc_check( nf90_get_var(ncid, varid, glon), 'get var longitude')
207 call nc_check( nf90_inq_varid(ncid,"latitude",varid) ,'inq varid latitude')
208 call nc_check( nf90_get_var(ncid, varid, glat), 'get var latitude')
209
210 ! if present, the data values from 'temperature'
211 nc_rc = nf90_inq_varid(ncid,"temperature",varid)
212 if (nc_rc == nf90_noerr) then
213 have_temp = .true.
214
215 call nc_check( nf90_get_var(ncid, varid, temperature, &
216 start=(/1,1,1,1/), count=(/1,1,ndepths,1/)), 'get var temperature')
217
218 ! for now, use the data qc from netcdf file
219 call nc_check( nf90_inq_varid(ncid,"TEMP_qparm",varid) ,'inq varid TEMP_qparam')
220 call nc_check( nf90_get_var(ncid, varid, t_qc, &
221 start=(/1,1/), count=(/1,ndepths/)), 'get var TEMP_qparam')
222
223 endif
224
225 ! if present, the data values from 'salinity'
226 nc_rc = nf90_inq_varid(ncid,"salinity",varid)
227 if (nc_rc == nf90_noerr) then
228 have_salt = .true.
229
230 call nc_check( nf90_get_var(ncid, varid, salinity, &
231 start=(/1,1,1,1/), count=(/1,1,ndepths,1/)), 'get var salinity')
232
233 ! for now, use the data qc from netcdf file
234
235 call nc_check( nf90_inq_varid(ncid,"PSAL_qparm",varid) ,'inq varid PSAL_qparam')
236 call nc_check( nf90_get_var(ncid, varid, s_qc, &
237 start=(/1,1/), count=(/1,ndepths/)), 'get var PSAL_qparam')
238
239 endif
240
241 call nc_check( nf90_close(ncid) , 'close file')
242
243 if (debug) then
244 print *, 'input file: ', trim(next_infile)
245 print *, 'ndepths: ', ndepths
246 print *, 'has temps, salinity: ', have_temp, have_salt
247 call print_date(obs_time, 'obs time')
248 endif
249
250 ! FIXME: these have no physical basis; selected to get us running
251 ! but need some domain expertise for guidance.
252 terr = 1.0 ! temp error = fixed at 1 degrees C
253 serr = 0.001 ! salinity error = 1 g/kg, which is 0.001 kg/kg
254
255 first_obs = .true.
256
257 obsloop: do k = 1, ndepths
258
259 ! check qc here. if bad, skip the rest of this block
260 read(t_qc(k:k), '(I1)') i_qc
261 if (have_temp .and. i_qc == 1) then
262
263 ! set qc to a good dart val
264 d_qc(1) = 0.0 ! for dart, a QC of 0 is good
265
266 ! set location - incoming obs are -180 to 180 in longitude;
267 ! dart wants 0 to 360.
268 if (glon < 0.0_r8) glon = glon + 360.0_r8
269 call set_obs_def_location(obs_def, &
270 set_location(glon, glat, obs_depth(k),vertisheight))
271 call set_obs_def_type_of_obs(obs_def, float_temperature)
272 call set_obs_def_time(obs_def, set_time(osec, oday))
273
274 call set_obs_def_error_variance(obs_def, terr * terr)
275 call set_obs_def_key(obs_def, obs_num)
276 call set_obs_def(obs, obs_def)
277
278 obs_val(1) = temperature(k)
279 call set_obs_values(obs, obs_val)
280 qc_val(1) = d_qc(1)
281 call set_qc(obs, qc_val)
282
283 ! first one, insert with no prev. otherwise, since all times are the
284 ! same for this column, insert with the prev obs as the starting point.
285 ! (the first insert with no prev means it will search for the right
286 ! time ordered starting point.)
287 if (first_obs) then
288 call insert_obs_in_seq(obs_seq, obs)
289 first_obs = .false.
290 else
291 call insert_obs_in_seq(obs_seq, obs, prev_obs)
292 endif
293 obs_num = obs_num+1
294 prev_obs = obs
295
296 if (.not. did_obs) did_obs = .true.
297 endif
298
299
300 ! check qc here. if bad, skip the rest of this block
301 read(s_qc(k:k), '(I1)') i_qc
302 if (have_salt .and. i_qc == 1) then
303
304 ! set qc to good dart val
305 d_qc(1) = 0.0 ! but for dart, a QC of 0 is good
306
307 ! set location - incoming obs are -180 to 180 in longitude;
308 ! dart wants 0 to 360.
309 if (glon < 0.0_r8) glon = glon + 360.0_r8
310 call set_obs_def_location(obs_def, &
311 set_location(glon, glat, obs_depth(k),vertisheight))
312 call set_obs_def_type_of_obs(obs_def, float_salinity)
313 call set_obs_def_time(obs_def, set_time(osec, oday))
314
315 call set_obs_def_error_variance(obs_def, serr * serr)
316 call set_obs_def_key(obs_def, obs_num)
317 call set_obs_def(obs, obs_def)
318
319 ! incoming obs are g/kg (practical salinity units - psu)
320 ! model works in kg/kg (model salinity units - msu) so convert here.
321 obs_val(1) = salinity(k) / 1000.0_r8
322 call set_obs_values(obs, obs_val)
323 qc_val(1) = d_qc(1)
324 call set_qc(obs, qc_val)
325
326 ! first one, insert with no prev. otherwise, since all times are the
327 ! same for this column, insert with the prev obs as the starting point.
328 ! (the first insert with no prev means it will search for the right
329 ! time ordered starting point.)
330 if (first_obs) then
331 call insert_obs_in_seq(obs_seq, obs)
332 first_obs = .false.
333 else
334 call insert_obs_in_seq(obs_seq, obs, prev_obs)
335 endif
336 obs_num = obs_num+1
337 prev_obs = obs
338
339 if (.not. did_obs) did_obs = .true.
340 endif
341
342 end do obsloop
343
344 filenum = filenum + 1
345
346end do fileloop
347
348! done with main loop. if we added any obs to the sequence, write it out.
349if (did_obs) then
350print *, 'ready to write, nobs = ', get_num_obs(obs_seq)
351 if (get_num_obs(obs_seq) > 0) &
352 call write_obs_seq(obs_seq, gtspp_out_file)
353
354 ! minor stab at cleanup, in the off chance this will someday get turned
355 ! into a subroutine in a module. probably not all that needs to be done,
356 ! but a start.
357 call destroy_obs(obs)
358 !call destroy_obs(prev_obs) ! is this identical to obs?
359 ! get core dumps here, not sure why?
360 if (get_num_obs(obs_seq) > 0) call destroy_obs_sequence(obs_seq)
361endif
362
363call error_handler(e_msg,'gtspp_to_obs','Finished successfully.',source,revision,revdate)
364call finalize_utilities()
365

373

376end program gtspp_to_obs
