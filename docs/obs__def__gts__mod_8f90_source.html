105subroutine get_expected_thickness(state_handle, ens_size, location, thickness, istatus)
106!-----------------------------------------------------------------------------
107! inputs:
108! state_vector: DART state vector
109!
110! output parameters:
111! thickness: modeled satem thickness between the specified layers. (unit: m)
112! istatus: =0 normal; =1 either or both t or q interpolation failed
113!---------------------------------------------
114! Hui Liu NCAR/IMAGE April 9, 2008
115!---------------------------------------------
116implicit none
117
118type(ensemble_type), intent(in) :: state_handle
119integer, intent(in) :: ens_size
120type(location_type), intent(in) :: location
121real(r8), intent(out) :: thickness(ens_size)
122integer, intent(out) :: istatus(ens_size)
123
124! local variables
125
126integer :: k
127integer, parameter :: nlevel = 9, num_press_int = 10
128real(r8) :: lon, lat, pressure, obsloc(3), obs_level(nlevel), press(num_press_int+1)
129real(r8) :: press_top, press_bot, press_int
130
131integer :: t_istatus(ens_size), q_istatus(ens_size)
132
133data obs_level/85000.0, 70000.0, 50000.0, 40000.0, 30000.0, 25000.0, 20000.0, 15000.0, 10000.0/
134
135real(r8), parameter:: rdrv1 = gas_constant_v/gas_constant - 1.0_r8
136
137real(r8) :: lon2, p
138type(location_type) :: location2
139integer :: which_vert
140real(r8) :: t(ens_size), q(ens_size), tv(ens_size)
141
142logical :: return_now
143
144if ( .not. module_initialized ) call initialize_module
145
146! Start out assuming passing status
147istatus = 0
148
149obsloc = get_location(location)
150lon = obsloc(1) ! degree: 0 to 360
151lat = obsloc(2) ! degree: -90 to 90
152pressure = obsloc(3) ! (Pa)
153
154press_bot = -999.9_r8
155press_top = -999.9_r8
156
157! find the bottom and top of the layer -
158! The 'bottom' is defined to be the incoming observation value.
159! The 'top' is the next higher observation level.
160! If the bottom is the highest observation level;
161! there is no top, and it is an ERROR.
162
163if(pressure > obs_level(1) ) then
164 press_bot = pressure
165 press_top = obs_level(1)
166endif
167
168do k = 1, nlevel-1
169 if( abs(pressure - obs_level(k)) < 0.01_r8 ) then
170 press_bot = obs_level(k)
171 press_top = obs_level(k+1)
172 exit
173 endif
174end do
175
176! define the vertical intervals for the integration of Tv. Use 10 sub_layers
177! for the vertical integration.
178
179press_int = (press_bot - press_top)/num_press_int
180
181press(1) = press_bot
182do k=2, num_press_int+1
183 press(k) = press(1) - press_int * (k-1)
184end do
185
186! define the location of the interpolated points.
187
188lon2 = lon
189if(lon > 360.0_r8 ) lon2 = lon - 360.0_r8
190if(lon < 0.0_r8 ) lon2 = lon + 360.0_r8
191
192which_vert = vertispressure
193thickness = 0.0_r8
194
195do k=2, num_press_int+1, 2
196 p = press(k)
197 location2 = set_location(lon2, lat, p, which_vert)
198
199 call interpolate(state_handle, ens_size, location2, qty_temperature, t, t_istatus)
200 call track_status(ens_size, t_istatus, thickness, istatus, return_now)
201 ! don't return until we have the corresponding q to test
202
203 call interpolate(state_handle, ens_size, location2, qty_specific_humidity, q, q_istatus)
204 call track_status(ens_size, q_istatus, thickness, istatus, return_now)
205
206
207 ! use a consistent error code for failed t or q instead of
208 ! the model specific return code.
209 where (istatus > 0) istatus = 1
210
211 ! t : Kelvin, from top to bottom
212 ! q : kg/kg, from top to bottom
213
214 where (istatus == 0)
215 tv = t * (1.0_r8 + rdrv1 * q ) ! virtual temperature
216 thickness = thickness + gas_constant/gravity * tv * log(press(k-1)/press(k+1))
217 end where
218
219 !> @todo is this magic number reasonable? jeff thinks no.
220 where ( abs(thickness) > 10000.0_r8 )
221 istatus = 2
222 thickness = missing_r8
223 end where
224
225end do
226
227return
228
229end subroutine get_expected_thickness
