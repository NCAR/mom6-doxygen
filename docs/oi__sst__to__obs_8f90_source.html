154 ! time is stored in the file 2 ways: as real(double) seconds since 1981/1/1,

193 endif
194
195 call create_3d_obs(lat(j), lon(i), 0.0_r8, vertissurface, sst(i,j), &

198
199 enddo obslooplon
200 enddo obslooplat
201
202 ! if we added any obs to the sequence, write it out to a file now.
203 if ( get_num_obs(obs_seq) > 0 ) then
204 if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
205 if (debug) print *, ' skipping = ', nmissing

207 else
208 write(string1,*)'no observations for output file'
209 write(string2,*)'"'//trim(output_file)//'"'
210 call error_handler(e_msg, routine, string1, text2=string2)
211 endif
212
213 call destroy_obs_sequence(obs_seq)
214
215end do timeloop
