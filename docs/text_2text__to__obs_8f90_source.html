7! An example program to create DART format observations from a text input file
8!
9! An example program that reads text (ascii) input lines and creates
10! DART observations from them. This is more of a template program.
11! You will have to modify it for your own use. In particular
12! you will have to adapt the read routine to match your white-space
13! separated values or fixed-width column data.
14!
15! This program now includes a small namelist. The items in the namelist
16! can be changed at runtime without having to recompile the program.
17! Having the input and output filenames in the namelist should make it
18! simpler to script the conversion of a series of files.
19!
20! history:
21! created 29 Mar 2010 nancy collins NCAR/IMAGe
22! updated 8 Feb 2016 minor changes to wind conversion, and updated
23! comments to try to be more helpful.
24! updated 29 Aug 2019 added namelist with example values
25!

28

50! variables that can be changed at runtime from a namelist
51! add any other options here that would be useful for your application.
52character(len=256) :: text_input_file = '../data/text_input_file'
53character(len=256) :: obs_out_file = 'obs_seq.out'
54logical :: debug = .false. ! set to .true. to print info

56namelist /text_to_obs_nml/ text_input_file, obs_out_file, debug

76! start of executable code

80! Read the namelist entry

90! time setup

93!! Some times are supplied as number of seconds since some reference
94!! date. To support that, set a base/reference time and then add
95!! the number of seconds to it. (time_types support adding two time
96!! types together, or adding a scalar to a time_type.)
97!! Here is an example of setting a reference date, giving the
98!! call: year, month, day, hours, mins, secs
99!ref_day0 = set_date(1970, 1, 1, 0, 0, 0)

102
103iunit = open_file(text_input_file, 'formatted', 'read')
104if (debug) print *, 'opened input file ' // trim(text_input_file)

107! each observation in this series will have a single observation value
108! and a quality control flag. the max possible number of obs needs to
109! be specified but it will only write out the actual number created.
110max_obs = 100000
111num_copies = 1
112num_qc = 1
113
114! call the initialization code, and initialize two empty observation types
115call static_init_obs_sequence()
116call init_obs(obs, num_copies, num_qc)
117call init_obs(prev_obs, num_copies, num_qc)
118first_obs = .true.

120! create a new, empty obs_seq file. you must give a max limit
121! on number of obs. increase the size if too small.
122call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
123
124! the first one needs to contain the string 'observation' and the
125! second needs the string 'QC'.
126call set_copy_meta_data(obs_seq, 1, 'observation')
127call set_qc_meta_data(obs_seq, 1, 'Data QC')
128
129! if you want to append to existing files (e.g. you have a lot of
130! small text files you want to combine), you can do it this way,
131! or you can use the obs_sequence_tool to merge a list of files
132! once they are in DART obs_seq format.
133
134! ! existing file found, append to it
135! inquire(file=obs_out_file, exist=file_exist)
136! if ( file_exist ) then
137! call read_obs_seq(obs_out_file, 0, 0, max_obs, obs_seq)
138! endif

140! Set the DART data quality control. 0 is good data.
141! increasingly larger QC values are more questionable quality data.
142qc = 0.0_r8
143
144obsloop: do ! no end limit - have the loop break when input ends
145
146 ! read in a line from the text file. What you need to create an obs:
147 ! location: lat, lon, and height in pressure or meters
148 ! time: when the observation was taken
149 ! type: from the DART list of obs types
150 ! error: very important - the instrument error plus representativeness error
151 ! (see html file for more info)
152
153 ! this code here assumes an input line has: a type (1/2), location, time, value, obs error
154 ! **this is where you will need to adapt this code for your input data values**
155
156 ! read in entire text line into a buffer
157 read(iunit, "(A)", iostat=rcio) input_line
158 if (rcio /= 0) then
159 if (debug) print *, 'got bad read code from input file, rcio = ', rcio
160 exit obsloop
161 endif
162
163 ! pull off the first 2 columns as an integer, to decode the type
164 read(input_line, "(I2)", iostat=rcio) otype
165 if (rcio /= 0) then
166 if (debug) print *, 'got bad read code trying to get obs type, rcio = ', rcio
167 exit obsloop
168 endif
169
170 if (debug) print *, 'next observation type = ', otype
171
172 ! for this example, assume there is an obs type, where otype=1 is
173 ! temperature measured at a location and a vertical height,
174 ! and if otype=2, a wind speed and direction in degrees (0-360)
175 ! measured at a location and a vertical pressure.
176
177 if (otype == 1) then
178 read(input_line(3:129), *, iostat=rcio) lat, lon, vert, &
179 year, month, day, hour, minute, second, &
180 temp, terr
181 if (rcio /= 0) then
182 if (debug) print *, 'got bad read code getting rest of temp obs, rcio = ', rcio
183 exit obsloop
184 endif
185 else
186 read(input_line(3:129), *, iostat=rcio) lat, lon, vert, &
187 year, month, day, hour, minute, second, &
188 wspeed, wdir, werr
189 if (rcio /= 0) then
190 if (debug) print *, 'got bad read code getting rest of wind obs, rcio = ', rcio
191 exit obsloop
192 endif
193 endif
194
195 if (debug) print *, 'next observation located at lat, lon = ', lat, lon
196
197 ! check the lat/lon values to see if they are ok. we require
198 ! longitudes to be 0 to 360.
199 if ( lat > 90.0_r8 .or. lat < -90.0_r8 ) cycle obsloop
200 if ( lon < 0.0_r8 .or. lon > 360.0_r8 ) cycle obsloop
201
202
203 ! if lon comes in between -180 and 180, use these lines instead:
204 !if ( lon > 180.0_r8 .or. lon < -180.0_r8 ) cycle obsloop
205 !if ( lon < 0.0_r8 ) lon = lon + 360.0_r8 ! changes into 0-360
206
207 ! put date into a dart time format
208 time_obs = set_date(year, month, day, hour, minute, second)
209
210 if (debug) call print_date(time_obs, 'next obs time is')
211
212 !! if time is given in seconds since some date, here's how to add it.
213 !time_obs = ref_day0 + time_obs
214
215 ! extract time of observation into gregorian day, sec.
216 call get_time(time_obs, osec, oday)
217
218 ! this example assumes there is an obs type, where otype=1 is
219 ! a temperature measured in height, and if otype=2, a wind speed
220 ! and direction with a vertical value in pressure. any observation
221 ! can use any of the vertical types; this is just an example.
222
223 if (otype == 1) then
224
225 ! height is in meters (gph)
226
227 ! make an obs derived type, and then add it to the sequence
228 call create_3d_obs(lat, lon, vert, vertisheight, temp, &
229 eval_temperature, terr, oday, osec, qc, obs)
230 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
231
232 if (debug) print *, 'added temperature obs to output seq'
233 else
234
235 ! DART assimilates wind as 2 separate U and V components. assimilating
236 ! "direction and speed" is difficult because direction is measured in
237 ! cyclic coordinates so you can't do simple statistics to get a mean value.
238
239 ! convert a wind speed & direction into the U and V components
240 ! and create 2 obs for it. assume vert is in mb or hectopascals,
241 ! convert to pascals. DART assumes all pressures are in pascals.
242 ! check your data source; usually wind direction is specified as the
243 ! direction the wind is coming from, increasing degrees going in a
244 ! clockwise circle. U and V wind components have the opposite sign.
245 uwnd = sin(wdir * deg2rad) * wspeed * (-1.0_r8)
246 vwnd = cos(wdir * deg2rad) * wspeed * (-1.0_r8)
247 uerr = werr
248 verr = werr
249
250 ! convert hectopascals to pascals.
251 vert = vert * 100.0_r8
252
253 call create_3d_obs(lat, lon, vert, vertispressure, uwnd, &
254 eval_u_wind_component, uerr, oday, osec, qc, obs)
255 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
256
257 call create_3d_obs(lat, lon, vert, vertispressure, vwnd, &
258 eval_v_wind_component, verr, oday, osec, qc, obs)
259 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
260
261 if (debug) print *, 'added 2 wind obs to output seq'
262 endif
263
264end do obsloop
265
266! if we added any obs to the sequence, write it out to a file now.
267if ( get_num_obs(obs_seq) > 0 ) then
268 if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
269 call write_obs_seq(obs_seq, obs_out_file)
270endif
271
272! end of main program
273call finalize_utilities()
274
275contains

280! create_3d_obs - subroutine that is used to create an observation
281! type from observation data.
282!
283! NOTE: assumes the code is using the threed_sphere locations module,
284! that the observation has a single data value and a single
285! qc value, and that this obs type has no additional required
286! data (e.g. gps and radar obs need additional data per obs)
287!
288! inputs:
289! lat - latitude of observation
290! lon - longitude of observation
291! vval - vertical coordinate
292! vkind - kind of vertical coordinate (pressure, level, etc)
293! obsv - observation value
294! otype - observation type
295! oerr - observation error
296! day - gregorian day
297! sec - gregorian second
298! qc - quality control value
299! outputs:
300! obs - observation type
301!
302! created Oct. 2007 Ryan Torn, NCAR/MMM
303! adapted for more generic use 11 Mar 2010, nancy collins, ncar/image
304!
305!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
306subroutine create_3d_obs(lat, lon, vval, vkind, obsv, otype, oerr, day, sec, qc, obs)
307use types_mod, only : r8
308use obs_def_mod, only : obs_def_type, set_obs_def_time, set_obs_def_type_of_obs, &
309 set_obs_def_error_variance, set_obs_def_location
310use obs_sequence_mod, only : obs_type, set_obs_values, set_qc, set_obs_def
311use time_manager_mod, only : time_type, set_time
312use location_mod, only : set_location
313
314 integer, intent(in) :: otype, vkind, day, sec
315 real(r8), intent(in) :: lat, lon, vval, obsv, oerr, qc
316 type(obs_type), intent(inout) :: obs
317
318real(r8) :: obs_val(1), qc_val(1)
319type(obs_def_type) :: obs_def
320
321call set_obs_def_location(obs_def, set_location(lon, lat, vval, vkind))
322call set_obs_def_type_of_obs(obs_def, otype)
323call set_obs_def_time(obs_def, set_time(sec, day))
324call set_obs_def_error_variance(obs_def, oerr * oerr)
325call set_obs_def(obs, obs_def)
326
327obs_val(1) = obsv
328call set_obs_values(obs, obs_val)
329qc_val(1) = qc
330call set_qc(obs, qc_val)
331
332end subroutine create_3d_obs

335!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
336!
337! add_obs_to_seq -- adds an observation to a sequence. inserts if first
338! obs, inserts with a prev obs to save searching if that's possible.
339!
340! seq - observation sequence to add obs to
341! obs - observation, already filled in, ready to add
342! obs_time - time of this observation, in dart time_type format
343! prev_obs - the previous observation that was added to this sequence
344! (will be updated by this routine)
345! prev_time - the time of the previously added observation
346! (will also be updated by this routine)
347! first_obs - should be initialized to be .true., and then will be
348! updated by this routine to be .false. after the first obs
349! has been added to this sequence.
350!
351! created Mar 8, 2010 nancy collins, ncar/image
352!
353!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
354subroutine add_obs_to_seq(seq, obs, obs_time, prev_obs, prev_time, first_obs)
355 use types_mod, only : r8
356 use obs_sequence_mod, only : obs_sequence_type, obs_type, insert_obs_in_seq
357 use time_manager_mod, only : time_type, operator(>=)
358
359 type(obs_sequence_type), intent(inout) :: seq
360 type(obs_type), intent(inout) :: obs, prev_obs
361 type(time_type), intent(in) :: obs_time
362 type(time_type), intent(inout) :: prev_time
363 logical, intent(inout) :: first_obs
364
365! insert(seq,obs) always works (i.e. it inserts the obs in
366! proper time format) but it can be slow with a long file.
367! supplying a previous observation that is older (or the same
368! time) as the new one speeds up the searching a lot.
369
370if(first_obs) then ! for the first observation, no prev_obs
371 call insert_obs_in_seq(seq, obs)
372 first_obs = .false.
373else
374 if(obs_time >= prev_time) then ! same time or later than previous obs
375 call insert_obs_in_seq(seq, obs, prev_obs)
376 else ! earlier, search from start of seq
377 call insert_obs_in_seq(seq, obs)
378 endif
379endif
380
381! update for next time
382prev_obs = obs
383prev_time = obs_time
384
385end subroutine add_obs_to_seq
