5!> Program to read true_state.nc and preassim.nc (or analysis.nc)
6!> and print out the time-mean total error and spread. The matlab macros
7!> do this with graphics, but if you want numerical output suitable for
8!> shell scripting/automation this should be easier to incorporate.
9!>
10!> Unlike the Matlab routine, this program treats every value in the state
11!> vector the same. There is no distinction between variables of different
12!> types (and possibly magnitudes), no weighting by area or volume based on
13!> the underlying grid, and no computing values by model level.
14!>
15!> The output includes other text; to collect just the error and spread lines
16!> grep standard output for 'Total'. This program also uses the standard
17!> DART utility routines that log both the output in dart_log.out and optionally
18!> the namelist values in dart_log.nml. If you are running this in a script
19!> or repeatedly in a loop, you may want to delete or truncate those files
20!> periodically.
21!>
22!> This program does compute any weighted errors.
23
24program compute_error
25
26use types_mod, only : r8, metadatalength
27
28use utilities_mod, only : initialize_utilities, &
29 error_handler, nmlfileunit, e_msg, e_err, &
30 find_namelist_in_file, &
31 check_namelist_read, finalize_utilities, &
32 do_nml_file, do_nml_term
33use netcdf_utilities_mod, only : nc_check
34use netcdf
35
36implicit none

38character(len=*), parameter :: source = 'compute_error.f90'
39
40character(len = 512) :: message1, message2
41
42integer :: iunit, model_size, io, ierr
43integer :: ntimes, skip_start, i1, i2
44integer :: true_copy, ens_mean_copy, ens_spread_copy
45integer :: varid
46
47real(r8), allocatable :: truth(:,:), ens_mean(:,:), ens_spread(:,:), zeros(:,:)
48real(r8), allocatable :: err_array(:), spread_array(:)
49
50real(r8) :: total_error, total_spread
51
52type fileinfo
53 integer :: ncid
54 integer :: numcopies
55 integer :: model_size
56 integer :: metadatacount
57 integer :: ntimes
58 integer :: sindex
59 integer :: eindex
60 real(r8), allocatable :: timevals(:)
61 character(len=metadatalength), allocatable :: metadata(:)
62 character(len=512) :: fname
63end type fileinfo

65type(fileinfo) :: t, d

68! Namelist variables
69
70character(len=256) :: diag_file_name = 'preassim.nc'
71character(len=256) :: truth_file_name = 'true_state.nc'
72integer :: skip_first_ntimes = 0
73
74namelist /compute_error_nml/ &
75 diag_file_name, &
76 truth_file_name, &
77 skip_first_ntimes

79!----------------------------------------------------------------
80! start of executable code
81!----------------------------------------------------------------
82
83call initialize_utilities('compute_error')
84
85
86! Read the namelist entry and print it
87call find_namelist_in_file("input.nml", "compute_error_nml", iunit)
88read(iunit, nml = compute_error_nml, iostat = io)
89call check_namelist_read(iunit, io, "compute_error_nml")
90
91if (do_nml_file()) write(nmlfileunit, nml=compute_error_nml)
92if (do_nml_term()) write( * , nml=compute_error_nml)
93
94! record names for error messages, mostly.
95t%fname = trim(truth_file_name)
96d%fname = trim(diag_file_name)
97
98! open truth, diag file
99ierr = nf90_open(path=trim(truth_file_name), mode=nf90_nowrite, ncid=t%ncid)
100call nc_check(ierr, 'compute_error', 'opening truth file')
101
102ierr = nf90_open(path=trim(diag_file_name), mode=nf90_nowrite, ncid=d%ncid)
103call nc_check(ierr, 'compute_error', 'opening diagnostic file')
104
105
106! collect the dimension ids and lengths we are going to use from the truth file:
107! copy, metadatalength, time and StateVariable.
108
109call findlengths(t)
110call findlengths(d)
111
112if (t%ntimes <= 0) then
113 write(message1, '(A)') 'file '//trim(t%fname)
114 call error_handler(e_err, 'findlengths', 'time dimension must have at least 1 value', &
115 source, text2=message1)
116endif
117if (d%ntimes <= 0) then
118 write(message1, '(A)') 'file '//trim(d%fname)
119 call error_handler(e_err, 'findlengths', 'time dimension must have at least 1 value', &
120 source, text2=message1)
121endif
122
123call confirm_modelsizes_match(t, d)
124
125! allocate space for data we are going to read, except for 'truth'.
126
127allocate(t%metadata(t%numcopies), t%timevals(t%ntimes))
128allocate(d%metadata(d%numcopies), d%timevals(d%ntimes))
129
130call getchardata_nc(t%ncid, "CopyMetaData", t%metadata)
131call getchardata_nc(d%ncid, "CopyMetaData", d%metadata)
132
133call getrealdata1d_nc(t%ncid, "time", t%timevals)
134call getrealdata1d_nc(d%ncid, "time", d%timevals)
135
136call findtimesubset(t, d, ntimes)
137model_size = t%model_size
138
139! allocate space for data now that we know the time overlap.
140allocate(truth(model_size, ntimes))
141allocate(ens_mean(model_size, ntimes), ens_spread(model_size, ntimes))
142allocate(zeros(model_size, ntimes), err_array(ntimes), spread_array(ntimes))
143
144! findcopyindex() dies with an error if the copy name is not found.
145call findcopyindex('CopyMetaData', t%metadata, t%numcopies, 'true state', t%fname, true_copy)
146call findcopyindex('CopyMetaData', d%metadata, d%numcopies, 'ensemble mean', d%fname, ens_mean_copy)
147call findcopyindex('CopyMetaData', d%metadata, d%numcopies, 'ensemble spread', d%fname, ens_spread_copy)
148
149
150! get the data - truth, mean, spread - needed for the error calculation
151ierr = nf90_inq_varid(t%ncid, "state", varid)
152call nc_check(ierr, 'compute_error', 'inq_varid state')
153
154ierr = nf90_get_var(t%ncid, varid, truth, &
155 start=(/ 1, true_copy, t%sindex /), &
156 count=(/ model_size, 1, t%ntimes /))
157call nc_check(ierr, 'compute_error', 'get_var true state')
158
159ierr = nf90_inq_varid(d%ncid, "state", varid)
160call nc_check(ierr, 'compute_error', 'inq_varid state')
161
162ierr = nf90_get_var(d%ncid, varid, ens_mean, &
163 start=(/ 1, ens_mean_copy, d%sindex /), &
164 count=(/ model_size, 1, d%ntimes /))
165call nc_check(ierr, 'compute_error', 'get_var state mean')
166ierr = nf90_get_var(d%ncid, varid, ens_spread, &
167 start=(/ 1, ens_spread_copy, d%sindex /), &
168 count=(/ model_size, 1, d%ntimes /))
169call nc_check(ierr, 'compute_error', 'get_var state spread')
170
171
172! compute the error. array is 'ntimes' long, each entry should be
173! sqrt of sum of (truth - diag) squared.
174! truth for spread is 0
175! then avg error value is err array / ntimes
176!
177! weighted error would be: sqrt of sum of ((truth - diag)^2 * weights)
178! this codes does not do weights.
179
180zeros(:,:) = 0.0_r8
181
182call find_error(truth, ens_mean, err_array, total_error)
183call find_error(zeros, ens_spread, spread_array, total_spread)
184
185! output
186
187call printout('Time-mean Ensemble Total Error: ', total_error, -1, -1.0_r8, &
188 'Time-mean Ensemble Total Spread: ', total_spread, -1, -1.0_r8)
189call error_handler(e_msg,'', '')
190
191call printout('First Error Value, Time Index/Value: ', err_array(1), 1, t%timevals(t%sindex), &
192 'First Spread Value, Time Index/Value: ', spread_array(1), 1, t%timevals(t%sindex))
193
194i1 = minloc(err_array, 1)
195i2 = minloc(spread_array, 1)
196call printout('Min Error Value, Time Index/Value: ', err_array(i1), i1, t%timevals(t%sindex+i1), &
197 'Min Spread Value, Time Index/Value: ', spread_array(i2), i2, t%timevals(t%sindex+i2))
198
199i1 = maxloc(err_array, 1)
200i2 = maxloc(spread_array, 1)
201call printout('Max Error Value, Time Index/Value: ', err_array(i1), i1, t%timevals(t%sindex+i1), &
202 'Max Spread Value, Time Index/Value: ', spread_array(i1), i2, t%timevals(t%sindex+i2))
203
204call printout('Last Error Value, Time Index/Value: ', err_array(ntimes), ntimes, t%timevals(t%eindex), &
205 'Last Spread Value, Time Index/Value: ', spread_array(ntimes), ntimes, t%timevals(t%eindex))
206
207! if the namelist 'skip_first_ntimes' is greater than 0, compute error again
208! without those values. intended use is to skip spinup spikes in error values.
209
210if (skip_first_ntimes > 0 .and. ntimes > skip_first_ntimes+1) then
211 skip_start = skip_first_ntimes+1
212 call find_error(truth(:, skip_start:ntimes), &
213 ens_mean(:, skip_start:ntimes), err_array(1:ntimes-skip_start+1), total_error)
214 call find_error(zeros(:, skip_start:ntimes), &
215 ens_spread(:, skip_start:ntimes), spread_array(1:ntimes-skip_start+1), total_spread)
216
217
218 message1 = construct_msg_string("Skip First", skip_first_ntimes, "Times,")
219 call error_handler(e_msg,'', '')
220 call printout(trim(message1)//' Mean Error: ', total_error, -1, -1.0_r8, &
221 trim(message1)//' Mean Spread: ', total_spread, -1, -1.0_r8)
222endif
223
224! FIXME: make an error handler call that just prints a blank line? or error_handler(E_BLANK)
225call error_handler(e_msg,'', '')
226
227! clean up
228
229deallocate(truth)
230deallocate(ens_mean, ens_spread, zeros, err_array, spread_array)
231
232ierr = nf90_close(t%ncid)
233call nc_check(ierr, 'compute_error', 'nc_close truth')
234ierr = nf90_close(d%ncid)
235call nc_check(ierr, 'compute_error', 'nc_close diag')
236
237
238call finalize_utilities('compute_error')
239

242!---------------------------------------------------
243
244! the routine that computes the time-series error array,
245! plus returning the mean of that array.
246
247subroutine find_error(truth, diag, err_array, total_err)
248
249real(r8), intent(in) :: truth(:,:)
250real(r8), intent(in) :: diag(:,:)
251real(r8), intent(out) :: err_array(:)
252real(r8), intent(out) :: total_err
253
254integer :: model_size, num_times, i, j
255real(r8), allocatable :: sumsq(:)
256
257model_size = size(truth, 1)
258num_times = size(truth, 2)
259
260!FIXME: are missing values in the truth or state possible?
261! check pop or clm to test this.
262
263allocate(sumsq(num_times))
264
265! in matlab:
266! err = sqrt( sum( (pred - verif).^2, 2) );
267
268sumsq(:) = 0.0_r8
269do j=1, num_times
270 do i=1, model_size
271 sumsq(j) = sumsq(j) + (truth(i,j)-diag(i,j))**2
272 enddo
273enddo
274
275err_array(:) = sqrt(sumsq)
276
277total_err = sum(err_array) / num_times
278
279deallocate(sumsq)
280
281end subroutine find_error

448! sets ntimes to the number of time vals in the common time array,
449! and also sets sindex and eindex in both this and that to indicate
450! the exact times which are overlapped.
451
452subroutine findtimesubset(this, that, ntimes)
453
454type(fileinfo), intent(inout) :: this
455type(fileinfo), intent(inout) :: that
456integer, intent(out) :: ntimes
457
458integer :: i, j
459real(r8) :: thistime
460logical :: timematch
461
462! set all the outputs here
463
464this%sindex = -1
465this%eindex = -1
466that%sindex = -1
467that%eindex = -1
468ntimes = 0
469
470!DEBUG print *, 'truth: ', this%ntimes, this%timevals(1), this%timevals(this%ntimes)
471!DEBUG print *, ' diag: ', that%ntimes, that%timevals(1), that%timevals(that%ntimes)
472
473! see if the common case is true: that both time arrays are the same
474! length and equal. if so, return happily.
475
476if (this%ntimes == that%ntimes) then
477 timematch = .true.
478 loop1: do i=1, this%ntimes
479 if (abs(this%timevals(i) - that%timevals(i)) > epsilon(thistime)) then
480 timematch = .false.
481 exit loop1
482 endif
483 enddo loop1
484 if (timematch) then
485 ntimes = this%ntimes
486 this%sindex = 1
487 that%sindex = 1
488 this%eindex = ntimes
489 that%eindex = ntimes
490 return
491 endif
492endif
493
494! see if another case is true - that these time lists are disjoint sets.
495
496if (this%timevals(1) > that%timevals(that%ntimes)) then
497 !DEBUG print *, this%timevals(1), that%timevals(that%ntimes), ' a > b '
498 write(message1, '(A)') 'all times in the diag file are before times in the truth file'
499 write(message2, '(A)') 'no common times in the two input files'
500 call error_handler(e_err, 'compare_lengths', message1, &
501 source, text2=message2)
502endif
503
504if (this%timevals(this%ntimes) < that%timevals(1)) then
505 !DEBUG print *, this%timevals(this%ntimes), that%timevals(1), ' a < b '
506 write(message1, '(A)') 'all times in the diag file are after times in the truth file'
507 write(message2, '(A)') 'no common times in the two input files'
508 call error_handler(e_err, 'compare_lengths', message1, &
509 source, text2=message2)
510endif
511
512! find which one has the earliest time and loop over the other looking for
513! a match of the start time.
514
515if (this%timevals(1) < that%timevals(1)) then
516
517 loop2: do i=1, this%ntimes
518 if (abs(this%timevals(i) - that%timevals(1)) <= epsilon(thistime)) then
519 this%sindex = i
520 that%sindex = 1
521 exit loop2
522 endif
523 enddo loop2
524 if (this%sindex < 0) goto 10 ! error
525
526else
527
528 loop3: do i=1, this%ntimes
529 if (abs(this%timevals(1) - that%timevals(i)) <= epsilon(thistime)) then
530 this%sindex = 1
531 that%sindex = i
532 exit loop3
533 endif
534 enddo loop3
535 if (this%sindex < 0) goto 10 ! error
536
537endif
538
539!DEBUG print *, 'set sindex ok, to vals', this%sindex, that%sindex
540
541! sindex is now set. start from sindex and stop when
542! you run out of times or the values differ.
543
544i = this%sindex
545j = that%sindex
546loop4: do
547 !DEBUG print *, i,j, this%timevals(i), that%timevals(j)
548
549 ! check to make sure the time intervals are the same in both files
550 if (abs(this%timevals(i) - that%timevals(j)) > epsilon(thistime)) goto 10 ! error
551
552 if (i+1 > this%ntimes) exit loop4
553 if (j+1 > that%ntimes) exit loop4
554
555 i = i + 1
556 j = j + 1
557
558enddo loop4
559
560this%eindex = i
561that%eindex = j
562
563!DEBUG print *, 'set eindex ok, to vals', this%eindex, that%eindex
564
565this%ntimes = this%eindex - this%sindex + 1
566that%ntimes = that%eindex - that%sindex + 1
567
568ntimes = this%ntimes
569!DEBUG print *, 'set ntimes to ', ntimes
570return
571
572! you only get here via an error goto
573
57410 continue
575
576write(message1, '(A)') 'there is an overlapping time period, but the exact times do not match'
577write(message2, '(A)') 'mismatched times between the two input files'
578call error_handler(e_err, 'compare_lengths', message1, &
579 source, text2=message2)
580
581end subroutine findtimesubset
