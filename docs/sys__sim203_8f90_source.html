7! UPDATE from 22 Sept. 2003 for obs space factor correction.
8! WARNING:WARNING:WARNING, when mean difference is much less than
9! the covariances, the factor hear are NOT independent of the mean
10! difference for the mean. BUT, This is just a secondary sampling issue.
11
12! Work done during last week of January 2002 (28 Jan init) to investigate
13! impacts of sampling error from small ensembles on update for a single
14! variable that is exactly observed; applicable to observation variable
15! priors. Small sample estimates of variance have approximately normal
16! (or is it exactly normal) error distributions. BUT, when one computes
17! the updated variance there is a bias. Of course, one must also account
18! for the increased uncertainty in the estimate of the mean resulting from
19! errors in the computation of the variance.
20
21! This piece looks at what large sample MC gives for correct updated
22! distribution statistics in preparation for correcting EAKF for the small
23! sample problems.

59subroutine sys_sim(input_var_ratio, n, n_samples, r, factor)
60
61use random_seq_mod, only : random_seq_type, init_random_seq, random_gaussian, &
62 twod_gaussians, random_uniform
63
64implicit none
65
66double precision, intent(in) :: input_var_ratio
67integer, intent(in) :: n, n_samples
68type(random_seq_type), intent(inout) :: r
69double precision, intent(out) :: factor
70
71double precision, allocatable :: rnum(:)
72double precision :: sigma_y_p, y_p, sigma_y_o, y_o
73double precision :: sum_mean, sum_var, var, sigma_y_u
74double precision :: y_u, sample_mean, sample_var, exact_mean
75double precision :: exact_var, correct_var, mean_ratio, var_ratio
76double precision :: sum_fraction, exact_fraction, fraction, fraction_ratio
77integer :: i, j
78
79! Given ratio of variance as only active parameter, construct all others
80sigma_y_o = 1.0
81sigma_y_p = sigma_y_o * input_var_ratio
82y_p = 0.0
83y_o = 1.0
84
85! What would answer be without sampling error?
86exact_var = (1.0 / (1.0 / sigma_y_p + 1.0 / sigma_y_o))
87exact_mean = exact_var * (y_p / sigma_y_p + y_o / sigma_y_o)
88exact_fraction = (exact_mean - y_p) / (y_o - y_p)
89
90! Allocate storage for computing sample variance
91allocate(rnum(n))
92
93! Initialize summations
94sum_mean = 0.0
95sum_var = 0.0
96sum_fraction = 0.0
97
98! Loop through the number of samples
99do i = 1, n_samples
100
101! Compute the sample variance for this ensemble size
102 do j = 1, n
103 rnum(j) = random_gaussian(r, y_p, sqrt(sigma_y_p))
104 end do
105
106! Compute the sample variance
107 call sample_mean_var(rnum, n, sample_mean, var)
108
109! Given this var and mean, compute updated var and mean
110 sigma_y_u = (1.0 / (1.0 / var + 1.0 / sigma_y_o))
111 y_u = sigma_y_u * (sample_mean / var + y_o / sigma_y_o)
112 fraction = (y_u - y_p) / (y_o - y_p)
113
114! Need to keep sum of the means, sum of the mean squared, sum of the var
115 sum_mean = sum_mean + y_u
116 sum_var = sum_var + sigma_y_u
117 sum_fraction = sum_fraction + fraction
118
119end do
120
121!write(*, *) 'Exact mean and var ', exact_mean, exact_var
122!write(*, *) 'Mean mean and var ', sum_mean / n_samples, sum_var / n_samples
123
124! Look at fraction that mean moves as another possible statistic
125fraction_ratio = (sum_fraction / n_samples) / exact_fraction
126!write(*, *) 'exact and mean fraction', exact_fraction, sum_fraction / n_samples
127!write(*, *) 'fraction ratio ', fraction_ratio
128
129mean_ratio = (sum_mean / n_samples) / exact_mean
130var_ratio = (sum_var / n_samples) / exact_var
131!write(*, *) 'mean var ratio ', mean_ratio, var_ratio
132
133! Returned factor is 1 / var_ratio
134factor = 1.0 / var_ratio
135
136! Verify that uncertainty is not increased
137!correct_var = exact_var / var_ratio
138!write(*, *) 'prior, exact, corr ', real(sigma_y_p), real(exact_var), &
139! real(correct_var)
140
141end subroutine sys_sim

145subroutine comp_correl(ens, n, correl)
146
147implicit none
148
149integer, intent(in) :: n
150double precision, intent(in) :: ens(2, n)
151double precision, intent(out) :: correl
152double precision :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
153
154
155sum_x = sum(ens(2, :))
156sum_y = sum(ens(1, :))
157sum_xy = sum(ens(2, :) * ens(1, :))
158sum_x2 = sum(ens(2, :) * ens(2, :))
159
160! Computation of correlation
161sum_y2 = sum(ens(1, :) * ens(1, :))
162
163correl = (n * sum_xy - sum_x * sum_y) / &
164 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
165
166end subroutine comp_correl

170subroutine sample_mean_var(x, n, mean, var)
171
172implicit none
173
174integer, intent(in) :: n
175double precision, intent(in) :: x(n)
176double precision, intent(out) :: mean, var
177
178double precision :: sx, s_x2
179
180sx = sum(x)
181s_x2 = sum(x * x)
182mean = sx / n
