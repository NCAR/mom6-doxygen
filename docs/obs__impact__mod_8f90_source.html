119subroutine create_impact_table(input_filename, output_filename, debug_flag)
120character(len=*), intent(in) :: input_filename
121character(len=*), intent(in) :: output_filename
122logical, intent(in), optional :: debug_flag
123
124! this is the routine that reads in the config file
125! and creates an output file that's suitable for reading
126! at runtime. so this is the 'create tool' part.
127
128type(type_toc) :: toc
129type(type_group) :: group_toc
130real(r8), allocatable :: table(:,:)
131integer :: funit
132
133! initialization and setup
134call initialize_module(debug)
135
136! set up space for the output table
137call allocate_impact_table(table)
138
139! read in types and qty strings, compute sizes,
140! allocate space for the toc and group types
141
142call init_strings_and_toc(toc, group_toc)
143
144
145this_state = state_undefined
146
147! read in ascii control file with and build table
148call error_handler(e_msg, 'obs_impact_tool', ' reading impacts from: '//trim(input_filename))
149
150
151linecount = 0
152!print *, 'file contents: '
153
154! loop until read returns nothing
155funit = open_file(input_filename, action='read')
156readloop: do
157 ! get the next line
158 read(funit, readformat, iostat=ios) readbuf
159 if (ios /= 0) exit readloop
160 linecount = linecount + 1
161
162!print *, 'read input line, was: '//trim(readbuf)
163
164 ! prepare msg in case of error
165 write(errline, '(A,I6,A)') 'error on line ', linecount, '; line contents: '
166
167 ! parse it into white-space separated words
168 call get_args_from_string(readbuf, wordcount, wordarray)
169 if (wordcount < 1) cycle readloop
170
171 ! count actual number of words (e.g. stop at comment char)
172 ! and note which ones are already in the TOC. do not add
173 ! new ones if they do not yet exist.
174 call search_toc(wordcount, wordarray, itemlist, toc)
175
176 ! change state based on keywords, add new group names to toc
177 call state_change(wordcount, wordarray, itemlist, toc, group_toc, this_state, otherwork)
178
179 if (otherwork) then
180 if (this_state == state_defgroup) then
181 ! add to groups if defining groups.
182 call add_to_group(wordcount, itemlist, current_group_id, group_toc, toc)
183 else if (this_state == state_defnotgroup) then
184 ! remove from groups if defining exclusion groups.
185 call sub_from_group(wordcount, itemlist, current_group_id, this_category, group_toc, toc)
186 else if (this_state == state_defimpact) then
187 ! build impact table based on what is going on now
188 call update_impact(table, wordcount, itemlist, wordarray, toc, group_toc)
189 endif
190 endif
191
192enddo readloop
193call close_file(funit)
194
195call error_handler(e_msg, 'obs_impact_tool', ' end of input')
196
197! anything still missing will be set to 1.0
198where (table(:,:) == missing_r8) table(:,:) = 1.0_r8
199
200call output_impact_table(output_filename, table, toc)
201
202if (.true.) then ! was debug
203 write(msgstring, *) 'closing file ', trim(output_filename)
204 call error_handler(e_msg, 'obs_impact_tool', msgstring)
205endif
206
207call finalize_tables(toc, group_toc, table)
208
209end subroutine create_impact_table

274 if (ios /= 0) exit readloop

280 write(errline, '(A,I6,A)') 'error on line ', linecount, '; line contents: '
281

283 read(readbuf, *, iostat=ios) typename, qtyname, rvalue

297where (table(:,:) == missing_r8) table(:,:) = 1.0_r8
298
299! debug only?
300do j=0, ubound(table, 2)
301 do i=1, ubound(table, 1)
302 if (table(i, j) /= 1.0_r8) then
303 write(errline, '(2A33,F12.6)') get_name_for_type_of_obs(i), get_name_for_quantity(j), table(i,j)
304 call error_handler(e_msg, 'read_impact_table', errline)

314! RUNTIME:
315subroutine set_impact(table, typename, qtyname, rvalue, allow_any_values, anyvals_string)
316
317real(r8), intent(inout) :: table(:,0:)
318character(len=*), intent(in) :: typename
319character(len=*), intent(in) :: qtyname
320real(r8), intent(in) :: rvalue
321logical, intent(in) :: allow_any_values
322character(len=*), intent(in), optional :: anyvals_string
323
324! change this to true to force values to be 0 or 1
325logical, save :: fully_on_or_off = .false.
326
327integer :: index1, index2
328
329! expect exactly 3 tokens on these lines:
330! type qty real(r8)
331
332index1 = get_index_for_type_of_obs(typename)
333index2 = get_index_for_quantity(qtyname)
334!print *, 'in set_impact, type, qty index values are: ', index1, index2
335
336if (index1 < 0) then
337 call error_handler(e_err, 'obs_impact', &
338 'first word on line is unrecognized specific obs TYPE', &
339 source, text2=errline, text3=readbuf)
340endif
341if (index2 < 0) then
342 call error_handler(e_err, 'obs_impact', &
343 'second word on line is unrecognized generic QTY', &
344 source, text2=errline, text3=readbuf)
345endif
346
347! options for the actual impact value:
348! 1. anything goes
349! 2. values restricted
350! a. must be 0.0 or 1.0 only
351! b. must be between 0.0 and 1.0 inclusive
352
353if (.not. allow_any_values) then
354 if (fully_on_or_off) then
355 if (rvalue /= 0.0_r8 .and. rvalue /= 1.0_r8) then
356 call error_handler(e_err, 'obs_impact', &
357 'impact values must be 0 or 1', &
358 source, text2=readbuf)
359 endif
360 else
361 if (present(anyvals_string)) then
362 msgstring3='set "'//trim(anyvals_string)//'=.true." in namelist to allow'
363 else
364 msgstring3=""
365 endif
366 if (rvalue < 0.0_r8 .or. rvalue > 1.0_r8) then
367 call error_handler(e_err, 'obs_impact', &
368 'impact values must be between 0 and 1, inclusive', &
369 source, text2=readbuf, text3=msgstring3)
370 endif
371 endif
372endif
373
374! ok, we have the first, second, and value. update the table.
375
376if ((table(index1, index2) /= missing_r8) .and. &
377 (table(index1, index2) /= rvalue)) then
378 write(msgstring, *) 'error; different impact factor already set for this pair'
379 write(msgstring2, *) 'previous value was', table(index1, index2), ' new value is ', rvalue
380 write(msgstring3, *) trim(typename), ' and ', trim(qtyname)
381 call error_handler(e_err, 'obs_impact', msgstring, &
382 source, text2=msgstring2, text3=msgstring3)
383
384endif
385
386table(index1, index2) = rvalue
387
388end subroutine set_impact

418subroutine init_strings_and_toc(toc, group_toc)
419
420type(type_toc), intent(inout) :: toc
421type(type_group), intent(inout) :: group_toc
422
423integer :: i, type_count, qty_count, ntoc
424integer :: groupsize, membersize
425character(len=obstypelength), allocatable :: knownqtys(:)
426character(len=obstypelength), allocatable :: knowntypes(:)
427
428! first, get all possible generic qtys from obs_kind_mod
429! BEWARE!! the first qty number is 0!!!!!
430qty_count = get_num_quantities()
431allocate(knownqtys(0:qty_count))
432do i=0, qty_count
433 knownqtys(i) = get_name_for_quantity(i)
434 call to_upper(knownqtys(i))
435enddo
436
437! then, get all possible specific types from obs_kind_mod
438type_count = get_num_types_of_obs()
439allocate(knowntypes(type_count))
440do i=1, type_count
441 knowntypes(i) = get_name_for_type_of_obs(i)
442 call to_upper(knowntypes(i))
443enddo
444
445
446! toc size - arbitrary size, could be larger
447ntoc = (type_count + qty_count) * 4
448
449! space for groups - both sizes completely arbitrary
450groupsize = qty_count * 2
451membersize = qty_count * 4
452group_toc%group_count = groupsize
453allocate(group_toc%groups(groupsize))
454do i=1, groupsize
455 group_toc%groups(i)%member_count = 0
456 allocate(group_toc%groups(i)%group_members(membersize))
457enddo
458
459allocate(toc%toc_entries(ntoc))
460toc%toc_count = 0
461
462call build_toc(qty_count, knownqtys, type_count, knowntypes, toc)
463
464deallocate(knownqtys, knowntypes)
465
466end subroutine init_strings_and_toc

737subroutine state_change(wordcount, wordarray, itemlist, toc, group_toc, this_state, otherwork)
738
739integer, intent(inout) :: wordcount
740character(len=*), intent(in) :: wordarray(:)
741integer, intent(in) :: itemlist(:)
742type(type_toc), intent(inout) :: toc
743type(type_group), intent(inout) :: group_toc
744integer, intent(inout) :: this_state
745logical, intent(out) :: otherwork
746
747! wordcount has been updated to take into account comment chars
748! words have been uppercased already, and itemlist is -1 if the
749! word is not yet in the toc.
750
751! if the current line is changing state, there is no other work
752! to be done. if it is not a state line, it is adding items to
753! a block or defining a relationship and needs to be processed further.
754otherwork = .false.
755
756! blank lines, comment lines, etc cannot change state
757if (wordcount < 1) return
758
759if (itemlist(1) < 0) then
760 call error_handler(e_err, 'state_change', &
761 'first word on line is unrecognized keyword, QTY, TYPE, or group name', &
762 source, text2=errline, text3=readbuf)
763endif
764
765
766select case (this_state)
767
768 case (state_undefined)
769 if (is_keyword(itemlist(1), 'GROUP', toc)) then
770 !GROUP - start definition
771 if (wordcount < 2) then
772 call error_handler(e_err, 'state_change', &
773 'GROUP keyword must be followed by a group name', &
774 source, text2=errline, text3=readbuf)
775 endif
776 if (itemlist(2) > 0) then
777 call error_handler(e_err, 'state_change', &
778 'group name '//trim(wordarray(2))//' already in use', &
779 source, text2=errline, text3=readbuf)
780 endif
781
782 this_state = state_defgroup
783 current_group_name = wordarray(2)
784 call to_upper(current_group_name)
785 current_group_id = current_group_id + 1
786 call add_toc(current_group_name, toc, entry_group, current_group_id)
787!print *, 'adding group name to toc, name,id:', trim(wordarray(2)), current_group_id
788
789 else if (is_keyword(itemlist(1), 'IMPACT', toc)) then
790 !IMPACT - start definition
791 if (wordcount > 1) then
792 call error_handler(e_err, 'state_change', &
793 'IMPACT keyword should not be followed by additional text', &
794 source, text2=errline, text3=readbuf)
795 endif
796
797 this_state = state_defimpact
798
799 else if (is_keyword(itemlist(1), 'END', toc)) then
800 !END - error if not inside another block
801 call error_handler(e_err, 'state_change', &
802 'END keyword found while not inside a GROUP or IMPACT block', &
803 source, text2=errline, text3=readbuf)
804
805 else
806 ! if we are already outside a block, there should not be any other text
807 call error_handler(e_err, 'state_change', &
808 'text found while not inside a GROUP or IMPACT block', &
809 source, text2=errline, text3=readbuf)
810 endif
811
812 case (state_defgroup, state_defnotgroup)
813 if (is_keyword(itemlist(1), 'GROUP', toc)) then
814 !GROUP while already in a group defn - error
815 call error_handler(e_err, 'state_change', 'GROUP keyword found inside a GROUP block', &
816 source, text2=errline, text3=readbuf)
817 else if (is_keyword(itemlist(1), 'IMPACT', toc)) then
818 !IMPACT while already in a group defn - error
819 call error_handler(e_err, 'state_change', 'IMPACT keyword found inside a GROUP block', &
820 source, text2=errline, text3=readbuf)
821 else if (is_keyword(itemlist(1), 'END', toc)) then
822 !END - change state if word 2 matches, else error
823 if (wordcount < 2) then
824 call error_handler(e_err, 'state_change', &
825 'END keyword found but missing second word GROUP', &
826 source, text2=errline, text3=readbuf)
827 endif
828 if (.not. is_keyword(itemlist(2), 'GROUP', toc)) then
829 call error_handler(e_err, 'state_change', &
830 'END keyword found but second word not GROUP', &
831 source, text2=errline, text3=readbuf)
832 endif
833
834 this_state = state_undefined
835
836 else if ((is_keyword(itemlist(1), 'ALL', toc)) .or. &
837 (is_keyword(itemlist(1), 'ALLQTYS', toc)) .or. &
838 (is_keyword(itemlist(1), 'ALLTYPES', toc))) then
839
840 !ALL and friends - add all, and then if word 2 is EXCEPT, take away
841 if (this_state == state_defnotgroup) then
842 call error_handler(e_err, 'state_change', &
843 'ALL keyword found but already excluding items', &
844 source, text2=errline, text3=readbuf)
845 endif
846
847 this_category = itemlist(1)
848
849 ! require group be empty to start.
850 if (get_group_count(current_group_id, group_toc) /= 0) then
851 call error_handler(e_err, 'state_change', &
852 'ALL EXCEPT must be first lines inside a group', &
853 source, text2=errline, text3=readbuf)
854 endif
855
856 ! add all qtys, types, or types+qtys to group.
857 call add_all_to_group(toc, current_group_id, itemlist(1), group_toc)
858
859 if (wordcount >= 2) then
860 if (.not. is_keyword(itemlist(2), 'EXCEPT', toc)) then
861 call error_handler(e_err, 'state_change', &
862 'ALL keyword found but second word not EXCEPT', &
863 source, text2=errline, text3=readbuf)
864 endif
865
866!print *, 'changing state to defnotgroup'
867 ! finish any qtys/groups on rest of this line.
868 if (wordcount > 2) &
869 call sub_from_group(wordcount-2, itemlist(3:wordcount), current_group_id, itemlist(1), group_toc, toc)
870
871 endif
872
873 ! subsequent lines remove qtys and groups from group.
874 this_state = state_defnotgroup
875
876 else
877 ! anything else is fine; no state change
878 otherwork = .true.
879
880 endif
881
882 case (state_defimpact)
883 if (is_keyword(itemlist(1), 'GROUP', toc)) then
884 !GROUP while already in a impact defn - error
885 call error_handler(e_err, 'state_change', 'GROUP keyword found inside an IMPACT block', &
886 source, text2=errline, text3=readbuf)
887 else if (is_keyword(itemlist(1), 'IMPACT', toc)) then
888 !IMPACT while already in a impact defn - error
889 call error_handler(e_err, 'state_change', 'IMPACT keyword found inside an IMPACT block', &
890 source, text2=errline, text3=readbuf)
891 else if (is_keyword(itemlist(1), 'END', toc)) then
892 !END - change state if word 2 matches, else error
893 if (wordcount < 2) then
894 call error_handler(e_err, 'state_change', &
895 'END keyword found but missing second word IMPACT', &
896 source, text2=errline, text3=readbuf)
897 endif
898 if (.not. is_keyword(itemlist(2), 'IMPACT', toc)) then
899 call error_handler(e_err, 'state_change', &
900 'END keyword found but second word not IMPACT', &
901 source, text2=errline, text3=readbuf)
902 endif
903
904 this_state = state_undefined
905
906 else
907 ! anything else is fine; no state change
908 otherwork = .true.
909
910 endif
911
912 case default
913 call error_handler(e_err,'state_change:','internal error: bad state value',source)
914end select
915
916end subroutine state_change

956
957integer, intent(in) :: wordcount
958integer, intent(in) :: itemlist(:)
959integer, intent(in) :: groupid
960type(type_group), intent(inout) :: group_toc
961type(type_toc), intent(inout) :: toc
962
963integer :: i, j, num_mem
964
965call check_groupindex(groupid, 'add_to_group', group_toc)
966
967do i = 1, wordcount
968 if (itemlist(i) < 0) then
969 call error_handler(e_err, 'add_to_group', &
970 'unrecognized item cannot be added to group', &
971 source, text2=errline, text3=readbuf)
972 endif
973
974 ! check for dups
975 do j=1, group_toc%groups(groupid)%member_count
976 if (group_toc%groups(groupid)%group_members(j) == itemlist(i)) then

979 source, text2=errline, text3=readbuf)
980 endif
981 enddo
982
983 ! make sure they are qtys or types, do not allow nested groups
984 if (get_toc_type(itemlist(i), toc) == entry_group) then
985 call error_handler(e_err, 'add_to_group', &
986 'group members must be DART QTYS or TYPES (nested groups not allowed)', &
987 source, text2=errline, text3=readbuf)
988 endif
989
990
991 num_mem = group_toc%groups(groupid)%member_count + 1
992
993 group_toc%groups(groupid)%member_count = num_mem
994 group_toc%groups(groupid)%group_members(num_mem) = itemlist(i)
995
996enddo
997

999end subroutine add_to_group

1050subroutine sub_from_group(wordcount, itemlist, groupid, category, group_toc, toc)
1051
1052integer, intent(in) :: wordcount
1053integer, intent(in) :: itemlist(:)
1054integer, intent(in) :: groupid
1055integer, intent(in) :: category
1056type(type_group), intent(inout) :: group_toc
1057type(type_toc), intent(inout) :: toc
1058
1059integer :: i, j, itemtype
1060logical :: isgroup
1061integer :: itemval, itemcount, subgroupid
1062logical :: doqtys, dotypes
1063
1064doqtys = .false.
1065dotypes = .false.
1066
1067if (is_keyword(category, 'ALL', toc)) then
1068 doqtys = .true.
1069 dotypes = .true.
1070else if (is_keyword(category, 'ALLQTYS', toc)) then
1071 doqtys = .true.
1072else if (is_keyword(category, 'ALLTYPES', toc)) then
1073 dotypes = .true.
1074else
1075 call error_handler(e_err, 'sub_from_group', &
1076 'internal error: ALLxxx keyword unrecognized', source)
1077endif
1078
1079call check_groupindex(groupid, 'sub_from_group', group_toc)
1080
1081do i = 1, wordcount
1082 if (itemlist(i) < 0) then
1083 call error_handler(e_err, 'sub_from_group', &
1084 'EXCEPT items must be known QTYS, TYPES, or groups', &
1085 source, text2=errline, text3=readbuf)
1086
1087 endif
1088
1089 ! see if itemlist(i) is an item or a group. if item, remove it.
1090 ! if group, iterate through the group and remove the member items.
1091 ! for now, do NOT allow nested groups, although at some point
1092 ! i guess we could start recursing. but it makes the
1093 ! error checking a nightmare, and hard to detect logical loops.
1094 ! for now, just say no.
1095
1096 call iteminfo(itemlist(i), itemval, isgroup, itemcount, toc, group_toc, qtyonly = .false.)
1097
1098 if (isgroup) subgroupid = itemval
1099
1100 do j=1, itemcount
1101
1102 if (isgroup) then
1103
1104 itemval = get_group_member_by_index(subgroupid, j, group_toc)
1105
1106 itemtype = get_toc_type(itemval, toc)
1107 if (.not. (itemtype == entry_dartqty .and. doqtys) .and. &
1108 .not. (itemtype == entry_darttype .and. dotypes)) then
1109 call error_handler(e_err, 'sub_from_group', &
1110 'error using EXCEPT; item '//trim(get_toc_name(itemval, toc))//' not a group member', &
1111 source, text2=errline, text3=readbuf)
1112 endif
1113
1114 else
1115 itemval = itemlist(i)
1116 if (.not. is_group_member(groupid, itemval, group_toc)) then
1117 call error_handler(e_err, 'sub_from_group', &
1118 'error using EXCEPT; item '//trim(get_toc_name(itemval, toc))//' not a member of group', &
1119 source, text2=errline, text3=readbuf)
1120 endif
1121 endif
1122
1123 call del_group_member(groupid, itemval, group_toc)
1124
1125 enddo
1126
1127enddo
1128
1129end subroutine sub_from_group

1313subroutine update_impact(table, wordcount, itemlist, wordarray, toc, group_toc)
1314
1315real(r8), intent(inout) :: table(:,0:)
1316integer, intent(in) :: wordcount
1317integer, intent(in) :: itemlist(:)
1318character(len=*), intent(in) :: wordarray(:)
1319type(type_toc), intent(in) :: toc
1320type(type_group), intent(in) :: group_toc
1321
1322real(r8) :: rvalue
1323integer :: itemval1, itemval2, groupid1, groupid2
1324integer :: groupitem1, groupitem2
1325integer :: itemcount1, itemcount2
1326logical :: isgroup1, isgroup2
1327integer :: i, j
1328
1329! expect exactly 3 tokens on these lines:
1330! qty_or_type_or_group qty_or_group real(r8)
1331! (where 2nd group cannot contain types)
1332
1333call check_impact_line(wordcount, itemlist, wordarray, rvalue)
1334
1335! ok, we have the first, second, and value. update the table.
1336! any combination of qty names or group names is allowed.
1337
1338call iteminfo(itemlist(1), itemval1, isgroup1, itemcount1, toc, group_toc, qtyonly=.false.)
1339call iteminfo(itemlist(2), itemval2, isgroup2, itemcount2, toc, group_toc, qtyonly=.true.)
1340
1341! both single items
1342if (.not. isgroup1 .and. .not. isgroup2) then
1343
1344 call fill_impact_table_values(table, itemval1, itemval2, rvalue, toc, itemlist(1), 'k/k')
1345
1346! single item to group
1347else if (.not. isgroup1 .and. isgroup2) then
1348 groupid2 = itemval2
1349 do i=1, itemcount2
1350 groupitem2 = get_group_member_by_index(groupid2, i, group_toc)
1351 call require_only_qtys_in_group(groupitem2, groupid2, toc)
1352 itemval2 = get_toc_value(groupitem2, toc)
1353 call fill_impact_table_values(table, itemval1, itemval2, rvalue, toc, itemlist(1), 'k/g')
1354 enddo
1355
1356! group to single item
1357else if (isgroup1 .and. .not. isgroup2) then
1358 groupid1 = itemval1
1359 itemval2 = get_toc_value(itemlist(2), toc)
1360 do i=1, itemcount1
1361 groupitem1 = get_group_member_by_index(groupid1, i, group_toc)
1362 itemval1 = get_toc_value(groupitem1, toc)
1363 call fill_impact_table_values(table, itemval1, itemval2, rvalue, toc, groupitem1, 'g/k')
1364 enddo
1365
1366! both groups
1367else if (isgroup1 .and. isgroup2) then
1368 groupid1 = itemval1
1369 groupid2 = itemval2
1370 do i=1, itemcount1
1371 groupitem1 = get_group_member_by_index(groupid1, i, group_toc)
1372 itemval1 = get_toc_value(groupitem1, toc)
