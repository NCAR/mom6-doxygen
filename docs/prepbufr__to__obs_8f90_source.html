170! the window start needs to be +1 second so no times
171! are in more than a single window.

192
193 call get_date(window_mid, fyear, fmonth, fday, fhour, fmin, fsec)

195

197 ! end bin time in the construct_obs_sequence routine. right now
198 ! the ascii intermediate files only have hours and no dates in them.
199 ! the current converter which creates them adds 24 to hours beyond
200 ! midnight for files which are centered on midnight and include obs

216 endif

218 write( input_name, input_filename_pattern) trim(input_filename_base), fyear, fmonth, fday, fhour
219 endif

221
222 write(*,*) ' input filename: "'//trim( input_name)//'"'
223 write(*,*) 'output filename: "'//trim(output_name)//'"'

236
237 call construct_obs_sequence(seq, obs_unit, gdays, window_start, window_end, inc_midnight)
238 call close_file(obs_unit)
239
240 ! construct input and output filenames
241 ! FIXME: make this consistent with prepbufr and the non-daily option:
242 next = window_mid + window_width
243 call get_date(next, fyear, fmonth, fday, fhour, fmin, fsec)
244 if (fhour == 0) then
245 hack = next - set_time(0, 1)
246 call get_date(hack, hyear, hmonth, hday, hhour, hmin, hsec)
247 write( input_name, input_filename_pattern) trim(input_filename_base), hyear, hmonth, hday, 24
248 else
249 write( input_name, input_filename_pattern) trim(input_filename_base), fyear, fmonth, fday, fhour
250 endif
251
252 write(*,*) ' opening aux input file: "'//trim(input_name)//'"'
253 obs_unit = open_file(input_name, form='formatted', action='read')
254
255 ! read the next available ascii intermediate file to collect any obs which are
256 ! exactly equal to the ending timestamp. this works for windows which are an even
257 ! multiple of 6H (the times in the original prepbufr files).
258 call construct_obs_sequence(seq, obs_unit, gdays, window_end, window_end, .false.)

260
261 ! output the sequence to a file
262 call write_obs_seq(seq, output_name)
263
264 ! release the memory of the seq.
265 call destroy_obs_sequence(seq)
266
267
268 window_start = window_end
269 window_end = window_start + window_width
270 window_mid = window_start + window_half
271 window_start = window_start + one_sec
272
273enddo
274
275call error_handler(e_msg,'prepbufr_to_obs','Finished successfully.',source,revision,revdate)
276call finalize_utilities()

284subroutine construct_obs_sequence (seq, iunit, gday, wstart, wend, wrap)
285!------------------------------------------------------------------------------
286! this function is to convert NCEP decoded BUFR data to DART sequence format
287!
288! FIXME: the prepbufr converter HAS to start putting a date as well as a time
289! in the ascii intermediate files, so i can jettison both the wrap and keep3
290! flags, and so we can have a single converter source (right now there is
291! prepbufr.f and prepbufr_03Z.f, which differ by about 3 lines.)
292!
293
294type(obs_sequence_type), intent(inout) :: seq
295integer, intent(in) :: iunit
296integer, intent(in) :: gday
297type(time_type), intent(in) :: wstart, wend
298logical, intent(in) :: wrap
299
300type(obs_type) :: obs, prev_obs
301integer :: i, num_copies, num_qc
302integer :: seconds
303integer :: hour, imin, sec
304integer :: obs_num
305integer :: syear, smonth, sday, shour, sminute, sseconds
306integer :: eyear, emonth, eday, ehour, eminute, eseconds
307type(time_type) :: current_day, time_obs, prev_time
308real(r8) :: bin_beg, bin_end
309
310! these can be initialized because they don't change
311! for multiple calls.
312integer, parameter :: num_fail_kinds = 8
313integer :: iskip(num_fail_kinds)
314integer, parameter :: fail_3Z = 1
315integer, parameter :: fail_timerange = 2
316integer, parameter :: fail_badloc = 3
317integer, parameter :: fail_areabox = 4
318integer, parameter :: fail_badkind = 5
319integer, parameter :: fail_notwanted = 6
320integer, parameter :: fail_badvert = 7
321integer, parameter :: fail_moisttype = 8
322character(len=32) :: skip_reasons(num_fail_kinds) = (/ &
323 'time too early (exactly 03Z) ', &
324 'time outside bin range ', &
325 'bad observation location ', &
326 'observation outside lat/lon box ', &
327 'unrecognized observation kind ', &
328 'obs type not on select list ', &
329 'bad vertical coordinate ', &
330 'unwanted moisture type ' /)
331
332
333integer :: rday, read_counter, io
334integer :: obs_prof, obs_kind, obs_kind_gen, which_vert, iqc, obstype, pc
335real(r8) :: obs_err, lon, lat, lev, zob, time, rcount, zob2
336real(r8) :: vloc, obs_value
337
338
339character(len = 8 ) :: obsdate
340character(len = 80) :: obsfile, label
341character(len = 6 ) :: subset
342logical :: pass, first_obs
343
344call get_date(wstart, syear, smonth, sday, shour, sminute, sseconds)
345call get_date(wend, eyear, emonth, eday, ehour, eminute, eseconds)
346if (debug) call print_date(wstart, 'window start')
347if (debug) call print_date(wend, 'window end')
348
349bin_beg = shour + sminute / 60.0_r8 + sseconds / 3600.0_r8
350bin_end = ehour + eminute / 60.0_r8 + eseconds / 3600.0_r8
351if (bin_end < bin_beg) then
352 bin_end = bin_end + 24.0_r8
353endif
354if (debug) print *, 'bin start, end: ', bin_beg, bin_end
355
356
357! Initialize the obs variable
358
359num_copies = 1
360num_qc = 1
361
362call init_obs(obs, num_copies, num_qc)
363call init_obs(prev_obs, num_copies, num_qc)
364
365obs_num = 0
366iskip(:) = 0
367first_obs = .true.
368
369! loop over all observations within the file
370!------------------------------------------------------------------------------
371
372read_counter = 1
373
374obsloop: do
375
376 read(iunit,880,end=200,iostat=io) obs_err, lon, lat, lev, zob, zob2, rcount, time, &
377 obstype, iqc, subset, pc
378 if (io /= 0) then
379 write(msgstring1,*)'read error was ',io,' for line ',read_counter
380 call error_handler(e_err,'construct_obs_sequence', msgstring1, source, revision, revdate)
381 endif
382
383 880 format(f5.2,2f9.4,e12.5,f7.2,f7.2,f9.0,f7.3,i4,i2,1x,a6,i2)
384
385 read_counter = read_counter + 1
386
387 !write(*, 880) obs_err, lon, lat, lev, zob, zob2, rcount, time, &
388 ! obstype, iqc, subset, pc
389
390!------------------------------------------------------------------------------
391 ! A 'day' is from 03:01Z of one day through 03Z of the next.
392 ! skip the observations at exact 03Z of the beginning of the day
393 ! (obs at 03Z the next day have a time of 27.)
394 ! (this should be deprecated eventually when we start putting dates
395 ! as well as times in the intermediate ascii files. for now, keep it.)
396 if(time == 3.0_r8 .and. daily) then
397 if (debug) write(*,*) 'invalid time. hours = ', time
398 iskip(fail_3z) = iskip(fail_3z) + 1
399 cycle obsloop
400 endif
401
402 ! reject obs outside the current time window
403 if(time < bin_beg .or. time > bin_end) then
404 if (debug) write(*,*) 'invalid time. hours = ', time
405 iskip(fail_timerange) = iskip(fail_timerange) + 1
406 cycle obsloop
407 endif
408
409 ! reject locations outside the valid values (shouldn't happen)
410 if((lon > 360.0_r8) .or. (lon < 0.0_r8) .or. &
411 (lat > 90.0_r8) .or. (lat < -90.0_r8)) then
412 if (debug) write(*,*) 'invalid location. lon,lat = ', lon, lat
413 iskip(fail_badloc) = iskip(fail_badloc) + 1
414 cycle obsloop
415 endif
416
417 ! reject observations outside the bounding box
418 if(lat < lat1 .or. lat > lat2 .or. &
419 .not. is_longitude_between(lon, lon1, lon2)) then
420 if (debug) write(*,*) 'invalid location. lon,lat = ', lon, lat
421 iskip(fail_areabox) = iskip(fail_areabox) + 1
422 cycle obsloop
423 endif
424
425 ! it seems when a machine writes out a number and then reads it back in
426 ! you should at least get consistent round-off errors. perhaps it is more
427 ! complex than it seems, but on the ibm xlf compiler, values of exactly
428 ! 90.0 and -90.0 are written out as (binary) doubles and when they are
429 ! read back in and compared to 90 and -90, they are not equal anymore.
430 ! the difference is in a very small but apparently significant digit,
431 ! so we error out in our 3d sphere locations module. this hack is a try
432 ! at adjusting the numbers by a small enough value that the location is
433 ! still the pole but isn't going to round out of range.
434 if (lat >= 89.9999_r8) then
435 lat = lat - 1.0e-12_r8
436 if (debug) write(*,*) 'lat adjusted down, now ', lat
437 else if (lat <= -89.9999_r8) then
438 lat = lat + 1.0e-12_r8
439 if (debug) write(*,*) 'lat adjusted up, now ', lat
440 endif
441
442 obs_prof = rcount/1000000
443
444
445! assign each observation the correct observation type
446!------------------------------------------------------------------------------
447
448 ! make sure we do not fall through the code below without setting
449 ! a valid obs kind (e.g. the obstype is one not listed)
450 obs_kind = -1
451
452 if(obs_prof == 1) then
453 obs_kind_gen = qty_temperature
454 if(obstype == 120 .or. obstype == 132) obs_kind = radiosonde_temperature
455 if(obstype == 130 .or. obstype == 131) obs_kind = aircraft_temperature
456 if(obstype == 133 ) obs_kind = acars_temperature
457 if(obstype == 161 .or. obstype == 163) obs_kind = atov_temperature
458 if(obstype == 171 .or. obstype == 173) obs_kind = atov_temperature
459 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_temperature
460 if(obstype == 181 .or. obstype == 183) obs_kind = land_sfc_temperature
461 endif
462
463 if(obs_prof == 5) then
464 if ( zob2 == 0.0_r8 .and. include_specific_humidity ) then
465 obs_kind_gen = qty_specific_humidity
466 if(obstype == 120 .or. obstype == 132) obs_kind = radiosonde_specific_humidity
467 if(obstype == 130 .or. obstype == 131) obs_kind = aircraft_specific_humidity
468 if(obstype == 133 ) obs_kind = acars_specific_humidity
469 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_specific_humidity
470 if(obstype == 181 .or. obstype == 183) obs_kind = land_sfc_specific_humidity
471 else if ( zob2 == 1.0_r8 .and. include_relative_humidity ) then
472 obs_kind_gen = qty_relative_humidity
473 if(obstype == 120 .or. obstype == 132) obs_kind = radiosonde_relative_humidity
474 if(obstype == 130 .or. obstype == 131) obs_kind = aircraft_relative_humidity
475 if(obstype == 133 ) obs_kind = acars_relative_humidity
476 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_relative_humidity
477 if(obstype == 181 .or. obstype == 183) obs_kind = land_sfc_relative_humidity
478 else if ( zob2 == 2.0_r8 .and. include_dewpoint ) then
479 obs_kind_gen = qty_dewpoint
480 if(obstype == 120 .or. obstype == 132) obs_kind = radiosonde_dewpoint
481 if(obstype == 130 .or. obstype == 131) obs_kind = aircraft_dewpoint
482 if(obstype == 133 ) obs_kind = acars_dewpoint
483 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_dewpoint
484 if(obstype == 181 .or. obstype == 183) obs_kind = land_sfc_dewpoint
485 endif
486 endif
487
488 if(obs_prof == 3) then
489 obs_kind_gen = qty_surface_pressure
490 if(obstype == 120 .or. obstype == 132) obs_kind = radiosonde_surface_altimeter
491 if(obstype == 180 .or. obstype == 182) obs_kind = marine_sfc_altimeter
492 if(obstype == 181 ) obs_kind = land_sfc_altimeter
493 endif
494
495 if(obs_prof == 2) then
496 obs_kind_gen = qty_u_wind_component
497 if(obstype == 220 .or. obstype == 232) obs_kind = radiosonde_u_wind_component
498 if(obstype == 221 ) obs_kind = radiosonde_u_wind_component
499 if(obstype == 230 .or. obstype == 231) obs_kind = aircraft_u_wind_component
500 if(obstype == 233 ) obs_kind = acars_u_wind_component
501 if(obstype == 242 .or. obstype == 243) obs_kind = sat_u_wind_component
502 if(obstype == 245 .or. obstype == 246) obs_kind = sat_u_wind_component
503 if(obstype == 252 .or. obstype == 253) obs_kind = sat_u_wind_component
504 if(obstype == 255 ) obs_kind = sat_u_wind_component
505 if(obstype == 280 .or. obstype == 282) obs_kind = marine_sfc_u_wind_component
506 if(obstype == 281 .or. obstype == 284) obs_kind = land_sfc_u_wind_component
507 endif
508
509 if(obs_prof == 9) then
510 obs_kind_gen = qty_v_wind_component
511 if(obstype == 220 .or. obstype == 232) obs_kind = radiosonde_v_wind_component
512 if(obstype == 221 ) obs_kind = radiosonde_v_wind_component
513 if(obstype == 230 .or. obstype == 231) obs_kind = aircraft_v_wind_component
514 if(obstype == 233 ) obs_kind = acars_v_wind_component
515 if(obstype == 242 .or. obstype == 243) obs_kind = sat_v_wind_component
516 if(obstype == 245 .or. obstype == 246) obs_kind = sat_v_wind_component
517 if(obstype == 252 .or. obstype == 253) obs_kind = sat_v_wind_component
518 if(obstype == 255 ) obs_kind = sat_v_wind_component
519 if(obstype == 280 .or. obstype == 282) obs_kind = marine_sfc_v_wind_component
520 if(obstype == 281 .or. obstype == 284) obs_kind = land_sfc_v_wind_component
521 endif
522
523 if (obs_kind < 0) then
524 ! the "real" fix if the record type is not found might actually be to
525 ! accept all record types within valid ranges, and depend on the first
526 ! preprocessing steps (in the prepbufr converter) to remove obs record
527 ! types which are not desired. for now, avoid giving them the wrong type
528 ! and quietly loop.
529 if (obs_prof == 5) then
530 if (debug) write(*,*) 'unwanted moisture obs_prof, skipping', obs_prof, obstype, zob2
531 iskip(fail_moisttype) = iskip(fail_moisttype) + 1
532 else
533 if (debug) write(*,*) 'unrecognized obs_prof or obstype, skipping', obs_prof, obstype
534 iskip(fail_badkind) = iskip(fail_badkind) + 1
535 endif
536 cycle obsloop
537 endif
538
539! check to see if this observation is desired
540!------------------------------------------------------------------------------
541
542 ! if select_obs is true, we are going to include all observations
543 ! and we skip the selection code below.
544 if(select_obs) then
545
546 ! assume we are going to ignore this obs, unless it is
547 ! specifically included by one of the selections below.
548 pass = .true.
549
550 ! select the specific NCEP obs types
551 if( (adpupa .and. (subset =='ADPUPA')) .or. &
552 (aircar .and. (subset =='AIRCAR')) .or. &
553 (aircft .and. (subset =='AIRCFT')) .or. &
554 (satemp .and. (subset =='SATEMP')) .or. &
555 (sfcshp .and. (subset =='SFCSHP')) .or. &
556 (adpsfc .and. (subset =='ADPSFC')) .or. &
557 (satwnd .and. (subset =='SATWND')) ) then
558
559 ! then select the obs kind requested
560 if( (obs_t .and. (obs_kind_gen == qty_temperature )) .or. &
561 (obs_u .and. (obs_kind_gen == qty_u_wind_component )) .or. &
562 (obs_v .and. (obs_kind_gen == qty_v_wind_component )) .or. &
563 (obs_ps .and. (obs_kind_gen == qty_surface_pressure)) .or. &
564 (obs_qv .and. (obs_kind_gen == qty_specific_humidity)) .or. &
565 (include_relative_humidity .and. (obs_kind_gen == qty_relative_humidity)) .or. &
566 (include_dewpoint .and. (obs_kind_gen == qty_dewpoint)) ) then
567 pass = .false.
568 endif
569
570 endif
571
572 ! if pass is still true, we want to ignore this obs.
573 if(pass) then
574 if (debug) write(*,*) 'obs skipped because not on wanted list. subset, obs_kind = ', subset, obs_kind_gen
575 iskip(fail_notwanted) = iskip(fail_notwanted) + 1
576 cycle obsloop
577 endif
578
579 endif
580
581! process this observation
582!------------------------------------------------------------------------------
583
584 obs_num = obs_num + 1
585
586 ! print a reassuring message after every Nth processed obs.
587 ! if requested, print in the form of a timestamp.
588 ! the default is just a plain string with the current obs count.
589 if(mod(obs_num, print_every_nth) == 0) then
590 write(label, *) 'obs count = ', obs_num
591 if (print_timestamps) then
592 call timestamp(string1=label, pos='brief')
593 else
594 write(*,*) trim(label)
595 endif
596 endif
597 if(obs_num == max_num_obs) then
598 print*, 'Max limit for observation count reached. Increase value in namelist'
599 stop
600 endif
601
602 ! set vertical coordinate for upper-air observations
603 if (subset == 'AIRCAR' .or. subset == 'AIRCFT' .or. &
604 subset == 'SATEMP' .or. &
605 subset == 'SATWND' .or. subset == 'ADPUPA' ) then
606 vloc = lev*100.0_r8 ! convert from mb to Pascal
607 which_vert = vertispressure
608 endif
609
610 ! set vertical coordinate for surface observations
611 if (subset == 'ADPSFC' .or. subset == 'SFCSHP') then
612 vloc = lev
613 ! some obs have elevation of 1e12 - toss those.
614 if ( subset == 'SFCSHP' .and. vloc > 4000.0_r8) then
615 iskip(fail_badvert) = iskip(fail_badvert) + 1
616 cycle obsloop
617 endif
618 which_vert = vertissurface
619 endif
620
621 ! set obs value and error if necessary
622 if ( obs_kind == land_sfc_altimeter .or. obs_kind == marine_sfc_altimeter &
623 .or. obs_kind == radiosonde_surface_altimeter ) then
624 vloc = lev ! station height, not used now for Ps obs
625 which_vert = vertissurface
626 obs_value = compute_altimeter(zob, vloc) ! altimeter is hPa
627 elseif(obs_kind_gen == qty_surface_pressure) then
628 obs_value = zob * 100.0_r8 ! for Ps variable only in Pascal
629 vloc = lev ! station height, not used now for Ps obs
630 which_vert = vertissurface
631 else if(obs_kind_gen == qty_specific_humidity) then
632 obs_err = obs_err*1.0e-3_r8
633 obs_value = zob*1.0e-3_r8 ! for Q variable to kg/kg
634 else
635 obs_value = zob ! for T, U, V, RH, Tdew
636 endif
637
638 if (time < 24.0_r8) then
639 if (wrap) then
640 rday = gday - 1
641 else
642 rday = gday
643 endif
644 seconds = time * 3600
645 else
646 rday = gday
647 seconds = (time - 24.0_r8) * 3600
648 endif
649
650 ! fill the time_obs variable to pass into add routine
651 time_obs = set_time(seconds, rday)
652
653 ! create the obs_def for this observation, add to sequence
654
655 call create_3d_obs(lat, lon, vloc, which_vert, obs_value, &
656 obs_kind, obs_err, rday, seconds, real(iqc,r8), obs)
657 call add_obs_to_seq(seq, obs, time_obs, prev_obs, prev_time, first_obs)
658
659end do obsloop
660
661200 continue
662
663print*, 'num obs used = ', obs_num, ' total obs skipped = ', sum(iskip)
664do i=1, num_fail_kinds
665 if (iskip(i) > 0) print *, iskip(i), 'skipped because ', skip_reasons(i)
666enddo
667
668end subroutine construct_obs_sequence
669
670
671end program prepbufr_to_obs
