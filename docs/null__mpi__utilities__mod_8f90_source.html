135!> Initialize the utilities module, and print out a message including the
136!> program name.
137subroutine initialize_mpi_utilities(progname, alternatename, communicator)
138
139character(len=*), intent(in), optional :: progname
140character(len=*), intent(in), optional :: alternatename
141integer, intent(in), optional :: communicator
142
143if ( module_initialized ) then
144 ! return without calling the code below multiple times. Print out a warning each
145 ! time this is called again because it may indicate an error in logic. In a well-
146 ! constructed program the initialize routine will only be called once.
147 write(errstring, *) 'initialize_mpi_utilities has already been called'
148 call error_handler(e_warn,'initialize_mpi_utilities', errstring, source)
149 return
150endif
151
152module_initialized = .true.
153
154! Initialize the module with utilities
155call initialize_utilities(progname, alternatename)
156
157if (present(progname)) then
158 if (len_trim(progname) <= len(saved_progname)) then
159 saved_progname = trim(progname)
160 else
161 saved_progname = progname(1:len(saved_progname))
162 endif
163endif
164
165! log info if requested
166
167if (r8 /= digits12) then
168 datasize = 4
169 write(errstring, *) "Using real * 4 for datasize of r8"
170 call error_handler(e_msg,'initialize_mpi_utilities: ',errstring, source)
171else
172 datasize = 8
173endif
174
175! non-MPI successfully initialized.
176call error_handler(e_msg,'initialize_mpi_utilities: ','Running single process', source)
177
178end subroutine initialize_mpi_utilities
