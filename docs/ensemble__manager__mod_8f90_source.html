124subroutine init_ensemble_manager(ens_handle, num_copies, &
125 num_vars, distribution_type_in, layout_type, transpose_type_in)
126
127type(ensemble_type), intent(out) :: ens_handle
128integer, intent(in) :: num_copies
129integer(i8), intent(in) :: num_vars
130integer, intent(in), optional :: distribution_type_in
131integer, intent(in), optional :: layout_type
132integer, intent(in), optional :: transpose_type_in ! no vars, transposable, transpose and duplicate
133
134integer :: iunit, io
135integer :: transpose_type
136
137! Distribution type controls pe layout of storage; Default is 1. 1 is only one implemented.
138if(.not. present(distribution_type_in)) then
139 ens_handle%distribution_type = 1
140else
141 ! Check for error: only type 1 implemented for now
142 if(distribution_type_in /= 1) call error_handler(e_err, 'init_ensemble_manager', &
143 'only distribution_type 1 is implemented', source)
144 ens_handle%distribution_type = distribution_type_in
145endif
146
147! First call to init_ensemble_manager must initialize module and read namelist
148if ( .not. module_initialized ) then
149 ! Initialize the module with utilities
150 module_initialized = .true.
151
152 ! Read the namelist entry
153 call find_namelist_in_file("input.nml", "ensemble_manager_nml", iunit)
154 read(iunit, nml = ensemble_manager_nml, iostat = io)
155 call check_namelist_read(iunit, io, "ensemble_manager_nml")
156
157 if (do_nml_file()) write(nmlfileunit, nml=ensemble_manager_nml)
158 if (do_nml_term()) write( * , nml=ensemble_manager_nml)
159
160 ! Get mpi information for this process; it's stored in module storage
161 num_pes = task_count()
162
163endif
164
165! Optional layout_type argument to assign how my_pe is related to my_task_id
166! layout_type can be set individually for each ensemble handle. It is not advisable to do this
167! because get_obs_ens assumes that the layout is the same for each ensemble handle.
168if(.not. present(layout_type) ) then
169 ens_handle%layout_type = layout ! namelist option
170else
171 ens_handle%layout_type = layout_type
172endif
173
174! Check for error: only layout_types 1 and 2 are implemented
175if (ens_handle%layout_type /= 1 .and. ens_handle%layout_type /=2 ) then
176 call error_handler(e_err, 'init_ensemble_manager', &
177 'only layout values 1 (standard), 2 (round-robin) allowed ', source)
178endif
179
180! Optional transpose type:
181! 1 not transposable - always copy complete
182! 2 transposable - has a vars array
183! 3 duplicatable - really only 1 copy, but this gets duplicated as vars array on every task during a transpose
184if (.not. present(transpose_type_in)) then
185 transpose_type = 1
186else
187 transpose_type = transpose_type_in
188endif
189
190ens_handle%transpose_type = transpose_type
191
192allocate(ens_handle%task_to_pe_list(num_pes))
193allocate(ens_handle%pe_to_task_list(num_pes))
194
195call assign_tasks_to_pes(ens_handle, num_copies, ens_handle%layout_type)
196ens_handle%my_pe = map_task_to_pe(ens_handle, my_task_id())
197
198! Set the global storage bounds for the number of copies and variables
199ens_handle%num_copies = num_copies
200ens_handle%num_vars = num_vars
201
202! For debugging, error checking
203ens_handle%id_num = global_counter
204global_counter = global_counter + 1
205
206! This controls which way the two transpose routines order their
207! loops: single sender with multiple receivers, or single receiver
208! with multiple senders. For small problems it doesn't matter;
209! for very large ensemble sizes and very large MPI task counts you
210! will want to profile each combination and pick the fastest.
211if (communication_configuration == 1) then
212 use_copy2var_send_loop = .true.
213 use_var2copy_rec_loop = .true.
214else if (communication_configuration == 2) then
215 use_copy2var_send_loop = .false.
216 use_var2copy_rec_loop = .true.
217else if (communication_configuration == 3) then
218 use_copy2var_send_loop = .true.
219 use_var2copy_rec_loop = .false.
220else if (communication_configuration == 4) then
221 use_copy2var_send_loop = .false.
222 use_var2copy_rec_loop = .false.
223else
224 write(msgstring, *) 'communication_configuration is', communication_configuration
225 call error_handler(e_err, 'init_ensemble_manager', &
226 'communication_configuration must be between 1 and 4', &
227 source, text2=msgstring)
228endif
229
230if(debug .and. my_task_id()==0) then
231 print*, 'pe_to_task_list', ens_handle%pe_to_task_list
232 print*, 'task_to_pe_list', ens_handle%task_to_pe_list
233endif
234
235! Figure out how the ensemble copies are partitioned
236call set_up_ens_distribution(ens_handle)
237
238ens_handle%num_extras = 0 ! This can be changed by calling set_num_extra_copies
239
240if (debug) call print_ens_handle(ens_handle)
241
242end subroutine init_ensemble_manager

246subroutine get_copy(receiving_pe, ens_handle, copy, vars, mtime)
247
248! The requested copy of the vars from the ensemble is stored into
249! vars on the receiving_pe process. Only the receiving pe and the pe
250! that actually stores the copy do anything. The vars storage only need be
251! allocated on the receiving_pe. Time only transferred if mtime present.
252! An example application of this routine would be for writing state
253! space diagnostics. The filter only uses a single writer to an open file,
254! currently process 0. To write out the ensemble mean state to the
255! prior and posterior netcdf state space diagnostic files, process 0
256! would need to request and receive the copy containing the ensemble
257! mean from whatever process stores it when processes have a subset
258! of copies of all variables.
259
260integer, intent(in) :: receiving_pe
261type(ensemble_type), intent(in) :: ens_handle
262integer, intent(in) :: copy
263real(r8), intent(out) :: vars(:)
264type(time_type), intent(out), optional :: mtime
265
266integer :: owner, owners_index
267
268! Verify that requested copy exists
269if(copy < 1 .or. copy > ens_handle%num_copies) then
270 write(msgstring, *) 'Requested copy: ', copy, ' is > maximum copy: ', ens_handle%num_copies
271 call error_handler(e_err,'get_copy', msgstring, source)
272endif
273
274! Make sure that vars has enough space to handle the answer
275if(ens_handle%my_pe == receiving_pe) then !HK I think only the receiver needs the space
276 if(size(vars) < ens_handle%num_vars) then
277 write(msgstring, *) 'Size of vars: ', size(vars), ' Must be at least ', ens_handle%num_vars
278 call error_handler(e_err,'get_copy', msgstring, source)
279 endif
280endif
281
282! Figure out which PE stores this copy and what its local storage index is
283call get_copy_owner_index(ens_handle, copy, owner, owners_index)
284
285!----------- Block of code that must be done by receiving pe -----------------------------
286if(ens_handle%my_pe == receiving_pe) then
287 ! If PE that stores is the same, just copy and return
288 if(ens_handle%my_pe == owner) then
289 vars = ens_handle%vars(:, owners_index)
290 if(present(mtime)) mtime = ens_handle%time(owners_index)
291 ! If I'm the receiving PE and also the owner, I'm all finished; return
292 return
293 endif
294
295 ! Otherwise, must wait to receive vars and time from storing pe
296 call receive_from(map_pe_to_task(ens_handle, owner), vars, mtime)
297endif
298
299!----- Block of code that must be done by PE that stores the copy IF it is NOT receiver -----
300if(ens_handle%my_pe == owner) then
301 ! Send copy to receiving pe
302
303 if(present(mtime)) then
304 call send_to(map_pe_to_task(ens_handle, receiving_pe), ens_handle%vars(:, owners_index), ens_handle%time(owners_index))
305 else
306 call send_to(map_pe_to_task(ens_handle, receiving_pe), ens_handle%vars(:, owners_index))
307 endif
308endif
309!------ End of block ---------------------------------------------------------------------
310
311end subroutine get_copy

315subroutine put_copy(sending_pe, ens_handle, copy, vars, mtime)
316
317! The vars on the sending_pe is stored on the pe
318! that stores this copy. Only the sending_pe and storing pe do anything. The
319! vars storage only needs to be allocated on the sending_pe. An example
320! use is when process 0 in the filter reads the observation keys from the
321! obs_sequence file and then needs to place this as the appropriate copy
322! in the obs_ensemble file. This copy is stored by some process that may
323! not be 0.
324
325integer, intent(in) :: sending_pe
326type(ensemble_type), intent(inout) :: ens_handle
327integer, intent(in) :: copy
328real(r8), intent(in) :: vars(:)
329type(time_type), intent(in), optional :: mtime
330
331integer :: owner, owners_index
332
333if(copy < 1 .or. copy > ens_handle%num_copies) then
334 write(msgstring, *) 'Requested copy: ', copy, ' is > maximum copy: ', ens_handle%num_copies
335 call error_handler(e_err,'put_copy', msgstring, source)
336endif
337
338! Make sure that num_vars has enough space to handle the answer
339if(ens_handle%num_vars < size(vars)) then
340 write(msgstring, *) 'Size of vars: ', size(vars), ' Cannot be more than ', ens_handle%num_vars
341 call error_handler(e_err,'put_copy', msgstring, source)
342endif
343
344! What PE stores this copy and what is its local storage index
345call get_copy_owner_index(ens_handle, copy, owner, owners_index)
346
347! Block of code that must be done by PE that is to send the copy
348if(ens_handle%my_pe == sending_pe) then
349 ! If PE that stores is the same, just copy and return
350 if(ens_handle%my_pe == owner) then
351 ens_handle%vars(:, owners_index) = vars
352 if(present(mtime)) ens_handle%time(owners_index) = mtime
353 ! If I'm the sending PE and also the owner, I'm all finished; return
354 return
355 endif
356
357 ! Otherwise, must send vars and possibly time to storing pe
358 call send_to(map_pe_to_task(ens_handle, owner), vars, mtime)
359
360endif
361
362! Block of code that must be done by PE that stores the copy IF it is NOT sender
363if(ens_handle%my_pe == owner) then
364 ! Need to receive copy from sending_pe
365 if(present(mtime)) then
366 call receive_from(map_pe_to_task(ens_handle, sending_pe), ens_handle%vars(:, owners_index), ens_handle%time(owners_index))
367 else
368 call receive_from(map_pe_to_task(ens_handle, sending_pe), ens_handle%vars(:, owners_index))
369 endif
370endif
371
372end subroutine put_copy

376subroutine broadcast_copy(ens_handle, copy, arraydata)
377
378! find which PE has the global copy number and have it broadcast
379! that copy to all the other PEs. arraydata is an output on
380! all PEs, even on the PE which is the owner it is separate
381! storage from the vars array in the ensemble handle.
382
383type(ensemble_type), intent(in) :: ens_handle
384integer, intent(in) :: copy
385real(r8), intent(out) :: arraydata(:)
386
387integer :: owner, owners_index
388
389if(copy < 1 .or. copy > ens_handle%num_copies) then
390 write(msgstring, *) 'Requested copy: ', copy, ' is > maximum copy: ', ens_handle%num_copies
391 call error_handler(e_err,'broadcast_copy', msgstring, source)
392endif
393
394! Make sure that arraydata has enough space to handle the answer
395if(size(arraydata) < ens_handle%num_vars) then
396 write(msgstring, *) 'Size of arraydata: ', size(arraydata), ' must be at least ', ens_handle%num_vars
397 call error_handler(e_err,'broadcast_copy', msgstring, source)
398endif
399
400! What PE stores this copy and what is its local storage index
401call get_copy_owner_index(ens_handle, copy, owner, owners_index)
402
403! First block of code that must be done by PE that is to send the copy
404if(ens_handle%my_pe == owner) then
405 arraydata = ens_handle%vars(:, owners_index)
406 call broadcast_send(map_pe_to_task(ens_handle, owner), arraydata)
407else
408 call broadcast_recv(map_pe_to_task(ens_handle, owner), arraydata)
409endif
410
411end subroutine broadcast_copy

469subroutine duplicate_ens(ens1, ens2, duplicate_time)
470
471type(ensemble_type), intent(in) :: ens1
472type(ensemble_type), intent(inout) :: ens2
473logical, intent(in) :: duplicate_time
474
475! Name was changed from copy_ens to avoid confusion in naming.
476! Should only be used if the ens1%vars storage is current (each pe has subset of
477! copies of all vars).
478! If duplicate_time is true, also copies the time information from ens1 to ens2.
479! If duplicate_time is false, the times in ens2 are left unchanged.
480
481! Check to make sure that the ensembles are compatible
482if(ens1%num_copies /= ens2%num_copies) then
483 write(msgstring, *) 'num_copies ', ens1%num_copies, ' and ', ens2%num_copies, &
484 'must be equal'
485 call error_handler(e_err,'duplicate_ens', msgstring, source)
486endif
487if(ens1%num_vars /= ens2%num_vars) then
488 write(msgstring, *) 'num_vars ', ens1%num_vars, ' and ', ens2%num_vars, &
489 'must be equal'
490 call error_handler(e_err,'duplicate_ens', msgstring, source)
491endif
492if(ens1%distribution_type /= ens2%distribution_type) then
493 write(msgstring, *) 'distribution_type ', ens1%distribution_type, ' and ', &
494 ens2%distribution_type, 'must be equal'
495 call error_handler(e_err,'duplicate_ens', msgstring, source)
496endif
497
498! Duplicate each copy that is stored locally on this process.
499ens2%vars = ens1%vars
500
501! Duplicate time if requested
502if(duplicate_time) ens2%time = ens1%time
503
504end subroutine duplicate_ens

578subroutine set_up_ens_distribution(ens_handle)
579
580! Figures out how to lay-out the copy complete and vars complete
581! distributions. The distribution_type identifies
582! different options. Only distribution_type 1 is implemented.
583! This puts every nth var or copy on a given processor where n is the
584! total number of processes.
585
586type (ensemble_type), intent(inout) :: ens_handle
587
588integer :: num_per_pe_below, num_left_over, i
589integer(i8) :: per_pe, suggest_pes
590
591! Check that there are enough pes for the state
592per_pe = ens_handle%num_vars / num_pes
593if (per_pe >= (huge(i)-100) ) then
594 suggest_pes = ( ens_handle%num_vars / (huge(i)) ) * 2
595 write(msgstring, '(A,I5,1X,A)') &
596 'not enough MPI tasks for the model size, suggest at least ' , &
597 suggest_pes, 'tasks'
598 call error_handler(e_err, 'set_up_ens_distribution', msgstring, source)
599endif
600
601! Option 1: Maximum separation for both vars and copies
602! Compute the total number of copies I'll get for var complete
603num_per_pe_below = ens_handle%num_copies / num_pes
604num_left_over = ens_handle%num_copies - num_per_pe_below * num_pes
605if(num_left_over >= (ens_handle%my_pe + 1)) then
606 ens_handle%my_num_copies = num_per_pe_below + 1
607else
608 ens_handle%my_num_copies = num_per_pe_below
609endif
610
611! Do the same thing for copy complete: figure out which vars I get

903subroutine all_vars_to_all_copies(ens_handle, label)
904
905! Converts from having subset of copies of all variables to having
906! all copies of a subset of variables on a given PE.
907!
908! updated version of the original routine. here, all tasks send
909! while 1 receives. original was all tasks receiving while 1 sends.
910! apparently the sends can overlap and the execution time is less.
911! HK Added namelist option (use_var2copy_rec_loop) to select updated
912! or original version of the routine.
913! Default: use updated version
914
915type (ensemble_type), intent(inout) :: ens_handle
916character (len=*), intent(in), optional :: label
917
918integer(i8), allocatable :: var_list(:)
919integer, allocatable :: copy_list(:)
920real(r8), allocatable :: transfer_temp(:)
921
922integer(i8) :: num_vars
923integer :: num_copies, my_num_vars, my_num_copies, my_pe
924integer :: max_num_vars, max_num_copies, num_copies_to_receive
925integer :: sending_pe, recv_pe, k, sv, num_vars_to_send, copy
926integer :: global_ens_index
927
928! only output if there is a label
929if (present(label)) then
930 call timestamp_message('vars_to_copies start: '//label, alltasks=.true.)
931endif
932
933! Accelerated version for single process
934if(num_pes == 1) then
935 ens_handle%copies = transpose(ens_handle%vars)
936 return
937end if
938
939! Short var definitions
940num_copies = ens_handle%num_copies
941num_vars = ens_handle%num_vars
942my_num_vars = ens_handle%my_num_vars
943my_num_copies = ens_handle%my_num_copies
944my_pe = ens_handle%my_pe
945
946! What is maximum number of vars stored on a copy complete pe?
947max_num_vars = get_max_num_vars(ens_handle, num_vars)
948
949! What is maximum number of copies stored on a var complete pe?
950max_num_copies = get_max_num_copies(ens_handle, num_copies)
951
952allocate(var_list(max_num_vars), transfer_temp(max_num_vars), &
953 copy_list(max_num_copies))
954
955if ( use_var2copy_rec_loop .eqv. .true. ) then ! use updated version
956
957 ! Loop to give each pe a turn to receive its copies
958 receiving_pe_loop: do recv_pe = 0, num_pes - 1
959 ! If I'm the receiving pe, do this block
960 if(my_pe == recv_pe) then
961
962 ! Figure out what piece to receive from each other PE and receive it
963 receive_from_each: do sending_pe = 0, num_pes - 1
964 call get_copy_list(ens_handle, num_copies, sending_pe, copy_list, num_copies_to_receive)
965
966 ! Loop to receive for each copy stored on my_pe
967 all_my_copies: do k = 1, num_copies_to_receive
968
969 global_ens_index = copy_list(k)
970
971 ! If sending_pe is receiving_pe, just copy
972 if(sending_pe == recv_pe) then
973 do sv = 1, my_num_vars
974 ens_handle%copies(global_ens_index, sv) = ens_handle%vars(ens_handle%my_vars(sv), k)
975 end do
976 else
977 if (num_copies_to_receive > 0) then
978 ! Otherwise, receive this part from the sending pe
979 call receive_from(map_pe_to_task(ens_handle, sending_pe), transfer_temp(1:my_num_vars))
980
981 ! Copy the transfer array to my local storage
982 ens_handle%copies(global_ens_index, :) = transfer_temp(1:my_num_vars)
983 endif
984 endif
985 end do all_my_copies
986 end do receive_from_each
987 else
988 ! I'm the sending PE, figure out what vars of my copies I'll send.
989 call get_var_list(ens_handle, num_vars, recv_pe, var_list, num_vars_to_send)
990
991 do k = 1, my_num_copies
992 do sv = 1, num_vars_to_send
993 ! Have to use temp because %var section is not contiguous storage
994 transfer_temp(sv) = ens_handle%vars(var_list(sv), k)
995 enddo
996 call send_to(map_pe_to_task(ens_handle, recv_pe), transfer_temp(1:num_vars_to_send))
997 end do
998
999 endif
1000 end do receiving_pe_loop
1001
1002else ! use older version
1003
1004 ! Loop to give each pe a turn to send its vars
1005 sending_pe_loop: do sending_pe = 0, num_pes - 1
1006 ! If I'm the sending pe, do this block
1007 if(my_pe == sending_pe) then
1008 ! Figure out what piece to send to each other PE and send it
1009 send_to_each: do recv_pe = 0, num_pes - 1
1010 call get_var_list(ens_handle, num_vars, recv_pe, var_list, num_vars_to_send)
1011
1012 if (num_vars_to_send > 0) then
1013 ! Loop to send these vars for each copy stored on my_pe
1014 all_my_copies_send_loop: do k = 1, my_num_copies
1015
1016 ! Fill up the transfer array
1017 do sv = 1, num_vars_to_send
1018 transfer_temp(sv) = ens_handle%vars(var_list(sv), k)
1019 end do
1020
1021 ! If sending_pe is receiving_pe, just copy
1022 if(sending_pe == recv_pe) then
1023 global_ens_index = ens_handle%my_copies(k)
1024 ens_handle%copies(global_ens_index, :) = transfer_temp(1:num_vars_to_send)
1025 else
1026 ! Otherwise, ship this off
1027 call send_to(map_pe_to_task(ens_handle, recv_pe), transfer_temp(1:num_vars_to_send))
1028 endif
1029 end do all_my_copies_send_loop
1030 endif
1031 end do send_to_each
1032
1033 else
1034 ! I'm not the sending PE, figure out what copies of my vars I'll receive from sending_pe
1035 call get_copy_list(ens_handle, num_copies, sending_pe, copy_list, num_copies_to_receive)
1036
1037 do copy = 1, num_copies_to_receive
1038 if (my_num_vars > 0) then
1039 ! Have to use temp because %copies section is not contiguous storage
1040 call receive_from(map_pe_to_task(ens_handle, sending_pe), transfer_temp(1:my_num_vars))
1041 ! Figure out which global ensemble member this is
1042 global_ens_index = copy_list(copy)
1043 ! Store this chunk in my local storage
1044 ens_handle%copies(global_ens_index, :) = transfer_temp(1:my_num_vars)
1045 endif
1046 end do
1047
1048 endif
1049
1050 end do sending_pe_loop
1051
1052endif
1053
1054
1055! Free up the temporary storage
1056deallocate(var_list, transfer_temp, copy_list)
1057
1058! only output if there is a label
1059if (present(label)) then
1060 call timestamp_message('vars_to_copies end: '//label, alltasks=.true.)
1061endif
1062
1063end subroutine all_vars_to_all_copies
