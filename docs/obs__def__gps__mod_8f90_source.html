89! Storage for the special information required for GPS RO observations
90!
91
92! Because we are currently only generating one observation type
93! (GPSRO_REFRACTIVITY), there must be enough of these to cover all gps
94! obs in all obs_seq files that are read in (e.g. for obs_diag if you
95! cover multiple days or weeks, you must have enough room for all of them.)
96! the local operator needs none of this additional info; the best approach
97! would be to keep a single QTY_GPSRO, but make 2 observation types.
98! the local has no additional metadata; the nonlocal needs one of these
99! allocated and filled in.
100integer :: max_gpsro_obs = 100000

109end type gps_nonlocal_type

124 subroutine initialize_module
125!------------------------------------------------------------------------------
126!
127! initialize global gps private key number and allocate space for obs data
128integer :: rc, iunit
129
130
131call register_module(source, revision, revdate)
132module_initialized = .true.
133
134! global count of all gps observations from any input file
135keycount = 0
136
137! Read the namelist entry
138call find_namelist_in_file("input.nml", "obs_def_gps_nml", iunit)
139read(iunit, nml = obs_def_gps_nml, iostat = rc)
140call check_namelist_read(iunit, rc, "obs_def_gps_nml")
141
142! Record the namelist values used for the run ...
143if (do_nml_file()) write(nmlfileunit, nml=obs_def_gps_nml)
144if (do_nml_term()) write( * , nml=obs_def_gps_nml)
145
146! find max number of gps obs which can be stored, and initialize type
147allocate(gps_data(max_gpsro_obs), stat = rc)
148if (rc /= 0) then
149 write(string1, *) 'initial allocation failed for gps observation data,', &
150 'itemcount = ', max_gpsro_obs
151 call error_handler(e_err,'initialize_module', string1, &
152 source, revision, revdate)
153endif
154
155end subroutine initialize_module

159 subroutine set_gpsro_ref(gpskey, nx, ny, nz, rfict0, ds, htop, subset0)
160!------------------------------------------------------------------------------
161!
162! increment key and set all private data for this observation
163
164integer, intent(out) :: gpskey
165real(r8), intent(in) :: nx, ny, nz, rfict0, ds, htop
166character(len=6), intent(in) :: subset0
167
168if ( .not. module_initialized ) call initialize_module
169
170keycount = keycount + 1
171gpskey = keycount
172
173if(gpskey > max_gpsro_obs) then
174 write(string1, *) 'key (',gpskey,') exceeds max_gpsro_obs (',max_gpsro_obs,')'
175 string2 = 'Increase max_gpsro_obs in input.nml &obs_def_gps_nml namelist.'
176 call error_handler(e_err,'read_gpsro_ref', string1, &
177 source, revision, revdate, text2=string2)
178endif
179
180gps_data(gpskey)%ray_direction(1) = nx
181gps_data(gpskey)%ray_direction(2) = ny
182gps_data(gpskey)%ray_direction(3) = nz
183gps_data(gpskey)%gpsro_ref_form = subset0
184
185gps_data(gpskey)%rfict = rfict0
186gps_data(gpskey)%step_size = ds
187gps_data(gpskey)%ray_top = htop

189end subroutine set_gpsro_ref

253 subroutine read_gpsro_ref(gpskey, ifile, fform)
254!------------------------------------------------------------------------------
255!
256! Every GPS observation has its own (metadata) gpskey.
257! When you read multiple gps observation sequence files, it is necessary
258! to track the total number of metadata gpskeys read, not just the number
259! in the current file.
260!
261
262integer, intent(out) :: gpskey
263integer, intent(in) :: ifile
264character(len=*), intent(in), optional :: fform
265
266integer :: keyin ! the metadata key in the current obs sequence
267
268real(r8) :: nx, ny, nz, rfict0, ds, htop
269character(len=6) :: subset0
270character(len=8) :: header
271
272if ( .not. module_initialized ) call initialize_module
273
274if (ascii_file_format(fform)) then
275 read(ifile, fmt='(a8, i8)') header, keyin ! throw away keyin
276 if(header /= 'gpsroref') then
277 call error_handler(e_err,'read_gpsro_ref', &
278 'Expected header "gpsroref" in input file', source, revision, revdate)
279 endif
280 read(ifile, *) rfict0, ds, htop, nx, ny, nz, subset0
281else
282 read(ifile) keyin ! read and throw away
283 read(ifile) rfict0, ds, htop, nx, ny, nz, subset0
284endif
285
286
287! increment key and set all private data for this observation
288call set_gpsro_ref(gpskey, nx, ny, nz, rfict0, ds, htop, subset0)
289
290end subroutine read_gpsro_ref

293subroutine interactive_gpsro_ref(gpskey)
294!----------------------------------------------------------------------
295!
296! Interactively prompt for the info needed to create a gps refractivity
297! observation. Increments the key number and returns it.
298
299integer, intent(out) :: gpskey
300
301real(r8) :: nx, ny, nz, rfict0, ds, htop
302character(len=6) :: subset0
303integer :: gpstype
304
305
306if ( .not. module_initialized ) call initialize_module
307
308!Now interactively obtain reflectivity type information
309! valid choices are local or non-local
310
311write(*, *)
312write(*, *) 'Beginning to inquire information on reflectivity type.'
313write(*, *)
314
315100 continue
316write(*, *) 'Enter 1 for local refractivity (GPSREF)'
317write(*, *) 'Enter 2 for non-local refractivity/excess phase delay (GPSEXC)'
318write(*, *)

320read(*,*) gpstype
321
322select case (gpstype)
323 case (1)
324 subset0 = 'GPSREF'
325 case (2)
326 subset0 = 'GPSEXC'
327 case default
328 write(*,*) 'Bad value, must enter 1 or 2'
329 goto 100
330end select
331
332if (gpstype == 2) then
333 ! FIXME: i have no idea what valid values are for any
334 ! of the following items, so i cannot add any error checking or
335 ! guidance for the user.
336
337 write(*, *)
338 write(*, *) 'Enter X, Y, Z value for ray direction'
339 write(*, *)
340 read(*,*) nx, ny, nz
341
342 write(*, *)
343 write(*, *) 'Enter local curvature radius'
344 write(*, *)
345 read(*,*) rfict0
346
347 write(*, *)
348 write(*, *) 'Enter step size'
349 write(*, *)
350 read(*,*) ds
351
352 write(*, *)
353 write(*, *) 'Enter ray top'
354 write(*, *)
355 read(*,*) htop
356else
357 nx = 0.0
358 ny = 0.0
359 nz = 0.0
360 rfict0 = 0.0
361 ds = 0.0
362 htop = 0.0
363endif
364
365! increment key and set all private data for this observation
366call set_gpsro_ref(gpskey, nx, ny, nz, rfict0, ds, htop, subset0)

368write(*, *)
369write(*, *) 'End of specialized section for gps observation data.'
370write(*, *) 'You will now have to enter the regular obs information.'
371write(*, *)

375!> Distributed version of get_expected_gpsro_ref
376 subroutine get_expected_gpsro_ref(state_handle, ens_size, location, gpskey, ro_ref, istatus)
377!------------------------------------------------------------------------------
378!
379! Purpose: Calculate GPS RO local refractivity or non_local (integrated)
380! refractivity (excess phase, Sergey Sokolovskiy et al., 2005)
381!------------------------------------------------------------------------------
382!
383! inputs:
384! state_vector: DART state vector
385!
386! output parameters:
387! ro_ref: modeled local refractivity (N-1)*1.0e6 or non_local
388! refractivity (excess phase, m)
389! (according to the input data parameter subset)
390! istatus: =0 normal; =1 outside of domain.
391!------------------------------------------------------------------------------
392! Author: Hui Liu
393! Version 1.1: June 15, 2004: Initial version CAM
394!
395! Version 1.2: July 29, 2005: revised for new obs_def and WRF
396!------------------------------------------------------------------------------
397
398type(ensemble_type), intent(in) :: state_handle
399integer, intent(in) :: ens_size
400type(location_type), intent(in) :: location
401integer, intent(in) :: gpskey
402real(r8), intent(out) :: ro_ref(ens_size)
403integer, intent(out) :: istatus(ens_size)
404
405! local variables. first set is per ensemble member.
406! second set is independent of a particular ensemble member.
407
408real(r8) :: phase(ens_size)
409real(r8) :: delta_phase1(ens_size), delta_phase2(ens_size)
410real(r8) :: ref_perigee(ens_size), ref1(ens_size), ref2(ens_size)
411real(r8) :: ref00(ens_size)
412integer :: this_istatus(ens_size)
413
414real(r8) :: nx, ny, nz ! unit tangent direction of ray at perigee
415real(r8) :: xo, yo, zo ! perigee location in Cartesian coordinate
416real(r8) :: height1, lat1, lon1
417real(r8) :: xx, yy, zz, dist_to_perigee
418real(r8) :: lon, lat, height, obsloc(3)
419integer :: iter
420logical :: return_now
421
422if ( .not. module_initialized ) call initialize_module
423
424if ( .not. is_vertical(location, "HEIGHT")) then
425 write(string1, *) 'vertical location must be height; gps obs key ', gpskey
426 call error_handler(e_err,'get_expected_gpsro_ref', string1, &
427 source, revision, revdate)
428endif
429
430obsloc = get_location(location)
431
432
433lon = obsloc(1) ! degree: 0 to 360
434lat = obsloc(2) ! degree: -90 to 90
435height = obsloc(3) ! (m)
436
437! to use track_status() start out with istatus all success.
438istatus = 0
439
440! calculate refractivity at perigee
441
442call ref_local(state_handle, ens_size, lat, lon, height, ref_perigee, this_istatus)
443call track_status(ens_size, this_istatus, ro_ref, istatus, return_now)
444if (return_now) return
445
446!>@todo we need to add a different obs type and kind for the local operator,
447!> so we can avoid allocating metadata because it isn't needed *at all*
448!> for the local operator. nsc 30oct2015
449choosetype: if(gps_data(gpskey)%gpsro_ref_form == 'GPSREF') then
450 ! use local refractivity
451 where (istatus == 0) ro_ref = ref_perigee * 1.0e6 ! in (N-1)*1.0e6, same with obs
452
453else ! gps_data(gpskey)%gpsro_ref_form == 'GPSEXC'
454
455 ! otherwise, use non_local refractivity(excess phase delay)
456
457 ! Initialization
458 phase = 0.0_r8
459 dist_to_perigee = 0.0_r8 ! distance to perigee from a point of the ray
460
461 nx = gps_data(gpskey)%ray_direction(1)
462 ny = gps_data(gpskey)%ray_direction(2)
463 nz = gps_data(gpskey)%ray_direction(3)
464
465 ! convert location of the perigee from geodetic to Cartesian coordinate
466
467 call geo2carte (height, lat, lon, xo, yo, zo, gps_data(gpskey)%rfict )
468
469 ! currently, use a straight line passing the perigee point as ray model.
470 ! later, more sophisticated ray models can be used.
471 !
472 ! Start the horizontal integrate of the model refractivity along a
473 ! straight line path in cartesian coordinate
474 !
475 ! (x-xo)/a = (y-yo)/b = (z-zo)/c, (a,b,c) is the line direction
476
477 ref1 = ref_perigee
478 ref2 = ref_perigee
479
480 iter = 0
481 integrate: do
482
483 iter = iter + 1
484 dist_to_perigee = dist_to_perigee + gps_data(gpskey)%step_size
485
486 ! integrate to one direction of the ray for one step
487 ! HK These are now different for each ensemble member
488 xx = xo + dist_to_perigee * nx
489 yy = yo + dist_to_perigee * ny
490 zz = zo + dist_to_perigee * nz
491
492 ! convert the location of the point to geodetic coordinates
493 ! height(m), lat, lon(deg)
494
495 call carte2geo(xx, yy, zz, height1, lat1, lon1, gps_data(gpskey)%rfict )
496 if (height1 >= gps_data(gpskey)%ray_top) exit integrate
497
498 ! get the refractivity at this ray point(ref00)
499 call ref_local(state_handle, ens_size, lat1, lon1, height1, ref00, this_istatus)
500 call track_status(ens_size, this_istatus, ro_ref, istatus, return_now)
501 if (return_now) return
502
503 ! get the excess phase due to this ray interval
504 where(istatus == 0) delta_phase1 = (ref1 + ref00) * gps_data(gpskey)%step_size * 0.5_r8
505
506 ! save the refractivity for integration of next ray interval
507 ref1 = ref00
508
509 ! integrate to the other direction of the ray
510 xx = xo - dist_to_perigee * nx
511 yy = yo - dist_to_perigee * ny
512 zz = zo - dist_to_perigee * nz
513
514 call carte2geo (xx, yy, zz, height1, lat1, lon1, gps_data(gpskey)%rfict )
515
516 ! get the refractivity at this ray point(ref00)
517 call ref_local(state_handle, ens_size, lat1, lon1, height1, ref00, this_istatus)
518 call track_status(ens_size, this_istatus, ro_ref, istatus, return_now)
519 if (return_now) return
520
521 ! get the excess phase due to this ray interval
522 where(istatus == 0) delta_phase2 = (ref2 + ref00) * gps_data(gpskey)%step_size * 0.5_r8
523
524 ! save the refractivity for integration of next ray interval
525 ref2 = ref00
526
527 where(istatus == 0) phase = phase + delta_phase1 + delta_phase2
528 ! print*, 'phase= ', phase, delta_phase1, delta_phase2
529
530 end do integrate
531
532 ! finish the integration of the excess phase along the ray
533
534 where(istatus == 0) ro_ref = phase ! in m
535
536 ! print*, 'xx = ', lon, lat, height, ro_ref
537
538endif choosetype
539
540! make sure return is missing_r8 if failure.
541!>@todo is the first line necessary? i believe the second one
542!> is a necessary test.
543where (istatus /= 0) ro_ref = missing_r8
544where (istatus == 0 .and. ro_ref < 0.0_r8)
545 istatus = 5
546 ro_ref = missing_r8
547endwhere
548
549end subroutine get_expected_gpsro_ref

551!> Distributed version. i removed the unused 'location' argument.
552! i changed this to be lat, lon, height - which is a more logical layout of
553! the location information. it used to be height, lat, lon - be careful if
554! anyone else calls this routine. it isn't public so no code outside this
555! module can be calling it. nsc 30 oct 2015

557subroutine ref_local(state_handle, ens_size, lat, lon, height, ref00, istatus0)
558!------------------------------------------------------------------------------
559!
560! Calculate local refractivity at any GPS ray point (lat, lon, height)
561!
562! inputs:
563! lat, lon, height: GPS observation location (units: degrees, degrees, meters)
564!
565! output:
566! ref00: modeled local refractivity at ray point(unit: N-1, ~1.0e-4 to e-6)
567!
568!------------------------------------------------------------------------------
569
570type(ensemble_type), intent(in) :: state_handle
571integer, intent(in) :: ens_size
572real(r8), intent(in) :: lon, lat, height
573real(r8), intent(out) :: ref00(ens_size)
574integer, intent(out) :: istatus0(ens_size)
575
576real(r8), parameter:: rd = 287.05_r8, rv = 461.51_r8, c1 = 77.6d-6 , &
577 c2 = 3.73d-1, rdorv = rd/rv
578
579real(r8) :: lon2
580real(r8) :: t(ens_size), q(ens_size), p(ens_size), ew(ens_size)
581integer :: this_istatus(ens_size)
582logical :: return_now
583type(location_type) :: location
584
585if ( .not. module_initialized ) call initialize_module
586
587! for integration of GPS ray path beyond the wraparound point
588lon2 = lon
589if(lon > 360.0_r8 ) lon2 = lon - 360.0_r8
590if(lon < 0.0_r8 ) lon2 = lon + 360.0_r8
591
592location = set_location(lon2, lat, height, vertisheight)
593
594! for sequential sets of calls to interpolate, start with a 0 istatus
595! and each call to track_status() will set new failures.
596istatus0 = 0
597ref00 = missing_r8
598
599call interpolate(state_handle, ens_size, location, qty_temperature, t, this_istatus)
600call track_status(ens_size, this_istatus, ref00, istatus0, return_now)
601if (return_now) return
602
603call interpolate(state_handle, ens_size, location, qty_specific_humidity, q, this_istatus)
604call track_status(ens_size, this_istatus, ref00, istatus0, return_now)
605if (return_now) return
606
607call interpolate(state_handle, ens_size, location, qty_pressure, p, this_istatus)
608call track_status(ens_size, this_istatus, ref00, istatus0, return_now)
609if (return_now) return
610
611! required variable units for calculation of GPS refractivity
612! t : Kelvin, from top to bottom
613! q : kg/kg, from top to bottom
614! p : mb
615
616where (istatus0 == 0)
617p = p * 0.01_r8 ! to mb
618
619ew = q * p/(rdorv + (1.0_r8-rdorv)*q )
620ref00 = c1*p/t + c2*ew/(t**2) ! (N-1)
621endwhere
622
623end subroutine ref_local

654 subroutine carte2geo (x1, x2, x3, s1, s2, s3, rfict0)
655!------------------------------------------------------------------------------
656!
657! Converts cartesian coordinates to geodetical.
658!
659! input parameters:
660! x - cartesian coordinates (x, y, z-coordinate, unit: m)
661!
662! output parameters:
663! s - geodetical coordinates
664! (height (m), latitude (deg), longitude (deg))
665! -90 to 90 0 to 360
666!------------------------------------------------------------------------------
667implicit none
668real(r8), intent(in) :: x1, x2, x3, rfict0
669real(r8), intent(out) :: s1, s2, s3
670
671real(r8), parameter :: crcl = 2.0_r8 * pi, &
672 crcl2 = 4.0_r8 * pi
673
674real(r8) :: rho, sphi, azmth
675
676if ( .not. module_initialized ) call initialize_module
677
678rho = sqrt(x1**2 + x2**2 + x3**2 )
679sphi = x3/rho
680s1 = rho - rfict0
681s2 = asin(sphi)
682azmth = atan2(x2, x1)
683s3 = mod((azmth + crcl2), crcl)
684
685s2 = s2 * rad2deg
686s3 = s3 * rad2deg
687
688end subroutine carte2geo
