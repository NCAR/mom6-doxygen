248function initialize_filenames(filename) result(file_handle)
249character(len=*), intent(in) :: filename(:)
250type(file_info_type) :: file_handle
251
252integer :: num_domains = 1
253integer :: imem
254
255character(len=256), allocatable :: file_array(:,:)
256character(len=512) :: my_base, my_desc
257
258allocate(file_array(num_ens, num_domains))
259file_array = reshape(filename, (/num_ens, num_domains/))
260
261call io_filenames_init(file_handle, &
262 ncopies = 1, &
263 cycling = .false., &
264 single_file = .false., &
265 restart_files = file_array)
266
267do imem = 1, num_ens
268 write(my_base,'(A,I2)') 'inens_', imem
269 write(my_desc,'(A,I2)') 'input ens', imem
270 call set_file_metadata(file_handle, &
271 cnum = imem, &
272 fnames = file_array(imem,:), &
273 basename = my_base, &
274 desc = my_desc)
275
276 call set_io_copy_flag(file_handle, &
277 cnum = imem, &
278 io_flag = read_copy)
279enddo
280
281end function initialize_filenames
