35!---------------------------------------------------------
36!> Gets all copies of an element of the state vector from the process who owns it
37!> Assumes ensemble complete. This differes from get_state as it now works on an
38!> array of state indices rather than a single index.
39subroutine get_state_array(x, my_index, state_ens_handle)
40
41real(r8), intent(out) :: x(data_count) !! all copies of an element of the state vector
42integer(i8), intent(in) :: my_index(data_count) !! index into state vector
43type(ensemble_type), intent(in) :: state_ens_handle
44
45real(r8) :: x_tmp(data_count) !! all copies of an element of the state vector
46logical :: not_done(data_count)
47integer :: i, e
48
49not_done = .true.
50
51! only grab state once unique state indices
52do e = 1, data_count
53 if (not_done(e)) then
54 x_tmp = get_state(my_index(e), state_ens_handle)
55 do i = e, data_count
56 if(my_index(i) == my_index(e)) then
57 x(i) = x_tmp(i)
58 not_done(i) = .false.
59 endif
60 enddo
61 endif
62enddo
63
64end subroutine get_state_array

71real(r8) :: x(data_count) !! all copies of an element of the state vector
72
73integer(i8), intent(in) :: my_index !! index into state vector
74type(ensemble_type), intent(in) :: ens_handle

86!---------------------------------------------------------
87!> Gets all copies of an element of the state vector from the process who owns it
88!> Assumes ensemble complete
89subroutine get_fwd(x, my_index, state_ens_handle)
90
91real(r8), intent(out) :: x(data_count) !! all copies of an element of the state vector
92integer(i8), intent(in) :: my_index !! index into state vector
93type(ensemble_type), intent(in) :: state_ens_handle
94
95integer :: owner_of_state !! task who owns the state
96integer :: element_index !! local index of element
97
98if (get_allow_transpose(state_ens_handle)) then
99 x = state_ens_handle%vars(my_index, 1:data_count)
100else
101
102 call get_var_owner_index(state_ens_handle, my_index, owner_of_state, element_index) ! pe
103
104 owner_of_state = map_pe_to_task(state_ens_handle, owner_of_state) ! task
105
106 if (my_task_id() == owner_of_state) then
107 !For future use if we need to access the local window
108 ! e.g. if %copies is modified while we are doing the forward operator.
109 ! Currently, %copies is static => same as what is in the window.
110 !x = get_local_state(element_index)
111 x = state_ens_handle%copies(1:data_count, element_index)
112 else
113 call get_from_fwd(owner_of_state, state_win, element_index, data_count, x)
114 endif
115endif
116
117end subroutine get_fwd
118
119!---------------------------------------------------------
120!> Gets all copies of an element of the state vector from the process who owns it
121!> Assumes ensemble complete
122subroutine get_mean(x, my_index, state_ens_handle)
123
124real(r8), intent(out) :: x(1) !! only grabing the mean
125integer(i8), intent(in) :: my_index !! index into state vector
126type(ensemble_type), intent(in) :: state_ens_handle
127
128integer :: owner_of_state !! task who owns the state
129integer :: element_index !! local index of element
130
131if (get_allow_transpose(mean_ens_handle)) then
132 x(1) = mean_ens_handle%vars(my_index, 1)
133else
134
135 call get_var_owner_index(state_ens_handle, my_index, owner_of_state, element_index) ! pe
136
137 owner_of_state = map_pe_to_task(state_ens_handle, owner_of_state) ! task
138
139 if (my_task_id() == owner_of_state) then
140 x(1) = mean_ens_handle%copies(1, element_index)
141 else
142 call get_from_mean(owner_of_state, mean_win, element_index, x(1))
143 endif
144
145endif
146
147end subroutine get_mean
