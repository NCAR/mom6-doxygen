6! Create synthetic observations evenly spaced on a sphere.
7!
8! This program creates observations without data. They have
9! a location, type, error variance, and a time. Run the output file
10! through create_fixed_network_seq if you want to create a time series
11! of these obs. Run the output through perfect_model_obs to add data.
12!
13! Alter as you wish to change obs types, number of types at each loc, etc.
14! If you change the obs types either set the obs error explicitly
15! or change the call to the obs_error module to match the correct obs type.
16!
17! There is a related matlab script in the obs_converters/even_sphere directory
18! which uses the same algorithm to distribute N points around a sphere. It has
19! additional plotting functions which may be useful.
20
21
22program create_sphere_obs

55! These default levels are standard radiosonde reporting pressures in hectopascals.
56real(r8), parameter :: default_levels(def_levels) = (/ 1000.0_r8, 925.0_r8, 850.0_r8, 700.0_r8, 500.0_r8, &
57 400.0_r8, 300.0_r8, 250.0_r8, 200.0_r8, 150.0_r8, &
58 100.0_r8, 70.0_r8, 50.0_r8, 30.0_r8, 20.0_r8, &
59 10.0_r8, 7.0_r8, 5.0_r8, 3.0_r8, 2.0_r8, 1.0_r8 /)
60
61! namelist variables
62
63! Number of roughly evenly distributed points in horizontal.
64! Final number of obs will be N * L times this, where N is the
65! number of different obs types at each location and L is the
66! number of pressure levels specified.
67
68integer :: number_of_locations
69
70! The level array here is specified in hectopascals (mb).
71! Observations are created and stored in pascals.
72! If this item is not altered by the namelist the default levels are used.
73
74real(r8) :: obs_levels(max_levels) = -1.0_r8
75
76! Date of obs
77integer :: year = 2008
78integer :: month = 10
79integer :: day = 1
80integer :: hour = 0
81
82
83namelist /create_sphere_obs_nml/ number_of_locations, &
84 obs_levels, &
85 year, month, day, hour
86
87
88
89! start of executable code
90
91call initialize_utilities('create_sphere_obs')
92
93call find_namelist_in_file("input.nml", "create_sphere_obs_nml", iunit)
94read(iunit, nml = create_sphere_obs_nml, iostat = io)
95call check_namelist_read(iunit, io, "create_sphere_obs_nml")
96
97if (do_nml_file()) write(nmlfileunit, nml=create_sphere_obs_nml)
98if (do_nml_term()) write( * , nml=create_sphere_obs_nml)
99
100! if the user doesn't set any levels, use the defaults.
101! otherwise, stop at the first -1 value or end of the array.
102
103if (all(obs_levels <= 0.0_r8)) then
104 obs_levels(1:def_levels) = default_levels
105 nlvls = def_levels
106else
107 nlvls = findloc(obs_levels, -1.0_r8, 1) - 1
108 if (nlvls <= 0) nlvls = size(obs_levels)
109endif
110
111! update this if you change the code to have more or fewer
112! obs types created at each location.
113ntypes = 3
114
115! Total number of observations at single time is: nlevels * nlocs * ntypes
116num_obs = nlvls * number_of_locations * ntypes
117
118! create a new (empty) sequence.
119! this program defines only the location, type, time, and error.
120! run perfect_model_obs to fill in the obs values if needed
121call init_obs_sequence(seq, 0, 0, num_obs)
122
123call init_obs(obs, 0, 0)
124call init_obs(prev_obs, 0, 0)
125
126first_obs = .true.
127
128call set_calendar_type(gregorian)
129time_obs = set_date(year, month, day, hour, 0, 0)
130call set_obs_def_time(obs_def, time_obs)
131
132
133inc = pi * (3.0_r8 - sqrt(5.0_r8))
134off = 2.0_r8 / number_of_locations
135
136do ob = 1, number_of_locations
137
138 y = (ob-1) * off - 1 + (off / 2)
139 r = sqrt(1 - y*y)
140 phi = (ob-1) * inc
141
142 x = cos(phi) * r
143 z = sin(phi) * r
144
145 ! Now convert to latitude and longitude, radians
146 ! first, then degrees
147
148 lon = atan2(y, x) + pi
149 lat = asin(z)
150
151 deglon = rad2deg * lon
152 deglat = rad2deg * lat
153
154 do lvl = 1, nlvls
155
156 ! in dart vertical locations using pressure are specified in units of pascals
157 call set_obs_def_location(obs_def, set_location(deglon, deglat, obs_levels(lvl)*100.0_r8, vertispressure))
158
159 ! the standard obs error routines take vertical locations in hectopascals/mb
160 t_err = rawin_temp_error(obs_levels(lvl))
161 call set_obs_def_error_variance(obs_def, t_err*t_err)
162 call set_obs_def_type_of_obs(obs_def, radiosonde_temperature)
163 call set_obs_def(obs, obs_def)
164 call add_obs_to_seq(seq, obs, time_obs, prev_obs, prev_time, first_obs)
165
166 w_err = rawin_wind_error(obs_levels(lvl))
167 call set_obs_def_error_variance(obs_def, w_err*w_err)
168 call set_obs_def_type_of_obs(obs_def, radiosonde_u_wind_component)
169 call set_obs_def(obs, obs_def)
170 call add_obs_to_seq(seq, obs, time_obs, prev_obs, prev_time, first_obs)
171
172 call set_obs_def_error_variance(obs_def, w_err*w_err)
173 call set_obs_def_type_of_obs(obs_def, radiosonde_v_wind_component)
174 call set_obs_def(obs, obs_def)
175 call add_obs_to_seq(seq, obs, time_obs, prev_obs, prev_time, first_obs)
176
177 enddo
178
179enddo
180
181call write_obs_seq(seq, 'even_sphere_seq.in')
182
183call finalize_utilities('create_sphere_obs')
184
185end program create_sphere_obs
