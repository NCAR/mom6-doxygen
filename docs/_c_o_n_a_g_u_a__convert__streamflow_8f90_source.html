135! -------------------------------------------------------
136! Start work
137! Initialize DART modules.
138call initialize_utilities('CONAGUA_convert_streamflow')
139
140call find_namelist_in_file('input.nml', 'CONAGUA_convert_streamflow_nml', iunit)
141read(iunit, nml = conagua_convert_streamflow_nml, iostat = io)
142call check_namelist_read(iunit, io, 'CONAGUA_convert_streamflow_nml')
143
144! Record the namelist values used for the run
145if (do_nml_file()) write(nmlfileunit, nml=conagua_convert_streamflow_nml)
146if (do_nml_term()) write( * , nml=conagua_convert_streamflow_nml)
147
148! time setup
149call set_calendar_type(gregorian)
150
151! each observation in this series will have a single observation value
152! and a quality control flag. the max possible number of obs needs to
153! be specified but it will only write out the actual number created.
154max_obs = 100000
155num_copies = 1
156num_qc = 1
157
158! call the initialization code, and initialize two empty observation types
159call static_init_obs_sequence()
160call init_obs(obs, num_copies, num_qc)
161call init_obs(prev_obs, num_copies, num_qc)
162first_obs = .true.
163
164! create a new, empty obs_seq file. you must give a max limit
165! on number of obs. increase the size if too small.
166call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
167
168! the first one needs to contain the string 'observation' and the
169! second needs the string 'QC'.
170call set_copy_meta_data(obs_seq, 1, 'observation')
171call set_qc_meta_data(obs_seq, 1, 'Data QC')
172
173! Several obs_seq quantities which are static for our observations:
174
175! surface obs. normally we set vert to the actual surface elevation,
176! we do not have it here, so set to 0 for now.
177vert = 0.0_r8
178
179! Set the DART data quality control. 0 is good data.
180! increasingly larger QC values are more questionable quality data.
181qc = 0.0_r8
182
183
184! -------------------------------------------------------
185! Read meta data file
186meta_data_file = trim(meta_data_file)
187meta_unit = open_file(meta_data_file, 'formatted', 'read')
188
189if (debug) then
190 write(string1,*) 'opened meta data file "'//trim(meta_data_file)//'"'
191 call error_handler(e_msg, source, string1)
192endif
193
194read(meta_unit, *) meta_header
195
196count_meta_stations = 0
197meta_loop: do i_rec = 1, max_stations
198 read(meta_unit, *, iostat=rcio) meta_data(i_rec)
199 if (rcio /= 0) then
200 if (debug) then
201 write(string1,*)'meta_loop: got bad read code getting line, rcio = ', rcio
202 call error_handler(e_msg, source, string1)
203 endif
204 exit meta_loop
205 endif
206 count_meta_stations = count_meta_stations + 1
207end do meta_loop
208call close_file(meta_unit)
209
210print *, ''
211print *, '----------------------------------'
212print *, 'reading meta data file: ', trim(meta_data_file)
213print *, ''
214print *, 'The following header fields were found: '
215do ii=1, n_meta_fields
216 write(*, fmt="(1x,a)", advance="no") trim(meta_header(ii)) // ', '
217end do
218print *,''
219print *,''
220print *, "count_meta_stations: ", count_meta_stations
221do ii = 1, count_meta_stations
222 print*, trim(meta_data(ii)%name), ': ', trim(meta_data(ii)%station)
223end do
224print *,''
225
226
227! -------------------------------------------------------
228! Pre-Process the data_file_list and check it's consistent with
229! 1) meta_data
230! 2) itself (all files at the same time resolution)
231
232list_unit = open_file(data_file_list, 'formatted', 'read')
233
234if (debug) then
235 write(string1,*) 'opened data_file_list "'//trim(data_file_list)//'"'
236 call error_handler(e_msg, source, string1)
237endif
238
239count_stations = 0
240list_loop: do i_rec = 1, max_stations
241 read(list_unit, '(a)', iostat=rcio) data_files(i_rec)
242 if (rcio /= 0) then
243 if (debug) then
244 write(string1,*)'list_loop: got bad read code getting line, rcio = ', rcio
245 call error_handler(e_msg, source, string1)
246 endif
247 exit list_loop
248 endif
249 count_stations = count_stations + 1
250end do list_loop
251call close_file(list_unit)
252
253! Validate that all the stations in the list are in the meta data.
254
255valid_loop: do ii = 1, count_stations
256 stn_file_name = data_files(ii)
257 stn_name_len = len_trim(stn_file_name)
258
259 ! As far as we can tell, CONAGUA stations are always 5 digits wide.
260 stn_resolution = stn_file_name((stn_name_len-10):(stn_name_len-9))
261 stn_name = stn_file_name((stn_name_len-8):(stn_name_len-4))
262
263 if (debug) then
264 write(string1,*)'stn_file_name substring is ', stn_file_name((stn_name_len-10):(stn_name_len-4))
265 write(string2,*)'station resolution is <',trim(stn_resolution),'>'
266 write(string3,*)'station name is <',trim(stn_name),'>'
267 call error_handler(e_msg, source, string1, text2=string2, text3=string3)
268 endif
269
270 if (.not. any(meta_data%name == stn_name)) then
271 write(string1,*)'Metadata not found for the requested station file: "'//trim(stn_file_name)//'"'
272 call error_handler(e_err, source, string1)
273 end if
274
275 if (ii == 1) then
276 hourly_resolution = (stn_resolution == 'hd')
277 else
278 if (hourly_resolution .and. stn_resolution .ne. 'hd') then
279 write(string1,*)"hourly resolution does not appear consistent across the data_file_list.txt"
280 call error_handler(e_err, source, string1)
281 end if
282 if (.not. hourly_resolution .and. stn_resolution .ne. 'dd') then
283 write(string1,*) "daily resolution does not appear consistent across the data_file_list.txt"
284 call error_handler(e_err, source, string1)
285 end if
286 end if
287
288end do valid_loop
289
290
291! -------------------------------------------------------
292! Daily observation processing.
293
294stn_daily_record_null%year = missing_int
295stn_daily_record_null%month = missing_int
296stn_daily_record_null%day_of_month_streamflow = missing_float
297
298station_loop: do i_stn = 1, count_stations
299
300 stn_file_name = data_files(i_stn)
301 stn_name_len = len(trim(stn_file_name))
302 stn_name = stn_file_name((stn_name_len-8):(stn_name_len-4))
303
304 wh_meta_stn = 0
305 wh_loop: do ii = 1, count_meta_stations
306 if (trim(meta_data(ii)%name) .eq. trim(stn_name)) then
307 wh_meta_stn = ii
308 exit wh_loop
309 end if
310 end do wh_loop
311
312 station_station = meta_data(wh_meta_stn)%station
313 lon = meta_data(wh_meta_stn)%lon
314 if (lon < 0.0_r8) lon = lon + 360.0_r8
315 lat = meta_data(wh_meta_stn)%lat
316
317 write(string1,*) ''
318 write(string2,*) '!-----------------------------------'
319 write(string3,*) 'Processing station ', i_stn, ': "'//trim(station_station)//'"'
320 call error_handler(e_msg,source,string1,text2=string2,text3=string3)
321
322 stn_unit = open_file(trim(stn_file_name), 'formatted', 'read')
323
324 write(string1,*) 'opened stn record file: "'//trim(stn_file_name)//'"'
325 call error_handler(e_msg,source,string1)
326
327 read(stn_unit, *, iostat=rcio) stn_daily_header
328
329 record_loop: do i_rec = 1, max_records
330 stn_daily_record = stn_daily_record_null
331 read(stn_unit, '(a)', iostat=rcio) line_in
332 if (rcio /= 0) then
333 if (debug) then
334 write(string1,*) 'recordloop: unable to read record for "'//trim(stn_name)//'"'
335 write(string2,*) ' rcio = ', rcio, '; record number = ', i_rec
336 call error_handler(e_msg,source,string1,text2=string2)
337 endif
338 exit record_loop
339 endif
340
341 ! This is a band-aid for records/lines which end with a missing value.
342 ! I'd prefer to keep it to a single comma so any "badly mangled" records might cause an error.
343 write(line_in, '(a)') trim(line_in) // ','
344
345 read(line_in, *, iostat=rcio) stn_daily_record
346 if (rcio /= 0) then
347 if (debug) then
348 write(string1,*) 'recordloop: unable to parse stn_daily_record for "'//trim(stn_name)//'"'
349 write(string2,*) 'rcio = ', rcio, '; record number = ', i_rec
350 call error_handler(e_msg,source,string1,text2=string2)
351 endif
352 exit record_loop
353 endif
354
355 year = stn_daily_record%year
356 month = stn_daily_record%month
357
358 day_loop: do day = 1, 31
359
360 obs_val = stn_daily_record%day_of_month_streamflow(day)
361
362 if (obs_val == missing_float .or. obs_val <= 0.000001_r8) then
363 cycle day_loop ! skip this recrod.
364 end if
365
366 ! Set the observation error variance here.
367 obs_err = max(obs_val*obs_fraction_for_error, obs_min_err_std)
368
369 time_obs = set_date(year, month, day, 0, 0, 0)
370 ! extract time of observation into gregorian day, sec.
371 call get_time(time_obs, osec, oday)
372
373 ! This routine can help tie gages to model spatial elements which can then
374 ! be used for more advanced localization.
375 ! Currently it is not used for gridded, though a layer in the Fulldom.nc file
376 ! could easily facilitate this. The bigger issue with gridded is lack of
377 ! connectivity approach currently (only calculated inside the model).
378 ! Just supplying dummy values now...
379
380 call set_streamflow_metadata(key, missing_gage_id, missing_link_id)
381
382 call create_3d_obs( &
383 lat, lon, vert, vertissurface, &
384 obs_val, &
385 stream_flow, &
386 obs_err, oday, osec, qc, obs, &
387 key )
388
389 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
390
391 end do day_loop
392
393 end do record_loop
394
395 call close_file(stn_unit)
396
397end do station_loop
398
399write(string1,*)'writing "'//trim(obs_out_file)//'"'
400write(string2,*)'obs_count = ', get_num_obs(obs_seq)
401
402! if we added any obs to the sequence, write it out to a file now.
403if ( get_num_obs(obs_seq) > 0 ) then
404 call error_handler(e_msg,source,string1,text2=string2)
405 call write_obs_seq(obs_seq, obs_out_file)
406else
407 write(string3,*)'no observations in output. Something is probably wrong.'
408 call error_handler(e_err,source,string1,text2=string2,text3=string3)
409endif
410
411! end of main program
412call finalize_utilities()
413
414end program conagua_convert_streamflow
