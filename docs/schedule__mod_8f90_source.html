109subroutine set_regular_schedule(schedule)
110! all of the necessary information to set a schedule is gotten from the namelist
111type(schedule_type), intent(out) :: schedule
112
113
114type(time_type) :: beg_time, end_time
115type(time_type) :: bininterval, binwidth
116type(time_type) :: timemin, timemax
117
118integer :: iepoch, nepochs, seconds, days
119
120character(len=32) :: str1, str2, str3
121
122if ( .not. module_initialized ) call schedule_init ! reads the namelist
123
124beg_time = set_date(first_bin_start(1), first_bin_start(2), &
125 first_bin_start(3), first_bin_start(4), &
126 first_bin_start(5), first_bin_start(6) )
127end_time = set_date(first_bin_end(1), first_bin_end(2), &
128 first_bin_end(3), first_bin_end(4), &
129 first_bin_end(5), first_bin_end(6) )
130timemax = set_date(last_bin_end(1), last_bin_end(2), &
131 last_bin_end(3), last_bin_end(4), &
132 last_bin_end(5), last_bin_end(6) )
133
134bininterval = set_time(bin_interval_seconds, bin_interval_days)
135
136! do some error-checking first
137
138if (end_time < beg_time) then
139 write(msgstring,*)'schedule_nml:first_bin_end must be at or after first_bin_start'
140 call error_handler(e_err,'set_regular_schedule',msgstring,source)
141endif
142
143if (timemax < end_time) then
144 write(msgstring,*)'schedule_nml:last_bin_end must be at or after first_bin_end'
145 call error_handler(e_err,'set_regular_schedule',msgstring,source)
146endif
147
148binwidth = beg_time - end_time
149call get_time(binwidth, seconds, days)
150binwidth = set_time(seconds, days)
151
152if (bininterval < binwidth) then
153 write(msgstring,*)'schedule_nml:bin interval must be >= bin width'
154 call error_handler(e_err,'set_regular_schedule',msgstring,source)
155endif
156
157! FIXME
158! Need to check that bin_interval is a multiple of the model advance step
159! for real assimilation experiments. This is not needed for observation-space
160! diagnostics or ...
161
162! Determine temporal bin characteristics.
163! The user input is not guaranteed to align on bin centers.
164! So -- we will assume the start time is correct and take strides till we
165! get past the last time of interest.
166! Nepochs will be the total number of time intervals of the period requested.
167
168timemin = end_time
169nepochs = 0
170nepochloop : do iepoch = 1,max_num_bins
171 if ( timemin > timemax ) exit nepochloop
172 nepochs = iepoch
173 timemin = timemin + bininterval
174enddo nepochloop
175
176if (do_output()) write(*,*)'Requesting ',nepochs,' assimilation periods.'
177
178if (nepochs < 1) then
179 write(msgstring,*)'schedule_nml:Requesting ZERO assimilation periods.'
180 call error_handler(e_err,'set_regular_schedule',msgstring,source)
181endif
182
183! Now that we know the number of assimilation epochs, allocate and fill.
184! Our assimilation bins start 1 second AFTER the stated time, and end ON
185! the stated time. If you specify a bin from 00Z to 03Z ... and again
186! from 03Z to 06Z ... the observation at 03Z is considered to be part
187! of the bin from 00Z to 03Z ONLY. Mathematically, ( 00Z, 03Z ]
188
189allocate(schedule%binstart( nepochs), schedule%binend( nepochs), &
190 schedule%epoch_start(nepochs), schedule%epoch_end(nepochs))
191
192schedule%binstart(1) = beg_time + set_time(1,0)
193schedule%binend( 1) = end_time
194schedule%binwidth = binwidth
195schedule%bininterval = bininterval
196schedule%current_bin = 0
197schedule%last_bin = .false.
198schedule%calendar = get_calendar_type()
199schedule%calendarstring = calendar
200schedule%num_bins = nepochs
201
202call get_time(schedule%binstart(1), seconds, days)
203schedule%epoch_start(1) = days + seconds/86400.0_digits12
204
205call get_time(schedule%binend( 1), seconds, days)
206schedule%epoch_end( 1) = days + seconds/86400.0_digits12
207
208binloop : do iepoch = 2,nepochs
209
210 schedule%binstart(iepoch) = schedule%binstart(iepoch-1) + bininterval
211 schedule%binend( iepoch) = schedule%binend( iepoch-1) + bininterval
212
213 call get_time(schedule%binstart(iepoch), seconds, days)
214 schedule%epoch_start(iepoch) = days + seconds/86400.0_digits12
215
216 call get_time(schedule%binend( iepoch), seconds, days)
217 schedule%epoch_end( iepoch) = days + seconds/86400.0_digits12
218
219enddo binloop
220
221if ( print_table ) then
222do iepoch = 1,nepochs
223 write( * ,*)
224 write(str1,'(''epoch '',i6,'' start'')')iepoch
225 write(str2,'(''epoch '',i6,'' center'')')iepoch
226 write(str3,'(''epoch '',i6,'' end'')')iepoch
227
228 call print_time( schedule%binstart(iepoch), str1)
229 call print_time( schedule%binend( iepoch), str3)
230
231 call print_date( schedule%binstart(iepoch), str1)
232 call print_date( schedule%binend( iepoch), str3)
233enddo
234write( * ,*)
235endif
236
237end subroutine set_regular_schedule

241subroutine get_timetype_from_schedule(mytime, schedule, iepoch, edge)
242
243type(time_type), intent(OUT) :: mytime
244type(schedule_type), intent(IN) :: schedule

248if (iepoch > schedule%num_bins) then
