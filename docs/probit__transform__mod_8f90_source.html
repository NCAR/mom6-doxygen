61subroutine transform_all_to_probit(ens_size, num_vars, state_ens, distribution_type, &
62 p, probit_ens, use_input_p, bounded_below, bounded_above, lower_bound, upper_bound)
63
64integer, intent(in) :: ens_size
65integer, intent(in) :: num_vars
66real(r8), intent(in) :: state_ens(:, :)
67integer, intent(in) :: distribution_type(num_vars)
68type(distribution_params_type), intent(inout) :: p(num_vars)
69real(r8), intent(out) :: probit_ens(:, :)
70logical, intent(in) :: use_input_p
71logical, intent(in) :: bounded_below, bounded_above
72real(r8), intent(in) :: lower_bound, upper_bound
73
74
75! NOTE THAT WILL MAKE HELEN CRAZY: THIS WORKS WITH THE INPUT CALLING ARGUMENTS FOR STATE_ENS AND
76! PROBIT_ENS BEING THE SAME. A TEMP IS USED TO AVOID OVERWRITING ISSUES. IS THIS YUCKY?
77
78! Note that the input and output arrays may have extra copies (first subscript). Passing sections of a
79! leading index could be inefficient for time and storage, so avoiding that for now.
80
81! Assumes that the bounds are the same for any variables that are BNRH for now
82! The bounds variables are not used for the normal case or the case where the input p is used
83
84integer :: i
85real(r8) :: temp_ens(ens_size)
86
87do i = 1, num_vars
88 call transform_to_probit(ens_size, state_ens(1:ens_size, i), distribution_type(i), &
89 p(i), temp_ens, use_input_p, bounded_below, bounded_above, lower_bound, upper_bound)
90 probit_ens(1:ens_size, i) = temp_ens
91end do
92
93end subroutine transform_all_to_probit

97subroutine transform_to_probit(ens_size, state_ens_in, distribution_type, p, &
98 probit_ens, use_input_p, bounded_below, bounded_above, lower_bound, upper_bound)
99
100integer, intent(in) :: ens_size
101real(r8), intent(in) :: state_ens_in(ens_size)
102integer, intent(in) :: distribution_type
103type(distribution_params_type), intent(inout) :: p
104real(r8), intent(out) :: probit_ens(ens_size)
105logical, intent(in) :: use_input_p
106logical, intent(in) :: bounded_below, bounded_above
107real(r8), intent(in) :: lower_bound, upper_bound
108
109real(r8) :: state_ens(ens_size)
110real(r8) :: probit_ens_temp(ens_size), state_ens_temp(ens_size), diff(ens_size)
111type(distribution_params_type) :: p_temp
112integer :: i
113
114! If not initialized, read in the namelist
115if(.not. module_initialized) call initialize_probit_transform
116
117! Fix bounds violations if requested
118if(fix_bound_violations) then
119 do i = 1, ens_size
120 state_ens(i) = fix_bounds(state_ens_in(i), bounded_below, bounded_above, &
121 lower_bound, upper_bound)
122 end do
123else
124 state_ens = state_ens_in
125endif
126
127! Set the type of the distribution in the parameters defined type
128p%distribution_type = distribution_type
129
130if(p%distribution_type == normal_distribution) then
131 ! No transformation is done for a normal
132 probit_ens = state_ens
133elseif(p%distribution_type == log_normal_distribution) then
134 call to_probit_log_normal(ens_size, state_ens, probit_ens)
135elseif(p%distribution_type == uniform_distribution) then
136 call to_probit_uniform(ens_size, state_ens, p, probit_ens, use_input_p, lower_bound, upper_bound)
137elseif(p%distribution_type == gamma_distribution) then
138 call to_probit_gamma(ens_size, state_ens, p, probit_ens, use_input_p, &
139 bounded_below, bounded_above, lower_bound, upper_bound)
140elseif(p%distribution_type == beta_distribution) then
141 call to_probit_beta(ens_size, state_ens, p, probit_ens, use_input_p, &
142 lower_bound, upper_bound)
143elseif(p%distribution_type == bounded_normal_rh_distribution) then
144 call to_probit_bounded_normal_rh(ens_size, state_ens, p, probit_ens, &
145 use_input_p, bounded_below, bounded_above, lower_bound, upper_bound)
146
147!----------------------------------------------------------------------------------
148! The following code block tests that the to/from probit calls are nearly inverse
149! for all of the calls made during an assimilation
150 if(do_inverse_check) then
151 if(.not. use_input_p) then
152 call to_probit_bounded_normal_rh(ens_size, state_ens, p_temp, probit_ens_temp, &
153 use_input_p, bounded_below, bounded_above, lower_bound, upper_bound)
154 call from_probit_bounded_normal_rh(ens_size, probit_ens_temp, p_temp, state_ens_temp)
155 diff = state_ens - state_ens_temp
156 if(abs(maxval(diff)) > 1.0e-8_r8) then
157 write(*, *) 'Maximum allowed value of probit to/from difference exceeded'
158 write(*, *) 'Location of minimum ensemble member ', minloc(state_ens)
159 write(*, *) 'Location of maximum ensemble member ', maxloc(state_ens)
160 do i = 1, ens_size
161 write(*, *) i, state_ens(i), state_ens_temp(i), diff(i)
162 enddo
163 stop
164 endif
165 endif
166
167 if(use_input_p) then
168 call to_probit_bounded_normal_rh(ens_size, state_ens, p, probit_ens_temp, &
169 use_input_p, bounded_below, bounded_above, lower_bound, upper_bound)
170 call from_probit_bounded_normal_rh(ens_size, probit_ens_temp, p, state_ens_temp)
171 diff = state_ens - state_ens_temp
172 if(abs(maxval(diff)) > 1.0e-8_r8) then
173 write(*, *) 'Maximum allowed value of probit to/from difference for input p exceeded'
174 write(*, *) 'Location of minimum ensemble member ', minloc(state_ens)
175 write(*, *) 'Location of maximum ensemble member ', maxloc(state_ens)
176 do i = 1, ens_size
177 write(*, *) i, state_ens(i), state_ens_temp(i), diff(i)
178 enddo
179 stop
180 endif
181
182 endif
183 endif
184!----------------------------------------------------------------------------------
185
186
187!!!elseif(p%distribution_type == PARTICLE_FILTER_DISTRIBUTION) then
188 !!!call to_probit_particle(ens_size, state_ens, p, probit_ens, use_input_p, &
189 !!!bounded_below, bounded_above, lower_bound, upper_bound)
190else
191 write(errstring, *) 'Illegal distribution type', p%distribution_type
192 call error_handler(e_err, 'transform_to_probit', errstring, source)
193endif
194
195end subroutine transform_to_probit

214subroutine to_probit_uniform(ens_size, state_ens, p, probit_ens, use_input_p, &
215 lower_bound_in, upper_bound_in)
216
217integer, intent(in) :: ens_size
218real(r8), intent(in) :: state_ens(ens_size)
219type(distribution_params_type), intent(inout) :: p
220real(r8), intent(out) :: probit_ens(ens_size)
221logical, intent(in) :: use_input_p
222real(r8), intent(in) :: lower_bound_in, upper_bound_in
223
224real(r8) :: lower_bound, upper_bound, d_range, quantile
225integer :: i
226
227! There is no distribution_mod for uniform at the moment so params setup is done here
228if(use_input_p) then
229 lower_bound = p%lower_bound
230 upper_bound = p%upper_bound
231else
232 lower_bound = lower_bound_in
233 upper_bound = upper_bound_in
234 ! Save the bounds in the distribution_params_type
235 p%lower_bound = lower_bound
236 p%upper_bound = upper_bound
237endif
238
239d_range = upper_bound - lower_bound
240do i = 1, ens_size
241 ! Transform to quantile; U(lower_bound, upper_bound) to U(0, 1)
242 quantile = (state_ens(i) - lower_bound) / d_range
243 ! Transform to probit/logit space
244 probit_ens(i) = probit_or_logit_transform(quantile)
245end do
246
247end subroutine to_probit_uniform

251subroutine to_probit_gamma(ens_size, state_ens, p, probit_ens, use_input_p, &
252 bounded_below, bounded_above, lower_bound, upper_bound)
253
254integer, intent(in) :: ens_size
255real(r8), intent(in) :: state_ens(ens_size)
256type(distribution_params_type), intent(inout) :: p
257real(r8), intent(out) :: probit_ens(ens_size)
258logical, intent(in) :: use_input_p
259logical, intent(in) :: bounded_below, bounded_above
260real(r8), intent(in) :: lower_bound, upper_bound
261
262! Probit transform for gamma.
263real(r8) :: quantile
264integer :: i
265
266! Bounds other than a lower bound at 0 not yet implemented for gamma distribution
267
268! Get the parameters for this distribution if not already available
269if(.not. use_input_p) then
270
271 ! In full generality, gamma must be bounded either below or above
272 if(.not. (bounded_below .neqv. bounded_above)) then
273 errstring = 'Gamma distribution requires either bounded above or below to be true'
274 call error_handler(e_err, 'to_probit_gamma', errstring, source)
275 endif
276
277 call set_gamma_params_from_ens(state_ens, ens_size, bounded_below, bounded_above, &
278 lower_bound, upper_bound, p)
279endif
280
281do i = 1, ens_size
282 ! First, get the quantile for this ensemble member
283 quantile = gamma_cdf_params(state_ens(i), p)
284 ! Transform to probit space
285 probit_ens(i) = probit_or_logit_transform(quantile)
286end do
287
288end subroutine to_probit_gamma

322subroutine to_probit_bounded_normal_rh(ens_size, state_ens, p, probit_ens, &
323 use_input_p, bounded_below, bounded_above, lower_bound, upper_bound)
324
325! Note that this is just for transforming back and forth, not for doing the RHF observation update
326! This means that we know a prior that the quantiles associated with the initial ensemble are
327! uniformly spaced which can be used to simplify transforming.
328
329! How to handle identical ensemble members is an open question for now. This is also a problem
330! for ensemble members that are identical to one of the bounds.
331
332integer, intent(in) :: ens_size
333real(r8), intent(in) :: state_ens(ens_size)
334type(distribution_params_type), intent(inout) :: p
335real(r8), intent(out) :: probit_ens(ens_size)
336logical, intent(in) :: use_input_p
337logical, intent(in) :: bounded_below, bounded_above
338real(r8), intent(in) :: lower_bound, upper_bound
339
340! Probit transform for bounded normal rh.
341integer :: i
342real(r8) :: quantile(ens_size)
343
344if(use_input_p) then
345 ! Do not know what to do if sd of original ensemble is 0 (or small, work on this later)
346 if(get_bnrh_sd(p) <= 0.0_r8) then
347 ! Just return the original ensemble
348 probit_ens = state_ens
349 return
350 endif
351
352 ! Get the quantiles for each of the ensemble members in a BNRH distribution
353 call bnrh_cdf_initialized_vector(state_ens, ens_size, p, quantile)
354
355else
356 ! Get all the info about the rank histogram cdf
357 call bnrh_cdf_params(state_ens, ens_size, bounded_below, bounded_above, &
358 lower_bound, upper_bound, p, quantile)
359
360 ! Do not know what to do if sd is 0 (or small, work on this later)
361 if(get_bnrh_sd(p) <= 0.0_r8) then
362 ! Just return the original ensemble
363 probit_ens = state_ens
364 return
365 endif
366
367endif
368
369! Transform the quantiles to probit space
370do i = 1, ens_size
371 probit_ens(i) = probit_or_logit_transform(quantile(i))
372end do
373
374end subroutine to_probit_bounded_normal_rh

470subroutine transform_from_probit(ens_size, probit_ens, p, state_ens)
471
472integer, intent(in) :: ens_size
473real(r8), intent(in) :: probit_ens(ens_size)
474type(distribution_params_type), intent(inout) :: p
475real(r8), intent(out) :: state_ens(ens_size)
476
477! If not initialized, read in the namelist
478if(.not. module_initialized) call initialize_probit_transform
479
480! Transform back to the original space
481if(p%distribution_type == normal_distribution) then
482 ! No need to do any transformation for a normal
483 state_ens = probit_ens
484elseif(p%distribution_type == log_normal_distribution) then
485 call from_probit_log_normal(ens_size, probit_ens, state_ens)
486elseif(p%distribution_type == uniform_distribution) then
487 call from_probit_uniform(ens_size, probit_ens, p, state_ens)
488elseif(p%distribution_type == gamma_distribution) then
489 call from_probit_gamma(ens_size, probit_ens, p, state_ens)
490elseif(p%distribution_type == beta_distribution) then
491 call from_probit_beta(ens_size, probit_ens, p, state_ens)
492elseif(p%distribution_type == bounded_normal_rh_distribution) then
493 call from_probit_bounded_normal_rh(ens_size, probit_ens, p, state_ens)
494!!!elseif(p%distribution_type == PARTICLE_FILTER_DISTRIBUTION) then
495 !!!call from_probit_particle(ens_size, probit_ens, p, state_ens)
496else
497 write(errstring, *) 'Illegal distribution type', p%distribution_type
498 call error_handler(e_err, 'transform_from_probit', errstring, source)
499 stop
500endif
501
502! Deallocate any allocatable storage that was used for this distribution
503call deallocate_distribution_params(p)
504
505end subroutine transform_from_probit

522subroutine from_probit_uniform(ens_size, probit_ens, p, state_ens)
523
524integer, intent(in) :: ens_size
525real(r8), intent(in) :: probit_ens(ens_size)
526type(distribution_params_type), intent(inout) :: p
527real(r8), intent(out) :: state_ens(ens_size)
528
529real(r8) :: quantile
530integer :: i
531
532do i = 1, ens_size
533 ! First, invert the probit to get a quantile
534 quantile = inv_probit_or_logit_transform(probit_ens(i))
535 ! Transform from U(0, 1) to U(lower_bound, upper_bound)
536 state_ens(i) = p%lower_bound + quantile * (p%upper_bound - p%lower_bound)
537end do
538
539end subroutine from_probit_uniform
