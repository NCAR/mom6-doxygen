6! Forward operator to compute total precipitable water in a column,
7! in centimeters, over the ocean. Can be used as an example of a
8! forward operator that loops over either fixed pressure levels or
9! over model levels.
10
11! This code is only correct for TPW over the ocean where the surface
12! is at 0m elevation. For TWP over land you must be able to compute
13! the surface elevation for any given lat/lon location. This code
14! assumes the model can return the surface pressure for any given
15! lat/lon location, and the specific humidity for a given location
16! where the vertical is either pressure or model levels.
17
18! keep in mind that fortran allows only 31 characters in parameter
19! definitions (which is what this string is going to be used for).
20! if the platform name gets longer than 5 chars, consider going
21! to something like xxx_TOTAL_PRECIP_WATER to give you room to
22! put in more descriptive platform names.

145subroutine get_expected_tpw(state_handle, ens_size, location, tpw, istatus)
146
147!------------------------------------------------------------------------------
148! Purpose: To calculate total precipitable water in a column over oceans.
149! inputs:
150! state_vector: DART state vector
151! location: Observation location
152!
153! output parameters:
154! tpw: total amount of liquid water (in cm) if all atmospheric water
155! vapor in the column was condensed.
156! istatus: 0 if ok, a positive value for error
157!------------------------------------------------------------------------------
158! Author: Hui Liu , Version 1.1: May 25, 2011 for WRF
159! updated by n. collins 14 june 2012
160!------------------------------------------------------------------------------
161
162type(ensemble_type), intent(in) :: state_handle
163integer, intent(in) :: ens_size
164type(location_type), intent(in) :: location
165real(r8), intent(out) :: tpw(ens_size)
166integer, intent(out) :: istatus(ens_size)
167
168! local variables
169real(r8) :: lon, lat, height, obsloc(3)
170type(location_type) :: location2
171
172! we'll compute the midpoint value for each pressure range, so allocate one
173! more than the number of expected values.
174real(r8) :: pressure(ens_size, max_pressure_intervals+1), qv(ens_size, max_pressure_intervals+1)
175real(r8) :: pressure_interval(ens_size), psfc(ens_size)
176integer :: which_vert, k, lastk, first_non_surface_level
177integer :: this_istatus(ens_size)
178logical :: return_now
179
180if ( .not. module_initialized ) call initialize_module
181
182tpw = missing_r8
183istatus = 0
184
185! check for bad values in the namelist which exceed the size of the
186! arrays that are hardcoded here.
187
188if (num_pressure_intervals > max_pressure_intervals) then
189 call error_handler(e_err, 'get_expected_tpw', &
190 'num_pressure_intervals greater than max allowed', &
191 source, revision, revdate, &
192 text2='increase max_pressure_intervals in obs_def_tpw_mod.f90', &
193 text3='and recompile.');
194endif
195
196! location is the lat/lon where we need to compute the column quantity
197obsloc = get_location(location)
198lon = obsloc(1) ! degree: 0 to 360
199lat = obsloc(2) ! degree: -90 to 90
200
201! get the pressure at the surface first.
202
203! This assumes the column is over an ocean where the surface
204! is at 0m elevation. If you are going to use this over land, the
205! height below must be the elevation (in m) of the surface at this location.
206which_vert = vertissurface
207height = 0.0
208location2 = set_location(lon, lat, height, which_vert)
209
210! interpolate the surface pressure and specific humidity at the desired location
211! assumes the values returned from the interpolation will be in these units:
212! surface pressure : Pa
213! moisture : kg/kg
214call interpolate(state_handle, ens_size, location2, qty_surface_pressure, pressure(:, 1), this_istatus)
215call track_status(ens_size, this_istatus, tpw, istatus, return_now)
216if (return_now) return
217
218call interpolate(state_handle, ens_size, location2, qty_specific_humidity, qv(:, 1), this_istatus)
219call track_status(ens_size, this_istatus, tpw, istatus, return_now)
220if (return_now) return
221
222! save this for use below
223psfc = pressure(:, 1)
224
225! there are two options for constructing the column of values. if 'model_levels'
226! is true, we query the model by vertical level number. the 'separate_surface_level'
227! flag should be set to indicate if the lowest level of the 3d grid is the
228! surface or if the surface values are a separate quantity below the 3d grid.
229
230if (model_levels) then
231
232 ! some models have a 3d grid of values and the lowest level contains
233 ! the surface quantities. others have a separate field for the
234 ! surface values and the 3d grid starts at some given elevation.
235 ! if the namelist value 'separate_surface_level' is true, we will
236 ! ask to interpolate a surface pressure first and then work up the
237 ! 3d column starting at level 1. if it is false, we assume level 1
238 ! was the surface pressure and we start here at level 2.
239
240 if (separate_surface_level) then
241 first_non_surface_level = 1
242 else
243 first_non_surface_level = 2
244 endif
245
246 ! construct a pressure column on model levels
247
248 ! call the model until the interpolation call fails (above the top level)
249 ! (this is not a fatal error unless the first call fails).
250 ! also exit the loop if the pressure is above the namelist-specified pressure top
251
252 lastk = 2
253 levels: do k=first_non_surface_level, 10000 ! something unreasonably large
254
255 ! call the model_mod to get the pressure and specific humidity at each level
256 ! from the model and fill out the pressure and qv arrays. the model must
257 ! support a vertical type of level number.
258
259 which_vert = vertislevel
260 location2 = set_location(lon, lat, real(k, r8), which_vert)
261!>@todo --- This may be different for each ensemble memeber ---
262 call interpolate(state_handle, ens_size, location2, qty_pressure, pressure(:, lastk), this_istatus)
263 call track_status(ens_size, this_istatus, tpw, istatus, return_now)
264 if (any(pressure(:, lastk) < pressure_top)) exit levels
265 if (return_now) return
266
267 call interpolate(state_handle, ens_size, location2, qty_specific_humidity, qv(:, lastk), this_istatus)
268 call track_status(ens_size, this_istatus, tpw, istatus, return_now)
269!--------------------------------------------------------------
270 if (return_now) return
271
272 lastk = lastk + 1
273 enddo levels
274
275 lastk = lastk - 1
276
277 ! if we got no valid values, set istatus and return here.
278 ! 'tpw' return value is already set to missing_r8
279 if (lastk == 1) then
280 istatus = 3
281 return
282 endif
283
284else
285
286 ! construct an explicit pressure column and get qv at each pressure level.
287 ! each column will be at a different set of heights because it
288 ! divides the surface pressure and 'pressure_top' evenly into
289 ! 'num_pressure_intervals'. so each column will have the same
290 ! number of samples, but each may be at different pressure values
291 ! depending on the surface pressure.
292 pressure_interval = (psfc - pressure_top)/num_pressure_intervals
293 lastk = num_pressure_intervals + 1
294
295 ! construct a pressure column at fixed pressure intervals
296 ! pressure(:, 1) is always the surface pressure.
297
298 do k=2, lastk
299 where (istatus == 0 ) pressure(:, k) = pressure(:, 1) - pressure_interval * (k-1)
300 end do
301
302 ! call the model_mod to get the specific humidity at each location from the model
303 ! and fill out the qv array.
304 do k=2, lastk
305
306 which_vert = vertispressure
307 !>@todo - there should be only a single location here. for now, use 1
308 !> but what to do in the general case? set a fixed top and bottom pressure??
309 location2 = set_location(lon, lat, pressure(1, k), which_vert)
310
311 call interpolate(state_handle, ens_size, location2, qty_specific_humidity, qv(:, k), this_istatus)
312 call track_status(ens_size, this_istatus, tpw, istatus, return_now)
313 if (return_now) return
314
315 enddo
316endif
317
318! whichever way the column was made (pressure levels or model levels),
319! sum the values in the column, computing the area under the curve.
320! pressure is in pascals (not hPa or mb), and moisture is in kg/kg.
321tpw = 0.0
322do k=1, lastk - 1
323 where (istatus == 0) &
324 tpw = tpw + 0.5 * (qv(:, k) + qv(:, k+1) ) * (pressure(:, k) - pressure(:, k+1) )
325enddo
326
327! convert to centimeters of water and return
328where (istatus == 0)
329 tpw = 100.0 * tpw /(density*gravity) ! -> cm
330elsewhere
331 tpw = missing_r8
332endwhere
333
334end subroutine get_expected_tpw

337end module obs_def_tpw_mod
