122!-----------------------------------------------------------------------
123! start of executable code
124!-----------------------------------------------------------------------
125
126call initialize_utilities('level4_to_obs')
127
128! Print module information to log file and stdout.
129call register_module(source, revision, revdate)
130
131! Read the namelist entry
132call find_namelist_in_file("input.nml", "level4_to_obs_nml", iunit)
133read(iunit, nml = level4_to_obs_nml, iostat = rcio)
134call check_namelist_read(iunit, rcio, "level4_to_obs_nml")
135
136! Record the namelist values used for the run ...
137if (do_nml_file()) write(nmlfileunit, nml=level4_to_obs_nml)
138if (do_nml_term()) write( * , nml=level4_to_obs_nml)
139
140! time setup
141call set_calendar_type(gregorian)
142offset = set_time(nint(abs(timezoneoffset)*3600.0_r8),0)
143prev_time = set_time(0, 0)
144
145write(string1, *) 'tower located at lat, lon, elev =', latitude, longitude, elevation
146write(string2, *) 'flux observations taken at =', flux_height,'m'
147if (verbose) call error_handler(e_msg,'level4_to_obs',string1,text2=string2)
148
149! check the lat/lon values to see if they are ok
150if (longitude < 0.0_r8) longitude = longitude + 360.0_r8
151
152if (( latitude > 90.0_r8 .or. latitude < -90.0_r8 ) .or. &
153 (longitude < 0.0_r8 .or. longitude > 360.0_r8 )) then
154
155 write (string2,*)'latitude should be [-90, 90] but is ',latitude
156 write (string3,*)'longitude should be [ 0,360] but is ',longitude
157
158 string1 ='tower location error in input.nml&level4_to_obs_nml'
159 call error_handler(e_err,'level4_to_obs', string1, source, revision, &
160 revdate, text2=string2,text3=string3)
161
162endif
163
164! We need to know the maximum number of observations in the input file.
165! Each line has info for the 3 observations we want.
166! The max possible number of obs needs to be specified but it
167! will only write out the actual number created.
168! Each observation in this series will have a single
169! observation value and a quality control flag.
170! Initialize two empty observations - one to track location
171! in observation sequence - the other is for the new observation.
172
173iunit = open_file(text_input_file, 'formatted', 'read')
174if (verbose) call error_handler(e_msg,'level4_to_obs','opened input file '//trim(text_input_file))
175
176nlines = count_file_lines(iunit)
177max_obs = 3*nlines
178num_copies = 1
179num_qc = 1
180first_obs = .true.
181
182call static_init_obs_sequence()
183call init_obs(obs, num_copies, num_qc)
184call init_obs(prev_obs, num_copies, num_qc)
185call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
186
187! the first one needs to contain the string 'observation' and the
188! second needs the string 'QC'.
189call set_copy_meta_data(obs_seq, 1, 'observation')
190call set_qc_meta_data( obs_seq, 1, 'Ameriflux QC')
191
192! The first line describes all the fields ... column headers, if you will
193
194rewind(iunit)
195call decode_header(iunit, nwords)
196
197obsloop: do iline = 2,nlines
198
199 ! read in entire text line into a buffer
200 read(iunit,'(A)',iostat=rcio) bigline
201 if (rcio < 0) exit obsloop
202 if (rcio > 0) then
203 write (string1,'(''Cannot read (error '',i3,'') line '',i8,'' in '',A)') &
204 rcio, iline, trim(text_input_file)
205 call error_handler(e_err,'main', string1, source, revision, revdate)
206 endif
207
208 input_line = adjustl(bigline)
209
210 ! parse the line into the tower structure (including the observation time)
211 call stringparse(input_line, nwords, iline)
212
213 if (iline <= 2) then
214 write(*,*)''
215 call print_date(tower%time_obs, ' first observation date (local time) is')
216 call print_time(tower%time_obs, ' first observation time (local time) is')
217 write(*,*)'first observation raw values: (column,string,value) timezone not applied'
218 write(*,*)tower%monthindex, tower%monthstring , tower%month
219 write(*,*)tower%dayindex , tower%daystring , tower%day
220 write(*,*)tower%hourindex , tower%hourstring , tower%hour
221 write(*,*)tower%doyindex , tower%doystring , tower%doy
222 write(*,*)tower%hindex , tower%hstring , tower%h
223 write(*,*)tower%hQCindex , tower%hQCstring , tower%hQC
224 write(*,*)tower%leindex , tower%lestring , tower%le
225 write(*,*)tower%leQCindex , tower%leQCstring , tower%leQC
226 write(*,*)tower%neeindex , tower%neestring , tower%nee
227 write(*,*)tower%neeQCindex, tower%neeQCstring , tower%neeQC
228 write(*,*)''
229
230 write(logfileunit,*)''
231 call print_date(tower%time_obs, ' first observation date (local time) is',logfileunit)
232 call print_time(tower%time_obs, ' first observation time (local time) is',logfileunit)
233 write(logfileunit,*)'first observation raw values: (column,string,value) timezone not applied'
234 write(logfileunit,*)tower%monthindex, tower%monthstring , tower%month
235 write(logfileunit,*)tower%dayindex , tower%daystring , tower%day
236 write(logfileunit,*)tower%hourindex , tower%hourstring , tower%hour
237 write(logfileunit,*)tower%doyindex , tower%doystring , tower%doy
238 write(logfileunit,*)tower%hindex , tower%hstring , tower%h
239 write(logfileunit,*)tower%hQCindex , tower%hQCstring , tower%hQC
240 write(logfileunit,*)tower%leindex , tower%lestring , tower%le
241 write(logfileunit,*)tower%leQCindex , tower%leQCstring , tower%leQC
242 write(logfileunit,*)tower%neeindex , tower%neestring , tower%nee
243 write(logfileunit,*)tower%neeQCindex, tower%neeQCstring , tower%neeQC
244 write(logfileunit,*)''
245 end if
246
247 call get_time(tower%time_obs, osec, oday)
248
249 if (verbose) then
250 write(string1,*)'obs time is (seconds,days) ',osec, oday,' obs date is '
251 call print_date(tower%time_obs, trim(string1))
252 call print_date(tower%time_obs, trim(string1),logfileunit)
253 endif
254
255 ! make an obs derived type, and then add it to the sequence
256 ! If the QC value is good, use the observation.
257 ! Increasingly larger QC values are more questionable quality data.
258 ! The observation errors are from page 183, Table 7.1(A) in
259 ! Chapter 7 of a book by A.D. Richardson et al. via Andy Fox.
260
261 if (tower%hQC <= maxgoodqc) then ! Sensible Heat Flux [W m-2]
262 oerr = 10.0_r8 + abs(tower%h)*0.22_r8
263 qc = real(tower%hQC,r8)
264 call create_3d_obs(latitude, longitude, flux_height, vertisheight, tower%h, &
265 tower_sensible_heat_flux, oerr, oday, osec, qc, obs)
266 call add_obs_to_seq(obs_seq, obs, tower%time_obs, prev_obs, prev_time, first_obs)
267 endif
268
269 if (tower%leQC <= maxgoodqc) then ! Latent Heat Flux [W m-2]
270 oerr = 10.0_r8 + abs(tower%le)*0.32_r8
271 qc = real(tower%leQC,r8)
272 call create_3d_obs(latitude, longitude, flux_height, vertisheight, tower%le, &
273 tower_latent_heat_flux, oerr, oday, osec, qc, obs)
274 call add_obs_to_seq(obs_seq, obs, tower%time_obs, prev_obs, prev_time, first_obs)
275 endif
276
277 if (tower%neeQC <= maxgoodqc) then ! Net Ecosystem Exchange [umol m-2 s-1]
278 if (tower%nee <= 0) then
279 oerr = (2.0_r8 + abs(tower%nee)*0.1_r8) * umol_to_gc
280 else
281 oerr = (2.0_r8 + abs(tower%nee)*0.4_r8) * umol_to_gc
282 endif
283 tower%nee = -tower%nee * umol_to_gc ! to match convention in CLM [gC m-2 s-1]
284 qc = real(tower%neeQC,r8)
285 call create_3d_obs(latitude, longitude, flux_height, vertisheight, tower%nee, &
286 tower_netc_eco_exchange, oerr, oday, osec, qc, obs)
287 call add_obs_to_seq(obs_seq, obs, tower%time_obs, prev_obs, prev_time, first_obs)
288 endif
289
290end do obsloop
291
292! if we added any obs to the sequence, write it out to a file now.
293if ( get_num_obs(obs_seq) > 0 ) then
294 write(string1,*)'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
295 if (verbose) call error_handler(e_msg,'level4_to_obs',string1)
296 call write_obs_seq(obs_seq, obs_out_file)
297endif
298
299! end of main program
300call finalize_utilities()
301
302contains

306!
307! create_3d_obs - subroutine that is used to create an observation
308! type from observation data.
309!
310! NOTE: assumes the code is using the threed_sphere locations module,
311! that the observation has a single data value and a single
312! qc value, and that this obs type has no additional required
313! data (e.g. gps and radar obs need additional data per obs)
314!
315! lat - latitude of observation
316! lon - longitude of observation
317! vval - vertical coordinate
318! vkind - kind of vertical coordinate (pressure, level, etc)
319! obsv - observation value
320! okind - observation kind
321! oerr - observation error (in units of standard deviation)
322! day - gregorian day
323! sec - gregorian second
324! qc - quality control value
325! obs - observation type
326!
327! created Oct. 2007 Ryan Torn, NCAR/MMM
328! adapted for more generic use 11 Mar 2010, nancy collins, ncar/image
329!
330!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
331subroutine create_3d_obs(lat, lon, vval, vkind, obsv, okind, oerr, day, sec, qc, obs)
332use obs_def_mod, only : obs_def_type, set_obs_def_time, set_obs_def_type_of_obs, &
333 set_obs_def_error_variance, set_obs_def_location
334use obs_sequence_mod, only : obs_type, set_obs_values, set_qc, set_obs_def
335use time_manager_mod, only : time_type, set_time
336use location_mod, only : set_location
337
338 integer, intent(in) :: okind, vkind, day, sec
339 real(r8), intent(in) :: lat, lon, vval, obsv, oerr, qc
340 type(obs_type), intent(inout) :: obs
341
342real(r8) :: obs_val(1), qc_val(1)
343type(obs_def_type) :: obs_def
344
345call set_obs_def_location(obs_def, set_location(lon, lat, vval, vkind))
346call set_obs_def_type_of_obs(obs_def, okind)
347call set_obs_def_time(obs_def, set_time(sec, day))
348call set_obs_def_error_variance(obs_def, oerr * oerr)
349call set_obs_def(obs, obs_def)
350
351obs_val(1) = obsv
352call set_obs_values(obs, obs_val)
353qc_val(1) = qc
354call set_qc(obs, qc_val)
355
356end subroutine create_3d_obs

359!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
360!
361! add_obs_to_seq -- adds an observation to a sequence. inserts if first
362! obs, inserts with a prev obs to save searching if that's possible.
363!
364! seq - observation sequence to add obs to
365! obs - observation, already filled in, ready to add
366! obs_time - time of this observation, in dart time_type format
367! prev_obs - the previous observation that was added to this sequence
368! (will be updated by this routine)
369! prev_time - the time of the previously added observation (will also
370! be updated by this routine)
371! first_obs - should be initialized to be .true., and then will be
372! updated by this routine to be .false. after the first obs
373! has been added to this sequence.
374!
375! created Mar 8, 2010 nancy collins, ncar/image
376!
377!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
378subroutine add_obs_to_seq(seq, obs, obs_time, prev_obs, prev_time, first_obs)
379
380use obs_sequence_mod, only : obs_sequence_type, obs_type, insert_obs_in_seq
381use time_manager_mod, only : time_type, operator(>=)
382
383type(obs_sequence_type), intent(inout) :: seq
384type(obs_type), intent(inout) :: obs, prev_obs
385type(time_type), intent(in) :: obs_time
386type(time_type), intent(inout) :: prev_time
387logical, intent(inout) :: first_obs
388
389! insert(seq,obs) always works (i.e. it inserts the obs in
390! proper time format) but it can be slow with a long file.
391! supplying a previous observation that is older (or the same
392! time) as the new one speeds up the searching a lot.
393
394if(first_obs) then ! for the first observation, no prev_obs
395 call insert_obs_in_seq(seq, obs)
396 first_obs = .false.
397else
398 if(obs_time >= prev_time) then ! same time or later than previous obs
399 call insert_obs_in_seq(seq, obs, prev_obs)
400 else ! earlier, search from start of seq
401 call insert_obs_in_seq(seq, obs)
402 endif
403endif
404
405! update for next time
406prev_obs = obs
407prev_time = obs_time
408
409end subroutine add_obs_to_seq

414! count_file_lines --
415! count the lines in a text file.
416! rewinds the unit after counting.
417!
418! iunit - handle to the already-open text file
419!
420! created May 2, 2012 Tim Hoar, NCAR/IMAGe
421!
422!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
423function count_file_lines(iunit)
424
425integer, intent(in) :: iunit
426integer :: count_file_lines
427
428integer :: i
429character(len=128) :: oneline
430
431integer, parameter :: tenmillion = 10000000
432rewind(iunit)
433
434count_file_lines = 0
435countloop : do i = 1,tenmillion
436
437 read(iunit,'(A)',iostat=rcio) oneline
438
439 if (rcio < 0) exit countloop ! end of file
440 if (rcio > 0) then

444 endif
445 count_file_lines = count_file_lines + 1
446
447enddo countloop
448rewind(iunit)
449
450if (count_file_lines >= tenmillion) then
451 write (string1,'('' suspiciously large number of lines '',i8)')count_file_lines
452 call error_handler(e_msg,'count_file_lines', string1, &
453 source, revision, revdate)
454endif
455
456end function count_file_lines

461subroutine decode_header(iunit,ncolumns)
462! Reads the first line of the header and parses the information.
463! And by parse, I mean determine which columns are the columns
464! of interest.
465
466integer, intent(in) :: iunit
467integer, intent(out) :: ncolumns
468
469integer, parameter :: maxwordlength = 30
470integer :: i,charcount,columncount,wordlength
471character(len=maxwordlength), dimension(:), allocatable :: columns
472integer, dimension(10) :: qc = 0
473
474! Read the line and strip off any leading whitespace.
475
476read(iunit,'(A)',iostat=rcio) bigline
477if (rcio /= 0) then
478 write(string1,*)'Cannot parse header. Begins <',trim(bigline(1:40)),'>'
479 call error_handler(e_err,'decode_header',string1, source, revision, revdate)
480endif
481
482input_line = adjustl(bigline)
483
484! Count how many commas are in the line - use this to determine how many columns
485
486charcount = countchar(input_line,',')
487ncolumns = charcount + 1
488allocate(columns(ncolumns))
489
490columncount = 0 ! track the number of columns
491wordlength = 0 ! number of characters in the column descriptor
492charcount = 0 ! the position of the (last) comma
493do i = 1,len_trim(input_line)
494 if (input_line(i:i) == ',') then
495 columncount = columncount + 1
496 if (wordlength > maxwordlength) then
497 write(string1,*)'unexpected long word ... starts <',&
498 input_line((i-wordlength):(i-1)),'>'
499 call error_handler(e_err,'decode_header',string1, source, revision, revdate)
500 endif
501 columns(columncount) = input_line((i-wordlength):(i-1))
502 write(string1,*) 'word(',columncount,') is ',columns(columncount)
503 if (verbose) call error_handler(e_msg,'decode_header',string1)
504 wordlength = 0
505 charcount = i
506 else
507 wordlength = wordlength + 1
508 endif
509enddo
510
511! There is one more column after the last comma
512
513if ((columncount+1) /= ncolumns) then
514 write(string1,*)'parsed wrong number of words ...'
515 write(string2,*)'expected ',ncolumns,' got ',columncount+1
516 call error_handler(e_err,'decode_header',string1,source,revision,revdate, &
517 text2=trim(string2), text3=trim(input_line))
518endif
519
520columns(ncolumns) = input_line((charcount+1):len_trim(input_line))
521
522write(string1,*)'word(',ncolumns,') is ',columns(ncolumns)
523if (verbose) call error_handler(e_msg,'decode_header',string1)
524
525! Finally get to the task at hand
526
527tower%monthindex = match(columns, tower%monthstring)
528tower%dayindex = match(columns, tower%daystring)
529tower%hourindex = match(columns, tower%hourstring)
530tower%doyindex = match(columns, tower%doystring)
531tower%hindex = match(columns, tower%hstring)
532tower%hQCindex = match(columns, tower%hQCstring)
533tower%leindex = match(columns, tower%lestring)
534tower%leQCindex = match(columns, tower%leQCstring)
535tower%neeindex = match(columns, tower%neestring)
536tower%neeQCindex = match(columns, tower%neeQCstring)
537
538! FIXME ... find a column marked 'year' or 'error_var' and use if possible.
539
540! Check to make sure we got all the indices we need
541
542qc( 1) = checkindex( tower%monthindex, tower%monthstring)
543qc( 2) = checkindex( tower%dayindex , tower%daystring)
544qc( 3) = checkindex( tower%hourindex , tower%hourstring)
545qc( 4) = checkindex( tower%doyindex , tower%doystring)
546qc( 5) = checkindex( tower%hindex , tower%hstring)
547qc( 6) = checkindex( tower%hQCindex , tower%hQCstring)
548qc( 7) = checkindex( tower%leindex , tower%lestring)
549qc( 8) = checkindex( tower%leQCindex , tower%leQCstring)
550qc( 9) = checkindex( tower%neeindex , tower%neestring)
551qc(10) = checkindex( tower%neeQCindex, tower%neeQCstring)
552
553if (any(qc /= 0) ) then
554 write(string1,*)'Did not find all the required column indices.'
555 call error_handler(e_err,'decode_header',string1, source, revision, revdate)
556endif
557
558! Summarize if desired
559
560if (verbose) then
561110 format('index for ',a20,' is ',i3)
562 write(*,110)tower%monthstring, tower%monthindex
563 write(*,110)tower%daystring , tower%dayindex
564 write(*,110)tower%hourstring , tower%hourindex
565 write(*,110)tower%doystring , tower%doyindex
566 write(*,110)tower%hstring , tower%hindex
567 write(*,110)tower%hQCstring , tower%hQCindex
568 write(*,110)tower%lestring , tower%leindex
569 write(*,110)tower%leQCstring , tower%leQCindex
570 write(*,110)tower%neestring , tower%neeindex
571 write(*,110)tower%neeQCstring, tower%neeQCindex
572endif
573
574deallocate(columns)
575
576end subroutine decode_header

580function countchar(str1,solo)
581! Count the number of instances of the single character in a character string.
582! useful when parsing a comma-separated list, for example.
