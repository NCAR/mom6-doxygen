204! ! Check for duplicate observations
205! do i = 1, nused

207! lat(n) == lat(used(i)) .and. &

238 end do
239 do i = 1, num_qc
240 call set_qc_meta_data(obs_seq, i, 'Data QC')
241 end do
242
243endif
244
245obsloop2: do i = 1, nused
246
247 ! get the next unique observation in sorted time order
248 n = used(sorted_used(i))
249
250 ! convert offset in seconds to a real time type again
251 time_obs = increment_time(time_ref, tobs(n))
252
253! if ( elev(n) /= elev_miss ) then
254! palt = pres_alt_to_pres(elev(n)) * 0.01_r8
255! else
256! palt = pres_alt_to_pres(def_elev) * 0.01_r8
257! endif
258
259 ! extract actual time of observation in file into oday, osec.
260 call get_time(time_obs, osec, oday)
261
262! ! add altimeter data to obs_seq
263! if ( sfcp(n) /= sfcp_miss .and. qc_sfcp(n) == 0 .and. elev(n) /= elev_miss ) then
264!
265! altim = compute_altimeter(sfcp(n) * 0.01_r8, elev(n))
266! if ( plid(n) == 0 ) then
267! oerr = fixed_marine_pres_error(palt)
268! else
269! oerr = moving_marine_pres_error(palt)
270! endif
271
272! if ( altim >= 890.0_r8 .and. altim <= 1100.0_r8 .and. oerr /= missing_r8 ) then
273!
274! call create_3d_obs(lat(n), lon(n), elev(n), VERTISSURFACE, altim, &
275! MARINE_SFC_ALTIMETER, oerr, oday, osec, qc(n), obs)
276! call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
277!
278! endif
279!
280! ! if surface pressure and elevation do not exist, use SLP.
281! else if ( slp(n) /= slp_miss .and. qc_slp(n) == 0 ) then
282!
283! altim = compute_altimeter(slp(n) * 0.01_r8, 0.0_r8)
284! if ( plid(n) == 0 ) then
285! oerr = fixed_marine_pres_error(palt)
286! else
287! oerr = moving_marine_pres_error(palt)
288! endif
289!
290
291 call create_3d_obs(lat(n), lon(n), pres(n), vertispressure, uwnd, &
292 radiosonde_u_wind_component, oerr, oday, osec, qc(n), obs)
293 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
294
295 call create_3d_obs(lat(n), lon(n), pres(n), vertispressure, vwnd, &
296 radiosonde_v_wind_component, oerr, oday, osec, qc(n), obs)

300100 continue

302end do obsloop2
303
304! if we added any obs to the sequence, write it now.

312function string_to_time(str_time)
313 character(len=*), intent(in) :: str_time
314 type(time_type) :: string_to_time
315
316integer :: yr, mo, dy, hr, mn, sc
317integer :: rc
318
319read(str_time, "(I4,I2,I2,X,I2,I2,I2)", iostat=rc) yr, mo, dy, hr, mn, sc
320if (rc /= 0) then
321 print *, 'error converting string to time: '//trim(str_time)
322 string_to_time = set_time_missing()
323 return
324endif
325
326string_to_time = set_date(yr, mo, dy, hr, mn, sc)
327
328end function string_to_time
