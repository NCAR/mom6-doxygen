7!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
8!
9! netCDF_to_obs
10!
11! This converter is intended to be used as example of how to convert
12! basic netCDF files to observation sequence files in one step.
13! The netCDF file(s) in question is/are (bi)monthly values,
14! so we want to put each time in a separate observation sequence file.
15! The header of the (hypothetical) netCDF file looks like this:
16!
17! ncdump -h GIMMS3g_LAI_Bimonthly_2000_2010_60min.nc
18! netcdf GIMMS3g_LAI_Bimonthly_2000_2010_60min {
19! dimensions:
20! time = 264 ;
21! lat = 180 ;
22! lon = 360 ;
23! nv = 2 ;
24! variables:
25! float LAI(time, lat, lon) ;
26! LAI:_FillValue = -9999.f ;
27! LAI:units = "m^2/m^2" ;
28! LAI:cell_methods = "time: mean" ;
29! LAI:standard_name = "leaf_area_index" ;
30! LAI:long_name = "Leaf Area Index" ;
31! double lat(lat) ;
32! lat:standard_name = "latitude" ;
33! lat:long_name = "latitude coordinate" ;
34! lat:units = "degrees_north" ;
35! double lon(lon) ;
36! lon:standard_name = "longitude" ;
37! lon:long_name = "longitude coordinate" ;
38! lon:units = "degrees_east" ;
39! double time(time) ;
40! time:units = "days since 2000-01-01 00:00:00" ;
41! time:calendar = "gregorian" ;
42! time:standard_name = "time" ;
43! time:climatology = "climatology_bounds" ;
44! int climatology_bounds(time, nv) ;
45! climatology_bounds:calendar = "standard" ;
46! climatology_bounds:description = "start and end time for each time stamp" ;
47! climatology_bounds:units = "days since 2000-01-01 00:00:00" ;
48!
49! // global attributes:
50! :source = "AVHRR GIMMS LAI3g version 2" ;
51!}
52!
53!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!

93character(len=*), parameter :: source = 'netCDF_to_obs.f90'
94
95integer, parameter :: num_copies = 1
96integer, parameter :: num_qc = 1
97
98character(len=256) :: next_infile
99character(len=512) :: string1, string2, string3
100
101integer :: ncid, io, iunit, filenum
102integer :: day, seconds
103integer :: num_new_obs
104integer :: i, j, nlat, nlon, itime
105integer :: obs_type_integer
106
107logical :: first_obs, from_list = .false.
108
109real(r8) :: qc, lat, lon, vertvalue
110real(r8) :: missing_value
111
112type(obs_sequence_type) :: obs_seq
113type(obs_type) :: obs, prev_obs
114type(time_type) :: obs_time
115
116real(r8), allocatable :: latitude(:), longitude(:)
117real(r8), allocatable :: observation(:,:)
118real(digits12), allocatable :: obstime(:)
119
120integer :: ndims, idimension, ntimes, timedimid
121integer :: ncstart( nf90_max_var_dims)
122integer :: nccount( nf90_max_var_dims)
123integer :: dimlens( nf90_max_var_dims)
124character(len=NF90_MAX_NAME) :: dimnames(nf90_max_var_dims)
125character(len=256) :: output_file
126
127!------------------------------------------------------------------------
128! Declare namelist parameters
129!------------------------------------------------------------------------
130
131character(len=256) :: input_file = ''
132character(len=256) :: input_file_list = ''
133character(len=256) :: output_file_base = 'obs_seq.out'
134character(len=256) :: observation_varname
135character(len=256) :: observation_type
136logical :: debug = .false.
137real(r8) :: lon_bounds(2) = (/ 0.0_r8, 360.0_r8 /)
138real(r8) :: lat_bounds(2) = (/ -90.0_r8, 90.0_r8 /)
139
140! must include representation error, which depends on model/data grid size,
141real(r8) :: obs_error_standard_deviation = 0.2
142
143namelist /netcdf_to_obs_nml/ &
144 input_file, input_file_list, output_file_base, &
145 observation_varname, observation_type, &
146 debug, obs_error_standard_deviation, &
147 lon_bounds, lat_bounds
148
149call set_calendar_type(gregorian)
150
151! read the necessary parameters from input.nml
152call initialize_utilities(source)
153call static_init_obs_sequence()
154
155call find_namelist_in_file('input.nml', 'netCDF_to_obs_nml', iunit)
156read(iunit, nml = netcdf_to_obs_nml, iostat = io)
157call check_namelist_read(iunit, io, 'netCDF_to_obs_nml')
158
159! Record the namelist values used for the run
160if (do_nml_file()) write(nmlfileunit, nml=netcdf_to_obs_nml)
161if (do_nml_term()) write( * , nml=netcdf_to_obs_nml)
162
163! cannot have both a single filename and a list;
164! the namelist must shut one off.
165
166if (input_file /= '' .and. input_file_list /= '') then
167 call error_handler(e_err, 'main', 'One of input_file or filelist must be NULL', &
168 source)
169endif
170
171if (input_file_list /= '') from_list = .true.
172
173!-----------------------------------------------------------------------
174! Match the string defining the observation_type to the integer code
175
176obs_type_integer = get_index_for_type_of_obs(observation_type)
177if (obs_type_integer < 0) then
178 write(string1,*)'observation_type "'//trim(observation_type)//'" is not a valid TYPE.'
179 write(string2,*)'valid TYPEs are summarized in obs_kind_mod.f90 after processing by preprocess.'
180 write(string3,*)'They are defined in the obs_def_XXXXX_mod.f90 files.'
181 call error_handler(e_err, 'main', string1, source, text2=string2, text3=string3)
182endif
183
184!-----------------------------------------------------------------------
185! main loop that does either a single file or a list of files
186! Each timestep is going to a separate output file.
187
188filenum = 1
189fileloop: do ! until out of files
190
191 ! get the single name, or the next name from a list
192 if (from_list) then
193 next_infile = get_next_filename(input_file_list, filenum)
194 else
195 next_infile = input_file
196 if (filenum > 1) next_infile = ''
197 endif
198
199 write(*,*)'filenum ',filenum,', is next_infile ="'//trim(next_infile)//'"'
200 if (next_infile == '') exit fileloop
201
202 ncid = nc_open_file_readonly(next_infile, 'netCDF_to_obs')
203
204 call nc_get_variable_info(ncid, observation_varname, &
205 ndims = ndims, &
206 dimlens = dimlens, &
207 dimnames= dimnames, &
208 context = 'netCDF_to_obs setup')
209
210 ! Each timestep will be written to a separate output file whose
211 ! name reflects the time. All other dimensions are used to calculate
212 ! the theoretical maximum number of observations in each file.
213
214 num_new_obs = 1
215 do idimension = 1,ndims
216 select case (dimnames(idimension))
217 case ('time','Time','TIME')
218 ntimes = dimlens(idimension)
219 ncstart(idimension) = 1
220 nccount(idimension) = 1
221 timedimid = idimension
222 case default
223 num_new_obs = num_new_obs * dimlens(idimension)
224 ncstart(idimension) = 1
225 nccount(idimension) = dimlens(idimension)
226 continue
227 end select
228 enddo
229
230 allocate(obstime(ntimes))
231
232 call nc_read_time(ncid, obstime)
233
234 nlat = nc_get_dimension_size(ncid,'lat')
235 nlon = nc_get_dimension_size(ncid,'lon')
236
237 allocate(latitude(nlat))
238 allocate(longitude(nlon))
239 allocate(observation(nlon,nlat)) ! just for one slab
240
241 call nc_get_variable(ncid, 'lon', longitude)
242 call nc_get_variable(ncid, 'lat', latitude)
243
244 ! ensure longitudes are [0,360]
245 where(longitude < 0.0_r8) longitude = longitude + 360.0_r8
246
247 timeloop : do itime = 1,ntimes
248
249 if (debug) write(*,*)'timestep ',itime,' of ',ntimes
250
251 first_obs = .true.
252
253 call init_obs( obs, num_copies, num_qc)
254 call init_obs( prev_obs, num_copies, num_qc)
255 call init_obs_sequence( obs_seq, num_copies, num_qc, num_new_obs)
256 call set_copy_meta_data(obs_seq, num_copies, 'observation')
257 call set_qc_meta_data( obs_seq, num_qc, 'original QC')
258
259 day = floor(obstime(itime))
260 seconds = nint((obstime(itime) - real(day,digits12)) * 86400.0_digits12)
261 obs_time = set_time(seconds,day)
262
263 ncstart(timedimid) = itime
264
265 if (debug) then
266 write(*,*)'ncstart is ',ncstart(1:ndims)
267 write(*,*)'nccount is ',nccount(1:ndims)
268 endif
269
270 call getvar_real_2d(ncid, observation_varname, observation, dmiss=missing_value, &
271 nc_start=ncstart(1:ndims), nc_count=nccount(1:ndims))
272
273 latloop: do j = 1, nlat
274 lat = latitude(j)
275 if ( lat < lat_bounds(1) ) cycle latloop
276 if ( lat > lat_bounds(2) ) cycle latloop
277 lonloop: do i = 1, nlon
278
279 lon = longitude(i)
280 if ( lon < lon_bounds(1) ) cycle lonloop
281 if ( lon > lon_bounds(2) ) cycle lonloop
282
283 if (observation(i,j) == missing_value ) cycle lonloop
284
285 !------------------------------------------------------------------
286 ! Create observation
287
288
289 vertvalue = -888888.0_r8 ! does not matter for LAI
290 qc = 0 ! they are all good
291
292 call create_3d_obs(lat, lon, vertvalue, vertisundef, &
293 observation(i,j), obs_type_integer, &
294 obs_error_standard_deviation, day, seconds, qc, obs)
295
296 ! The insert operation is fast if given a time-ordered starting point.
297 ! Since all these observations are at the same time, just use the
298 ! previous observation as the time ordered starting point.
299 if (first_obs) then
300 call insert_obs_in_seq(obs_seq, obs)
301 first_obs = .false.
302 else
303 call insert_obs_in_seq(obs_seq, obs, prev_obs)
304 endif
305 prev_obs = obs
306
307 enddo lonloop
308 enddo latloop
309
310 ! if we added any obs to the sequence, write it out.
311 if (get_num_obs(obs_seq) > 0) then
312
313 call construct_filename(output_file_base, obs_time, output_file)
314
315 write(string1,*) 'writing ', get_num_obs(obs_seq), ' observations to "'//trim(output_file)//'"'
316 call error_handler(e_msg, 'main', string1, source)
317
318 call write_obs_seq(obs_seq, output_file)
319
320 else
321 write(string1,*) ' no observations for timestep ', itime
322 write(string2,*) ' in file "'//trim(next_infile)//'"'
323 call print_date(obs_time,' no observations at this date')
324 call print_time(obs_time,' no observations at this time')
325 call error_handler(e_warn, 'main', string1, source, text2=string2)
326 endif
327
328 ! Little known fact that if obs = prev_obs, destroying one of them destroys them both
329 call destroy_obs(obs)
330 ! if (get_num_obs(obs_seq) > 0) call destroy_obs_sequence(obs_seq)
331 call destroy_obs_sequence(obs_seq)
332
333 ! a little whitespace helps output readability
334 write(*,*)''
335 write(logfileunit,*)''
336
337 end do timeloop
338
339 deallocate(latitude)
340 deallocate(longitude)
341 deallocate(observation)
342
343 call nc_close_file(ncid)
344 filenum = filenum + 1
345
346enddo fileloop
347
348
349call error_handler(e_msg, 'main','Finished successfully.',source)
350call finalize_utilities()
351
352contains

354!-----------------------------------------------------------------------
355!> read the time variable from a netCDF file, applying 'common' attributes
356!> like the 'days since ....' and 'calendar' ...
357!> If the variable is not named 'time', 'Time', or 'TIME', the optional
358!> argument may be used to specify the variable name.
359!>
360
361subroutine nc_read_time(ncid, timearray, varname)
362
363integer, intent(in) :: ncid
364real(digits12), intent(out) :: timearray(:)
365character(len=*), optional, intent(in) :: varname
366
367character(len=NF90_MAX_NAME) :: timevariablename
368character(len=256) :: unitstring
369type(time_type) :: calendar_start
370integer :: ios, year, month, days, hours, minutes, seconds
371real(digits12) :: timeorigin
372integer :: ntimes
373
374! Search for the time variable name.
375write(string1,*)'time variable is not one of "time", "Time", or "TIME"'
376if (present(varname)) then
377 timevariablename = adjustl(varname)
378 if (.not. nc_variable_exists(ncid,timevariablename)) then
379 write(string1,*)'time variable is not "'//trim(timevariablename)//'"'
380 call error_handler(e_err,'nc_read_time',string1,source,text2='variable not found')
381 endif
382elseif (nc_variable_exists(ncid,'time')) then
383 timevariablename = 'time'
384elseif (nc_variable_exists(ncid,'Time')) then
385 timevariablename = 'Time'
386elseif (nc_variable_exists(ncid,'TIME')) then
387 timevariablename = 'TIME'
388else
389 call error_handler(e_err,'nc_read_time',string1,source,text2='variable not found')
390endif
391
392! FIXME set the calendar, check for legitimate values
393! calendar = nc_get_calendar(ncid,timevariablename)
394!
395! call error_handler(E_ERR, 'nc_read_time', &
396! 'calendar type "'//trim(calendar)//'" unsupported by nc_read_time() routine', &
397! source, text2=string3)
398
399call nc_get_variable(ncid,timevariablename,timearray,'nc_read_time')
400call nc_get_attribute_from_variable(ncid, timevariablename, 'units', unitstring, 'nc_read_time')
401
402if (unitstring(1:10) == 'days since') then
403
404 read(unitstring,'(11x,i4,5(1x,i2))',iostat=ios)year,month,days,hours,minutes,seconds
405 if (ios /= 0) then
406 write(string1,*)'Unable to interpret time "units" attribute. Error status was ',ios
407 write(string2,*)'expected "days since YYYY-MM-DD HH:MM:SS", got "'//trim(unitstring)//'"'
408 call error_handler(e_err, 'nc_read_time:', string1, source, text2=string2)
409 endif
410
411else if (unitstring(1:13) == 'seconds since') then
412
413 read(unitstring,'(14x,i4,5(1x,i2))',iostat=ios)year,month,days,hours,minutes,seconds
414 if (ios /= 0) then
415 write(string1,*)'Unable to interpret time "units" attribute. Error status was ',ios
416 write(string2,*)'expected "seconds since YYYY-MM-DD HH:MM:SS", got "'//trim(unitstring)//'"'
417 call error_handler(e_err, 'read_model_time:', string1, source, text2=string2)
418 endif
419
420else
421
422 write(string1,*) 'looking for "days since" or "seconds since" in the "units" attribute'
423 write(string2,*) 'units read as "'//trim(unitstring)//'"'
424 call error_handler(e_err, 'read_model_time:', 'unable to set start of calendar', &
425 source, text2=string1, text3=string2)
426
427endif
428
429! set_date is relative to the calendar in use already
430calendar_start = set_date(year, month, days, hours, minutes, seconds)
431
432call get_time(calendar_start, seconds, days)
433
434timeorigin = days + seconds/86400.0_r8
435timearray = timearray + timeorigin
436
437! Just confirm that the first time is what we expect.
438! ignore the reuse of 'calendar_start'
439if ( debug ) then
440
441 days = floor(timearray(1))
442 seconds = nint((timearray(1) - real(days,digits12)) * 86400.0_digits12)
443 calendar_start = set_time(seconds,days)
444
445 call print_time(calendar_start,'First time in file is')
446 call print_date(calendar_start,'First date in file is')
447
448 ntimes = size(timearray)
449
450 days = floor(timearray(ntimes))
451 seconds = nint((timearray(ntimes) - real(days,digits12)) * 86400.0_digits12)
452 calendar_start = set_time(seconds,days)
453
454 call print_time(calendar_start,'Last time in file is')
455 call print_date(calendar_start,'Last date in file is')
456
457endif
458
459end subroutine nc_read_time

466subroutine construct_filename(base, mytime, filename)
467
468character(len=*), intent(in) :: base
469type(time_type), intent(in) :: mytime
470character(len=*), intent(out) :: filename
471
472integer :: year, month, days, hours, minutes, seconds
473
