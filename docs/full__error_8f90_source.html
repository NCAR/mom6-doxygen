5!> Correct covariances for fixed ensemble sizes.
6!> See Anderson, J. L., 2011: Localization and Sampling Error Correction
7!> in Ensemble Kalman Filter Data Assimilation.

10program full_error
11
12! This version of the program reads the ensemble size and base filename
13! for the output from a namelist.
14
15use types_mod, only : r8
16use utilities_mod, only : open_file, close_file, error_handler, &
17 find_namelist_in_file, check_namelist_read, &
18 do_nml_file, do_nml_term, nmlfileunit, &
19 initialize_utilities, finalize_utilities, e_err
20use random_seq_mod, only : random_seq_type, init_random_seq, &
21 twod_gaussians
22
23implicit none
24
25character(len=*), parameter :: source = 'full_error.f90'
26
27integer, parameter :: num_times = 1
28integer, parameter :: num_samples = 100000000
29
30! ---------------
31! namelist items
32
33integer :: ens_size = 80
34character(len=256) :: output_filename = 'final_full'
35
36namelist /full_error_nml/ ens_size, output_filename
37
38
39type (random_seq_type) :: ran_id
40real(r8), allocatable :: pairs(:,:), temp(:)
41
42real(r8) :: zero_2(2) = 0.0, cov(2, 2)
43real(r8) :: t_correl, sample_correl, alpha, beta
44real(r8) :: s_mean(2), s_var(2), reg_mean, reg_sd, t_sd1, t_sd2, true_correl_mean
45real(r8) :: tcorrel_sum(201), reg_sum(201), reg_2_sum(201)
46
47integer :: i, j, k, bin_num, bin_count(201)
48integer :: iunit, io
49
50character(len=16) :: formstring
51character(len=256) :: outname
52character(len=512) :: errstring
53
54!
55! start of executable code
56!
57
58call initialize_utilities('full_error')
59
60! Read the namelist entry
61call find_namelist_in_file("input.nml", "full_error_nml", iunit)
62read(iunit, nml = full_error_nml, iostat = io)
63call check_namelist_read(iunit, io, "full_error_nml")
64
65! Record the namelist values used for the run
66if (do_nml_file()) write(nmlfileunit, nml=full_error_nml)
67if (do_nml_term()) write( * , nml=full_error_nml)
68
69call close_file(iunit)
70
71call init_random_seq(ran_id)
72
73write(*, *) 'stats for ensemble size ', ens_size
74
75allocate(pairs(2, ens_size), temp(ens_size))
76
77bin_count = 0
78tcorrel_sum = 0.0_r8
79reg_sum = 0.0_r8
80reg_2_sum = 0.0_r8
81
82! Uniformly distributed sequence of true correlations
83do j = 1, num_samples
84 ! Assume that true correlation is uniform [-1,1]
85 t_correl = -1.0_r8 + 2.0_r8 * ((j - 1) / (num_samples - 1.0_r8))
86 if(j > num_samples) t_correl = 0.0_r8
87
88 do k = 1, num_times
89
90 t_sd1 = 1.0_r8
91 t_sd2 = 1.0_r8
92
93 ! Generate the covariance matrix for this correlation
94 cov(1, 1) = t_sd1**2
95 cov(2, 2) = t_sd2**2
96 cov(1, 2) = t_correl * t_sd1 * t_sd2
97 cov(2, 1) = cov(1, 2)
98
99 ! Loop to generate an ensemble size sample from this correl
100 ! Generate a random sample of size ens_size from something with this correlation
101 do i = 1, ens_size
102 call twod_gaussians(ran_id, zero_2, cov, pairs(:, i))
103 enddo
104
105 ! Compute the sample correlation
106 call comp_correl(pairs, ens_size, sample_correl)
107
108 ! Bin statistics for each 0.01 sample correlation bin;
109 ! start with just mean true correlation
110 bin_num = (sample_correl + 1.0_r8) / 0.01_r8 + 1
111 ! Keep a sum of the sample_correl and squared to compute standard deviation
112 tcorrel_sum(bin_num) = tcorrel_sum(bin_num) + t_correl
113 !-----------------
114 ! Also interested in finding out what the spurious variance reduction factor is
115 ! First need to compute sample s.d. for the obs and unobs variable
116 do i = 1, 2
117 temp = pairs(i, :)
118 call sample_mean_var(temp, ens_size, s_mean(i), s_var(i))
119 enddo
120
121 !-----------------
122 reg_sum(bin_num) = reg_sum(bin_num) + t_correl * sqrt(s_var(2) / s_var(1))
123 reg_2_sum(bin_num) = reg_2_sum(bin_num) + (t_correl * sqrt(s_var(2) / s_var(1)))**2
124
125 bin_count(bin_num) = bin_count(bin_num) + 1
126
127 enddo
128enddo
129
130! make the size of the integer output .X, .XX, .XXX, etc
131! depending on the number of decimal digits in the value.
132if (ens_size < 10) then
133 formstring = '(2A,I1)'
134else if (ens_size < 100) then
135 formstring = '(2A,I2)'
136else if (ens_size < 1000) then
137 formstring = '(2A,I3)'
138else if (ens_size < 10000) then
139 formstring = '(2A,I4)'
140else if (ens_size < 100000) then
141 formstring = '(2A,I5)'
142else
143 formstring = '(2A,I8)'
144endif
145
146! filename
147write(outname, formstring) trim(output_filename), '.', ens_size
148
149! text file, overwrite existing file if present
150iunit = open_file(outname, 'formatted', 'write')
151
152! always generate exactly 200 entries in the output file
153do i = 1, 200
154 ! must have at least 2 counts to compute a std dev
155 if(bin_count(i) <= 1) then
156 write(errstring, *) 'Bin ', i, ' has ', bin_count(i), ' counts'
157 call error_handler(e_err,'full_error', errstring, &
158 source, text2="All bins must have at least 2 counts")
159 endif
160
161 ! Compute the standard deviation of the true correlations
162 true_correl_mean = tcorrel_sum(i) / bin_count(i)
163 reg_mean = reg_sum(i) / bin_count(i)
164 reg_sd = sqrt((reg_2_sum(i) - bin_count(i) * reg_mean**2) / (bin_count(i) - 1))
165
166 if(reg_sd <= 0.0_r8) then
167 alpha = 1.0_r8
168 else
169 !!!beta = reg_mean**2 / reg_sd**2
170 ! Correct for bias in the standard deviation for very small ensembles, too
171 beta = reg_mean**2 / (reg_sd**2 * (1.0_r8 + 1.0_r8 / ens_size))
172 alpha = beta / (1.0_r8 + beta)
173 endif
174
175 write(*, *) 'bin, count, mean ', i, bin_count(i), true_correl_mean, alpha
176 write(iunit, 10) i, bin_count(i), true_correl_mean, alpha
177enddo
178
17910 format (i4,i9,2g25.14)
180
181call close_file(iunit)
182
183! print out just to stdout how many counts, if any, fell beyond bin 200
184write(*, *) 'bin, count, mean ', 201, bin_count(201), 0, 0
185
186deallocate(pairs, temp)
187
188call finalize_utilities()
189
190! end of main program
191
192contains
193

196subroutine comp_correl(ens, n, correl)
197
198integer, intent(in) :: n
199real(r8), intent(in) :: ens(2, n)
200real(r8), intent(out) :: correl
201
202real(r8) :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
203
204
205sum_x = sum(ens(2, :))
206sum_y = sum(ens(1, :))
207sum_xy = sum(ens(2, :) * ens(1, :))
208sum_x2 = sum(ens(2, :) * ens(2, :))
209
210! Computation of correlation
211sum_y2 = sum(ens(1, :) * ens(1, :))
212
213correl = (n * sum_xy - sum_x * sum_y) / &
214 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
215
216end subroutine comp_correl
217
218!----------------------------------------------------------------
219
220subroutine sample_mean_var(x, n, mean, var)
221
222integer, intent(in) :: n
223real(r8), intent(in) :: x(n)
224real(r8), intent(out) :: mean, var
225
226real(r8) :: sx, s_x2
227
228sx = sum(x)
229s_x2 = sum(x * x)
230mean = sx / n
231var = (s_x2 - sx**2 / n) / (n - 1)
232
233end subroutine sample_mean_var
234
235
236end program full_error
