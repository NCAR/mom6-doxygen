71!> Subroutine initialize_qc
72!>
73!> read the quality control namelist
74!------------------------------------------------------------------------------
75subroutine initialize_qc()
76
77character(len=512) :: msgstring
78
79integer :: iunit, io
80
81! Read the namelist entry
82call find_namelist_in_file("input.nml", "quality_control_nml", iunit)
83read(iunit, nml = quality_control_nml, iostat = io)
84call check_namelist_read(iunit, io, "quality_control_nml")
85
86! Record the namelist values used for the run ...
87if (do_nml_file()) write(nmlfileunit, nml=quality_control_nml)
88if (do_nml_term()) write( * , nml=quality_control_nml)
89
90if(do_output()) then
91 write(msgstring, '(A,I4)') 'Will reject obs with Data QC larger than ',nint(input_qc_threshold)
92 call error_handler(e_msg,'quality_control_mod:', msgstring)
93
94 ! Let user know what settings the outlier threshold has, and data qc.
95 if (outlier_threshold <= 0.0_r8) then
96 write(msgstring, '(A)') 'No observation outlier threshold rejection will be done'
97 else
98 write(msgstring, '(A,F12.6,A)') 'Will reject obs values more than', outlier_threshold, ' sigma from mean'
99 endif
100 call error_handler(e_msg,'quality_control_mod:', msgstring)
101
102 ! if doing something special with outlier threshold, say so
103 if (enable_special_outlier_code) then
104 call error_handler(e_msg,'quality_control_mod:', &
105 'special outlier threshold handling enabled', source)
106 endif
107
108endif
109
110end subroutine initialize_qc

162!------------------------------------------------------------------------------
163!> Subroutine get_dart_qc
164!>
165!> Consolidate forward operator qc
166!> find the min and max istatus values across all ensemble members. These are
167!> either set by dart code, or returned by the model-specific model_interpolate()
168!> routine, or by forward operator code in obs_def_xxx_mod files.
169!>
170!------------------------------------------------------------------------------
171subroutine get_dart_qc(istatus, ens_size, assimilate_this_ob, evaluate_this_ob, isprior, dart_qc)
172
173integer, intent(in) :: ens_size !! number of ensemble members
174integer, intent(in) :: istatus(ens_size) !! 0=ok, >0 is error, <0 reserved for system use
175logical, intent(in) :: assimilate_this_ob !! true if obs was assimilated
176logical, intent(in) :: evaluate_this_ob !! true if obs was evaluated
177logical, intent(in) :: isprior !! true for prior pass; false for posterior
178integer, intent(inout) :: dart_qc !! resulting dart qc
179
180logical :: failed_fop, inconsistent
181
182! If we didn't call the forward operators, return now.
183if (dart_qc == dartqc_bad_incoming_qc) return
184
185failed_fop = (maxval(istatus(:)) > 0)
186inconsistent = (minval(istatus(:)) /= 0)
187
188! Now do a case statement to figure out what the qc result should be
189! For prior, have to test for a bunch of stuff
190! FIXME: note that this case statement doesn't cover every possibility;
191! if there's an error in the code and a minus value gets into the forward
192! operator istatus without being caught, it will fail all cases below.
193! add another line for 'internal inconsistency' to be safe.
194
195if(isprior) then
196
197 ! only do this for the priors. the posterior failures
198 ! don't change whether the prior worked or not.
199
200 if((.not. assimilate_this_ob) .and. (.not. evaluate_this_ob)) then
201 dart_qc = dartqc_not_in_namelist
202 else if(failed_fop) then ! At least one forward operator failed
203 dart_qc = dartqc_failed_fop
204 else if(evaluate_this_ob) then ! Observation to be evaluated only
205 dart_qc = dartqc_eval_good_fop
206 else if(.not. inconsistent) then ! All clear, assimilate this ob
207 dart_qc = dartqc_assim_good_fop
208
209 else ! 'should not happen'
210 dart_qc = -99 ! inconsistent istatus codes
211 call error_handler(e_err, 'get_dart_qc: ', &
212 'internal error, should not happen: bad else in qc case', source)
213 endif
214
215else ! posterior
216
217 ! For failed posterior, only update qc if prior was successful
218 if(failed_fop) then
219 ! Both the following 2 tests and assignments were on single executable lines,
220 ! but one compiler (gfortran) was confused by this, so they were put in
221 ! if/endif blocks.
222 if(dart_qc == dartqc_assim_good_fop) then
223 dart_qc = dartqc_assim_failed_post_fop
224
225 else if(dart_qc == dartqc_eval_good_fop) then
226 dart_qc = dartqc_eval_failed_post_fop
227
228 endif
229 endif
230
231endif
232
233end subroutine get_dart_qc

235!------------------------------------------------------------------------------
236!> Subroutine check_outlier_threshold
237!>
238!> Check on the outlier threshold quality control. If existing DART QC
239!> is good and the obs is outside the threshold, set the QC to indicate that.
240!------------------------------------------------------------------------------
241subroutine check_outlier_threshold(obs_prior_mean, obs_prior_var, obs_val, obs_err_var, &
242 obs_seq, this_obs_key, dart_qc)
243
244real(r8), intent(in) :: obs_prior_mean !! prior observation mean
245real(r8), intent(in) :: obs_prior_var !! prior observation variance
246real(r8), intent(in) :: obs_val !! observation value
247real(r8), intent(in) :: obs_err_var !! observation error variance
248type(obs_sequence_type), intent(in) :: obs_seq !! observation sequence
249integer, intent(in) :: this_obs_key !! index for this observation
250integer, intent(inout) :: dart_qc !! possibly modified DART QC
251
252real(r8) :: error, diff_sd, ratio
253
254logical :: failed
255
256! FIX ME: could be a loop if we consider groups
257
258! Check on the outlier threshold quality control:
259! BUG FIX: this was using the incoming qc threshold before.
260! it should only be doing the outlier test on dart qc values of 0 or 1.
261! if there is already a different qc code set, leave it alone.
262! only if it is still successful (assim or eval, 0 or 1), then check
263! for failing outlier test.
264
265! If bad QC flag exit here.
266if ( .not. good_dart_qc(dart_qc)) return
267
268error = obs_prior_mean - obs_val
269diff_sd = sqrt(obs_prior_var + obs_err_var)
270
271if (diff_sd /= 0.0_r8) then
272 ratio = abs(error / diff_sd)
273else
274 ratio = 0.0_r8
275endif
276
277! if special handling requested, pass in the outlier ratio for this obs,
278! the default outlier threshold value, and enough info to extract the specific
279! obs type for this obs. the function should return .true. if this is an
280! outlier, .false. if it is ok.
281if (enable_special_outlier_code) then
282 failed = failed_outlier(ratio, this_obs_key, obs_seq)
283else
284 if ( outlier_threshold < 0 ) return ! don't modify dart_qc if no outlier check
285 failed = (ratio > outlier_threshold)
286endif
287
288if (failed) then
289 dart_qc = dartqc_failed_outlier_test
290endif
291
292end subroutine check_outlier_threshold

294!------------------------------------------------------------------------------
295!> Function failed_outlier
296!>
297!> Intended to be easy to modify if needed to use other criteria
298!> when evaluating the outlier threshold. Set the namelist to
299!> turn this call on (enable_special_outlier_code) and then add your
300!> own tests to this section. Return 'true' if the tests fail,
301!> return 'false' if the obs is ok.
302!------------------------------------------------------------------------------
303function failed_outlier(ratio, this_obs_key, seq)
304
305! return true if the observation value is too far away from the ensemble mean
306! and should be rejected and not assimilated.
307
308real(r8), intent(in) :: ratio !! observation error ratio
309integer, intent(in) :: this_obs_key !! index for this observation
310type(obs_sequence_type), intent(in) :: seq !! the observation sequence
311logical :: failed_outlier !! true if outlier test fails
312
313! the default test is: if (ratio > outlier_threshold) failed_outlier = .true.
314! but you can add code here to do different tests for different observation
315! types. this function is only called if this namelist item is set to true:
316! enable_special_outlier_code = .true.
317! in the &quality_control_nml namelist. it is intended to be customized by the user.
318
319integer :: this_obs_type
320
321type(obs_def_type) :: obs_def
322type(obs_type), save :: observation
323
324logical :: first_time = .true.
325
326! make sure there's space to hold the observation. this is a memory
327! leak in that we never release this space, but we only allocate it
328! one time so it doesn't grow. if you are going to access the values
329! of the observation or the qc values, you must change the 0, 0 below
330! to match what's in your obs_seq file. the example code below uses
331! only things in the obs_def derived type and so doesn't need space
332! allocated for either copies or qcs.
333if (first_time) then
334 call init_obs(observation, 0, 0)
335 first_time = .false.
336endif
337
338! if you want to do something different based on the observation specific type:
339call get_obs_from_key(seq, this_obs_key, observation)
340call get_obs_def(observation, obs_def)
341this_obs_type = get_obs_def_type_of_obs(obs_def)
342
343! note that this example uses the specific type (e.g. RADIOSONDE_TEMPERATURE)
344! to make decisions. you have the observation so any other part (e.g. the
345! time, the value, the error) is available to you as well.
346
347select case (this_obs_type)
348
349! example of specifying a different threshold value for one obs type:
350! case (RADIOSONDE_TEMPERATURE)
351! if (ratio > some_other_value) then
352! failed_outlier = .true.
353! else
354! failed_outlier = .false.
355! endif
356
357! accept all values of this observation type no matter how far
358! from the ensemble mean:
359! case (AIRCRAFT_U_WIND_COMPONENT, AIRCRAFT_V_WIND_COMPONENT)
360! failed_outlier = .false.
361
362 case default
363 if (ratio > outlier_threshold) then
364 failed_outlier = .true.
365 else
366 failed_outlier = .false.
367 endif
368 ! outlier_threshold < 0 means don't do outlier check.
369 if ( outlier_threshold < 0 ) failed_outlier = .false.
370
371end select
372
373end function failed_outlier

392 good_dart_qc = .false.
