6!> A collection of interfaces to the MPI (Message Passing Interface)
7!> multi-processor communication library routines.
8!>
9!> This file and its companion file, null_mpi_utilities_mod.f90,
10!> are the only modules in the DART system that should be calling
11!> the MPI library routines directly. This isolates the MPI calls
12!> and allows programs to swap in the null version to compile the
13!> same source files into a serial program.
14!>
15!> The names of these routines were intentionally picked to be
16!> more descriptive to someone who doesn't the MPI interfaces.
17!> e.g. MPI_AllReduce() may not immediately tell a user what
18!> it does, but broadcast_minmax() is hopefully more helpful.
19!>
20!> If you add any routines or change any arguments in this file
21!> you must make the same changes in the null version. These two
22!> modules have the same module name and must have identical
23!> public routines and calling formats.
24!>
25!> All MPI routines are called from here. There is a companion
26!> file which has the same module name and entry points but all
27!> routines are stubs. This allows a single-task version of the
28!> code to be compiled without the MPI libraries.

214!> Initialize MPI and query it for global information. Make a duplicate
215!> communicator so that any user code which wants to call MPI will not
216!> interfere with any outstanding asynchronous requests, accidental tag
217!> matches, etc. This routine must be called before any other routine in
218!> this file, and it should not be called more than once (but it does have
219!> defensive code in case that happens.)
220
221subroutine initialize_mpi_utilities(progname, alternatename, communicator)
222
223character(len=*), intent(in), optional :: progname
224character(len=*), intent(in), optional :: alternatename
225type(mpi_comm), intent(in), optional :: communicator
226
227integer :: errcode, iunit
228logical :: already
229
230if ( module_initialized ) then
231 ! return without calling the code below multiple times
232 write(errstring, *) 'initialize_mpi_utilities has already been called'
233 call error_handler(e_warn,'initialize_mpi_utilities', errstring, source)
234 return
235endif
236
237! prevent any other code from calling into this init
238! routine and causing overlapping code execution
239module_initialized = .true.
240
241! some implementations of mpich need this to happen before any I/O is done.
242! this makes the startup sequence very tricky.
243! still, allow for the possibility that the user code has already initialized
244! the MPI libs and do not try to call initialize again.
245errcode = -999
246call mpi_initialized(already, errcode)
247if (errcode /= mpi_success) then
248 write(*, *) 'MPI_Initialized returned error code ', errcode
249 call exit(-99)
250endif
251if (.not.already) then
252 call mpi_init(errcode)
253 if (errcode /= mpi_success) then
254 write(*, *) 'MPI_Init returned error code ', errcode
255 call exit(-99)
256 endif
257endif
258
259if (.not. present(communicator)) then
260 ! give this a temporary initial value, in case we call the abort code.
261 ! later we will dup the world comm and use a private comm for our comms.
262 my_local_comm = mpi_comm_world
263else
264 my_local_comm = communicator
265 given_communicator = .true.
266endif
267
268call mpi_comm_rank(my_local_comm, myrank, errcode)
269if (errcode /= mpi_success) then
270 write(*, *) 'MPI_Comm_rank returned error code ', errcode
271 call exit(-99)
272endif
273
274! pass the arguments through so the utilities can log the program name
275! only PE0 gets to output, whenever possible.
276if (myrank == 0) then
277 call initialize_utilities(progname, alternatename, .true.)
278else
279 call initialize_utilities(progname, alternatename, .false.)
280endif
281
282! save a copy of the initial program name for use in finalize.
283! make sure it is not too long to fit into our variable.
284if (present(progname)) then
285 if (len_trim(progname) <= len(saved_progname)) then
286 saved_progname = trim(progname)
287 else
288 saved_progname = progname(1:len(saved_progname))
289 endif
290endif
291
292! if logging, add this info to the log
293! (must come after regular utils are initialized)
294
295! this must come AFTER the standard utils are initialized.
296! Read the DART namelist for the mpi_utilities.
297if (read_namelist) then
298 call find_namelist_in_file('input.nml', 'mpi_utilities_nml', iunit)
299 read(iunit, nml = mpi_utilities_nml, iostat = errcode)
300 call check_namelist_read(iunit, errcode, "mpi_utilities_nml")
301else
302 errstring = ' !must edit assimilation_code/modules/utilities/mpi_utilities_mod.f90 to enable this namelist'
303 if (do_nml_file()) write(nmlfileunit, '(A)') trim(errstring)
304 if (do_nml_term()) write( * , '(A)') trim(errstring)
305endif
306
307! Record the namelist values used for the run
308if (do_nml_file()) write(nmlfileunit, nml=mpi_utilities_nml)
309if (do_nml_term()) write( * , nml=mpi_utilities_nml)
310
311
312! duplicate the world communicator to isolate us from any other user
313! calls to MPI. All subsequent mpi calls here will use the local communicator
314! and not the global world comm.
315if (.not. given_communicator .and. create_local_comm) then
316 call mpi_comm_dup(mpi_comm_world, my_local_comm, errcode)
317 if (errcode /= mpi_success) then
318 write(errstring, '(a,i8)') 'MPI_Comm_dup returned error code ', errcode
319 call error_handler(e_err,'initialize_mpi_utilities', errstring, source)
320 endif
321endif
322
323! find out who we are (0 to N-1).
324call mpi_comm_rank(my_local_comm, myrank, errcode)
325if (errcode /= mpi_success) then
326 write(errstring, '(a,i8)') 'MPI_Comm_rank returned error code ', errcode
327 call error_handler(e_err,'initialize_mpi_utilities', errstring, source)
328endif
329
330! number of tasks (if 10, returns 10. task id numbers go from 0-9)
331call mpi_comm_size(my_local_comm, total_tasks, errcode)
332if (errcode /= mpi_success) then
333 write(errstring, '(a,i8)') 'MPI_Comm_size returned error code ', errcode
334 call error_handler(e_err,'initialize_mpi_utilities', errstring, source)
335endif
336
337! tell the utilities module what task number we are.
338call set_tasknum(myrank)
339
340! Turn off non-critical log messages from all but task 0, for performance
341! and for sanity (e.g. only one copy of informational messages). can be
342! overridden by namelist if desired.
343if (all_tasks_print) then
344 call set_output(.true.) ! everyone gets to talk
345else
346 if (myrank /= 0) call set_output(.false.) ! process 0 speaks for all
347endif
348
349! Users have the option of redefining the DART r8 kind to be the same size
350! as r4. But when we call the MPI routines we have to know what MPI type
351! to use. The internal dart kind 'digits12' is always defined to be *8, so
352! if the parameters r8 and digits12 are the same value, use *8. Otherwise
353! assume r4 and r8 were defined to be the same and use *4. (Some users
354! do this to cut down on the space requirements.)
355
356if (r8 == digits12) then
357 datasize = mpi_real8
358 !print *, "using real * 8 for datasize of r8"
359else
360 datasize = mpi_real4
361 if (myrank == 0) then
362 write(errstring, *) "Using real * 4 for datasize of r8"
363 call error_handler(e_msg,'initialize_mpi_utilities: ',errstring, source)
364 endif
365endif
366
367! create a type we can use for integer(i8) calls
368longinttype = mpi_integer8
369! or:
370!call MPI_Type_Create_F90_Integer(15, longinttype, errcode)
371!if (errcode /= MPI_SUCCESS) then
372! write(errstring, '(a,i8)') 'MPI_Type_Create_F90_Integer returned error code ', errcode
373! call error_handler(E_ERR,'initialize_mpi_utilities', errstring, source)
374!endif
375
376
377! in async 4 mode, where the controlling job (usually filter) and the
378! model are both mpi tasks and they handshake via named pipes, the tasks
379! writing and reading the named pipes must be on the same node as the
380! one running the start script. many MPI implementations (mpich, mvapich)
381! put task 0 on the first node, with the script. some (openmpi)
382! lay them out in reverse order, and task N-1 is on the same node
383! as the script. if this is wrong, things will hang when the
384! filter task tries to advance the model - it won't be able to
385! write the 'go ahead' message to the pipe. set this via namelist
386! to match what works on your system.
387if (reverse_task_layout) then
388 head_task = total_tasks - 1
389else
390 head_task = 0 ! normal case
391endif
392
393! MPI successfully initialized. Log for the record how many tasks.
394if (verbose) write(*,*) "PE", myrank, ": MPI successfully initialized"
395
396if (myrank == 0) then
397 write(errstring, *) 'Running with ', total_tasks, ' MPI processes.'
398 call error_handler(e_msg,'initialize_mpi_utilities: ',errstring, source)
399endif
400
401end subroutine initialize_mpi_utilities

405!> Shut down MPI cleanly. This must be done before the program exits; on
406!> some implementations of MPI the final I/O flushes are not done until this
407!> is called. The optional argument can prevent us from calling MPI_Finalize,
408!> so that user code can continue to use MPI after this returns. Calling other
409!> routines in this file after calling finalize will invalidate your warranty.
410
411subroutine finalize_mpi_utilities(callfinalize, async)
412 logical, intent(in), optional :: callfinalize
413 integer, intent(in), optional :: async
414
415integer :: errcode
416logical :: dofinalize
417
418if ( .not. module_initialized ) then
419 write(errstring, *) 'initialize_mpi_utilities() must be called first'
420 call error_handler(e_err,'finalize_mpi_utilities', errstring, source)
421endif
422
423! give the async=4 case a chance to tell the script to shut down.
424if (present(async)) then
425 call finished_task(async)
426endif
427
428! close the log files and write a timestamp
429if (saved_progname /= '') then
430 call finalize_utilities(saved_progname)
431else
432 call finalize_utilities()
433endif
434
435! For the SGI implementation of MPI in particular, sync before shutting
436! down MPI. Once MPI is shut down, no other output can be written; it causes
437! tasks to hang if they try. Make sure all tasks are here and ready to
438! close down at the same time.
439call task_sync()
440
441if (.not. given_communicator) then
442 ! Release the private communicator we created at init time.
443 if (my_local_comm /= mpi_comm_world) then
444 call mpi_comm_free(my_local_comm, errcode)
445 if (errcode /= mpi_success) then
446 write(errstring, '(a,i8)') 'MPI_Comm_free returned error code ', errcode
447 call error_handler(e_err,'finalize_mpi_utilities', errstring, source)
448 endif
449 endif
450 my_local_comm = mpi_comm_world
451endif
452
453! If the optional argument is not given, or is given and is true,
454! shut down mpi. Only if the argument is specified and is false do we
455! skip the finalization.
456if (.not.present(callfinalize)) then
457 dofinalize = .true.
458else
459 dofinalize = callfinalize
460endif
461
462! Normally we shut down MPI here. If the user tells us not to shut down MPI
463! they must call this routine from their own code before exiting.
464if (dofinalize) then
465 if (verbose) write(*,*) "PE", myrank, ": MPI finalize being called now"
466 call mpi_finalize(errcode)
467 if (errcode /= mpi_success) then
468 write(errstring, '(a,i8)') 'MPI_Finalize returned error code ', errcode
469 call error_handler(e_err,'finalize_mpi_utilities', errstring, source)
470 endif
471endif
472
473! NO I/O after calling MPI_Finalize. on some MPI implementations
474! this can hang the job.
475
476end subroutine finalize_mpi_utilities

545!> Send the srcarray to the destination id.
546!> If time is specified, it is also sent in a separate communications call.
547!> This is a synchronous call; it will not return until the destination has
548!> called receive to accept the data. If the send_to/receive_from calls are
549!> not paired correctly the code will hang.
550
551subroutine send_to(dest_id, srcarray, time, label)
552 integer, intent(in) :: dest_id
553 real(r8), intent(in) :: srcarray(:)
554 type(time_type), intent(in), optional :: time
555 character(len=*), intent(in), optional :: label
556
557integer :: tag, errcode
558integer :: itime(2)
559integer(i8) :: itemcount, offset, nextsize
560real(r8), allocatable :: tmpdata(:)
561
562if (verbose) write(*,*) "PE", myrank, ": start of send_to "
563
564if ( .not. module_initialized ) then
565 write(errstring, *) 'initialize_mpi_utilities() must be called first'
566 call error_handler(e_err,'send_to', errstring, source)
567endif
568
569! simple idiotproofing
570if ((dest_id < 0) .or. (dest_id >= total_tasks)) then
571 write(errstring, '(a,i8,a,i8)') "destination task id ", dest_id, &
572 "must be >= 0 and < ", total_tasks
573 call error_handler(e_err,'send_to', errstring, source)
574endif
575
576itemcount = size(srcarray,kind=i8)
577
578if (present(label)) then
579 write(*,*) trim(label)//" PE", myrank, ": send_to itemsize ", itemcount, " dest ", dest_id
580else if (verbose) then
581 write(*,*) "PE", myrank, ": send_to itemsize ", itemcount, " dest ", dest_id
582endif
583
584! use my task id as the tag; unused at this point.
585tag = myrank
586
587if (make_copy_before_sendrecv) allocate(tmpdata(min(itemcount, sndrcv_maxsize)))
588
589
590if (itemcount <= sndrcv_maxsize) then
591
592 if (verbose) write(*,*) "PE", myrank, ": send_to ", itemcount, " dest ", dest_id
593
594 if (.not. make_copy_before_sendrecv) then
595 call mpi_ssend(srcarray, int(itemcount,i4), datasize, dest_id, tag, &
596 my_local_comm, errcode)
597 else
598 ! this copy should be unneeded, but on the intel fortran 9.0 compiler and mpich
599 ! on one platform, calling this subroutine with an array section resulted in some
600 ! apparently random memory corruption. making a copy of the data into a local,
601 ! contiguous buffer before send and receive fixed the corruption. this shouldn't
602 ! have been needed, and is a performance/memory sink.
603
604 tmpdata = srcarray
605 call mpi_ssend(tmpdata, int(itemcount,i4), datasize, dest_id, tag, &
606 my_local_comm, errcode)
607 endif
608else
609 ! there are a few places in the code where we send/receive a full state vector.
610 ! as these get really large, they may exceed the limits of the MPI library.
611 ! break large messages up into smaller chunks if needed.
612 offset = 1
613 do while (offset <= itemcount)
614 if (itemcount-offset >= sndrcv_maxsize) then
615 nextsize = sndrcv_maxsize
616 else if (itemcount-offset == 0) then
617 nextsize = 1
618 else
619 nextsize = itemcount - offset
620 endif
621
622 if (verbose) write(*,*) 'sending array items ', offset, ' thru ' , offset + nextsize - 1

624 if (.not. make_copy_before_sendrecv) then
625 call mpi_ssend(srcarray(offset:offset+nextsize-1), int(nextsize,i4), datasize, dest_id, tag, &
626 my_local_comm, errcode)
627 else
628 tmpdata = srcarray(offset:offset+nextsize-1)
629 call mpi_ssend(tmpdata, int(nextsize,i4), datasize, dest_id, tag, &
630 my_local_comm, errcode)
631 endif
632 if (errcode /= mpi_success) then
633 write(errstring, '(a,i8)') 'MPI_Ssend returned error code ', errcode
634 call error_handler(e_err,'send_to', errstring, source)
635 endif
636 offset = offset + nextsize
637 enddo
638endif
639
640if (errcode /= mpi_success) then
641 write(errstring, '(a,i8)') 'MPI_Ssend returned error code ', errcode
642 call error_handler(e_err,'send_to', errstring, source)
643endif

646if (present(time)) then
647 if (verbose) write(*,*) "PE", myrank, ": time present"
648 call get_time(time, itime(1), itime(2))
649 call mpi_ssend(itime, 2, mpi_integer, dest_id, tag*2, my_local_comm, errcode)
650 if (errcode /= mpi_success) then

665!-----------------------------------------------------------------------------
666
667!> Receive data into the destination array from the src task.
668!> If time is specified, it is received in a separate communications call.
669!> This is a synchronous call; it will not return until the source has
670!> sent the data. If the send_to/receive_from calls are not paired correctly
671!> the code will hang.
672
673subroutine receive_from(src_id, destarray, time, label)
674 integer, intent(in) :: src_id
675 real(r8), intent(inout) :: destarray(:)
676 type(time_type), intent(out), optional :: time
677 character(len=*), intent(in), optional :: label
678
679integer :: tag, errcode
680integer :: itime(2)
681type(mpi_status) :: status
682integer(i8) :: itemcount, offset, nextsize
683real(r8), allocatable :: tmpdata(:)
684
685if (verbose) write(*,*) "PE", myrank, ": start of receive_from "
686
687if ( .not. module_initialized ) then
688 write(errstring, *) 'initialize_mpi_utilities() must be called first'
689 call error_handler(e_err,'receive_from', errstring, source)
690endif
691
692! simple idiotproofing
693if ((src_id < 0) .or. (src_id >= total_tasks)) then
694 write(errstring, '(a,i8,a,i8)') "source task id ", src_id, &
695 "must be >= 0 and < ", total_tasks
696 call error_handler(e_err,'receive_from', errstring, source)
697endif
698
699itemcount = size(destarray,kind=i8)
700
701if (present(label)) then
702 write(*,*) trim(label)//" PE", myrank, ": receive_from itemsize ", itemcount, " src ", src_id
703else if (verbose) then
704 write(*,*) "PE", myrank, ": receive_from itemsize ", itemcount, " src ", src_id
705endif
706
707! send_to uses its own id as the tag.
708tag = src_id
709
710if (verbose) write(*,*) "PE", myrank, ": receive ", itemcount, " src ", src_id
711
712if (make_copy_before_sendrecv) allocate(tmpdata(min(itemcount,sndrcv_maxsize)))
713
714
715if (itemcount <= sndrcv_maxsize) then
716 if (.not. make_copy_before_sendrecv) then
717 call mpi_recv(destarray, int(itemcount,i4), datasize, src_id, mpi_any_tag, &
718 my_local_comm, status, errcode)
719 else
720 ! this copy should be unneeded, but on the intel fortran 9.0 compiler and mpich
721 ! on one platform, calling this subroutine with an array section resulted in some
722 ! apparently random memory corruption. making a copy of the data into a local,
723 ! contiguous buffer before send and receive fixed the corruption. this shouldn't
724 ! have been needed, and is a performance/memory sink.
725
726 call mpi_recv(tmpdata, int(itemcount,i4), datasize, src_id, mpi_any_tag, &
727 my_local_comm, status, errcode)
728 destarray = tmpdata
729 endif
730else
731 ! there are a few places in the code where we send/receive a full state vector.
732 ! as these get really large, they may exceed the limits of the MPI library.
733 ! break large messages up into smaller chunks if needed.
734 offset = 1
735 do while (offset <= itemcount)
736 if (itemcount-offset >= sndrcv_maxsize) then
737 nextsize = sndrcv_maxsize
738 else if (itemcount-offset == 0) then
739 nextsize = 1
740 else
741 nextsize = itemcount - offset
742 endif
743
744 if (verbose) write(*,*) 'recving array items ', offset, ' thru ' , offset + nextsize - 1
745
746 if (.not. make_copy_before_sendrecv) then
747 call mpi_recv(destarray(offset:offset+nextsize-1), int(nextsize,i4), datasize, src_id, mpi_any_tag, &
748 my_local_comm, status, errcode)
749 else
750 call mpi_recv(tmpdata, int(nextsize,i4), datasize, src_id, mpi_any_tag, &
751 my_local_comm, status, errcode)
752 destarray(offset:offset+nextsize-1) = tmpdata(1:nextsize)
753 endif
754 if (errcode /= mpi_success) then
755 write(errstring, '(a,i8)') 'MPI_Recv returned error code ', errcode
756 call error_handler(e_err,'receive_from', errstring, source)
757 endif
758 offset = offset + nextsize
759 end do
760endif
761
762if (errcode /= mpi_success) then
763 write(errstring, '(a,i8)') 'MPI_Recv returned error code ', errcode
764 call error_handler(e_err,'receive_from', errstring, source)
765endif
766
767if (verbose) write(*,*) "PE", myrank, ": received from ", src_id
768
769! if time specified, call MPI again to send the 2 time ints.
770if (present(time)) then
771 if (verbose) write(*,*) "PE", myrank, ": time present"
772 call mpi_recv(itime, 2, mpi_integer, src_id, tag*2, &
773 my_local_comm, status, errcode)
774 if (errcode /= mpi_success) then
775 write(errstring, '(a,i8)') 'MPI_Recv returned error code ', errcode
776 call error_handler(e_err,'receive_from', errstring, source)
777 endif
778 if (itime(2) < 0) then
779 write(errstring, '(a,i8)') 'seconds in settime were < 0; using 0'
780 call error_handler(e_msg,'receive_from', errstring, source)
781 time = set_time(itime(1), 0)
782 else
783 time = set_time(itime(1), itime(2))
784 endif
785 if (verbose) write(*,*) "PE", myrank, ": received time from ", src_id
786endif
787
788if (make_copy_before_sendrecv) deallocate(tmpdata)
789
790if (verbose) write(*,*) "PE", myrank, ": end of receive_from "
791
792
793end subroutine receive_from

796!-----------------------------------------------------------------------------
797! NOTE: so far we only seem to be sending real data, not integer, and 1D arrays.
798! this could be overloaded to send 2D arrays, and ints if needed.
799
800!> The data array values on the root task will be broadcast to every other
801!> task. When this routine returns, all tasks will have the contents of the
802!> root array in their own arrays. Thus 'array' is intent(in) on root, and
803!> intent(out) on all other tasks.
804
805subroutine array_broadcast(array, root, icount)
806 real(r8), intent(inout) :: array(:)
807 integer, intent(in) :: root
808 integer, intent(in), optional :: icount
809
810integer :: itemcount, errcode, offset, nextsize
811real(r8), allocatable :: tmpdata(:)
812
813if ( .not. module_initialized ) then
814 write(errstring, *) 'initialize_mpi_utilities() must be called first'
815 call error_handler(e_err,'array_broadcast', errstring, source)
816endif
817
818! simple idiotproofing
819if ((root < 0) .or. (root >= total_tasks)) then
820 write(errstring, '(a,i8,a,i8)') "root task id ", root, &
821 "must be >= 0 and < ", total_tasks
822 call error_handler(e_err,'array_broadcast', errstring, source)
823endif
824
825! if the actual data to be sent is shorter than the size of 'array',
826! there are performance advantages to sending only the actual data
827! and not the full array size. (performance tested on an ibm machine.)
828! calling code must determine if this is the case and pass in a length
829! shorter than the array size.
830if (present(icount)) then
831 if (icount > size(array)) then
832 write(errstring, '(a,i12)') "number of items to broadcast: ", icount
833 write(errstring1, '(a,i12)') "cannot be larger than the array size: ", size(array)
834 call error_handler(e_err,'array_broadcast', errstring, source, &
835 text2=errstring1)
836 endif
837 itemcount = icount
838else
839 itemcount = size(array)
840endif
841
842if (verbose .and. myrank == root) write(*,*) "PE", myrank, ": bcast itemsize from here ", itemcount
843
844! comment this in only if you really have to. it will flood the output with
845! messages from every task except one.
846!if (verbose .and. myrank /= root) write(*,*) "PE", myrank, ": bcast itemsize ", itemcount, " root ", root
847
848if (make_copy_before_broadcast) allocate(tmpdata(min(itemcount,bcast_maxsize)))
849
850! at least one user has run into a limit in the MPI implementation where you
851! cannot broadcast too large an array. if the size of this array is too large,
852! broadcast it in chunks until all the data has been processed
853if (itemcount <= bcast_maxsize) then
854 if (.not. make_copy_before_broadcast) then
855 call mpi_bcast(array, itemcount, datasize, root, my_local_comm, errcode)
856 else
857 if (my_task_id() == root) tmpdata = array
858 call mpi_bcast(tmpdata, itemcount, datasize, root, my_local_comm, errcode)
859 if (my_task_id() /= root) array = tmpdata
860 endif
861else
862 offset = 1
863 do while (offset <= itemcount)
864 if (itemcount-offset >= bcast_maxsize) then
865 nextsize = bcast_maxsize
866 else if (itemcount-offset == 0) then
867 nextsize = 1
868 else
869 nextsize = itemcount - offset
870 endif
871
872 if (verbose) write(*,*) 'bcasting array items ', offset, ' thru ' , offset + nextsize - 1
873
874 ! test this - are array sections going to cause array temps to be created?
875 if (.not. make_copy_before_broadcast) then
876 call mpi_bcast(array(offset:offset+nextsize-1), nextsize, datasize, root, my_local_comm, errcode)
877 else
878 if (my_task_id() == root) tmpdata = array(offset:offset+nextsize-1)
879 call mpi_bcast(tmpdata, nextsize, datasize, root, my_local_comm, errcode)
880 if (my_task_id() /= root) array(offset:offset+nextsize-1) = tmpdata
881 endif
882 if (errcode /= mpi_success) then
883 write(errstring, '(a,i8)') 'MPI_Bcast returned error code ', errcode
884 call error_handler(e_err,'array_broadcast', errstring, source)
885 endif
886 offset = offset + nextsize
887 end do
888endif
889
890if (errcode /= mpi_success) then
891 write(errstring, '(a,i8)') 'MPI_Bcast returned error code ', errcode
892 call error_handler(e_err,'array_broadcast', errstring, source)
893endif
894
895if (make_copy_before_broadcast) deallocate(tmpdata)
896
897end subroutine array_broadcast

1741!-----------------------------------------------------------------------------
1742! general system util wrappers.
1743!-----------------------------------------------------------------------------
1744
1745!> Use the system() command to execute a command string.
1746!> Will wait for the command to complete and returns an
1747!> error code unless you end the command with & to put
1748!> it into background. Function which returns the rc
1749!> of the command, 0 being all is ok.
1750!>
1751!> allow code to test the theory that maybe the system call is
1752!> not reentrant on some platforms. if serialize is set and
1753!> is true, do each call serially.
1754
1755function shell_execute(execute_string, serialize)
1756 character(len=*), intent(in) :: execute_string
1757 logical, intent(in), optional :: serialize
1758 integer :: shell_execute
1759
1760logical :: all_at_once
1761integer :: errcode, dummy(1)
1762type(mpi_status) :: status
1763
1764if (verbose) async2_verbose = .true.
1765
1766! default to everyone running concurrently, but if set and not true,
1767! serialize the calls to system() so they do not step on each other.
1768if (present(serialize)) then
1769 all_at_once = .not. serialize
1770else
1771 all_at_once = .true.
1772endif
1773
1774if (async2_verbose) write(*,*) "PE", myrank, ": system string is: ", trim(execute_string)
1775shell_execute = -1
1776
1777! this is the normal (default) case
1778if (all_at_once) then
1779
1780 ! all tasks call system at the same time
1781 call do_system(execute_string, shell_execute)
1782 if (async2_verbose) write(*,*) "PE", myrank, ": execution returns, rc = ", shell_execute
1783
1784 return
1785endif
1786
1787! only one task at a time calls system, and all wait their turn by
1788! making each task wait for a message from the (N-1)th task.
1789
1790! this is used only to signal; the value it contains is unused.
1791dummy = 0
1792
1793if (myrank == 0) then
1794
1795 ! my turn to execute
1796 call do_system(execute_string, shell_execute)
1797 if (async2_verbose) write(*,*) "PE", myrank, ": execution returns, rc = ", shell_execute
1798
1799 if (total_tasks > 1) then
1800 ! tell next task it can continue
1801 call mpi_send(dummy, 1, mpi_integer, 1, 1, my_local_comm, errcode)
1802 if (errcode /= mpi_success) then
1803 write(errstring, '(a,i8)') 'MPI_Send returned error code ', &
1804 errcode
1805 call error_handler(e_err,'shell_execute', errstring, source)
1806 endif
1807 endif
1808
1809else if (myrank /= (total_tasks-1)) then
1810 ! wait for (me-1) to tell me it is my turn
1811 call mpi_recv(dummy, 1, mpi_integer, myrank-1, myrank, &
1812 my_local_comm, status, errcode)
1813 if (errcode /= mpi_success) then
1814 write(errstring, '(a,i8)') 'MPI_Recv returned error code ', errcode
1815 call error_handler(e_err,'shell_execute', errstring, source)
1816 endif
1817
1818 ! my turn to execute
1819 call do_system(execute_string, shell_execute)
1820 if (async2_verbose) write(*,*) "PE", myrank, ": execution returns, rc = ", shell_execute
1821
1822 ! and now tell (me+1) to go
1823 call mpi_send(dummy, 1, mpi_integer, myrank+1, myrank+1, my_local_comm, errcode)
1824 if (errcode /= mpi_success) then
1825 write(errstring, '(a,i8)') 'MPI_Send returned error code ', &
1826 errcode
1827 call error_handler(e_err,'shell_execute', errstring, source)
1828 endif
1829else
1830 ! last task, no one else to send to.
1831 call mpi_recv(dummy, 1, mpi_integer, myrank-1, myrank, &
1832 my_local_comm, status, errcode)
1833 if (errcode /= mpi_success) then
1834 write(errstring, '(a,i8)') 'MPI_Recv returned error code ', errcode
1835 call error_handler(e_err,'shell_execute', errstring, source)
1836 endif
1837
1838 ! my turn to execute
1839 call do_system(execute_string, shell_execute)
1840 if (async2_verbose) write(*,*) "PE", myrank, ": execution returns, rc = ", shell_execute
1841
1842endif
1843
1844end function shell_execute
