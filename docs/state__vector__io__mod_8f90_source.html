7!> \defgroup state_vector_io_mod state_vector_io_mod
8!> @{ \brief Routines for reading and writing the model state.

10!> Read_state() and write_state() are the major routines in this module.

336!-----------------------------------------------------------------------
337!> Single state space inflation from file value is only index 1 of
338!> inflation vars array (mean and sd).
339!> This routine find the owner of the 1st element in the vars array
340!> The owner broadcasts the values of single state space inflation
341!> to all other tasks who then update their copies array.
342!> Note filling both mean and sd values if at least one of mean
343!> or sd is read from file. If one is set from a namelist the copies
344!> array is overwritten in fill_inf_from_namelist_value()
345
346!>@todo We need to refactor this, not sure where it is being used
347subroutine fill_single_inflate_val_from_read(ens_handle, &
348 prior_inflate_handle, post_inflate_handle)
349
350type(ensemble_type), intent(inout) :: ens_handle
351type(adaptive_inflate_type), intent(in) :: prior_inflate_handle
352type(adaptive_inflate_type), intent(in) :: post_inflate_handle
353
354integer :: owner, owners_index
355integer(i8) :: first_element
356real(r8), allocatable :: inf_array(:) ! 2 or 4 values
357integer :: inf_count
358logical :: return_me ! flag to return if not read any inflation values from files
359
360integer :: PRIOR_INF_MEAN
361integer :: PRIOR_INF_SD
362integer :: POST_INF_MEAN
363integer :: POST_INF_SD
364
365! Return if not doing single state space inflation
366if (.not. do_single_ss_inflate(prior_inflate_handle) .and. &
367 .not. do_single_ss_inflate(post_inflate_handle)) return
368
369return_me = .true.
370! Return if not reading any state space inflation values from files
371if ( do_single_ss_inflate(prior_inflate_handle)) then
372 if (mean_from_restart(prior_inflate_handle)) return_me = .false.
373 if (sd_from_restart(prior_inflate_handle)) return_me = .false.
374endif
375if ( do_single_ss_inflate(post_inflate_handle)) then
376 if (mean_from_restart(post_inflate_handle)) return_me = .false.
377 if (sd_from_restart(post_inflate_handle)) return_me = .false.
378endif
379if (return_me) return
380
381! At least some single state space inflation values are being read from files.
382inf_count = 0
383if (do_single_ss_inflate(prior_inflate_handle)) inf_count = 2
384if (do_single_ss_inflate(post_inflate_handle)) inf_count = inf_count + 2
385
386allocate(inf_array(inf_count)) ! for sending and recveiving inflation values
387
388! Find out who owns the first element of vars array
389first_element = 1
390call get_var_owner_index(ens_handle, first_element, owner, owners_index)
391
392prior_inf_mean = get_inflation_mean_copy(prior_inflate_handle)
393prior_inf_sd = get_inflation_sd_copy( prior_inflate_handle)
394post_inf_mean = get_inflation_mean_copy(post_inflate_handle)
395post_inf_sd = get_inflation_sd_copy( post_inflate_handle)
396
397if (ens_handle%my_pe == owner) then
398 if (do_single_ss_inflate(prior_inflate_handle) .and. &
399 do_single_ss_inflate(post_inflate_handle)) then
400 inf_array(1) = ens_handle%copies(prior_inf_mean, owners_index)
401 inf_array(2) = ens_handle%copies(prior_inf_sd, owners_index)
402 inf_array(3) = ens_handle%copies(post_inf_mean, owners_index)
403 inf_array(4) = ens_handle%copies(post_inf_sd, owners_index)
404
405 elseif (do_single_ss_inflate(post_inflate_handle) .and. &
406 .not. do_single_ss_inflate(post_inflate_handle)) then
407
408 inf_array(1) = ens_handle%copies(prior_inf_mean, owners_index)
409 inf_array(2) = ens_handle%copies(prior_inf_sd, owners_index)
410
411 elseif(.not. do_single_ss_inflate(post_inflate_handle) .and. &
412 do_single_ss_inflate(post_inflate_handle)) then
413
414 inf_array(1) = ens_handle%copies(post_inf_mean, owners_index)
415 inf_array(2) = ens_handle%copies(post_inf_sd, owners_index)
416
417 endif
418
419 call broadcast_send(map_pe_to_task(ens_handle, owner), inf_array)
420
421else
422
423 call broadcast_recv(map_pe_to_task(ens_handle, owner), inf_array)
424
425 if (do_single_ss_inflate(prior_inflate_handle) .and. &
426 do_single_ss_inflate(post_inflate_handle)) then
427
428 ens_handle%copies(prior_inf_mean, owners_index) = inf_array(1)
429 ens_handle%copies(prior_inf_sd, owners_index) = inf_array(2)
430 ens_handle%copies(post_inf_mean, owners_index) = inf_array(3)
431 ens_handle%copies(post_inf_sd, owners_index) = inf_array(4)
432
433 elseif (do_single_ss_inflate(prior_inflate_handle) .and. &
434 .not. do_single_ss_inflate(post_inflate_handle)) then
435
436 ens_handle%copies(prior_inf_mean, owners_index) = inf_array(1)
437 ens_handle%copies(prior_inf_sd, owners_index) = inf_array(2)
438
439 elseif(.not. do_single_ss_inflate(prior_inflate_handle) .and. &
440 do_single_ss_inflate(post_inflate_handle)) then
441
442 ens_handle%copies(post_inf_mean, owners_index) = inf_array(1)
443 ens_handle%copies(post_inf_sd, owners_index) = inf_array(2)
444
445 endif
446
447endif
448
449end subroutine fill_single_inflate_val_from_read
