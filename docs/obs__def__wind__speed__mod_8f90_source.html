103subroutine get_expected_windspeed(state_handle, ens_size, location, wspd, istatus)
104
105type(ensemble_type), intent(in) :: state_handle
106integer, intent(in) :: ens_size
107type(location_type), intent(in) :: location
108real(r8), intent(out) :: wspd(ens_size)
109integer, intent(out) :: istatus(ens_size)
110
111! Forward operator for windspeed. The argument list to this routine
112! must match the call in the GET_EXPECTED_OBS_FROM_DEF section above.
113
114real(r8) :: uwind(ens_size) ! zonal wind component
115real(r8) :: vwind(ens_size) ! meridional wind component
116integer :: this_istatus(ens_size)
117logical :: return_now
118
119if ( .not. module_initialized ) call initialize_module
120
121istatus = 0 ! to use track_status, it must start out 0
122
123! Zonal wind at this location - this calls the model_mod code.
124call interpolate(state_handle, ens_size, location, qty_u_wind_component, uwind, this_istatus)
125call track_status(ens_size, this_istatus, wspd, istatus, return_now)
126if (return_now) return
127
128! Meridional wind at this location - this calls the model_mod code.
129call interpolate(state_handle, ens_size, location, qty_v_wind_component, vwind, istatus)
130call track_status(ens_size, this_istatus, wspd, istatus, return_now)
131if (return_now) return
132
133! The actual forward operator computation. This is the value that
134! will be returned. istatus (the return code) of 0 is good,
135! any value > 0 indicates an error. (values < 0 reserved for
136! system use.)
137
138where (istatus == 0) wspd = sqrt(uwind**2 + vwind**2)
139
140end subroutine get_expected_windspeed
