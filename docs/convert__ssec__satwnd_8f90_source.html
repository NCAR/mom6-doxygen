19use types_mod, only : r8, missing_r8
20use utilities_mod, only : get_unit
21use time_manager_mod, only : time_type, set_calendar_type, gregorian, &
22 increment_time, get_time, set_date, operator(-)

25 static_init_obs_sequence, init_obs, write_obs_seq, &
26 append_obs_to_seq, init_obs_sequence, get_num_obs, &
27 set_copy_meta_data, set_qc_meta_data
28use meteor_mod, only : wind_dirspd_to_uv

33implicit none

35character(len=16), parameter :: ssec_sat_file = 'satwnd_input.txt'
36character(len=129), parameter :: sat_wind_file = 'obs_seq.satwnd'
37
38integer, parameter :: nmaxwnd = 50000, & ! maximum number of vectors
39 num_copies = 1, & ! number of copies in sequence
40 num_qc = 1 ! number of QC entries

42character(len=129) :: header
43character(len=8) :: datein
44character(len=6) :: sat
45character(len=4) :: band, hourin
46
47logical :: iruse, visuse, wvuse, swiruse, file_exist, qifile, eefile, &
48 userfqc, useqiqc, useeeqc, first_obs
49
50integer :: in_unit, i, oday, osec, nused, iyear, imonth, iday, ihour, &
51 imin, qctype
52real(r8) :: lat, lon, pres, wdir, wspd, uwnd, vwnd, oerr, latu(nmaxwnd), &
53 lonu(nmaxwnd), prsu(nmaxwnd), qc, qcthresh, rfqc, qiqc, eeqc
54
55type(obs_sequence_type) :: obs_seq
56type(obs_type) :: obs, prev_obs
57type(time_type) :: time_obs, prev_time
58
59print*,'Do you want to include IR, VISIBLE, WV data, and SW IR? (T/F, 4 times)'
60read*, iruse, visuse, wvuse, swiruse
61
62call set_calendar_type(gregorian)
63first_obs = .true.
64
65qifile = .false. ; eefile = .false.
66userfqc = .false. ; useqiqc = .false. ; useeeqc = .false.
67
68in_unit = get_unit()
69open(unit=in_unit, file = ssec_sat_file, status='old')
70read(in_unit,'(a100)') header
71
72if ( header(81:82) == 'ee' ) then
73
74 print*,'Enter the QC type (0=none, 1=rf, 2=qi, 3=ee) and threshhold'
75 read*, qctype, qcthresh
76
77 select case ( qctype )
78 case (1) ! Regression Factor QC
79 userfqc = .true.
80 case (2) ! QI format QC
81 useqiqc = .true.
82 case (3)
83 useeeqc = .true.
84 end select
85 eefile = .true.
86
87else if ( header(75:76) == 'qi' ) then
88
89 print*,'Enter the QC type (0=none, 1=rf, 2=qi) and threshhold'
90 read*, qctype, qcthresh
91
92 select case ( qctype )
93 case (1) ! Regression Factor QC
94 userfqc = .true.
95 case (2) ! QI format QC
96 useqiqc = .true.
97 end select
98 qifile = .true.
99
100end if
101
102! either read existing obs_seq or create a new one
103call static_init_obs_sequence()
104call init_obs(obs, num_copies, num_qc)
105inquire(file=sat_wind_file, exist=file_exist)
106if ( file_exist ) then
107
108 call read_obs_seq(sat_wind_file, 0, 0, 2*nmaxwnd, obs_seq)
109
110else
111
112 call init_obs_sequence(obs_seq, num_copies, num_qc, 2*nmaxwnd)
113 do i = 1, num_copies
114 call set_copy_meta_data(obs_seq, i, 'SSEC observation')
115 end do
116 do i = 1, num_qc
117 call set_qc_meta_data(obs_seq, i, 'Data QC')
118 end do
119
120end if
121
122nused = 0
123obsloop: do
124
125 if ( qifile ) then ! QI format file
126
127 read(in_unit,*,END=200) band, sat, datein, hourin, lat, lon, &
128 pres, wspd, wdir, rfqc, qiqc
129 if ( userfqc .and. rfqc > qcthresh ) cycle obsloop
130 if ( useqiqc .and. qiqc < qcthresh ) cycle obsloop
131
132 else if ( eefile ) then ! EE format file
133
134 read(in_unit,*,END=200) band, sat, datein, hourin, lat, lon, &
135 pres, wspd, wdir, rfqc, qiqc, eeqc
136 if ( userfqc .and. (rfqc > qcthresh) ) cycle obsloop
137 if ( useqiqc .and. (qiqc < qcthresh) ) cycle obsloop
138 if ( useeeqc .and. (eeqc > qcthresh) ) cycle obsloop
139
140 else ! file without QC information
141
142 read(in_unit,*,END=200) band, sat, datein, hourin, lat, lon, pres, wspd, wdir
143
144 end if
145
146 read(datein(1:4), fmt='(i4)') iyear
147 read(datein(5:6), fmt='(i2)') imonth
148 read(datein(7:8), fmt='(i2)') iday
149 read(hourin(1:2), fmt='(i2)') ihour
150 read(hourin(3:4), fmt='(i2)') imin
151 time_obs = set_date(iyear, imonth, iday, ihour, imin, 0)
152 call get_time(time_obs, osec, oday)
153
154 ! check the satellite channel
155 if ( trim(adjustl(band)) == 'IR' .and. (.not. iruse) ) cycle obsloop
156 if ( trim(adjustl(band)) == 'SWIR' .and. (.not. swiruse) ) cycle obsloop
157 if ( trim(adjustl(band)) == 'VIS' .and. (.not. visuse) ) cycle obsloop
158 if ( trim(adjustl(band(1:2))) == 'WV' .and. (.not. wvuse) ) cycle obsloop
159 lon = -lon ; if ( lon < 0.0_r8 ) lon = lon + 360.0_r8
160
161 ! check to make sure this observation has not been used before
162 do i = 1, nused
163 if ( lon == lonu(i) .and. lat == latu(i) .and. pres == prsu(i) ) cycle obsloop
164 end do
165 qc = 1.0_r8
166
167! if ( trim(adjustl(band(1:2))) == 'WV' ) then
168! oerr = sat_wv_wind_error(pres)
169! else
170 oerr = sat_wind_error(pres)
171! end if
172
173 ! perform sanity checks on observation errors and values
174 if ( oerr == missing_r8 .or. wdir < 0.0_r8 .or. wdir > 360.0_r8 .or. &
175 wspd < 0.0_r8 .or. wspd > 120.0_r8 ) cycle obsloop
176
177 call wind_dirspd_to_uv(wdir, wspd, uwnd, vwnd)
178
179 call create_3d_obs(lat, lon, pres * 100.0_r8, vertispressure, uwnd, &
180 sat_u_wind_component, oerr, oday, osec, qc, obs)
181 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
182
183 call create_3d_obs(lat, lon, pres * 100.0_r8, vertispressure, vwnd, &
184 sat_v_wind_component, oerr, oday, osec, qc, obs)
185 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
186
187 nused = nused + 1
188 latu(nused) = lat
189 lonu(nused) = lon
190 prsu(nused) = pres
191
192end do obsloop
193200 continue
194
195close( in_unit)
196
197if ( get_num_obs(obs_seq) > 0 ) call write_obs_seq(obs_seq, sat_wind_file)
198
199end program
