8
9!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
10!
11! convert_gpspw - program that reads ground-based GPS
12! precipitable water from Suominet sites
13! in netCDF and writes a DART
14! obs_seq file using the DART library routines.
15!
16! created Nov. 2014 Soyoung Ha, NCAR/MMM
17!
18!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
19
20use types_mod, only : r8
21use utilities_mod, only : initialize_utilities, finalize_utilities, &
22 find_namelist_in_file, check_namelist_read
23use netcdf_utilities_mod, only : nc_open_file_readonly, nc_close_file, &
24 nc_get_global_attribute
25use time_manager_mod, only : time_type, set_calendar_type, set_date, set_time, &
26 increment_time, get_time, get_date, operator(-), gregorian
27use location_mod, only : vertisundef
28use obs_sequence_mod, only : obs_sequence_type, obs_type, read_obs_seq, &
29 static_init_obs_sequence, init_obs, write_obs_seq, &
30 init_obs_sequence, get_num_obs, &
31 set_copy_meta_data, set_qc_meta_data
32use obs_kind_mod, only : gps_precipitable_water
33use obs_utilities_mod, only : getvar_real, get_or_fill_qc, add_obs_to_seq, &
34 create_3d_obs, getvar_int, getdimlen, getvar_real_2d, &
35 getvar_int_2d, query_varname, set_missing_name
36
37implicit none
38
39character(len=20), parameter :: gpspw_netcdf_file = 'gpspw_input.nc'
40character(len=129), parameter :: gpspw_out_file = 'obs_seq.gpspw'
41character(len=129) :: gpspw_outfile
42
43integer, parameter :: num_copies = 1, & ! number of copies in sequence
44 num_qc = 1 ! number of QC entries
45
46integer :: iunit, io
47integer :: ncid, nstn, n, i, oday, osec, nused, index, ntime, it
48logical :: file_exist, first_obs
49real(r8) :: qc
50real(r8) :: pwv_miss = -999.
51!real(r8) :: pwverr_miss = -999.
52
53!character(len=129), allocatable :: stationID(:)
54real(r8), allocatable :: lat(:), lon(:), elev(:), toff(:)
55!real(r8), allocatable :: latu(:), lonu(:), levu(:), tobu(:)
56real(r8), allocatable :: pwv(:,:), pwv_err(:,:)
57!real(r8), allocatable :: wdel(:), tdel(:), rain(:)
58!real(r8), allocatable :: wdir(:), wspd(:), rh(:), psfc(:), tsfc(:)
59real(r8) :: obs_window
60
61type(obs_sequence_type) :: obs_seq
62type(obs_type) :: obs, prev_obs
63type(time_type) :: comp_day0, time_obs, prev_time
64
65integer, parameter :: max_name = 256
66character(len=MAX_NAME) :: varname(5)
67
68! For the data resource, check Readme in data/ directory.
69! Sumoinet data provides observation times only as offset [sec] from 00Z
70! at each gregorian day. Thus, the actual observation date is read from
71! global attributes available in the file.
72! The Suominet data has two different data regions - either over CONUS or the
73! whole globe. We add the region name to the output file name (as well as
74! the observation time).
75! In this converter, we can process either hourly conus data or daily global
76! data depending on the global_data flag.
77! There are other surface variables available at the stations if one wants
78! to read from this dataset. For now, we only get pwv.
79
80integer :: max_num_obs = 1000000
81real(r8) :: obs_window_hr = 6 ! read data only every 6 hr from 00Z.
82logical :: global_data = .true.
83
84namelist /convert_gpspw_nml/ global_data, max_num_obs, obs_window_hr
85! include_wet_delay, include_total_delay, &
86! include_surface_pressure, include_relative_humidity, &
87! include_surface_temperature, include_rain_accumulation, &
88! include_wind_speed, include_wind_direction
89
90character(len=6) :: ftail
91character(len=19):: sdate
92character(len= 8):: ymd ! YYYYMMDD
93character(len=10):: ymdh ! YYYYMMDDHH
94integer :: iyear, iday, ihour
95integer :: iyr, imo, idy, ihr, imn, isc
96
97!------------
98! start of executable code
99!------------
100
101call initialize_utilities('convert_gpspw')
102
103call find_namelist_in_file("input.nml", "convert_gpspw_nml", iunit)
104read(iunit, nml = convert_gpspw_nml, iostat = io)
105call check_namelist_read(iunit, io, "mpas_obs_preproc_nml")
106
107
108first_obs = .true.
109
110ncid = nc_open_file_readonly(gpspw_netcdf_file, 'convert_gpspw')
111call nc_get_global_attribute(ncid, 'start_date', sdate)
112read(sdate,'(i4,1x,i3,1x,i2)') iyear, iday, ihour
113
114call set_calendar_type(gregorian)
115comp_day0 = set_date(iyear,1,1,0,0,0)
116call get_time(comp_day0, osec, oday)
117!print *,'DATE for ',iyear,' 01-01_00:00 => (oday, osec): ',oday, osec
118oday = oday + iday - 1
119comp_day0 = set_time(ihour*3600,oday)
120call get_date(comp_day0, iyr, imo, idy, ihr, imn, isc)
121
122! Final output file name
123if( global_data ) then
124 ftail = '.globe'
125 write(ymd,'(I4,2I2.2)') iyr, imo, idy
126 gpspw_outfile = trim(gpspw_out_file) // ftail // '.' // trim(ymd)
127 print *,'OBS_DATE: ',oday,' gregorian days which is ',ymd
128else
129 ftail = '.conus'
130 write(ymdh,'(I4,3I2.2)') iyr, imo, idy, ihr
131 gpspw_outfile = trim(gpspw_out_file) // ftail // '.' // trim(ymdh)
132 print *,'OBS_DATE: ',oday,' gregorian days which is ',ymdh
133endif
134print *,'Output file: ',trim(gpspw_outfile)
135
136
137call getdimlen(ncid, "station", nstn)
138call getdimlen(ncid, "time_offset", ntime)
139call set_missing_name("missing_value")
140
141obs_window = obs_window_hr * 3600.0_r8 ! obs frequency (from hours to seconds)
142
143allocate( lat(nstn))
144allocate( lon(nstn))
145allocate(elev(nstn))
146allocate(toff(ntime))
147allocate( pwv(ntime,nstn)) ; allocate(pwv_err(ntime,nstn)) ! for multiple obs times
148
149! Set the DART data quality control. Be consistent with NCEP codes;
150! 0 is 'must use', 1 is good, no reason not to use it.
151qc = 0.0_r8
152
153! read in the data arrays
154
155! we have gpspw data files which have different names for the
156! lat/lon/elev/obs arrays in the netcdf file. there doesn't seem
157! to be a global attr to say which one is in use, so for now try
158! both options.
159
160varname(1) = 'lat'
161call query_varname(ncid, 1, varname, index, force=.true.)
162call getvar_real(ncid, varname(index), lat ) ! station latitude
163
164varname(1) = 'lon'
165call query_varname(ncid, 1, varname, index, force=.true.)
166call getvar_real(ncid, varname(index), lon ) ! station longitude
167
168varname(1) = 'height'
169call query_varname(ncid, 1, varname, index, force=.true.)
170call getvar_real(ncid, varname(index), elev ) ! station elevation
171
172varname(1) = 'pwv'
173call query_varname(ncid, 1, varname, index, force=.true.)
174call getvar_real_2d(ncid, varname(index), pwv ) ! precipitable water
175
176varname(1) = 'pwv_err'
177call query_varname(ncid, 1, varname, index, force=.true.)
178call getvar_real_2d(ncid, varname(index), pwv_err ) ! pwv obs error
179
180varname(1) = 'time_offset' ! "PWV window midpoint time delta from start_time"
181call query_varname(ncid, 1, varname, index, force=.true.)
182call getvar_real(ncid, varname(index), toff ) ! obs time offset in seconds
183
184! if user says to use them, read in QCs if present
185
186! either read existing obs_seq or create a new one
187call static_init_obs_sequence()
188call init_obs(obs, num_copies, num_qc)
189call init_obs(prev_obs, num_copies, num_qc)
190
191inquire(file=gpspw_outfile, exist=file_exist)
192
193if ( file_exist ) then
194
195 ! existing file found, append to it
196 call read_obs_seq(gpspw_outfile, 0, 0, nstn, obs_seq)
197
198else
199
200 ! create a new one
201 call init_obs_sequence(obs_seq, num_copies, num_qc, max_num_obs)
202 do i = 1, num_copies
203 call set_copy_meta_data(obs_seq, i, 'Ground GPS PWV observation')
204 end do
205 do i = 1, num_qc
206 call set_qc_meta_data(obs_seq, i, 'Data QC')
207 end do
208
209endif
210
211
212timloop: do it = 1, ntime
213
214 if(obs_window_hr > 0) then
215 if(mod(toff(it),obs_window).ne.0) cycle timloop
216 endif
217
218 ! compute time of observation
219 time_obs = increment_time(comp_day0, nint(toff(it)))
220
221 ! extract actual time of observation in file into oday, osec.
222 call get_time(time_obs, osec, oday)
223 print*,'time_obs for it = ',it,oday, osec
224
225 nused = 0
226 stnloop: do n = 1, nstn
227
228 ! check the lat/lon values to see if they are ok
229 if ( lat(n) > 90.0_r8 .or. lat(n) < -90.0_r8 ) cycle stnloop
230 if ( lon(n) > 180.0_r8 .or. lon(n) < -180.0_r8 ) cycle stnloop
231
232 if ( lon(n) < 0.0_r8 ) lon(n) = lon(n) + 360.0_r8
233
234 ! add pwv data to obs_seq
235 if ( pwv(it,n) /= pwv_miss .and. pwv(it,n) > 0.0_r8 ) then
236
237 ! Ha: pwv observation error should be at least 1.5 mm.
238 if ( pwv_err(it,n) < 1.5 ) pwv_err(it,n) = 1.5
239
240 call create_3d_obs(lat(n), lon(n), elev(n), vertisundef, pwv(it,n), &
241 gps_precipitable_water, pwv_err(it,n), oday, osec, qc, obs)
242 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
243 nused = nused + 1
244
245 endif
246
247 end do stnloop
248 print*, 'nused =',nused
249
250end do timloop
251
252deallocate(lat)
253deallocate(lon)
254deallocate(elev)
255deallocate(toff)
256deallocate(pwv)
257deallocate(pwv_err)
258
259call nc_close_file(ncid, 'convert_gpspw')
260
261! if we added any obs to the sequence, write it now.
262if ( get_num_obs(obs_seq) > 0 ) call write_obs_seq(obs_seq, gpspw_outfile)
263
264! end of main program
265call finalize_utilities()
266
267
268end program
