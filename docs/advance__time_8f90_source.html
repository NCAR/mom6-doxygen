5!> Compute with time quantities
6!>
7!> The advance_time program computes the resulting time when either
8!> adding or subtracting time intervals. The increments can be
9!> expressed in days, hours, minutes or seconds. The output can be
10!> formatted as native WRF, CESM, Julian or Gregorian format.
11!>
12!> Reads input from standard input to be more portable, since older
13!> versions of iargc() weren't standardized.
14!>
15!> Based on the WRF da_advance_cymdh utility.
16!>
17!> All time computations call DART time manager.
18!>
19!> - has accuracy down to second,
20!> - can use day/hour/minute/second (with/without +/- sign) to advance time,
21!> - can digest various input date format if it still has the right order (ie. cc yy mm dd hh nn ss)
22!> - can digest flexible time increment
23!> - can output in wrf date format (ccyy-mm-dd_hh:nn:ss)
24!> - can specify output date format
25!> - can output Julian day
26!> - can output Gregorian days and seconds (since year 1601)
27!> - can output in CESM time format (ccyy-mm-dd-fffff where fffff is seconds of day)
28!>
29!> Examples
30!>
31!> - advance 12 h
32!>
33!> echo 20070730 12 | advance_time
34!>
35!> - go back 1 day 2 hours 30 minutes and 30 seconds
36!>
37!> echo 2007073012 -1d2h30m30s | advance_time
38!>
39!> - go back 3 hours 30 minutes less 1 second
40!>
41!> echo 2007073012 1s-3h30m | advance_time
42!>
43!> - advance 2 days and 1 second, output in wrf date format (three ways)
44!>
45!> echo 200707301200 2d1s -w | advance_time
46!>
47!> echo 2007-07-30_12:00:00 2d1s -w | advance_time
48!>
49!> echo 200707301200 2d1s -f ccyy-mm-dd_hh:nn:ss | advance_time
50!>
51!> - advance 120 h, and print year and Julian day
52!>
53!> echo 2007073006 120 -j | advance_time
54!>
55!> - advance 120 h, print year, Julian day, hour, minute and second
56!>
57!> echo 2007073006 120 -J | advance_time
58!>
59!> - print Gregorian day and second (since year 1601)
60!>
61!> echo 2007073006 0 -g | advance_time
62!>
63!> - print CESM format time (ccyy-mm-dd-fffff where fffff is sec of day)
64!>
65!> echo 2007073006 0 -c | advance_time
66!>
67!> @todo if run with no inputs ... it just hangs. Can we make it fail straight away?

255!-----------------------------------------------------------------------
256!> removes non-numeric characters from a date string
257!>
258!> @param[in] datein character string containing the date. May include
259!> dashes, colons, etc.
260!> @return character string with only numeric characters
261
262function parsedate(datein)
263character(len=*), intent(in) :: datein
264
265character(len=14) :: parsedate
266character(len=1 ) :: ch
267integer :: n, i
268
269parsedate = '00000000000000'
270
271i=0
272do n = 1, len_trim(datein)
273 ch = datein(n:n)
274 if (ch >= '0' .and. ch <= '9') then
275 i=i+1
276 parsedate(i:i)=ch
277 end if
278end do
279
280if (i == 13) then
281 parsedate(14:14) = ''
282 return ! CESM format
283else if (parsedate(11:14) == '0000') then
284 parsedate(11:14) = ''
285else if(parsedate(13:14) == '00') then
286 parsedate(13:14) = ''
287end if

289end function parsedate

291!-----------------------------------------------------------------------
292!> extracts the day,hour,minutes and seconds from the second input argument
293!>
294!> @param[in] dt character string with the temporal offset (the second input argument).
295!> @param[out] dday the day
296!> @param[out] dh the hour
297!> @param[out] dn the minute
298!> @param[out] ds the second
299
300subroutine parsedt(dt,dday,dh,dn,ds)
301character(len=*), intent(in) :: dt
302integer, intent(out) :: dday
303integer, intent(out) :: dh
304integer, intent(out) :: dn
305integer, intent(out) :: ds
306
307character(len=1 ) :: ch
308integer :: n,i,d,s,nounit
309
310! initialize time and sign
311nounit=1
312dday=0
313dh=0
314dn=0
315ds=0
316d=0
317s=1
318
319do n = 1, len_trim(dt)
320 ch = dt(n:n)
321 select case (ch)
322 case ('0':'9')
323 read(ch,fmt='(i1)') i
324 d=d*10+i
325 case ('-')
326 s=-1
327 case ('+')
328 s=1
329 case ('d')
330 nounit=0
331 dday=dday+d*s
332 d=0
333 case ('h')
334 nounit=0
335 dh=dh+d*s
336 d=0
337 case ('n','m')
338 nounit=0
339 dn=dn+d*s
340 d=0
341 case ('s')
342 nounit=0
343 ds=ds+d*s
344 d=0
345 case default
346 continue
347 end select
348end do
349
350if (nounit==1) dh=d*s
351
352end subroutine parsedt
