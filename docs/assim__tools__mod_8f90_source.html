208subroutine assim_tools_init()
209
210integer :: iunit, io, i, j
211integer :: num_special_cutoff, type_index
212logical :: cache_override = .false.
213
214
215! do this up front
216module_initialized = .true.
217
218! give these guys initial values at runtime *before* we read
219! in the namelist. this is to help detect how many items are
220! actually given in the namelist.
221special_localization_obs_types(:) = 'null'
222special_localization_cutoffs(:) = missing_r8
223
224! Read the namelist entry
225call find_namelist_in_file("input.nml", "assim_tools_nml", iunit)
226read(iunit, nml = assim_tools_nml, iostat = io)
227call check_namelist_read(iunit, io, "assim_tools_nml")
228
229! Write the namelist values to the log file
230if (do_nml_file()) write(nmlfileunit, nml=assim_tools_nml)
231if (do_nml_term()) write( * , nml=assim_tools_nml)
232
233! Forcing distributed_mean for single processor.
234! Note null_win_mod.f90 ignores distibute_mean.
235if (task_count() == 1) distribute_mean = .true.
236
237if(spread_restoration) then
238 write(msgstring, *) 'The spread_restoration option is not supported in this version of ', &
239 'DART. Contact the DAReS team if this option is needed '
240 call error_handler(e_err,'assim_tools_init:', msgstring, source)
241endif
242
243! allocate a list in all cases - even the ones where there is only
244! a single cutoff value. note that in spite of the name these
245! are specific types (e.g. RADIOSONDE_TEMPERATURE, AIRCRAFT_TEMPERATURE)
246! because that's what get_close() is passed. and because i've confused
247! myself several times -- we define generic kinds starting at 0, but
248! the specific types are autogenerated and always start at 1. so the
249! cutoff list is never (0:num_types); it is always (num_types).
250num_types = get_num_types_of_obs()
251allocate(cutoff_list(num_types))
252cutoff_list(:) = cutoff
253has_special_cutoffs = .false.
254
255! Go through special-treatment observation kinds, if any.
256num_special_cutoff = 0
257j = 0
258do i = 1, max_items
259 if(special_localization_obs_types(i) == 'null') exit
260 if(special_localization_cutoffs(i) == missing_r8) then
261 write(msgstring, *) 'cutoff value', i, ' is uninitialized.'
262 call error_handler(e_err,'assim_tools_init:', &
263 'special cutoff namelist for types and distances do not match', &
264 source, &
265 text2='kind = '//trim(special_localization_obs_types(i)), &
266 text3=trim(msgstring))
267 endif
268 j = j + 1
269enddo
270num_special_cutoff = j
271
272if (num_special_cutoff > 0) has_special_cutoffs = .true.
273
274do i = 1, num_special_cutoff
275 type_index = get_index_for_type_of_obs(special_localization_obs_types(i))
276 if (type_index < 0) then
277 write(msgstring, *) 'unrecognized TYPE_ in the special localization namelist:'
278 call error_handler(e_err,'assim_tools_init:', msgstring, source, &
279 text2=trim(special_localization_obs_types(i)))
280 endif
281 cutoff_list(type_index) = special_localization_cutoffs(i)
282end do
283
284! cannot cache previous obs location if different obs types have different
285! localization radii. change it to false, and warn user why.
286if (has_special_cutoffs .and. close_obs_caching) then
287 cache_override = .true.
288 close_obs_caching = .false.
289endif
290
291if(sampling_error_correction) then
292 sec_table_size = get_sampling_error_table_size()
293 allocate(exp_true_correl(sec_table_size), alpha(sec_table_size))
294 ! we can't read the table here because we don't have access to the ens_size
295endif
296
297is_doing_vertical_conversion = (has_vertical_choice() .and. vertical_localization_on())
298
299call log_namelist_selections(num_special_cutoff, cache_override)
300
301end subroutine assim_tools_init

530 obs_qc = obs_ens_handle%copies(obs_global_qc_copy, i)
531 ! Only do conversion of qc if forward operator is good
532 if(nint(obs_qc) == 0) then
533 ! Need to specify what kind of prior to use for each
534 call probit_dist_info(my_obs_kind(i), .false., .false., dist_for_obs, &
535 bounded_below, bounded_above, lower_bound, upper_bound)
536
537 ! Convert all my obs (extended state) variables to appropriate probit space
538 call transform_to_probit(ens_size, obs_ens_handle%copies(1:ens_size, i), dist_for_obs, &
539 obs_dist_params(i), probit_ens, .false., &
540 bounded_below, bounded_above, lower_bound, upper_bound)
541 obs_ens_handle%copies(1:ens_size, i) = probit_ens
542 endif
543end do
544
545! Initialize the method for getting state variables close to a given ob on my process
546if (has_special_cutoffs) then
547 call get_close_init(gc_state, my_num_state, 2.0_r8*cutoff, my_state_loc, 2.0_r8*cutoff_list)
548else
549 call get_close_init(gc_state, my_num_state, 2.0_r8*cutoff, my_state_loc)
550endif
551
552! Initialize the method for getting obs close to a given ob on my process
553if (has_special_cutoffs) then
554 call get_close_init(gc_obs, my_num_obs, 2.0_r8*cutoff, my_obs_loc, 2.0_r8*cutoff_list)
555else
556 call get_close_init(gc_obs, my_num_obs, 2.0_r8*cutoff, my_obs_loc)
557endif
558
559if (close_obs_caching) then
560 ! Initialize last obs and state get_close lookups, to take advantage below
561 ! of sequential observations at the same location (e.g. U,V, possibly T,Q)
562 ! (this is getting long enough it probably should go into a subroutine. nsc.)
563 last_base_obs_loc = set_location_missing()
564 last_base_states_loc = set_location_missing()
565 last_num_close_obs = -1
566 last_num_close_states = -1
567 num_close_obs_cached = 0
568 num_close_states_cached = 0
569 num_close_obs_calls_made = 0
570 num_close_states_calls_made = 0
571endif
572
573allow_missing_in_state = get_missing_ok_status()
574
575! Loop through all the (global) observations sequentially
576sequential_obs: do i = 1, obs_ens_handle%num_vars
577 ! Some compilers do not like mod by 0, so test first.
578 if (print_every_nth_obs > 0) nth_obs = mod(i, print_every_nth_obs)
579
580 ! If requested, print out a message every Nth observation
581 ! to indicate progress is being made and to allow estimates
582 ! of how long the assim will take.
583 if (nth_obs == 0) then
584 write(msgstring, '(2(A,I8))') 'Processing observation ', i, &
585 ' of ', obs_ens_handle%num_vars
586 if (print_timestamps == 0) then
587 call error_handler(e_msg,'filter_assim',msgstring)
588 else
589 call timestamp(trim(msgstring), pos="brief")
590 endif
591 endif
592
593 ! Every pe has information about the global obs sequence
594 call get_obs_from_key(obs_seq, keys(i), observation)
595 call get_obs_def(observation, obs_def)
596 base_obs_loc = get_obs_def_location(obs_def)
597 obs_err_var = get_obs_def_error_variance(obs_def)
598 base_obs_type = get_obs_def_type_of_obs(obs_def)
599 if (base_obs_type > 0) then
600 base_obs_kind = get_quantity_for_type_of_obs(base_obs_type)
601 else
602 call get_state_meta_data(-1 * int(base_obs_type,i8), dummyloc, base_obs_kind) ! identity obs
603 endif
604 ! Get the value of the observation
605 call get_obs_values(observation, obs, obs_val_index)
606
607 ! Find out who has this observation and where it is
608 call get_var_owner_index(ens_handle, int(i,i8), owner, owners_index)
609
610 ! Following block is done only by the owner of this observation
611 !-----------------------------------------------------------------------
612 if(ens_handle%my_pe == owner) then
613 ! each task has its own subset of all obs. if they were converted in the
614 ! vertical up above, then we need to broadcast the new values to all the other
615 ! tasks so they're computing the right distances when applying the increments.
616 if (is_doing_vertical_conversion) then
617 vertvalue_obs_in_localization_coord = query_location(my_obs_loc(owners_index), "VLOC")
618 whichvert_obs_in_localization_coord = query_location(my_obs_loc(owners_index), "WHICH_VERT")
619 else
620 vertvalue_obs_in_localization_coord = 0.0_r8
621 whichvert_obs_in_localization_coord = 0
622 endif
623
624 obs_qc = obs_ens_handle%copies(obs_global_qc_copy, owners_index)
625
626 ! Only value of 0 for DART QC field should be assimilated
627 if_qc_is_okay: if(nint(obs_qc) ==0) then
628 ! Note that these are before DA starts, so can be different from current obs_prior
629 orig_obs_prior_mean = obs_ens_handle%copies(obs_prior_mean_start: &
630 obs_prior_mean_end, owners_index)
631 orig_obs_prior_var = obs_ens_handle%copies(obs_prior_var_start: &
632 obs_prior_var_end, owners_index)
633
634 ! If QC is okay, convert this observation ensemble from probit to regular space
635 call transform_from_probit(ens_size, obs_ens_handle%copies(1:ens_size, owners_index) , &
636 obs_dist_params(owners_index), obs_ens_handle%copies(1:ens_size, owners_index))
637
638 obs_prior = obs_ens_handle%copies(1:ens_size, owners_index)
639 endif if_qc_is_okay
640
641 !Broadcast the info from this obs to all other processes
642 ! orig_obs_prior_mean and orig_obs_prior_var only used with adaptive inflation
643 ! my_inflate and my_inflate_sd only used with single state space inflation
644 ! vertvalue_obs_in_localization_coord and whichvert_real only used for vertical
645 ! coordinate transformation
646 whichvert_real = real(whichvert_obs_in_localization_coord, r8)
647 call broadcast_send(map_pe_to_task(ens_handle, owner), obs_prior, &
648 orig_obs_prior_mean, orig_obs_prior_var, &
649 scalar1=obs_qc, scalar2=vertvalue_obs_in_localization_coord, &
650 scalar3=whichvert_real, scalar4=my_inflate, scalar5=my_inflate_sd)
651
652 ! Next block is done by processes that do NOT own this observation
653 !-----------------------------------------------------------------------
654 else
655 call broadcast_recv(map_pe_to_task(ens_handle, owner), obs_prior, &
656 orig_obs_prior_mean, orig_obs_prior_var, &
657 scalar1=obs_qc, scalar2=vertvalue_obs_in_localization_coord, &
658 scalar3=whichvert_real, scalar4=my_inflate, scalar5=my_inflate_sd)
659 whichvert_obs_in_localization_coord = nint(whichvert_real)
660
661 endif
662 !-----------------------------------------------------------------------
663
664 ! Everybody is doing this section, cycle if qc is bad
665 if(nint(obs_qc) /= 0) cycle sequential_obs
666
667 !> all tasks must set the converted vertical values into the 'base' version of this loc
668 !> because that's what we pass into the get_close_xxx() routines below.
669 if (is_doing_vertical_conversion) &
670 call set_vertical(base_obs_loc, vertvalue_obs_in_localization_coord, whichvert_obs_in_localization_coord)
671
672 ! Compute observation space increments for each group
673 do group = 1, num_groups
674 grp_bot = grp_beg(group); grp_top = grp_end(group)
675 call obs_increment(obs_prior(grp_bot:grp_top), grp_size, obs(1), &
676 obs_err_var, base_obs_kind, obs_inc(grp_bot:grp_top), inflate, my_inflate, &
677 my_inflate_sd, net_a(group))
678 obs_post(grp_bot:grp_top) = obs_prior(grp_bot:grp_top) + obs_inc(grp_bot:grp_top)
679
680 ! Convert both the prior and posterior to probit space (efficiency for prior???)
681 ! Running probit space with groups needs to be studied more carefully
682 ! EFFICIENCY NOTE: FOR RHF, THE OBS_INCREMENT HAS TO DO A SORT
683 ! THE POSTERIOR WOULD HAVE THE SAME RANK STATISTICS, SO THIS SORT WOULD BE THE SAME
684 ! THE SECOND CONVERT_TO_PROBIT CAN BE MUCH MORE EFFICIENT USING A SORT
685 ! SHOULD FIGURE OUT A WAY TO PASS THE SORT ORDER
686 ! NOTE 2: THIS CONVERSION IS USING THE INFO FROM THE CURRENT (UPDATED) PRIOR ENSEMBLE. THIS
687 ! IS GENERALLY GOING TO BE A DIFFERENT PROBIT TRANSFORMED ENSEMBLE THAN THE ONE THAT WAS JUST
688 ! CONVERTED FROM PROBIT SPACE BY THE PROCESS THAT OWNS THIS OBSERVATION.
689
690 ! Need to specify what kind of prior to use for obs being assimilated
691 call probit_dist_info(base_obs_kind, .false., .false., dist_for_obs, &
692 bounded_below, bounded_above, lower_bound, upper_bound)
693
694 ! Convert the prior and posterior for this observation to probit space
695 call transform_to_probit(grp_size, obs_prior(grp_bot:grp_top), dist_for_obs, &
696 temp_dist_params, probit_obs_prior(grp_bot:grp_top), .false., &
697 bounded_below, bounded_above, lower_bound, upper_bound)
698 call transform_to_probit(grp_size, obs_post(grp_bot:grp_top), dist_for_obs, &
699 temp_dist_params, probit_obs_post(grp_bot:grp_top), .true., &
700 bounded_below, bounded_above, lower_bound, upper_bound)
701 ! Free up the storage used for this transform
702 call deallocate_distribution_params(temp_dist_params)
703
704 ! Copy back into original storage
705 obs_prior(grp_bot:grp_top) = probit_obs_prior(grp_bot:grp_top)
706 obs_post(grp_bot:grp_top) = probit_obs_post(grp_bot:grp_top)
707 ! Recompute obs_inc in probit space
708 obs_inc(grp_bot:grp_top) = obs_post(grp_bot:grp_top) - obs_prior(grp_bot:grp_top)
709
710 ! Also compute prior mean and variance of obs for efficiency here
711 obs_prior_mean(group) = sum(obs_prior(grp_bot:grp_top)) / grp_size
712 obs_prior_var(group) = sum((obs_prior(grp_bot:grp_top) - obs_prior_mean(group))**2) / &
713 (grp_size - 1)
714 if (obs_prior_var(group) < 0.0_r8) obs_prior_var(group) = 0.0_r8
715 end do
716
717 ! Compute updated values for single state space inflation
718 if(local_single_ss_inflate) then
719 ! Update for each group separately
720 do group = 1, num_groups
721 call update_single_state_space_inflation(inflate, my_inflate, my_inflate_sd, &
722 ens_handle%copies(ens_sd_copy, 1), orig_obs_prior_mean(group), &
723 orig_obs_prior_var(group), obs(1), obs_err_var, grp_size, inflate_only)
724 end do
725 endif
726
727 ! Adaptive localization needs number of other observations within localization radius.
728 ! Do get_close_obs first, even though state space increments are computed before obs increments.
729 call get_close_obs_cached(gc_obs, base_obs_loc, base_obs_type, &
730 my_obs_loc, my_obs_kind, my_obs_type, num_close_obs, close_obs_ind, close_obs_dist, &
731 ens_handle, last_base_obs_loc, last_num_close_obs, num_close_obs_cached, &
732 num_close_obs_calls_made)
733
734 ! set the cutoff default, keep a copy of the original value, and avoid
735 ! looking up the cutoff in a list if the incoming obs is an identity ob
736 ! (and therefore has a negative kind). specific types can never be 0;
737 ! generic kinds (not used here) start their numbering at 0 instead of 1.
738 if (base_obs_type > 0) then
739 cutoff_orig = cutoff_list(base_obs_type)
740 else
741 cutoff_orig = cutoff
742 endif
743
744 ! JLA, could also cache for adaptive_localization which may be expensive?
745 call adaptive_localization_and_diags(cutoff_orig, cutoff_rev, adaptive_localization_threshold, &
746 adaptive_cutoff_floor, num_close_obs, close_obs_ind, close_obs_dist, my_obs_type, &
747 i, base_obs_loc, obs_def, localization_unit)
748
749 ! Find state variables on my process that are close to observation being assimilated
750 call get_close_state_cached(gc_state, base_obs_loc, base_obs_type, &
751 my_state_loc, my_state_kind, my_state_indx, num_close_states, close_state_ind, close_state_dist, &
752 ens_handle, last_base_states_loc, last_num_close_states, num_close_states_cached, &
753 num_close_states_calls_made)
754 !call test_close_obs_dist(close_state_dist, num_close_states, i)
755
756 ! Loop through to update each of my state variables that is potentially close
757 state_update: do j = 1, num_close_states
758 state_index = close_state_ind(j)
759
760 if ( allow_missing_in_state ) then
761 ! Don't allow update of state ensemble with any missing values
762 if (any(ens_handle%copies(1:ens_size, state_index) == missing_r8)) cycle state_update
763 endif
764
765 ! Compute the covariance localization and adjust_obs_impact factors (module storage)
766 final_factor = cov_and_impact_factors(base_obs_loc, base_obs_type, my_state_loc(state_index), &
767 my_state_kind(state_index), close_state_dist(j), cutoff_rev)
768
769 if(final_factor <= 0.0_r8) cycle state_update
770
771 call obs_updates_ens(ens_size, num_groups, ens_handle%copies(1:ens_size, state_index), &
772 my_state_loc(state_index), my_state_kind(state_index), obs_prior, obs_inc, &
773 obs_prior_mean, obs_prior_var, base_obs_loc, base_obs_type, obs_time, &
774 net_a, grp_size, grp_beg, grp_end, i, &
775 my_state_indx(state_index), final_factor, correl, local_varying_ss_inflate, inflate_only)
776
777 ! Compute spatially-varying state space inflation
778 if(local_varying_ss_inflate) then
779 do group = 1, num_groups
780 call update_varying_state_space_inflation(inflate, &
781 ens_handle%copies(ens_inf_copy, state_index), &
782 ens_handle%copies(ens_inf_sd_copy, state_index), &
783 ens_handle%copies(ens_sd_copy, state_index), &
784 orig_obs_prior_mean(group), orig_obs_prior_var(group), obs(1), &
785 obs_err_var, grp_size, final_factor, correl(group), inflate_only)
786 end do
787 endif
788 end do state_update
789
790 if(.not. inflate_only) then
791 ! Now everybody updates their obs priors (only ones after this one)
792 obs_update: do j = 1, num_close_obs
793 obs_index = close_obs_ind(j)
794
795 ! Only have to update obs that have not yet been used
796 if(my_obs_indx(obs_index) > i) then
797
798 ! If forward observation operator failed, no need to update unassimilated observations
799 if (any(obs_ens_handle%copies(1:ens_size, obs_index) == missing_r8)) cycle obs_update
800
801 ! Compute the covariance localization and adjust_obs_impact factors (module storage)
802 final_factor = cov_and_impact_factors(base_obs_loc, base_obs_type, my_obs_loc(obs_index), &
803 my_obs_kind(obs_index), close_obs_dist(j), cutoff_rev)
804
805 if(final_factor <= 0.0_r8) cycle obs_update
806
807 call obs_updates_ens(ens_size, num_groups, obs_ens_handle%copies(1:ens_size, obs_index), &
808 my_obs_loc(obs_index), my_obs_kind(obs_index), obs_prior, obs_inc, &
809 obs_prior_mean, obs_prior_var, base_obs_loc, base_obs_type, obs_time, &
810 net_a, grp_size, grp_beg, grp_end, i, &
811 -1*my_obs_indx(obs_index), final_factor, correl, .false., inflate_only)
812 endif
813 end do obs_update
814 endif
815end do sequential_obs
816
817! Do the inverse probit transform for state variables
818call transform_all_from_probit(ens_size, ens_handle%my_num_vars, ens_handle%copies, &
819 state_dist_params, ens_handle%copies)
820
821! Every pe needs to get the current my_inflate and my_inflate_sd back
822if(local_single_ss_inflate) then
823 ens_handle%copies(ens_inf_copy, :) = my_inflate
824 ens_handle%copies(ens_inf_sd_copy, :) = my_inflate_sd
825end if
826
827! Free up the storage
828call destroy_obs(observation)
829call get_close_destroy(gc_state)
830call get_close_destroy(gc_obs)
831
832! do some stats - being aware that unless we do a reduce() operation
833! this is going to be per-task. so only print if something interesting
834! shows up in the stats? maybe it would be worth a reduce() call here?
835
836! Assure user we have done something
837if (print_trace_details >= 0) then
838 write(msgstring, '(A,I8,A)') 'Processed', obs_ens_handle%num_vars, ' total observations'
839 call error_handler(e_msg,'filter_assim:',msgstring)
840endif
841
842! diagnostics for stats on saving calls by remembering obs at the same location.
843! change .true. to .false. in the line below to remove the output completely.
844if (close_obs_caching) then
845 if (num_close_obs_cached > 0 .and. do_output()) then
846 print *, "Total number of calls made to get_close_obs for obs/states: ", &
847 num_close_obs_calls_made + num_close_states_calls_made
848 print *, "Total number of calls avoided to get_close_obs for obs/states: ", &
849 num_close_obs_cached + num_close_states_cached
850 if (num_close_obs_cached+num_close_obs_calls_made+ &
851 num_close_states_cached+num_close_states_calls_made > 0) then
852 print *, "Percent saved: ", 100.0_r8 * &
853 (real(num_close_obs_cached+num_close_states_cached, r8) / &
854 (num_close_obs_calls_made+num_close_obs_cached + &
855 num_close_states_calls_made+num_close_states_cached))
856 endif
857 endif
858endif
859
860!call test_state_copies(ens_handle, 'end')
861
862! Close the localization diagnostics file
863if(output_localization_diagnostics .and. my_task_id() == 0) call close_file(localization_unit)
864
865! get rid of mpi window
866call free_mean_window()
867
868! deallocate space
869deallocate(close_obs_dist, &
870 my_obs_indx, &
871 my_obs_kind, &
872 my_obs_type, &
873 close_obs_ind, &
874 vstatus, &
875 my_obs_loc)
876
877deallocate(close_state_dist, &
878 my_state_indx, &
879 close_state_ind, &
880 my_state_kind, &
881 my_state_loc)
882
883! end dealloc
884
885end subroutine filter_assim

889subroutine obs_increment(ens_in, ens_size, obs, obs_var, obs_kind, obs_inc, &
890 inflate, my_cov_inflate, my_cov_inflate_sd, net_a)
891
892! Given the ensemble prior for an observation, the observation, and
893! the observation error variance, computes increments and adjusts
894! observation space inflation values
895
896integer, intent(in) :: ens_size
897real(r8), intent(in) :: ens_in(ens_size), obs, obs_var
898integer, intent(in) :: obs_kind
899real(r8), intent(out) :: obs_inc(ens_size)
900type(adaptive_inflate_type), intent(inout) :: inflate
901real(r8), intent(inout) :: my_cov_inflate, my_cov_inflate_sd
902real(r8), intent(out) :: net_a
903
904real(r8) :: ens(ens_size), inflate_inc(ens_size)
905real(r8) :: prior_mean, prior_var, new_val(ens_size)
906integer :: i, ens_index(ens_size), new_index(ens_size)
907
908real(r8) :: rel_weights(ens_size)
909
910integer :: filter_kind
911logical :: bounded_below, bounded_above
912real(r8) :: lower_bound, upper_bound
913
914! Declarations for bounded rank histogram filter
915real(r8) :: likelihood(ens_size), like_sum
916
917! Copy the input ensemble to something that can be modified
918ens = ens_in
919
920! Null value of net spread change factor is 1.0
921net_a = 0.0_r8
922
923! Compute prior variance and mean from sample
924prior_mean = sum(ens) / ens_size
925prior_var = sum((ens - prior_mean)**2) / (ens_size - 1)
926
927! If observation space inflation is being done, compute the initial
928! increments and update the inflation factor and its standard deviation
929! as needed. my_cov_inflate < 0 means don't do any of this.
930if(do_obs_inflate(inflate)) then
931 ! If my_cov_inflate_sd is <= 0, just retain current my_cov_inflate setting
932 if(my_cov_inflate_sd > 0.0_r8) &
933 ! Gamma set to 1.0 because no distance for observation space
934 call update_inflation(inflate, my_cov_inflate, my_cov_inflate_sd, prior_mean, &
935 prior_var, ens_size, obs, obs_var, gamma_corr = 1.0_r8)
936
937 ! Now inflate the ensemble and compute a preliminary inflation increment
938 call inflate_ens(inflate, ens, prior_mean, my_cov_inflate, prior_var)
939 ! Keep the increment due to inflation alone
940 inflate_inc = ens - ens_in
941
942 ! Need to recompute variance if non-deterministic inflation (mean is unchanged)
943 if(.not. deterministic_inflate(inflate)) &
944 prior_var = sum((ens - prior_mean)**2) / (ens_size - 1)
945endif
946
947! Gets information about the increment method (filter_kind, bounds) for the current observation.
948call obs_inc_info(obs_kind, filter_kind, bounded_below, bounded_above, &
949 lower_bound, upper_bound)
950
951! The first three options in the next if block of code may be inappropriate for
952! some more general filters; need to revisit
953! If obs_var == 0, delta function. The mean becomes obs value with no spread.
954! If prior_var == 0, obs has no effect. The increments are 0.
955! If both obs_var and prior_var == 0 there is no right thing to do, so Stop.
956if ((obs_var == 0.0_r8) .and. (prior_var == 0.0_r8)) then
957
958 ! fail if both obs variance and prior spreads are 0.
959 write(msgstring, *) 'Observation value is ', obs, ' ensemble mean value is ', prior_mean
960 write(msgstring2, *) 'The observation has 0.0 error variance, and the ensemble members have 0.0 spread.'
961 write(msgstring3, *) 'These require inconsistent actions and the algorithm cannot continue.'
962 call error_handler(e_err, 'obs_increment', msgstring, &
963 source, text2=msgstring2, text3=msgstring3)
964
965else if (obs_var == 0.0_r8) then
966
967 ! new mean is obs value, so increments are differences between obs
968 ! value and current value. after applying obs, all state will equal obs.
969 obs_inc(:) = obs - ens
970
971else if (prior_var == 0.0_r8) then
972
973 ! if all state values are the same, nothing changes.
974 obs_inc(:) = 0.0_r8
975
976else
977
978 ! Call the appropriate filter option to compute increments for ensemble
979 ! note that at this point we've taken care of the cases where either the
980 ! obs_var or the prior_var is 0, so the individual routines no longer need
981 ! to have code to test for those cases.
982 if(filter_kind == eakf) then
983 call obs_increment_eakf(ens, ens_size, prior_mean, prior_var, &
984 obs, obs_var, obs_inc, net_a)
985 else if(filter_kind == enkf) then
986 call obs_increment_enkf(ens, ens_size, prior_var, obs, obs_var, obs_inc)
987 else if(filter_kind == kernel) then
988 call obs_increment_kernel(ens, ens_size, obs, obs_var, obs_inc)
989 else if(filter_kind == obs_particle) then
990 call obs_increment_particle(ens, ens_size, obs, obs_var, obs_inc)
991 else if(filter_kind == unbounded_rhf) then
992 call obs_increment_rank_histogram(ens, ens_size, prior_var, obs, obs_var, obs_inc)
993 else if(filter_kind == gamma_filter) then
994 call obs_increment_gamma(ens, ens_size, prior_mean, prior_var, obs, obs_var, obs_inc)
995 !--------------------------------------------------------------------------
996 else if(filter_kind == bounded_normal_rhf) then
997
998 ! Use bounded normal likelihood; Could use an arbitrary likelihood
999 do i = 1, ens_size
1000 likelihood(i) = get_truncated_normal_like(ens(i), obs, obs_var, &
1001 bounded_below, bounded_above, lower_bound, upper_bound)
1002 end do
1003
1004 ! Normalize the likelihood here
1005 like_sum = sum(likelihood)
1006 ! If likelihood underflow, assume flat likelihood, so no increments
1007 if(like_sum <= 0.0_r8) then
1008 obs_inc = 0.0_r8
1009 return
1010 else
1011 likelihood = likelihood / like_sum
1012 endif
1013
1014 call obs_increment_bounded_norm_rhf(ens, likelihood, ens_size, prior_var, &
1015 obs_inc, bounded_below, bounded_above, lower_bound, upper_bound)
1016
1017 ! Do test of inversion for an uninformative likelihood
1018 !!!t_likelihood = 1.0
1019 !!!t_likelihood = t_likelihood / sum(t_likelihood)

1571 mean_factor = (1.0_r8 - alpha(sec_table_size)) * fract + alpha(sec_table_size)
1572else
1573 ! given the ifs above, the floor() computation below for low_indx
1574 ! should always result in a value in the range 1 to 199. but if this
1575 ! code is compiled with r8=r4 (single precision reals) it turns out
1576 ! to be possible to get values a few bits below 0 which results in
1577 ! a very large negative integer. the limit tests below ensure the
1578 ! index stays in a legal range.
1579 low_indx = floor((scorrel + 0.995_r8) / 0.01_r8 + 1.0_r8)
1580 if (low_indx < 1) low_indx = 1
1581 if (low_indx > 199) low_indx = 199
1582 low_correl = -0.995_r8 + (low_indx - 1) * 0.01_r8
1583 low_exp_correl = exp_true_correl(low_indx)
1584 low_alpha = alpha(low_indx)
1585 high_indx = low_indx + 1
1586 high_correl = low_correl + 0.01_r8
1587 high_exp_correl = exp_true_correl(high_indx)
1588 high_alpha = alpha(high_indx)
1589 fract = (scorrel - low_correl) / (high_correl - low_correl)
1590 correl = (high_exp_correl - low_exp_correl) * fract + low_exp_correl
1591 mean_factor = (high_alpha - low_alpha) * fract + low_alpha
1592endif
1593
1594expected_true_correl = correl
1595
1596! Don't want Monte Carlo interpolation problems to put us outside of a
1597! ratio between 0 and 1 for expected_true_correl / sample_correl
1598! If they have different signs, expected should just be 0
1599if(expected_true_correl * scorrel <= 0.0_r8) then
1600 expected_true_correl = 0.0_r8
1601else if(abs(expected_true_correl) > abs(scorrel)) then
1602 ! If same sign, expected should not be bigger in absolute value
1603 expected_true_correl = scorrel
1604endif
1605
1606end subroutine get_correction_from_table

1609subroutine obs_increment_rank_histogram(ens, ens_size, prior_var, &
1610 obs, obs_var, obs_inc)
1611!------------------------------------------------------------------------
1612!
1613! Does observation space update by approximating the prior distribution by
1614! a rank histogram. Prior and posterior are assumed to have 1/(n+1) probability
1615! mass between each ensemble member. The tails are assumed to be gaussian with
1616! a variance equal to sample variance of the entire ensemble and a mean
1617! selected so that 1/(n+1) of the mass is in each tail.
1618!
1619! The likelihood between the extreme ensemble members is approximated by
1620! quadrature. Two options are available and controlled by the namelist entry
1621! rectangular_quadrature. If this namelist is true than the likelihood between
1622! a pair of ensemble members is assumed to be uniform with the average of
1623! the likelihood computed at the two ensemble members. If it is false then
1624! the likelihood between two ensemble members is approximated by a line
1625! connecting the values of the likelihood computed at each of the ensemble
1626! members (trapezoidal quadrature).
1627!
1628! Two options are available for approximating the likelihood on the tails.
1629! If gaussian_likelihood_tails is true that the likelihood is assumed to
1630! be N(obs, obs_var) on the tails. If this is false, then the likelihood
1631! on the tails is taken to be uniform (to infinity) with the value at the
1632! outermost ensemble members.
1633!
1634! A product of the approximate prior and approximate posterior is taken
1635! and new ensemble members are located so that 1/(n+1) of the mass is between
1636! each member and on the tails.
1637
1638integer, intent(in) :: ens_size
1639real(r8), intent(in) :: ens(ens_size), prior_var, obs, obs_var
1640real(r8), intent(out) :: obs_inc(ens_size)
1641
1642integer :: i, e_ind(ens_size), lowest_box, j
1643real(r8) :: prior_sd, var_ratio, umass, left_amp, right_amp, norm_const
1644real(r8) :: left_mean, right_mean
1645real(r8) :: mass(ens_size + 1), like(ens_size), cumul_mass(0:ens_size + 1)
1646real(r8) :: nmass(ens_size + 1)
1647real(r8) :: new_mean_left, new_mean_right, prod_weight_left, prod_weight_right
1648real(r8) :: new_var_left, new_var_right, new_sd_left, new_sd_right
1649real(r8) :: new_ens(ens_size), mass_sum
1650real(r8) :: x(ens_size)
1651real(r8) :: like_dense(2:ens_size), height(2:ens_size)
1652real(r8) :: dist_for_unit_sd
1653real(r8) :: a, b, c, hright, hleft, r1, r2, adj_r1, adj_r2
1654
1655! Do an index sort of the ensemble members; Will want to do this very efficiently
1656call index_sort(ens, e_ind, ens_size)
1657x = ens(e_ind)
1658
1659! Define normal PDF constant term
1660norm_const = 1.0_r8 / sqrt(2.0_r8 * pi * obs_var)
1661! Compute likelihood for each ensemble member; just evaluate the gaussian
1662do i = 1, ens_size
1663 like(i) = norm_const * exp(-1.0_r8 * (x(i) - obs)**2 / (2.0_r8 * obs_var))
1664end do
1665
1666! Compute approx likelihood each interior bin (average of bounding likelihoods)
1667do i = 2, ens_size
1668 like_dense(i) = ((like(i - 1) + like(i)) / 2.0_r8)
1669end do
1670
1671! Compute the s.d. of the ensemble for getting the gaussian tails
1672prior_sd = sqrt(prior_var)
1673
1674! For unit normal, find distance from mean to where cdf is 1/(n+1)
1675! Lots of this can be done once in first call and then saved
1676dist_for_unit_sd = inv_weighted_normal_cdf(1.0_r8, 0.0_r8, 1.0_r8, &
1677 1.0_r8 / (ens_size + 1.0_r8))
1678dist_for_unit_sd = -1.0_r8 * dist_for_unit_sd
1679
1680! Have variance of tails just be sample prior variance
1681! Mean is adjusted so that 1/(n+1) is outside
1682left_mean = x(1) + dist_for_unit_sd * prior_sd
1683! Same for right tail
1684right_mean = x(ens_size) - dist_for_unit_sd * prior_sd
1685
1686if(gaussian_likelihood_tails) then
1687 !*************** Block to do Gaussian-Gaussian on tail **************
1688 ! Compute the product of the obs likelihood gaussian with the priors
1689 ! Left tail gaussian first
1690 var_ratio = obs_var / (prior_var + obs_var)
1691 new_var_left = var_ratio * prior_var
1692 new_sd_left = sqrt(new_var_left)
1693 new_mean_left = var_ratio * (left_mean + prior_var*obs / obs_var)
1694 ! REMEMBER, this product has an associated weight which must be taken into account
1695 ! See Anderson and Anderson for this weight term (or tutorial kernel filter)
1696 prod_weight_left = exp(-0.5_r8 * (left_mean**2 / prior_var + &
1697 obs**2 / obs_var - new_mean_left**2 / new_var_left)) / &
1698 sqrt(prior_var + obs_var) / sqrt(2.0_r8 * pi)
1699 ! Determine how much mass is in the updated tails by computing gaussian cdf
1700 mass(1) = normal_cdf(x(1), new_mean_left, new_sd_left) * prod_weight_left
1701
1702 ! Same for the right tail
1703 var_ratio = obs_var / (prior_var + obs_var)
1704 new_var_right = var_ratio * prior_var
1705 new_sd_right = sqrt(new_var_right)
1706 new_mean_right = var_ratio * (right_mean + prior_var*obs / obs_var)
1707 prod_weight_right = exp(-0.5_r8 * (right_mean**2 / prior_var + &
1708 obs**2 / obs_var - new_mean_right**2 / new_var_right)) / &
1709 sqrt(prior_var + obs_var) / sqrt(2.0_r8 * pi)
1710 ! Determine how much mass is in the updated tails by computing gaussian cdf
1711 mass(ens_size + 1) = (1.0_r8 - normal_cdf(x(ens_size), new_mean_right, new_sd_right)) * &
1712 prod_weight_right
1713
1714 !************ End Block to do Gaussian-Gaussian on tail **************
1715else
1716 !*************** Block to do flat tail for likelihood ****************
1717 ! Flat tails: THIS REMOVES ASSUMPTIONS ABOUT LIKELIHOOD AND CUTS COST
1718 new_sd_left = prior_sd
1719 new_mean_left = left_mean
1720 prod_weight_left = like(1)
1721 mass(1) = like(1) / (ens_size + 1.0_r8)
1722
1723 ! Same for right tail
1724 new_sd_right = prior_sd
1725 new_mean_right = right_mean
1726 prod_weight_right = like(ens_size)
1727 mass(ens_size + 1) = like(ens_size) / (ens_size + 1.0_r8)
1728 !*************** End block to do flat tail for likelihood ****************
1729endif
1730
1731! The mass in each interior box is the height times the width
1732! The height of the likelihood is like_dense
1733! For the prior, mass is 1/(n+1), and mass = height x width so...
1734! The height of the prior is 1 / ((n+1) width); multiplying by width leaves 1/(n+1)
1735
1736! In prior, have 1/(n+1) mass in each bin, multiply by mean likelihood density
1737! to get approximate mass in updated bin
1738do i = 2, ens_size
1739 mass(i) = like_dense(i) / (ens_size + 1.0_r8)
1740 ! Height of prior in this bin is mass/width; Only needed for trapezoidal
1741 ! If two ensemble members are the same, set height to -1 as flag
1742 if(x(i) == x(i - 1)) then
1743 height(i) = -1.0_r8
1744 else
1745 height(i) = 1.0_r8 / ((ens_size + 1.0_r8) * (x(i) - x(i-1)))
1746 endif
1747end do
1748
1749! Now normalize the mass in the different bins to get a pdf
1750mass_sum = sum(mass)
1751nmass = mass / mass_sum
1752
1753! Get the weight for the final normalized tail gaussians
1754! This is the same as left_amp=(ens_size + 1)*nmass(1)
1755left_amp = prod_weight_left / mass_sum
1756! This is the same as right_amp=(ens_size + 1)*nmass(ens_size + 1)
1757right_amp = prod_weight_right / mass_sum
1758
1759! Find cumulative mass at each box boundary and middle boundary
1760cumul_mass(0) = 0.0_r8
1761do i = 1, ens_size + 1
1762 cumul_mass(i) = cumul_mass(i - 1) + nmass(i)
1763end do
1764
1765! Begin intenal box search at bottom of lowest box, update for efficiency
1766lowest_box = 1
1767
1768! Find each new ensemble members location
1769do i = 1, ens_size
1770 ! Each update ensemble member has 1/(n+1) mass before it
1771 umass = (1.0_r8 * i) / (ens_size + 1.0_r8)
1772
1773 ! If it is in the inner or outer range have to use normal
1774 if(umass < cumul_mass(1)) then
1775 ! It's in the left tail
1776 ! Get position of x in weighted gaussian where the cdf has value umass
1777 new_ens(i) = inv_weighted_normal_cdf(left_amp, new_mean_left, new_sd_left, &
1778 umass)
1779 else if(umass > cumul_mass(ens_size)) then
1780 ! It's in the right tail
1781 ! Get position of x in weighted gaussian where the cdf has value umass
1782 new_ens(i) = inv_weighted_normal_cdf(right_amp, new_mean_right, new_sd_right, &
1783 1.0_r8 - umass)
1784 ! Coming in from the right, use symmetry after pretending its on left
1785 new_ens(i) = new_mean_right + (new_mean_right - new_ens(i))
1786 else
1787 ! In one of the inner uniform boxes.
1788 find_box:do j = lowest_box, ens_size - 1
1789 ! Find the box that this mass is in
1790 if(umass >= cumul_mass(j) .and. umass <= cumul_mass(j + 1)) then
1791
1792 if(rectangular_quadrature) then
1793 !********* Block for rectangular quadrature *******************
1794 ! Linearly interpolate in mass
1795 new_ens(i) = x(j) + ((umass - cumul_mass(j)) / &
1796 (cumul_mass(j+1) - cumul_mass(j))) * (x(j + 1) - x(j))
1797 !********* End block for rectangular quadrature *******************
1798
1799 else
1800
1801 !********* Block for trapezoidal interpolation *******************
1802 ! Assume that mass has linear profile, quadratic interpolation
1803 ! If two ensemble members are the same, just keep that value
1804 if(height(j + 1) < 0) then
1805 new_ens(i) = x(j)
1806 else
1807 ! Height on left side and right side
1808 hleft = height(j + 1) * like(j) / mass_sum
1809 hright = height(j + 1) * like(j + 1) / mass_sum
1810 ! Will solve a quadratic for desired x-x(j)
1811 ! a is 0.5(hright - hleft) / (x(j+1) - x(j))
1812 a = 0.5_r8 * (hright - hleft) / (x(j+1) - x(j))
1813 ! b is hleft
1814 b = hleft
1815 ! c is cumul_mass(j) - umass
1816 c = cumul_mass(j) - umass
1817 ! Use stable quadratic solver
1818 call solve_quadratic(a, b, c, r1, r2)
1819 adj_r1 = r1 + x(j)
1820 adj_r2 = r2 + x(j)
1821 if(adj_r1 >= x(j) .and. adj_r1 <= x(j+1)) then
1822 new_ens(i) = adj_r1
1823 elseif (adj_r2 >= x(j) .and. adj_r2 <= x(j+1)) then
1824 new_ens(i) = adj_r2
1825 else
1826 msgstring = 'Did not get a satisfactory quadratic root'
1827 call error_handler(e_err, 'obs_increment_rank_histogram', msgstring, &
1828 source)
1829 endif
1830 endif
1831 !********* End block for quadratic interpolation *******************
1832
1833 endif
1834
1835 ! Don't need to search lower boxes again
1836 lowest_box = j
1837 exit find_box
1838 end if
1839 end do find_box
1840 endif
1841end do
1842
1843! Convert to increments for unsorted
1844do i = 1, ens_size
1845 obs_inc(e_ind(i)) = new_ens(i) - x(i)
1846end do
1847
1848end subroutine obs_increment_rank_histogram

1853subroutine update_ens_from_weights(ens, ens_size, rel_weight, ens_inc)
1854!------------------------------------------------------------------------
1855! Given relative weights for an ensemble, compute increments for the
1856! ensemble members. Assumes that prior distributon is equal uniform mass
1857! between each ensemble member. On the edges, have a normal with the
1858! sample mean and s.d. BUT normalized by a factor alpha so that only
1859! 1/(2*ens_size) of the total mass lies on each flank.
1860
1861integer, intent(in) :: ens_size
1862real(r8), intent(in) :: ens(ens_size), rel_weight(ens_size)
1863real(r8), intent(out) :: ens_inc(ens_size)
1864
1865integer :: i, j, lowest_box
1866integer :: e_ind(ens_size)
1867real(r8) :: x(1:2*ens_size - 1), cumul_mass(1:2*ens_size - 1), new_ens(ens_size)
1868real(r8) :: sort_inc(ens_size), updated_mass(2 * ens_size)
1869real(r8) :: sx, prior_mean, prior_var, prior_sd, mass
1870real(r8) :: total_mass_left, total_mass_right, alpha(2)
1871
1872! Initialize assim_tools_module if needed
1873if (.not. module_initialized) call assim_tools_init()
1874
1875call error_handler(e_err,'update_ens_from_weight','Routine needs testing.', &
1876 source, text2='Talk to Jeff before using.')
1877
1878! Do an index sort of the ensemble members
1879call index_sort(ens, e_ind, ens_size)
1880
1881! Have half boxes between all ensembles in the interior
1882! Total number of mass boxes is 2*ens_size
1883
1884! Compute the points that bound all the updated mass boxes; start with ensemble
1885do i = 1, ens_size
1886 x(2*i - 1) = ens(e_ind(i))
1887end do
1888! Compute the mid-point interior boundaries; these are halfway between ensembles
1889do i = 2, 2*ens_size - 2, 2
1890 x(i) = (x(i - 1) + x(i + 1)) / 2.0_r8
1891end do
1892
1893! Compute the mean and s.d. of the prior ensemble to handle wings
1894sx = sum(ens)
1895prior_mean = sx / ens_size
1896prior_var = sum((ens - prior_mean)**2) / (ens_size - 1)
1897prior_sd = sqrt(prior_var)
1898
1899! Need to normalize the wings so they have 1/(2*ens_size) mass outside
1900! Use cdf to find out how much mass is left of 1st member, right of last
1901total_mass_left = normal_cdf(ens(e_ind(1)), prior_mean, prior_sd)
1902total_mass_right = 1.0_r8 - normal_cdf(ens(e_ind(ens_size)), prior_mean, prior_sd)
1903
1904! Find the mass in each division given the initial equal partition and the weights
1905updated_mass(1) = rel_weight(e_ind(1)) / (2.0_r8 * ens_size)
1906updated_mass(2 * ens_size) = rel_weight(e_ind(ens_size)) / (2.0_r8 * ens_size)
1907do i = 2, 2*ens_size - 2, 2
1908 updated_mass(i) = rel_weight(e_ind(i / 2)) / (2.0_r8 * ens_size)
1909end do
1910do i = 3, 2*ens_size - 1, 2
1911 updated_mass(i) = rel_weight(e_ind((i+1) / 2)) / (2.0_r8 * ens_size)
1912end do
1913
1914! Normalize the mass; (COULD IT EVER BE 0 necessitating error check?)
1915updated_mass = updated_mass / sum(updated_mass)
1916
1917! Find a normalization factor to get tail mass right
1918if(total_mass_left > 0.0_r8) then
1919 alpha(1) = updated_mass(1) / total_mass_left
1920else
1921 alpha(1) = 0.0_r8
1922endif
1923if(total_mass_right > 0.0_r8) then
1924 alpha(2) = updated_mass(2 * ens_size) / total_mass_right
1925else
1926 alpha(2) = 0.0_r8
1927endif
1928
1929! Find cumulative mass at each box boundary and middle boundary
1930cumul_mass(1) = updated_mass(1)
1931do i = 2, 2*ens_size - 1
1932 cumul_mass(i) = cumul_mass(i - 1) + updated_mass(i)
1933end do
1934
1935! Get resampled position an inefficient way
1936! Need 1/ens_size between each EXCEPT for outers which get half of this
1937mass = 1.0_r8 / (2.0_r8 * ens_size)
1938
1939do i = 1, ens_size
1940 ! If it's in the inner or outer range have to use normal
1941 if(mass < cumul_mass(1)) then
1942 ! In the first normal box
1943 new_ens(i) = inv_weighted_normal_cdf(alpha(1), prior_mean, prior_sd, mass)
1944 else if(mass > cumul_mass(2*ens_size - 1)) then
1945 ! In the last normal box; Come in from the outside
1946 new_ens(i) = inv_weighted_normal_cdf(alpha(2), prior_mean, prior_sd, 1.0_r8 - mass)
1947 new_ens(i) = prior_mean + (prior_mean - new_ens(i))
1948 else
1949 ! In one of the inner uniform boxes. Make this much more efficient search?
1950 lowest_box = 1
1951 find_box:do j = lowest_box, 2 * ens_size - 2
1952 ! Find the box that this mass is in
1953 if(mass >= cumul_mass(j) .and. mass <= cumul_mass(j + 1)) then
1954 new_ens(i) = x(j) + ((mass - cumul_mass(j)) / (cumul_mass(j+1) - cumul_mass(j))) * &
1955 (x(j + 1) - x(j))
1956 ! Don't need to search lower boxes again
1957 lowest_box = j
1958 exit find_box
1959 end if
1960 end do find_box
1961 endif
1962 ! Want equally partitioned mass in update with exception that outermost boxes have half
1963 mass = mass + 1.0_r8 / ens_size
1964end do
1965
1966! Can now compute sorted increments
1967do i = 1, ens_size
1968 sort_inc(i) = new_ens(i) - ens(e_ind(i))
1969end do
1970
1971! Now, need to convert to increments for unsorted
1972do i = 1, ens_size
1973 ens_inc(e_ind(i)) = sort_inc(i)
1974end do
1975
1976end subroutine update_ens_from_weights

1980 obs_prior, obs_inc, obs_prior_mean, obs_prior_var, obs_loc, obs_type, obs_time, &
1981 net_a, grp_size, grp_beg, grp_end, reg_factor_obs_index, &

1983

1988integer, intent(in) :: ens_kind

2006

2017 obs_prior_var(group), obs_inc(grp_bot:grp_top), ens(grp_bot:grp_top), grp_size, &

2028 reg_factor_obs_index, reg_factor_ens_index)

2030endif

2033if(.not. inflate_only) ens = ens + final_factor * increment
