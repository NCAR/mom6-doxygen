338function compute_diff(ens_mems, ens_mean)
339real(r8), intent(in) :: ens_mems(num_copies,my_num_vars)
340real(r8), intent(in) :: ens_mean(my_num_vars)
341real(r8) :: compute_diff(ens_size)
342
343real(r8), allocatable :: local_diffs(:), adiff(:), global_diff(:)
344
345allocate(adiff(my_num_vars))
346allocate(local_diffs(ens_size), global_diff(ens_size))
347
348do copies = 1, ens_size
349 select case (difference_method)
350
351 ! simple absolute difference
352 case (1)
353 where(useindex)
354 adiff(:) = abs(ens_mean(:) - ens_mems(copies,:))
355 endwhere
356
357 local_diffs(copies) = sum(adiff(:))
358
359 ! normalized absolute difference
360 case (2)
361
362 where (useindex)
363 where (ens_mean(:) /= 0.0_r8)
364 adiff(:) = abs((ens_mean(:) - ens_mems(copies,:))/ens_mean(:))
365 elsewhere
366 adiff(:) = abs(ens_mems(copies,:))
367 endwhere
368 endwhere
369
370 local_diffs(copies) = sum(adiff(:))
371
372 ! simple rms difference
373 case (3)
374
375 where (useindex) adiff(:) = ens_mean(:) - ens_mems(copies, :)
376
377 local_diffs(copies) = sum(adiff * adiff)
378
379 ! normalized rms difference
380 case (4)
381
382 where (useindex)
383 where (ens_mean(:) /= 0.0_r8)
384 adiff = (ens_mean(:) - ens_mems(copies,:)) / ens_mean(:)
385 elsewhere
386 adiff = ens_mems(copies,:)
387 endwhere
388 endwhere
389
390 local_diffs(copies) = sum(adiff * adiff)
391
392 case default
393 write(msgstring, *) 'Valid values for difference_method are 1-4, value is', difference_method
394 call error_handler(e_err,'closest_member_tool','Bad value for difference_method', &
395 source, text2=msgstring)
396 end select
397enddo
398
399call send_sum_to(local_diffs, 0, global_diff)
400
401if(my_task_id() == 0) then
402 ! need to square total difference after local sums are computed
403 if (difference_method == 3 .or. difference_method == 4) then
404 where(global_diff > 0) global_diff = sqrt(global_diff)
405 endif
406endif
407
408compute_diff = global_diff
409
410deallocate(adiff, global_diff, local_diffs)
411
412end function compute_diff
413
414end program closest_member_tool
