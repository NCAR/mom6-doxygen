118subroutine create_output_filename(l2bname, ofname)
119!-------------------------------------------------
120! The L2b filenames have a very long extension that
121! records when the data was published - not very interesting
122! for our purposes. replace with something DART-y.
123character(len=*), intent(IN) :: l2bname
124character(len=*), intent(OUT) :: ofname
125
126integer :: i, strlen, extstart
127
128strlen = len_trim(l2bname)
129
130dotloop : do i = strlen,1,-1
131 if (l2bname(i:i) == '.' ) then
132 extstart = i
133 exit dotloop
134 endif
135enddo dotloop
136
137ofname = l2bname(1:extstart)//'obs_seq_out'

139end subroutine create_output_filename

143function real_obs_sequence ( orbit, lon1, lon2, lat1, lat2, &
144 row_thin, col_thin )
145!------------------------------------------------------------------------------
146! this function is to prepare data to DART sequence format
147!
148type(orbit_type), intent(in) :: orbit
149real(r8), intent(in) :: lon1, lon2, lat1, lat2
150integer, intent(in) :: row_thin, col_thin
151
152integer :: max_num=max_wvc*max_rows*2
153
154type(obs_sequence_type) :: real_obs_sequence
155type(obs_def_type) :: obs_def
156type(obs_type) :: obs, prev_obs
157
158integer :: i, irow, iwvc, iamb
159integer, PARAMETER :: num_copies = 1
160!integer, PARAMETER :: num_qc = 3
161integer, PARAMETER :: num_qc = 1
162integer :: days, seconds
163integer :: obs_num
164integer :: which_vert, uobstype, vobstype
165
166real(r4) :: speed, dir
167real(r8) :: lon, lat, vloc
168real(r8) :: u_obs, v_obs, u_var, v_var
169real(r8) :: aqc(num_qc)
170real(r8) :: sintheta, costheta, dirvar, speedvar
171
172type(time_type) :: time, pre_time
173
174if ( .not. module_initialized ) call initialize_module
175
176! Initialize an obs_sequence
177call init_obs_sequence(real_obs_sequence, num_copies, num_qc, max_num)
178
179! set meta data of obs_seq
180do i = 1, num_copies
181 call set_copy_meta_data(real_obs_sequence, i, 'observation')
182end do
183
184call set_qc_meta_data(real_obs_sequence, 1, 'QC flag - wvc_quality_flag')
185!call set_qc_meta_data(real_obs_sequence, 2, 'IMUDH - mp_rain_probability')
186!call set_qc_meta_data(real_obs_sequence, 3, 'NOF Index - nof_rain_index')
187
188! Initialize the obs variables
189call init_obs( obs, num_copies, num_qc)
190call init_obs(prev_obs, num_copies, num_qc)
191
192! assign each observation the correct observation type
193uobstype = get_index_for_type_of_obs('QKSWND_U_WIND_COMPONENT')
194vobstype = get_index_for_type_of_obs('QKSWND_V_WIND_COMPONENT')
195if(( uobstype < 1) .or. ( vobstype < 1)) then
196 msgstring = 'unknown observation type [QKSWND_U_WIND_COMPONENT] ... dying ...'
197 call error_handler(e_err,'real_obs_sequence',msgstring,source,revision,revdate)
198endif
199
200! loop over all observations within the file
201!------------------------------------------------------------------------------
202
203obs_num = 0
204vloc = 10 ! all L2b data is 10m height
205which_vert = vertisheight
206
207!rowloop: do irow=400,403
208rowloop: do irow=1,max_rows
209
210 ! if we're going to subset rows, cycle here
211 if (row_thin > 0) then
212 if (modulo(irow, row_thin) /= 1) cycle rowloop
213 endif
214
215 time = orbit%row_time(irow)
216 call get_time(time, seconds, days)
217
218 wvcloop: do iwvc=1,max_wvc
219
220 ! if we're going to subset columns, cycle here
221 if (col_thin > 0) then
222 if (modulo(iwvc, col_thin) /= 1) cycle wvcloop
223 endif
224
225 ! no ambiguities means no retrieval
226
227 if (orbit%num_ambigs(iwvc,irow) < 1) cycle wvcloop
228
229 ! only use data flagged as 'best'
230 ! this should be namelist controlled at some point
231 !
232 if ( orbit%wvc_quality_flag(iwvc,irow) /= 0 ) cycle wvcloop
233
234 aqc(1) = orbit%wvc_quality_flag( iwvc,irow)
235! aqc(2) = orbit%mp_rain_probability(iwvc,irow)
236! aqc(3) = orbit%nof_rain_index( iwvc,irow)
237
238 lat = orbit%wvc_lat(iwvc,irow) ! valid range [-90.00, 90.00]
239 lon = orbit%wvc_lon(iwvc,irow) ! valid range [ 0.00, 359.99]
240
241 ! verify the location is not outside valid limits
242 if((lon > 360.0_r8) .or. (lon < 0.0_r8) .or. &
243 (lat > 90.0_r8) .or. (lat < -90.0_r8)) then
244 write(*,*)'WARNING : invalid location. wvc,row,lon,lat = ', iwvc,irow,lon,lat
245 cycle wvcloop
246 endif
247
248 ! reject observations outside the bounding box (allowing wrapping)
249 if(( lat < lat1) .or. ( lat > lat2 ) .or. &
250 (.not. is_longitude_between(lon, lon1, lon2))) cycle wvcloop
251
252 ! QuikSCAT uses the oceanographic/flow convention ...
253 ! 0.0 is TOWARD the north - in direct contradiction to
254 ! atmospheric convention. Convert by adding 180 modulo 360
255
256 iamb = orbit%wvc_selection( iwvc,irow)
257
258 ! using the 'selected wind' ... aka highest-ranked ambiguity,
259
260! speed = orbit%wind_speed(iamb,iwvc,irow)
261! dir = mod(orbit%wind_dir( iamb,iwvc,irow)+180.0_r4, 360.0_r4)
262
263 ! Use this block for the DIRTH selection
264
265 speed = orbit%wind_speed_selection(iwvc,irow)
266 dir = mod(orbit%wind_dir_selection( iwvc,irow) + 180.0_r4, 360.0_r4)
267
268 if ( speed < 1.0_r4 ) cycle wvcloop ! everything unreliable
269
270 ! The requirements for QuikSCAT were 2 m/s speed (rms) over 3-20m/s
271 ! or 10% of the speed from 20-30 m/s and 20 degrees (rms) 3-30 m/s
272
273 dirvar = (20.0_r4*deg2rad)**2 ! 20 degree (rms) by spec
274 speedvar = max(2.0_r4, speed*0.1_r4)**2
275
276 sintheta = sin(dir*deg2rad)
277 costheta = cos(dir*deg2rad)
278
279 ! converting the speed and direction and uncertainties to U,V
280 ! is a bit tedious for the uncertainties. Really could/should
281 ! have a native type for speed and direction and a forward
282 ! operator that takes U,V and calculates speed/dir ...(trivial)
283 ! left for a future upgrade TJH
284
285 ! U = R*sin(theta)
286 ! V = R*cos(theta)
287 ! theta ~ theta0 + etheta ... etheta ~ (0,dirvar)
288 ! R ~ R0 + espeed ... espeed ~ (0,speedvar)
289 ! U ~ sin(theta0+etheta)(R0 + espeed)
290 ! expand via Taylor ...
291 ! U ~ sin(theta0)R0 + R0*cos(theta0)*etheta + sin(theta0)*espeed + ...
292 ! The first term is the mean.
293 ! The variance of a constant times a random variable is the
294 ! constant-squared times the variance of the random variable.
295 ! The random variables in the higher order terms is etheta and espeed.
296
297 u_obs = speed * sintheta
298 v_obs = speed * costheta
299
300 u_var = ((speed*costheta)**2)*dirvar + (sintheta**2)*speedvar
301 v_var = ((speed*sintheta)**2)*dirvar + (costheta**2)*speedvar
302
303 ! verify the location is not outside valid limits
304 if((lon > 360.0_r8) .or. (lon < 0.0_r8) .or. &
305 (lat > 90.0_r8) .or. (lat < -90.0_r8)) then
306 write(*,*) 'invalid location. lon,lat = ', lon, lat
307 cycle wvcloop
308 endif
309
310 obs_num = obs_num + 1
311
312 ! create the obs_def for this observation, add to sequence
313 !---------------------------------------------------------------
314
315 call real_obs(num_copies, num_qc, obs, lon, lat, vloc, u_obs, &
316 u_var, aqc, uobstype, which_vert, seconds, days)
317
318 if(obs_num == 1) then ! for the first observation
319 call insert_obs_in_seq(real_obs_sequence, obs)
320 call copy_obs(prev_obs, obs)
321 pre_time = time
322 else ! not the first observation
323 if(time == pre_time) then ! same time as previous observation
324 call insert_obs_in_seq(real_obs_sequence, obs, prev_obs)
325 call copy_obs(prev_obs, obs)
326 else ! not the same time
327 call insert_obs_in_seq(real_obs_sequence, obs)
328 call copy_obs(prev_obs, obs)
329 pre_time = time
330 endif
331 endif
332
333 ! The (paired) V observation must be at the same time,
334 ! and cannot be first, so it is easier.
335 call real_obs(num_copies, num_qc, obs, lon, lat, vloc, v_obs, &
336 v_var, aqc, vobstype, which_vert, seconds, days)
337 call insert_obs_in_seq(real_obs_sequence, obs, prev_obs)
338 call copy_obs(prev_obs, obs)
339
340 obs_num = obs_num + 1
341
342 if ( debug ) then
343 write(*,*)irow, iwvc, aqc, lat, lon, &
344 sqrt(u_obs**2+v_obs**2), atan(u_obs/v_obs)*rad2deg
345 endif
346
347 enddo wvcloop
348enddo rowloop
349
350! Print a little summary
351print*, 'obs used = ', obs_num, ' obs skipped = ', 2*max_wvc*max_rows - obs_num
352
353if ( get_first_obs(real_obs_sequence, obs) ) then
354 call get_obs_def(obs, obs_def)
355 pre_time = get_obs_def_time(obs_def)
356 call print_time(pre_time,' first time in sequence is ')
357 call print_date(pre_time,' first date in sequence is ')
358endif
359if( get_last_obs(real_obs_sequence, obs)) then
360 call get_obs_def(obs, obs_def)
361 time = get_obs_def_time(obs_def)
362 call print_time(time,' last time in sequence is ')
363 call print_date(time,' last date in sequence is ')
364endif
365print*, ''
366
367end function real_obs_sequence

427subroutine read_qscat2b(l2b_file, orbit)
428!====================================================================
429!
430! Description:
431!
432! This file contains 3 subroutines in order to read the
433! QuikSCAT Level 2B data in Hierarchical Data Format (HDF).
434! The subroutines are as follows.
435!
436! 1. read_attrib_byname(): a subroutine to read the name
437! and value(s) of a global attribute
438! referenced by its name.
439!
440! 2. read_timetags(): a subroutine to read the timetag info
441! contained in the HDF VDATA
442!
443! 3. extract_sds(): a subroutine to read the contents of an
444! SDS from an HDF file
445!
446! NOTES:
447! 1. Please refer all questions concerning this program and
448! QuikSCAT data obtained from the JPL PO.DAAC to
449! qscat@podaac.jpl.nasa.gov.
450!
451! 2. The HDF library must be installed before this program will
452! work properly. The HDF library and further information
453! about HDF may be obtained from the National Center for
454! Supercomputing Applications (NCSA) at http://hdf.ncsa.uiuc.edu.
455!
456! 3. The L2B data are read in their entirety. Examples of reading
457! the QuikSCAT data by slabs can be found in read_qscat1b.f
458! read_qscat2a.f.
459!
460!====================================================================
461
462character(len=128), intent(in) :: l2b_file
463type(orbit_type), intent(out) :: orbit
464
465! Set Parameters
466
467integer :: dfacc_rdonly
468parameter(dfacc_rdonly = 1)
469
470! Define Variables
471
472! character(len=1) :: product(8)
473character(len=21) :: timetags(max_rows)
474
475real(r4), dimension(MAX_WVC,MAX_ROWS) :: datmat
476
477integer :: sd_id, retn, sfstart, sfend
478integer :: irow, iwvc, iamb
479! integer :: ntype, nval
480
481integer :: year, doy, hh, mm
482real(r4) :: ss
483type(time_type) :: basetime, offset
484
485if ( .not. module_initialized ) call initialize_module
486
487call error_handler(e_msg,'read_qscat2b','FILENAME: '//trim(l2b_file))
488
489if ( .not. file_exist(l2b_file)) then
490 call error_handler(e_err,'read_qscat2b', &
491 'QuikSCAT L2B file does not exist.', source, revision, revdate)
492endif
493
494! Open the HDF input file and initiate the SD interface
495sd_id = sfstart(l2b_file,dfacc_rdonly)
496
497! Make sure that the file is a QuikSCAT Level 2B file
498! I've never been able to make this work with F90 ... crazy games are afoot.
499!----------------------------------------------------------------------
500! call read_attrib_byname(sd_id,'ShortName',ntype,nval,product)
501! write(*,*) product
502! write(*,*) product .ne. 'QSCATL2B'
503!if ( product .ne. 'QSCATL2B') then
504! print *,'The input file is not a QuikSCAT Level 2B file'
505! print *,'*** Aborting program ***'
506! stop
507!endif
508
509! Read the timetag info contained in the HDF VDATA
510call read_timetags(l2b_file, timetags)
511
512! Read each SDS in its entirety. For an example of reading
513! the QuikSCAT SDS data in slabs, please refer to read_qscat2a.f.
514
515irow=1
516call extract_sds(sd_id,'wvc_row', irow, max_rows, orbit%wvc_row)
517call extract_sds(sd_id,'wvc_lat', irow, max_rows, orbit%wvc_lat)
518call extract_sds(sd_id,'wvc_lon', irow, max_rows, orbit%wvc_lon)
519call extract_sds(sd_id,'atten_corr', irow, max_rows, orbit%atten_corr)
520call extract_sds(sd_id,'model_speed', irow, max_rows, orbit%model_speed)
521call extract_sds(sd_id,'model_dir', irow, max_rows, orbit%model_dir)
522call extract_sds(sd_id,'wind_speed', irow, max_rows, orbit%wind_speed)
523call extract_sds(sd_id,'wind_dir', irow, max_rows, orbit%wind_dir)
524call extract_sds(sd_id,'wind_speed_err', irow, max_rows, orbit%wind_speed_err)
525call extract_sds(sd_id,'wind_dir_err', irow, max_rows, orbit%wind_dir_err)
526call extract_sds(sd_id,'max_likelihood_est', irow, max_rows, orbit%max_likelihood_est)
527call extract_sds(sd_id,'wind_speed_selection',irow, max_rows, orbit%wind_speed_selection)
528call extract_sds(sd_id,'wind_dir_selection', irow, max_rows, orbit%wind_dir_selection)
529call extract_sds(sd_id,'mp_rain_probability', irow, max_rows, orbit%mp_rain_probability)
530call extract_sds(sd_id,'srad_rain_rate', irow, max_rows, orbit%srad_rain_rate)
531call extract_sds(sd_id,'wvc_index', irow, max_rows, datmat)
532 orbit%wvc_index = nint(datmat)
533call extract_sds(sd_id,'num_in_fore', irow, max_rows, datmat)
534 orbit%num_in_fore = nint(datmat)
535call extract_sds(sd_id,'num_in_aft', irow, max_rows, datmat)
536 orbit%num_in_aft = nint(datmat)
537call extract_sds(sd_id,'num_out_fore', irow, max_rows, datmat)
538 orbit%num_out_fore = nint(datmat)
539call extract_sds(sd_id,'num_out_aft', irow, max_rows, datmat)
540 orbit%num_out_aft = nint(datmat)
541call extract_sds(sd_id,'wvc_quality_flag', irow, max_rows, datmat)
542 orbit%wvc_quality_flag = int(datmat)
543call extract_sds(sd_id,'num_ambigs', irow, max_rows, datmat)
544 orbit%num_ambigs = nint(datmat)
545call extract_sds(sd_id,'wvc_selection', irow, max_rows, datmat)
546 orbit%wvc_selection = nint(datmat)
547call extract_sds(sd_id,'nof_rain_index', irow, max_rows, datmat)
548 orbit%nof_rain_index = nint(datmat)
549
550! Convert time tag to a dart timetype.
551!00000000011111111112
552!12345678901234567890
553!2008-001T01:16:24.584
554
555do irow = 1,max_rows
556 read(timetags(irow),'( i4)')year
557 read(timetags(irow),'( 5x, i3)')doy
558 read(timetags(irow),'( 9x, i2)')hh
559 read(timetags(irow),'(12x, i2)')mm

747!====================================================================
748! EXTRACT_SDS: a subroutine to read the contents of an
749! SDS from an HDF file
750!
751! 5/12/1998 R.S. Dunbar
752!
753! Revisions:
754! 7/1999 Code adapted to read input in bytes as well as ints
755! and floats. Comments were also added. K.L. Perry
756!
757! 3/2000 Corrected code for 8-bit unsigned integers. "buffer"
758! was used instead of "buffer2". K.L. Perry
759!
760! 5/2000 Changed MAX_BUF_SIZE from 1000000 to 10000000.
761! Corrected code for 32-bit unsigned integers. Created
762! "buffer3" array of int*4 to correctly read in uint32.
763! K.L. Perry
764!
765!====================================================================
766 subroutine extract_sds(sd_id,in_var,irec,slab_size,out_var)
767
768 integer :: MAX_BUF_SIZE
769 parameter(max_buf_size=10000000)
770 integer :: sd_id,sds_index,sds_id,retn
771 integer :: rank,dim_sizes(3),data_type,nattrs,num_type
772 integer :: edge(3),stride(3),start(3),irec,slab_size
773 real(digits12) :: cal, cal_err, off, off_err
774 integer :: iprod,i,itmp
775
776 character*(*) in_var
777 character(len=256) :: name
778 integer :: sfn2index,sfselect,sfginfo,sfrdata,sfgcal,sfendacc
779
780 integer*2 :: buffer(MAX_BUF_SIZE)
781 byte buffer2(max_buf_size)
782 integer*4 :: buffer3(MAX_BUF_SIZE)
783 real(r4) :: out_var(*)
784
785 if ( .not. module_initialized ) call initialize_module
786
787! Search for the index of "in_var"
788 sds_index = sfn2index(sd_id, in_var)
789
790! Select data set corresponding to the returned index
791 sds_id = sfselect(sd_id,sds_index)
792 retn = sfginfo(sds_id,name,rank,dim_sizes,data_type,nattrs)
793
794 do i=1,rank
795 edge(i)=dim_sizes(i)
796 start(i)=0
797 stride(i)=1
798 enddo
799 edge(rank)=slab_size
800 start(rank)=irec-1
801
802 iprod=1
803 do i=1,rank
804 iprod=iprod*edge(i)
805 enddo
806
807! Get the calibration and offset values of input
808 retn = sfgcal(sds_id,cal,cal_err,off,off_err,num_type)
809
810! Read Arrays which are not float32 or int8 or uint8 or uint32
811 if ((data_type.ne. 5).and.(data_type.ne.20).and. &
812 (data_type.ne.21).and.(data_type.ne.25)) then
813
814! Read the data set into the "buffer" array
815 retn=sfrdata(sds_id,start,stride,edge,buffer)
816
817! Calibrate the output
818 do i=1,iprod
819! Correct for 16-bit unsigned integers
820 if ((data_type.eq.23).and.(buffer(i).lt.0)) then
821 out_var(i)=buffer(i)+65536.0
822
823! No correction needed for signed or positive unsigned integers
824 else
825 out_var(i)=buffer(i)
826 endif
827
828 out_var(i)=out_var(i)*cal
829 enddo
830
831! Read uint32 arrays.
832 else if (data_type.eq.25) then
833
834! Read the data set into the "buffer3" uint32 array
835 retn=sfrdata(sds_id,start,stride,edge,buffer3)
836
837! Calibrate the output
838 do i=1,iprod
839! Correct for 32-bit unsigned integers
840 if ((data_type.eq.25).and.(buffer3(i).lt.0)) then
841 out_var(i)=buffer3(i)+4294967296.0
842 else
843 out_var(i)=buffer3(i)
844 endif
845 out_var(i)=out_var(i)*cal
846 enddo
847
848! Read int8 and uint8 arrays.
849 else if ((data_type.eq.20).or.(data_type.eq.21)) then
850
851! Read the data set into the "buffer2" byte-array
852 retn=sfrdata(sds_id,start,stride,edge,buffer2)
853
854! Calibrate the output
855 do i=1,iprod
856
857! Correct for 8-bit unsigned integers
858 itmp=buffer2(i)
859 if ((data_type.eq.21).and.(buffer2(i).lt.0)) then
860 itmp=itmp+256
861 out_var(i)=itmp
862
863! No correction needed for signed or positive unsigned integers
864 else
865 out_var(i)=itmp
866 endif
867
868 out_var(i)=out_var(i)*cal
869 enddo
870
871 else
872! Read float32 arrays directly into the "out_var" array
873 retn=sfrdata(sds_id,start,stride,edge,out_var)
874
875! Calibrate the output
876 do i=1,iprod
877 out_var(i)=out_var(i)*cal
878 enddo
879 endif
880
881! Terminate access to the array data set.
882 retn = sfendacc(sds_id)
883 end subroutine extract_sds
