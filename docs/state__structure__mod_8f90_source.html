7module state_structure_mod

9!> \defgroup state_structure_mod state_structure_mod
10!> @{ @brief Contains the stucture of the state vector and routines to access this structure
11!>
12!> Usage:
13!> call add_domain() in static_init_model()
14!> - must be called.
15!> - may be called more than once in static_init_model if a model has
16!> more than one domain (e.g. WRF).
17!>
18!> The add_domain() call adds a 'domain' to the state. This may be a component in
19!> the case of XCESM or another coupled model.
20!>
21!> There are three ways to add a domain (these are overloaded as add_domain):
22!> * add_domain_blank. This takes model size as an argument.
23!>
24!> * add_domain_from_file. This takes a netcdf file and a list of variables
25!>
26!> * add_domain_from_spec. This makes a skeleton structure for a domain. Dimensions
27!> for each variable must be added using add_dimension_to_variable(). This is intended
28!> to be used to create netcdf output for models like bgrid_solo that are spun up.
29!> Usage: call add_domain_from_spec
30!> for each variable call add_dimension_to_variable() for each dimension in the variable
31!> call finished_adding_domain_from_spec
32!>
33!> There are optional arguments to add_domain_from_spec and add_domain_from_file. These
34!> are:
35!> * kind_list - dart kind of each variable
36!> * clamp_vals - upper and lower bounds each variable (missing_r8 for no clamping)
37!> * update_list - true/false. Write out the variable. Default is .true.
38!>
39!> The state_structure hierarchy is:
40!> state -> domain -> variable
41!> Each time add_domain is called another domain is added to the state.
42!> Outside code can query with the state structure using domain_id and variable_id as
43!> arguments to accessor funcitons.
44!> Domain_id and variable_id are integers currently. It may be better to make domain_id
45!> a private type so other information can be stored in the domain_id.
46!>
47!> Interaction with the state_stucture depends on which module is accessing the structure.
48!> For example, model_mod querying the number of dimensions a variable is not concerned
49!> with the unlimited dimension, but state_vector_io_mod is concerned with the unlimtied
50!> dimension.
51!> The variable type has an io_type inside it to hold io information
52!> The io accessor functions have get_io_* in their name.
53!>
54!> get_dart_vector_index() and its inverse get_model_variable_indices() link
55!> a model x,y,z to dart index. The order of the state vector is no longer under model_mod
56!> control. Beware when converting model_mods such as CAM that transform the order of
57!> variables after reading from a netcdf file. There may be calculations in model_mod
58!> that are assuming a transformed order which no longer exists.

320function add_domain_from_file(info_file, num_vars, var_names, kind_list, clamp_vals, update_list) result(dom_id)
321
322character(len=*), intent(in) :: info_file
323integer, intent(in) :: num_vars
324character(len=*), intent(in) :: var_names(num_vars)
325integer, intent(in), optional :: kind_list(num_vars)
326real(r8), intent(in), optional :: clamp_vals(num_vars, 2)
327logical, intent(in), optional :: update_list(num_vars)
328integer :: dom_id
329
330integer :: ivar

332! add to domains
333call assert_below_max_num_domains('add_domain_from_file')
334state%num_domains = state%num_domains + 1
335!>@todo dom_id should be a handle.
336dom_id = state%num_domains
337
338! save information about the information file
339state%domain(dom_id)%info_file = info_file
340state%domain(dom_id)%method = 'file'
341

361if ( present(kind_list) ) call set_dart_kinds (dom_id, num_vars, kind_list)
362if ( present(clamp_vals) ) call set_clamping (dom_id, num_vars, clamp_vals)

560!-------------------------------------------------------------------------------
561!> load dimension information and calculate variable and domain sizes
562!-------------------------------------------------------------------------------
563subroutine load_variable_sizes(ncfile, domain)
564
565integer, intent(in) :: ncfile ! netdcf file id - should this be part of the domain handle?
566type(domain_type), intent(inout) :: domain
567
568integer :: ivar, jdim, num_vars, num_dims !< loop variables
569integer :: variable_size
570integer :: ret ! netcdf return value
571integer(i8) :: index_start, domain_size
572
573domain_size = 0
574num_vars = domain%num_variables
575index_start = state%model_size + 1
576
577do ivar = 1, num_vars
578
579 ! from netcdf id load variable dimension and ids
580 ret = nf90_inquire_variable(ncfile, domain%variable(ivar)%io_info%varid, &
581 ndims = domain%variable(ivar)%io_info%io_numdims, &
582 dimids = domain%variable(ivar)%io_info%io_dimIds, &
583 xtype = domain%variable(ivar)%io_info%xtype)
584
585 call nc_check(ret, 'load_variable_sizes, inq_variable', &
586 trim(domain%variable(ivar)%varname))
587
588 variable_size = 1
589 num_dims = domain%variable(ivar)%io_info%io_numdims
590
591 do jdim = 1, num_dims
592
593 ! load dimension names and lengths
594 ret = nf90_inquire_dimension(ncfile, domain%variable(ivar)%io_info%io_dimIds(jdim), &
595 name = domain%variable(ivar)%dimname(jdim), &
596 len = domain%variable(ivar)%io_info%dimlens(jdim))
597
598 domain%variable(ivar)%dimlens(jdim) = domain%variable(ivar)%io_info%dimlens(jdim)
599
600 call nc_check(ret, 'load_variable_sizes, inq_dimension', &
601 trim(domain%variable(ivar)%dimname(jdim)))
602
603 !>@todo FIXME we'll have to document that no user can have a 'member' dimension
604 !>in their own netcdf files. it's more reasonable to indicate 'time' is special
605 !>but this code doesn't know if its a user file or one we wrote. but without
606 !>skipping these dimensions you get the wrong variable size because it includes
607 !>all times and all members in the size. this needs to be revisited later.
608
609 if ((domain%variable(ivar)%dimname(jdim) == 'time') .or. &
610 (domain%variable(ivar)%dimname(jdim) == 'member')) cycle
611
612 variable_size = variable_size * domain%variable(ivar)%dimlens(jdim)
613
614 enddo
615
616
617 ! to be consistent this needs to ignore both 'time' and 'member' for
618 ! files we write, and newer netcdf libs support multiple unlimited dims
619 ! so this does need to change. but it works, somehow, as-is so leave it
620 ! for now.
621 ! subtract the unlimited dimension if it exists
622 !>@todo : how to handle models with multiple unlimited dimensions?
623 !> nf90_inquire returns the first unlimited dimension id which
624 !> is the slowest varying dimension. For now am assuming that
625 !> there can only be one unlimited dimension. Just subtract
626 !> to get 'spatial' dimensions.
627 if ( any(domain%variable(ivar)%io_info%io_dimIds(1:num_dims) == domain%unlimDimId) ) then
628 domain%variable(ivar)%numdims = num_dims - 1
629 domain%variable(ivar)%var_has_unlim = .true.
630 else
631 domain%variable(ivar)%numdims = num_dims
632 endif
633
634 ! member is not a spatial domain but could be included in a single file
635 do jdim = 1, num_dims
636 if ( domain%variable(ivar)%dimname(jdim) == 'member') then
637 domain%variable(ivar)%numdims = domain%variable(ivar)%numdims - 1
638 endif
639 enddo
640
641 domain%variable(ivar)%var_size = variable_size
642
643 ! first and last location of variable in the state index
644 domain%variable(ivar)%index_start = index_start
645 domain%variable(ivar)%index_end = index_start + variable_size - 1
646
647 ! update counters
648 domain_size = domain_size + variable_size
649 index_start = domain%variable(ivar)%index_end + 1
650
651enddo
652
653domain%dom_size = domain_size
654state%model_size = state%model_size + domain_size
655
656end subroutine load_variable_sizes

658!-------------------------------------------------------------------------------
659!> Identify the unique dimensions within a domain
660!> I think you can sort dimension ids and remove duplicates
661!> What if there are 0 dimensions?
662!-------------------------------------------------------------------------------
663subroutine load_unique_dim_info(dom_id)
664
665integer, intent(in) :: dom_id ! domain identifier
666
667integer, allocatable :: array_of_dimids(:)
668integer, allocatable :: array_of_lengths(:)
669integer, allocatable :: array_of_indices(:)
670character(len=NF90_MAX_NAME), allocatable :: array_of_names(:)
671logical, allocatable :: unique(:)
672
673integer :: ivar, jdim, jdim_dom
674integer :: count_dims, ndims, nvars, ndims_dom
675integer :: num_unique
676
677num_unique = 0
678
679ndims_dom = get_domain_num_dims(dom_id)
680nvars = get_num_variables(dom_id)
681
682allocate(array_of_dimids(ndims_dom))
683allocate(array_of_names(ndims_dom))
684allocate(array_of_lengths(ndims_dom))
685allocate(array_of_indices(ndims_dom))
686allocate(unique(ndims_dom))
687
688count_dims = 1
689
690do ivar = 1, nvars
691 ndims = get_io_num_dims(dom_id, ivar)
692 do jdim = 1, ndims
693 array_of_dimids(count_dims) = state%domain(dom_id)%variable(ivar)%io_info%io_dimIds(jdim)
694 array_of_names(count_dims) = state%domain(dom_id)%variable(ivar)%dimname(jdim)
695 array_of_lengths(count_dims) = state%domain(dom_id)%variable(ivar)%dimlens(jdim)
696 count_dims = count_dims + 1
697 enddo
698enddo
699
700call index_sort(array_of_dimids, array_of_indices, ndims_dom)
701
702count_dims = 1
703unique(:) = .false.
704unique(1) = .true.
705
706do jdim_dom = 2, ndims_dom
707 if( array_of_dimids(array_of_indices(jdim_dom)) /= &
708 array_of_dimids(array_of_indices(jdim_dom-1)) ) then
709
710 count_dims = count_dims + 1
711 unique(jdim_dom) = .true.
712 endif
713enddo
714
715state%domain(dom_id)%num_unique_dims = count_dims
716
717allocate(state%domain(dom_id)%unique_dim_names(count_dims))
718allocate(state%domain(dom_id)%unique_dim_length(count_dims))
719allocate(state%domain(dom_id)%original_dim_IDs(count_dims))
720
721count_dims = 1
722
723do jdim_dom = 1, ndims_dom
724 if(unique(jdim_dom)) then
725 state%domain(dom_id)%unique_dim_names( count_dims) = array_of_names( array_of_indices(jdim_dom))
726 state%domain(dom_id)%unique_dim_length(count_dims) = array_of_lengths(array_of_indices(jdim_dom))
727 state%domain(dom_id)%original_dim_IDs( count_dims) = array_of_dimids( array_of_indices(jdim_dom))
728 count_dims = count_dims + 1
729 endif
730enddo
731
732deallocate(array_of_dimids, array_of_names, array_of_lengths, array_of_indices, unique)
733
734end subroutine load_unique_dim_info

736!-------------------------------------------------------------------------------
737!> Check to see if the template file has any of the common cf-conventions
738!>

742!> * short_name
743!> * _FillValue
744!> * missing_value
745!> * add_offset
746!> * scale_factor
747!>
748!> If they exist, load them up into the state structure.
749!-------------------------------------------------------------------------------
750
751subroutine load_common_cf_conventions(domain)
752
753type(domain_type), intent(inout) :: domain
754
755integer :: ivar
756integer :: nvars
757
758! netcdf variables
759integer :: ret, ncid, VarID
760integer :: var_xtype
761integer :: cf_spvalINT
762real(r4) :: cf_spvalR4
763real(digits12) :: cf_spvalR8
764real(digits12) :: cf_scale_factor, cf_add_offset
765character(len=512) :: ncFilename
766character(len=NF90_MAX_NAME) :: var_name
767character(len=NF90_MAX_NAME) :: cf_long_name, cf_short_name, cf_units
768
769character(len=*), parameter :: routine = 'load_common_cf_conventions'
770
771ncfilename = domain%info_file
772
773ret = nf90_open(ncfilename, nf90_nowrite, ncid)
774call nc_check(ret, routine,'nf90_open '//trim(ncfilename))
775
776! determine attributes of each variable in turn
777
778nvars = domain%num_variables
779
780do ivar = 1, nvars
781 var_name = domain%variable(ivar)%varname
782
783 ret = nf90_inq_varid(ncid, trim(var_name), varid)
784 call nc_check(ret, routine, 'inq_varid '//trim(var_name))
785
786 ! If the short_name, long_name and/or units attributes are set, get them.
787 ! They are not REQUIRED by DART but are nice to keep around if they are present.
788
789 if( nf90_inquire_attribute( ncid, varid, 'long_name') == nf90_noerr ) then
790 ret = nf90_get_att(ncid, varid, 'long_name' , cf_long_name)
791 call nc_check(ret, routine, 'get_att long_name '//trim(var_name))
792 domain%variable(ivar)%io_info%long_name = cf_long_name
793 endif
794
795 if( nf90_inquire_attribute( ncid, varid, 'short_name') == nf90_noerr ) then
796 ret = nf90_get_att(ncid, varid, 'short_name' , cf_short_name)
797 call nc_check(ret, routine, 'get_att short_name '//trim(var_name))
798 domain%variable(ivar)%io_info%short_name = cf_short_name
799 endif
800
801 if( nf90_inquire_attribute( ncid, varid, 'units') == nf90_noerr ) then
802 ret = nf90_get_att(ncid, varid, 'units' , cf_units)
803 call nc_check(ret, routine, 'get_att units '//trim(var_name))
804 domain%variable(ivar)%io_info%units = cf_units
805 endif
806
807 ! Saving any FillValue, missing_value attributes ...
808 ! Also stuff them into the 'r8' slots to facilitate simpler clamp_variable
809 ! implementation. (Since we coerce the DART state to 'r8')
810
811 var_xtype = domain%variable(ivar)%io_info%xtype
812 select case (var_xtype)
813 case ( nf90_int )
814 ! Sometimes the attributes are specified as GLOBAL attributes
815 ret = nf90_get_att(ncid, nf90_global, '_FillValue', cf_spvalint)
816 if (ret == nf90_noerr) then
817 domain%variable(ivar)%io_info%spvalINT = cf_spvalint
818 domain%variable(ivar)%io_info%spvalR8 = real(cf_spvalint,r8)
819 domain%variable(ivar)%io_info%has_FillValue = .true.
820 endif
821 ret = nf90_get_att(ncid, nf90_global, 'missing_value', cf_spvalint)
822 if (ret == nf90_noerr) then
823 domain%variable(ivar)%io_info%missingINT = cf_spvalint
824 domain%variable(ivar)%io_info%spvalR8 = real(cf_spvalint,r8)
825 domain%variable(ivar)%io_info%has_missing_value = .true.
826 endif
827 ! Usually the attributes are specified as variable attributes
828 ret = nf90_get_att(ncid, varid, '_FillValue', cf_spvalint)
829 if (ret == nf90_noerr) then
830 domain%variable(ivar)%io_info%spvalINT = cf_spvalint
831 domain%variable(ivar)%io_info%spvalR8 = real(cf_spvalint,r8)
832 domain%variable(ivar)%io_info%has_FillValue = .true.
833 endif
834 ret = nf90_get_att(ncid, varid, 'missing_value', cf_spvalint)
835 if (ret == nf90_noerr) then
836 domain%variable(ivar)%io_info%missingINT = cf_spvalint
837 domain%variable(ivar)%io_info%missingR8 = real(cf_spvalint,r8)
838 domain%variable(ivar)%io_info%has_missing_value = .true.
839 endif
840
841 case ( nf90_float )
842 ret = nf90_get_att(ncid, nf90_global, '_FillValue', cf_spvalr4)
843 if (ret == nf90_noerr) then
844 domain%variable(ivar)%io_info%spvalR4 = cf_spvalr4
845 domain%variable(ivar)%io_info%spvalR8 = real(cf_spvalr4,r8)
846 domain%variable(ivar)%io_info%has_FillValue = .true.
847 endif
848 ret = nf90_get_att(ncid, nf90_global, 'missing_value', cf_spvalr4)
849 if (ret == nf90_noerr) then
850 domain%variable(ivar)%io_info%missingR4 = cf_spvalr4
851 domain%variable(ivar)%io_info%spvalR8 = real(cf_spvalr4,r8)
852 domain%variable(ivar)%io_info%has_missing_value = .true.
853 endif
854 ret = nf90_get_att(ncid, varid, '_FillValue', cf_spvalr4)
855 if (ret == nf90_noerr) then
856 domain%variable(ivar)%io_info%spvalR4 = cf_spvalr4
857 domain%variable(ivar)%io_info%spvalR8 = real(cf_spvalr4,r8)
858 domain%variable(ivar)%io_info%has_FillValue = .true.
859 endif
860 ret = nf90_get_att(ncid, varid, 'missing_value', cf_spvalr4)
861 if (ret == nf90_noerr) then
862 domain%variable(ivar)%io_info%missingR4 = cf_spvalr4
863 domain%variable(ivar)%io_info%missingR8 = real(cf_spvalr4,r8)
864 domain%variable(ivar)%io_info%has_missing_value = .true.
865 endif
866
867 case ( nf90_double )
868
869 ! If r8 = r4,
870 ! the missing_value must be present in both missingR4 and missingR8
871 ! ditto for _FillValue.
872 ! This satisfies the overloaded operator 'get_missing_value, get_FillValue'
873
874 ret = nf90_get_att(ncid, nf90_global, '_FillValue', cf_spvalr8)
875 if (ret == nf90_noerr) then
876 domain%variable(ivar)%io_info%spvalR4 = cf_spvalr8
877 domain%variable(ivar)%io_info%spvalR8 = cf_spvalr8
878 domain%variable(ivar)%io_info%has_FillValue = .true.
879 endif
880 ret = nf90_get_att(ncid, nf90_global, 'missing_value', cf_spvalr8)
881 if (ret == nf90_noerr) then
882 domain%variable(ivar)%io_info%missingR4 = cf_spvalr8
883 domain%variable(ivar)%io_info%missingR8 = cf_spvalr8
884 domain%variable(ivar)%io_info%has_missing_value = .true.
885 endif
886 ret = nf90_get_att(ncid, varid, '_FillValue', cf_spvalr8)
887 if (ret == nf90_noerr) then
888 domain%variable(ivar)%io_info%spvalR4 = cf_spvalr8
889 domain%variable(ivar)%io_info%spvalR8 = cf_spvalr8
890 domain%variable(ivar)%io_info%has_FillValue = .true.
891 endif
892 ret = nf90_get_att(ncid, varid, 'missing_value', cf_spvalr8)
893 if (ret == nf90_noerr) then
894 domain%variable(ivar)%io_info%missingR4 = cf_spvalr8
895 domain%variable(ivar)%io_info%missingR8 = cf_spvalr8
896 domain%variable(ivar)%io_info%has_missing_value = .true.
897 endif
898
899 case DEFAULT
900 write(string1,*) ' unsupported netcdf variable type : ', var_xtype
901 call error_handler(e_err,routine,string1,source)
902 end select
903
904 ! If the variable has one or the other, no problem.
905 ! If the variable has both _FillValue and missing_value attributes, the
906 ! values must be the same or we are lost. DART only supports one missing
907 ! value code. When we go to write, we have no way of knowing which value
908 ! to use as a replacement for the DART missing code.
909
910 if ( domain%variable(ivar)%io_info%has_missing_value .and. &
911 domain%variable(ivar)%io_info%has_FillValue ) then
912
913 if ( domain%variable(ivar)%io_info%missingR8 /= &
914 domain%variable(ivar)%io_info%spvalR8 ) then
915
916 write(string1, *) trim(var_name)//' missing_value /= _FillValue '
917 write(string2,*) 'missing_value is ', domain%variable(ivar)%io_info%missingR8
918 write(string3,*) '_FillValue is ', domain%variable(ivar)%io_info%spvalR8
919 call error_handler(e_err,'set_dart_missing_value:',string1, &
920 source, text2=string2, text3=string3)
921 endif
922
923 endif
924
925 !>@todo FIXME : Not supporting scale factor or offset at the moment, so just error.
926 !> To fully support netCDF I/O we need to
927 !> pack and unpack the variable if these attributes exist.
928 if (nf90_get_att(ncid, varid, 'scale_factor', cf_scale_factor) == nf90_noerr) then
929 domain%variable(ivar)%io_info%scale_factor = cf_scale_factor
930 write(string1,*) 'scale_factor not supported at the moment'
931 write(string2,*) 'contact DART if you would like to get this to work'
932 call error_handler(e_err,routine,string1,source,text2=string2)
933 endif
934
935 if (nf90_get_att(ncid, varid, 'add_offset', cf_add_offset) == nf90_noerr) then
936 domain%variable(ivar)%io_info%add_offset = cf_add_offset
937 write(string1,*) 'add_offset not supported at the moment'
938 write(string2,*) 'contact DART if you would like to get this to work'
939 call error_handler(e_err,routine,string1,source,text2=string2)
940 endif
941
942enddo
943
944! close netcdf file
945ret = nf90_close(ncid)
946call nc_check(ret, routine, 'nf90_close', trim(ncfilename))
947
948end subroutine load_common_cf_conventions

951!> TIEGCM, the top level of each variable is the boundary condition.
952!> So the top level should not be part of the state.
953!> How general does this need to be? At the moment it is just a hack to
954!> be able to proceed with TIEGCM.
955!>
956!> This slicing needs to be on the the DART state structure, not the io netcdf
957!> structure.
958
959subroutine hyperslice_domain(dom_id, dim_name, new_length)
960
961integer, intent(in) :: dom_id
962character(len=*), intent(in) :: dim_name
963integer, intent(in) :: new_length
964
965integer :: ivar, jdim
966integer(i8) :: orig_domain_size, orig_model_size, index_start
967integer(i8) :: domain_size, variable_size
968
969orig_domain_size = get_domain_size(dom_id)
970orig_model_size = state%model_size
971! Start of the domain remains the same
972index_start = state%domain(dom_id)%variable(1)%index_start
973
974! Reduce the dimension size
975do ivar = 1, get_num_variables(dom_id)
976 do jdim = 1, get_num_dims(dom_id, ivar)
977 if (get_dim_name(dom_id, ivar, jdim) == dim_name) then
978 state%domain(dom_id)%variable(ivar)%dimlens(jdim) = new_length
979 endif
980 enddo
981enddo
982
983! need to modify state index accessing for each variable:
984! variable_size
985! index_start, index_end
986! model size
987! domain size
988
989domain_size = 0
990! variable size
991do ivar = 1, get_num_variables(dom_id)
992
993 variable_size = 1
994
995 do jdim = 1, get_num_dims(dom_id, ivar)
996 !HK there is a todo about this, what is going on with time and member?
997 if ((state%domain(dom_id)%variable(ivar)%dimname(jdim) == 'time') .or. &
998 (state%domain(dom_id)%variable(ivar)%dimname(jdim) == 'member')) cycle
999
1000 variable_size = variable_size *state%domain(dom_id)%variable(ivar)%dimlens(jdim)
1001 enddo
1002
1003 state%domain(dom_id)%variable(ivar)%var_size = variable_size
1004
1005 ! first and last location of variable in the state index
1006 state%domain(dom_id)%variable(ivar)%index_start = index_start
1007 state%domain(dom_id)%variable(ivar)%index_end = index_start + variable_size - 1
1008
1009 ! update counters
1010 domain_size = domain_size + variable_size
1011 index_start = state%domain(dom_id)%variable(ivar)%index_end + 1
1012
1013enddo
1014
1015! update state structure with new domain size and new model size
1016state%domain(dom_id)%dom_size = domain_size
1017state%model_size = orig_model_size - orig_domain_size + domain_size
1018
1019end subroutine hyperslice_domain

1443!> Given a dart state index, return the iloc, jloc, kloc location of the local variable
1444
1445
1446subroutine get_model_variable_indices(index_in, iloc, jloc, kloc, var_id, dom_id, kind_index, kind_string)
1447
1448integer(i8), intent(in) :: index_in
1449integer, intent(out) :: iloc
1450integer, intent(out) :: jloc
1451integer, intent(out) :: kloc
1452integer, intent(out), optional :: var_id
1453integer, intent(out), optional :: dom_id
1454integer, intent(out), optional :: kind_index
1455character(len=*), intent(out), optional :: kind_string
1456
1457integer :: local_ind
1458integer(i8) :: index_abs, index_start, index_end
1459integer :: ndims, nvars, ndomains
1460integer :: idom, ivar
1461integer :: domid, varid
1462integer, allocatable :: dsize(:)
1463
1464! identity obs come in with a negative value - absolute
1465! value into the state vector.
1466index_abs = abs(index_in)
1467
1468varid = -1
1469ndomains = get_num_domains()
1470
1471! get the local variable indices
1472findvariable : do idom = 1, ndomains
1473 nvars = get_num_variables(idom)
1474 do ivar = 1, nvars
1475
1476 index_start = get_index_start(idom,ivar)
1477 index_end = get_index_end(idom,ivar)

1479 if ((index_abs .ge. index_start) .and. &
1480 (index_abs .le. index_end) ) then
1481

1483 varid = ivar
1484 domid = idom
1485 if(debug) print*, 'index_start, index_end', index_start, index_end
1486

1488
1489 endif

1501

1521 source, text2=string2)
1522endif

1525 write(*,'(A)') '---------------------------------------------'
1526 write(*,'(A, I10)') 'index_in = ', index_in
1527 write(*,'(A, I2, A, I2)') ' domid = ', domid, ' varid = ', varid
1528 write(*,'(A, 3I10, A)') ' (i, j, k) = (', iloc, jloc, kloc, ')'

1533 var_id = varid
1534
1535if (present(dom_id)) &
1536 dom_id = domid
1537

1539 kind_string = get_kind_string(domid,varid)
1540
1541if (present(kind_index)) &
1542 kind_index = get_kind_index(domid,varid)
