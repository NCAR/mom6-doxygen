7! do a simple test that the mean distance between sample
8! draws from two gaussian distributions are close to the
9! expected value. indirect test of the random number generator
10! which is used in the gaussian random numbers.

32real(r8) :: r1, r2, dist2, mean_dist2

52
53write(*, *) ''
54write(*, wformati) 'sample size = ', n
55write(*, *) ''
56write(*, wformat2) 'mean, stddev 1 = ', 0.0_r8, sd1
57write(*, wformat2) 'mean, stddev 2 = ', mean, sd2
58write(*, *) ''
59
60
61call init_random_seq(r,iseed)
62mean_dist2 = 0.0_r8
63do i = 1, n
64 r1 = random_gaussian(r, 0.0_r8, sd1)
65 r2 = random_gaussian(r, mean, sd2)
66 dist2 = (r1 - r2)**2
67 mean_dist2 = mean_dist2 + dist2
68end do
69
70write(*, wformat1) 'predicted distance = ', sqrt(mean**2 + sd1**2 + sd2**2)
71write(*, wformat1) 'sample mean distance = ', sqrt(mean_dist2 / n)
72write(*, *) ''
73
74call finalize_utilities()

76
77end program test_diff
