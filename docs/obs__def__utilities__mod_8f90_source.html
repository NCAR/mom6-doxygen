34!----------------------------------------------------------------------
35!> track_status can be used to keep track of the status of
36!> each ensemble member during multiple calls to model_interpolate
37!> for a given obs_def.
38!> It assumes that you are starting with istatus(:) = 0
39!> If debugging, return_now is only set to true if all istatuses are non-zero
40!> If not debugging, return_now is set to true if any istatues are non-zero
41!> and any remaining zero istatues are set to 1.
42subroutine track_status(ens_size, val_istatus, val_data, istatus, return_now)
43
44integer, intent(in) :: ens_size
45integer, intent(in) :: val_istatus(ens_size)
46real(r8), intent(inout) :: val_data(ens_size) !! expected_obs for obs_def
47integer, intent(inout) :: istatus(ens_size) !! istatus for obs_def
48logical, intent(out) :: return_now
49
50where (istatus == 0) istatus = val_istatus
51where (istatus /= 0) val_data = missing_r8
52
53return_now = .false.
54if (debug) then
55 if( all(istatus /= 0))then
56 return_now = .true.
57 val_data(:) = missing_r8
58 endif
59else
60 if( any(istatus /= 0)) then
61 return_now = .true.
62 val_data(:) = missing_r8
63 where (istatus == 0) istatus = 1
64 endif
65endif
66
67end subroutine track_status
