7program convert_madis_mesonet
8
9!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
10!
11! convert_madis_mesonet - program that reads a MADIS netCDF land
12! surface observation file and writes a DART
13! obs_seq file using the DART library routines.
14! This version works on generic MESONET files.
15!
16! created Dec. 2007 Ryan Torn, NCAR/MMM
17! modified Dec. 2008 Soyoung Ha and David Dowell, NCAR/MMM
18! - added dewpoint as an output variable
19! - added relative humidity as an output variable
20!
21! modified to include QC_flag check (Soyoung Ha, NCAR/MMM, 08-04-2009)
22! split from generic surface converter since the METAR and MESONET
23! files differ slightly in format. (Glen Romine, NCAR/MMM Feb 2010)
24!
25! modified to use a common set of utilities, better netcdf error checks,
26! able to insert obs with any time correctly (not only monotonically
27! increasing times) nancy collins, ncar/image 11 march 2010
28!
29! keep original obs times, make source for all converters as similar
30! as possbile. nancy collins, ncar/image 26 march 2010
31!
32!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
33

35use utilities_mod, only : initialize_utilities, finalize_utilities
36use netcdf_utilities_mod, only : nc_open_file_readonly, nc_close_file
37use time_manager_mod, only : time_type, set_calendar_type, set_date, &
38 increment_time, get_time, operator(-), gregorian
39use location_mod, only : vertissurface
40use obs_sequence_mod, only : obs_sequence_type, obs_type, read_obs_seq, &
41 static_init_obs_sequence, init_obs, write_obs_seq, &
42 init_obs_sequence, get_num_obs, &
43 set_copy_meta_data, set_qc_meta_data
44use meteor_mod, only : sat_vapor_pressure, specific_humidity, &
45 wind_dirspd_to_uv, invert_altimeter, pres_alt_to_pres, &
46 temp_and_dewpoint_to_rh
47use obs_err_mod, only : land_temp_error, land_wind_error, &
48 land_pres_error, land_rel_hum_error
49use dewpoint_obs_err_mod, only : dewpt_error_from_rh_and_temp, &
50 rh_error_from_dewpt_and_temp
51use obs_kind_mod, only : land_sfc_u_wind_component, land_sfc_v_wind_component, &
52 land_sfc_temperature, land_sfc_specific_humidity, &
53 land_sfc_dewpoint, land_sfc_relative_humidity, &
54 land_sfc_altimeter
55use sort_mod, only : index_sort
56use obs_utilities_mod, only : getvar_real, get_or_fill_qc, add_obs_to_seq, &
57 create_3d_obs, getvar_int, getdimlen, set_missing_name, &
58 is_variable_integer

60use netcdf
61
62implicit none
63
64character(len=16), parameter :: surface_netcdf_file = 'mesonet_input.nc'
65character(len=129), parameter :: surface_out_file = 'obs_seq.mesonet'
66
67! the following logical parameters control which water-vapor variables appear in the output file,
68! whether to use the NCEP error or Lin and Hubbard (2004) moisture error model, and if the
69! input file has data quality control fields, whether to use or ignore them.
70logical, parameter :: lh_err = .false.
71logical, parameter :: include_specific_humidity = .true.
72logical, parameter :: include_relative_humidity = .false.
73logical, parameter :: include_dewpoint = .false.
74logical, parameter :: use_input_qc = .true.
75
76integer, parameter :: num_copies = 1, & ! number of copies in sequence
77 num_qc = 1 ! number of QC entries
78
79integer :: ncid, nobs, nvars, n, i, oday, osec, nused
80logical :: file_exist, first_obs
81real(r8) :: alti_miss, tair_miss, tdew_miss, wdir_miss, wspd_miss, uwnd, &
82 vwnd, palt, qobs, qsat, rh, oerr, pres, qerr, qc
83
84integer, allocatable :: tobs(:), tused(:), used(:), sorted_used(:)
85real(r8), allocatable :: lat(:), lon(:), elev(:), alti(:), tair(:), &
86 tdew(:), wdir(:), wspd(:), tobs_r(:)
87integer, allocatable :: qc_alti(:), qc_tair(:), qc_tdew(:), qc_wdir(:), qc_wspd(:)
88
89type(obs_sequence_type) :: obs_seq
90type(obs_type) :: obs, prev_obs
91type(time_type) :: comp_day0, time_obs, prev_time
92
93
94!----------------------------------------------------------------
95
96call initialize_utilities('convert_madis_mesonet')
97
98
99! put the reference date into DART format
100call set_calendar_type(gregorian)
101comp_day0 = set_date(1970, 1, 1, 0, 0, 0)
102
103first_obs = .true.
104
105
106ncid = nc_open_file_readonly(surface_netcdf_file, 'convert_madis_mesonet')
107
108call getdimlen(ncid, "recNum", nobs)
109call set_missing_name("missing_value")
110
111allocate( lat(nobs)) ; allocate( lon(nobs))
112allocate(elev(nobs)) ; allocate(alti(nobs))
113allocate(tair(nobs)) ; allocate(tdew(nobs))
114allocate(wdir(nobs)) ; allocate(wspd(nobs))
115allocate(used(nobs)) ; allocate(sorted_used(nobs))
116allocate(tobs(nobs)) ; allocate(tobs_r(nobs))
117allocate(tused(nobs))
118
119nvars = 4
120if (include_specific_humidity) nvars = nvars + 1
121if (include_relative_humidity) nvars = nvars + 1
122if (include_dewpoint) nvars = nvars + 1
123
124allocate(qc_alti(nobs))
125allocate(qc_tair(nobs)) ; allocate(qc_tdew(nobs))
126allocate(qc_wdir(nobs)) ; allocate(qc_wspd(nobs))
127
128! read in the data arrays
129call getvar_real(ncid, "latitude", lat ) ! latitude
130call getvar_real(ncid, "longitude", lon ) ! longitude
131call getvar_real(ncid, "elevation", elev ) ! elevation
132call getvar_real(ncid, "altimeter", alti, alti_miss) ! altimeter setting
133call getvar_real(ncid, "temperature", tair, tair_miss) ! air temperature
134call getvar_real(ncid, "dewpoint", tdew, tdew_miss) ! dew-point temperature
135call getvar_real(ncid, "windDir", wdir, wdir_miss) ! wind direction
136call getvar_real(ncid, "windSpeed", wspd, wspd_miss) ! wind speed
137
138! recent versions of MADIS files seem to have changed time to a real
139! perhaps to avoid overflowing an integer value?
140if (is_variable_integer(ncid, "observationTime")) then
141 call getvar_int (ncid, "observationTime", tobs ) ! observation time as integer
142else
143 call getvar_real(ncid, "observationTime", tobs_r ) ! observation time as real
144 tobs = nint(tobs_r) ! and convert to nearest second
145endif
146
147! if user says to use them, read in QCs if present
148if (use_input_qc) then
149 call get_or_fill_qc(ncid, "altimeterQCR", qc_alti)
150 call get_or_fill_qc(ncid, "temperatureQCR", qc_tair)
151 call get_or_fill_qc(ncid, "dewpointQCR", qc_tdew)
152 call get_or_fill_qc(ncid, "windDirQCR", qc_wdir)
153 call get_or_fill_qc(ncid, "windSpeedQCR", qc_wspd)
154else
155 qc_alti = 0
156 qc_tair = 0 ; qc_tdew = 0
157 qc_wdir = 0 ; qc_wspd = 0
158endif
159
160call nc_close_file(ncid, 'convert_madis_mesonet')
161
162
163! either read existing obs_seq or create a new one
164call static_init_obs_sequence()
165call init_obs(obs, num_copies, num_qc)
166call init_obs(prev_obs, num_copies, num_qc)
167
168inquire(file=surface_out_file, exist=file_exist)
169
170if ( file_exist ) then
171
172 ! existing file found, append to it
173 call read_obs_seq(surface_out_file, 0, 0, nvars*nobs, obs_seq)
174
175else
176
177 ! create a new one
178 call init_obs_sequence(obs_seq, num_copies, num_qc, nvars*nobs)
179 do i = 1, num_copies
180 call set_copy_meta_data(obs_seq, i, 'MADIS observation')
181 end do
182 do i = 1, num_qc
183 call set_qc_meta_data(obs_seq, i, 'Data QC')
184 end do
185
186endif
187
188! Set the DART data quality control. Be consistent with NCEP codes;
189! 0 is 'must use', 1 is good, no reason not to use it.
190qc = 1.0_r8
191
192nused = 0
193obsloop1: do n = 1, nobs
194
195 ! check the lat/lon values to see if they are ok
196 if ( lat(n) > 90.0_r8 .or. lat(n) < -90.0_r8 ) cycle obsloop1
197 if ( lon(n) > 180.0_r8 .or. lon(n) < -180.0_r8 ) cycle obsloop1
198
199 ! change lon from -180 to 180 into 0-360
200 if ( lon(n) < 0.0_r8 ) lon(n) = lon(n) + 360.0_r8
201
202 ! check for duplicate observations
203 do i=1, nused
204 if ( lon(n) == lon(used(i)) .and. &
205 lat(n) == lat(used(i)) .and. &
206 tobs(n) == tobs(used(i)) ) cycle obsloop1
207 enddo
208
209 nused = nused + 1
210 used(nused) = n
211 tused(nused) = tobs(n)
212
213enddo obsloop1
214
215! sort into time order
216call index_sort(tused, sorted_used, nused)
217
218obsloop2: do i = 1, nused
219
220 ! get the next unique observation in sorted time order
221 n = used(sorted_used(i))
222
223
224 ! compute time of observation
225 time_obs = increment_time(comp_day0, tobs(n))
226
227 palt = pres_alt_to_pres(elev(n)) * 0.01_r8
228
229 ! extract actual time of observation in file into oday, osec.
230 call get_time(time_obs, osec, oday)
231
232 ! add altimeter data to obs_seq
233 if ( alti(n) /= alti_miss .and. qc_alti(n) == 0 ) then
234
235 pres = invert_altimeter(alti(n) * 0.01_r8, elev(n))
236 oerr = land_pres_error(palt)
237 if ( alti(n) >= 89000.0_r8 .and. &
238 alti(n) <= 110000.0_r8 .and. oerr /= missing_r8 ) then
239
240 call create_3d_obs(lat(n), lon(n), elev(n), vertissurface, alti(n) * 0.01_r8, &
241 land_sfc_altimeter, oerr, oday, osec, qc, obs)
242 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
243
244 endif
245
246 endif
247
248 ! add wind component data to obs_seq
249 if ( wdir(n) /= wdir_miss .and. qc_wdir(n) == 0 .and. &
250 wspd(n) /= wspd_miss .and. qc_wspd(n) == 0 ) then
251
252 call wind_dirspd_to_uv(wdir(n), wspd(n), uwnd, vwnd)
253 oerr = land_wind_error(palt)
254 if ( abs(uwnd) < 150.0_r8 .and. abs(vwnd) < 150.0_r8 .and. oerr /= missing_r8 ) then
255
256 call create_3d_obs(lat(n), lon(n), elev(n), vertissurface, uwnd, &
257 land_sfc_u_wind_component, oerr, oday, osec, qc, obs)
258 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
259
260 call create_3d_obs(lat(n), lon(n), elev(n), vertissurface, vwnd, &
261 land_sfc_v_wind_component, oerr, oday, osec, qc, obs)
262 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
263
264 endif
265
266 endif
267
268 ! add air temperature data to obs_seq
269 if ( tair(n) /= tair_miss .and. qc_tair(n) == 0 ) then
270
271 oerr = land_temp_error(palt)
272 if ( tair(n) >= 200.0_r8 .and. tair(n) <= 335.0_r8 .and. oerr /= missing_r8 ) then
273
274 call create_3d_obs(lat(n), lon(n), elev(n), vertissurface, tair(n), &
275 land_sfc_temperature, oerr, oday, osec, qc, obs)
276 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
277
278 endif
279
280 endif
281
282 ! if the air, dewpoint are ok, then see which of
283 ! the three types of moisture obs to generate.
284 if ( tair(n) /= tair_miss .and. qc_tair(n) == 0 .and. &
285 tdew(n) /= tdew_miss .and. qc_tdew(n) == 0 ) then
286
287 ! before we start computing things based on the dewpoint,
288 ! make sure it isn't larger than the air temp. if it is
289 ! more than a degree larger, skip it completely. if it is
290 ! less, set them equal and continue.
291 if (tdew(n) > tair(n)) then
292 if (tdew(n) > tair(n) + 1.0_r8) cycle obsloop2
293 tdew(n) = tair(n)
294 endif
295
296 ! add specific humidity to obs_seq
297 if ( include_specific_humidity .and. &
298 alti(n) /= alti_miss .and. qc_alti(n) == 0 ) then
299
300 qobs = specific_humidity(sat_vapor_pressure(tdew(n)), pres * 100.0_r8)
301 qsat = specific_humidity(sat_vapor_pressure(tair(n)), pres * 100.0_r8)
302 if ( lh_err ) then
303 qerr = rh_error_from_dewpt_and_temp(tair(n), tdew(n))
304 else
305 qerr = land_rel_hum_error(pres, tair(n), qobs / qsat)
306 endif
307 oerr = max(qerr * qsat, 0.0001_r8)
308
309 if ( qobs > 0.0_r8 .and. &
310 qobs <= 0.07_r8 .and. qerr /= missing_r8 ) then
311
312 call create_3d_obs(lat(n), lon(n), elev(n), vertissurface, qobs, &
313 land_sfc_specific_humidity, oerr, oday, osec, qc, obs)
314 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
315
316 endif
317
318 endif
319
320 ! add relative humidity data to obs_seq
321 if ( include_relative_humidity ) then
322
323 rh = temp_and_dewpoint_to_rh(tair(n), tdew(n))
324 if ( lh_err ) then
325 oerr = rh_error_from_dewpt_and_temp(tair(n), tdew(n))
326 else
327 oerr = land_rel_hum_error(pres, tair(n), rh)
328 endif
329
330 if ( rh > 0.0_r8 .and. &
331 rh <= 1.5_r8 .and. oerr /= missing_r8 ) then
332
333 call create_3d_obs(lat(n), lon(n), elev(n), vertissurface, rh, &
334 land_sfc_relative_humidity, oerr, oday, osec, qc, obs)
335 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
336
337 endif
338
339 endif
340
341 ! add dew-point temperature data to obs_seq
342 if ( include_dewpoint ) then
343
344 rh = temp_and_dewpoint_to_rh(tair(n), tdew(n))
345 oerr = dewpt_error_from_rh_and_temp(tair(n), rh)
346
347 if ( rh > 0.0_r8 .and. &
348 rh <= 1.5_r8 .and. oerr /= missing_r8 ) then
349
350 call create_3d_obs(lat(n), lon(n), elev(n), vertissurface, tdew(n), &
351 land_sfc_dewpoint, oerr, oday, osec, qc, obs)
352 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
353
354 endif
355
356 ! print*, 'temp (C), rh (%), oerr: ', tair(n)-273.15_r8, rh*100.0_r8, oerr
357
358 endif
359
360 endif ! quality control/missing check on tair, tdew
361
362end do obsloop2
363
364! if we added any obs to the sequence, write it now.
365if ( get_num_obs(obs_seq) > 0 ) call write_obs_seq(obs_seq, surface_out_file)
366
367! end of main program
368call finalize_utilities()
369
370
371end program convert_madis_mesonet
