63!------------------------------------------------------------------------------
64!> Subroutine get_obs_ens_distrib_state
65!>
66!> Computes the forward observation operators and related quality
67!> control indicators.
68!>
69!------------------------------------------------------------------------------
70subroutine get_obs_ens_distrib_state(ens_handle, obs_fwd_op_ens_handle, &
71 qc_ens_handle, seq, keys, obs_val_index, input_qc_index, &
72 OBS_ERR_VAR_COPY, OBS_VAL_COPY, OBS_KEY_COPY, &
73 OBS_GLOBAL_QC_COPY, OBS_EXTRA_QC_COPY, OBS_MEAN_COPY, &
74 OBS_VAR_COPY, isprior, prior_qc_copy)
75
76type(ensemble_type), intent(inout) :: ens_handle !! state ensemble handle
77type(ensemble_type), intent(inout) :: obs_fwd_op_ens_handle !! observation forward operator handle
78type(ensemble_type), intent(inout) :: qc_ens_handle !! quality control handle
79type(obs_sequence_type), intent(in) :: seq !! the observation sequence
80integer, intent(in) :: keys(:) !! observation key numbers
81integer, intent(in) :: obs_val_index !! copy index for observation value
82integer, intent(in) :: input_qc_index !! qc index for incoming QC value
83integer, intent(in) :: obs_err_var_copy !! ensemble copy number for observation error variances
84integer, intent(in) :: obs_val_copy !! ensemble copy number for observation values
85integer, intent(in) :: obs_key_copy !! ensemble copy number for observation keys
86integer, intent(in) :: obs_global_qc_copy !! ensemble copy number for ??
87integer, intent(in) :: obs_extra_qc_copy !! ensemble copy number for ??
88integer, intent(in) :: obs_mean_copy !! ensemble copy number for obs mean values
89integer, intent(in) :: obs_var_copy !! ensemble copy number for obs variance
90logical, intent(in) :: isprior !! true for prior eval; false for posterior
91real(r8), intent(inout) :: prior_qc_copy(:) !! array instead of ensemble copy ??
92
93real(r8) :: input_qc(1), obs_value(1), obs_err_var
94
95real(r8), allocatable :: expected_obs(:) !Also regular obs now?
96
97integer :: i, j, k !! index variables
98integer :: thiskey(1)
99integer :: global_obs_num, global_qc_value
100integer :: num_copies_to_calc
101integer :: global_ens_index
102integer :: ens_size
103
104integer, allocatable :: istatus(:)
105integer, allocatable :: var_istatus(:)
106integer, allocatable :: my_copy_indices(:) ! The global ens index for each copy a task has
107
108logical :: evaluate_this_ob, assimilate_this_ob
109
110type(time_type) :: dummy_time
111
112type(obs_def_type) :: obs_def
113type(obs_type) :: observation
114
115! IMPORTANT, IT IS ASSUMED THAT ACTUAL ENSEMBLES COME FIRST
116! It is also assumed that the ensemble members are in the same
117! order in each of the handles
118
119num_copies_to_calc = copies_in_window(ens_handle)
120
121allocate(istatus(num_copies_to_calc))
122allocate(expected_obs(num_copies_to_calc))
123allocate(my_copy_indices(num_copies_to_calc))
124
125istatus = 999123
126expected_obs = missing_r8
127
128! Set up access to the state
129call create_state_window(ens_handle, obs_fwd_op_ens_handle, qc_ens_handle)
130
131ens_size = ens_handle%num_copies - ens_handle%num_extras
132
133if(get_allow_transpose(ens_handle)) then ! giant if for transpose or distributed forward op
134
135 my_copy_indices(:) = ens_handle%my_copies(1:num_copies_to_calc) ! var-complete forward operators
136
137 ! Loop through all observations in the set
138 all_observations: do j = 1, obs_fwd_op_ens_handle%num_vars
139 ! Get the information on this observation by placing it in temporary
140 call get_obs_from_key(seq, keys(j), observation)
141 call get_obs_def(observation, obs_def)
142 ! Check to see if this observation fails input qc test
143 call get_qc(observation, input_qc, input_qc_index)
144
145 ! Get the observation value and error variance
146 call get_obs_values(observation, obs_value(1:1), obs_val_index)
147 obs_err_var = get_obs_def_error_variance(obs_def)
148
149 ! Loop through all copies stored by this process and set values as needed
150 do k = 1, obs_fwd_op_ens_handle%my_num_copies
151 ! See if this is the copy for error variance or observed value
152 global_ens_index = obs_fwd_op_ens_handle%my_copies(k)
153 if(global_ens_index == obs_err_var_copy) then
154 ! This copy is the instrument observation error variance; read and
155 ! store
156 obs_fwd_op_ens_handle%vars(j, k) = obs_err_var
157
158 else if(global_ens_index == obs_val_copy) then
159 ! This copy is the observation from the instrument; read and store
160 obs_fwd_op_ens_handle%vars(j, k) = obs_value(1)
161 else if(global_ens_index == obs_key_copy) then
162 obs_fwd_op_ens_handle%vars(j, k) = keys(j)
163 else if(global_ens_index == obs_global_qc_copy .and. isprior) then
164 obs_fwd_op_ens_handle%vars(j, k) = 0.0_r8
165 endif
166 enddo
167
168 ! If it is bad, set forward operator status value to -99 and return missing_r8 for obs_value
169 ! PAR THIS SUBROUTINE SHOULD EVENTUALLY GO IN THE QUALITY CONTROL MODULE
170 if(.not. input_qc_ok(input_qc(1), global_qc_value)) then
171
172 qc_ens_handle%vars(j, :) = 0
173
174 do k=1, obs_fwd_op_ens_handle%my_num_copies
175 global_ens_index = obs_fwd_op_ens_handle%my_copies(k)
176 ! Update prior/post obs values, mean, etc - but leave the key copy
177 ! and the QC copy alone.
178 if ((global_ens_index /= obs_key_copy) .and. &
179 (global_ens_index /= obs_extra_qc_copy) .and. &
180 (global_ens_index /= obs_global_qc_copy)) then
181 ! JH test if the global_ens_index is our extra bad qc copy
182 obs_fwd_op_ens_handle%vars(j, k) = missing_r8
183 endif
184 if (global_ens_index == obs_global_qc_copy) then
185 ! JH set extra fwd_op_ens to bad incoming qc
186 obs_fwd_op_ens_handle%vars(j, k) = global_qc_value
187 endif
188 enddo
189
190 ! No need to do anything else for a failed observation
191 cycle all_observations
192 endif
193
194 thiskey(1) = keys(j)
195
196 if(qc_ens_handle%my_num_copies > 0) then
197 call get_expected_obs_distrib_state(seq, thiskey, &
198 dummy_time, isprior, istatus, &
199 assimilate_this_ob, evaluate_this_ob, ens_handle, num_copies_to_calc, my_copy_indices, expected_obs)
200 obs_fwd_op_ens_handle%vars(j, 1:num_copies_to_calc) = expected_obs
201 else ! need to know whether it was assimilate or evaluate this ob.
202
203 ! TJH istatus is not set in here, yet it is in the other half of the if statement.
204 ! TJH Consequently, initializing it after it gets allocated.
205 call assim_or_eval(seq, thiskey(1), ens_handle%num_vars, assimilate_this_ob, evaluate_this_ob)
206
207 endif
208
209 do k=1, obs_fwd_op_ens_handle%my_num_copies
210 global_ens_index = obs_fwd_op_ens_handle%my_copies(k)
211
212 ! Storing assimilate_this_obs, evaluate_this_ob, or ob not in
213 ! namelist in OBS_EXTRA_QC_COPY
214 if (global_ens_index == obs_extra_qc_copy) then
215 if (assimilate_this_ob) then
216 obs_fwd_op_ens_handle%vars(j, k) = dartqc_assim_good_fop
217 else if (evaluate_this_ob) then
218 obs_fwd_op_ens_handle%vars(j, k) = dartqc_eval_good_fop
219 else
220 obs_fwd_op_ens_handle%vars(j, k) = dartqc_not_in_namelist
221 endif
222 endif
223 enddo
224
225 ! TJH if qc_ens_handle%my_num_copies <= 0) istatus was not defined
226 qc_ens_handle%vars(j, 1:num_copies_to_calc) = istatus(:)
227
228 call check_forward_operator_istatus(num_copies_to_calc, assimilate_this_ob, &
229 evaluate_this_ob, istatus, expected_obs, keys(j))
230
231 end do all_observations
232
233else ! distributed state
234
235 do i = 1, num_copies_to_calc
236 my_copy_indices(i) = i ! copy-complete fwd operator so indices are 1 to ens_size
237 enddo
238
239 ! Loop through all my observations in the set
240 my_observations: do j = 1, obs_fwd_op_ens_handle%my_num_vars
241
242 ! convert the local obs number to global obs number
243 global_obs_num = obs_fwd_op_ens_handle%my_vars(j)
244 thiskey(1) = keys(global_obs_num)
245
246 call get_obs_from_key(seq, keys(global_obs_num), observation)
247 call get_obs_def(observation, obs_def)
248
249 ! Get the information on this observation by placing it in temporary
250 ! storage. Check to see if this observation fails input qc test
251 call get_qc(observation, input_qc, input_qc_index)
252
253 if (isprior) then
254 ! this should only need to be done once, in the prior, right?
255 obs_fwd_op_ens_handle%copies(obs_key_copy, j) = thiskey(1)
256
257 ! Check to see if the incoming data qc is bad
258 if(.not. input_qc_ok(input_qc(1), global_qc_value)) then
259 obs_fwd_op_ens_handle%copies(obs_global_qc_copy, j) = global_qc_value
260 obs_fwd_op_ens_handle%copies(1:ens_size,j) = missing_r8
261 ! No need to do anything else for bad incoming qc
262 cycle my_observations
263 endif
264 else ! posterior
265 ! Check to see if the incoming data qc is bad
266 if(.not. input_qc_ok(input_qc(1), global_qc_value)) then
267 cycle my_observations
268 endif
269 endif
270
271 ! Get the observation value and error variance
272 call get_obs_values(observation, obs_value(1:1), obs_val_index)
273
274 obs_err_var = get_obs_def_error_variance(obs_def)
275
276 call get_expected_obs_distrib_state(seq, thiskey, &
277 dummy_time, isprior, istatus, &
278 assimilate_this_ob, evaluate_this_ob, ens_handle, num_copies_to_calc, my_copy_indices, expected_obs)
279
280 obs_fwd_op_ens_handle%copies(1:num_copies_to_calc, j) = expected_obs
281
282 ! collect dart qc
283 global_qc_value = nint(obs_fwd_op_ens_handle%copies(obs_global_qc_copy, j))
284
285 call get_dart_qc(istatus, num_copies_to_calc, assimilate_this_ob, evaluate_this_ob, &
286 isprior, global_qc_value)
287
288 ! update the dart qc, error variance and for observed value
289 obs_fwd_op_ens_handle%copies(obs_global_qc_copy, j) = global_qc_value
290 obs_fwd_op_ens_handle%copies(obs_err_var_copy , j) = obs_err_var
291 obs_fwd_op_ens_handle%copies(obs_val_copy , j) = obs_value(1)
292
293 qc_ens_handle%copies(:, j) = istatus
294
295 call check_forward_operator_istatus(num_copies_to_calc, assimilate_this_ob, evaluate_this_ob, &
296 istatus, expected_obs, thiskey(1))
297
298 end do my_observations
299
300endif
301
302! End access to the state
303call free_state_window(ens_handle, obs_fwd_op_ens_handle, qc_ens_handle)
304
305
306! QC Section:
307! * Consolidate QC for non-distributed forward operator
308! * Check outlier threshold (prior only)
309! * Set mean and var to missing_r8 if any forward operator failed.
310! The failure test is any qc_ens_handle%copies = 0.
311! Previously we were testing for any obs_fwd_op_ens_handle%copies = missing_r8 but
312! some models (e.g. CAM) return a non-zero QC, but a forward operator value (not missing_r8)
313
314if (get_allow_transpose(ens_handle)) then
315 ! Extra step for non-distributed to consolidate qc values for
316 ! each ensemble member into global_qc_value
317 allocate(var_istatus(qc_ens_handle%num_copies))
318
319 my_obs: do j = 1, obs_fwd_op_ens_handle%my_num_vars
320 ! collect dart qc
321 var_istatus = qc_ens_handle%copies(:,j)
322
323 assimilate_this_ob = (obs_fwd_op_ens_handle%copies(obs_extra_qc_copy, j) == dartqc_assim_good_fop )
324 evaluate_this_ob = (obs_fwd_op_ens_handle%copies(obs_extra_qc_copy, j) == dartqc_eval_good_fop)
325 global_qc_value = nint(obs_fwd_op_ens_handle%copies(obs_global_qc_copy, j))
326
327 ! update the status
328 call get_dart_qc(var_istatus, qc_ens_handle%num_copies, assimilate_this_ob, evaluate_this_ob, &
329 isprior, global_qc_value)
330 obs_fwd_op_ens_handle%copies(obs_global_qc_copy, j) = global_qc_value
331
332 enddo my_obs
333
334 deallocate(var_istatus)
335
336endif
337
338call compute_copy_mean_var(obs_fwd_op_ens_handle, 1, qc_ens_handle%num_copies, &
339 obs_mean_copy, obs_var_copy)
340
341!>@todo Science quesion: Should groups be considered separately for outlier threshold test?
342qc_loop: do j = 1, obs_fwd_op_ens_handle%my_num_vars
343
344 if (isprior) then
345
346 global_qc_value = nint(obs_fwd_op_ens_handle%copies(obs_global_qc_copy, j))
347 ! check_outlier_threshold could change the global_qc_value
348 call check_outlier_threshold(obs_fwd_op_ens_handle%copies(obs_mean_copy, j), &
349 obs_fwd_op_ens_handle%copies(obs_var_copy, j), &
350 obs_fwd_op_ens_handle%copies(obs_val_copy, j), &
351 obs_fwd_op_ens_handle%copies(obs_err_var_copy, j), seq, &
352 nint(obs_fwd_op_ens_handle%copies(obs_key_copy, j)), global_qc_value)
353
354 obs_fwd_op_ens_handle%copies(obs_global_qc_copy, j) = global_qc_value
355
356 endif
357
358 ! for either prior or posterior, if any forward operator failed,
359 ! reset the mean/var to missing_r8, regardless of the DART QC status
360 if (any(qc_ens_handle%copies(:, j) /= 0)) then
361 obs_fwd_op_ens_handle%copies(obs_mean_copy, j) = missing_r8
362 obs_fwd_op_ens_handle%copies(obs_var_copy, j) = missing_r8
363 endif
364
365end do qc_loop
366
367if (isprior) call get_single_copy(obs_fwd_op_ens_handle, obs_global_qc_copy, prior_qc_copy)
368
369deallocate(expected_obs, istatus, my_copy_indices)
370
371end subroutine get_obs_ens_distrib_state
372
373!------------------------------------------------------------------------------
374!> Subroutine get_expected_obs_distrib_state
375!>
376!> Compute forward operator for set of obs in sequence for distributed state vector.
377!>@todo does this need to be for a set of obs?
378!>
379!------------------------------------------------------------------------------
380subroutine get_expected_obs_distrib_state(seq, keys, state_time, isprior, &
381 istatus, assimilate_this_ob, evaluate_this_ob, state_ens_handle, num_ens, copy_indices, expected_obs)
382
383type(obs_sequence_type), intent(in) :: seq !! the observation sequence
384integer, intent(in) :: keys(:) !! list of obs numbers
385type(time_type), intent(in) :: state_time !! time state vector data is valid for
386logical, intent(in) :: isprior !! true if prior; false if posterior
387integer, intent(in) :: num_ens !! number of ensemble members (decl must preceed use)
388integer, intent(out) :: istatus(num_ens) !! FO return codes; 0=ok, >0 is error, <0 reserved for system use
389logical, intent(out) :: assimilate_this_ob !! true if assimilated; false otherwise
390logical, intent(out) :: evaluate_this_ob !! true if evaluated; false otherwise
391type(ensemble_type), intent(in) :: state_ens_handle !! state ensemble handle
392integer, intent(in) :: copy_indices(num_ens) !! ??
393real(r8), intent(inout) :: expected_obs(num_ens) !! the array of computed forward operator values
394
395type(obs_type) :: obs
396type(obs_def_type) :: obs_def
397character(len=32) :: state_size_string, obs_key_string, identity_obs_string
398
399integer :: obs_kind_ind
400integer :: num_obs, i
401
402num_obs = size(keys)
403
404! NEED to initialize istatus to okay value
405istatus = 0
406
407! Initialize the observation type
408!!! Can actually init with the correct size here if wanted
409call init_obs(obs, 0, 0)
410
411!>@todo do you ever use this with more than one obs?
412do i = 1, num_obs
413 call get_obs_from_key(seq, keys(i), obs)
414 call get_obs_def(obs, obs_def)
415
416 obs_kind_ind = get_obs_def_type_of_obs(obs_def)
417
418 !location = get_obs_def_location(obs_def)
419
420 ! Check in kind for negative for identity obs
421 if(obs_kind_ind < 0) then
422 if ( -obs_kind_ind > state_ens_handle%num_vars ) then
423 write(state_size_string, *) state_ens_handle%num_vars
424 write(obs_key_string, *) keys(i)
425 write(identity_obs_string, *) -obs_kind_ind
426 write(string1, *) 'unable to compute forward operator for obs number '//trim(adjustl(obs_key_string))
427 write(string2, *) 'identity index '//trim(adjustl(identity_obs_string))//&
428 ' must be between 1 and the state size of '//trim(adjustl(state_size_string))
429 call error_handler(e_err, 'get_expected_obs', string1, source, text2=string2)
430 endif
431
432 expected_obs = get_state(-1*int(obs_kind_ind,i8), state_ens_handle)
433
434 ! FIXME : we currently have no option to eval only identity obs,
435 ! or select to skip their assimilation via namelist.
436 assimilate_this_ob = .true.; evaluate_this_ob = .false.
437
438 else ! do forward operator for this kind
439
440 call get_expected_obs_from_def_distrib_state(state_ens_handle, num_ens, copy_indices, keys(i), &
441 obs_def, obs_kind_ind, state_time, isprior, &
442 assimilate_this_ob, evaluate_this_ob, expected_obs, istatus)
443
444 endif
445end do
446
447! need to free any observation specific storage that
448! might have been allocated.
449call destroy_obs(obs)
450
451end subroutine get_expected_obs_distrib_state

493!------------------------------------------------------------------------------
494!> Checks for errors from the model_mod forward operator
495!> Possible errors:
496!> * Successful istatus but missing_r8 for forward operator
497!> * Negative istatus
498!> This routine calls the error handler (E_ERR) if either of these happen.
499
500subroutine check_forward_operator_istatus(num_fwd_ops, assimilate_ob, evaluate_ob, istatus, expected_obs, thiskey)
501
502integer, intent(in) :: num_fwd_ops
503logical, intent(in) :: assimilate_ob
504logical, intent(in) :: evaluate_ob
505integer, intent(in) :: istatus(num_fwd_ops)
506real(r8), intent(in) :: expected_obs(num_fwd_ops)
507integer, intent(in) :: thiskey
508
509integer :: copy
510
511! Check for errors from model_mod forward operator
512do copy = 1, num_fwd_ops
513
514 ! Successful istatus but missing_r8 for forward operator
515 if(istatus(copy) == 0) then
516 if ((assimilate_ob .or. evaluate_ob) .and. (expected_obs(copy) == missing_r8)) then
517 write(string1, *) 'istatus was 0 (OK) but forward operator returned missing value.'
518 write(string2, *) 'observation number ', thiskey
519 call error_handler(e_err,'check_forward_operator_istatus', string1, &
520 source, text2=string2)
521 endif
522 ! Negative istatus
523 else if (istatus(copy) < 0) then
524 write(string1, *) 'istatus must not be <0 from forward operator. 0=OK, >0 for error'
525 write(string2, *) 'observation number ', thiskey
526 call error_handler(e_err,'check_forward_operator_istatus', string1, &
527 source, text2=string2)
528 endif
529
530enddo
531
532end subroutine check_forward_operator_istatus
