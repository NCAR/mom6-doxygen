18!> Check the return code from a pnetcdf call and print out information
19!> if the call failed.
20subroutine pnet_check(istatus, subr_name, context)
21
22integer, intent(in) :: istatus !! pnetcdf return code
23character(len=*), intent(in) :: subr_name
24character(len=*), intent(in), optional :: context
25
26character(len=129) :: error_msg
27
28! if no error, nothing to do here. we are done.
29if( istatus == nf_noerr ) return
30
31! something wrong. construct an error string and call the handler.
32
33! context is optional, but is very useful if specified.
34! if context + error code > 129, the assignment will truncate.
35if (present(context) ) then
36 error_msg = trim(context) // ': ' // trim(nfmpi_strerror(istatus))
37else
38 error_msg = nfmpi_strerror(istatus)
39endif
40
41! this does not return
42call error_mesg(subr_name, error_msg, fatal)
43
