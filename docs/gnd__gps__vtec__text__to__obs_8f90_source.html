12! in a text-based dataset - either white-space separated values or

42! things which can/should be in the gnd_gps_vtec_text_to_obs_nml
43
44character(len=64) :: text_input_file = 'Density_3deg_02_335_2p2.ascii'
45character(len=64) :: obs_out_file = 'obs_seq.out'
46logical :: debug = .true. ! set to .true. to print info
47
48namelist /gnd_gps_vtec_text_to_obs_nml/ &
49 text_input_file, &
50 obs_out_file, &
51 debug
52
53character (len=200) :: input_line ! GPS TEC data is 153 wide, but give myself some room
54
55integer :: oday, osec, rcio, iunit
56integer :: year, month, day, hour, minute, second
57integer :: num_copies, num_qc, max_obs
58
59logical :: file_exist, first_obs
60
61real(r8) :: temp, terr, qc
62real(r8) :: lat, lon, vert
63
64type(obs_sequence_type) :: obs_seq
65type(obs_type) :: obs, prev_obs
66type(time_type) :: time_obs, prev_time
67
68integer :: ignore_i ! integers we read but don't use
69real(r8) :: mi,ma ! calculate the min and max of the dataset on the fly
70
71! start of executable code
72
73call initialize_utilities('gnd_gps_vtec_text_to_obs')
74
75write(*,*)
76write(*,*) "WARNING, BE CAREFUL: the TEC files from different dates might have different number of columns!"
77write(*,*) "For example, on some dates in 2002 there are 13 columns, whereas in 2006 there are 10 "
78write(*,*) "If you are getting some weird data/results, this could be the issue "
79write(*,*) "To fix it, change ../gnd_gps_vtec_text_to_obs.f90:line~175 and recompile with ./quickbuild.sh"
80write(*,*)
81
82! Set the DART data quality control. 0 is good data.
83! increasingly larger QC values are more questionable quality data.
84qc = 0.0_r8
85mi = 1.0e32 ! set initial minimum as large positive
86ma =-1.0e32 ! set initial maximum as large negative
87
88! each observation in this series will have a single observation value
89! and a quality control flag. the max possible number of obs needs to
90! be specified but it will only write out the actual number created.
91max_obs = 10000000
92num_copies = 1
93num_qc = 1
94
95! time setup
96call set_calendar_type(gregorian)
97
98! Read the DART namelist for this model
99call find_namelist_in_file('input.nml', 'gnd_gps_vtec_text_to_obs_nml', iunit)
100read(iunit, nml = gnd_gps_vtec_text_to_obs_nml, iostat = rcio)
101call check_namelist_read(iunit, rcio, 'gnd_gps_vtec_text_to_obs_nml')
102
103! call the initialization code, and initialize two empty observation types
104call static_init_obs_sequence()
105call init_obs(obs, num_copies, num_qc)
106call init_obs(prev_obs, num_copies, num_qc)
107first_obs = .true.
108
109! create a new, empty obs_seq file. you must give a max limit
110! on number of obs. increase the size if too small.
111call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
112
113! the first one needs to contain the string 'observation' and the
114! second needs the string 'QC'.
115call set_copy_meta_data(obs_seq, 1, 'observation')
116call set_qc_meta_data(obs_seq, 1, 'Data QC')
117
118! if you want to append to existing files (e.g. you have a lot of
119! small text files you want to combine), you can do it this way,
120! or you can use the obs_sequence_tool to merge a list of files
121! once they are in DART obs_seq format.
122
123! ! existing file found, append to it
124! inquire(file=obs_out_file, exist=file_exist)
125! if ( file_exist ) then
126! call read_obs_seq(obs_out_file, 0, 0, max_obs, obs_seq)
127! endif
128
129! open input text file
130iunit = open_file(text_input_file, 'formatted', 'read')
131if (debug) print *, 'opened input file ' // trim(text_input_file)
132
133read(iunit, "(A)", iostat=rcio) input_line ! skip the first line as it is a header
134
135obsloop: do ! FIXME no end limit - have the loop break when input ends
136
137 ! read in a line from the text file. What you need to create an obs:
138 ! location: lat, lon, and height in pressure or meters
139 ! time: when the observation was taken
140 ! type: from the DART list of obs types
141 ! error: very important - the instrument error plus representativeness error
142 ! (see html file for more info)
143
144 ! read the whole line
145 read(iunit, "(A)", iostat=rcio) input_line
146
147 if (rcio /= 0) then
148 if (debug) print *, 'got bad read code from input file, rcio = ', rcio
149 exit obsloop
150 endif
151
152 !BE CAREFUL: the files from different years have different number of columns!
153 !for example, on some dates in 2002 it is 13, in 2006 it is 10
154
155 ! assume here is a line from http://madrigal.haystack.mit.edu/cgi-bin/madrigal/displayCatHead.py?expName=experiments/
156 !2002/gps/30nov02&fileName=gps021130g.002&expTitle=World-wide%20Vertical%20Total%20Electron%20Content&displayLevel=0
157 !here is the header
158 !YEAR MONTH DAY HOUR MIN SEC UT1_UNIX UT2_UNIX RECNO GDLAT GLON TEC DTEC
159 !here is a sample line (white space removed by hand)
160 !2002 12 1 0 2 30 1038700800 1038701100 2 -82.00 -178.00 8.40000e+00 5.00000e+00
161
162 read(input_line, *, iostat=rcio) year, month, day, hour, minute, second, &
163 ignore_i, ignore_i, ignore_i, lat, lon, temp, terr
164
165 !temp is the TEC value
166 !terr is the STANDARD DEVIATION (measured in TECU) uncertainty associated with that measurement
167
168 ! replacing "harsh" exit with a "soft" cycle because GPS data has a lot of "missing" values
169 ! which we simply skip (i.e. throw away)
170 if (rcio /= 0) then
171 if (debug) print *, 'got bad read code getting rest of temp obs, rcio = ', rcio
172 cycle obsloop
173 endif
174
175 if (debug) print *, 'next observation located at lat, lon = ', lat, lon
176
177 ! check the lat/lon values to see if they are ok
178 if ( lat >= 90.0_r8 .or. lat <= -90.0_r8 ) cycle obsloop
179 if ( lon > 180.0_r8 .or. lon < -180.0_r8 ) cycle obsloop
180 if ( lon < 0.0_r8 ) lon = lon + 360.0_r8 ! changes into 0-360
181
182 ! after all checks were done on this data point, update the min and max
183 mi = min(mi, temp)
184 ma = max(ma, temp)
185
186 ! put date into a dart time format
187 time_obs = set_date(year, month, day, hour, minute, second)
188
189 if (debug) call print_date(time_obs, 'next obs time is')
190
191 ! extract time of observation into gregorian day, sec.
192 call get_time(time_obs, osec, oday)
193
194 !height is tricky since ground TEC measurement is an integral over height, so it has no height. I'll set it to
195 !350 000m, but in model_mod I'll make TEC measurement localized only in horizontal (no vertical).
196
197 vert=350000.0
198 call create_3d_obs(lat, lon, vert, vertisheight, temp, &
199 gnd_gps_vtec, terr, oday, osec, qc, obs)
200 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
201 if (debug) print *, 'added an obs to output seq, now the total count is', get_num_obs(obs_seq)
202
203enddo obsloop
204
205! if we added any obs to the sequence, write it out to a file now.
206if ( get_num_obs(obs_seq) > 0 ) then
207 if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
208 print *, 'writing obs_seq, obs_count, min, max = ', get_num_obs(obs_seq), mi, ma
209 call write_obs_seq(obs_seq, obs_out_file)
210endif
211
212! end of main program
213call finalize_utilities()
214
215end program gnd_gps_vtec_text_to_obs
