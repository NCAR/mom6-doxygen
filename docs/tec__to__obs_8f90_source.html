7! An example program to create DART format observations from a tec input file
8!
9! An example program that reads tec (ascii) input lines and creates
10! DART observations from them. This is more of a template program.
11! You will have to modify it for your own use. In particular
12! you will have to adapt the read routine to match your white-space
13! separated values or fixed-width column data.
14!
15! This program now includes a small namelist. The items in the namelist
16! can be changed at runtime without having to recompile the program.
17! Having the input and output filenames in the namelist should make it
18! simpler to script the conversion of a series of files.
19!
20! history:
21! created 29 Mar 2010 nancy collins NCAR/IMAGe
22! updated 8 Feb 2016 minor changes to wind conversion, and updated
23! comments to try to be more helpful.
24! updated 29 Aug 2019 added namelist with example values
25!
26
27program tec_to_obs
28
29use types_mod, only : r8, pi, deg2rad
30use utilities_mod, only : initialize_utilities, finalize_utilities, &
31 open_file, close_file, &
32 find_namelist_in_file, check_namelist_read, &
33 error_handler, e_err, e_msg, nmlfileunit, &
34 do_nml_file, do_nml_term
35use time_manager_mod, only : time_type, set_calendar_type, set_date, &
36 operator(>=), increment_time, get_time, &
37 operator(-), gregorian, operator(+), print_date
38use location_mod, only : vertisheight, vertispressure
39use obs_sequence_mod, only : obs_sequence_type, obs_type, read_obs_seq, &
40 static_init_obs_sequence, init_obs, write_obs_seq, &
41 init_obs_sequence, get_num_obs, &
42 set_copy_meta_data, set_qc_meta_data
43
44use obs_utilities_mod, only : create_3d_obs, add_obs_to_seq
45
46! Change these to the actual types of observations you have.
47use obs_kind_mod, only : eval_u_wind_component, eval_v_wind_component, &
48 eval_temperature
49
50implicit none
51
52! variables that can be changed at runtime from a namelist
53! add any other options here that would be useful for your application.
54character(len=256) :: tec_input_file = '../data/tec_input_file'
55character(len=256) :: obs_out_file = 'obs_seq.out'
56logical :: debug = .false. ! set to .true. to print info
57
58namelist /tec_to_obs_nml/ tec_input_file, obs_out_file, debug
59
60
61! local variables
62character (len=129) :: input_line
63
64integer :: oday, osec, rcio, iunit, otype
65integer :: year, month, day, hour, minute, second
66integer :: num_copies, num_qc, max_obs
67
68logical :: file_exist, first_obs
69
70real(r8) :: temp, terr, qc, wdir, wspeed, werr
71real(r8) :: lat, lon, vert, uwnd, uerr, vwnd, verr
72
73type(obs_sequence_type) :: obs_seq
74type(obs_type) :: obs, prev_obs
75type(time_type) :: ref_day0, time_obs, prev_time
76
77
78! start of executable code
79
80call initialize_utilities('tec_to_obs')
81
82! Read the namelist entry
83call find_namelist_in_file("input.nml", "tec_to_obs_nml", iunit)
84read(iunit, nml = tec_to_obs_nml, iostat = rcio)
85call check_namelist_read(iunit, rcio, "tec_to_obs_nml")
86
87! Record the namelist values used for the run
88if (do_nml_file()) write(nmlfileunit, nml=tec_to_obs_nml)
89if (do_nml_term()) write( * , nml=tec_to_obs_nml)
90
91
92! time setup
93call set_calendar_type(gregorian)
94
95!! Some times are supplied as number of seconds since some reference
96!! date. To support that, set a base/reference time and then add
97!! the number of seconds to it. (time_types support adding two time
98!! types together, or adding a scalar to a time_type.)
99!! Here is an example of setting a reference date, giving the
100!! call: year, month, day, hours, mins, secs
101!ref_day0 = set_date(1970, 1, 1, 0, 0, 0)
102
103! open input tec file
104
105iunit = open_file(tec_input_file, 'formatted', 'read')
106if (debug) print *, 'opened input file ' // trim(tec_input_file)
107
108
109! each observation in this series will have a single observation value
110! and a quality control flag. the max possible number of obs needs to
111! be specified but it will only write out the actual number created.
112max_obs = 100000
113num_copies = 1
114num_qc = 1
115
116! call the initialization code, and initialize two empty observation types
117call static_init_obs_sequence()
118call init_obs(obs, num_copies, num_qc)
119call init_obs(prev_obs, num_copies, num_qc)
120first_obs = .true.
121
122! create a new, empty obs_seq file. you must give a max limit
123! on number of obs. increase the size if too small.
124call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
125
126! the first one needs to contain the string 'observation' and the
127! second needs the string 'QC'.
128call set_copy_meta_data(obs_seq, 1, 'observation')
129call set_qc_meta_data(obs_seq, 1, 'Data QC')
130
131! if you want to append to existing files (e.g. you have a lot of
132! small tec files you want to combine), you can do it this way,
133! or you can use the obs_sequence_tool to merge a list of files
134! once they are in DART obs_seq format.
135
136! ! existing file found, append to it
137! inquire(file=obs_out_file, exist=file_exist)
138! if ( file_exist ) then
139! call read_obs_seq(obs_out_file, 0, 0, max_obs, obs_seq)
140! endif
141
142! Set the DART data quality control. 0 is good data.
143! increasingly larger QC values are more questionable quality data.
144qc = 0.0_r8
145
146obsloop: do ! no end limit - have the loop break when input ends
147
148 ! read in a line from the tec file. What you need to create an obs:
149 ! location: lat, lon, and height in pressure or meters
150 ! time: when the observation was taken
151 ! type: from the DART list of obs types
152 ! error: very important - the instrument error plus representativeness error
153 ! (see html file for more info)
154
155 ! this code here assumes an input line has: a type (1/2), location, time, value, obs error
156 ! **this is where you will need to adapt this code for your input data values**
157
158 ! read in entire tec line into a buffer
159 read(iunit, "(A)", iostat=rcio) input_line
160 if (rcio /= 0) then
161 if (debug) print *, 'got bad read code from input file, rcio = ', rcio
162 exit obsloop
163 endif
164
165 ! pull off the first 2 columns as an integer, to decode the type
166 read(input_line, "(I2)", iostat=rcio) otype
167 if (rcio /= 0) then
168 if (debug) print *, 'got bad read code trying to get obs type, rcio = ', rcio
169 exit obsloop
170 endif
171
172 if (debug) print *, 'next observation type = ', otype
173
174 ! for this example, assume there is an obs type, where otype=1 is
175 ! temperature measured at a location and a vertical height,
176 ! and if otype=2, a wind speed and direction in degrees (0-360)
177 ! measured at a location and a vertical pressure.
178
179 if (otype == 1) then
180 read(input_line(3:129), *, iostat=rcio) lat, lon, vert, &
181 year, month, day, hour, minute, second, &
182 temp, terr
183 if (rcio /= 0) then
184 if (debug) print *, 'got bad read code getting rest of temp obs, rcio = ', rcio
185 exit obsloop
186 endif
187 else
188 read(input_line(3:129), *, iostat=rcio) lat, lon, vert, &
189 year, month, day, hour, minute, second, &
190 wspeed, wdir, werr
191 if (rcio /= 0) then
192 if (debug) print *, 'got bad read code getting rest of wind obs, rcio = ', rcio
193 exit obsloop
194 endif
195 endif
196
197 if (debug) print *, 'next observation located at lat, lon = ', lat, lon
198
199 ! check the lat/lon values to see if they are ok. we require
200 ! longitudes to be 0 to 360.
201 if ( lat > 90.0_r8 .or. lat < -90.0_r8 ) cycle obsloop
202 if ( lon < 0.0_r8 .or. lon > 360.0_r8 ) cycle obsloop
203
204
205 ! if lon comes in between -180 and 180, use these lines instead:
206 !if ( lon > 180.0_r8 .or. lon < -180.0_r8 ) cycle obsloop
207 !if ( lon < 0.0_r8 ) lon = lon + 360.0_r8 ! changes into 0-360
208
209 ! put date into a dart time format
210 time_obs = set_date(year, month, day, hour, minute, second)
211
212 if (debug) call print_date(time_obs, 'next obs time is')
213
214 !! if time is given in seconds since some date, here's how to add it.
215 !time_obs = ref_day0 + time_obs
216
217 ! extract time of observation into gregorian day, sec.
218 call get_time(time_obs, osec, oday)
219
220 ! this example assumes there is an obs type, where otype=1 is
221 ! a temperature measured in height, and if otype=2, a wind speed
222 ! and direction with a vertical value in pressure. any observation
223 ! can use any of the vertical types; this is just an example.
224
225 if (otype == 1) then
226
227 ! height is in meters (gph)
228
229 ! make an obs derived type, and then add it to the sequence
230 call create_3d_obs(lat, lon, vert, vertisheight, temp, &
231 eval_temperature, terr, oday, osec, qc, obs)
232 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
233
234 if (debug) print *, 'added temperature obs to output seq'
235 else
236
237 ! DART assimilates wind as 2 separate U and V components. assimilating
238 ! "direction and speed" is difficult because direction is measured in
239 ! cyclic coordinates so you can't do simple statistics to get a mean value.
240
241 ! convert a wind speed & direction into the U and V components
242 ! and create 2 obs for it. assume vert is in mb or hectopascals,
243 ! convert to pascals. DART assumes all pressures are in pascals.
244 ! check your data source; usually wind direction is specified as the
245 ! direction the wind is coming from, increasing degrees going in a
246 ! clockwise circle. U and V wind components have the opposite sign.
247 uwnd = sin(wdir * deg2rad) * wspeed * (-1.0_r8)
248 vwnd = cos(wdir * deg2rad) * wspeed * (-1.0_r8)
249 uerr = werr
250 verr = werr
251
252 ! convert hectopascals to pascals.
253 vert = vert * 100.0_r8
254
255 call create_3d_obs(lat, lon, vert, vertispressure, uwnd, &
256 eval_u_wind_component, uerr, oday, osec, qc, obs)
257 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
258
259 call create_3d_obs(lat, lon, vert, vertispressure, vwnd, &
260 eval_v_wind_component, verr, oday, osec, qc, obs)
261 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
262
263 if (debug) print *, 'added 2 wind obs to output seq'
264 endif
265
266end do obsloop
267
268! if we added any obs to the sequence, write it out to a file now.
269if ( get_num_obs(obs_seq) > 0 ) then
270 if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
271 call write_obs_seq(obs_seq, obs_out_file)
272endif
273
274! end of main program
275call finalize_utilities()
276
277end program tec_to_obs
