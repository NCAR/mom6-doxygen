7program convert_tpw
8
9! convert MODIS observations of Total Precipitable Water into
10! DART observation sequence files. since the original observations
11! are very dense, this program has options to bin both in time and
12! in space. THIS CODE ASSUMES ALL ELEVATIONS ARE AT 0.0 meters.
13! this is true for all ocean obs. if you have land obs, you will
14! have to get the hght of the surface elevation at the lat/lon and
15! use that instead.
16!
17! assumes data is in text files which match the following read line:
18! read(iunit, '(f11.6, f13.5, f10.4, 4x, i4, 4i3, f7.3)') &
19! lat, lon, tpw, iyear, imonth, iday, ihour, imin, seconds
20!
21! constructs a input filename based on:
22! ObsBase/InfilePrefix + YYYYMMDD + InfileSuffix
23!
24! constructs a output filename based on:
25! ./OutfilePrefix + YYYYMMDD + OutfileSuffix
26!
27! any of the prefix or suffixes can be '' (blank)
28!
29! FIXME:
30! cannot loop past month boundaries. also does not handle time windows
31! that span day boundaries, in the sense that it reads one day at a time
32! and outputs one obs_seq file for each day. if a time bin crosses the
33! day boundary it should read in the next day, keep binning the obs for
34! the rest of the current time bin, and write out an obs_seq file that
35! has a full bin of all available observations. the current code starts
36! at 0Z and ends at 0Z and so you should construct your time bins carefully
37! around the day boundaries.

158! Initialize the obs_sequence module
159
160call static_init_obs_sequence()

163! Read the namelist
164!----------------------------------------------------------------------
165
166call find_namelist_in_file('input.nml', 'convert_tpw_nml', iunit)
167read(iunit, nml = convert_tpw_nml, iostat = io)
168call check_namelist_read(iunit, io, 'convert_tpw_nml')
169
170! Record the namelist values used for the run ...
171if (do_nml_file()) write(nmlfileunit, nml=convert_tpw_nml)
172if (do_nml_term()) write( * , nml=convert_tpw_nml)
173
174! some simple error checking - if you specify either of the longitude
175! limits, since it's a cyclic value you have to specify both. the included
176! area will start at min_lon and go east until max_lon. for latitude
177! you can let one default if you only want to set a min or a max.
178if (((min_lon /= missing_r8) .and. (max_lon == missing_r8)) .or. &
179 ((min_lon == missing_r8) .and. (max_lon /= missing_r8))) then
180 write(msgstring, *) 'if you set the min or max limit on longitude you must specify both'
181 call error_handler(e_err,'convert_tpw', msgstring, source, revision, revdate)
182endif
183
184! convert a string into an observation type number
185obstype = get_index_for_type_of_obs(observation_name)
186if (obstype < 0) then
187 write(msgstring, *) 'unrecognized observation type ', trim(observation_name)
188 call error_handler(e_err,'convert_tpw', msgstring, source, revision, revdate, &
189 text2='check the obs_defs in your &preprocess list')
190endif
191
192call set_calendar_type(gregorian)
193
194! allocate arrays which depend on input values in the namelist
195allocate(tpw_all(max_obs), this_bin(max_obs))
196
197!--------------------------------------
198
199dayloop: do idd = start_day, start_day + total_days - 1
200
201 ! set up for next obs_seq for next day
202
203 call init_obs(obs, num_copies, num_qc)
204 call init_obs(prev_obs, num_copies, num_qc)
205 first_obs = .true.
206 num_obs = 0
207
208 call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
209
210 call set_copy_meta_data(obs_seq, 1, 'MODIS observation')
211 call set_qc_meta_data(obs_seq, 1, 'MODIS QC')
212
213 ! construct the input and output filenames
214 write(obsdate, '(i4.4, i2.2, i2.2)') start_year, start_month, idd
215 write(infile, '(A)') trim(obsbase)//'/'//trim(infileprefix)//obsdate//trim(infilesuffix)
216 write(*, '(A)') 'reading file: '//trim(infile)
217 write(outfile, '(A)') trim(outfileprefix)//obsdate//trim(outfilesuffix)
218 write(*, '(A)') 'creating file: '//trim(outfile)
219
220 iunit = open_file(infile,'formatted',action='read')
221
222 nrec = 0
223
224 ! read in all observation records and fill the tpw type array
225 readloop: do
226
227 read(iunit, '(f11.6, f13.5, f10.4, 4x, i4, 4i3, f7.3)', iostat=io) &
228 lat, lon, tpw, iyear, imonth, iday, ihour, imin, seconds
229 if (io /= 0) exit readloop
230
231 nrec = nrec + 1
232
233 if(nrec > max_obs) then
234 write(msgstring, *) 'number of input observations larger than "max_obs" limit'
235 call error_handler(e_err,'convert_tpw', msgstring, source, revision, revdate, &
236 text2='increase parameter "max_obs" in namelist and rerun')
237 endif
238
239 tpw_all(nrec)%lat = lat
240 tpw_all(nrec)%lon = lon
241 tpw_all(nrec)%tpw_value = tpw
242 tpw_all(nrec)%time = set_date(iyear, imonth, iday, ihour, imin, int(seconds))
243 tpw_all(nrec)%fract_hours = ihour + real(imin)/60.0_r8 + seconds/3600.0_r8
244 tpw_all(nrec)%been_used = .false.
245
246 enddo readloop
247
248 call close_file(iunit)
249
250 write(msgstring, '(A,I12)') 'number of observations in input file is ', nrec
251 call error_handler(e_msg,'', msgstring, source, revision, revdate)
252
253! ---------------------------------------------------
254! select observations for specific time interval
255! ---------------------------------------------------
256
257 ibin = 0
258
259 timebinloop: do
260
261 bin_beg = time_bin_start + ibin*time_bin_interval - time_bin_half_width
262 bin_end = time_bin_start + ibin*time_bin_interval + time_bin_half_width
263 if (bin_beg >= time_bin_end) exit timebinloop
264
265 ibin = ibin + 1
266
267 ! make a list of which tpw index numbers are in this time bin
268
269 nbin = 0
270 nrecloop: do nr = 1, nrec
271
272 if(tpw_all(nr)%fract_hours <= bin_beg .or. tpw_all(nr)%fract_hours > bin_end) cycle nrecloop
273
274 nbin = nbin + 1
275 this_bin(nbin) = nr
276
277 enddo nrecloop
278
279 write(msgstring, '(A,F6.3,A,F6.3,A,I12)') 'num obs in time bin ', &
280 bin_beg, ' to ', bin_end, ' hrs is ', nbin
281 call error_handler(e_msg,'', msgstring, source, revision, revdate)
282
283 ! if there are no observations in this bin, go straight to the next bin
284 if (nbin == 0) cycle timebinloop
285
286 ! main loop over all observations in this time bin
287 obsloop2: do kk = 1, nbin
288
289 tpw_base = tpw_all(this_bin(kk))
290
291 ! if we have already averaged this observation in with another
292 ! then skip it.
293 if (tpw_base%been_used) cycle obsloop2
294
295 lonk = tpw_base%lon
296
297 ! this will always return true, but if the longitude box crosses the
298 ! prime meridian it will add 360 to all values on the east of the line.
299 ! when we compute an average the points will have consistent values.
300 ! (e.g. we will never average 355 and 5 and get 180 instead of 0.)
301 lon_between = is_longitude_between(lonk, lonk-delta_lon_box, &
302 lonk+delta_lon_box, newlon=nlonk)
303
304 tpw_base%lon = nlonk
305
306 ! use this observation as the base for averaging any other
307 ! close obs nearby
308 num_avg = 1
309 tpw_avg = tpw_base
310
311 ! average time by tracking the difference, in seconds, between the
312 ! times of all the obs being averaged together.
313 sec_diff = 0
314
315 ! loop over the remaining observations
316 ! do the average of the nearby observations
317 otherobs: do nn = (kk+1), nbin
318
319 tpw_next = tpw_all(this_bin(nn))
320
321 ! if we have already averaged this observation in with another
322 ! then skip it.
323 if (tpw_next%been_used) cycle otherobs
324
325 ! for latitudes you can do a simple subtraction. since longitudes
326 ! are cyclic the computation is more complex. nloni will be returned
327 ! as a value between 0 and 720 to guarentee that values in the delta_lon
328 ! region are contiguous (no -180/180 or 359/0 discontinuous vals).
329 dlat = abs(tpw_base%lat - tpw_next%lat)
330 if (dlat > delta_lat_box) cycle otherobs
331
332 lon_between = is_longitude_between(tpw_next%lon, lonk-delta_lon_box, &
333 lonk+delta_lon_box, newlon=nloni)
334 if (.not. lon_between) cycle otherobs
335
336 ! obs is close enough to average
337
338 num_avg = num_avg + 1
339 tpw_avg%lat = tpw_avg%lat + tpw_next%lat
340 tpw_avg%lon = tpw_avg%lon + nloni ! possibly + 360 if near prime meridian
341 tpw_avg%tpw_value = tpw_avg%tpw_value + tpw_next%tpw_value
342
343 ! do the average by tracking the difference in seconds between the
344 ! original time and all the times being averaged in. this is to
345 ! avoid adding a lot of large numbers of seconds together and running
346 ! into roundoff error. at the end, divide the total difference in secs
347 ! by the count and add (or subtract) that from the initial time.
348
349 ! when subtracting two time types, the time mgr always returns
350 ! positive vals, so we have to test to see if we need to add or
351 ! subtract from the running total difference.
352 time_diff = tpw_next%time - tpw_base%time
353 call get_time(time_diff, osec)
354
355 if (tpw_next%time > tpw_base%time) then
356 sec_diff = sec_diff + osec
357 else
358 sec_diff = sec_diff - osec
359 endif
360
361 ! since we have averaged this obs in with the others, remove
362 ! it from the original list so it isn't used again.
363 tpw_all(this_bin(nn))%been_used = .true.
364
365 enddo otherobs
366
367 ! compute the average values, including location and time
368
369 if (num_avg > 1) then
370 lat = tpw_avg%lat / real(num_avg)
371 lon = tpw_avg%lon / real(num_avg)
372 tpw = tpw_avg%tpw_value / real(num_avg)
373
374 ! sec_diff is the total running difference, in seconds, of all
375 ! the obs from the base time. divide by the obs count to get
376 ! the average diff from the base time.
377 sec_diff = nint(real(sec_diff) / real(num_avg))
378 if (sec_diff > 0) then
379 obs_time = increment_time(tpw_base%time, sec_diff)
380 else if (sec_diff < 0) then
381 obs_time = decrement_time(tpw_base%time, sec_diff)
382 else
383 obs_time = tpw_base%time
384 endif
385 else
386 lat = tpw_base%lat
387 lon = tpw_base%lon
388 tpw = tpw_base%tpw_value
389 obs_time = tpw_base%time
390 endif
391
392 ! at this point longitudes are between 0 and 720; subtract
393 ! 360 if > 360 to get back into range of 0 to 360.
394 if (lon > 360.0_r8) lon = lon - 360.0_r8
395
396
397 ! optionally select a range of lat/lon.
398 ! now that we have the average lat and lon, test them against the limits.
399 ! we do this test last so the location of the average lat/lon is guarenteed
400 ! to be within the bounds; possibly individual obs being averaged with this
401 ! one may be slightly out of the range depending on the box size.
402 if (min_lat /= missing_r8) then
403 if (lat .lt. min_lat) cycle obsloop2
404 endif
405 if (max_lat /= missing_r8) then
406 if (lat .gt. max_lat) cycle obsloop2
407 endif
408 if ((min_lon /= missing_r8) .and. (max_lon /= missing_r8)) then
409 if (.not. is_longitude_between(lon, min_lon, max_lon)) cycle obsloop2
410 endif
411
412
413 qc = 0.0_r8 ! quality control - all good.
414 hght = 0.0_r8 ! This only works over the ocean; over land need elevation
415 oerr = 0.5 ! observation error in cm
416 call get_time(obs_time, osec, oday)
417
418 call create_3d_obs(lat, lon, hght, vertissurface, tpw, &
419 obstype, oerr, oday, osec, qc, obs)
420 call add_obs_to_seq(obs_seq, obs, obs_time, prev_obs, prev_time, first_obs)
421 num_obs = num_obs + 1
422
423 enddo obsloop2
424
425 enddo timebinloop
426
427
428 if (num_obs > 0) then
429 write(msgstring, '(A,I12)') 'number of observations in output file is ', num_obs
430 call error_handler(e_msg,'', msgstring, source, revision, revdate)
431 call write_obs_seq(obs_seq, outfile)
432 else
433 call error_handler(e_msg,'', 'not creating output file because 0 observations found', &
434 source, revision, revdate)
435 endif
436
437 call destroy_obs(obs)
438
439enddo dayloop
440
441deallocate(tpw_all, this_bin)
442
443call finalize_utilities()
444
445end program
