7! a test of reseeding the random sequence generator, to see
8! if different seeds generate overlapping sequences of values.
9! one use case in the regular dart code is when running
10! perfect_model_obs and generating random draws of gaussian
11! error values. if the program is restarted each time between
12! separate model advances, if the same seed is used each time
13! then the errors will be the same. the code now reseeds the
14! random number generator with a value based on the state data
15! timestamp. this seed seems to generate random values for
16! observation errors which are not repeating.
17!
18! this program was also helpful in finding a bug in the
19! time_manager code with one compiler. it was generating
20! long repeating sequences of values which lead to the
21! discovery that the seconds were being rounded off to 0
22! when setting a time type and different times would result
23! in the same initial seed.

169subroutine test2(nreps, per, start_time, delta_time, unif, fname)
170 integer, intent(in) :: nreps, per

172 logical, intent(in) :: unif

179open(unit=unitnum, file=fname, action='write')

182nextseed = generate_seed(t)
183call init_random_seq(seq1, nextseed)
184
185print *, ' '
186if (calendar_name /= 'NO_CALENDAR') then
187 call print_date(t, 'start time for reseed loop')
188else
189 call print_time(t, 'start time for reseed loop')
190endif
191call print_time(delta_time, 'delta time for reseed loop')
192print *, 'reseeding every ', per, ' times through the loop'
193
194j = 1
195do i=1, nreps
196 if (unif) then
197 write(unitnum, *) random_uniform(seq1)
198 else
199 write(unitnum, *) random_gaussian(seq1, 0.0_r8, 1.0_r8)
200 endif
201 if (j == per) then
202 t = t + delta_time
203 nextseed = generate_seed(t)
204 call init_random_seq(seq1, nextseed)
205 j = 1
206 else
207 j = j + 1
208 endif
209end do
210
211if (calendar_name /= 'NO_CALENDAR') then
212 call print_date(t, 'end time for reseed loop')
213else
214 call print_time(t, 'end time for reseed loop')
215endif
216print *, ' '
217
218close(unitnum)

224! search to see if reseeding the sequence causes it to repeat
225! sequences from other seeds
226
227subroutine test3()
228
229real(r8), allocatable :: history(:)
230real(r8) :: next_val
231integer :: i, j, k, nextseed
232integer, allocatable :: seedhist(:)
233type(time_type) :: base_time, state_time, delta_time, delta_time2
234type(random_seq_type) :: seq1
235
236
237base_time = set_time(0, 148000)
238state_time = base_time
239delta_time = set_time(1)
240delta_time2 = set_time(0, 1)
241
242print *, ' '
243if (calendar_name /= 'NO_CALENDAR') then
244 call print_date(state_time, 'start time for history loop')
245else
246 call print_time(state_time, 'start time for history loop')
247endif
248call print_time(delta_time, 'delta time for history loop')
249call print_time(delta_time2, 'delta2 time for history loop')
250print *, ' '
251
252! generate 10000 consecutive rand nums
253allocate(history(10000), seedhist(10000))
254
255state_time = base_time
256nextseed = generate_seed(state_time)
257call init_random_seq(seq1, nextseed)
258
259do i=1, 10000
260 history(i) = random_uniform(seq1)
261enddo
262
263! now generate 100 values from various seeds and see if
264! they overlap in any of the previously generated vals.
265state_time = base_time + delta_time
266do i=1, 10000
267 nextseed = generate_seed(state_time)
268 call init_random_seq(seq1, nextseed)
269
270 do j=1, 100
271 next_val = random_uniform(seq1)
272 do k=1, 10000-1
273 if (history(k) == next_val) then
274 print *, 'found match, ', next_val, i, j, k
275 next_val = random_uniform(seq1)
276 print *, 'next val from ran ', next_val, ' next in seq is ', history(k+1)
277 if (history(k+1) == next_val) then
278 print *, 'found 2 matching values in a row: ', history(k), history(k+1), i, j, k
279 endif
280 endif
281 enddo
282 enddo
283
284 state_time = state_time + delta_time
285
286enddo
287
288deallocate(history)
289
290end subroutine test3
