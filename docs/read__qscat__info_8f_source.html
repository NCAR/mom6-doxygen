6c====================================================================
7c Filename: read_qscat_info.f
8c
9c Usage:
10c
11c To run this program, use the following command:
12c
13c your_computer% read_qscat_info <filename>
14c
15c where "<filename>" is the name of the QuikSCAT input file
16c
17c Description:
18c
19c This file contains 2 subroutines in order to read the
20c Hierarchical Data Format (HDF) attributes in the
21c QuikSCAT data. The subroutines are as follows.
22c
23c 1. GET_TYPE: a subroutine which returns a string
24c containing the data type in ascii given
25c the definition value (see HDF documentation
26c for conversions).
27c
28c 2. READ_ATTRIBUTE: a subroutine to read the name and
29c value(s) of a global attribute referenced
30c by its index.
31c
32c NOTES:
33c 1. Please refer all questions concerning this program and
34c QuikSCAT data obtained from the JPL PO.DAAC to
35c qscat@podaac.jpl.nasa.gov.
36c
37c 2. The HDF library must be installed before this program will
38c work properly. The HDF library and further information
39c about HDF may be obtained from the National Center for
40c Supercomputing Applications (NCSA) at http://hdf.ncsa.uiuc.edu.
41c
42c 7/1/1999 R.S. Dunbar, K.L. Perry
43c Copyright 1999, California Institute of Technology
44c
45c DART $Id$

144c GET_TYPE: a subroutine which returns a string containing the
145c data type in ascii given the definition value
146c (see HDF documentation for conversions).
147c
148c 7/22/1999 K.L. Perry
149c====================================================================
150 subroutine get_type(data_type,string_data_type)
151
152 integer data_type
153 character string_data_type*10
154
155 if (data_type.eq.3) then
156 string_data_type='uchar'
157 else if (data_type.eq.4) then
158 string_data_type='char'
159 else if (data_type.eq.5) then
160 string_data_type='float32'
161 else if (data_type.eq.20) then
162 string_data_type='int8'
163 else if (data_type.eq.21) then
164 string_data_type='uint8'
165 else if (data_type.eq.22) then
166 string_data_type='int16'
167 else if (data_type.eq.23) then
168 string_data_type='uint16'
169 else if (data_type.eq.24) then
170 string_data_type='int32'
171 else if (data_type.eq.25) then
172 string_data_type='uint32'
173 else if (data_type.eq.26) then
174 string_data_type='int64'
175 else if (data_type.eq.27) then
176 string_data_type='uint64'
177 else
178 string_data_type='UNKNOWN'
179 write(*,*) 'Abort Program: Data Type Unknown'
180 stop
181 endif
182
183 end

193 subroutine read_attribute(sd_id,attr_index,attr_name,
194 & adata_type,n_values,fvalues)
195
196 integer MAX_NC_NAME
197 parameter(max_nc_name=256)
198
199 integer sd_id,retn,adata_type,n_values
200 integer attr_index,count,retn,n,oldn
201 integer sfgainfo,sfrattr
202
203 character*(*) fvalues(*)
204 character attr_name*(MAX_NC_NAME),attr_data*512
205 character*(MAX_NC_NAME) values(25)
206 character cr
207
208c Get information about the file attribute
209 retn = sfgainfo(sd_id,attr_index,attr_name,adata_type,count)
210
211c Read the attribute data
212 retn = sfrattr(sd_id,attr_index,attr_data)
213
214 cr = char(10)
215 ival = 0
216 oldn = 1
217 5 continue
218
219c QuikSCAT attributes have atleast three lines:
220c metadata type, array size and metadata contents
221c Use "blank spaces" to identify the end of a line
222
223 n = index(attr_data(oldn:(count-1)),cr)
224
225c Read all of the metadata lines
226 if (n .eq. 0) then
227 ival=ival+1
228 values(ival) = attr_data(oldn:(count-1))
229 goto 99
230 else
231 ival=ival+1
232 values(ival) = attr_data(oldn:(oldn+n-2))
233 endif
234 oldn=n+oldn
235 goto 5
236
237 99 continue
238
239 n_values = ival - 2
240 do i=1,n_values
241 fvalues(i) = values(i+2)
242 enddo
243
244 return
245 end
