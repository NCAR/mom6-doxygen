111! Anything in this routine is executed exactly once. It could be used

376subroutine get_expected_1d_integral(state_handle, ens_size, location, igrkey, val, istatus)
377
378type(ensemble_type), intent(in) :: state_handle
379integer, intent(in) :: ens_size
380type(location_type), intent(in) :: location
381integer, intent(in) :: igrkey
382real(r8), intent(out) :: val(ens_size)
383integer, intent(out) :: istatus(ens_size)
384
385! The forward operator interface for this type of observation. It is
386! called with a state vector, a location, and a key to identify which
387! observation is being processed. The return 'val' is the expected
388! observation value, and istatus is the return code. 0 is ok,
389! > 0 signals an error, and < 0 values are reserved for system use.
390! The call to 'interpolate()' below calls the forward operator in
391! whatever model this code has been compiled with.
392
393type(location_type) :: location2
394integer :: i
395real(r8) :: range, loc, bottom, dx, x
396integer :: j !< loop variable
397integer :: point_istatus(ens_size)
398logical :: return_now ! used to return early if an interpoaltion fails
399real(r8), dimension(ens_size) :: sum, dist, weight, weight_sum
400
401if ( .not. module_initialized ) call initialize_module
402
403! Make sure key is within valid range
404call check_valid_key(igrkey, 'GIVEN', 'get_expected_1d_integral')
405
406! Figure out the total range of the integrated funtion (1 is max)
407range = 4.0_r8 * half_width(igrkey)
408if(range > 1.0_r8) range = 1.0_r8
409if(debug) print*, 'range is ', range
410
411! Get the location value
412loc = get_location(location)
413if(debug) print*, 'loc base is ', loc
414
415! Compute the bottom and top of the range
416bottom = loc - range / 2.0_r8
417if(bottom < 0.0_r8) bottom = bottom + 1.0_r8
418if(debug) print*, 'bottom is ', bottom
419
420! Next figure out where to put all the points
421dx = range / (num_points(igrkey) - 1)
422if(debug) print*, 'dx is ', dx
423
424! Loop to compute the value at each point, then multiply by localization
425! to get weighted integral
426sum = 0.0_r8
427weight_sum = 0.0_r8
428istatus(:) = 0 ! so you can track istatus from 1 to num_points
429
430do i = 1, num_points(igrkey)
431 x = bottom + (i - 1) * dx
432 if(x > 1.0_r8) x = x - 1.0_r8
433 location2 = set_location(x)
434 call interpolate(state_handle, ens_size, location2, qty_state_variable, val, point_istatus)
435 call track_status(ens_size, point_istatus, val, istatus, return_now)
436 do j = 1, ens_size
437 if (istatus(j) == 0) then
438 dist(j) = abs(loc - x)
439 if(dist(j) > 0.5_r8) dist(j) = 1.0_r8 - dist(j)
440 weight(j) = comp_cov_factor(dist(j), half_width(igrkey), &
441 localization_override = localization_type(igrkey))
442 sum(j) = sum(j) + weight(j) * val(j)
443 weight_sum(j) = weight_sum(j) + weight(j)
444 endif
445 enddo
446
447enddo
448
449where (istatus == 0) val = sum / weight_sum
450
451if(debug) print*, 'get_expected_1d_integral key is ', igrkey
452if(debug) print*, 'metadata values are: ', half_width(igrkey), num_points(igrkey), localization_type(igrkey)
453if(debug) print*, 'return value for forward operator is ', val
454if(debug) print*, 'return status (0 good; >0 error; <0 reserved for system use) is ', istatus
455
456end subroutine get_expected_1d_integral

459subroutine get_expected_power(state_handle, ens_size, location, powkey, val, istatus)
460
461type(ensemble_type), intent(in) :: state_handle
462integer, intent(in) :: ens_size
463type(location_type), intent(in) :: location
464integer, intent(in) :: powkey
465real(r8), intent(out) :: val(ens_size)
466integer, intent(out) :: istatus(ens_size)
467
468integer :: i
469real(r8) :: pval(ens_size)
470
471! The forward operator interface for this type of observation. It is
472! called with a state vector, a location, and a key to identify which
473! observation is being processed. The return 'val' is the expected
474! observation value, and istatus is the return code. 0 is ok,
475! > 0 signals an error, and < 0 values are reserved for system use.
476! The call to 'interpolate()' below calls the forward operator in
477! whatever model this code has been compiled with.
478
479if ( .not. module_initialized ) call initialize_module
480
481! Make sure key is within valid range
482call check_valid_key_power(powkey, 'GIVEN', 'get_expected_power')
483
484! Interpolate the raw state to the location for each ensemble member
485call interpolate(state_handle, ens_size, location, qty_state_variable, pval, istatus)
486
487if(power(powkey) == int(power(powkey))) then
488 ! Integer power, just use standard definition
489 val = pval ** power(powkey)
490else
491 ! For non-integer powers, fix up values for negative bases
492 do i = 1, ens_size
493 if(pval(i) >= 0.0_r8) then
494 val(i) = pval(i) ** power(powkey)
495 else
496 val(i) = -1.0_r8 * (-1.0_r8 * pval(i)) ** power(powkey)
497 endif
498 end do
499endif
500
501if(debug) print*, 'get_expected_power key is ', powkey
502if(debug) print*, 'metadata value is: ', power(powkey)
503if(debug) print*, 'return value for forward operator is ', val
504if(debug) print*, 'return status (0 good; >0 error; <0 reserved for system use) is ', istatus
505
506end subroutine get_expected_power

601subroutine check_valid_key(igrkey, what, fromwhere)
602 integer, intent(in) :: igrkey
603 character(len=*), intent(in) :: what, fromwhere
604
605! Internal subroutine that verifies that we haven't incremented the key value
606! past the size of the allocated space, or that a routine hasn't been called
607! with a out-of-range key (which would indicate an internal error of some kind).
608! If an error is found, a fatal message is printed and this routine doesn't return.
609! The 'what' argument is either 'GIVEN' for a key value that's passed in from
