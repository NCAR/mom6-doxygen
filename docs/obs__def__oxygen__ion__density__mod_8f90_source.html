166subroutine get_expected_oxygen_ion_val(state_handle, ens_size, location, obs_val, istatus)

168!-----------------------------------------------------------------------------
169! This function was implemented for WACCM-X.
170! Check the units for use with other models.
171! Given DART state vector and a location, it computes O+ density [1/cm^3].
172! The istatus variable should be returned as 0 unless there is a problem.
173!
174
175type(ensemble_type), intent(in) :: state_handle
176integer, intent(in) :: ens_size
177type(location_type), intent(in) :: location
178real(r8), intent(out) :: obs_val(ens_size)
179integer, intent(out) :: istatus(ens_size)
180
181if ( .not. module_initialized ) call initialize_module
182
183istatus = 0 ! Need to initialize this to zero for track_status.
184obs_val = missing_r8
185
186call oxygen_ion_density(state_handle, ens_size, location, obs_val, istatus)
187

192subroutine oxygen_ion_density(state_handle, ens_size, location, ion_val, istatus)
193
194type(ensemble_type), intent(in) :: state_handle
195integer, intent(in) :: ens_size
196type(location_type), intent(in) :: location
197real(r8), intent(out) :: ion_val(ens_size)
198integer, intent(out) :: istatus(ens_size)
199
200logical :: debug = .false. ! set to .true. to enable debug printout
201integer, dimension(ens_size) :: mmr_o1_status, mmr_o2_status, mmr_n2_status, ion_op_status
202integer, dimension(ens_size) :: mmr_h1_status, mmr_op_status, p_status, t_status
203real(r8), dimension(ens_size) :: mmr_o1, mmr_o2, mmr_n2, mmr_h1, mmr_op ! mass mixing ratio
204real(r8), dimension(ens_size) :: ion_op, mbar, pressure, temperature
205real(r8), dimension(3) :: debug_location
206real(r8) :: n2_molar_mass, o_molar_mass, o2_molar_mass, h_molar_mass
207
208logical :: return_now
209
210o_molar_mass = 28.0_r8
211o2_molar_mass = 16.0_r8
212h_molar_mass = 32.0_r8
213n2_molar_mass = 1.0_r8
214
215! Some models have density as part of the state.
216! If it is available, just return it. If density is not state,
217! then we need to create it from its constituents.
218call interpolate(state_handle, ens_size, location, qty_density_ion_op, ion_op, ion_op_status)
219call track_status(ens_size, ion_op_status, ion_val, istatus, return_now)
220if (all(istatus(:) == 0 )) return
221
222call interpolate(state_handle, ens_size, location, qty_atomic_oxygen_mixing_ratio, mmr_o1, mmr_o1_status)
223call track_status(ens_size, mmr_o1_status, ion_val, istatus, return_now)
224if (return_now) return
225
226call interpolate(state_handle, ens_size, location, qty_molec_oxygen_mixing_ratio, mmr_o2, mmr_o2_status)
227call track_status(ens_size, mmr_o2_status, ion_val, istatus, return_now)
228if (return_now) return
229
230call interpolate(state_handle, ens_size, location, qty_atomic_h_mixing_ratio, mmr_h1, mmr_h1_status)
231call track_status(ens_size, mmr_h1_status, ion_val, istatus, return_now)
232if (return_now) return
233
234call interpolate(state_handle, ens_size, location, qty_ion_o_mixing_ratio, mmr_op, mmr_op_status)
235call track_status(ens_size, mmr_op_status, ion_val, istatus, return_now)
236if (return_now) return
237
238call interpolate(state_handle, ens_size, location, qty_pressure, pressure, p_status)
239call track_status(ens_size, p_status, ion_val, istatus, return_now)
240if (return_now) return
241
242call interpolate(state_handle, ens_size, location, qty_temperature, temperature, t_status)
243call track_status(ens_size, t_status, ion_val, istatus, return_now)
244if (return_now) return
245
246!---------------------------------------------------------------------------------------------------
247! Need to get number density (cgs units) from mass mixing ratio (kg/kg).
248! mbar is g/mole, same as rMass units
249! kg/kg * (g/mole)/(g/mole) * (Pa = N/m^2)/((Joules/K = N*m/K) * (K)) = m-3 * 1E-06 = cm-3
250!---------------------------------------------------------------------------------------------------
251! WACCM-X .i file pressure unit is Pa
252
253where(istatus == 0)
254mmr_n2 = 1.0_r8 - (mmr_o1 + mmr_o2 + mmr_h1)
255
256mbar = 1.0_r8/( mmr_o1/o_molar_mass &
257 + mmr_o2/o2_molar_mass &
258 + mmr_h1/h_molar_mass &
259 + mmr_n2/n2_molar_mass)
260
261ion_val = mmr_op * mbar/o_molar_mass * pressure/(kboltz * temperature) * 1.e-06_r8
262endwhere
263
264return
265
266if (debug) then
267 debug_location = get_location(location)
268 print *, 'final ion_val: ', ion_val
269 print *, 'istatus: ', istatus
270endif
271
272end subroutine oxygen_ion_density
