10!
11! convert_aviso - program that reads a netCDF file from CMEMS/Aviso+
12! containing L3 (along-track) sea level anomalies from
13! any of three supported platforms and writes a DART
14! observation sequence file.
15!
16! http://marine.copernicus.eu/services-portfolio/access-to-products/
17! product: SEALEVEL_GLO_SLA_L3_REP_OBSERVATIONS_008_018
18!
19!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!

54character(len=256), parameter :: source = &
55 "$URL$"
56character(len=32 ), parameter :: revision = "$Revision$"
57character(len=128), parameter :: revdate = "$Date$"
58
59logical, parameter :: use_input_qc = .false.
60
61integer, parameter :: num_copies = 1, & ! number of copies in sequence
62 num_qc = 1 ! number of QC entries
63
64integer :: ncid, varid, ios
65integer :: ntime, n, i, oday, osec, nused, idx
66integer :: iyear, imonth, iday, ihour, imin, isec
67logical :: first_obs

69real(r8) :: oerr, qc, missing

75type(time_type) :: comp_day0, time_obs, prev_time

77character(len=100) :: platform, attvalue
78
79character(len=256) :: re_file
80character(len=256) :: profiler_out_file = 'obs_seq.aviso'
81character(len=256) :: input_file = '/glade/p/work/fredc/OBS/SLA_alongtrack/dt_global_j1_sla_vfec_20080101_20140106.nc'
82
83character(len=512) :: string1, string2, string3

85!-----------------------------------------------------------------------
86! start of executable code
87!-----------------------------------------------------------------------
88
89call initialize_utilities('convert_aviso')
90
91! command line argument
92call get_command_argument(1, input_file)
93
94if (input_file == '') then
95 write(string1,*)'.. Require a command-line argument specifying the input file.'
96 write(string2,*)'Must be a fully-qualified file name.'
97 write(string3,*)'USAGE: convert_aviso <input_file_name>'
98 call error_handler(e_err, 'convert_aviso', string1, &
99 source, revision, revdate, text2=string2, text3=string3)
100endif
101
102write(string1,*)'Converting input file "'//trim(input_file)//'"'
103call error_handler(e_msg, 'convert_aviso', string1)
104
105first_obs = .true.
106
107call nc_check( nf90_open(input_file, nf90_nowrite, ncid), &
108 'convert_aviso', 'opening file "'//trim(input_file)//'"')
109
110call nc_check( nf90_get_att(ncid, nf90_global, "platform", platform), &
111 'convert_aviso', 'reading global attribute platform')
112
113call getdimlen(ncid, "time", ntime)
114
115allocate( lat(ntime))
116allocate( lon(ntime))
117allocate( sla(ntime))
118allocate(time(ntime))
119
120call get_int_as_r8( ncid, "latitude", lat)
121call get_int_as_r8( ncid, "longitude", lon)
122call get_short_as_r8(ncid, "SLA", sla, missing)
123
124idx = index(input_file,'sla')
125re_file = input_file(1:idx-1) // 'RE' // input_file(idx+3:len(input_file))
126
127call nc_check( nf90_open(re_file, nf90_nowrite, ncid), &
128 'convert_aviso', 'opening file '//trim(re_file))
129
130allocate(re(ntime))
131
132call get_short_as_r8(ncid, "RE", re, missing)
133
134! Convert the time to DART type
135
136ios = nf90_inq_varid(ncid, 'time', varid)
137call nc_check(ios, 'convert_aviso', 'inq_varid "time"')
138
139ios = nf90_get_var(ncid, varid, time)
140call nc_check(ios, 'convert_aviso', 'get_var "time"')
141
142ios = nf90_get_att(ncid, varid, 'units', attvalue)
143call nc_check(ios, 'convert_aviso', 'get_att time "units"')
144
145! time:units = "days since 1950-01-01 00:00:00" ;
146! 1234567890
147
148if (attvalue(1:10) /= 'days since') then
149 write(string1,*)'expecting time units of [days since ... ]'
150 write(string2,*)'read time units of ['//trim(attvalue)//']'
151 call error_handler(e_err, 'convert_aviso', string1, &
152 source, revision, revdate, text2=string2)
153endif
154
155read(attvalue,'(11x,i4,5(1x,i2))',iostat=ios)iyear,imonth,iday,ihour,imin,isec
156if (ios /= 0) then
157 write(string1,*)'Unable to read time units. Error status was ',ios
158 write(string2,*)'expected "days since YYYY-MM-DD HH:MM:SS"'
159 write(string3,*)'was "'//trim(attvalue)//'"'
160 call error_handler(e_err, 'convert_aviso', string1, &
161 source, revision, revdate, text2=string2, text3=string3)
162endif
163
164call set_calendar_type(gregorian)
165comp_day0 = set_date(iyear, imonth, iday, ihour, imin, isec)
166
167call static_init_obs_sequence()
168call init_obs(obs, num_copies, num_qc)
169call init_obs(prev_obs, num_copies, num_qc)
170
171! create a new one
172call init_obs_sequence(obs_seq, num_copies, num_qc, ntime)
173do i = 1, num_copies
174 call set_copy_meta_data(obs_seq, i, 'observation')
175end do
176do i = 1, num_qc
177 call set_qc_meta_data(obs_seq, i, 'Data QC')
178end do
179
180! Set the DART data quality control. Be consistent with NCEP codes;
181! 0 is 'must use', 1 is good, no reason not to use it.
182qc = 1.0_r8
183
184! add sla component data to obs_seq
185nused = 0
186staloop: do n = 1, ntime
187
188 if ( sla(n) == missing) cycle staloop
189 if ( lat(n) > 90.0_r8 .or. lat(n) < -90.0_r8 ) cycle staloop
190 if ( lon(n) > 360.0_r8 .or. lon(n) < 0.0_r8 ) cycle staloop
191
192 ! compute time of observation
193 oday = int(time(n))
194 osec = nint((time(n) - real(oday,r8))*86400.0_r8)
195 time_obs = set_time(osec, oday) + comp_day0
196 call get_time(time_obs, osec, oday)
197
198 oerr = 0.03_r8 ! this works mostly for surface to 100mb
199
200 select case (platform)
201
202 case("Jason-1")
203 call create_3d_obs(lat(n), lon(n), 0.0_r8, vertissurface, sla(n), &
204 j1_sea_surface_anomaly, re(n), oday, osec, qc, obs)
205 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
206 case("Envisat")
207 call create_3d_obs(lat(n), lon(n), 0.0_r8, vertissurface, sla(n), &
208 en_sea_surface_anomaly, re(n), oday, osec, qc, obs)
209 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
210 case("Geosat Follow On")
211 call create_3d_obs(lat(n), lon(n), 0.0_r8, vertissurface, sla(n), &
212 gfo_sea_surface_anomaly, re(n), oday, osec, qc, obs)
213 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
214 case default
215 string1 = 'Unknown "platform" from "'//trim(input_file)//'"'
216 string2 = 'read as "'//trim(platform)//'"'
217 string3 = 'must be "Jason-1", "Envisat", or "Geosat Follow On"'
218 call error_handler(e_err, 'query_varname', string1, &
219 source, revision, revdate, text2=string2, text3=string3)
220 end select
221
222 nused = nused + 1
223
224end do staloop
225
226! need to wait to close file because in the loop it queries the
227! report types.
228call nc_check(nf90_close(ncid),'convert_aviso','closing "'//trim(input_file)//'"')
229
230! if we added any obs to the sequence, write it now.
231if ( get_num_obs(obs_seq) > 0 ) call write_obs_seq(obs_seq, profiler_out_file)
232
233call finalize_utilities()
234
235end program convert_aviso
