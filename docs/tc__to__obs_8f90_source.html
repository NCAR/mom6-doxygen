38! version controlled file description for error handling, do not edit
39character(len=256), parameter :: source = &
40 "$URL$"
41character(len=32 ), parameter :: revision = "$Revision$"
42character(len=128), parameter :: revdate = "$Date$"
43
44
45! IMPORTANT! These are the observational errors for position (in degrees),
46! minimum sea level pressure (in hectopascals/mb), and max wind (in m/s).
47! They are hardcoded here; to change their values either edit them here
48! or add code to make them dependent on the magnitude of the obs values.
49
50real(r8), parameter :: tc_pos_error = 0.30_r8, &
51 tc_pmin_error = 3.00_r8, &
52 tc_wmax_error = 3.00_r8
53
54
55logical :: file_exist, first_obs
56real(r8) :: lat, lon, qc
57real(r8) :: mslp, mwnd, wsval
58integer :: fhr, istat, io
59integer :: iyear, imonth, iday, ihour
60integer :: oday, osec, rcio, iunit
61integer :: num_copies, num_qc, max_obs
62
63character(len=256) :: input_line, msgstring
64character(len=10) :: yyyymmddhh
65character(len=4) :: ftype
66character(len=2) :: basin, tcnum, stypei
67character(len=1) :: ew, ns
68
69type(obs_sequence_type) :: obs_seq
70type(obs_type) :: obs, prev_obs
71type(time_type) :: time_obs, prev_time
72
73! things in the namelist
74character(len=256) :: input_atcf_file = 'input.txt'
75character(len=128) :: fileformat = 'b-deck'
76character(len=256) :: obs_out_file = 'obs_seq.out'
77logical :: append_to_existing_file = .false.
78logical :: debug = .false. ! set to .true. to print more info
79
80
81namelist /tc_to_obs_nml/ &
82 input_atcf_file, fileformat, obs_out_file, append_to_existing_file, debug
83
84
85! start of executable code
86
87call initialize_utilities('tc_to_obs')
88
89! Read the namelist entry
90call find_namelist_in_file("input.nml", "tc_to_obs_nml", iunit)
91read(iunit, nml = tc_to_obs_nml, iostat = io)
92call check_namelist_read(iunit, io, "tc_to_obs_nml")
93
94! Record the namelist values used for the run ...
95if (do_nml_file()) write(nmlfileunit, nml=tc_to_obs_nml)
96if (do_nml_term()) write( * , nml=tc_to_obs_nml)
97
98! time setup
99call set_calendar_type(gregorian)
100
101! open input text file
102iunit = open_file(input_atcf_file, 'formatted', 'read')
103if (debug) print *, 'opened input ATCF file ' // trim(input_atcf_file)
104
105! each observation in this series will have a single observation value
106! and a quality control flag. the max possible number of obs needs to
107! be specified but it will only write out the actual number created.
108max_obs = 100000
109num_copies = 1
110num_qc = 1
111
112! call the initialization code, and initialize two empty observation types
113call static_init_obs_sequence()
114call init_obs(obs, num_copies, num_qc)
115call init_obs(prev_obs, num_copies, num_qc)
116first_obs = .true.
117
118! if you want to append to existing files (e.g. you have a lot of
119! small text files you want to combine), you can do it here,
120! or you can use the obs_sequence_tool to merge a list of files
121! once they are in DART obs_seq format.
122
123! existing file found, append to it if user requests this via the namelist
124inquire(file=obs_out_file, exist=file_exist)
125
126if ( file_exist .and. append_to_existing_file) then
127 if (debug) print *, 'found existing obs_seq file, appending to ' // trim(obs_out_file)
128 call read_obs_seq(obs_out_file, 0, 0, max_obs, obs_seq)
129else
130 ! create a new, empty obs_seq file. you must give a max limit
131 ! on number of obs. increase the size if too small.
132 if (debug) print *, 'creating new obs_seq file for output ' // trim(obs_out_file)
133 call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
134
135 ! the first one needs to contain the string 'observation' and the
136 ! second needs the string 'QC'.
137 call set_copy_meta_data(obs_seq, 1, 'observation')
138 call set_qc_meta_data(obs_seq, 1, 'Data QC')
139endif
140
141! Set the DART data quality control. 0 is good data.
142! increasingly larger QC values are more questionable quality data.
143qc = 0.0_r8
144
145obsloop: do ! no end limit - have the loop break when input ends
146
147 if (debug) print *, ''
148
149 ! read in a line from the text file. What you need to create an obs:
150 ! location: lat, lon, and for these types of obs the vertical is 'UNDEF'
151 ! time: when the observation was taken
152 ! type: from the DART list of obs types, typically VORTEX_xxx
153 ! error: very important - the instrument error plus model representativeness error
154 ! (see html doc file for more info)
155
156 ! read the entire text line into a buffer here. this makes it easier to
157 ! parse lines which may have variable numbers of fields. the later read()
158 ! then reads out of this character variable.
159 read(iunit, "(A)", iostat=rcio) input_line
160 if (rcio /= 0) then
161 if (debug) print *, 'read code from input file, rcio = ', rcio
162 if (debug) print *, '(normal if we have reached the end of file)'
163 exit obsloop
164 endif
165
166 if (debug) print *, 'next input line: ' // trim(input_line)
167
168 select case(fileformat)
169 case ("b-deck")
170 ! read the various data fields out of the most recently read input line
171 read(input_line, 21, iostat=istat ) basin, tcnum, yyyymmddhh, ftype, fhr, &
172 lat, ns, lon, ew, mwnd, mslp, stypei, wsval
173
17421 format(a2,1x,a2,1x,a10,1x,a4,1x,i1,1x, &
175 f5.2,a1,1x,f6.2,a1,1x,f5.1,1x,f6.1,1x,a2)
176
177 read(yyyymmddhh(1:4), fmt='(i4)') iyear
178 read(yyyymmddhh(5:6), fmt='(i2)') imonth
179 read(yyyymmddhh(7:8), fmt='(i2)') iday
180 read(yyyymmddhh(9:10), fmt='(i2)') ihour
181
182 if ( ns == 'S' ) lat = -1.0_r8 * lat
183 if ( ew == 'W' ) lon = -1.0_r8 * lon
184
185 mslp = mslp * 100.0_r8 ! pressures in dart are in pascals
186
187 time_obs = set_date(iyear, imonth, iday, ihour, 0, 0)
188 call get_time(time_obs, osec, oday)
189
190 ! case ("a-deck")
191 ! case ("other")
192 ! add different read format lines for other data layouts here
193
194 case default
195 write(msgstring, *) 'unrecognized fileformat string: "'//trim(fileformat)//'"'
196 call error_handler(e_err,'tc_to_obs', msgstring, source, revision, revdate)
197 end select
198
199
200 ! check the lat/lon values to see if they are ok
201 if ( lat > 90.0_r8 .or. lat < -90.0_r8 ) cycle obsloop
202 if ( lon > 360.0_r8 .or. lon < -180.0_r8 ) cycle obsloop
203 if ( lon < 0.0_r8 ) lon = lon + 360.0_r8 ! dart needs lon 0-360
204
205 if (debug) call print_date(time_obs, ' time of observations: ')
206
207 ! make an obs derived type, and then add it to the sequence
208 if (debug) print *, ' adding vortex latitude observation, value,err (in degrees) = ', lat, tc_pos_error
209 call create_3d_obs(lat, lon, 0.0_r8, vertisundef, lat, &
210 vortex_lat, tc_pos_error, oday, osec, qc, obs)
211 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
212
213 ! make an obs derived type, and then add it to the sequence
214 if (debug) print *, ' adding vortex longitude observation, value,err (in degrees) = ', lon, tc_pos_error
215 call create_3d_obs(lat, lon, 0.0_r8, vertisundef, lon, &
216 vortex_lon, tc_pos_error, oday, osec, qc, obs)
217 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
218
219 ! make an obs derived type, and then add it to the sequence
220 if (debug) print *, ' adding min slp observation, value,err (in pascals) = ', mslp, tc_pmin_error*100.0_r8
221 call create_3d_obs(lat, lon, 0.0_r8, vertisundef, mslp, &
222 vortex_pmin, tc_pmin_error*100.0_r8, oday, osec, qc, obs)
223 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
224
225 if (mwnd > 0.0_r8) then
226 ! make an obs derived type, and then add it to the sequence
227 if (debug) print *, ' adding max wind observation, value,err (in m/sec) = ', mwnd, tc_wmax_error
228 call create_3d_obs(lat, lon, 0.0_r8, vertisundef, mwnd, &
229 vortex_wmax, tc_wmax_error, oday, osec, qc, obs)
230 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
231 endif
232
233end do obsloop
234
235! if we added any obs to the sequence, write it out to a file now.
236if ( get_num_obs(obs_seq) > 0 ) then
237 if (debug) print *, 'writing output obs_seq file, obs_count = ', get_num_obs(obs_seq)
238 call write_obs_seq(obs_seq, obs_out_file)
239endif
240
241! end of main program
242call finalize_utilities()
243
244
245end program tc_to_obs
