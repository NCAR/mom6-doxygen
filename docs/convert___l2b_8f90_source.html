7program convert_l2b
8
9! Initial program to read the raw ocean observations and insert them
10! into an observation sequence. To make things easy ... we will mandate
11! an assimilatin interval of 1 day - so all observations will be
12! redefined to occur at NOON on the day they were observed.
13

90seq = real_obs_sequence(orbit, lon1, lon2, lat1, lat2, &
91 along_track_thin, cross_track_thin ) ! convert structure to a sequence
92call write_obs_seq(seq, dartfile)
93call destroy_obs_sequence(seq) ! release the memory of the seq
94
95call error_handler(e_msg,'convert_L2b','Finished successfully.',source,revision,revdate)
96call finalize_utilities()
97
98end program convert_l2b
