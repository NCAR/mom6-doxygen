5program sys_sim201

7! MODIFIED Sep12 2003 from sys_sim5 to look at uncertainty in motion
8! of mean in observation space with hierarchical ensembles.
9!
10! Work done during last week of January 2002 (28 Jan init) to investigate
11! impacts of sampling error from small ensembles on update for a single
12! variable that is exactly observed; applicable to observation variable
13! priors. Small sample estimates of variance have approximately normal
14! (or is it exactly normal) error distributions. BUT, when one computes
15! the updated variance there is a bias. Of course, one must also account
16! for the increased uncertainty in the estimate of the mean resulting from
17! errors in the computation of the variance.
18
19! This piece looks at what large sample MC gives for correct updated
20! distribution statistics in preparation for correcting EAKF for the small
21! sample problems.
22
23use random_seq_mod, only : random_seq_type, init_random_seq, random_gaussian, &
24 twod_gaussians, random_uniform
25
26implicit none
27
28type (random_seq_type) :: r
29double precision, allocatable :: rnum(:)
30double precision :: sigma_y_p, y_p, sigma_y_o, y_o, var_of_var
31double precision :: sum_mean, sum_mean_2, sum_var, var, sigma_y_u
32double precision :: y_u, sample_mean, sample_var, inf_var, inf_u
33double precision :: temp_mean, pre_var_sum, mean_var
34integer :: n, n_samples, i, j
35
36! Initialize repeatable random sequence
37call init_random_seq(r)
38
39! For now have a set of free parameters that may be too large
40write(*, *) 'Input prior variance of observation variable'
41read(*, *) sigma_y_p
42!!!write(*, *) 'Input prior value of observation variable'
43!!!read(*, *) y_p
44! Make prior value always 0 to cut down a parameter
45y_p = 0.0
46write(*, *) 'Input variance of prior variance from groups'
47read(*, *) var_of_var
48write(*, *) 'Input variance of observing instrument'
49read(*, *) sigma_y_o
50!!!write(*, *) 'input value of observation '
51!!!read(*, *) y_o
52! This also doesn't matter; make it 1.0
53y_o = 1.0
54write(*, *) 'Input ensemble size'
55read(*, *) n
56
57write(*, *) 'Input number of samples for statistics'
58read(*, *) n_samples
59
60! Initialize summations
61sum_mean = 0.0
62sum_mean_2 = 0.0
63sum_var = 0.0
64pre_var_sum = 0.0
65
66! Loop through the number of samples
67do i = 1, n_samples
68
69! We are given the sample variance, what is a sample of the underlying
70! variance distribution
71 var = random_gaussian(r, sigma_y_p, sqrt(var_of_var))
72
73 pre_var_sum = pre_var_sum + var
74
75
76
77! Given this var, compute updated var and mean
78 sigma_y_u = (1.0 / (1.0 / var + 1.0 / sigma_y_o))
79 y_u = sigma_y_u * (y_p / var + y_o / sigma_y_o)
80
81
82if(var < 0.0) write(*, *) 'var is , y_u, si', real(var), real(y_u), real(sigma_y_u)
83
84
85! Need to keep sum of the means, sum of the mean squared, sum of the var
86 sum_mean = sum_mean + y_u
87 sum_mean_2 = sum_mean_2 + (y_u)**2
88 sum_var = sum_var + sigma_y_u
89
90end do
91
92! Output the mean and variance corresponding to finite sample
93! See working notes, 29 Jan 2002
94sample_mean = sum_mean / n_samples
95sample_var = (sum_var + sum_mean_2) / n_samples - sample_mean**2
96write(*, *) 'sample mean and var ', real(sample_mean), real(sample_var)
97
98! What would happen without these assumptions
99inf_var = (1.0 / (1.0 / sigma_y_p + 1.0 / sigma_y_o))
100inf_u = inf_var * (y_p / sigma_y_p + y_o / sigma_y_o)
101write(*, *) 'infini mean and var ', real(inf_u), real(inf_var)
102
103mean_var = sum_var / n_samples
104write(*, *) 'mean var i s ', real(mean_var)
105write(*, *) 'pre mean var ', real(pre_var_sum / n_samples)
106
107write(*, *) '-----------------'
108write(*, *) 'normalized sample var - mean var ', &
109 real((sample_var - sum_var / n_samples) / (y_o - y_p)**2)
110write(*, *) '-------------------------------'
111write(*, *) 'ratio of mean_var to prior var ', real(mean_var /sigma_y_p)
112write(*, *) 'expected ratio of mean to prior ', real(inf_var / sigma_y_p)
113write(*, *) '-------------------------------'
114
115write(*, *) 'fraction of movement ', real((sample_mean - y_o) / (y_p - y_o))
116write(*, *) 'expected fraction of movement ', real((inf_u - y_o) / (y_p - y_o))
117
118
119end program sys_sim201

123subroutine comp_correl(ens, n, correl)
124
125implicit none
126
127integer, intent(in) :: n
128double precision, intent(in) :: ens(2, n)
129double precision, intent(out) :: correl
130double precision :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
131
132
133sum_x = sum(ens(2, :))
134sum_y = sum(ens(1, :))
135sum_xy = sum(ens(2, :) * ens(1, :))
136sum_x2 = sum(ens(2, :) * ens(2, :))
137
138! Computation of correlation
139sum_y2 = sum(ens(1, :) * ens(1, :))
140
141correl = (n * sum_xy - sum_x * sum_y) / &
142 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
143
144end subroutine comp_correl
