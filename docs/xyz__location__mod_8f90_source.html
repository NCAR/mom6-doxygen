42! This version supports only regularly spaced boxes. it originally had code to
43! use an octree division of the space, but finding nearest boxes on each side
44! (where there might be multiples) was too complex. i still think its a good
45! idea but need a better data structure to capture the relationships of leaving
46! any box via a face and listing all the other boxes that adjoin that face.
47!
48! the basic idea for octrees was to divide each dim in half until N numbers of filled
49! boxes, or octree reaches some max depth. give some threshold where you don't
50! divide a box with less than N points in it.
51
52! contrast with kD-trees (divide along dimensions, not points), and there are
53! two types of octrees - PR (point region) where the regions split at an
54! explicit point, vs MX tree where the split is defined to be at the center
55! of the region.
56
57! if the underlying geometry is spherical, there will be many many empty boxes
58! if we uniformly divide up space, and worse, existing locations will be
59! clustered in a few boxes.
60
61

146function xyz_get_dist(loc1, loc2, type1, kind2)
147
148! returns the distance between 2 locations
149
150! the 3rd name is a specific type, the 4th is a generic kind.
151! These types/kinds are part of
152! the interface in case user-code wants to do a more sophisticated distance
153! calculation based on the base type or target kind. In the usual case this
154! code still doesn't use the types, but there's an undocumented feature that
155! allows you to maintain the original vertical normalization even when
156! changing the cutoff distance in the horizontal. For that to work we
157! do need to know the type, and we use the type of loc1 to control it.
158!
159
160type(xyz_location_type), intent(in) :: loc1, loc2
161integer, optional, intent(in) :: type1, kind2
162real(r8) :: xyz_get_dist
163
164real(r8) :: x_dif, y_dif, z_dif
165real(r8) :: mag1, mag2, norm
166
167if ( .not. module_initialized ) call initialize_module
168
169if (line_of_sight_distance) then
170
171 ! straight line in 3d space
172x_dif = loc1%x - loc2%x
173y_dif = loc1%y - loc2%y
174z_dif = loc1%z - loc2%z
175
176xyz_get_dist = sqrt(x_dif * x_dif + y_dif * y_dif + z_dif * z_dif)
177
178 !print *, 'los dist: ', xyz_get_dist
179
180else
181
182 ! great circle distance
183 mag1 = sqrt(loc1%x * loc1%x + loc1%y * loc1%y + loc1%z * loc1%z)
184 mag2 = sqrt(loc2%x * loc2%x + loc2%y * loc2%y + loc2%z * loc2%z)
185 norm = (loc1%x*loc2%x + loc1%y*loc2%y + loc1%z*loc2%z) / (mag1 * mag2)
186
187 xyz_get_dist = sphere_radius * acos(norm)
188
189 !print *, 'mag1, 2, norm, acos, radius: ', mag1, mag2, norm, acos(norm), sphere_radius
190 !print *, ' gc dist: ', xyz_get_dist
191
192endif
193
194end function xyz_get_dist

215subroutine xyz_get_ll_location(loc, radius, lon, lat)
216
217! Given a cartesian x, y, z coordinate relative to an origin at
218! the center of the sphere, using a fixed radius specified by the
219! caller, return the corresponding lon, lat location in degrees.
220
221type(xyz_location_type) :: loc
222real(r8), intent(in) :: radius
223real(r8), intent(out) :: lon, lat
224
225real(r8) :: rlat, rlon
226
227! don't call this in performance-critical code.
228! the arc versions of routines are expensive.
229
230rlat = pi/2.0_r8 - acos(loc%z/radius)
231rlon = atan2(loc%y,loc%x)
232if (rlon < 0) rlon = rlon + pi*2
233
234lat = rlat * rad2deg
235lon = rlon * rad2deg
236
237end subroutine xyz_get_ll_location

309subroutine xyz_get_close_init(gc, maxdist, num, locs)
310
311type(xyz_get_close_type), intent(inout) :: gc
312real(r8), intent(in) :: maxdist
313integer, intent(in) :: num
314type(xyz_location_type), intent(in) :: locs(num)
315integer :: i, j, k, cum_start, l
316integer :: x_box(num), y_box(num), z_box(num)
317integer :: tstart(nx, ny, nz)
318
319if ( .not. module_initialized ) call initialize_module
320
321! set the default value.
322gc%maxdist = maxdist
323!print *, 'setting maxdist to ', maxdist
324
325! Allocate the storage for the grid dependent boxes
326allocate(gc%box%count(nx,ny,nz), gc%box%start(nx,ny,nz))
327gc%box%count = -1
328gc%box%start = -1
329
330! Allocate storage for loc number dependent part
331allocate(gc%box%loc_box(num))
332gc%box%loc_box(:) = -1
333
334! Set the value of num_loc in the structure
335gc%num = num
336
337! If num == 0, no point in going any further.
338if (num == 0) return
339
340! Determine where the boxes should be for this set of loc and maxdist
341call find_box_ranges(gc, locs, num)
342
343! Begin by computing the number of locations in each box in x,y,z
344gc%box%count = 0
345do i = 1, num
346
347!print *, i, locs(i)%x, locs(i)%y, locs(i)%z
348 x_box(i) = floor((locs(i)%x - gc%box%bot_x) / gc%box%x_width) + 1
349 if(x_box(i) > nx) x_box(i) = nx
350 if(x_box(i) < 1) x_box(i) = 1
351
352 y_box(i) = floor((locs(i)%y - gc%box%bot_y) / gc%box%y_width) + 1
353 if(y_box(i) > ny) y_box(i) = ny
354 if(y_box(i) < 1) y_box(i) = 1
355
356 z_box(i) = floor((locs(i)%z - gc%box%bot_z) / gc%box%z_width) + 1
357 if(z_box(i) > nz) z_box(i) = nz
358 if(z_box(i) < 1) z_box(i) = 1
359
360 gc%box%count(x_box(i), y_box(i), z_box(i)) = gc%box%count(x_box(i), y_box(i), z_box(i)) + 1
361!print *, 'adding count to box ', x_box(i), y_box(i), z_box(i), &
362! gc%box%count(x_box(i), y_box(i), z_box(i))
363end do
364
365! Figure out where storage for each boxes members should begin
366cum_start = 1
367do i = 1, nx
368 do j = 1, ny
369 do k = 1, nz
370 gc%box%start(i, j, k) = cum_start
371 cum_start = cum_start + gc%box%count(i, j, k)
372 end do
373 end do
374end do
375
376! Now we know how many are in each box, get a list of which are in each box
377tstart = gc%box%start
378do i = 1, num
379 gc%box%loc_box(tstart(x_box(i), y_box(i), z_box(i))) = i
380 tstart(x_box(i), y_box(i), z_box(i)) = tstart(x_box(i), y_box(i), z_box(i)) + 1
381end do
382
383do i = 1, nx
384 do j = 1, ny
385 do k = 1, nz
386!if (gc%box%count(i,j,k) > 0) print *, i,j,k, gc%box%count(i,j,k), gc%box%start(i,j,k)
387 do l=1, gc%box%count(i,j,k)
388!print *, l, gc%box%loc_box(l)
389 enddo
390 end do
391 end do
392end do
393
394
395end subroutine xyz_get_close_init

496!> search all boxes which are potentially close enough
497
498subroutine find_nearest_boxes(gc, base_loc, loc_list, nearest, rc, dist)
499 type(xyz_get_close_type), intent(in), target :: gc
500 type(xyz_location_type), intent(in) :: base_loc
501 type(xyz_location_type), intent(in) :: loc_list(:)
502 integer, intent(out) :: nearest
503 integer, intent(out) :: rc
504 real(r8), optional, intent(out) :: dist
505
506
507integer :: x_box, y_box, z_box, i, j, k, l
508integer :: start_x, end_x, start_y, end_y, start_z, end_z
509integer :: n_in_box, st, t_ind(1), ghost, n_found
510real(r8) :: this_dist, mindist(1)
511
512! First, set the intent out arguments to a missing value
513nearest = -99
514t_ind = -99
515rc = -1
516n_found = 0
517mindist = 1e38_r8 ! something big and positive.
518if (present(dist)) dist = 1e38_r8 ! ditto
519
520! the list of locations in the loc_list() argument must be the same
521! as the list of locations passed into get_close_init(), so
522! gc%num and size(loc_list) better be the same. if the list changes
523! you have to destroy the old gc and init a new one.
524if (size(loc_list) /= gc%num) then
525 write(errstring,*)'loc_list() array must match one passed to xyz_get_close_init()'
526 call error_handler(e_err, 'find_nearest_boxes', errstring, source)
527endif
528
529! If num == 0, no point in going any further.
530if (gc%num == 0) return
531
532!--------------------------------------------------------------
533
534! Begin by figuring out which box the base loc is in
535x_box = floor((base_loc%x - gc%box%bot_x) / gc%box%x_width) + 1
536if(x_box > nx) x_box = nx
537if(x_box < 1) x_box = 1
538y_box = floor((base_loc%y - gc%box%bot_y) / gc%box%y_width) + 1
539if(y_box > ny) y_box = ny
540if(y_box < 1) y_box = 1
541z_box = floor((base_loc%z - gc%box%bot_z) / gc%box%z_width) + 1
542if(z_box > nz) z_box = nz
543if(z_box < 1) z_box = 1
544
545!print *, 'base_loc box ', x_box, y_box, z_box
546!print *, 'nx, ny, nz = ', nx, ny, nz
547
548! If it is not in any box, then it is more than the maxdist away from everybody
549if(x_box > nx .or. x_box < 1) return
550if(y_box > ny .or. y_box < 1) return
551if(z_box > nz .or. z_box < 1) return
552
553!print *, 'good box'
554
555
556! First, search all points in the box that contains the base loc
557call do_this_box(gc, x_box, y_box, z_box, base_loc, loc_list, 1, t_ind, mindist, n_found)
558
559! if box small enough that no points match, expand search.
560! and for now, it is quite possible that points in a neighboring
561! box are closer than all points in the current box. so search
562! the neighbors until we are far enough away that we know we
563! have the nearest ones.
564ghost = 0
565
56610 continue
567if (ghost == 0 .or. n_found < 1) then
568 ghost = ghost + 1
569
570 start_x = x_box - ghost
571 if (start_x < 1) start_x = 1
572 end_x = x_box + ghost
573 if (end_x > nx) end_x = nx
574
575 start_y = y_box - ghost
576 if (start_y < 1) start_y = 1
577 end_y = y_box + ghost
578 if (end_y > ny) end_y = ny
579
580 start_z = z_box - ghost
581 if (start_z < 1) start_z = 1
582 end_z = z_box + ghost
583 if (end_z > nz) end_z = nz
584
585 !print *, 'looping from '
586 !print *, 'x: ', start_x, end_x
587 !print *, 'y: ', start_y, end_y
588 !print *, 'z: ', start_z, end_z
589
590 ! Next, loop through each box that is close to this box
591 do i = start_x, end_x
592 do j = start_y, end_y
593 do k = start_z, end_z
594
595 call do_this_box(gc, i, j, k, base_loc, loc_list, 1, t_ind, mindist, n_found)
596
597 end do
598 end do
599end do
600
601 if (n_found < 1) then
602 ! if we have searched the entire space, punt.
603 if (start_x == 1 .and. end_x == nx .and. &
604 start_y == 1 .and. end_y == ny .and. &
605 start_z == 1 .and. end_z == nz) return
606
607 ! repeat search with larger radius of boxes
608 goto 10
609 endif
610endif
611
612nearest = t_ind(1)
613rc = 0
614
615! if the caller asked for the actual distance, return it

622subroutine find_nearest_n_boxes(gc, base_loc, loc_list, n_wanted, nearest, n_found, rc, dist)
623 type(xyz_get_close_type), intent(in), target :: gc
624 type(xyz_location_type), intent(in) :: base_loc
625 type(xyz_location_type), intent(in) :: loc_list(:)
626 integer, intent(in) :: n_wanted
627 integer, intent(out) :: nearest(:)
628 integer, intent(out) :: n_found
629 integer, intent(out) :: rc
630 real(r8), optional, intent(out) :: dist(:)
631
632! find the nearest N points in the get close list to the specified point
633
634integer :: x_box, y_box, z_box, i, j, k, l
635integer :: start_x, end_x, start_y, end_y, start_z, end_z
636integer :: n_in_box, st, t_ind, ghost
637real(r8) :: this_dist, mindist(n_wanted), largest_dist
638logical :: been_searched(nx, ny, nz)
639
640! First, set the intent out arguments to a missing value
641nearest = -99
642n_found = 0
643rc = -1
644mindist = 1e38_r8 ! something big and positive.
645largest_dist = 1e38_r8 ! ditto
646if (present(dist)) dist(:) = 1e38_r8 ! ditto
647
648! make sure they want something - else return or fail?
649if (n_wanted < 1) then
650 write(errstring,*)'n_wanted must be 1 or larger'
651 call error_handler(e_err, 'find_nearest_n_boxes', errstring, source)
652endif
653
654if (present(dist)) then
655 if (size(dist) < n_wanted) then
656 write(errstring,*)'dist(:) array size must be equal to or larger than n_wanted'
657 call error_handler(e_err, 'find_nearest_n_boxes', errstring, source)
658 endif
659endif
660
661! the list of locations in the loc_list argument must be the same
662! as the list of locations passed into get_close_init(), so
663! gc%num and size(loc_list better be the same. if the list changes
664! you have to destroy the old gc and init a new one.
665if (size(loc_list) /= gc%num) then
666 write(errstring,*)'loc_list() array must match one passed to xyz_get_close_init()'
667 call error_handler(e_err, 'find_nearest_n_boxes', errstring, source)
668endif
669
670! If num == 0, no point in going any further.
671if (gc%num == 0) return
672
673!--------------------------------------------------------------
674
675! Begin by figuring out which box the base loc is in
676x_box = floor((base_loc%x - gc%box%bot_x) / gc%box%x_width) + 1
677if(x_box > nx) x_box = nx
678if(x_box < 1) x_box = 1
679y_box = floor((base_loc%y - gc%box%bot_y) / gc%box%y_width) + 1
680if(y_box > ny) y_box = ny
681if(y_box < 1) y_box = 1
682z_box = floor((base_loc%z - gc%box%bot_z) / gc%box%z_width) + 1
683if(z_box > nz) z_box = nz
684if(z_box < 1) z_box = 1
685
686!print *, 'base_loc box ', x_box, y_box, z_box
687!print *, 'nx, ny, nz = ', nx, ny, nz
688
689! If it is not in any box, then is this an error? what do we do about maxdist? FIXME
690if(x_box > nx .or. x_box < 1) return
691if(y_box > ny .or. y_box < 1) return
692if(z_box > nz .or. z_box < 1) return
693
694!print *, 'good box'
695
696been_searched(:,:,:) = .false.
697
698! find the closest N in the box that contains the base loc
699call do_this_box(gc, x_box, y_box, z_box, base_loc, loc_list, n_wanted, nearest, mindist, n_found)
700been_searched(x_box, y_box, z_box) = .true.
701
702! if box small enough that not enough points match, expand search.
703! also, it is quite possible that points in a neighboring
704! box are closer than some points in the current box. so search
705! the neighbors until we are far enough away that we know we
706! have the nearest ones. FIXME: be smarter about this.
707! have it set a flag for boxes it has already checked?
708ghost = 0
709
71010 continue
711if (ghost == 0 .or. n_found < n_wanted) then
712 ghost = ghost + 1
713
714 start_x = x_box - ghost
715 if (start_x < 1) start_x = 1
716 end_x = x_box + ghost
717 if (end_x > nx) end_x = nx
718
719 start_y = y_box - ghost
720 if (start_y < 1) start_y = 1
721 end_y = y_box + ghost
722 if (end_y > ny) end_y = ny
723
724 start_z = z_box - ghost
725 if (start_z < 1) start_z = 1
726 end_z = z_box + ghost
727 if (end_z > nz) end_z = nz
728
729 !print *, 'looping from '
730 !print *, 'x: ', start_x, end_x
731 !print *, 'y: ', start_y, end_y
732 !print *, 'z: ', start_z, end_z
733
734 ! Next, loop through each box that is close to this box
735 do i = start_x, end_x
736 do j = start_y, end_y
737 do k = start_z, end_z
738
739 if (been_searched(i,j,k)) cycle
740
741 call do_this_box(gc, i, j, k, base_loc, loc_list, n_wanted, nearest, mindist, n_found)
742 been_searched(i,j,k) = .true.
743
744 end do
745 end do
746 end do
747
748 if (n_found < n_wanted) then
749 ! if we have searched the entire space, punt.
750 if (start_x == 1 .and. end_x == nx .and. &
751 start_y == 1 .and. end_y == ny .and. &
752 start_z == 1 .and. end_z == nz) return
753
754 ! repeat search with larger radius of boxes
755 goto 10
756 endif
757endif
758
759! if they asked for the explicit distances, return them
760if (present(dist)) dist(:) = mindist(:)
761
762end subroutine find_nearest_n_boxes

766subroutine do_this_box(gc, i, j, k, base_loc, loc_list, n_wanted, nearest, dist, n_found)
767 type(xyz_get_close_type), intent(in), target :: gc
768 integer, intent(in) :: i, j, k
769 type(xyz_location_type), intent(in) :: base_loc
770 type(xyz_location_type), intent(in) :: loc_list(:)
771 integer, intent(in) :: n_wanted
772 integer, intent(inout) :: nearest(:)
773 real(r8), intent(inout) :: dist(:)
774 integer, intent(inout) :: n_found !< how many we already have
775
776
777integer :: n_in_box, st, l, m, n, t_ind
778real(r8) :: this_dist
779logical :: this_one_is_wanted
780
781! Box to search is i,j,k
782n_in_box = gc%box%count(i, j, k)
783st = gc%box%start(i,j,k)
784
785! Loop to check how close all locs in the box are; add those that are close
786do l = 1, n_in_box
787
788 t_ind = gc%box%loc_box(st - 1 + l)
789 ! print *, 'l, t_ind = ', l, t_ind
790
791 this_dist = xyz_get_dist(base_loc, loc_list(t_ind))
792 ! print *, 'this_dist = ', this_dist
793
794 ! if we haven't filled up the number of wanted near points,
795 ! or this new distance is smaller than the largest one, we are
796 ! going to be adding it. make these separate tests since if
797 ! n_found is 0, you'll get an out-of-bounds on dist(n_found)
798 this_one_is_wanted = .false.
799 if(n_found < n_wanted) then
800 this_one_is_wanted = .true.
801 else if (this_dist < dist(n_found)) then
802 this_one_is_wanted = .true.
803 endif
804
805 if (this_one_is_wanted) then
806 if (n_found < n_wanted) n_found = n_found + 1
807 do m=1, n_wanted ! updated value
808 if (this_dist >= dist(m)) cycle
809 if (m == n_found) then
810 nearest(m) = t_ind
811 dist(m) = this_dist
812 exit
813 else
814 do n=n_found, m+1, -1
815 nearest(n) = nearest(n-1)
816 dist(n) = dist(n-1)
817 enddo
818 nearest(m) = t_ind
819 dist(m) = this_dist
820 exit
821 endif
822 enddo
823 endif
824end do ! n_in_box
825
826end subroutine do_this_box
