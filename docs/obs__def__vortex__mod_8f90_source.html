114! Return vortex info according to whichinfo
115! whichinfo='lat', vinfo = center lat
116! whichinfo='lon', vinfo = center lon

122type(location_type), intent(in) :: location

126
127if ( .not. module_initialized ) call initialize_module
128
129if (whichinfo == 'lat') then
130 call interpolate(state_handle, ens_size, location, qty_vortex_lat, vinfo, istatus)
131elseif (whichinfo == 'lon') then
132 call interpolate(state_handle, ens_size, location, qty_vortex_lon, vinfo, istatus)
133elseif (whichinfo == 'pmi') then

135elseif (whichinfo == 'wma') then
136 call interpolate(state_handle, ens_size, location, qty_vortex_wmax, vinfo, istatus)
