66 obs_u, obs_v, obs_t, obs_ps, obs_qv, obs_z, daily_file, lon1, lon2, &
67 lat1, lat2, obs_time, include_specific_humidity, &
68 include_relative_humidity, include_dewpoint, include_surface_pressure

70! ----------------------------------------------------------------------
71! Select observation types using NCEP categories (when select_obs /= 0).
72! ADPUPA: upper-air reports (mostly radiosonde plus few dropsonde, PIBAL)
73! AIRCFT: Conv. (AIREP, PIREP) and ASDAR aircraft reports
74! AIRCAR: ACARS sircraft reports
75! SATEMP: ATOVS retrived temperature
76! SFCSHP: SURFACE MARINE reports
77! ADPSFC: SURFACE LAND SYNOPTIC STATION reports
78! SATWND: Satellite derived wind reports
79! ----------------------------------------------------------------------
80! Select variables of U, V, T, QV, PS using the logicals:
81! obs_U obs_V obs_PS obs_T obs_QV obs_Z
82! ----------------------------------------------------------------------
83
84! start of executable program code
85
86call initialize_utilities('create_real_obs')
87call register_module(source,revision,revdate)
88
89! Initialize the obs_sequence module ...
90
91call static_init_obs_sequence()
92
93! Read the namelist entry
94call find_namelist_in_file("input.nml", "ncepobs_nml", iunit)
95read(iunit, nml = ncepobs_nml, iostat = io)
96call check_namelist_read(iunit, io, "ncepobs_nml")
97
98! Record the namelist values used for the run ...
99if (do_nml_file()) write(nmlfileunit, nml=ncepobs_nml)
100if (do_nml_term()) write( * , nml=ncepobs_nml)
101
102! Loop through the days interested.
103
104do ii = 1, tot_days
105
106 day1 = day + ii - 1
107
108 ! define observation filename
109 write(obsdate, '(i4.4,i2.2,i2.2)') year, month, day1
110
111 ! set the obs sequence of the day (daily or 6 hourly)
112 if(daily_file) then
113 kbeg = 5
114 kend = 5
115 output_name = 'obs_seq'//obsdate
116 else
117 kbeg = 1
118 kend = 4
119 endif
120
121 do kkk = kbeg, kend
122
123 if (daily_file) then
124 hour1 = ''
125 else
126 hour1 = obstime(kkk)
127 end if
128
129 seq = real_obs_sequence(year, month, day1, hour1, max_num, select_obs, &
130 obsbase, adpupa, aircar, aircft, satemp, sfcshp, adpsfc, satwnd, &
131 obs_u, obs_v, obs_t, obs_ps, obs_qv, obs_z, include_specific_humidity, &
132 include_relative_humidity, include_dewpoint, include_surface_pressure, &
133 bin_beg(kkk), bin_end(kkk), lon1, lon2, lat1, lat2, obs_time)
134
135 ! output the daily sequence to a file
136 if(.not. daily_file) output_name = 'obs_seq'//obsdate//obstime(kkk)
137 call write_obs_seq(seq, output_name)
138
139 ! release the memory of the seq.
140 call destroy_obs_sequence(seq)
141
142 enddo
143
144enddo
145
146call error_handler(e_msg,'create_real_obs','Finished successfully.',source,revision,revdate)
147call finalize_utilities()
148
149end program create_real_obs
