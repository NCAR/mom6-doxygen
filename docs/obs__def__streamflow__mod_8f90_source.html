284subroutine interactive_streamflow_metadata(key)
285
286integer, intent(out) :: key

288character(len=metadatalength) :: userinput

364!----------------------------------------------------------------------
365!> If the allocatable metadata arrays are not big enough ... enlarge them
366
367subroutine grow_metadata(key, routine)
368
369integer, intent(in) :: key
370character(len=*), intent(in) :: routine
371
372integer :: orglength
373type(gage_metadata), allocatable, dimension(:) :: safe_metadata
374
375! fine -- no problem.
376if ((key > 0) .and. (key <= maxstreamkey)) return
377
378orglength = maxstreamkey
379maxstreamkey = 2 * orglength
380
381! Check for some error conditions.
382if (key < 1) then
383 write(string1, *) 'key (',key,') must be >= 1'
384 call error_handler(e_err,routine,string1,source,revision,revdate)
385elseif (key >= 2*maxstreamkey) then
386 write(string1, *) 'key (',key,') really unexpected.'
387 write(string2, *) 'doubling storage will not help.'
388 call error_handler(e_err,routine,string1,source,revision,revdate, &
389 text2=string2)
390endif
391
392! News. Tell the user we are increasing storage.
393write(string1, *) 'key (',key,') exceeds MAXstreamkey (',orglength,')'
394write(string2, *) 'Increasing MAXstreamkey to ',maxstreamkey
395call error_handler(e_msg,routine,string1,source,revision,revdate,text2=string2)
396
397allocate(safe_metadata(orglength))
398safe_metadata(:) = observation_metadata(:)
399
400deallocate(observation_metadata)
401 allocate(observation_metadata(maxstreamkey))
402
403observation_metadata(1:orglength) = safe_metadata(:)
404observation_metadata(orglength+1:maxstreamkey) = missing_metadata
405
406deallocate(safe_metadata)
407
408end subroutine grow_metadata
