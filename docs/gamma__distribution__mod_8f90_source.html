34subroutine test_gamma
35
36! This routine provides limited tests of the numerics in this module. It begins
37! by comparing a handful of cases of the pdf and cdf to results from Matlab. It
38! then tests the quality of the inverse cdf for a single shape/scale pair. Failing
39! these tests suggests a serious problem. Passing them does not indicate that
40! there are acceptable results for all possible inputs.
41
42real(r8) :: x, y, inv
43real(r8) :: mean, variance, sd, gamma_shape, gamma_scale, max_diff
44integer :: i
45
46! Comparative results for a handful of cases from MATLAB21a
47real(r8) :: pdf_diff(7), cdf_diff(7)
48real(r8) :: mshape(7) = [1.0_r8, 2.0_r8, 3.0_r8, 5.0_r8, 9.0_r8, 7.5_r8, 0.5_r8]
49real(r8) :: mscale(7) = [2.0_r8, 2.0_r8, 2.0_r8, 1.0_r8, 0.5_r8, 1.0_r8, 1.0_r8]
50real(r8) :: mx(7) = [1.0_r8, 2.0_r8, 3.0_r8, 4.0_r8, 5.0_r8, 6.0_r8, 7.0_r8]
51! Generated by matlab gampdf(mx, mshape, mscale)
52real(r8) :: mpdf(7) = [0.303265329856317_r8, 0.183939720585721_r8, 0.125510715083492_r8, &
53 0.195366814813165_r8, 0.225198064298040_r8, 0.151385201555322_r8, &
54 0.000194453010092_r8]
55! Generated by matlab gamcdf(mx, mshape, mscale)
56real(r8) :: mcdf(7) = [0.393469340287367_r8, 0.264241117657115_r8, 0.191153169461942_r8, &
57 0.371163064820127_r8, 0.667180321249281_r8, 0.320970942909585_r8, &
58 0.999817189367018_r8]
59
60! Compare to matlab
61write(*, *) 'Absolute value of differences should be less than 1e-15'
62do i = 1, 7
63 pdf_diff(i) = gamma_pdf(mx(i), mshape(i), mscale(i)) - mpdf(i)
64 cdf_diff(i) = gamma_cdf(mx(i), mshape(i), mscale(i), .true., .false., 0.0_r8, missing_r8) - mcdf(i)
65 write(*, *) i, pdf_diff(i), cdf_diff(i)
66end do
67
68! Input a mean and variance
69mean = 10.0_r8
70sd = 1.0_r8
71variance = sd**2
72
73! Get shape and scale
74gamma_shape = mean**2 / variance
75gamma_scale = variance / mean
76
77! Test the inversion of the cdf over +/- 5 standard deviations around mean
78max_diff = -1.0_r8
79do i = 0, 1000
80 x = mean + ((i - 500.0_r8) / 500.0_r8) * 5.0_r8 * sd
81 y = gamma_cdf(x, gamma_shape, gamma_scale, .true., .false., 0.0_r8, missing_r8)
82 inv = inv_gamma_cdf(y, gamma_shape, gamma_scale, .true., .false., 0.0_r8, missing_r8)
83 max_diff = max(abs(x-inv), max_diff)
84end do
85
86write(*, *) '----------------------------'
87write(*, *) 'max difference in inversion is ', max_diff
88write(*, *) 'max difference should be less than 1e-11'
89
90end subroutine test_gamma

132function gamma_pdf(x, gamma_shape, gamma_scale)
133
134! Returns the probability density of a gamma function with shape and scale
135! at the value x
136
137real(r8) :: gamma_pdf
138real(r8), intent(in) :: x, gamma_shape, gamma_scale
139
140! All inputs must be nonnegative
141if(x < 0.0_r8 .or. gamma_shape < 0.0_r8 .or. gamma_scale < 0.0_r8) then
142 gamma_pdf = failed_value
143else
144 gamma_pdf = x**(gamma_shape - 1.0_r8) * exp(-x / gamma_scale) / &
145 (gamma(gamma_shape) * gamma_scale**gamma_shape)

195function gammad (x, p)
196
197implicit none
198
199real(r8) :: gammad
200real(r8), intent(in) :: x
201real(r8), intent(in) :: p
202
203!*****************************************************************************80
204!
205!! GAMMAD computes the Incomplete Gamma Integral
206!
207! Modified:
208!
209! 20 January 2008
210!
211! Author:
212!
213! Original FORTRAN77 version by B Shea.
214! FORTRAN90 version by John Burkardt.
215!
216! Reference:
217!
218! B Shea,
219! Algorithm AS 239:
220! Chi-squared and Incomplete Gamma Integral,
221! Applied Statistics,
222! Volume 37, Number 3, 1988, pages 466-473.
223!
224! Parameters:
225!
226! Input, real ( kind = 8 ) X, P, the parameters of the incomplete
227! gamma ratio. 0 <= X, and 0 < P.
228!
229!
230! Output, real ( kind = 8 ) GAMMAD, the value of the incomplete
231! Gamma integral.
232!
233
234real(r8) :: a, b, c, an, arg, pn(6), rn
235real(r8), parameter :: elimit = - 88.0_r8
236real(r8), parameter :: oflo = 1.0e+37_r8
237real(r8), parameter :: plimit = 1000.0_r8
238real(r8), parameter :: tol = 1.0e-14_r8
239real(r8), parameter :: xbig = 1.0e+08_r8
240
241! x zero returns zero
242if(x == 0.0_r8) then
243 gammad = 0.0_r8
244elseif(xbig < x) then
245 ! If X is large set GAMMAD = 1.
246 gammad = 1.0_r8
247elseif(plimit < p) then
248! If P is large, use a normal approximation.
249 pn(1) = 3.0_r8 * sqrt(p) * ((x / p)**(1.0_r8 / 3.0_r8) + &
250 1.0_r8 / (9.0_r8 * p) - 1.0_r8)
251 gammad = normal_cdf(pn(1), 0.0_r8, 1.0_r8)
252elseif(x <= 1.0_r8 .or. x < p) then
253! Use Pearson's series expansion.
254! Original note: (Note that P is not large enough to force overflow in logAM).
255 arg = p * log(x) - x - log(gamma(p + 1.0_r8))
256 c = 1.0_r8
257 gammad = 1.0_r8
258 a = p
259
260 do
261 a = a + 1.0_r8
262 c = c * x / a
263 gammad = gammad + c
264 if(c <= tol) exit
265 end do
266
267 arg = arg + log(gammad)
268
269 if(elimit <= arg) then
270 gammad = exp(arg)
271 else
272 gammad = 0.0_r8
273 end if
274else
275 ! Use a continued fraction expansion.
276 arg = p * log(x) - x - log(gamma(p))
277 a = 1.0_r8 - p
278 b = a + x + 1.0_r8
279 c = 0.0_r8
280 pn(1) = 1.0_r8
281 pn(2) = x
282 pn(3) = x + 1.0_r8
283 pn(4) = x * b
284 gammad = pn(3) / pn(4)
285
286 do
287 a = a + 1.0_r8
288 b = b + 2.0_r8
289 c = c + 1.0_r8
290 an = a * c
291 pn(5) = b * pn(3) - an * pn(1)
292 pn(6) = b * pn(4) - an * pn(2)
293
294 if (pn(6) /= 0.0_r8) then
295 rn = pn(5) / pn(6)
296 if(abs(gammad - rn) <= min(tol, tol * rn)) exit
297 gammad = rn
298 end if
299
300 pn(1) = pn(3)
301 pn(2) = pn(4)
302 pn(3) = pn(5)
303 pn(4) = pn(6)
304
305 ! Re-scale terms in continued fraction if terms are large.
306 if (oflo <= abs(pn(5))) pn(1:4) = pn(1:4) / oflo
307
308 end do
309
310 arg = arg + log(gammad)
311
312 if (elimit <= arg) then
313 gammad = 1.0_r8 - exp(arg)
314 else
315 gammad = 1.0_r8
316 endif
317endif
318
319end function gammad

323function random_gamma(r, rshape, rscale)
324
325! Note that this provides same qualitative functionality as a similarly named
326! routine in the random_seq_mod that uses a rejection algorithm. However, once
327! we have an inverse cdf function for a distribution, it is possible to generate
328! random numbers by first getting a draw from a U(0, 1) and then inverting these
329! quantiles to get an actual value
330
331type(random_seq_type), intent(inout) :: r
332real(r8), intent(in) :: rshape
333real(r8), intent(in) :: rscale
334real(r8) :: random_gamma
335
336real(r8) :: quantile
337if (rshape <= 0.0_r8) then
338 write(errstring, *) 'Shape parameter must be positive, was ', rshape
339 call error_handler(e_err, 'random_gamma', errstring, source)
340endif
341
342if (rscale <= 0.0_r8) then
343 write(errstring, *) 'Scale parameter (scale=1/rate) must be positive, was ', rscale
344 call error_handler(e_err, 'random_gamma', errstring, source)
345endif
346
347! Draw from U(0, 1) to get a quantile
348quantile = random_uniform(r)
349! Invert cdf to get a draw from gamma
350random_gamma = inv_gamma_cdf(quantile, rshape, rscale, .true., .false., 0.0_r8, missing_r8)
351
352end function random_gamma

356subroutine gamma_shape_scale(x, num, gamma_shape, gamma_scale)
357
358integer, intent(in) :: num
359real(r8), intent(in) :: x(num)
360real(r8), intent(out) :: gamma_shape
361real(r8), intent(out) :: gamma_scale
362
363! This subroutine computes a shape and scale from a sample
364! It first computes the mean and sd, then converts
365! Note that this is NOT the maximum likelihood estimator from the sample
366! and computing that would be an alternative method to get shape and scale
367
368real(r8) :: mean, variance
369
370mean = sum(x) / num
371variance = sum((x - mean)**2) / (num - 1)
372
373call gamma_mn_var_to_shape_scale(mean, variance, gamma_shape, gamma_scale)
374
