11! convert_gpsro_bufr - program that reads a COSMIC GPS observation
12! in BUFR format and writes the data to a DART
13! observation sequence file.
14!
15! For platforms with little-endian binary file format (e.g. Intel, AMDÂ®, and
16! non-MIPS SGI processors), you may need to first run the byte-swapping program
17! DART/observations/NCEP/prep_bufr/exe/grabbufr.x over the bufr file
18! before running this program. For the info, check
19! DART/observations/NCEP/prep_bufr/prep_bufr.html.
20!
21! created August 2015 Soyoung Ha, NCAR/MMM
22!
23! Tested with gdas.gpsro.2012060100 for local operator only.

322!------------------------------------------------------------------------
323! main loop that does either a single file or a list of files
324!------------------------------------------------------------------------
325
326filenum = 1
327fileloop: do ! until out of files
328
329 ! do some bookkeeping to see how many scans are ignored and for what reason.
330 ! the print for these is inside the fileloop so the reset has to be inside
331 ! as well. if you want totals for the entire run (including multiple files)
332 ! take these out of the loop and move the prints out of the loop at the bottom
333
334 obs_num_byfile = 0
335 nprof_gps = 0
336 nprof_bytime = 0
337 nprof_cdaac_bad = 0
338 nprof_gras_bad = 0
339 nprof_levs_bad = 0

342 ! get the single name, or the next name from a list
343 if (from_list) then

351 !------------------------------------------------------------------------
352 ! Open file for input

356 !------------------------------------------------------------------------
357 ! Reading bufr data
358 !------------------------------------------------------------------------
359 call openbf(unit_in,'IN',unit_in) ! BUFRLIB
360 call datelen(10) ! BUFRLIB

364 ! Big loop over the bufr file
365 num_message=0
366 msg_report: do while (ireadmg(unit_in,subsetb,idate) == 0) ! BUFRLIB
367 num_message = num_message + 1
368 num_subset = 0
369 if (debug) write(*,'(I10,I5,a10)') idate,num_message,subsetb

371 read_loop: do while (ireadsb(unit_in) == 0) ! BUFRLIB
372 num_subset = num_subset+1
373
374 ! Extract header information
375 call ufbint(unit_in,hdr,n1ahdr,1,iret,hdr1a) ! BUFRLIB
376 call ufbint(unit_in,qfro,1,1,iret,nemo) ! BUFRLIB
377
378 ! observation time in minutes
379 iyear = hdr(1) ! year
380 imonth = hdr(2) ! month
381 iday = hdr(3) ! day
382 ihour = hdr(4) ! hour
383 imin = hdr(5) ! minute
384 pcc=hdr(6) ! profile per cent confidence
385 roc=hdr(7) ! Earth local radius of curvature
386 said=hdr(8) ! Satellite identifier (receiver)

388
389 if (debug) write(*,*) iyear, imonth, iday, ihour, imin

391 call get_time(obs_time, osec, oday)
392
393 ! Check the satellite list in convinfo
394 ikx = 0
395 find_loop: do i = 1, num_satid
396 if ( said == gps_satid(i) ) then
397 ikx = 1
398 exit find_loop
399 endif
400 enddo find_loop
401 if (ikx==0) then
402 nprof_nosat = nprof_nosat + 1
403 cycle read_loop
404 endif
405
406 ! Check profile quality flags
407 if ( ((said > 739).and.(said < 746)).or.(said == 820).or.(said == 786)) then !CDAAC processing
408 if(pcc==0.0) then
409 if (debug) write(*,*)'convert_gpsro_bufr: 1 bad profile said=',said,'ptid=',ptid,&
410 ' SKIP this report'
411 nprof_cdaac_bad = nprof_cdaac_bad + 1
412 cycle read_loop
413 endif
414 endif
415
416 if ((said == 4).or.(said == 3).or.(said == 421).or.(said == 440).or.&
417 (said == 821)) then ! GRAS SAF processing
418 call upftbv(unit_in,nemo,qfro,mxib,ibit,nib) ! BUFRLIB
419 lone = .false.
420 if(nib > 0) then
421 do i=1,nib
422 if(ibit(i)== 6) then
423 lone = .true.
424 exit
425 endif
426 enddo
427 endif
428 if(lone) then
429 if (debug) write(*,*) 'convert_gpsro_bufr: 2 bad profile said=',said,'ptid=',ptid,&
430 ' SKIP this report'
431 nprof_gras_bad = nprof_gras_bad + 1
432 cycle read_loop
433 endif
434 endif
435
436 ! Read observations
437 call ufbint(unit_in,nreps_this_roseq2,1,maxlevs,nreps_roseq1,'{ROSEQ2}') ! BUFRLIB
438 call ufbseq(unit_in,data1b,50,maxlevs, nlevs,'ROSEQ1') ! bending angle ! BUFRLIB
439 call ufbseq(unit_in,data2a,50,maxlevs,nlevels,'ROSEQ3') ! refractivity ! BUFRLIB
440 if (nlevs/=nlevels) then
441 if (debug) write(*,'(A,2I5,2(A,I5),A)') 'convert_gpsro_bufr: *WARNING* said,ptid=',said,ptid,&
442 ' with gps_bnd nlevs=', nlevs,&
443 ' and gps_ref nlevels=',nlevels,&
444 ' SKIP this report'
445 nprof_levs_bad = nprof_levs_bad + 1
446 cycle read_loop
447 endif
448
449 ! Exclude observations outside the time window.
450 if ( obs_window_hr > 0.0 ) then
451 if (obs_time <= window_min .or. obs_time > window_max ) then
452 num_excluded_bytime = num_excluded_bytime + nlevels
453 nprof_bytime = nprof_bytime + 1
454 cycle read_loop
455 end if
456 end if
457
458 ! Increment report counters
459 nprof_gps = nprof_gps + 1 ! count reports in bufr file
460 if (debug) write(*,'(5(A8,I5))') ' nprof =',nprof_gps,' said =',said,' ptid=',ptid,&
461 ' nlevs =',nlevs,' nlevels =',nlevels
462
463 ! Loop over nlevels in profile
464 obsloop2: do k = 1, nlevels
465
466 rlat=data1b(1,k) ! earth relative latitude (degrees)
467 rlon=data1b(2,k) ! earth relative longitude (degrees)
468 height=data2a(1,k) ! geometric height above geoid (m)
469 ref=data2a(2,k) ! refractivity obs (units of N)
470 referr=data2a(4,k) ! gps ref obs error (units of N)
471 ref_pccf=data2a(6,k) ! percent confidence (%)
472
473 if(use_bending_angle) then ! for later
474
475 ! Loop over number of replications of ROSEQ2 nested inside ROSEQ1
476 nk = nreps_this_roseq2(k)
477 do i=1,nk
478 m=(6*i)-2
479 freq_chk=data1b(m,k) ! frequency (hertz)
480 if(nint(freq_chk).ne.0) cycle ! do not want non-zero freq., go on
481 ! to next replication of ROSEQ2
482 freq=data1b(m,k)
483 impact=data1b(m+1,k) ! impact parameter (m)
484 bend=data1b(m+2,k) ! bending angle (rad)
485 bend_error=data1b(m+4,k) ! RMSE in bending angle (rad)
486 enddo
487 bend_pccf=data1b((6*nk)+4,k) ! percent confidence for this ROSEQ1 replication
488
489 endif !if(use_bending_angle) then
490
491 ! Preliminary (sanity) checks for bad and missing data
492 good=.true.
493 if ((abs(rlat)>90.0_r8).or.(abs(rlon)>360.0_r8)) good=.false.
494 if ((ref>=1.e+9_r8).or.(ref<=0.0_r8)) good=.false.
495 if ((height<=ray_hbot).or.(height>=ray_htop)) good=.false.
496 ! Remove MetOP/GRAS ata below 8 km (following GSI)
497 if ((height<=8000.0_r8).and.((said==4).or.(said==3))) good=.false.
498
499 if(use_bending_angle) then
500 if ((bend>=1.e+9_r8).or.(bend<=0.0).or.(impact>=1.e+9_r8).or.(impact<roc)) then
501 good=.false.
502 endif
503 endif
504
505 ! If observation is "good" print out sample data
506 if(good) then
507
508 obsval = ref ! unit "N"
509
510 geopot = compute_geopotential_height(height,rlat)
511
512 if(obs_error_in_gsi) then
513 !oerr = 0.01_r8 * obsval
514 oerr = gsi_refractivity_error(height, rlat, if_global, gsi_error_inv_factor)
515 else
516 oerr = 0.01_r8 * ref_obserr_kuo_percent(height * 0.001_r8) * obsval
517 endif
518 subset = 'GPSREF'
519
520 if(use_bending_angle) then
521 obsval = bend
522 if((impact-roc) <= 10000.0_r8) then
523 oerr=(-bend*0.09/10000.0_r8)*(impact-roc)+bend*1.e-1_r8
524 else
525 oerr=max(7.e-6_r8,bend*1.e-2_r8)
526 endif
527 endif
528
529 if(rlon==360.0_r8) rlon=0.0_r8
530 if(rlon < 0.0_r8) rlon=rlon+360.0_r8
531
532 if(debug.and.(k.lt.10)) then
533 if(use_bending_angle) then
534 write(*,'(A2,I5,4f9.2,f20.3)') 'k=',k,rlat,rlon,height,bend,oerr
535 else
536 write(*,'(A2,I5,6f9.2)') 'k=',k,rlat,rlon,height,ref,oerr,referr
537 endif
538 endif
539
540 if(convert_to_geopotential_height) height = geopot
541
542 ! the first arg here is intent(out), and is passed back in to set_obs_def_key()
543 call set_gpsro_ref(gps_obs_num, nx, ny, nz, rfict, ray_ds, ray_htop, subset)

652!
653! ref_obserr_kuo_percent - function that computes the observation
654! error for a refractivity observation.
655! These numbers are taken from a Kuo paper.
656!
657! hght - height of refractivity observation
658!
659! created Hui Liu NCAR/MMM
660! modified June 2008, Ryan Torn NCAR/MMM
661! modified August 2015, Soyoung Ha NCAR/MMM
662!
663!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
664function ref_obserr_kuo_percent(hght)
665
666use types_mod, only : r8
667
668implicit none
669
670integer, parameter :: nobs_level = 22 ! maximum number of obs levels
671
672real(r8), intent(in) :: hght
673
674integer :: k, k0
675real(r8) :: hght0, ref_err(nobs_level), obs_ht(nobs_level), ref_obserr_kuo_percent
676
677! observation error heights (km) and errors
678data obs_ht/17.98_r8, 16.39_r8, 14.97_r8, 13.65_r8, 12.39_r8, 11.15_r8, &
679 9.95_r8, 8.82_r8, 7.82_r8, 6.94_r8, 6.15_r8, 5.45_r8, &
680 4.83_r8, 4.27_r8, 3.78_r8, 3.34_r8, 2.94_r8, 2.59_r8, &
681 2.28_r8, 1.99_r8, 1.00_r8, 0.00_r8/
682
683data ref_err/0.48_r8, 0.56_r8, 0.36_r8, 0.28_r8, 0.33_r8, 0.41_r8, &
684 0.57_r8, 0.73_r8, 0.90_r8, 1.11_r8, 1.18_r8, 1.26_r8, &
685 1.53_r8, 1.85_r8, 1.81_r8, 2.08_r8, 2.34_r8, 2.43_r8, &
686 2.43_r8, 2.43_r8, 2.43_r8, 2.43_r8/
687
688hght0 = max(hght, 0.0_r8)
689
690do k = 1, nobs_level
691
692 k0 = k
693 if ( hght0 >= obs_ht(k) ) exit
694
695end do
696
697if(k0.eq.1) then ! HA
698 ref_obserr_kuo_percent = ref_err(k0)
699else
700 ref_obserr_kuo_percent = ref_err(k0) + (ref_err(k0) - ref_err(k0-1)) / &
701 (obs_ht(k0)-obs_ht(k0-1)) * (hght0-obs_ht(k0))
702endif
703
704return
705end function ref_obserr_kuo_percent

707!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
708!
709! compute_geopotential_height
710! subroutine converts geometric height to geopotential height
711!
712! Input:
713! H -- input real value geometric height [m]
714! lat -- latitude in degree
715!
716! Output:
717! Z -- output real value geopotential height [m]
718!
719!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
720function compute_geopotential_height(H, lat)
721 real(r8), intent(in) :: h
722 real(r8), intent(in) :: lat
723 real(r8) :: compute_geopotential_height
724
725! -----------------------------------------------------------------------*/
726 real(r8) :: pi2, latr
727 real(r8) :: semi_major_axis, semi_minor_axis, grav_polar, grav_equator
728 real(r8) :: earth_omega, grav_constant, flattening, somigliana
729 real(r8) :: grav_ratio, sin2, termg, termr, grav, eccentricity
730
731! Parameters below from WGS-84 model software inside GPS receivers.
732 parameter(semi_major_axis = 6378.1370d3) ! (m)
733 parameter(semi_minor_axis = 6356.7523142d3) ! (m)
734 parameter(grav_polar = 9.8321849378) ! (m/s2)
735 parameter(grav_equator = 9.7803253359) ! (m/s2)
736 parameter(earth_omega = 7.292115d-5) ! (rad/s)
737 parameter(grav_constant = 3.986004418d14) ! (m3/s2)
738 parameter(grav = 9.80665d0) ! (m/s2) WMO std g at 45 deg lat
739 parameter(eccentricity = 0.081819d0) ! unitless
740 parameter(pi2 = 3.14159265358979d0/180.d0)
741
742! Derived geophysical constants
743 parameter(flattening = (semi_major_axis-semi_minor_axis) / semi_major_axis)
744
745 parameter(somigliana = (semi_minor_axis/semi_major_axis)*(grav_polar/grav_equator)-1.d0)
746
747 parameter(grav_ratio = (earth_omega*earth_omega * &
748 semi_major_axis*semi_major_axis * semi_minor_axis)/grav_constant)
749
750! Sanity Check
751 if(lat.gt.90 .or. lat.lt.-90) then
752 print*, 'compute_geopotential_height: Latitude is not between -90 and 90 degrees: ',lat
753 return
754 endif
755
756 latr = lat * (pi2) ! in radians
757 sin2 = sin(latr) * sin(latr)
758 termg = grav_equator * ( (1.d0+somigliana*sin2) / &
759 sqrt(1.d0-eccentricity*eccentricity*sin2) )
760 termr = semi_major_axis / (1.d0 + flattening + grav_ratio - 2.d0*flattening*sin2)
761
762 compute_geopotential_height = (termg/grav)*((termr*h)/(termr+h))
763 !print '(A,3f15.5)','compute_geopotential_height:',&
764 ! H,compute_geopotential_height,H-compute_geopotential_height
765
766end function compute_geopotential_height

772!
773! Input:
774! H -- input real value geometric height [m]

779! gsi_refractivity_error -- output refractivity observation error [N]

781!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
782function gsi_refractivity_error(H, lat, is_it_global, factor)
783 real(r8), intent(in) :: h

811 else
812 rerr = -1.2_r8 + 0.065_r8 * zkm + 0.021_r8 * zkm ** 2
813 endif
814 endif
815
816 endif ! if(is_it_global) then
817
818 gsi_refractivity_error = 1./(abs(exp(rerr))*factor)
819 !gsi_refractivity_error = 1./abs(exp(rerr))
