5program convert_amsu_l1
6
7! Program to convert the AMSU/A 'AIRABRAD' brightness temperatures
8! from netCDF to a DART observation sequence file.
9!
10! See the REAMDE.rst in this directory for more information, but the
11! data citation information for the dataset is
12!
13! Title: AIRS/Aqua L1B AMSU (A1/A2) geolocated and calibrated brightness temperatures V005
14! Version: 005
15! Creator: AIRS project
16! Publisher: Goddard Earth Sciences Data and Information Services Center (GES DISC)
17! Release Date: 2007-07-26T00:00:00.000Z
18! Linkage: https://disc.gsfc.nasa.gov/datacollection/AIRABRAD_005.html
19!
20! The data are originally distributed in HDF-EOS format and can be converted to netCDF
21! by the h4tonccf_nc4 program available from http://hdfeos.org/software/h4cflib.php
22
23use types_mod, only : r8, deg2rad, pi
24
25use obs_sequence_mod, only : obs_sequence_type, write_obs_seq, &
26 static_init_obs_sequence, destroy_obs_sequence, &
27 print_obs_seq_summary, get_num_obs
28
29use utilities_mod, only : initialize_utilities, &
30 error_handler, finalize_utilities, e_err, e_msg, &
31 find_namelist_in_file, check_namelist_read, &
32 do_nml_file, do_nml_term, set_filename_list, &
33 nmlfileunit, get_next_filename
34
35use amsua_bt_mod, only : amsua_bt_granule
36
37use amsua_netcdf_support_mod, only : initialize_amsua_netcdf, &
38 channel_list_to_indices, &
39 read_amsua_bt_netcdf_granule, &

47integer, parameter :: maxfiles = 512
48integer, parameter :: amsua_bt_channel = 15
49
50! ----------------------------------------------------------------------
51! Declare namelist parameters
52! ----------------------------------------------------------------------
53
54character(len=256) :: l1_files(maxfiles) = ''
55character(len=256) :: l1_file_list = ''
56character(len=256) :: outputfile = 'obs_seq.amsua'
57
58real(r8) :: lon1 = 0.0_r8, & ! lower longitude bound
59 lon2 = 360.0_r8, & ! upper longitude bound
60 lat1 = -90.0_r8, & ! lower latitude bound
61 lat2 = 90.0_r8 ! upper latitude bound
62
63character(len=8) :: channel_list(amsua_bt_channel) = 'null'
64integer :: cross_track_thin = 0
65integer :: along_track_thin = 0
66logical :: append_output = .true.
67integer :: verbose = 0 ! 0 is quiet, 3 is lots of output
68
69namelist /convert_amsu_l1_nml/ l1_files, l1_file_list, &
70 outputfile, &
71 lon1, lon2, lat1, lat2, &
72 channel_list, &
73 cross_track_thin, &
74 along_track_thin, &
75 append_output, &
76 verbose
77
78! ----------------------------------------------------------------------
79! Declare local parameters
80! ----------------------------------------------------------------------
81
82integer :: io, iunit, ifile
83integer :: filecount
84integer :: max_num
85integer :: num_inserted
86type(amsua_bt_granule) :: granule
87type(obs_sequence_type) :: big_sequence, small_sequence
88logical :: use_channels(amsua_bt_channel) = .false.
89
90! version controlled file description for error handling, do not edit
91character(len=*), parameter :: source = 'convert_amsu_L1.f90'
92
93character(len=512) :: string1
94
95! ----------------------------------------------------------------------
96! start of executable program code
97! ----------------------------------------------------------------------
98
99call initialize_utilities('convert_amsu_L1')
100call static_init_obs_sequence()
101
102!----------------------------------------------------------------------
103! Read the namelist
104!----------------------------------------------------------------------
105
106call find_namelist_in_file('input.nml', 'convert_amsu_L1_nml', iunit)
107read(iunit, nml = convert_amsu_l1_nml, iostat = io)
108call check_namelist_read(iunit, io, 'convert_amsu_L1_nml')
109
110! Record the namelist values used for the run ...
111if (do_nml_file()) write(nmlfileunit, nml=convert_amsu_l1_nml)
112if (do_nml_term()) write( * , nml=convert_amsu_l1_nml)
113
114! when this routine returns, the l1_files variable will have
115! all the filenames, regardless of which way they were specified.
116filecount = set_filename_list(l1_files, l1_file_list, "convert_amsu_L1")
117
118! use the first file to initialize the module
119call initialize_amsua_netcdf(l1_files(1),verbose)
120
121! FIXME ... maybe ... are all the files guaranteed to have the
122! same channels - I think so.
123call channel_list_to_indices(channel_list, use_channels)
124
125! Figure out how many (more) observations there may be in the output.
126! It is impossible to predict how the lat/lon subsetting may affect this
127! number, but the thinning and channels can be taken into account.
128
129max_num = max_possible_obs(filecount, cross_track_thin, along_track_thin, use_channels)
130
131! either read existing obs_seq or create a new one
132
133call append_or_create(append_output,outputfile,max_num,big_sequence)
134
135! read from netCDF file into a derived type that holds all the information
136! convert derived type information to DART sequence for that file
137! combine that sequence to the output sequence
138! release the file sequence memory in preparation for creating it new next iteration
139
140do ifile = 1,filecount
141
142 call read_amsua_bt_netcdf_granule(l1_files(ifile), granule)
143
144 call make_obs_sequence(small_sequence, granule, lon1, lon2, lat1, lat2, &
145 use_channels, along_track_thin, cross_track_thin)
146
147 num_inserted = combine_sequences(small_sequence, big_sequence, l1_files(ifile))
148
149 if (verbose > 2) then
150 call print_obs_seq_summary(small_sequence)
151 elseif (verbose > 1) then
152 write(string1,*)trim(l1_files(ifile))//' added ',num_inserted,' observations.'
153 call error_handler(e_msg,string1,'')
154 endif
155
156 call destroy_obs_sequence(small_sequence)
157
158enddo
159
160! Report on the current status of the entire observation sequence
161if (verbose > 0) then
162 call print_obs_seq_summary(big_sequence)
163endif
164
165! DISCUSSION
166! There is some discussion in the group about whether this program should
167! error out or continue if there are no observations in the sequence.
168! If used in a batch job, maybe you're OK with not having an output observation
169! sequence for this collection of files -OR- you might want to know that there
170! are no viable observations for a particular collection of files.
171! If you want to allow this program to exit without returning an error code,
172! change E_ERR to E_MSG in the following call to error_handler().
173
174! Write the sequence to a disk file
175if ( get_num_obs(big_sequence) > 0 ) then
176 call write_obs_seq(big_sequence, outputfile)
177else
178 call error_handler(e_err,'NO OBSERVATIONS TO WRITE',source, &
179 text2='see the DISCUSSION comment in the source to exit without error.')
180endif
181
182! release the sequence memory
183call destroy_obs_sequence(big_sequence)
184
185call error_handler(e_msg, source, 'Finished successfully.',source)
186call finalize_utilities()
187
188end program convert_amsu_l1
