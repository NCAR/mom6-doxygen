66 io, iunit, nobs, filenum, dummy, numrejected

68logical :: file_exist, first_obs, from_list = .false.

70 qc, lato, lono, hghto, refro, azimo, wght, nx, ny, &
71 nz, rfict, obsval, phs, obs_val(1), qc_val(1)

81!------------------------------------------------------------------------
82! Declare namelist parameters
83!------------------------------------------------------------------------
84
85integer, parameter :: nmaxlevels = 200 ! max number of observation levels
86
87logical :: local_operator = .true. ! see html file for more on non/local
88logical :: use_original_kuo_error = .false. ! alternative is use lidia c's version
89real(r8) :: obs_levels(nmaxlevels) = -1.0_r8
90real(r8) :: ray_ds = 5000.0_r8 ! delta stepsize (m) along ray, nonlocal op
91real(r8) :: ray_htop = 15000.0_r8 ! max height (m) for nonlocal op
92character(len=256) :: gpsro_netcdf_file = 'cosmic_gps_input.nc'
93character(len=256) :: gpsro_netcdf_filelist = ''
94character(len=256) :: gpsro_out_file = 'obs_seq.gpsro'
95
96namelist /convert_cosmic_gps_nml/ obs_levels, local_operator, ray_ds, &
97 ray_htop, gpsro_netcdf_file, &
98 gpsro_netcdf_filelist, gpsro_out_file, &
99 use_original_kuo_error
100
101! initialize some values
102obs_num = 1
103qc = 0.0_r8
104first_obs = .true.
105call set_calendar_type(gregorian)
106
107! read the necessary parameters from input.nml
108call initialize_utilities()
109
110call find_namelist_in_file("input.nml", "convert_cosmic_gps_nml", iunit)
111read(iunit, nml = convert_cosmic_gps_nml, iostat = io)
112call check_namelist_read(iunit, io, "convert_cosmic_gps_nml")
113
114! Record the namelist values used for the run
115if (do_nml_file()) write(nmlfileunit, nml=convert_cosmic_gps_nml)
116if (do_nml_term()) write( * , nml=convert_cosmic_gps_nml)
117
118! namelist checks for sanity
119if (.not. use_original_kuo_error .and. .not. local_operator) then
120 call error_handler(e_err, 'convert_cosmic_gps_cdf', &
121 'New error values only implemented for local operator', &
122 source, revision, revdate)
123endif
124
125! count observation levels, make sure observation levels increase from 0
126nlevels = 0
127levelloop: do k = 1, nmaxlevels
128 if ( obs_levels(k) == -1.0_r8 ) exit levelloop
129 nlevels = k
130end do levelloop
131do k = 2, nlevels
132 if ( obs_levels(k-1) >= obs_levels(k) ) then
133 call error_handler(e_err, 'convert_cosmic_gps_cdf', &
134 'Observation levels should increase', &
135 source, revision, revdate)
136 end if
137end do
138
139! cannot have both a single filename and a list; the namelist must
140! shut one off.
141if (gpsro_netcdf_file /= '' .and. gpsro_netcdf_filelist /= '') then
142 call error_handler(e_err, 'convert_cosmic_gps_cdf', &
143 'One of gpsro_netcdf_file or filelist must be NULL', &
144 source, revision, revdate)
145endif
146if (gpsro_netcdf_filelist /= '') from_list = .true.
147
148! need to know a reasonable max number of obs that could be added here.
149if (from_list) then
150 call find_textfile_dims(gpsro_netcdf_filelist, nfiles, dummy)
151 num_new_obs = nlevels * nfiles
152else
153 num_new_obs = nlevels
154endif
155
156! either read existing obs_seq or create a new one
157call static_init_obs_sequence()
158call init_obs(obs, num_copies, num_qc)
159call init_obs(prev_obs, num_copies, num_qc)
160inquire(file=gpsro_out_file, exist=file_exist)
161if ( file_exist ) then
162
163 write(msgstring, *) "found existing obs_seq file, appending to ", trim(gpsro_out_file)
164 write(msgstring2, *) "adding up to a maximum of ", num_new_obs, " new observations"
165 call error_handler(e_msg, 'convert_cosmic_gps_cdf', msgstring, &
166 source, revision, revdate, text2=msgstring2)
167 call read_obs_seq(gpsro_out_file, 0, 0, num_new_obs, obs_seq)
168
169else
170
171 write(msgstring, *) "no previously existing obs_seq file, creating ", trim(gpsro_out_file)
172 write(msgstring2, *) "with up to a maximum of ", num_new_obs, " observations"
173 call error_handler(e_msg, 'convert_cosmic_gps_cdf', msgstring, &
174 source, revision, revdate, text2=msgstring2)
175
176 call init_obs_sequence(obs_seq, num_copies, num_qc, num_new_obs)
177 call set_copy_meta_data(obs_seq, 1, 'COSMIC GPS Observation')
178 call set_qc_meta_data(obs_seq, 1, 'COSMIC QC')
179
180end if
181
182
183! main loop that does either a single file or a list of files.
184! the data is currently distributed as a single profile per file.
185
186filenum = 1
187numrejected = 0
188
189fileloop: do ! until out of files
190
191 ! get the single name, or the next name from a list
192 if (from_list) then
193 next_infile = get_next_filename(gpsro_netcdf_filelist, filenum)
194 else
195 next_infile = gpsro_netcdf_file
196 if (filenum > 1) next_infile = ''
197 endif
198 if (next_infile == '') exit fileloop
199
200 ! open the next occultation profile
201 call nc_check( nf90_open(next_infile, nf90_nowrite, ncid), 'file open', next_infile)
202
203 ! this is new with the cosmic 2013 reprocessed obs. they have included profiles
204 ! which have failed their QC checking but added a global attribute 'bad' which
205 ! is char "1" for bad profiles. if there is no 'bad' attribute, or if there is one
206 ! and the value is "0", then it is ok to proceed. otherwise close the file and
207 ! cycle the loop to pick up the next file.
208
209 io = nf90_get_att(ncid,nf90_global,'bad', badqc)
210 if ((io == nf90_noerr) .and. (badqc /= "0")) then
211 ! profile with a bad QC
212 numrejected = numrejected + 1
213 call nc_check( nf90_close(ncid) , 'close file', next_infile)
214 filenum = filenum + 1
215 cycle fileloop
216 endif
217
218 ! process the profile
219 call nc_check( nf90_get_att(ncid,nf90_global,'year', iyear) ,'get_att year', next_infile)
220 call nc_check( nf90_get_att(ncid,nf90_global,'month', imonth),'get_att month', next_infile)
221 call nc_check( nf90_get_att(ncid,nf90_global,'day', iday) ,'get_att day', next_infile)
222 call nc_check( nf90_get_att(ncid,nf90_global,'hour', ihour) ,'get_att hour', next_infile)
223 call nc_check( nf90_get_att(ncid,nf90_global,'minute',imin) ,'get_att minute', next_infile)
224 call nc_check( nf90_get_att(ncid,nf90_global,'second',isec) ,'get_att second', next_infile)
225
226 time_obs = set_date(iyear, imonth, iday, ihour, imin, isec)
227 call get_time(time_obs, osec, oday)
228
229 call nc_check( nf90_inq_dimid(ncid, "MSL_alt", varid), 'inq dimid MSL_alt', next_infile)
230 call nc_check( nf90_inquire_dimension(ncid, varid, name, nobs), 'inq dim MSL_alt', next_infile)
231
232 call nc_check( nf90_get_att(ncid,nf90_global,'lon', glon) ,'get_att lon', next_infile)
233 call nc_check( nf90_get_att(ncid,nf90_global,'lat', glat) ,'get_att lat', next_infile)
234 call nc_check( nf90_get_att(ncid,nf90_global,'rfict',rfict),'get_att rfict', next_infile)
235 rfict = rfict * 1000.0_r8
236
237 allocate(hghtp(nobs)) ; allocate(refrp(nobs))
238 allocate( lat(nobs)) ; allocate( lon(nobs))
239 allocate(hght(nobs)) ; allocate(refr(nobs))
240 allocate(azim(nobs)) ; allocate(benda(nobs))
241
242 ! read the latitude array
243 call nc_check( nf90_inq_varid(ncid, "Lat", varid) ,'inq varid Lat', next_infile)
244 call nc_check( nf90_get_var(ncid, varid, lat) ,'get var Lat', next_infile)
245
246 ! read the latitude array
247 call nc_check( nf90_inq_varid(ncid, "Lon", varid) ,'inq varid Lon', next_infile)
248 call nc_check( nf90_get_var(ncid, varid, lon) ,'get var Lon', next_infile)
249
250 ! read the altitude array
251 call nc_check( nf90_inq_varid(ncid, "MSL_alt", varid) ,'inq varid MSL_alt', next_infile)
252 call nc_check( nf90_get_var(ncid, varid, hght) ,'get_var MSL_alt', next_infile)
253 call nc_check( nf90_get_att(ncid, varid, '_FillValue', hght_miss) ,'get_att _FillValue MSL_alt', next_infile)
254
255 ! read the refractivity
256 call nc_check( nf90_inq_varid(ncid, "Ref", varid) ,'inq varid Ref', next_infile)
257 call nc_check( nf90_get_var(ncid, varid, refr) ,'get var Ref', next_infile)
258 call nc_check( nf90_get_att(ncid, varid, '_FillValue', refr_miss) ,'get_att _FillValue Ref', next_infile)
259
260 ! note about bending angle:
261 ! it is currently unused but people are interested in trying to assimilate this
262 ! instead of refractivity. i don't think we have a forward operator for it yet.
263
264 ! read the bending angle
265 call nc_check( nf90_inq_varid(ncid, "Bend_ang", varid) ,'inq varid Bend_ang', next_infile)
266 call nc_check( nf90_get_var(ncid, varid, benda) ,'get var Bend_ang', next_infile)
267 call nc_check( nf90_get_att(ncid, varid, '_FillValue', benda_miss) ,'get_att _FillValue benda', next_infile)
268
269 ! read the azimuth of occultation plane - only used for non-local operator
270 call nc_check( nf90_inq_varid(ncid, "Azim", varid) ,'inq varid Azim', next_infile)
271 call nc_check( nf90_get_var(ncid, varid, azim) ,'get var Azim', next_infile)
272 call nc_check( nf90_get_att(ncid, varid, '_FillValue', azim_miss) ,'get_att _FillValue Azim', next_infile)
273
274 call nc_check( nf90_close(ncid) , 'close file', next_infile)
275
276 !where(hght(1: < -998)) then
277 ! write(*,*) 'height array:'
278 ! write(*,*) hght
279 !endwhere
280 !where (any(refr < -998)) then
281 ! write(*,*) 'refractivity array:'
282 ! write(*,*) refr
283 !endwhere
284 !where (any(benda < -998)) then
285 ! write(*,*) 'bending angle array:'
286 ! write(*,*) benda
287 !endwhere
288
289 ! convert units here.
290 hghtp(:) = hght(:) * 1000.0_r8
291 refrp(:) = refr(:) * 1.0e-6_r8
292
293 obsloop2: do k = 1, nlevels
294
295 call interp_height_wght(hght, obs_levels(k), nobs, zloc, wght)
296 if ( zloc < 1 ) cycle obsloop2
297
298 ! lon(zloc) and lon(zloc+1) range from -180 to +180
299 ! call a subroutine to handle the wrap point.
300 lono = compute_lon_wrap(lon(zloc), lon(zloc+1), wght)
301 lato = wght * lat(zloc) + (1.0_r8 - wght) * lat(zloc+1)
302 hghto = wght * hght(zloc) + (1.0_r8 - wght) * hght(zloc+1)
303 hghto = hghto * 1000.0_r8
304 refro = wght * refr(zloc) + (1.0_r8 - wght) * refr(zloc+1)
305 azimo = wght * azim(zloc) + (1.0_r8 - wght) * azim(zloc+1)
306
307 if ( local_operator ) then
308
309 nx = 0.0_r8
310 ny = 0.0_r8
311 nz = 0.0_r8
312 ray_ds = 0.0_r8
313 ray_htop = 0.0_r8
314 rfict = 0.0_r8
315
316 obsval = refro
317 if (use_original_kuo_error) then
318 oerr = 0.01_r8 * ref_obserr_kuo_percent(hghto * 0.001_r8) * obsval
319 else
320 oerr = gsi_refractivity_error(hghto, lato, is_it_global=.true., factor=1.0_r8)
321 endif
322 subset = 'GPSREF'
323
324 else
325
326 ! compute tangent unit vector
327 call tanvec01(lono, lato, azimo, nx, ny, nz)
328
329 ! compute the excess phase
330 call excess(refrp, hghtp, lono, lato, hghto, nx, &
331 ny, nz, rfict, ray_ds, ray_htop, phs, nobs)
332
333 ! if too high, phs will return as 0. cycle loop here.
334 if (phs <= 0) cycle obsloop2
335
336 obsval = phs
337 oerr = 0.01_r8 * excess_obserr_percent(hghto * 0.001_r8) * obsval
338 !DEBUG print *, 'hghto,obsval,perc,err=', hghto, obsval, &
339 !DEBUG excess_obserr_percent(hghto * 0.001_r8), oerr
340 subset = 'GPSEXC'
341
342 end if
343
344 call set_gpsro_ref(obs_num, nx, ny, nz, rfict, ray_ds, ray_htop, subset)
345 call set_obs_def_location(obs_def,set_location(lono,lato,hghto,vertisheight))
346 call set_obs_def_type_of_obs(obs_def, gpsro_refractivity)
347 call set_obs_def_time(obs_def, set_time(osec, oday))
348 call set_obs_def_error_variance(obs_def, oerr * oerr)
349 call set_obs_def_key(obs_def, obs_num)
350 call set_obs_def(obs, obs_def)
351
352 obs_val(1) = obsval
353 call set_obs_values(obs, obs_val)
354 qc_val(1) = qc
355 call set_qc(obs, qc_val)
356
357 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
358
359 obs_num = obs_num+1
360
361 end do obsloop2
362
363 ! clean up and loop if there is another input file
364 deallocate( lat, lon, hght, refr, azim, benda, hghtp, refrp)
365
366 filenum = filenum + 1
367
368end do fileloop
369
370! done with main loop. if we added any new obs to the sequence, write it out.
371if (obs_num > 1) call write_obs_seq(obs_seq, gpsro_out_file)
372
373! cleanup memory
374call destroy_obs_sequence(obs_seq)
375call destroy_obs(obs) ! do not destroy prev_obs, which is same as obs
376
377write(msgstring, *) 'processed ', filenum-1, ' total profiles'
378call error_handler(e_msg, 'convert_cosmic_gps_obs', msgstring, source, revision, revdate)
379
380if (numrejected > 0) then
381 write(msgstring, *) numrejected, ' profiles rejected for bad incoming quality control'
382 call error_handler(e_msg, 'convert_cosmic_gps_obs', msgstring, source, revision, revdate)
383endif
384
385call finalize_utilities()
386
387! END OF MAIN ROUTINE
388
389contains

393!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
394!
395! excess - subroutine that computes the excess phase based on the
396! method of Sergey et al. (2004), eq. 1 and 2
397!
398! refp - refractivity profile
399! hghtp - height profile of refractivity observations
400! lon - longitude of GPS perigee point
401! lat - latitude of GPS perigee point
402! height - height of GPS perigee point
403! nx - x component of tangent point of ray
404! ny - y component of tangent point of ray
405! nz - z component of tangent point of ray
406! rfict - local curvature radius
407! ds - increment of excess computation - intent(in) now
408! htop - heighest level of computation - intent(in) now
409! phs - excess phase
410! nobs - number of points in profile
411!
412! created June 2004, Hui Liu NCAR/MMM
413! modified Ryan Torn, NCAR/MMM
414! updated Nancy Collins, NCAR/IMAGe
415! quit before computing segment when one endpoint is above top, not after
416! pass delta step and htop in as input args instead of out.
417!
418!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
419subroutine excess(refp, hghtp, lon, lat, height, nx, ny, nz, &
420 rfict, ds, htop, phs, nobs)
421
422use types_mod, only : r8
423
424implicit none
425
426integer, intent(in) :: nobs
427real(r8), intent(in) :: refp(nobs), hghtp(nobs), lon, lat, height, nx, &
428 ny, nz, rfict, ds, htop
429real(r8), intent(out) :: phs
430
431integer :: iter
432real(r8) :: xo, yo, zo, ref00, ref1, ref2, ss, xx, yy, zz, height1, &
433 lat1, lon1, dphs1, dphs2
434
435! make sure height is not already above requested top of integration
436if( height >= htop ) then
437 phs = 0.0_r8 ! excess phase
438 return
439endif
440
441! convert location of perigee from geodetic to Cartesian coordinate
442call geo2car(height, lat, lon, xo, yo, zo, rfict)
443
444! get refractivity at perigee
445call ref_int(height, refp, hghtp, ref00, nobs)
446
447! integrate refractivity along a straight line path in cartesian coordinate
448! (x-xo)/a = (y-yo)/b = (z-zo)/c, (a,b,c) is the line direction
449ref1 = ref00 ; ref2 = ref00
450
451! initialization
452phs = 0.0_r8 ! excess phase
453ss = 0.0_r8 ! distance to perigee from a ray point
454iter = 0
455
456do
457
458 iter = iter + 1
459 ss = ss + ds
460
461 ! integrate to one direction of the line for one step
462 xx = xo + ss * nx
463 yy = yo + ss * ny
464 zz = zo + ss * nz
465
466 ! convert the location of the point to geodetic coordinates
467 ! height(m), lat, lon(deg)
468 call car2geo(xx, yy, zz, height1, lat1, lon1, rfict)
469 if( height1 >= htop ) exit ! break out of loop if above level
470
471 call ref_int(height1, refp, hghtp, ref00, nobs)
472 dphs1 = (ref1 + ref00) * ds / 2.0_r8
473
474 ref1 = ref00
475
476 ! integrate to the other direction of the line for one step
477 xx = xo - ss * nx
478 yy = yo - ss * ny
479 zz = zo - ss * nz
480
481 call car2geo(xx, yy, zz, height1, lat1, lon1, rfict)
482 call ref_int(height1, refp, hghtp, ref00, nobs)
483 dphs2 = (ref2 + ref00) * ds / 2.0_r8
484
485 ref2 = ref00
486
487 phs = phs + dphs1 + dphs2
488
489end do
490
491return
492end subroutine excess

494!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
495!
496! car2geo - subroutine that converts cartesian coordinates to
497! geodetical.
498!
499! x1 - x coordinate
500! x2 - y coordinate
501! x3 - z coordinate
502! s1 - geodetical height
503! s2 - geodetical latitude
504! s3 - geodetical longitude
505! rfict - local curvature radius
506!
507! created Hui Liu NCAR/MMM
508! Modified June 2008, Ryan Torn NCAR/MMM
509!
510!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
511subroutine car2geo (x1, x2, x3, s1, s2, s3, rfict)
512
513use types_mod, only : r8, rad2deg, pi
514
515implicit none
516
517real(r8), intent(in) :: x1, x2, x3, rfict
518real(r8), intent(out) :: s1, s2, s3
519
520real(r8) :: rho, azmth
521
522rho = sqrt(x1 * x1 + x2 * x2 + x3 * x3)
523s1 = rho - rfict
524s2 = asin(x3 / rho)
525azmth = atan2(x2, x1)
526s3 = mod((azmth + 4.0_r8 * pi), 2.0_r8 * pi)
527
528s2 = s2 * rad2deg
529s3 = s3 * rad2deg
530
531return
532end subroutine car2geo

534!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
535!
536! geo2car - subroutine that converts geodetical coordinates to
537! cartesian.

630!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
631!
632! ref_int - subroutine that calculates the refractivity at any
633! particular point.
634!
635! height - GPS observation height (m)
636! refp - refractivity profile
637! hghtp - height profile
638! lref - local refractivity value
639! nobs - number of observations in the profile
640!
641! created Hui Liu NCAR/MMM
642! updated June 2008, Ryan TORN NCAR/MMM
643!
644!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
645subroutine ref_int(height, refp, hghtp, lref, nobs)
646
647use types_mod, only : r8
648
649implicit none
650
651integer, intent(in) :: nobs
652real(r8), intent(in) :: height, refp(nobs), hghtp(nobs)
653real(r8), intent(out) :: lref
654
655integer :: bot_lev, k
656real(r8) :: fract
657
658bot_lev = -1
659fract = 0.0_r8
660
661! make sure it's not higher than the highest available level.
662if (height > hghtp(1)) then
663 write(msgstring, *) 'requested height is ', height, '; highest available is ', hghtp(1)
664 call error_handler(e_err, 'bad level, above highest in file', &
665 source, revision, revdate, text2=msgstring)
666endif
667
668! Search down through height levels
669heights: do k = 2, nobs
670 if ( height >= hghtp(k) ) then
671 bot_lev = k
672 fract = (hghtp(k) - height) / (hghtp(k) - hghtp(k-1))
673 exit heights
674 endif
675end do heights
676
677! the hghtp() array is currently an interpolated list of levels
678! and on at least 1 PGI compiler version computing the lowest value
679! rounds enough that a height exactly equal to the lowest level
680! compares as less than instead of equal. so test and if it's very very
681! close to the lowest level then return it as equal; otherwise it's
682! an internally inconsistent input file.
683if (bot_lev < 0) then
684 if (abs(height - hghtp(nobs)) < 0.00001_r8) then
685 bot_lev = nobs
686 fract = 0.0_r8
687 else
688 write(msgstring, *) 'requested height is ', height, '; lowest available is ', hghtp(nobs)
689 call error_handler(e_err, 'bad level, below lowest in file', &
690 source, revision, revdate, text2=msgstring)
691 endif
692endif
693
694lref = (1.0_r8 - fract) * refp(bot_lev) + fract * refp(bot_lev-1)
695
696return
697end subroutine ref_int

700!
701! tanvec01 - subroutine that computes the unit vector tangent of the
702! ray at the perigee.
703!
704! lon0 - longitude of the tangent point
705! lat0 - latitude of the tangent point
706! azim0 - angle between occultation plane from north
707! uz - x component of tangent vector at tangent point of array
708! uy - y component of tangent vector at tangent point of array
709! uz - z component of tangent vector at tangent point of array
710!
711! created Hui Liu, NCAR/MMM
712! modified June 2008, Ryan Torn NCAR/MMM
713!
714!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
715subroutine tanvec01(lon0, lat0, azim0, ux, uy, uz)
716
717use types_mod, only : r8, deg2rad
718
719implicit none
720
721real(r8), intent(in) :: lon0, lat0, azim0
722real(r8), intent(out) :: ux, uy, uz
723
724real(r8) :: zz0(3), lon, lat, azim, rtp(3), rnm(3), rno(3), uon(3), vlen0
725
726zz0(1) = 0.0_r8 ; zz0(2) = 0.0_r8 ; zz0(3) = 1.0_r8
727lon = lon0 * deg2rad ; lat = lat0 * deg2rad ; azim = azim0 * deg2rad
728
729rtp(1) = cos(lat) * cos(lon)
730rtp(2) = cos(lat) * sin(lon)
731rtp(3) = sin(lat)
732
733! compute unit vector normal to merdion plane through tangent point
734call vprod(rtp, zz0, rnm)
735vlen0 = sqrt(rnm(1)*rnm(1) + rnm(2)*rnm(2) + rnm(3)*rnm(3))
736rnm(:) = rnm(:) / vlen0
737
738! compute unit vector toward north from perigee point
739call vprod(rnm, rtp, rno)
740vlen0 = sqrt(rno(1)*rno(1) + rno(2)*rno(2) + rno(3)*rno(3))
741rno(:) = rno(:) / vlen0
742
743! rotate the vector rno around rtp for a single azim to get tangent vector
744call spin(rno, rtp, azim, uon)
745vlen0 = sqrt(uon(1)*uon(1) + uon(2)*uon(2) + uon(3)*uon(3))
746ux = uon(1) / vlen0 ; uy = uon(2) / vlen0 ; uz = uon(3) / vlen0
747
748return
749end subroutine tanvec01

751!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
752!
753! spin - subroutine that rotates vector v1 around vs clockwise by
754! by a specified angle.
755!
756! v1 - vector to rotate
757! vs - vector to rotate about
758! a - angle to rotate v1 around
759! v2 - output vector after rotation
760!
761! created Hui Liu NCAR/MMM
762! modified June 2008, Ryan Torn NCAR/MMM
763!
764!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
765subroutine spin(v1, vs, a, v2)
766
767use types_mod, only : r8, deg2rad
768
769implicit none
770
771real(r8), intent(in) :: v1(3), vs(3), a
772real(r8), intent(out) :: v2(3)
773
774real(r8) :: vsabs, vsn(3), a1, a2, a3, s(3,3)
775
776! Calculation of the unit vector for the rotation
777vsabs = sqrt(vs(1)*vs(1) + vs(2)*vs(2) + vs(3)*vs(3))
778vsn(:) = vs(:) / vsabs
779
780! Calculation the rotation matrix
781a1 = cos(a) ; a2 = 1.0_r8 - a1 ; a3 = sin(a)
782s(1,1) = a2 * vsn(1) * vsn(1) + a1
783s(1,2) = a2 * vsn(1) * vsn(2) - a3 * vsn(3)
784s(1,3) = a2 * vsn(1) * vsn(3) + a3 * vsn(2)
785s(2,1) = a2 * vsn(2) * vsn(1) + a3 * vsn(3)
786s(2,2) = a2 * vsn(2) * vsn(2) + a1
787s(2,3) = a2 * vsn(2) * vsn(3) - a3 * vsn(1)
788s(3,1) = a2 * vsn(3) * vsn(1) - a3 * vsn(2)
789s(3,2) = a2 * vsn(3) * vsn(2) + a3 * vsn(1)
790s(3,3) = a2 * vsn(3) * vsn(3) + a1
791
792! Compute the rotated vector
793v2(:) = s(:,1) * v1(1) + s(:,2) * v1(2) + s(:,3) * v1(3)
794
795return
796end subroutine spin

827!
828! compute_lon_wrap - interpolate between 2 longitude values taking
829! into account the wrap at -180 degrees
830!
831! lon1, lon2 - longitude in degrees between -180 and +180
832! weight - interpolation weight between lon1 and lon2 (0 to 1)
833!
834! returns interpolated longitude between 0 and 360 degrees.
835!
836! if the longitudes are the same sign (both negative or both positive)
837! then do the interpolation with the original values. if the signs
838! are different then we need to decide if they are crossing 0 (where we
839! still use the original values) or if they are crossing the -180/180 line
840! and we have to wrap the negative value.
841
842! to decide between the 0 and 180 cases, take the positive value and subtract
843! the negative value (which adds it on) and see if the sum is > 180. if not,
844! we're at the 0 crossing and we do nothing. if yes, then we add 360 to the
845! negative value and interpolate between two positive values. in either case
846! once we have the result, if it's < 0 add 360 so the longitude returned is
847! between 0 and 360 in longitude.
848!
849! this does not try to do anything special if the profile is tracking directly
850! over one of the poles. this is because at the exact poles all longitudes are
851! identical, so being off in the longitude in any direction won't be a large
852! difference in real distance.
853!
854! created nancy collins NCAR/IMAGe
855!
856!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
857function compute_lon_wrap(lon1, lon2, weight)
858
859real(r8), intent(in) :: lon1, lon2, weight
860real(r8) :: compute_lon_wrap
861
862real(r8) :: lon1a, lon2a, lono
863
864! r/w temporaries in case we have to change the value.
865lon1a = lon1
866lon2a = lon2
867
868
869! if different signs and crossing the -180/180 boundary, add 360
870! to the negative value.
871if (lon1 <= 0.0_r8 .and. lon2 >= 0.0_r8) then
872 if (lon2 - lon1 > 180.0_r8) lon1a = lon1a + 360.0_r8
873else if (lon1 >= 0.0_r8 .and. lon2 <= 0.0_r8) then
874 if (lon1 - lon2 > 180.0_r8) lon2a = lon2a + 360.0_r8
875endif
876
877! linear interpolation, and make return value between 0 and 360.
878lono = weight * lon1a + (1.0_r8 - weight) * lon2a
879if (lono < 0.0_r8) lono = lono + 360.0_r8
880
881compute_lon_wrap = lono
882
883end function compute_lon_wrap

1067! compute_geopotential_height
1068! subroutine converts geometric height to geopotential height
1069!
1070! Input:
1071! H -- input real value geometric height [m]
1072! lat -- latitude in degree
1073!
1074! Output:
1075! Z -- output real value geopotential height [m]
1076!
1077!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
1078function compute_geopotential_height(H, lat)
1079 real(r8), intent(in) :: h
1080 real(r8), intent(in) :: lat
1081 real(r8) :: compute_geopotential_height
1082
1083! -----------------------------------------------------------------------*/
1084 real(r8) :: pi2, latr
1085 real(r8) :: semi_major_axis, semi_minor_axis, grav_polar, grav_equator
1086 real(r8) :: earth_omega, grav_constant, flattening, somigliana
1087 real(r8) :: grav_ratio, sin2, termg, termr, grav, eccentricity
1088
1089! Parameters below from WGS-84 model software inside GPS receivers.
1090 parameter(semi_major_axis = 6378.1370d3) ! (m)
1091 parameter(semi_minor_axis = 6356.7523142d3) ! (m)
1092 parameter(grav_polar = 9.8321849378) ! (m/s2)
1093 parameter(grav_equator = 9.7803253359) ! (m/s2)
1094 parameter(earth_omega = 7.292115d-5) ! (rad/s)
1095 parameter(grav_constant = 3.986004418d14) ! (m3/s2)
1096 parameter(grav = 9.80665d0) ! (m/s2) WMO std g at 45 deg lat
1097 parameter(eccentricity = 0.081819d0) ! unitless
1098 parameter(pi2 = 3.14159265358979d0/180.d0)
1099
1100! Derived geophysical constants
1101 parameter(flattening = (semi_major_axis-semi_minor_axis) / semi_major_axis)
1102
1103 parameter(somigliana = (semi_minor_axis/semi_major_axis)*(grav_polar/grav_equator)-1.d0)
1104
1105 parameter(grav_ratio = (earth_omega*earth_omega * &
1106 semi_major_axis*semi_major_axis * semi_minor_axis)/grav_constant)
1107
1108! Sanity Check
1109 if(lat.gt.90 .or. lat.lt.-90) then
1110 print*, 'compute_geopotential_height: Latitude is not between -90 and 90 degrees: ',lat
1111 return
1112 endif
1113
1114 latr = lat * (pi2) ! in radians
1115 sin2 = sin(latr) * sin(latr)
1116 termg = grav_equator * ( (1.d0+somigliana*sin2) / &
1117 sqrt(1.d0-eccentricity*eccentricity*sin2) )
1118 termr = semi_major_axis / (1.d0 + flattening + grav_ratio - 2.d0*flattening*sin2)
1119
1120 compute_geopotential_height = (termg/grav)*((termr*h)/(termr+h))
1121 !print '(A,3f15.5)','compute_geopotential_height:',&
1122 ! H,compute_geopotential_height,H-compute_geopotential_height
1123
1124end function compute_geopotential_height
