75! From Ibrahim - 19 May 2009
76! "Concerning the radar observations: we can use a format very similar to what we
77! have been using for all the other types of observations:
78!
79!LAT LON DEPTH ANGLE OBS_value Z_LOC ERR_obs QC_flag TYPE_obs STARTDATE1 STARTDATE2
80!
81! where here we only add "ANGLE": the angle of the radial observation.
82!
83! Now what we need is to compute the radial velocity R_model from the state vector
84! (which already contains the horizontal velocities U and V) using
85!
86! 'OBS_value' = U*cos(angle) + V*sin(angle)
87!
88! The angles are in degrees (0 from the east) and the observations in m/s.
89! so to compute 'OBS_value' we need to read the observation file to get the ANGLE,
90! and that's basically the only difference with the assimilation of the other
91! types of observations. We also need to introduce the new obs type
92! (HFRADAR_RADIAL_VELOCITY)."
93
94! changed all occurences of xxx_radial_vel to xxx_hf_radial_vel (for high
95! frequency) since otherwise the external names here collide with those
96! external names in the radar module (for weather radars, e.g. nexrad). nsc.

200subroutine initialize_module
201
202! Called once to set values and allocate space
203
204integer :: iunit, io, rc
205
206! Prevent multiple calls from executing this code more than once.
207if (module_initialized) return
208
209module_initialized = .true.
210
211! Log the version of this source file.
212call register_module(source, revision, revdate)
213
214! Read the namelist entry.
215call find_namelist_in_file("input.nml", "obs_def_ocean_nml", iunit)
216read(iunit, nml = obs_def_ocean_nml, iostat = io)
217call check_namelist_read(iunit, io, "obs_def_ocean_nml")
218
219! Record the namelist values used for the run ...
220if (do_nml_file()) write(nmlfileunit, nml=obs_def_ocean_nml)
221if (do_nml_term()) write( * , nml=obs_def_ocean_nml)
222
223! Allocate space for the auxiliary information associated with each obs
224! This code must be placed after reading the namelist, so the user can
225! increase or decrease the number of obs supported and use more or less
226! memory at run time.
227allocate(radial_vel_metadata(max_radial_vel_obs), stat = rc)
228if (rc /= 0) then
229 write(msgstring, *) 'initial allocation failed for radial vel obs data,', &
230 'itemcount = ', max_radial_vel_obs
231 call error_handler(e_err,'initialize_module', msgstring, &
232 source, revision, revdate)
233endif
234
235end subroutine initialize_module

245subroutine read_hf_radial_vel(velkey, ifile, fform)
246
247! Main read subroutine for the radial velocity observation auxiliary data.
248! The beam_angle is expected to have units of degrees.
249
250integer, intent(out) :: velkey
251integer, intent(in) :: ifile
252character(len=*), intent(in), optional :: fform
253
254logical :: is_asciifile
255integer :: instrument_id
256real(r8) :: beam_angle
257integer :: oldkey
258
259! instrument id: Arbitrary number to distinguish different sources of
260! the data. Not used in any of this code, but carried along in case
261! it is useful in postprocessing or diagnostics.
262
263if ( .not. module_initialized ) call initialize_module
264
265is_asciifile = ascii_file_format(fform)
266
267if (is_asciifile) then
268 ! Read the character identifier for verbose formatted output
269 read(ifile, fmt="(a)") header
270 str1 = adjustl(header)
271
272 if(str1(1:7) /= 'HFRADAR') then
273 write(msgstring,*)"Expected header 'HFRADAR' in input file, got ", str1(1:7)
274 call error_handler(e_err,'read_hf_radial_vel', &
275 msgstring, source, revision, revdate)
276 endif
277endif
278
279! read_location is a DART library routine that expects an optional string
280! arg, but the other two read routines are local to this module and we can
281! tell them exactly what format to be reading because we already know it.
282instrument_id = read_instrument_id(ifile, is_asciifile)
283beam_angle = read_beam_angle(ifile, is_asciifile)
284
285! Read in the velkey for this particular observation, however, it will
286! be discarded and a new, unique key will be generated in the set routine.
287if (is_asciifile) then
288 read(ifile, *) oldkey
289else
290 read(ifile) oldkey
291endif
292
293! Generate new unique radial velocity observation key, and set the contents
294! of the private defined type.
295call set_hf_radial_vel(velkey, instrument_id, beam_angle )
296
297end subroutine read_hf_radial_vel
298

302
303subroutine write_hf_radial_vel(velkey, ifile, fform)
304
305! Write radial velocity auxiliary information to the obs_seq file.
306
307integer, intent(in) :: velkey, ifile
308character(len=*), intent(in), optional :: fform
309
310logical :: is_asciifile
311integer :: instrument_id
312real(r8) :: beam_angle
313
314if ( .not. module_initialized ) call initialize_module
315
316! Simple error check on key number before accessing the array
317call velkey_out_of_range(velkey,'write_hf_radial_vel')
318
319is_asciifile = ascii_file_format(fform)
320
321if (is_asciifile) then
322 ! Write the 7 character identifier for verbose formatted output
323 write(ifile, "('HFRADAR')")
324endif
325
326! Extract the values for this key and call the appropriate write routines.
327instrument_id = radial_vel_metadata(velkey)%instrument_id
328beam_angle = radial_vel_metadata(velkey)%beam_angle
329
330! These two write routines are local to this module, and we have already
331! figured out if it is a unformatted/binary file or formatted/ascii, so
332! go ahead and pass that info directly down to the routines.
333call write_instrument_id(ifile, instrument_id, is_asciifile)
334call write_beam_angle(ifile, beam_angle, is_asciifile)
335
336! Write out the velkey used for this run, however this will be discarded
337! when this observation is read in and a new key will be generated.
338! (It may be useful for correlating error messages or identifying particular
339! observations so we are leaving it as part of the aux data.)
340if (is_asciifile) then
341 write(ifile, *) velkey
342else
343 write(ifile) velkey
344endif
345
346end subroutine write_hf_radial_vel

412function read_beam_angle(ifile, is_asciiformat)
413
414! Reads beam_angle from file that was written by write_beam_angle.
415! See read_hf_radial_vel for additional discussion.
416
417integer, intent(in) :: ifile
418logical, intent(in) :: is_asciiformat

420
421real(r8) :: beam_angle
422
423if ( .not. module_initialized ) call initialize_module
424
425beam_angle = missing_r8
426
427if (is_asciiformat) then
428 read(ifile, "(a)" ) header

430

444 write(msgstring,*) "beam_angle value must be between 0 and 360, got: ", &
445 beam_angle
446 call error_handler(e_err, 'read_beam_angle', msgstring, &
447 source, revision, revdate)
448endif
449
450! set function return value
451read_beam_angle = beam_angle

608subroutine get_expected_hf_radial_vel(state_handle, ens_size, location, velkey, &
609 radial_vel, istatus)
610
611! This is the main forward operator routine for radar Doppler velocity.
612
613type(ensemble_type), intent(in) :: state_handle
614integer, intent(in) :: ens_size
615type(location_type), intent(in) :: location
616integer, intent(in) :: velkey
617real(r8), intent(out) :: radial_vel(ens_size)
618integer, intent(out) :: istatus(ens_size)
619
620
621! Given a location and the state vector from one of the ensemble members,
622! compute the model-predicted radial velocity that would be observed
623! at that location. The value is returned in 'radial_vel'.
624! 'istatus' is the return code. 0 is success; any positive value signals an
625! error (different values can be used to indicate different error types).
626! Negative istatus values are reserved for internal use only by DART.
627!
628! The along-beam component of the 3-d ocean surface velocity is computed
629! from the u, v fields plus the beam_angle:
630! return_value = U*cos(angle) + V*sin(angle)
