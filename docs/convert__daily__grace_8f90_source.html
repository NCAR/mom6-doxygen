10!
11! convert_daily_grace - reads daily GRACE TOTAL WATER STORAGE
12! observations and writes a DART obs_seq file.
13!
14! One day in, one day out
15!
16! The input data to this routine have been preprocessed to convert
17! the monthly GRACE total water storage anomalies to DAILY total water storage.
18! Part of the preprocessing is to convert the monthly anomalies to daily
19! anomalies, and then it is required to add in the long-term mean total water
20! storage from whatever model you are using.

122! call the initialization code, and initialize two empty observation types

164
165 ! the first one needs to contain the string 'observation' and the

178 ! A little helpful logging
179 if (verbose) then
180 write(*,*)'converting "',trim(filename_seq_list(ifile)),'"'

182 write(string2,*)'"'//trim(output_name)//'" time is'
183 call print_date(time_obs, str=string1)
184 call print_time(time_obs, str=string2)
185 endif
186
187 countloop: do icount=1,counts
188
189 if ( tws(icount) < 0.0_r8 ) cycle countloop
190
191 rlat = lat(icount)
192 rlon = lon(icount)
193
194 ! ensure the lat/lon values are in range
195 if ( rlat > 90.0_r8 .or. rlat < -90.0_r8 ) cycle countloop
196 if ( rlon > 360.0_r8 .or. rlon < 0.0_r8 ) cycle countloop
197
198 ! make an obs derived type, and then add it to the sequence
199
200 call create_3d_obs(rlat, rlon, 0.0_r8, vertisundef, tws(icount), &

203 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
204
205 enddo countloop
206
207
208 ! if we added any obs to the sequence, write it out to a file now.
209 if ( get_num_obs(obs_seq) > 0 ) then
210 if (verbose) then
211 write(string1,*)'"'//trim(output_name)//'" has ', &
212 get_num_obs(obs_seq), ' observations.'
213 call error_handler(e_msg, routine, string1)
214 endif
215 call write_obs_seq(obs_seq, output_name)
216 endif
217
218 deallocate(tws)
219 deallocate(err)
220 deallocate(lon)
221 deallocate(lat)
222
223 call destroy_obs_sequence(obs_seq)
224
225enddo fileloop
226
227! end of main program
228call finalize_utilities()

233function check_input_files(input_list, output_list)
234! Read a list of files to process
235character(len=*), intent(in) :: input_list
236character(len=*), dimension(:), intent(out) :: output_list
237integer :: check_input_files
238
239character(len=256) :: input_line
240character(len=256) :: ladjusted
241integer, parameter :: maxlines = 500 ! Long; change from 1000
242integer :: iline
243
244iunit = open_file(trim(input_list), 'formatted', 'read')
245
246check_input_files = 0
247filenameloop: do iline = 1,maxlines ! a lot of lines
248
249 ! read in entire text line into a buffer
250 read(iunit, "(A)", iostat=iocode) input_line
251 if (iocode > 0) then
252 write(string1,*) 'While reading ', trim(input_list)
253 write(string2,*) 'got read code (iostat) = ', iocode,' around line ',iline
254 call error_handler(e_err, 'Check_Input_Files', string1, &
255 source, revision, revdate, text2=string2)
256 elseif (iocode < 0) then
257 ! Normal end of file
258 exit filenameloop
259 else
260 check_input_files = check_input_files + 1
261
262 ladjusted = adjustl(input_line)
263 if ( file_exist(trim(ladjusted)) ) then
264 output_list(check_input_files) = trim(ladjusted)
265 else
266 write(string1,*)'file does not exist.'
267 call error_handler(e_err,'Check_Input_Files',&
268 string1,source,revision,revdate,text2=trim(ladjusted))
269 endif
270 endif
271
272enddo filenameloop
273
274if (check_input_files >= maxlines-1 ) then
275 write(string1,*)'Too many files to process. Increase MAXLINES and try again.'
276 call error_handler(e_err,'Check_Input_Files',string1,source,revision,revdate)
277endif
278
279end function check_input_files

284subroutine get_yyyymmdd(filename, year, month, day)
285character(len=*), intent(in) :: filename
286integer, intent(out) :: year, month, day
287
288! The filenames END with "GRACE_1d_2003-01-01"
289! There is no file extension.
290
291integer :: strlen, index1, io
292
293strlen = len_trim(filename)
294index1 = strlen - 10 + 1 ! YYYY-MM-DD is 10 chars
295
296read(filename(index1:strlen),"(I4,1X,I2,1X,I2)",iostat=io) year, month, day
297if (io /= 0) then
298 write(string1,*)'Unable to read year, month, day from filename.'
299 write(string2,*)'filename was "'//trim(filename)//'"'
300 write(string3,*)'Expected last 10 chars to contain the date, they are "' &
301 & //filename(index1:strlen)//'"'
302 call error_handler(e_err, 'get_yyymmdd', string1, &
303 source, revision, revdate, text2=string2, text3=string3)
304endif
305
306end subroutine get_yyyymmdd
