21!
22! dewpt_error_from_rh_and_temp - computes estimated uncertainty in
23! a dewpoint observation, when the dewpoint is derived from
24! temperature and relative humidity observations
25!
26! created Dec. 2008 David Dowell, NCAR/MMM
27!
28!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
29
30function dewpt_error_from_rh_and_temp(tmpk, rh_in)
31
32! reference: Lin and Hubbard 2004, Journal of Applied Meteorology, 821-825
33
34real(r8) :: dewpt_error_from_rh_and_temp
35real(r8), intent(in) :: tmpk ! temperature (Kelvin)
36real(r8), intent(in) :: rh_in ! relative humidity (0.00-1.00)
37
38real(r8), parameter :: rh_error = 0.05_r8 ! guess for instrument + representativeness error
39real(r8), parameter :: t_error = 1.0_r8 ! guess for instrument + representativeness error
40
41real(r8), parameter :: rh_min = 0.20_r8
42real(r8), parameter :: rh_max = 1.00_r8
43real(r8), parameter :: delta_rh = 0.01_r8 ! perturbation for finite differencing
44real(r8), parameter :: delta_t = 0.1_r8 ! perturbation for finite differencing
45real(r8) :: rh
46real(r8) :: rh1
47real(r8) :: rh2
48real(r8) :: t1
49real(r8) :: t2
50real(r8) :: td_deriv_rh
51real(r8) :: td_deriv_t
52
53
54if ( ( rh_in < 0.00_r8 ) .or. ( rh_in > 1.00_r8 ) ) then
55 print*,'dewpt_error_from_rh_and_temp: bad rh ',rh_in
56 stop
57end if
58
59! The uncertainty in dewpoint derived from temperature and relative humidity grows without bound
60! as the relative humidity approaches 0. A lower bound on the input relative humidity will be
61! applied to keep the calculation bounded and to acknowledge uncertainty in the relative humidity
62! measurement.
63
64rh = max(rh_in, rh_min)
65
66! finite-difference approximations of derivatives
67rh1 = max(rh - delta_rh, rh_min)
68rh2 = min(rh + delta_rh, rh_max)
69t1 = tmpk - delta_t
70t2 = tmpk + delta_t
71td_deriv_rh = ( rh_and_temp_to_dewpoint(rh2, tmpk) - rh_and_temp_to_dewpoint(rh1, tmpk) ) / (rh2-rh1)
72td_deriv_t = ( rh_and_temp_to_dewpoint(rh, t2) - rh_and_temp_to_dewpoint(rh, t1) ) / (t2-t1)
73
74dewpt_error_from_rh_and_temp = sqrt( td_deriv_t*td_deriv_t * t_error*t_error &
75 +td_deriv_rh*td_deriv_rh * rh_error*rh_error)
76
77return
78end function dewpt_error_from_rh_and_temp
79

81! GSR - Added rh error function following above
82function rh_error_from_dewpt_and_temp(tmpk, dewpt)
83
84! reference: Lin and Hubbard 2004, Journal of Applied Meteorology, 821-825
85
86real(r8) :: rh_error_from_dewpt_and_temp
87real(r8), intent(in) :: tmpk ! temperature (Kelvin)
88real(r8), intent(in) :: dewpt ! dewpoint temperature (Kelvin)
89
90real(r8), parameter :: td_error = 1.50_r8 ! guess for instrument + representativeness error
91real(r8), parameter :: t_error = 1.0_r8 ! guess for instrument + representativeness error
92
93real(r8), parameter :: delta_td = 0.1_r8 ! perturbation for finite differencing
94real(r8), parameter :: delta_t = 0.1_r8 ! perturbation for finite differencing
95real(r8) :: td1
96real(r8) :: td2
97real(r8) :: t1
98real(r8) :: t2
99real(r8) :: rh_deriv_td
100real(r8) :: rh_deriv_t
101
102
103if ( ( dewpt > tmpk ) ) then
104 print*,'rh_error_from_dewpt_and_temp: bad dewpt ',dewpt, tmpk
105 stop
106end if
107
108! finite-difference approximations of derivatives
109td1 = dewpt - delta_td
110td2 = min(dewpt + delta_td, tmpk + delta_t)
111t1 = tmpk - delta_t
112t2 = tmpk + delta_t
113rh_deriv_td = ( temp_and_dewpoint_to_rh(tmpk, td2) - temp_and_dewpoint_to_rh(tmpk, td1) ) / (td2-td1)
114rh_deriv_t = ( temp_and_dewpoint_to_rh(t2, dewpt) - temp_and_dewpoint_to_rh(t1, dewpt) ) / (t2-t1)
115
116rh_error_from_dewpt_and_temp = sqrt( rh_deriv_t*rh_deriv_t * t_error*t_error &
117 +rh_deriv_td*rh_deriv_td * td_error*td_error)
118
119return
120end function rh_error_from_dewpt_and_temp
121
