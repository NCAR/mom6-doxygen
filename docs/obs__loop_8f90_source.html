5!> Simple program that opens an obs_seq file and loops over the observations
6!> and copies them to a new output file. This is intended to be a
7!> template for programs that want to alter existing obs in some simple way.
8!>
9!> NOTE:
10!> Observations in a sequence file are processed in ascending time order.
11!> For efficiency at creation time, sequence files support observations in a
12!> different physical order than temporal order. This program will reorder
13!> the observations to be in temporal order, so the output observations
14!> may have a different observation index number than in the input file.

144! blank line, start of actually creating output file

147! Initialize individual observation variables

158enddo
159do j=1, num_qc_in

162enddo

171! Must pass a copy of incoming obs to insert_obs_in_seq.
172!
173!-------------------------------------------------------------------------------

175num_inserted = 0
176
177if ( get_first_obs(seq_in, obs_in) ) then
178
179 is_this_last = .false.

181
182 obsloop : do while ( .not. is_this_last )
183
184 obs_in = next_obs_in
185
186 ! obs_out will be modified when it is inserted in the output sequence
187 ! so we have to make a copy of obs_in before modifiying it.
188 obs_out = obs_in
189
190
191 ! ----------- MODIFY HERE ----------------
192 ! modify obs_out to match what you need
193
194 ! If you need something in the obs_def type, here's how you do that.
195 ! This just queries the location and then makes a call to unpackage
196 ! the location into an array.
197
198 ! call get_obs_def(obs_out, this_obs_def)
199 ! obs_loc = get_obs_def_location(this_obs_def)
200 ! obs_loc_array = get_location(obs_loc)
201
202 ! If you change something in the observation definition
203 ! you have to update the actual observation.
204
205 ! call set_obs_def(obs_out, this_obs_def)
206
207 ! if you do NOT want to insert this observation in the output
208 ! file, set a condition that skips the next block. you *must*
209 ! still call get_next_obs() before looping.
210
211 ! ----------- MODIFY HERE ----------------
212
213
214 if (.true.) then
215
216 ! Since the stride through the observation sequence file is always
217 ! guaranteed to be in temporally-ascending order, we can use the
218 ! 'previous' observation as the starting point to search for the
219 ! correct insertion point. This speeds up the insert code a lot.
220
221 if (num_inserted > 0) then
222 call insert_obs_in_seq(seq_out, obs_out, prev_obs_out)
223 else
224 call insert_obs_in_seq(seq_out, obs_out)
225 endif
226
227 prev_obs_out = obs_out ! update position in seq for next insert
228 num_inserted = num_inserted + 1
229
230 if (print_every > 0) then
231 if (mod(num_inserted,print_every) == 0) then
232 print*, 'inserted number ',num_inserted,' of ',size_seq_out
233 endif
234 endif
235
236 endif
237
238 ! ---- THIS MUST BE EXECUTED BEFORE THE NEXT LOOP ----
239
240 call get_next_obs(seq_in, obs_in, next_obs_in, is_this_last)
241
242 enddo obsloop
243
244else
245 write(msgstring, *)'no first observation in ',trim(filename_in)
246 call error_handler(e_msg,'main', msgstring,source)
247endif
248
249if (.not. print_only) then
250 print*, '--------- Obs seqs '
251 print*, 'Number of obs input sequence : ', size_seq_in
252 print*, 'Number of obs copied to output : ', num_inserted
253 print*, '---------------------------------------------------------'
254endif
255
256write(msgstring, *) 'Starting to process output sequence file ', &
257 trim(filename_out)
258call error_handler(e_msg,'main',msgstring,source)
259
260print*, 'Number of obs in the output seq file :', get_num_key_range(seq_out)
261
262call print_obs_seq_summary(seq_out, filename_out)
263if (.not. print_only) then
264 call write_obs_seq(seq_out, filename_out)
265else
266 write(msgstring,*) 'Output sequence file not created; print_only in namelist is .true.'
267 call error_handler(e_msg,'', msgstring)
268endif

271

275call destroy_obs(next_obs_in )
276call destroy_obs( obs_out)
