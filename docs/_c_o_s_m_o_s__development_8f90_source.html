63character(len=256), parameter :: source = &
64 "$URL$"
65character(len=32 ), parameter :: revision = "$Revision$"

68!-----------------------------------------------------------------------
69! Namelist with default values
70!-----------------------------------------------------------------------
71
72character(len=256) :: site_metadata_file = 'COSMIC_parlist.nc'
73character(len=128) :: text_input_file = 'textdata.input'
74character(len=128) :: obs_out_file = 'obs_seq.out'
75character(len=128) :: sitename = 'missing'
76integer :: year = -1
77real(r8) :: maxgoodqc = 3.0_r8
78logical :: verbose = .false.
79
80namelist /cosmos_development_nml/ site_metadata_file, text_input_file, &
81 obs_out_file, sitename, year, maxgoodqc, verbose

97! The cosmosdata type holds all the information from the text file
98! from the COSMOS instrument. This file format is known best to Rafael.
99! The files using this format are not generally available to the public.
100
101type cosmosdata
102 type(time_type) :: time_obs
103 character(len=20) :: monthstring = 'month'
104 character(len=20) :: daystring = 'day'
105 character(len=20) :: hourstring = 'hour'
106 character(len=20) :: doystring = 'doy'
107 character(len=20) :: neutronstring = 'neutron_fluxAVE'
108 character(len=20) :: neutronstdstring = 'neutron_fluxSTD'
109 character(len=20) :: neutronqcstring = 'neutron_fluxQC'
110 integer :: monthindex

112 integer :: hourindex
113 integer :: doyindex
114 integer :: neutronindex

117 integer :: month
118 integer :: day

122 real(r8) :: neutronstd

124end type cosmosdata

126type(cosmosdata) :: cosmos ! we only need one of these.

139 real(r8) :: alpha ! Ratio of Fast Neutron Creation Factor
140 real(r8) :: l1 ! High Energy Soil Attenuation Length
141 real(r8) :: l2 ! High Energy Water Attenuation Length

150!-----------------------------------------------------------------------
151! start of executable code
152!-----------------------------------------------------------------------
153
154call initialize_utilities('COSMOS_development')
155
156! Print module information to log file and stdout.
157call register_module(source, revision, revdate)
158
159! Read the namelist entry
160call find_namelist_in_file("input.nml", "COSMOS_development_nml", iunit)
161read(iunit, nml = cosmos_development_nml, iostat = rcio)
162call check_namelist_read(iunit, rcio, "COSMOS_development_nml")
163
164! Record the namelist values used for the run ...
165if (do_nml_file()) write(nmlfileunit, nml=cosmos_development_nml)
166if (do_nml_term()) write( * , nml=cosmos_development_nml)
167
168! time setup
169call set_calendar_type(gregorian)
170prev_time = set_time(0, 0)
171
172! Read the COSMOS metadata/parameters for each site.
173! These will be added to the metadata for neutron intensity observations.
174nsites = read_cosmos_metadata(site_metadata_file)
175siteindx = find_site_index(sitename)
176bd = cosmos_metadata(siteindx)%bd
177lattwat = cosmos_metadata(siteindx)%lattwat
178n = cosmos_metadata(siteindx)%N
179alpha = cosmos_metadata(siteindx)%alpha
180l1 = cosmos_metadata(siteindx)%L1
181l2 = cosmos_metadata(siteindx)%L2
182l3 = cosmos_metadata(siteindx)%L3
183l4 = cosmos_metadata(siteindx)%L4
184
185if (verbose) print *, 'COSMOS site located at lat, lon, elev =', &
186 cosmos_metadata(siteindx)%latitude, &
187 cosmos_metadata(siteindx)%longitude, &
188 cosmos_metadata(siteindx)%elevation
189
190! We need to know the maximum number of observations in the input file.
191! Each line has info for a single observation we want (COSMOS neutron counts).
192! Each observation in this series will have a single
193! observation value, its standard deviation, and a quality control flag.
194! Initialize two empty observations - one to track location
195! in observation sequence - the other is for the new observation.
196
197call find_textfile_dims(text_input_file, max_obs)
198
199iunit = open_file(text_input_file, 'formatted', 'read')
200if (verbose) print *, 'opened input file ' // trim(text_input_file)
201
202num_copies = 1
203num_qc = 1
204first_obs = .true.
205
206call static_init_obs_sequence()
207call init_obs( obs, num_copies, num_qc)
208call init_obs( prev_obs, num_copies, num_qc)
209call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
210
211! the first one needs to contain the string 'observation' and the
212! second needs the string 'QC'.
213call set_copy_meta_data(obs_seq, 1, 'observation')
214call set_qc_meta_data( obs_seq, 1, 'COSMOS QC')
215
216! The first line describes all the fields ... column headers, if you will
217call decode_header(iunit)
218
219obsloop: do iline = 2,max_obs
220
221 ! read in entire text line into a buffer
222 read(iunit,'(A)',iostat=rcio) input_line
223 if (rcio < 0) exit obsloop
224 if (rcio > 0) then
225 write (string1,'(''Cannot read (error '',i3,'') line '',i8,'' in '',A)') &
226 rcio, iline, trim(text_input_file)
227 call error_handler(e_err,'main', string1, source, revision, revdate)
228 endif
229
230 ! parse the line into the cosmos structure (including the observation time)
231 call stringparse(input_line, iline)
232
233 if (iline <= 2) then
234 write(*,*)''
235 write(*,*)'Check of the first observation: (column,string,value)'
236 write(*,*)cosmos%monthindex , cosmos%monthstring , cosmos%month
237 write(*,*)cosmos%dayindex , cosmos%daystring , cosmos%day
238 write(*,*)cosmos%hourindex , cosmos%hourstring , cosmos%hour
239 write(*,*)cosmos%doyindex , cosmos%doystring , cosmos%doy
240 write(*,*)cosmos%neutronindex , cosmos%neutronstring , cosmos%neutron
241 write(*,*)cosmos%neutronstdindex , cosmos%neutronstdstring , cosmos%neutronstd
242 write(*,*)cosmos%neutronQCindex , cosmos%neutronQCstring , cosmos%neutronQC
243 call print_date(cosmos%time_obs, 'observation date is')
244 call print_time(cosmos%time_obs, 'observation time is')
245 end if
246
247 if (verbose) call print_date(cosmos%time_obs, 'obs time is')
248
249 call get_time(cosmos%time_obs, osec, oday)
250
251 ! make an obs derived type, and then add it to the sequence
252 ! If the QC value is good, use the observation.
253 ! Increasingly larger QC values are more questionable quality data.
254 ! Quality Control Flags by Rafael Rosolem (rosolem@email.arizona.edu)
255 ! QC flags define as: 0 = BEST (corrected for water vapor)
256 ! 1 = OK (NOT corrected for water vapor)
257 ! 2 = BAD (not reliable/consistent)
258 ! 3 = MISSING (missing data, i.e. -9999)
259
260 if (cosmos%neutronQC <= maxgoodqc) then ! COSMOS neutron counts
261 oerr = cosmos%neutronstd
262 qc = real(cosmos%neutronQC,r8)
263
264 call set_cosmos_metadata(obsindx, bd, lattwat, n, alpha, l1, l2, l3, l4)
265
266 call create_3d_obs(cosmos_metadata(siteindx)%latitude, &
267 cosmos_metadata(siteindx)%longitude, 0.0_r8, vertissurface, &
268 cosmos%neutron, cosmos_neutron_intensity, oerr, oday, osec, qc, obsindx, obs)
269
270 call add_obs_to_seq(obs_seq, obs, cosmos%time_obs, prev_obs, prev_time, first_obs)
271 endif
272
273end do obsloop
274
275! if we added any obs to the sequence, write it out to a file now.
276if ( get_num_obs(obs_seq) > 0 ) then
277 if (verbose) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
278 call write_obs_seq(obs_seq, obs_out_file)
279endif
280
281! end of main program
282call finalize_utilities()
283
284contains

286!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
287!
288! create_3d_obs - subroutine that is used to create an observation
289! type from observation data.
290!
291! NOTE: assumes the code is using the threed_sphere locations module,
292! that the observation has a single data value and a single
293! qc value.
294!
295! lat - latitude of observation
296! lon - longitude of observation
297! vval - vertical coordinate
298! vkind - kind of vertical coordinate (pressure, level, etc)
299! obsv - observation value
300! okind - observation kind
301! oerr - observation error
302! day - gregorian day
303! sec - gregorian second
304! qc - quality control value
305! key - index to metadata in obs_def_COSMOS_mod arrays
306! obs - observation type
307!
308! extended from the observations/utilities/obs_utilities_mod.f90 v 5601
309! to support the extra metadata -- TJH
310!
311!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
312subroutine create_3d_obs(lat, lon, vval, vkind, obsv, okind, oerr, day, sec, qc, key, obs)
313integer, intent(in) :: okind, vkind, day, sec
314real(r8), intent(in) :: lat, lon, vval, obsv, oerr, qc
315integer, intent(in) :: key
316type(obs_type), intent(inout) :: obs
317
318real(r8) :: obs_val(1), qc_val(1)
319type(obs_def_type) :: obs_def
320
321call set_obs_def_location(obs_def, set_location(lon, lat, vval, vkind))
322call set_obs_def_type_of_obs(obs_def, okind)
323call set_obs_def_time(obs_def, set_time(sec, day))
324call set_obs_def_error_variance(obs_def, oerr * oerr)
325call set_obs_def_key(obs_def, key)
326call set_obs_def(obs, obs_def)
327
328obs_val(1) = obsv
329call set_obs_values(obs, obs_val)
330qc_val(1) = qc
331call set_qc(obs, qc_val)
332
333end subroutine create_3d_obs

364subroutine stringparse(str1,linenum)
365! just declare everything as reals and chunk it
366
367character(len=*), intent(in) :: str1
368integer , intent(in) :: linenum
369
370real(r8), dimension(7) :: values
371integer :: iday, ihour, imin, isec, seconds
372
373values = missing_r8
374
375read(str1,*,iostat=rcio) values
376if (rcio /= 0) then
377 write(string1,*)'Cannot parse line',linenum,'. Begins <',trim(str1(1:40)),'>'
378 call error_handler(e_err,'stringparse',string1, source, revision, revdate)
379endif
380
381! Stuff what we want into the cosmos structure
382!
383cosmos%month = nint(values(cosmos%monthindex ))
384cosmos%day = nint(values(cosmos%dayindex ))
385cosmos%hour = values(cosmos%hourindex )
386cosmos%doy = values(cosmos%doyindex )
387cosmos%neutron = values(cosmos%neutronindex )
388cosmos%neutronstd = values(cosmos%neutronstdindex )
389cosmos%neutronQC = nint(values(cosmos%neutronQCindex ))
390
391! decode the time - there is baggage here to determine minutes and seconds
392! which might be useful in the future but not now.
393
394iday = int(cosmos%doy)
395ihour = int(cosmos%hour)
396seconds = nint((cosmos%hour - real(ihour,r8))*3600)
397imin = seconds / 60
398isec = seconds - imin * 60
399
400cosmos%time_obs = set_date(year, cosmos%month, cosmos%day, ihour, imin, isec)
401
402! The QC values can be 'missing' ... in which case the values are too
403
404if (cosmos%neutronQC < 0) cosmos%neutronQC = maxgoodqc + 1000
405
406end subroutine stringparse

411 function read_cosmos_metadata(site_metadata_file)
412!----------------------------------------------------------------------------
413! nsites = read_cosmos_metadata(site_metadata_file)
414!
415! Read the list of parameters for every site we know about and
416! return the number of sites we know about.
417
418integer :: read_cosmos_metadata
419character(len=*), intent(in) :: site_metadata_file
420
421integer :: strlength
422integer :: ncid, isite, varid
423integer, dimension(NF90_MAX_VAR_DIMS) :: dimids
424character(len=metadatalength) :: sitename
425
426! Check to make sure the required parameter file exists
427
428if ( .not. file_exist(site_metadata_file) ) then
429 write(string1,*) 'COSMIC parameter file [', trim(site_metadata_file),'] does not exist.'
430 call error_handler(e_err,'read_cosmos_metadata',string1,source,revision,revdate)
431endif
432
433call nc_check(nf90_open(site_metadata_file, nf90_nowrite, ncid), &
434 'read_cosmos_metadata', 'open '//trim(site_metadata_file))
435
436call nc_check(nf90_inq_dimid(ncid, 'nsites', dimids(1)), &
437 'read_cosmos_metadata','inq_dimid nsites '//trim(site_metadata_file))
438call nc_check(nf90_inquire_dimension(ncid, dimids(1), len=read_cosmos_metadata), &
439 'read_cosmos_metadata','inquire_dimension nsites '//trim(site_metadata_file))
440
441call nc_check(nf90_inq_dimid(ncid, 'strlength', dimids(2)), &
442 'read_cosmos_metadata','inq_dimid strlength '//trim(site_metadata_file))
443call nc_check(nf90_inquire_dimension(ncid, dimids(2), len=strlength), &
444 'read_cosmos_metadata','inquire_dimension strlength '//trim(site_metadata_file))
445
446call nc_check(nf90_inq_varid(ncid, 'sitenames', varid), &
447 'read_cosmos_metadata','inq_varid sitenames '//trim(site_metadata_file))
448
449allocate(cosmos_metadata(read_cosmos_metadata))
450
451do isite = 1,read_cosmos_metadata
452
453 sitename = ''
454
455 write(string1,*)'get_var sitename site ',isite,' '//trim(site_metadata_file)
456 call nc_check(nf90_get_var(ncid, varid, sitename(1:strlength), &
457 start=(/1, isite/), count=(/strlength,1/)), 'read_cosmos_metadata', string1)
458
459 cosmos_metadata(isite)%sitename = trim(sitename(1:strlength))
460
461 call fill_site(ncid, isite)
462
463enddo
464
465call nc_check(nf90_close(ncid), 'read_cosmos_metadata', 'close '//trim(site_metadata_file))
466
467if (verbose) then
468do isite = 1,read_cosmos_metadata
469 write(*,*)
470 write(*,*)'site name ',cosmos_metadata(isite)%sitename
471 write(*,*)'longitude ',cosmos_metadata(isite)%longitude
472 write(*,*)'latitude ',cosmos_metadata(isite)%latitude
473 write(*,*)'elevation ',cosmos_metadata(isite)%elevation
474 write(*,*)'bd ',cosmos_metadata(isite)%bd
475 write(*,*)'lattwat ',cosmos_metadata(isite)%lattwat
476 write(*,*)'N ',cosmos_metadata(isite)%N
477 write(*,*)'alpha ',cosmos_metadata(isite)%alpha
478 write(*,*)'L1 ',cosmos_metadata(isite)%L1
479 write(*,*)'L2 ',cosmos_metadata(isite)%L2
480 write(*,*)'L3 ',cosmos_metadata(isite)%L3
481 write(*,*)'L4 ',cosmos_metadata(isite)%L4
482 write(*,*)
483enddo
484endif
485
486end function read_cosmos_metadata

491
492integer, intent(in) :: ncid, siteindx
493
494call get_var(ncid, siteindx, 'longitude', cosmos_metadata(siteindx)%longitude)
495call get_var(ncid, siteindx, 'latitude' , cosmos_metadata(siteindx)%latitude )
496call get_var(ncid, siteindx, 'elevation', cosmos_metadata(siteindx)%elevation)
497call get_var(ncid, siteindx, 'bd' , cosmos_metadata(siteindx)%bd )
498call get_var(ncid, siteindx, 'lattwat' , cosmos_metadata(siteindx)%lattwat )
499call get_var(ncid, siteindx, 'N' , cosmos_metadata(siteindx)%N )
500call get_var(ncid, siteindx, 'alpha' , cosmos_metadata(siteindx)%alpha )
501call get_var(ncid, siteindx, 'L1' , cosmos_metadata(siteindx)%L1 )
502call get_var(ncid, siteindx, 'L2' , cosmos_metadata(siteindx)%L2 )
503call get_var(ncid, siteindx, 'L3' , cosmos_metadata(siteindx)%L3 )
504call get_var(ncid, siteindx, 'L4' , cosmos_metadata(siteindx)%L4 )
505
506! Ensure that longitudes are [0, 360)
507if (cosmos_metadata(siteindx)%longitude < 0.0_r8) &
508 cosmos_metadata(siteindx)%longitude = cosmos_metadata(siteindx)%longitude + 360.0_r8
509if (cosmos_metadata(siteindx)%longitude == 360.0_r8) &
510 cosmos_metadata(siteindx)%longitude = 0.0_r8
511
512cosmos_metadata(siteindx)%location = set_location(cosmos_metadata(siteindx)%longitude, &
513 cosmos_metadata(siteindx)%latitude, &
514 cosmos_metadata(siteindx)%elevation, &
515 vertissurface)
516end subroutine fill_site

541 function find_site_index(sitename)
542!-------------------------------------------------------------------------------
543! siteIndx = find_site_index(sitename)
544
545integer :: find_site_index
546character(len=*), intent(in) :: sitename
547
548integer :: i, siteid
549character(len=metadatalength) :: testsite, insite
550
551insite = trim(sitename)
552call to_upper(insite) ! convert to upper case - in place
553
554siteid = 0
555
556sitenames : do i = 1,nsites
557
558 testsite = trim(cosmos_metadata(i)%sitename)
559 call to_upper(testsite)
560
561 if (testsite == insite) then
562 siteid = i
563 exit sitenames
564 endif
565
566enddo sitenames
567
568if (siteid == 0) then
569 string1 = 'Unable to find case-insensitive match for site <'//trim(sitename)//'>'
570 string2 = 'Make sure '//trim(site_metadata_file)//' has metadata for this site. Run:'
571 string3 = 'ncdump -v sitenames '//trim(site_metadata_file)
572 call error_handler(e_err,'find_site_index', string1, source, revision, revdate, &
573 text2=string2,text3=string3)
574endif
575
576if (verbose) then
577 write(*,*)
578 write(*,*)'site <'//trim(cosmos_metadata(siteid)%sitename)//'> is number ',siteid
579 write(*,*)'longitude ',cosmos_metadata(siteid)%longitude
580 write(*,*)'latitude ',cosmos_metadata(siteid)%latitude
581 write(*,*)'elevation ',cosmos_metadata(siteid)%elevation
582 write(*,*)'bd ',cosmos_metadata(siteid)%bd
583 write(*,*)'lattwat ',cosmos_metadata(siteid)%lattwat
