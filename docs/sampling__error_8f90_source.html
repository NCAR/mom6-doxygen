106subroutine comp_correl(ens, n, correl)
107
108integer, intent(in) :: n
109real(digits12), intent(in) :: ens(2, n)
110real(digits12), intent(out) :: correl
111
112real(digits12) :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
113
114sum_x = sum(ens(2, :))
115sum_y = sum(ens(1, :))
116sum_xy = sum(ens(2, :) * ens(1, :))
117sum_x2 = sum(ens(2, :) * ens(2, :))
118
119! Computation of correlation
120sum_y2 = sum(ens(1, :) * ens(1, :))
121
122correl = (n * sum_xy - sum_x * sum_y) / &
123 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
124
125end subroutine comp_correl

131integer, intent(in) :: n
