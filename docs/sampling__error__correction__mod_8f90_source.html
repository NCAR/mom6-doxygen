82subroutine read_sampling_error_correction(requested_ens_size, true_correl_mean, alpha)
83
84integer, intent(in) :: requested_ens_size
85real(r8), intent(out) :: true_correl_mean(:), alpha(:)
86
87integer :: ncid, indx
88
89if (.not. module_initialized) call init_sampling_error_correction()
90
91ncid = open_input_file(input_filename)
92
93indx = lookup_ens_index(ncid, nens, requested_ens_size)
94
95if (indx < 0) then
96 write(msgstring, *) 'file "'//trim(input_filename)//'" does not contain a entry for ensemble size ', &
97 requested_ens_size
98 write(msgstring1, *) 'You can add one to the existing file with the "gen_sampling_err_table" program'
99 call error_handler(e_err, 'read_sampling_error_correction:', 'unsupported ensemble size requested', &
100 source, text2=msgstring, text3=msgstring1)
101endif
102
103if (size(true_correl_mean(:)) /= nentries .or. &
104 size(alpha(:)) /= nentries) then
105 write(msgstring, *) 'one or both arrays "true_correl_mean" and "alpha" are not allocated correctly'
106 write(msgstring1, *) 'they must be size ', nentries, ' but are ', size(true_correl_mean), ' and ', size(alpha)
107 call error_handler(e_err, 'read_sampling_error_correction:', 'error in output array size', &
108 source, text2=msgstring, text3=msgstring1)
109endif
110
111call read_input_file(ncid, indx, true_correl_mean, alpha)
112
113call close_input_file(ncid, input_filename)
114
115end subroutine read_sampling_error_correction

158function lookup_ens_index(ncid, num_ens, requested_ens_size)
159
160integer, intent(in) :: ncid
161integer, intent(in) :: num_ens
162integer, intent(in) :: requested_ens_size
163
164integer :: lookup_ens_index
165integer :: i, indx, id
166integer, allocatable :: index_array(:)
167
168allocate(index_array(num_ens))
169
170call query_sec_data(ncid, 'ens_sizes', id)
171call read_sec_data_int(ncid, 1, 'ens_sizes', id, index_array)
172
173indx = -1
174do i=1, num_ens
175 if (index_array(i) == requested_ens_size) then
176 indx = i
177 exit
178 endif
179enddo
180
181lookup_ens_index = indx
182deallocate(index_array)
183
184end function lookup_ens_index
