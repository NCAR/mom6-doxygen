153!------------------------------------------------------------------------
154
155!> return a random draw from a 2D multivariate gaussian distribution
156!> with the specified 2 means and covariances.
157
158subroutine twod_gaussians(r, mean, cov, rnum)
159
160type(random_seq_type), intent(inout) :: r
161real(r8), intent(in) :: mean(2)
162real(r8), intent(in) :: cov(2, 2)
163real(r8), intent(out) :: rnum(2)
164
165real(r8) :: a11, a21, a22, x1, x2
166
167if ( .not. module_initialized ) call initialize_module
168
169! Use method from Knuth, exercise 13, section 3.4.1 to generate random
170! numbers with this mean and covariance
171
172a11 = sqrt(cov(1, 1))
173a21 = cov(1, 2) / a11
174a22 = sqrt(cov(2, 2) - a21**2)
175
176! Two base independent gaussian deviates
177x1 = ran_gauss(r)
178x2 = ran_gauss(r)
179
180! Use these to generate correlated
181rnum(1) = mean(1) + a11 * x1
182rnum(2) = mean(2) + a21 * x1 + a22 * x2
183
184end subroutine twod_gaussians

188!> return a random draw from a gamma distribution
189!> with the specified shape and scale parameter, often
190!> denoted with the symbols kappa and theta.
191!>
192!> note that there are multiple common formulations of
193!> the second parameter to this function. if you have
194!> 'rate' instead of 'scale' specify: 1.0 / rate
195!> for the scale parameter. shape and rate are often
196!> denoted with the symbols alpha and beta.
197!>
198!> The distribution mean should be: shape * rate,
199!> and the variance: shape * (rate^2).

305!> A random congruential random number generator of the Mersenne Twister type
306!> See GNU Scientific Library code.
307
308subroutine init_ran(s, seed)
309 integer, intent(in) :: seed
310 type(random_seq_type), intent(out) :: s
311
312 integer :: i, sd
313
314! Initialize the generator for use with
315! repeatable sequences
316
317sd = seed
318if (sd == 0) sd = 4357 ! do not allow seed to be 0, use default
319
320s%mt(0) = iand(int(sd,i8), full32_mask)
321
322! See Knuth's "Art of Computer Programming" Vol. 2, 3rd Ed. p.106
323! for multiplier.
324
325do i=1, n-1
326 s%mt(i) = 1812433253_i8 * ieor(s%mt(i-1), ishft(s%mt(i-1), -30_i8)) + i
327
328 s%mt(i) = iand(s%mt(i), full32_mask)
329end do
330
331s%mti = n
332s%lastg = 0.0_r8
333s%gset = .false.
334
335end subroutine init_ran

339!> A random congruential random number generator from
340!> the GNU Scientific Library (The Mersenne Twister MT19937 varient.)
341
342function ran_unif(s)
343 type(random_seq_type), intent(inout) :: s
344 real(r8) :: ran_unif
345
346integer :: kk
347integer(i8) :: k, y, m1
348
349! original c code:
350! define MAGIC(y) (((y)&0x1) ? 0x9908b0dfUL : 0)
351
352if (s%mti >= n) then
353 ! generate N words at a time
354 do kk = 0, n-m-1
355 y = ior(iand(s%mt(kk), upper_mask), iand(s%mt(kk + 1), lower_mask))
356 if (iand(y,1_i8) == 1_i8) then
357 m1 = magic
358 else
359 m1 = 0_i8
360 endif
361 s%mt(kk) = ieor(s%mt(kk + m), ieor(ishft(y,-1_i8), m1))
362
363! original c code:
364! unsigned long y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
365! mt[kk] = mt[kk + M] ^ (y >> 1) ^ MAGIC(y);
366
367 enddo
368
369 do kk = n-m, n-2
370 y = ior(iand(s%mt(kk), upper_mask), iand(s%mt(kk + 1), lower_mask))
371 if (iand(y,1_i8) == 1_i8) then
372 m1 = magic
373 else
374 m1 = 0_i8
375 endif
376 s%mt(kk) = ieor(s%mt(kk + (m-n)), ieor(ishft(y,-1_i8), m1))
377
378! original c code:
379! unsigned long y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
380! mt[kk] = mt[kk + (M - N)] ^ (y >> 1) ^ MAGIC(y);
381
382 enddo
383
384 y = ior(iand(s%mt(n-1), upper_mask), iand(s%mt(0), lower_mask))
385 if (iand(y,1_i8) == 1_i8) then
386 m1 = magic
387 else
388 m1 = 0_i8
389 endif
390 s%mt(n-1) = ieor(s%mt(m-1), ieor(ishft(y,-1_i8), m1))
391
392! original c code:
393! unsigned long y = (mt[N - 1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
394! mt[N - 1] = mt[M - 1] ^ (y >> 1) ^ MAGIC(y);
395
396 s%mti = 0
397endif
398
399! Tempering
400
401k = s%mt(s%mti)
402
403k = ieor(k, ishft(k, -11_i8))
404k = ieor(k, iand(ishft(k, 7_i8), c1))
405k = ieor(k, iand(ishft(k, 15_i8), c2))
406k = ieor(k, ishft(k, -18_i8))
407
408! original c code:
409! k ^= (k >> 11);
410! k ^= (k << 7) & 0x9d2c5680UL;
411! k ^= (k << 15) & 0xefc60000UL;
412! k ^= (k >> 18);
413
414s%mti = s%mti + 1
415
416! at this point we have an integer value for k
417! this routine returns 0.0 <= real < 1.0, so do
418! the divide here. return range: [0,1).
419
420ran_unif = real(real(k, digits12) / 4294967296.0_digits12, r8)
421
422end function ran_unif
423
424!------------------------------------------------------------------------
425
426!> Polar (Box-Mueller) method; See Knuth v2, 3rd ed, p122
427!> Returns a N(0, 1) random number draw from a gaussian distribution
428
429function ran_gauss(s)
430
431type(random_seq_type), intent(inout) :: s
432real(r8) :: ran_gauss
433
434real(digits12) :: x, y, r2, t
435integer :: lc
436
437! each pass through this code generates 2 values. if we have one
438! saved from a previous call, return it without doing any new work.
439! otherwise, generate 2 new values; return one and save the other.
440
441if (s%gset) then
442 ran_gauss = s%lastg
443 s%gset = .false.
444else
445
446 lc = 0
44710 continue
448 lc = lc + 1
449
450 ! choose x,y in uniform square (-1,-1) to (+1,+1)
451 x = -1.0_digits12 + 2.0_digits12 * ran_unif(s)
452 y = -1.0_digits12 + 2.0_digits12 * ran_unif(s)
453
454 ! repeat if it is outside the unit circle or at origin
455 r2 = x*x + y*y
456 if (r2 >= 1.0_digits12 .or. r2 == 0.0_digits12) then
457 if (lc > 100) then
458 write(errstring, *) 'x, y = ', x, y
459 call error_handler(e_err, 'ran_gauss', &
460 'if both x and y are -1, random number generator probably not initialized', &
461 source, text2 = errstring);
462 endif
463 goto 10
464 endif
465
466 t = sqrt(-2.0_digits12 * log(r2) / r2)
467 s%lastg = real(x * t, r8)
468 s%gset = .true.
469 ran_gauss = real(y * t, r8)
470endif
471
472end function ran_gauss

474!-------------------------------------------------------------------
475
476!> Gamma(a,b) generator from the GNU Scientific library.
477!>
478!> There are three different parameterizations in common use:
479!> 1. With a shape parameter k and a scale parameter .
480!> 2. With a shape parameter = k and an inverse scale parameter = 1/, called a rate parameter.
481!> 3. With a shape parameter k and a mean parameter = k/.
482!>
483!> warning!
484!> this is the private, internal interface and it uses the first
485!> parameterization, kappa and theta, like the original GSL code.
486!> our public interfaces use the words 'shape', 'scale' and 'rate'
487!> to try to be more clear about which parameterization we're using.
488!>
489!> this internal routine assumes the caller has already verified that
490!> the shape and scale are positive.
491
492recursive function ran_gamma (r, rshape, rscale) result(g)
493
494type(random_seq_type), intent(inout) :: r
495real(r8), intent(in) :: rshape
496real(r8), intent(in) :: rscale
497real(r8) :: g
498
499real(r8) :: x, v, u
500real(r8) :: c, d
501
502
503! we chose the public interface to take 'rate' since that
504! seems to be more common in the kalman world. but the original
505! code is based on a scale parameter, so that's what we use here.
506
507if (rshape < 1.0) then
508 u = random_uniform(r)
509 g = ran_gamma(r, 1.0_r8 + rshape, rscale) * u ** (1.0_r8 / rshape)
510 return
511endif
512
513d = rshape - 1.0_r8 / 3.0_r8
514c = (1.0_r8 / 3.0_r8) / sqrt(d)
515v = -1.0_r8
516
517outer: do while (.true.)
518 v = -1.0_r8
519 do while (v <= 0.0_r8)
520 x = random_gaussian(r, 0.0_r8, 1.0_r8)
521 v = 1.0_r8 + c * x
522 enddo
523
524 v = v * v * v;
525 u = random_uniform(r);
526
527 if (u < 1 - 0.0331_r8 * x * x * x * x) exit outer
528
529 if (log(u) < 0.5_r8 * x * x + d * (1 - v + log(v))) exit outer
530
531enddo outer
532
533g = rscale * d * v
534
535end function ran_gamma
