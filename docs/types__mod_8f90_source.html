40!----------------------------------------------------------------------------
41! Attributes for variable kinds -- declaring sizes explicitly means we know
42! exactly what precision we are using and are not relying on compiler flags
43! to set the defaults for real and int.
44!
45! the call is: selected_int_kind(p, r)
46! where p = decimal_precision, and r = decimal_exponent_range
47!
48! below, "external32_size" = number of bytes
49!
50!
51! from the MPI documentation, they say:
52! (this block of text is originally from the MPI documentation but it seems to apply
53! to the compiler intrinsic functions SELECTED_REAL_KIND and SELECTED_INT_KIND)
54!
55! for reals, SELECTED_REAL_KIND(p, r) values result in:
56!
57! if (p > 33) or (r > 4931) then external32 representation
58! is undefined
59! else if (p > 15) or (r > 307) then external32_size = 16
60! else if (p > 6) or (r > 37) then external32_size = 8
61! else external32_size = 4
62!
63!
64! for integers, SELECTED_INT_KIND(r) results in:
65!
66! if (r > 38) then external32 representation is undefined
67! else if (r > 18) then external32_size = 16
68! else if (r > 9) then external32_size = 8
69! else if (r > 4) then external32_size = 4
70! else if (r > 2) then external32_size = 2
71! else external32_size = 1
72!
73!----------------------------------------------------------------------------
74
75! integer precision:
76integer, parameter :: i2 = selected_int_kind(3)
77integer, parameter :: i4 = selected_int_kind(8)
78integer, parameter :: i8 = selected_int_kind(13)
79
80! real precision:
81! TO RUN WITH REDUCED PRECISION REALS (and use correspondingly less memory)
82! comment OUT the r8 definition below and use the second one:
83integer, parameter :: r4 = selected_real_kind(6,30)
84integer, parameter :: r8 = selected_real_kind(12) ! 8 byte reals
85!integer, parameter :: r8 = r4 ! alias r8 to r4
86
87! complex precision:
88integer, parameter :: c4 = selected_real_kind(6,30)
89integer, parameter :: c8 = selected_real_kind(12)
