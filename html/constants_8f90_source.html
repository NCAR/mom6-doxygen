284 subroutine init_constants(regional)
285!$$$ subprogram documentation block
286! . . . .
287! subprogram: init_constants set regional or global constants
288! prgmmr: treadon org: np23 date: 2004-03-02
289!
290! abstract: This routine sets constants specific to regional or global
291! applications of the gsi
292!
293! program history log:
294! 2004-03-02 treadon
295! 2004-06-16 treadon, documentation
296! 2004-10-28 treadon - use intrinsic TINY function to set value
297! for smallest machine representable positive
298! number
299! 2004-12-03 treadon - move derived constants to init_constants_derived
300! 2005-03-03 treadon - add implicit none
301!
302! input argument list:
303! regional - if .true., set regional gsi constants;
304! otherwise (.false.), use global constants
305!
306! output argument list:
307!
308! attributes:
309! language: f90
310! machine: ibm rs/6000 sp
311!
312!$$$
313 implicit none
314
315 logical,intent(in ) :: regional
316
317 real(r_kind) reradius,g,r_d,r_v,cliq_wrf
318
319! Define regional constants here
320 if (regional) then
321
322! Name given to WRF constants
323 reradius = one/6370.e03_r_kind
324 g = 9.81_r_kind
325 r_d = 287.04_r_kind
326 r_v = 461.6_r_kind
327 cliq_wrf = 4190.0_r_kind
328 cp_mass = 1004.67_r_kind
329
330! Transfer WRF constants into unified GSI constants
331 rearth = one/reradius
332 grav = g
333 rd = r_d
334 rv = r_v
335 cv = cp-r_d
336 cliq = cliq_wrf
337 rd_over_cp_mass = rd / cp_mass
338
339! Define global constants here
340 else
341 rearth = 6.3712e+6_r_kind
342 grav = 9.80665e+0_r_kind
343 rd = 2.8705e+2_r_kind
344 rv = 4.6150e+2_r_kind
345 cv = 7.1760e+2_r_kind
346 cliq = 4.1855e+3_r_kind
347 cp_mass= zero
348 rd_over_cp_mass = zero
349 endif
350
351
352! Now define derived constants which depend on constants
353! which differ between global and regional applications.
354
355! Constants related to ozone assimilation
356 ozcon = grav*21.4e-9_r_kind
357 rozcon= one/ozcon
358
359! Constant used in vertical integral for precipitable water
360 tpwcon = 100.0_r_kind/grav
361
362! Derived atmospheric constants
363 fv = rv/rd-one ! used in virtual temperature equation
364 dldt = cvap-cliq
365 xa = -(dldt/rv)
366 xai = -(dldti/rv)
367 xb = xa+hvap/(rv*ttp)
368 xbi = xai+hsub/(rv*ttp)
369 eps = rd/rv
370 epsm1 = rd/rv-one
371 omeps = one-eps
372 factor1 = (cvap-cliq)/rv
373 factor2 = hvap/rv-factor1*t0c
374 cpr = cp*rd
375 el2orc = hvap*hvap/(rv*cp)
376 rd_over_g = rd/grav
377 rd_over_cp = rd/cp
378 g_over_rd = grav/rd
379
380 return
381 end subroutine init_constants

383 subroutine gps_constants(use_compress)
384!$$$ subprogram documentation block
385! . . . .
386! subprogram: gps_constants set Bevis or Rueger refractive index coeff
387! prgmmr: cucurull org: np23 date: 2010-08-25
388!
389! abstract: This routine sets constants for the refractivity equation. GSI uses Bevis
390! coefficients when the compressibility factors option is turned off
391! and uses Rueger coefficients otherwise.
392!
393! program history log:
394! 2010-08-25 cucurull
395! 2010-08-25 cucurull, documentation
396!
397! input argument list:
398! compress - if .true., set Rueger coefficients;
399! otherwise (.false.), use Bevis coefficients
400!
401! output argument list:
402!
403! attributes:
404! language: f90
405! machine: ibm rs/6000 sp
406!
407!$$$
408 implicit none
409
410 logical,intent(in ) :: use_compress
411
412! Define refractive index coefficients here
413 if (use_compress) then
414
415 ! Constants for gpsro data (Rueger 2002)
416 n_a = 77.6890_r_kind ! K/mb
417 n_b = 3.75463e+5_r_kind ! K^2/mb
418 n_c = 71.2952_r_kind ! K/mb
419 else
420 ! Constants for gpsro data (Bevis et al 1994)
421 n_a = 77.60_r_kind ! K/mb
422 n_b = 3.739e+5_r_kind ! K^2/mb
423 n_c = 70.4_r_kind ! K/mb
424 endif
425
426 return
427 end subroutine gps_constants
