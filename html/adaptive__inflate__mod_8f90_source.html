39! Manages both observation space and state space inflation
40! Handles initial values and restarts, diagnostic output, and computations
41! Algorithm options at present include a single fixed observation space,
42! a single fixed state space adaptive inflation,
43! and a spatially-varying state space inflation that carries
44! a mean and variance for the state space inflation at each point.
45
46! There are many length 2 variables declared to hold items. Some of them
47! refer to the concept of prior and posterior. Those that do can be referenced
48! with the PRIOR_INF and POSTERIOR_INF - which MUST have values 1 and 2, do not change.
49
50integer, parameter :: prior_inf = 1
51integer, parameter :: posterior_inf = 2

218!> Initializes an adaptive_inflate_type
219
220subroutine adaptive_inflate_init(inflate_handle, inf_flavor, mean_from_restart, &
221 sd_from_restart, output_inflation, deterministic, &
222 inf_initial, sd_initial, inf_lower_bound, inf_upper_bound, &
223 sd_lower_bound, sd_max_change, missing_ok, label)
224
225type(adaptive_inflate_type), intent(inout) :: inflate_handle
226integer, intent(in) :: inf_flavor
227logical, intent(in) :: mean_from_restart
228logical, intent(in) :: sd_from_restart
229logical, intent(in) :: output_inflation
230logical, intent(in) :: deterministic
231real(r8), intent(in) :: inf_initial, sd_initial
232real(r8), intent(in) :: inf_lower_bound, inf_upper_bound
233real(r8), intent(in) :: sd_lower_bound
234real(r8), intent(in) :: sd_max_change
235logical, intent(in) :: missing_ok
236character(len = *), intent(in) :: label
237
238! random value
239integer, save :: salt = 139
240
241! Record the module version if this is first initialize call
242if(.not. initialized) then
243 initialized = .true.
244endif
245
246! If non-deterministic inflation is being done, need to initialize random sequence.
247! use the task id number (plus 1 since they start at 0) to set the initial seed.
248! NOTE: non-deterministic inflation does NOT reproduce as process count is varied!
249!> this used to set the same seed for prior & posterior - add a constant value
250!> in case it's called a second time.
251if(.not. deterministic) then
252 call init_random_seq(inflate_handle%ran_seq, my_task_id()+1 + salt)
253 salt = salt + 1000
254endif
255
256! Load up the structure first to keep track of all details of this inflation type
257inflate_handle%inflation_flavor = inf_flavor
258inflate_handle%inflation_sub_flavor = inf_flavor ! see code below
259inflate_handle%output_restart = output_inflation
260inflate_handle%deterministic = deterministic
261inflate_handle%inflate = inf_initial
262inflate_handle%sd = sd_initial
263inflate_handle%inf_lower_bound = inf_lower_bound
264inflate_handle%inf_upper_bound = inf_upper_bound
265inflate_handle%sd_lower_bound = sd_lower_bound
266inflate_handle%sd_max_change = sd_max_change
267inflate_handle%allow_missing_in_clm = missing_ok
268inflate_handle%mean_from_restart = mean_from_restart
269inflate_handle%sd_from_restart = sd_from_restart
270
271!Overwriting the initial value of inflation with 1.0
272!as in other inflation flavors (usually start from 1).
273!This is required for RTPS because inf_initial(2)
274!is not really the inflation factor but rather the weighting
275!parameter, say alpha:
276!RTPS: lambda = alpha * (sd_b - sd_a) / sd_a + 1
277!where; sd_b (sd_a): prior (posteriro) spread
278if(inf_flavor == relaxation_to_prior_spread) inflate_handle%inflate = 1.0_r8
279
280! Prior and posterior are intialized to false
281if (trim(label)=='Prior') inflate_handle%prior = .true.
282if (trim(label)=='Posterior') inflate_handle%posterior = .true.
283
284! ENHANCED_SS_INFLATION is a subset of VARYING_SS_INFLATION. modify the main flavor here.
285if (inf_flavor == enhanced_ss_inflation) then
286 inflate_handle%inflation_flavor = varying_ss_inflation
287endif
288
289! Cannot support non-determistic inflation and an inf_lower_bound < 1
290if(.not. deterministic .and. inf_lower_bound < 1.0_r8) then
291 write(string1, *) 'Cannot have non-deterministic inflation and inf_lower_bound < 1'
292 call error_handler(e_err, 'adaptive_inflate_init', string1, source)
293endif
294
295! give these distinctive values; if inflation is being used
296! (e.g. inf_flavor > 0) then they should be set in all cases.
297inflate_handle%minmax_mean(:) = missing_r8
298inflate_handle%minmax_sd(:) = missing_r8
299
300! State space inflation is read in the IO routine read_state.
301
302! Read type 1 (observation space inflation)
303if(inf_flavor == obs_inflation) then
304
305 write(string1, *) 'No longer supporting observation space inflation ', &
306 '(i.e. inf_flavor = 1).'
307 write(string2, *) 'Please contact dart@ucar.edu if you would like to use ', &
308 'observation space inflation'
309 call error_handler(e_err, 'adaptive_inflate_init', string1, source, text2=string2)
310
311endif
312
313end subroutine adaptive_inflate_init

407!> Make sure the combination of inflation options are legal
408
409subroutine validate_inflate_options(inf_flavor, inf_damping, inf_initial_from_restart, &
410 inf_sd_initial_from_restart, inf_deterministic, inf_sd_max_change, &
411 do_prior_inflate, do_posterior_inflate, output_inflation, &
412 compute_posterior)
413
414integer, intent(in) :: inf_flavor(2)
415real(r8), intent(inout) :: inf_damping(2)
416logical, intent(inout) :: inf_initial_from_restart(2)
417logical, intent(inout) :: inf_sd_initial_from_restart(2)
418logical, intent(inout) :: inf_deterministic(2)
419real(r8), intent(in) :: inf_sd_max_change(2)
420logical, intent(out) :: do_prior_inflate
421logical, intent(out) :: do_posterior_inflate
422logical, intent(out) :: output_inflation
423logical, intent(in) :: compute_posterior
424
425integer :: i
426character(len=32) :: string(2)
427
428do_prior_inflate = .false.
429do_posterior_inflate = .false.
430output_inflation = .false.
431
432! for error messages
433string(prior_inf) = 'Prior'
434string(posterior_inf) = 'Posterior'
435
436do i = prior_inf, posterior_inf
437 if(inf_flavor(i) < no_inflation .or. inf_flavor(i) > enhanced_ss_inflation) then
438 write(string1, *) 'inf_flavor=', inf_flavor(i), ' Must be 0, 1, 2, 3, 4, or 5 '
439 call error_handler(e_err,'validate_inflate_options', string1, source, &
440 text2='Inflation type for '//string(i))
441 endif
442
443 if(inf_damping(i) < 0.0_r8 .or. inf_damping(i) > 1.0_r8) then
444 write(string1, *) 'inf_damping=', inf_damping(i), ' Must be 0.0 <= d <= 1.0'
445 call error_handler(e_err,'validate_inflate_options', string1, source, &
446 text2='Inflation damping for '//string(i))
447 endif
448end do
449
450! Check to see if state space inflation is turned on
451if (inf_flavor(prior_inf) /= no_inflation .and. &
452 inf_flavor(prior_inf) /= obs_inflation) do_prior_inflate = .true.
453if (inf_flavor(posterior_inf) /= no_inflation .and. &
454 inf_flavor(posterior_inf) /= obs_inflation) do_posterior_inflate = .true.
455
456if (do_prior_inflate .or. do_posterior_inflate) output_inflation = .true.
457
458! Observation space inflation not currently supported
459if(inf_flavor(prior_inf) == obs_inflation .or. &
460 inf_flavor(posterior_inf) == obs_inflation) &
461 call error_handler(e_err, 'validate_inflate_options', &
462 'observation space inflation (type 1) not currently supported', source, &
463 text2='contact DART developers if you are interested in using it.')
464
465! Relaxation-to-prior-spread (RTPS) is only an option for posterior inflation
466if(inf_flavor(prior_inf) == relaxation_to_prior_spread) &
467 call error_handler(e_err, 'validate_inflate_options', &
468 'RTPS inflation (type 4) only supported for Posterior inflation', source)
469
470! Cannot select posterior options if not computing posterior
471if(.not. compute_posterior .and. inf_flavor(posterior_inf) /= no_inflation) then
472 write(string1, *) 'cannot enable posterior inflation if not computing posterior values'
473 call error_handler(e_err,'validate_inflate_options', string1, source, &
474 text2='"compute_posterior" is false; posterior inflation flavor must be 0')
475endif
476
477! RTPS needs a single parameter from namelist: inf_initial(2).
478! Do not read in any files. Also, no damping.
479! Warn the user if they try to set different values in the namelist.
480if (inf_flavor(posterior_inf) == relaxation_to_prior_spread) then
481 if (inf_initial_from_restart(posterior_inf) .or. inf_sd_initial_from_restart(posterior_inf)) &
482 call error_handler(e_msg, 'validate_inflate_options:', &
483 'RTPS inflation (type 4) overrides posterior inflation restart file with value in namelist', &
484 text2='posterior inflation standard deviation value not used in RTPS')
485
486 ! Get parameter from namelist inf_initial(2), not from file
487 ! inf_sd not used in this algorithm
488 inf_initial_from_restart(posterior_inf) = .false.
489 inf_sd_initial_from_restart(posterior_inf) = .false.
490
491 if (.not. inf_deterministic(posterior_inf)) &
492 call error_handler(e_msg, 'validate_inflate_options:', &
493 'RTPS inflation (type 4) overrides posterior inf_deterministic with .true.')
494 inf_deterministic(posterior_inf) = .true. ! this algorithm is deterministic
495
496 if (inf_damping(posterior_inf) /= 1.0_r8) &
497 call error_handler(e_msg, 'validate_inflate_options:', &
498 'RTPS inflation (type 4) disables posterior inf_damping')
499 inf_damping(posterior_inf) = 1.0_r8 ! no damping
500endif
501
502! enhanced inflation checks - this is before we set the subflavor in the structure.
503if (inf_flavor(prior_inf) == enhanced_ss_inflation .or. &
504 inf_flavor(posterior_inf) == enhanced_ss_inflation) then
505
506 ! check inf_sd_max_change() for valid range
507 do i=prior_inf, posterior_inf
508 if (inf_sd_max_change(i) < 1.0_r8 .or. inf_sd_max_change(i) > 2.0_r8) then
509 write(string1, *) 'inf_sd_max_change=', inf_sd_max_change(i), &
510 ' Must be 1.0 <= X <= 2.0'
511 call error_handler(e_err,'validate_inflate_options', string1, source, &
512 text2='Inflation stddev max change for '//string(i))
513 endif
514 enddo
515endif
516
517end subroutine validate_inflate_options

520!-------------------------------------------------------------------------------
521!> Inflates subset of ensemble members given mean and inflate
522!> Selects between deterministic and stochastic inflation
523
524subroutine inflate_ens(inflate_handle, ens, mean, inflate, var_in, fsprd, asprd)
525
526type(adaptive_inflate_type), intent(inout) :: inflate_handle
527real(r8), intent(inout) :: ens(:)
528real(r8), intent(in) :: mean
529real(r8), intent(inout) :: inflate
530real(r8), optional, intent(in) :: var_in
531real(r8), optional, intent(in) :: fsprd, asprd
532
533integer :: i, ens_size
534real(r8) :: rand_sd, var, sd_inflate
535
536! it's possible to have MISSING_R8s in the state vector now.
537! so we need to be able to avoid changing MISSING_R8 values by inflation here.
538if (inflate_handle%allow_missing_in_clm) then
539 if (any(ens == missing_r8)) return
540endif
541
542if(inflate_handle%deterministic) then
543
544 if ( do_rtps_inflate(inflate_handle)) then
545 if ( .not. present(fsprd) .or. .not. present(asprd)) then
546 write(string1, *) 'missing arguments for RTPS inflation, should not happen'
547 call error_handler(e_err,'inflate_ens',string1,source)
548 endif
549 ! only inflate if spreads are > 0
550 if ( asprd .gt. 0.0_r8 .and. fsprd .gt. 0.0_r8) &
551 inflate = 1.0_r8 + inflate * ((fsprd-asprd) / asprd)
552 ens = ens * inflate + mean * (1.0_r8 - inflate)
553 else
554
555 ! Spread the ensemble out linearly for deterministic
556 ! Following line can lead to inflation of 1.0 changing ens on some compilers
557 !!! ens = (ens - mean) * sqrt(inflate) + mean
558 ! Following gives 1.0 inflation having no impact on known compilers
559 sd_inflate = sqrt(inflate)
560 ens = ens * sd_inflate + mean * (1.0_r8 - sd_inflate)
561
562 endif
563
564else
565 ! Use a stochastic algorithm to spread out.
566 ens_size = size(ens)
567
568 ! If var is not present, go ahead and compute it here.
569 if(.not. present(var_in)) then
570 var = sum((ens - mean)**2) / (ens_size - 1)
571 else
572 var = var_in
573 endif
574
575 ! To increase the variance of the prior ensemble to the appropriate level
576 ! probably want to keep the mean fixed by shifting AND do a sort
577 ! on the final prior/posterior increment pairs to avoid large regression
578 ! error as per stochastic filter algorithms. This might help to avoid
579 ! problems with generating gravity waves in the Bgrid model, for instance.
580
581 ! The following code does not do the sort.
582
583 ! Figure out required sd for random noise being added
584 ! Don't allow covariance deflation in this version
585 if(inflate > 1.0_r8) then
586 rand_sd = sqrt(inflate*var - var)
587 ! Add random sample from this noise into the ensemble
588 do i = 1, ens_size
589 ens(i) = random_gaussian(inflate_handle%ran_seq, ens(i), rand_sd)
590 end do
591 ! Adjust the mean back to the original value
592 ens = ens - (sum(ens) / ens_size - mean)
593 endif
594endif
595
596end subroutine inflate_ens

599!-------------------------------------------------------------------------------
600!> This routine is given information from an inflate type, scalar values for inflate
601!> and inflate_sd, the ensemble prior_mean and prior_var for an observation, the
602!> ensemble (or group) size, the observed value, observational error variance,
603!> and gamma_corr (see below for description). It computes updated values for the
604!> inflate and inflate_sd items using algorithms described in filter.html (including
605!> references to papers).
606!>
607!> The gamma_corr parameter gives the localized prior correlation times the localization
608!> which is computed in the assim_tools routine filter_assim. For single state
609!> space inflation it is 1.0.
610
611subroutine update_inflation(inflate_handle, inflate, inflate_sd, prior_mean, prior_var, &
612 ens_size, obs, obs_var, gamma_corr)
613
614type(adaptive_inflate_type), intent(in) :: inflate_handle
615real(r8), intent(inout) :: inflate, inflate_sd
616real(r8), intent(in) :: prior_mean, prior_var
617integer, intent(in) :: ens_size
618real(r8), intent(in) :: obs, obs_var, gamma_corr
619
620real(r8) :: new_inflate, new_inflate_sd
621integer :: inf_type
622
623! If the inflate_sd not positive, keep everything the same
624if(inflate_sd <= 0.0_r8) return
625
626! A lower bound on the updated inflation sd and an upper bound
627! on the inflation itself are provided in the inflate_handle.
628
629! select which method to update with
630if (do_enhanced_ss_inflate(inflate_handle)) then
631 inf_type = gha2017
632else
633 inf_type = and2009
634endif
635
636! Use bayes theorem to update
637call bayes_cov_inflate(ens_size, inf_type, prior_mean, prior_var, obs, obs_var, inflate, &
638 inflate_sd, gamma_corr, inflate_handle%sd_lower_bound, inflate_handle%sd_max_change, &

644if(inflate > inflate_handle%inf_upper_bound) inflate = inflate_handle%inf_upper_bound

648if(inflate_sd < inflate_handle%sd_lower_bound) inflate_sd = inflate_handle%sd_lower_bound

655subroutine update_single_state_space_inflation(inflate, inflate_mean, inflate_sd, &
656 ss_inflate_base, orig_obs_prior_mean, orig_obs_prior_var, obs, obs_err_var, &
657 ens_size, inflate_only)
658
659type(adaptive_inflate_type), intent(in) :: inflate
660real(r8), intent(inout) :: inflate_mean
661real(r8), intent(inout) :: inflate_sd
662real(r8), intent(in) :: ss_inflate_base

669

671
672! If either inflation or sd is not positive, not really doing inflation
673if(inflate_mean <= 0.0_r8 .or. inflate_sd <= 0.0_r8) return

675! For case with single spatial inflation, use gamma = 1.0_r8
676gamma = 1.0_r8

678! This is one of many places that assumes linear state/obs relation
679! over range of ensemble; Essentially, we are removing the inflation
680! which has already been applied in filter to see what inflation should

917 new_cov_inflate_sd = lambda_sd

921 ! This comes from the assumption that the mode of the IG is the mean/mode

933 shape_old, rate, gamma_corr, new_cov_inflate)

937 abs(density_2) <= tiny(0.0_r8) .OR. &
938 density_1 /= density_1 .OR. density_1 /= density_1 .OR. &
939 density_2 /= density_2 .OR. density_2 /= density_2) then
940 new_cov_inflate_sd = lambda_sd
941 return
942 endif

945 ratio = density_1 / density_2
946 omega = log(new_cov_inflate )/new_cov_inflate + 1.0_r8/new_cov_inflate - &
947 log(new_cov_inflate+lambda_sd)/new_cov_inflate - 1.0_r8/(new_cov_inflate+lambda_sd)

949 shape_new = rate_new / new_cov_inflate - 1.0_r8
950
951 ! Finally, get the sd of the IG posterior
952 if (shape_new <= 2.0_r8) then
953 new_cov_inflate_sd = lambda_sd
954 return
955 endif
956 new_cov_inflate_sd = sqrt(rate_new**2 / ( (shape_new-1.0_r8)**2 * (shape_new-2.0_r8) ))
957
958 ! If the updated variance is more than xx% the prior variance, keep the prior unchanged
959 ! for stability reasons. Also, if the updated variance is NaN (not sure why this
960 ! can happen; never did when developing this code), keep the prior variance unchanged.
961 if ( new_cov_inflate_sd > sd_max_change_in*lambda_sd .OR. &
962 new_cov_inflate_sd /= new_cov_inflate_sd) then
963 new_cov_inflate_sd = lambda_sd
964 return
965 endif
966 endif
967
968else
969 write(string1, *) 'Internal error, should not happen. Illegal value for bayes type.'
970 call error_handler(e_err, 'bayes_cov_inflate', string1, source)
971endif

973end subroutine bayes_cov_inflate

977!> Used to update density by taking approximate gaussian product
978!> original routine.
979
980function compute_new_density(dist_2, sigma_p_2, sigma_o_2, &
981 lambda_mean, lambda_sd, gamma, lambda)
982
983real(r8), intent(in) :: dist_2, sigma_p_2, sigma_o_2
984real(r8), intent(in) :: lambda_mean, lambda_sd, gamma, lambda
985real(r8) :: compute_new_density
986
987real(r8) :: theta_2, theta
988real(r8) :: exponent_prior, exponent_likelihood
989
990
991! Compute probability of this lambda being correct
992exponent_prior = (lambda - lambda_mean)**2 / (-2.0_r8 * lambda_sd**2)
993
994! Compute probability that observation would have been observed given this lambda
995theta_2 = (1.0_r8 + gamma * (sqrt(lambda) - 1.0_r8))**2 * sigma_p_2 + sigma_o_2
996theta = sqrt(theta_2)
997
998exponent_likelihood = dist_2 / ( -2.0_r8 * theta_2)
999
1000! Compute the updated probability density for lambda
1001! Have 1 / sqrt(2 PI) twice, so product is 1 / (2 PI)
1002compute_new_density = exp(exponent_likelihood + exponent_prior) / &
1003 (2.0_r8 * pi * lambda_sd * theta)
1004
1005end function compute_new_density

1011function enh_compute_new_density(dist_2, ens_size, sigma_p_2, sigma_o_2, &
1012 alpha, beta, gamma_corr, lambda)
1013
1014! Used to update density by taking approximate gaussian product
1015real(r8), intent(in) :: dist_2
1016integer , intent(in) :: ens_size
1017real(r8), intent(in) :: sigma_p_2, sigma_o_2, gamma_corr, lambda

1019real(r8) :: enh_compute_new_density

1025exp_prior = - beta / lambda
1026
1027! Compute probability that observation would have been observed given this lambda

1058!> original linear_bayes routine
1059
1060subroutine linear_bayes(dist_2, sigma_p_2, sigma_o_2, lambda_mean, lambda_sd_2, gamma, &
1061 new_cov_inflate)
1062
1063real(r8), intent(in) :: dist_2, sigma_p_2, sigma_o_2, lambda_mean, lambda_sd_2
1064real(r8), intent(in) :: gamma
1065real(r8), intent(inout) :: new_cov_inflate
1066
1067real(r8) :: theta_bar_2, u_bar, like_exp_bar, v_bar, like_bar, like_prime, theta_bar
1068real(r8) :: a, b, c, plus_root, minus_root, dtheta_dlambda
1069
1070! Compute value of theta at current lambda_mean
1071theta_bar_2 = (1.0_r8 + gamma * (sqrt(lambda_mean) - 1.0_r8))**2 * sigma_p_2 + sigma_o_2
1072theta_bar = sqrt(theta_bar_2)
1073! Compute constant coefficient for likelihood at lambda_bar
1074u_bar = 1.0_r8 / (sqrt(2.0_r8 * pi) * theta_bar)
1075! Compute exponent of likelihood at lambda_bar
1076like_exp_bar = dist_2 / (-2.0_r8 * theta_bar_2)
1077! Compute exponential part of likelihood at lambda_bar
1078v_bar = exp(like_exp_bar)
1079! Compute value of likelihood at current lambda_bar value
1080like_bar = u_bar * v_bar
1081
1082! If like_bar goes to 0, can't do anything, so just keep current values
1083if(like_bar <= 0.0_r8) then
1084 new_cov_inflate = lambda_mean
1085 return
1086endif
1087
1088! Next compute derivative of likelihood at this point
1089
1090! First compute d/dlambda of theta evaluated at lambda_mean
1091! Verified correct by finite difference, 1 January, 2006
1092dtheta_dlambda = 0.5_r8 * sigma_p_2 * gamma *(1.0_r8 - gamma + gamma*sqrt(lambda_mean)) / &
1093 (theta_bar * sqrt(lambda_mean))
1094like_prime = (u_bar * v_bar * dtheta_dlambda / theta_bar) * (dist_2 / theta_bar_2 - 1.0_r8)
1095
1096! If like_prime goes to 0, can't do anything, so just keep current values
1097if(like_prime == 0.0_r8) then
1098 new_cov_inflate = lambda_mean
1099 return
1100endif
1101
1102a = 1.0_r8
1103b = like_bar / like_prime - 2.0_r8 * lambda_mean
1104c = lambda_mean**2 -lambda_sd_2 - like_bar * lambda_mean / like_prime
1105
1106! Use nice scaled quadratic solver to avoid precision issues
1107call solve_quadratic(a, b, c, plus_root, minus_root)
1108
1109! Do a check to pick closest root
1110if(abs(minus_root - lambda_mean) < abs(plus_root - lambda_mean)) then
1111 new_cov_inflate = minus_root
1112else
1113 new_cov_inflate = plus_root
1114endif
1115
1116end subroutine linear_bayes

1122subroutine enh_linear_bayes(dist_2, sigma_p_2, sigma_o_2, &
1123 lambda_mean, gamma_corr, ens_size, beta, new_cov_inflate)
1124
1125real(r8), intent(in) :: dist_2, sigma_p_2, sigma_o_2, lambda_mean
1126real(r8), intent(in) :: gamma_corr, beta
1127real(r8), intent(inout) :: new_cov_inflate
1128
1129integer :: ens_size
1130real(r8) :: theta_bar_2, like_bar, like_prime, theta_bar
1131real(r8) :: a, b, c, plus_root, minus_root, deriv_theta
1132real(r8) :: fac1, fac2, like_ratio
1133
1134! Scaling factors
1135fac1 = (1.0_r8 + gamma_corr * (sqrt(lambda_mean) - 1.0_r8))**2
1136fac2 = -1.0_r8 / ens_size
1137
1138! Compute value of theta at current lambda_mean
1139if ( fac1 < abs(fac2) ) fac2 = 0.0_r8
1140theta_bar_2 = (fac1+fac2) * sigma_p_2 + sigma_o_2
1141theta_bar = sqrt(theta_bar_2)
1142
1143! Compute constant coefficient for likelihood at lambda_bar
1144like_bar = exp(- 0.5_r8 * dist_2 / theta_bar_2) / (sqrt(2.0_r8 * pi) * theta_bar)
1145
1146! If like_bar goes to 0, can't do anything, so just keep current values
1147! Density at current inflation value must be positive
1148if(like_bar <= 0.0_r8) then
1149 new_cov_inflate = lambda_mean
1150 return
1151endif
1152
1153! Next compute derivative of likelihood at this point
1154deriv_theta = 0.5_r8 * sigma_p_2 * gamma_corr * ( 1.0_r8 - gamma_corr + &
1155 gamma_corr * sqrt(lambda_mean) ) / ( theta_bar * sqrt(lambda_mean) )
1156like_prime = like_bar * deriv_theta * (dist_2 / theta_bar_2 - 1.0_r8) / theta_bar
1157
1158! If like_prime goes to 0, can't do anything, so just keep current values
1159! We're dividing by the derivative in the quadratic equation, so this
1160! term better non-zero!
1161if( like_prime == 0.0_r8 .OR. &
1162 abs(like_bar) <= tiny(0.0_r8) .OR. &
1163 abs(like_prime) <= tiny(0.0_r8) ) then
1164 new_cov_inflate = lambda_mean
1165 return
1166endif
1167like_ratio = like_bar / like_prime
1168
1169a = 1.0_r8 - lambda_mean / beta
1170b = like_ratio - 2.0_r8 * lambda_mean
1171c = lambda_mean**2 - like_ratio * lambda_mean
1172
1173! Use nice scaled quadratic solver to avoid precision issues
1174call solve_quadratic(a, b, c, plus_root, minus_root)
1175
1176! Do a check to pick closest root
1177if(abs(minus_root - lambda_mean) < abs(plus_root - lambda_mean)) then
1178 new_cov_inflate = minus_root
1179else
1180 new_cov_inflate = plus_root
1181endif

1183! Do a final check on the sign of the updated factor
1184! Sometimes the factor can be very small (almost zero)
1185! From the selection process above it can be negative
1186! if the positive root is far away from it.
1187! As such, keep the current factor value
1188if(new_cov_inflate <= 0.0_r8 .OR. new_cov_inflate /= new_cov_inflate) &
1189 new_cov_inflate = lambda_mean
