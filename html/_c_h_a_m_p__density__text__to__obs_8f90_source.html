7program champ_density_text_to_obs
8
9!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
10!
11! CHAMP_density_text_to_obs - a program that only needs minor customization to read
12! in a text-based dataset - either white-space separated values or
13! fixed-width column data.
14!
15! created 29 Mar 2010 nancy collins NCAR/IMAGe
16!
17!+ modified 15 Aug 2012 Alexey Morozov (Univ. of Michigan), alexeymor at google mail
18!
19!+ It is designed to read CHAMP ascii files
20!+ For example of input files, see Density_3deg_02_335.ascii in work folder, which is taken from
21! http://sisko.colorado.edu/sutton/data/ver2.2/champ/density/2002/ascii/
22!+ This program reads the name of the text file, obs_seq file, and "debug" from input.nml
23!+ APPENDS new observations to existing obs_seq.out - see line 130ish
24!+ (but not if you change the obs_out_file in input.nml)
25!+ For added convenience, see convert.sh in work folder, which runs this program repeatedly
26!+ to convert+append many CHAMP files
27!+ Implemented the suggestion about times starting from weird points (like 335th day in 2002)
28!+ - see lines 190ish
29!
30!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
31
32use types_mod, only : r8, pi, deg2rad
33
34use utilities_mod, only : initialize_utilities, finalize_utilities, &
35 open_file, close_file, find_namelist_in_file, check_namelist_read

37use time_manager_mod, only : time_type, set_calendar_type, set_date, &
38 operator(>=), increment_time, get_time, set_time, &
39 operator(-), gregorian, operator(+), print_date

41use location_mod, only : vertisheight
42
43use obs_sequence_mod, only : obs_sequence_type, obs_type, read_obs_seq, &
44 static_init_obs_sequence, init_obs, write_obs_seq, &
45 init_obs_sequence, get_num_obs, &
46 set_copy_meta_data, set_qc_meta_data
47
48use obs_utilities_mod, only : create_3d_obs, add_obs_to_seq
49
50use obs_kind_mod, only : sat_rho !this is density
51
52implicit none
53
54! things which can/should be in the text_to_obs_nml
55
56character(len=64) :: text_input_file = 'Density_3deg_02_335_2p2.ascii'
57character(len=64) :: obs_out_file = 'obs_seq.out'
58logical :: debug = .true.
59
60namelist /champ_density_text_to_obs_nml/ &
61 text_input_file, &
62 obs_out_file, &
63 debug
64
65
66character (len=200) :: input_line !162 is the width of CHAMPdens2.2 files, but to be safe do 200
67
68integer :: oday, osec, rcio, iunit
69integer :: year, day, second
70integer :: num_copies, num_qc, max_obs
71
72logical :: file_exist, first_obs
73
74real(r8) :: temp, terr, qc
75real(r8) :: lat, lon, vert
76
77real(r8) :: second_r !CHAMP seconds are reals instead of ints
78
79!variables to be discarded (only needed so that the read line works)
80integer :: ignore_i
81real :: ignore_r
82
83type(obs_sequence_type) :: obs_seq
84type(obs_type) :: obs, prev_obs
85type(time_type) :: comp_day0, time_obs, prev_time
86
87! start of executable code
88
89call initialize_utilities('CHAMP_density_text_to_obs')
90
91! Read the DART namelist for this model
92call find_namelist_in_file('input.nml', 'CHAMP_density_text_to_obs_nml', iunit)
93read(iunit, nml = champ_density_text_to_obs_nml, iostat = rcio)
94call check_namelist_read(iunit, rcio, 'CHAMP_density_text_to_obs_nml')
95
96! time setup
97call set_calendar_type(gregorian)
98
99
100! open input text file
101
102iunit = open_file(text_input_file, 'formatted', 'read')
103if (debug) print *, 'opened input file ' // trim(text_input_file)
104
105
106! each observation in this series will have a single observation value
107! and a quality control flag. the max possible number of obs needs to
108! be specified but it will only write out the actual number created.
109max_obs = 100000
110num_copies = 1
111num_qc = 1
112
113! call the initialization code, and initialize two empty observation types
114call static_init_obs_sequence()
115call init_obs(obs, num_copies, num_qc)
116call init_obs(prev_obs, num_copies, num_qc)
117first_obs = .true.
118
119! create a new, empty obs_seq file. you must give a max limit
120! on number of obs. increase the size if too small.
121call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
122
123! the first one needs to contain the string 'observation' and the
124! second needs the string 'QC'.
125call set_copy_meta_data(obs_seq, 1, 'observation')
126call set_qc_meta_data(obs_seq, 1, 'Data QC')
127
128! if you want to append to existing files (e.g. you have a lot of
129! small text files you want to combine), you can do it this way,
130! or you can use the obs_sequence_tool to merge a list of files
131! once they are in DART obs_seq format.
132
133! existing file found, append to it
134inquire(file=obs_out_file, exist=file_exist)
135if ( file_exist ) then
136 call read_obs_seq(obs_out_file, 0, 0, max_obs, obs_seq)
137endif
138
139! Set the DART data quality control. 0 is good data.
140! increasingly larger QC values are more questionable quality data.
141qc = 0.0_r8
142
143! first two lines are just text (description), so just skip them
144read(iunit, "(A)", iostat=rcio) input_line
145read(iunit, "(A)", iostat=rcio) input_line
146
147obsloop: do ! no end limit - have the loop break when input ends
148
149 ! read in a line from the text file. What you need to create an obs:
150 ! location: lat, lon, and height in pressure or meters
151 ! time: when the observation was taken
152 ! type: from the DART list of obs types
153 ! error: very important - the instrument error plus representativeness error
154 ! (see html file for more info)
155
156 ! read the whole line into a buffer
157 read(iunit, "(A)", iostat=rcio) input_line
158 if (rcio /= 0) then
159 if (debug) print *, 'got bad read code from input file, rcio = ', rcio
160 exit obsloop
161 endif
162
163 ! assume here is a line from sisko.colorado.edu/sutton/data/ver2.2/champ/density/2002/ascii/,
164 !data format is:
165 !+ 1)year(2I), 2)day(3I), 3)second(8.3F), 4)round(lat), 5)lat(d,-90 90), 6)lon(d,-180 180), 7)alt(km),
166 !+ 8)LT, 9)Mlat, 10)Mlon, 11)MLT, 12)Rho(Density!), 13)MSISRho400, 14)MSISRho410, 15)MSISRhoSat
167 !+ 16)Rho uncertainty (I'm guessing std deviation from units: kg/m^3) 17)points averaged over
168 !+ 18)points needing interpolation 19)coeff of drag averaged over bin
169
170 read(input_line, *, iostat=rcio) &
171 year, day, second_r, ignore_i, lat, lon, vert, &
172 ignore_r, ignore_r, ignore_r, ignore_r, temp, ignore_r, ignore_r, ignore_r, &
173 terr, ignore_i, &
174 ignore_i, ignore_r
175
176 vert=vert*1000 !DART needs alt in m, whereas in champ files it's in km
177
178 if (rcio /= 0) then
179 if (debug) print *, 'got bad read code getting rest of temp obs, rcio = ', rcio
180 exit obsloop
181 endif
182
183 if (debug) print *, 'this observation located at lat, lon = ', lat, lon
184
185 ! if lon comes in between -180 and 180, use these lines instead:
186 if ( lat > 90.0_r8 .or. lat < -90.0_r8 ) cycle obsloop
187 if ( lon > 180.0_r8 .or. lon < -180.0_r8 ) cycle obsloop
188 if ( lon < 0.0_r8 ) lon = lon + 360.0_r8 ! changes into 0-360
189
190 ! put date into a dart time format
191
192 year = 2000 + year !because year in file is (2I) - 2 digits
193 second = nint(second_r)
194
195 !! some times are supplied as number of seconds since some reference
196 !! date. This is an example of how to support that.
197 !! put the reference date into DART format
198 comp_day0 = set_date(year, 1, 1, 0, 0, 0)
199 time_obs = comp_day0 + set_time(second, day-1)
200
201 ! extract time of observation into gregorian day, sec.
202 call get_time(time_obs, osec, oday)
203
204 if (debug) call print_date(time_obs, 'this obs time is')
205
206 ! height is in kilometers (yardstick)
207 ! make an obs derived type, and then add it to the sequence
208
209 call create_3d_obs(lat, lon, vert, vertisheight, temp, &
210 sat_rho, terr, oday, osec, qc, obs)
211 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
212
213 if (debug) print *, 'added RHO obs to output seq'
214
215end do obsloop
216
217! if we added any obs to the sequence, write it out to a file now.
218if ( get_num_obs(obs_seq) > 0 ) then
219 !if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
220 print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
221 call write_obs_seq(obs_seq, obs_out_file)
222endif
223
224! end of main program
225call finalize_utilities()
226
227end program champ_density_text_to_obs
