140!------------------------------------------------------------------
141!> Verify that the namelist was filled in correctly, and check
142!> that there are valid entries for the dart_kind.
143!> Returns a table with columns:
144!>
145!> netcdf_variable_name ; dart_kind_string
146
147subroutine verify_state_variables( state_variables, ngood, table, kind_list)
148
149character(len=*), intent(inout) :: state_variables(:)
150integer, intent(out) :: ngood
151character(len=*), intent(out) :: table(:,:)
152integer, intent(out) :: kind_list(:) ! kind number
153
154integer :: nrows, i
155character(len=256) :: varname
156character(len=32) :: kindstr
157
158nrows = size(table,1)
159
160ngood = 0
161
162if ( state_variables(1) == ' ' ) then ! no model_variables namelist provided
163 string1 = 'model_nml:model_variables not specified'
164 call error_handler(e_err,'verify_state_variables',string1,source,revision,revdate)
165endif
166
167myloop : do i = 1, nrows
168
169 varname = trim(state_variables(2*i -1))
170 kindstr = trim(state_variables(2*i ))
171
172 table(i,1) = trim(varname)
173 table(i,2) = trim(kindstr)
174
175 if ( table(i,1) == ' ' .and. table(i,2) == ' ') exit myloop ! Found end of list.
176
177 if ( table(i,1) == ' ' .and. table(i,2) == ' ') then
178 string1 = 'model_nml:model_variables not fully specified'
179 call error_handler(e_err,'verify_state_variables',string1,source,revision,revdate)
180 endif
181
182 ! Make sure DART kind is valid
183
184 kind_list(i) = get_index_for_quantity(kindstr)
185 if( kind_list(i) < 0 ) then
186 write(string1,'(''there is no obs_kind <'',a,''> in obs_kind_mod.f90'')') trim(kindstr)
187 call error_handler(e_err,'verify_state_variables',string1,source,revision,revdate)
188 endif
189
190 ! Record the contents of the DART state vector
191
192 if (verbose) then
193 write(string1,'(A,I2,6A)') 'variable ',i,' is ',trim(varname), ', ', trim(kindstr)
194 call error_handler(e_msg,'verify_state_variables',string1,source,revision,revdate)
195 endif
196
197 ngood = ngood + 1
198enddo myloop
199
200end subroutine verify_state_variables

204function initialize_filenames(filename, IO_STRING) result (file_handle)
