131subroutine get_expected_dew_point(state_handle, ens_size, location, key, td, istatus)
132
133type(ensemble_type), intent(in) :: state_handle
134integer, intent(in) :: ens_size
135type(location_type), intent(in) :: location
136integer, intent(in) :: key
137real(r8), intent(out) :: td(ens_size) ! dewpoint (K)
138integer, intent(out) :: istatus(ens_size)
139
140integer :: ipres
141real(r8) :: qv(ens_size) ! water vapor mixing ratio (kg/kg)
142real(r8) :: e_mb(ens_size) ! water vapor pressure (mb)
143real(r8), PARAMETER :: e_min = 0.001_r8 ! threshold for minimum vapor pressure (mb),
144 ! to avoid problems near zero vapor pressure in Bolton's equation
145real(r8) :: p_pa(ens_size) ! pressure (Pa)
146real(r8) :: p_mb(ens_size) ! pressure (mb)
147real(r8) :: log_term(ens_size) ! Intermediate term in computation of dewpoint temperature
148!> @todo make strings longer
149character(len=129) :: errstring
150logical :: return_now
151integer :: qv_istatus(ens_size)
152integer :: ipres_istatus(ens_size)
153
154if ( .not. module_initialized ) call initialize_module
155
156istatus = 0
157
158if(key == 1) then
159 ipres = qty_pressure
160elseif(key == 2) then
161 ipres = qty_surface_pressure
162else
163 write(errstring,*)'key has to be 1 (upper levels) or 2 (2-meter), got ',key
164 call error_handler(e_err,'get_expected_dew_point', errstring, &
165 source, revision, revdate)
166endif
167
168call interpolate(state_handle, ens_size, location, ipres, p_pa, ipres_istatus)
169call track_status(ens_size, ipres_istatus, td, istatus, return_now)
170if (return_now) return
171
172call interpolate(state_handle, ens_size,location, qty_vapor_mixing_ratio, qv, qv_istatus)
173call track_status(ens_size, qv_istatus, td, istatus, return_now)
174if (return_now) return
175
176
177where (qv < 0.0_r8 .or. qv >= 1.0_r8)
178 istatus = 1
179 td = missing_r8
180end where
181
182
183
184!------------------------------------------------------------------------------
185! Compute water vapor pressure.
186!------------------------------------------------------------------------------
187where (istatus == 0 ) ! To avoid possible FPE with missing_r8
188
189 p_mb = p_pa * 0.01_r8
190
191 e_mb = qv * p_mb / (0.622_r8 + qv)
192 e_mb = max(e_mb, e_min)
193
194 !------------------------------------------------------------------------------
195 ! Use Bolton's approximation to compute dewpoint.
196 ! Bolton, David, 1980: The Computation of Equivalent Potential Temperature.
197 ! Monthly Weather Review. 108 (7): 1046-1053.
198 ! Bolton does not explicitly have this formula, but the pieces are there.
199 ! He uses the three constant values used here.
200 ! Other authors, and the Wikipedia page on dewpoint, provide this formula
201 ! explicitly and suggest 6.1121 instead of 6.112.
202 !------------------------------------------------------------------------------
203
204 ! The following expression can fail numerically for dewpoints very close to 0 C
205 !td = t_kelvin + (243.5_r8 / ((17.67_r8 / log(e_mb/6.112_r8)) - 1.0_r8) )
206
207 ! A numerically robust formula that avoids the failure near dewpoints of 0 C
208 log_term = log(e_mb / 6.112_r8)
209 td = t_kelvin + 243.5_r8 * log_term / (17.67_r8 - log_term)
210
211elsewhere
212 td = missing_r8
213end where
214
215end subroutine get_expected_dew_point
