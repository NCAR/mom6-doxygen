35subroutine test_beta
36
37! This routine provides limited tests of the numerics in this module. It begins
38! by comparing a handful of cases of the pdf and cdf to results from Matlab. It
39! then tests the quality of the inverse cdf for a single shape/scale pair. Failing
40! these tests suggests a serious problem. Passing them does not indicate that
41! there are acceptable results for all possible inputs.
42
43real(r8) :: x, y, p, inv
44real(r8) :: alpha, beta, max_diff
45integer :: i
46
47! Comparative results for a handful of cases from MATLAB21a
48real(r8) :: pdf_diff(7), cdf_diff(7)
49real(r8) :: malpha(7) = [0.5_r8, 5.0_r8, 1.0_r8, 2.0_r8, 2.0_r8, 5.0_r8, 0.5_r8]
50real(r8) :: mbeta(7) = [0.5_r8, 1.0_r8, 3.0_r8, 2.0_r8, 5.0_r8, 2.0_r8, 1.0_r8]
51real(r8) :: mx(7) = [0.1_r8, 0.3_r8, 0.5_r8, 0.6_r8, 0.7_r8, 0.8_r8, 0.9_r8]
52! Generated by matlab betapdf(mx, malpha, mbeta)
53real(r8) :: mpdf(7) = [1.061032953945969_r8, 0.040500000000000_r8, 0.750000000000000_r8, &
54 1.440000000000000_r8, 0.170100000000000_r8, 2.457600000000000_r8, &
55 0.527046276694730_r8]
56! Generated by matlab gamcdf(mx, malpha, mbeta)
57real(r8) :: mcdf(7) = [0.204832764699133_r8, 0.002430000000000_r8, 0.875000000000000_r8, &
58 0.648000000000000_r8, 0.989065000000000_r8, 0.655360000000000_r8, &
59 0.948683298050514_r8]
60
61! Compare to matlab
62write(*, *) 'Absolute value of differences should be less than 1e-15'
63do i = 1, 7
64 pdf_diff(i) = beta_pdf(mx(i), malpha(i), mbeta(i)) - mpdf(i)
65 cdf_diff(i) = beta_cdf(mx(i), malpha(i), mbeta(i), 0.0_r8, 1.0_r8) - mcdf(i)
66 write(*, *) i, pdf_diff(i), cdf_diff(i)
67end do
68
69! Test many x values for cdf and inverse cdf for a single set of alpha and beta
70alpha = 5.0_r8
71beta = 2.0_r8
72
73max_diff = -1.0_r8
74do i = 0, 1000
75 x = i / 1000.0_r8
76 p = beta_pdf(x, alpha, beta)
77 y = beta_cdf(x, alpha, beta, 0.0_r8, 1.0_r8)
78 inv = inv_beta_cdf(y, alpha, beta, 0.0_r8, 1.0_r8)
79 max_diff = max(abs(x - inv), max_diff)
80end do
81
82write(*, *) '----------------------------'
83write(*, *) 'max difference in inversion is ', max_diff
84write(*, *) 'max difference should be less than 1e-14'
85
86end subroutine test_beta

102function inv_beta_cdf(quantile, alpha, beta, lower_bound, upper_bound) result(x)
103
104real(r8) :: x
105real(r8), intent(in) :: quantile
106real(r8), intent(in) :: alpha, beta
107real(r8), intent(in) :: lower_bound, upper_bound
108
109! Given a quantile, finds the value of x for which the scaled beta cdf
110! with alpha and beta has approximately this quantile
111
112type(distribution_params_type) :: p
113
114if (alpha <= 0.0_r8 .or. beta <= 0.0_r8) then
115 errstring = 'Negative input beta parameters'
116 call error_handler(e_err, 'inv_beta_cdf', errstring, source)
117endif
118
119p%params(1) = alpha; p%params(2) = beta
120! Beta must be bounded on both sides
121p%lower_bound = lower_bound; p%upper_bound = upper_bound
122
123x = inv_beta_cdf_params(quantile, p)
124
125! Undo the scaling
126x = x * (upper_bound - lower_bound) + lower_bound
127
128end function inv_beta_cdf

132function beta_pdf(x, alpha, beta)
133
134! Returns the probability density of a beta function with alpha and beta
135! at the value x
136
137! Returns a large negative value if called with illegal values
138
139real(r8) :: beta_pdf
140real(r8), intent(in) :: x, alpha, beta
141
142real(r8) :: gamma_ratio
143
144! Parameters alpha and beta must be positive
145if(alpha <= 0.0_r8 .or. beta <= 0.0_r8) then
146 beta_pdf = failed_value
147elseif(x < 0.0 .or. x > 1.0_r8) then
148 beta_pdf = failed_value
149elseif(alpha == 1.0_r8 .and. x == 0.0_r8) then
150 ! Tricky stuff for x = 0 or 1;
151 beta_pdf = beta
152elseif(beta == 1.0_r8 .and. x == 1.0_r8) then
153 beta_pdf = alpha
154elseif(alpha < 1.0_r8 .and. x == 0.0_r8) then
155 beta_pdf = failed_value
156elseif(beta < 1.0_r8 .and. x == 1.0_r8) then
157 beta_pdf = failed_value
158else
159 ! Use definition via gammas since this is a Fortran intrinsic
160 gamma_ratio = gamma(alpha) * gamma(beta) / gamma(alpha + beta)
161 beta_pdf = x**(alpha - 1.0_r8) * (1.0_r8 - x)**(beta - 1.0_r8) / gamma_ratio
162endif
163
164end function beta_pdf

183function beta_cdf(x, alpha, beta, lower_bound, upper_bound)
184
185! Returns the cumulative distribution of a beta function with alpha and beta
186! at the value x
187
188! Returns a large negative value if called with illegal values
189
190real(r8) :: beta_cdf
191real(r8), intent(in) :: x, alpha, beta
192real(r8), intent(in) :: lower_bound, upper_bound
193
194! Parameters must be positive
195if(alpha <= 0.0_r8 .or. beta <= 0.0_r8) then
196 beta_cdf = failed_value
197elseif(x < 0.0_r8 .or. x > 1.0_r8) then
198 ! x must be in 0 1
199 beta_cdf = failed_value
200elseif(x == 0.0_r8) then
201 beta_cdf = 0.0_r8
202elseif(x == 1.0_r8) then
203 beta_cdf = 1.0_r8
204elseif (x > (alpha + 1.0_r8)/(alpha + beta + 2.0_r8)) then
205 beta_cdf = (1.0_r8 - incomplete_beta(beta, alpha, 1.0_r8 - x))
206else
207 beta_cdf = incomplete_beta(alpha, beta, x)
208endif

214function random_beta(r, alpha, beta)
215
216! Note that this provides same qualitative functionality as a similarly named
217! routine in the random_seq_mod that uses a rejection algorithm. However, once
218! we have an inverse cdf function for a distribution, it is possible to generate
219! random numbers by first getting a draw from a U(0, 1) and then inverting these
220! quantiles to get an actual value
221
222type(random_seq_type), intent(inout) :: r
223real(r8), intent(in) :: alpha
224real(r8), intent(in) :: beta
225real(r8) :: random_beta
226
227real(r8) :: quantile
228if (alpha <= 0.0_r8) then
229 write(errstring, *) 'Alpha parameter must be positive, was ', alpha
230 call error_handler(e_err, 'random_beta', errstring, source)
231endif
232
233if (beta <= 0.0_r8) then
234 write(errstring, *) 'Beta parameter must be positive, was ', beta
235 call error_handler(e_err, 'random_beta', errstring, source)
236endif
237
238! Draw from U(0, 1) to get a quantile
239quantile = random_uniform(r)
240! Invert cdf to get a draw from beta
241random_beta = inv_beta_cdf(quantile, alpha, beta, 0.0_r8, 1.0_r8)
242
243end function random_beta

247function incomplete_beta(a,b,x)
248
249! Computes an approximation of the incomplete beta integral using the continued
250! fraction evaluation routine also found in numerical recipes
251
252real(r8) :: incomplete_beta
253real(r8), intent(in) :: a, b, x
254
255real(r8), parameter :: tiny = 1.0e-30
256real(r8), parameter :: stop = 1.0e-14
257integer, parameter :: max_iter = 100
258
259real(r8) :: front, f, c, d, numerator, cd
260integer :: m, iter
261
262if (x < 0.0_r8 .or. x > 1.0_r8) then
263 errstring = 'Input value for x is not between 0 - 1'
264 call error_handler(e_err, 'incomplete_beta', errstring, source)
265endif
266
267! Set a default failed value
268incomplete_beta = missing_r8
269
270front = exp(log(x)*a + log(1.0_r8-x)*b - log_beta(a, b)) / a
271f = 1.0_r8
272c = 1.0_r8
273d = 0.0_r8
274
275do iter = 0, max_iter
276 m = floor(iter/2.0_r8)
277
278 ! Initial step numerator is 1
279 if (iter == 0) then
280 numerator = 1.0_r8
281 ! Algorithm has an odd and even iteration step
282 else if (mod(iter, 2) == 0) then
283 numerator = (m*(b - m)*x)/((a + 2.0_r8*m - 1.0_r8)*(a + 2.0_r8*m))
284 else
285 numerator = -((a + m)*(a + b + m)*x)/((a + 2.0_r8*m)*(a + 2.0_r8*m + 1.0_r8))
286 end if
287
288 d = 1.0_r8 + (numerator * d)
289 if (abs(d) < tiny) d = tiny
290 d = 1.0_r8 / d
291
292 c = 1.0_r8 + (numerator/c)
293 if (abs(c) < tiny) c = tiny
294
295 cd = c*d
296 f = cd*f
297
298 if (abs(1.0_r8 - cd) < stop) then
299 incomplete_beta = front * (f - 1.0_r8)
300 return
301 end if
302end do
303
304! Error if failed to converge
305errstring = 'Alg. did not converge'
306call error_handler(e_err, 'incomplete_beta', errstring, source)
307
308end function incomplete_beta

356subroutine beta_alpha_beta(x, num, alpha, beta)
357
358! Computes the alpha and beta parameters for a beta distribution from an ensemble
359! Assumes the ensemble members are confined to [0, 1]
360! This may not be the maximum likelihood estimate
361
362integer, intent(in) :: num
363real(r8), intent(in) :: x(num)
364real(r8), intent(out) :: alpha
365real(r8), intent(out) :: beta
366
367real(r8) :: mean, variance
368
369mean = sum(x) / num
370variance = sum((x - mean)**2) / (num- 1)
371! Get alpha and beta
372alpha = mean**2 * (1.0_r8 - mean) / variance - mean
373beta = alpha * (1.0_r8 / mean - 1.0_r8)
374
375end subroutine beta_alpha_beta
