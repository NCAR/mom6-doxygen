12! in a text-based dataset - either white-space separated values or

40implicit none

42character(len=64), parameter :: text_input_file = 'text.txt'
43character(len=64), parameter :: obs_out_file = 'obs_seq.out'
44
45logical, parameter :: debug = .false. ! set to .true. to print info
46
47character (len=129) :: input_line
48
49integer :: oday, osec, rcio, iunit, otype
50integer :: year, month, day, hour, minute, second
51integer :: num_copies, num_qc, max_obs
52
53logical :: first_obs
54
55real(r8) :: temp, terr, qc
56real(r8) :: lat, lon, vert
57
58type(obs_sequence_type) :: obs_seq
59type(obs_type) :: obs, prev_obs
60type(time_type) :: time_obs, prev_time
61
62! start of executable code
63
64call initialize_utilities('text_to_obs')
65
66! time setup
67call set_calendar_type(gregorian)
68
69! each observation in this series will have a single observation value
70! and a quality control flag. the max possible number of obs needs to
71! be specified but it will only write out the actual number created.
72max_obs = 100000
73num_copies = 1
74num_qc = 1
75
76! Set the DART data quality control. 0 is good data.
77! increasingly larger QC values are more questionable quality data.
78qc = 0.0_r8
79
80! call the initialization code, and initialize two empty observation types
81call static_init_obs_sequence()
82call init_obs(obs, num_copies, num_qc)
83call init_obs(prev_obs, num_copies, num_qc)
84first_obs = .true.
85
86! create a new, empty obs_seq file. you must give a max limit
87! on number of obs. increase the size if too small.
88call init_obs_sequence(obs_seq, num_copies, num_qc, max_obs)
89
90! the first one needs to contain the string 'observation' and the
91! second needs the string 'QC'.
92call set_copy_meta_data(obs_seq, 1, 'observation')
93call set_qc_meta_data(obs_seq, 1, 'Data QC')
94
95! if you want to append to existing files (e.g. you have a lot of
96! small text files you want to combine), you can do it this way,
97! or you can use the obs_sequence_tool to merge a list of files
98! once they are in DART obs_seq format.
99
100! ! existing file found, append to it
101! inquire(file=obs_out_file, exist=file_exist)
102! if ( file_exist ) then
103! call read_obs_seq(obs_out_file, 0, 0, max_obs, obs_seq)
104! endif
105
106! open input text file
107iunit = open_file(text_input_file, 'formatted', 'read')
108if (debug) print *, 'opened input file ' // trim(text_input_file)
109
110obsloop: do ! no end limit - have the loop break when input ends
111
112 ! read in a line from the text file. What you need to create an obs:
113 ! location: lat, lon, and height in pressure or meters
114 ! time: when the observation was taken
115 ! type: from the DART list of obs types
116 ! error: very important - the instrument error plus representativeness error
117 ! (see html file for more info)
118
119 ! assume here a line is a type, location, time, value, obs error
120
121 ! read in entire text line into a buffer
122 read(iunit, "(A)", iostat=rcio) input_line
123 if (rcio /= 0) then
124 if (debug) print *, 'got bad read code from input file, rcio = ', rcio
125 exit obsloop
126 endif
127
128 ! pull off the first value as an integer, to decode the type
129 read(input_line, *, iostat=rcio) otype
130 if (rcio /= 0) then
131 if (debug) print *, 'got bad read code trying to get obs type, rcio = ', rcio
132 exit obsloop
133 endif
134
135 if (debug) print *, 'next observation type = ', otype
136
137 ! for this example, assume there is an obs type, where otype=1 is
138 ! a temperature, and if otype=37, there's Rho (neutral density)
139 ! The input text file has these as their own hardcoded convention.
140
141 if (otype == 1) then !Temperature
142 read(input_line, *, iostat=rcio) otype, lat, lon, vert, &
143 year, month, day, hour, minute, second, &
144 temp, terr
145 if (rcio /= 0) then
146 if (debug) print *, 'got bad read code getting rest of temp obs, rcio = ', rcio
147 exit obsloop
148 endif
149
150 elseif (otype == 37) then !Rho
151 read(input_line, *, iostat=rcio) otype, lat, lon, vert, &
152 year, month, day, hour, minute, second, &
153 temp, terr
154 if (rcio /= 0) then
155 if (debug) print *, 'got bad read code getting rest of temp obs, rcio = ', rcio
156 exit obsloop
157 endif
158
159 else !WHAT?
160
161 ! no method defined to convert this type
162 write(*,*) "NOOOOO, DON'T DO THIS: no method defined to convert this type = ", otype
163
164 endif
165
166 if (debug) print *, 'next observation located at lat, lon = ', lat, lon
167
168 ! check the lat/lon values to see if they are ok
169 if ( lat > 90.0_r8 .or. lat < -90.0_r8 ) cycle obsloop
170 if ( lon < 0.0_r8 .or. lon > 360.0_r8 ) cycle obsloop
171
172 ! put date into a dart time format
173 time_obs = set_date(year, month, day, hour, minute, second)
174
175 if (debug) call print_date(time_obs, 'next obs time is')
176
177 ! extract time of observation into gregorian day, sec.
178 call get_time(time_obs, osec, oday)
179
180 ! this example assumes there is an obs type, where otype=1 is
181 ! a temperature measured in height, and if otype=38, there's a TEC obs
182 if (otype == 1) then !Temperature
183
184 ! height is in meters (gph)
185 ! make an obs derived type, and then add it to the sequence
186 call create_3d_obs(lat, lon, vert, vertisheight, temp, &
187 sat_temperature, terr, oday, osec, qc, obs)
188 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
189 if (debug) print *, 'added Temperature obs to output seq'
190
191 elseif (otype == 37) then !SAT_RHO=37, but probably only in GITM
192 ! height is in meters (gph)
193 call create_3d_obs(lat, lon, vert, vertisheight, temp, &
194 sat_rho, terr, oday, osec, qc, obs)
195 call add_obs_to_seq(obs_seq, obs, time_obs, prev_obs, prev_time, first_obs)
196 if (debug) print *, 'added Rho obs to output seq'
197
198
199 else !WHAT???
200
201 ! no method defined to convert this type
202 write(*,*) 'no method defined to convert this type = ', otype
203
204 endif
205
206enddo obsloop
207
208! if we added any obs to the sequence, write it out to a file now.
209if ( get_num_obs(obs_seq) > 0 ) then
210 if (debug) print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
211 print *, 'writing obs_seq, obs_count = ', get_num_obs(obs_seq)
212 call write_obs_seq(obs_seq, obs_out_file)
213endif
214
215! end of main program
216call finalize_utilities()
217
218end program text_to_obs
