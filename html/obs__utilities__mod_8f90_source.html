239!> lat - latitude of observation (all the rest out)
240!> lon - longitude of observation

245!> oerr - observation error

253subroutine query_3d_obs(obs, lat, lon, vval, vkind, obsv, okind, oerr, day, sec, qc)
254

257 real(r8), intent(out) :: lat, lon, vval, obsv, oerr, qc

273

279! now get the value and qc from the obs_type.

288!--------------------------------------------------------------------
289!> adds an observation to a sequence. inserts if first obs, inserts
290!> with a prev obs to save searching if that's possible.
291!>
292!> seq - observation sequence to add obs to
293!> obs - observation, already filled in, ready to add
294!> obs_time - time of this observation, in dart time_type format
295!> prev_obs - the previous observation that was added to this sequence
296!> (will be updated by this routine)
297!> prev_time - the time of the previously added observation (will also
298!> be updated by this routine)
299!> first_obs - should be initialized to be .true., and then will be
300!> updated by this routine to be .false. after the first obs
301!> has been added to this sequence.
302!>
303!> created Mar 8, 2010 nancy collins, ncar/image
304
305subroutine add_obs_to_seq(seq, obs, obs_time, prev_obs, prev_time, first_obs)
306
307 type(obs_sequence_type), intent(inout) :: seq
308 type(obs_type), intent(inout) :: obs, prev_obs
309 type(time_type), intent(in) :: obs_time
310 type(time_type), intent(inout) :: prev_time
311 logical, intent(inout) :: first_obs
312
313! insert(seq,obs) always works (i.e. it inserts the obs in
314! proper time format) but it can be slow with a long file.
315! supplying a previous observation that is older (or the same
316! time) as the new one speeds up the searching a lot.
317
318if(first_obs) then ! for the first observation, no prev_obs
319 call insert_obs_in_seq(seq, obs)
320 first_obs = .false.
321else
322 if(obs_time >= prev_time) then ! same time or later than previous obs
323 call insert_obs_in_seq(seq, obs, prev_obs)
324 else ! earlier, search from start of seq
325 call insert_obs_in_seq(seq, obs)
326 endif
327endif
328
329! update for next time
330prev_obs = obs
331prev_time = obs_time
332
333end subroutine add_obs_to_seq

402!--------------------------------------------------------------------
403!> sets the name of the attribute that describes missing values.
404!>
405!> the suggested rule is data that was never written to the netcdf
406!> variable will have a value equal to "_FillValue", and this should
407!> be added as an attribute on the variable when it is created.
408!> the "missing_data" attribute indicates data that was written
409!> but is not valid or present, e.g. land grid points in an ocean
410!> data field.
411!>
412!> but in practice there are problems. one is that users may not
413!> add any attributes and have a private convention for missing or
414!> invalid data (e.g. 99999 or some such pattern).
415!>
416!> the netcdf files can be created with an option to NOT prefill
417!> variables with the _FillValue. this speeds execution when the
418!> code expects to fully write all variables. but if there are
419!> paths through the code where a variable isn't written by mistake,
420!> the contents are undefined.
421!>
422!> and finally, some users mix 'missing_value' and "_FillValue" usage,
423!> writing the value of "_FillValue" to indicate missing data.
424!>
425!> in this code all we care about is knowing what data value
426!> indicates data is missing for any reason. if the code uses
427!> a different attribute value, set it with this routine.
428!>
429!> name - string name of attribute that holds the missing value
430!>
431!> created 11 Mar 2010, nancy collins, ncar/image
432!> (this usage comment updated 26 sep 2019, nsc)
433
434subroutine set_missing_name(name)
435
436 character(len = *), intent(in) :: name
437
438 if (len(name) > nf90_max_name) then
439 write(string1,*) 'name must be less than ', nf90_max_name, ' chars long'
440 write(string2,*) 'name is ', len(name), ' long'
441 call error_handler(e_err,'set_missing_name:',string1,source,text2=string2)
442 endif
443
444 missing_name = name
445
446end subroutine set_missing_name

449!--------------------------------------------------------------------
450!>inquires, gets the variable, and fills
451!> in the missing value attribute if that arg is present.
452!> gets the entire array, no start or count specified.
453!>
454!> ncid - open netcdf file handle
455!> varname - string name of netcdf variable
456!> darray - output array. real(r8)
457!> dmiss - value that signals a missing value real(r8), optional
458!>
459!> created 11 Mar 2010, nancy collins, ncar/image
460
461subroutine getvar_real(ncid, varname, darray, dmiss)
462
463 integer, intent(in) :: ncid
464 character(len = *), intent(in) :: varname
465 real(r8), intent(out) :: darray(:)
466 real(r8), optional, intent(out) :: dmiss
467
468integer :: varid, nfrc
469real(r8) :: fill, miss
470logical :: found_miss
471
472! read the data for the requested array, and optionally get the fill value
473call nc_check( nf90_inq_varid(ncid, varname, varid), &
474 'getvar_real', 'inquire var '// trim(varname))
475call nc_check( nf90_get_var(ncid, varid, darray), &
476 'getvar_real', 'getting var '// trim(varname))
477
478! the logic here is:
479! if the user hasn't asked about the fill value, don't look for any of
480! these attributes and just return.
481
482! if the user has told us another attribute name to look for, try that first.
483! it's currently NOT an error if it's not there. then second, look for the
484! official '_FillValue' attr. if it's found, return it as the missing value.
485!
486! if there are both, overwrite the missing value with the fill value and return
487! the fill value as the 'dmiss' argument.
488!
489! if neither are there, set dmiss to the DART missing value. this could also
490! be an error, but default to being permissive for now.
491
492if (present(dmiss)) then
493 dmiss = missing_r8
494 found_miss = .false.
495
496 ! if user defined another attribute name for missing vals
497 ! look for it first, and make it an error if it's not there?
498 if (missing_name /= '') then
499 nfrc = nf90_get_att(ncid, varid, missing_name, miss)
500 if (nfrc == nf90_noerr) then
501 found_miss = .true.
502 dmiss = miss
503 endif
504 endif
505
506 ! this would make it a fatal error if not found
507 !call nc_check( nf90_get_att(ncid, varid, missing_name', miss), &
508 ! 'getvar_real', 'getting attr "//trim(missing_name)//" for '//trim(varname))
509
510 ! the predefined netcdf fill value.
511 nfrc = nf90_get_att(ncid, varid, '_FillValue', fill)
512 if (nfrc == nf90_noerr) then
513 if (.not. found_miss) then
514 found_miss = .true.
515 dmiss = fill
516 else
517 ! found both, set all to fill value
518 where(darray .eq. miss) darray = fill
519 dmiss = fill
520 endif
521 endif
522
523 ! if you wanted an error if you specified dmiss and neither attr are
524 ! there, you'd test found_miss here. if it's still false, none were there.
525
526endif
527
528end subroutine getvar_real

531!--------------------------------------------------------------------
532!> get_short_as_r8 - subroutine that inquires, gets the variable, and fills
533!> in the missing value attribute if that arg is present.
534!> gets the entire array, no start or count specified.
535!>
536!> FIXME: this code handles scale and offset ok, but like most of the other
537!> routines here it doesn't check for 'missing_value' which many obs files
538!> have either in addition to the _FillValue or instead of it.
539!> e.g. a dump from a MADIS mesonet file:
540!>
541!> double observationTime ( recNum )
542!> long_name : time of observation
543!> units : seconds since 1970-1-1 00:00:00.0
544!> _FillValue : 3.40282346e+38
545!> missing_value : -9999
546!> standard_name : time
547!>
548!> i've most commonly encountered the "missing_value" in the actual data
549!> for obs files, not the fill value. we need a routine that looks for missing,
550!> then fill, and decides what to do if there are both. or use the routine
551!> that was already here 'set_missing_value' and the calling code tells us
552!> which attribute name this particular netcdf file is using.
553!>
554!> ncid - open netcdf file handle
555!> varname - string name of netcdf variable
556!> darray - output array. real(r8)
557!> dmiss - value that signals a missing value real(r8), optional
558
559subroutine get_short_as_r8(ncid, varname, darray, dmiss)
560
561integer, intent(in) :: ncid
562character(len=*), intent(in) :: varname
563real(r8), intent(out) :: darray(:)
564real(r8), optional, intent(out) :: dmiss
565
566integer :: varid
567integer(i2) :: shortarray(size(darray))
568integer(i2) :: fillvalue
569real(r8) :: scale_factor, add_offset
570integer :: offset_exists, scaling_exists, fill_exists
571!>@todo FIXME need missing_exists or something
572
573! read the data for the requested array, and get the fill value
574call nc_check( nf90_inq_varid(ncid, varname, varid), &
575 'get_short_as_r8', 'inquire var "'// trim(varname)//'"')
576
577 offset_exists = nf90_get_att(ncid, varid, 'add_offset', add_offset)
578scaling_exists = nf90_get_att(ncid, varid, 'scale_factor', scale_factor)
579 fill_exists = nf90_get_att(ncid, varid, '_FillValue', fillvalue)
580! miss_exists = nf90_get_att(ncid, varid, 'missing_value', miss_value)
581
582call nc_check( nf90_get_var(ncid, varid, shortarray), &
583 'get_short_as_r8', 'getting var '// trim(varname))
584
585darray = real(shortarray,r8)
586
587
588if (fill_exists == nf90_noerr) then ! FillValue exists
589
590 if ( (offset_exists == nf90_noerr) .and. (scaling_exists == nf90_noerr) ) then
591 where (shortarray /= fillvalue) darray = darray * scale_factor + add_offset
592 elseif (offset_exists == nf90_noerr) then
593 where (darray /= fillvalue) darray = darray * scale_factor
594 elseif (scaling_exists == nf90_noerr) then
595 where (darray /= fillvalue) darray = darray + add_offset
596 endif
597
598 if (present(dmiss)) dmiss = real(fillvalue,r8)
599
600else
601
602 if ( (offset_exists == nf90_noerr) .and. (scaling_exists == nf90_noerr) ) then
603 darray = darray * scale_factor + add_offset
604 elseif (offset_exists == nf90_noerr) then
605 darray = darray * scale_factor
606 elseif (scaling_exists == nf90_noerr) then
607 darray = darray + add_offset
608 endif
609
610 if (present(dmiss)) dmiss = missing_r8
611
612endif
613
614end subroutine get_short_as_r8

1042!> scale_factor
1043!>

1046!>
1047!> If valid values are specified using the valid_min, valid_max, valid_range,

1050!> scale_factor and add_offset are applied.
1051!>
1052!> add_offset
1053!>
1054!> If present for a variable, this number is to be added to the data after it is read
1055!> by the application that accesses the data. If both scale_factor and add_offset
1056!> attributes are present, the data are first scaled before the offset is added.
1057!> The attributes scale_factor and add_offset can be used together to provide simple
1058!> data compression to store low-resolution floating-point data as small integers in
1059!> a netCDF dataset. When scaled data are written, the application should first subtract
1060!> the offset and then divide by the scale factor, rounding the result to the nearest
1061!> integer to avoid a bias caused by truncation towards zero.
1062!>
1063!> When scale_factor and add_offset are used for packing, the associated variable
1064!> (containing the packed data) is typically of type byte or short, whereas the unpacked
1065!> values are intended to be of type float or double. The attributes scale_factor and
1066!> add_offset should both be of the type intended for the unpacked data, e.g. float or double.
