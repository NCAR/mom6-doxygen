321!-----------------------------------------------------------------------
322!> Match the observation stationId string to the RouteLink gage string to
323!> determine which gage has the location matching the stationId.
324!>
325!> If the gage is not one we desire, set the id to 0 to just skip it.
326
327function find_matching_gage_index(counter) result(id)
328
329integer, intent(in) :: counter
330integer :: id
331
332integer :: i
333
334id = 0 ! Indicate the station has no matching gage or is not wanted.
335
336links : do i = 1,nlinks
337 if (station_strings(counter) == gage_strings(i)) then
338 if (verbose > 0) then
339 write(string1,*)'identified station "',station_strings(counter),'"'
340 call error_handler(e_msg, 'find_matching_gage_index:', string1)
341 endif
342 id = i
343 exit links
344 endif
345enddo links
346
347if (id == 0 .and. verbose > 1) then
348 write(string1,*)'Unable to match station id for obs #',counter,' "',station_strings(counter),'"'
349 call error_handler(e_msg, 'find_matching_gage_index', string1, source, revision, revdate)
350endif
351
352if (id == 0) return ! unable to find location information
353
354! If no subsetting gage list is specified, we want them all.
355! If the gage is desired, we want it.
356! Just return with whatever id we have already found.
357if ( n_wanted_gages == 0 ) then
358 continue
359else if( any(desired_gages == station_strings(counter)) ) then
360 continue
361else
362 ! We do not want the gage
363 id = 0
364endif
365
366end function find_matching_gage_index

572function estimate_total_obs_count(file_list,nfiles) result (num_obs)
573
574character(len=*), intent(in) :: file_list
575integer, intent(in) :: nfiles
576integer :: num_obs
577
578! Rather than opening up each file and getting the actual number of possible
579! observations in each file, we are just going to extrapolate from a single
580! file and add a fudge factor. There does not seem to be much variability in
581! the number of possible observations from file-to-file.
582
583character(len=*), parameter :: routine = 'estimate_total_obs_count'
584integer :: iunit, io, ncid, nobs
585character(len=256) :: input_file
586
587iunit = open_file(file_list,form='formatted',action='read')
588read(iunit,'(A)', iostat=io) input_file
589if (io /= 0 ) then
590 write(string1,*) 'Unable to read input file from "'//trim(file_list)//'"'
591 call error_handler(e_err,routine,string1,source,revision,revdate)
592endif
593call close_file(iunit)
594
595! Need to know about how many observations are in each file.
596io = nf90_open(input_file, nf90_nowrite, ncid)
597call nc_check(io, routine, 'opening data file "'//trim(input_file)//'"' )
598call getdimlen(ncid, 'stationIdInd', nobs)
599io = nf90_close(ncid)
600call nc_check(io, routine, 'closing file "'//trim(input_file)//'"' )
601
602! We need to know how many observations there may be.
603! Specifying too many is not really a problem.
604! I am adding 20%
605
606num_obs = 1.2_r8 * nobs * nfiles
607
608end function estimate_total_obs_count
