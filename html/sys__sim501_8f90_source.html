7! See work from 13 October, 2003. Test to evaluate how to look_for_bias in
8! prior ensemble distribution and observation in observation space.
9
10use random_seq_mod, only : random_seq_type, init_random_seq, random_gaussian, &
11 twod_gaussians, random_uniform
12
13implicit none
14
15type (random_seq_type) :: r
16double precision :: correl, cov(2, 2), zero_2(2) = 0.0, correl_mean, correl_var
17double precision :: q, alpha, alpha2, mean_abs_correl
18double precision, allocatable :: pairs(:, :), sample_correl(:)
19integer :: ens_size, n_samples, i, j, k
20
21! Initialize repeatable random sequence
22call init_random_seq(r)
23
24! For now have a set of free parameters that may be too large
25write(*, *) 'Input ensemble size'
26read(*, *) ens_size
27write(*, *) 'Input number of samples for statistics'
28read(*, *) n_samples
29
30! Allocate storage for sample from this correlation
31allocate(pairs(2, ens_size), sample_correl(n_samples))
32
33! Loop through 101 correlaton values
34do k = 1, 101
35 correl = (k - 1.0) * 0.01
36 ! Generate the covariance matrix for this correlation
37 cov(1, 1) = 1.0; cov(2, 2) = 1.0
38 cov(1, 2) = correl; cov(2, 1) = correl
39
40 ! Loop for each sample size
41 do j = 1, n_samples
42 ! Loop to generate an ensemble size sample from this correl
43 ! Generate a random sample of size ens_size from something with this correlation
44 do i = 1, ens_size
45 call twod_gaussians(r, zero_2, cov, pairs(:, i))
46 end do
47
48 ! Compute the sample correlation
49 call comp_correl(pairs, ens_size, sample_correl(j))
50 !!! write(*, *) 'sample correl is ', sample_correl(j)
51 end do
52
53 ! Compute a mean and standard deviation of the sample correlations
54 call sample_mean_var(sample_correl, n_samples, correl_mean, correl_var)
55 q = sqrt(correl_var) / abs(correl_mean)
56 write(*, *) 'correl mean and sd', correl_mean, sqrt(correl_var)
57 write(*, *) 'ratio of sd to mean is ', q
58
59 ! Also compute the mean of the absolute value of correl
60 mean_abs_correl = sum(abs(sample_correl)) / n_samples
61 write(*, *) 'correl, mean_abs_correl ', correl, mean_abs_correl
62
63 ! Value of alpha (regression confidence factor) is computed as limit of
64 ! group size m -> infinity from paper: alpha = 1 / (Q^2 + 1)
65 alpha = 1.0 / (q**2 + 1)
66 if(alpha < 0) alpha = 0.0
67 ! Maybe I should really be doing this in limit of m = 1, single sample?
68 alpha2 = 1.0 - q**2
69 if(alpha2 < 0.0) alpha2 = 0.0
70 write(*, *) correl, alpha, alpha2
71end do
72
73end program sys_sim501

77subroutine comp_correl(ens, n, correl)

79implicit none

81integer, intent(in) :: n
82double precision, intent(in) :: ens(2, n)
83double precision, intent(out) :: correl
84double precision :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
85
86
87sum_x = sum(ens(2, :))
88sum_y = sum(ens(1, :))
89sum_xy = sum(ens(2, :) * ens(1, :))
90sum_x2 = sum(ens(2, :) * ens(2, :))

92! Computation of correlation

94
95correl = (n * sum_xy - sum_x * sum_y) / &
96 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
