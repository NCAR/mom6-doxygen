163! Process each input observation sequence file in turn, optionally

169! Read the namelist entry
170call find_namelist_in_file("input.nml", "obs_sequence_tool_nml", iunit)
171read(iunit, nml = obs_sequence_tool_nml, iostat = io)
172call check_namelist_read(iunit, io, "obs_sequence_tool_nml")

179! check it first before overwriting it in the call to set_filename_list().

181 write(msgstring1, *) '"num_input_files" is a DEPRECATED namelist item and is ignored.'

190! this is a logical. if the calendar is gregorian it prints out times in
191! both (day,second) format and date (year/month/day/hour/minute/second) format.
192! otherwise it just prints (day,second). if users want other calendar type
193! support this should be renamed to 'calendar_type' and be a character string.
194if (gregorian_cal) call set_calendar_type(gregorian)
195
196! See if the user is restricting the obs types to be processed, and set up
197! the values if so.
198num_obs_input_types = 0
199do i = 1, max_obs_input_types
200 if ( len(obs_types(i)) .eq. 0 .or. obs_types(i) .eq. "" ) exit
201 num_obs_input_types = i
202enddo

205else

207endif

209! See if the user wants to suppress the precomputed forward operator values
210! for any/all observation types with precomputed forward operator values
211
212num_fo_types_to_suppress = 0
213do i = 1, max_obs_input_types
214 if ( len(remove_precomputed_fo_values(i)) == 0 .or. &
215 remove_precomputed_fo_values(i) == "" ) exit
216 num_fo_types_to_suppress = i
217enddo
218if (num_fo_types_to_suppress > 0) call setup_fo_suppress_list(fo_write_types)
219
220! See if the user is restricting the obs locations to be processed, and set up
221! the values if so. Note that if any of the values are not missing_r8, all of
222! them must have good (and consistent) values. i don't have code in here yet
223! to idiotproof this, but i should add it.
224
225restrict_by_location = .false.
226if ((minval(min_box).ne.missing_r8) .or. (maxval(min_box).ne.missing_r8) .or. &
227 (minval(max_box).ne.missing_r8) .or. (maxval(max_box).ne.missing_r8)) then
228 restrict_by_location = .true.
229 min_loc = set_location(min_box)
230 max_loc = set_location(max_box)
231endif
232
233! 3d sphere box check - locations module dependent, but an important one.
234if ((min_lat /= -90.0_r8) .or. (max_lat /= 90.0_r8) .or. &
235 (min_lon /= 0.0_r8) .or. (max_lon /= 360.0_r8)) then
236 ! do not allow namelist to set BOTH min/max box and by lat/lon.
237 if (restrict_by_location) then
238 call error_handler(e_err,'obs_sequence_tool', &
239 'use either lat/lon box or min/max box but not both', source)
240 endif
241 restrict_by_location = .true.
242 if ((trim(locationname) /= 'loc3Dsphere') .and. &
243 (trim(locationname) /= 'loc2Dsphere')) then
244 call error_handler(e_err,'obs_sequence_tool', &
245 'can only use lat/lon box with 2d/3d sphere locations', source)
246 endif
247 ! simple err checks before going on; try to catch radians vs degrees or
248 ! just plain junk or typos.
249 if (min_lat >= max_lat) then
250 call error_handler(e_err,'obs_sequence_tool', &
251 'min_lat must be less than max_lat', source)
252 endif
253 if (min_lat < -90.0_r8) then
254 call error_handler(e_err,'obs_sequence_tool', &
255 'min_lat cannot be less than -90.0 degrees', source)
256 endif
257 if (max_lat > 90.0_r8) then
258 call error_handler(e_err,'obs_sequence_tool', &
259 'max_lat cannot be greater than 90.0 degrees', source)
260 endif
261 ! this is ok - e.g.-180 to 180
262 !if (min_lon < 0.0_r8) then
263 ! call error_handler(E_ERR,'obs_sequence_tool', &
264 ! 'min_lon cannot be less than 0.0 degrees', source)
265 !endif
266 if (min_lon > 360.0_r8) then
267 call error_handler(e_err,'obs_sequence_tool', &
268 'min_lon cannot be greater than 360.0 degrees', source)
269 endif
270 if (max_lon > 360.0_r8) then
271 call error_handler(e_err,'obs_sequence_tool', &
272 'max_lon cannot be greater than 360.0 degrees', source)
273 endif
274
275 ! it is risky to allow == operations on floating point numbers.
276 ! if someone wants to select only obs exactly on a particular lon,
277 ! force them to do an interval [min-epsilon, max-epsilon].
278 ! move this test BEFORE the modulo so that if afterwards they are
279 ! the same (e.g. 0 and 360) that's ok -- it will mean all longitudes
280 ! will be accepted.
281 if (min_lon == max_lon) then
282 call error_handler(e_err,'obs_sequence_tool', &
283 'min_lon cannot exactly equal max_lon', source)
284 endif
285
286 ! do not test for min < max, because lon can wrap around 0. ensure that
287 ! the values after here are [0, 360) and the compare routine knows about wrap.
288 ! must use modulo() and not just mod() to handle negative values correctly;
289 ! the result of modulo() is positive even if the input is negative.
290 min_lon = modulo(min_lon,360.0_r8)
291 max_lon = modulo(max_lon,360.0_r8)
292
293 ! create real location_type objects, knowing we are running with the
294 ! 3d sphere locations module.
295 min_box(1) = min_lon
296 min_box(2) = min_lat
297 min_box(3) = 0.0_r8
298 min_box(4) = -2.0_r8 !! FIXME: VERTISUNDEF, but not all loc mods have it
299 min_loc = set_location(min_box)
300 max_box(1) = max_lon
301 max_box(2) = max_lat
302 max_box(3) = 0.0_r8
303 max_box(4) = -2.0_r8 !! FIXME: VERTISUNDEF, but not all loc mods have it
304 max_loc = set_location(max_box)
305endif
306
307!%! ! SPECIAL: cut off all GPS obs below the given height
308!%! ! see comments in select_gps_by_height() for more info.
309!%! if (min_gps_height /= missing_r8) then
310!%! restrict_by_height = .true.
311!%! else
312!%! restrict_by_height = .false.
313!%! endif
314
315
316! See if the user is restricting the data values or qc to be processed,
317! and if so, set up
318if ((min_qc /= missing_r8) .or. (max_qc /= missing_r8)) then
319 if (len(trim(qc_metadata)) == 0) then
320 call error_handler(e_err,'obs_sequence_tool', &
321 'must specify the metadata name of a QC field', source)
322 endif
323 restrict_by_qc = .true.
324else
325 restrict_by_qc = .false.
326endif

328if ((min_copy /= missing_r8) .or. (max_copy /= missing_r8)) then
329 if (len(trim(copy_metadata)) == 0) then
330 call error_handler(e_err,'obs_sequence_tool', &
331 'must specify the metadata name of a copy field', source)
332 endif
333 restrict_by_copy = .true.
334else
335 restrict_by_copy = .false.
336endif
337
338! See if the user is requesting edit of either the number of copies
339! or qc values, or reordering them.
340editing_obs = .false.
341if (edit_copies) then
342 ! count up how many copies and qc there are going out.
343 do i = 1, size(new_copy_index)
344 if (new_copy_index(i) < 0) exit
345 copy_index_len = copy_index_len + 1
346 enddo
347 editing_obs = .true.
348endif
349if (edit_qcs) then
350 do i = 1, size(new_qc_index)
351 if (new_qc_index(i) < 0) exit
352 qc_index_len = qc_index_len + 1
353 enddo
354 editing_obs = .true.
355endif
356
357! check to see if we are going to trim the sequence by time
358if(first_obs_seconds >= 0 .or. first_obs_days >= 0) then
359 first_obs_time = set_time(first_obs_seconds, first_obs_days)
360 trim_first = .true.
361else
362 trim_first = .false.
363endif
364if(last_obs_seconds >= 0 .or. last_obs_days >= 0) then
365 last_obs_time = set_time(last_obs_seconds, last_obs_days)
366 trim_last = .true.
367else
368 trim_last = .false.
369endif
370
371if (trim_first) then
372 call print_time(first_obs_time, 'Excluding observations before: ')
373 if (gregorian_cal) &
374 call print_date(first_obs_time, ' which is Gregorian day: ')
375endif
376if (trim_last) then
377 call print_time(last_obs_time, 'Excluding observations after: ')
378 if (gregorian_cal) &
379 call print_date(last_obs_time, ' which is Gregorian day: ')
380endif
381
382if (trim_first .and. trim_last) then
383 if (first_obs_time > last_obs_time) then
384 call error_handler(e_err,'obs_sequence_tool', &
385 'first time cannot be later than last time', source)
386 endif
387endif
388
389! check to see if user gave us lists of metadata that should be
390! considered the same across different obs_seq files. count the
391! length of each list. used in the compare_metadata() routine.
392matching_copy_metadata = .false.
393if (synonymous_copy_list(1) /= '') then
394 matching_copy_metadata = .true.
395 do i = 1, size(synonymous_copy_list)
396 if (synonymous_copy_list(i) == '') exit
397 matching_copy_limit = i
398 enddo
399endif
400matching_qc_metadata = .false.
401if (synonymous_qc_list(1) /= '') then
402 matching_qc_metadata = .true.
403 do i = 1, size(synonymous_qc_list)
404 if (synonymous_qc_list(i) == '') exit
405 matching_qc_limit = i
406 enddo
407endif
408
409! end of namelist processing and setup
410
411! Read header information for the sequences to see if we need
412! to accomodate additional copies or qc values from subsequent sequences.
413! Also, calculate how many observations to be added to the first sequence.
414num_copies_out = 0
415num_qc_out = 0
416size_seq_out = 0
417
418! TWO PASS algorithm; open each file, trim it if requested, and count
419! the number of actual observations. then the output file can be
420! created with the correct size, and as observations are put into it
421! they'll be sorted, and unused obs will be removed.
422
423! pass 1:
424
425first_seq = -1
426do i = 1, num_input_files
427
428 if ( len(filename_seq(i)) .eq. 0 .or. filename_seq(i) .eq. "" ) then
429 call error_handler(e_err,'obs_sequence_tool', &
430 'num_input_files and filename_seq mismatch',source)
431 endif
432
433 ! count up the number of observations we are going to eventually have.
434 ! if all the observations in a file are not part of the linked list, the
435 ! output number of observations might be much smaller than the total size in
436 ! the header. it is slower, but go ahead and read in the entire sequence
437 ! and count up the real number of obs - trim_seq will do the count even if
438 ! it is not trimming in time. this allows us to create an empty obs_seq
439 ! output file of exactly the right size.
440
441 call read_obs_seq_header(filename_seq(i), num_copies_in, num_qc_in, &
442 size_seq_in, max_num_obs, file_id, read_format, pre_i_format, &
443 close_the_file = .true.)
444
445 call read_obs_seq(filename_seq(i), 0, 0, 0, seq_in)
446 call trim_seq(seq_in, trim_first, first_obs_time, trim_last, last_obs_time, &
447 filename_seq(i), .true., remaining_obs_count)
448 call destroy_obs_sequence(seq_in)
449 if (remaining_obs_count == 0) then
450 process_file(i) = .false.
451 write(msgstring1,*) 'No obs used from input sequence file ', trim(filename_seq(i))
452 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
453 cycle
454 else
455 process_file(i) = .true.
456 size_seq_in = remaining_obs_count
457 endif
458
459 ! First sequence with valid data gets to set up the format of the output.
460 ! This only matters if we keep the 'non-exact match of metadata' in the
461 ! compare_metadata() subroutine below.
462 if ( first_seq < 0 ) then
463 first_seq = i
464 ! set up allowing number of copies or qcs to be changed or reordered.
465 if (edit_copies) then
466 ! do a tiny bit of sanity checking here.
467 do j = 1, copy_index_len
468 if ((new_copy_index(j) > num_copies_in) .or. &
469 (new_copy_index(j) < 0)) then ! was < 1 here and below
470 write(msgstring1,*)'new_copy_index values must be between 0 and ', num_copies_in
471 call error_handler(e_err,'obs_sequence_tool', msgstring1, source)
472
473 endif
474 enddo
475 if (copy_index_len > num_copies_in) then
476 write(msgstring1,*)'WARNING: more output copies than input; some are being replicated'
477 call error_handler(e_msg,'obs_sequence_tool', msgstring1)
478 endif
479 num_copies_out = copy_index_len
480 else
481 num_copies_out = num_copies_in
482 do j = 1, num_copies_in
483 new_copy_index(j) = j
484 enddo
485 endif
486 if (edit_qcs) then
487 do j = 1, qc_index_len
488 if ((new_qc_index(j) > num_qc_in) .or. &
489 (new_qc_index(j) < 0)) then ! was < 1 here and below
490 write(msgstring1,*)'new_qc_index values must be between 0 and ', num_qc_in
491 call error_handler(e_err,'obs_sequence_tool', msgstring1, source)
492
493 endif
494 enddo
495 if (qc_index_len > num_qc_in) then
496 write(msgstring1,*)'WARNING: more output qcs than input; some are being replicated'
497 call error_handler(e_msg,'obs_sequence_tool', msgstring1)
498 endif
499 num_qc_out = qc_index_len
500 else
501 num_qc_out = num_qc_in
502 do j = 1, num_qc_in
503 new_qc_index(j) = j
504 enddo
505 endif
506 size_seq_out = size_seq_in
507 else
508 size_seq_out = size_seq_out + size_seq_in
509 endif
510
511enddo
512
513! no valid obs found? if the index value is still negative, we are
514! still waiting to process the first one and never found one.
515if (first_seq < 0 .or. size_seq_out == 0) then
516 msgstring1 = 'All input files are empty or all obs excluded by time/type/location'
517 call error_handler(e_err,'obs_sequence_tool',msgstring1,source)
518endif
519
520! pass 2:
521
522! blank line, start of actually creating output file
523call error_handler(e_msg,' ',' ')
524
525! Initialize individual observation variables
526call init_obs( obs_in, num_copies_in, num_qc_in )
527call init_obs(next_obs_in, num_copies_in, num_qc_in )
528call init_obs( obs_out, num_copies_out, num_qc_out)
529call init_obs(prev_obs_out, num_copies_out, num_qc_out)
530
531total_num_inserted = 0
532
533! Read obs seq to be added, and insert obs from it to the output seq
534first_seq = -1
535do i = 1, num_input_files
536
537 if (.not. process_file(i)) cycle
538
539 write(msgstring1,*) 'Starting to process input sequence file ', trim(filename_seq(i))
540 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
541
542 call read_obs_seq(filename_seq(i), 0, 0, 0, seq_in)
543
544 ! If you get here, there better be observations in this file which
545 ! are going to be used (the process_file flag wouldn't be set otherwise.)
546 call trim_seq(seq_in, trim_first, first_obs_time, trim_last, last_obs_time, &
547 filename_seq(i), .false., remaining_obs_count)
548
549 ! This would be an error at this point.
550 if(remaining_obs_count == 0) then
551 call destroy_obs_sequence(seq_in)
552 write(msgstring1, *) 'Internal error trying to process file ', trim(filename_seq(i))
553 call error_handler(e_err,'obs_sequence_tool',msgstring1,source)
554 endif
555
556 ! create the output sequence here based on the first input file
557 if (first_seq < 0) then
558 call init_obs_sequence(seq_out, num_copies_out, num_qc_out, size_seq_out)
559 do j=1, num_copies_out
560 if (edit_copy_metadata) then
561 meta_data = new_copy_metadata(j)
562 if (new_copy_index(j) > 0) then
563 write(msgstring1, *)'replacing original copy meta_data: ' // &
564 trim(get_copy_meta_data(seq_in, new_copy_index(j)))
565 else
566 write(msgstring1, *)'replacing original copy meta_data: ' // &
567 trim(get_copy_meta_data(seq_out, j))
568 endif
569 write(msgstring2, *)' with: ' // trim(meta_data)
570 if (new_copy_index(j) /= j) then
571 write(msgstring3, *)'original index was ', new_copy_index(j), ', now ', j
572 else
573 msgstring3 = 'copy index has not changed'
574 endif
575 call error_handler(e_msg,'obs_sequence_tool',msgstring1,&
576 text2=msgstring2,text3=msgstring3)
577
578 else
579 if (new_copy_index(j) > 0) then
580 meta_data = get_copy_meta_data(seq_in, new_copy_index(j))
581 else
582 meta_data = get_copy_meta_data(seq_out, j)
583 endif
584 if (new_copy_index(j) /= j) then
585 write(msgstring1, *)'copy with meta_data: ' // trim(meta_data)
586 write(msgstring2, *)'moved from index ', new_copy_index(j), ' to ', j
587 call error_handler(e_msg,'obs_sequence_tool',msgstring1,text2=msgstring2)
588 endif
589 endif
590 call set_copy_meta_data(seq_out, j, meta_data)
591 enddo
592 do j=1, num_qc_out
593 if (edit_qc_metadata) then
594 meta_data = new_qc_metadata(j)
595 if (new_qc_index(j) > 0) then
596 write(msgstring1, *)'replacing original qc meta_data: ' // &
597 trim(get_qc_meta_data(seq_in, new_qc_index(j)))
598 else
599 write(msgstring1, *)'replacing original qc meta_data: ' // &
600 trim(get_qc_meta_data(seq_out, j))
601 endif
602 write(msgstring2, *)' with: ' // trim(meta_data)
603 if (new_qc_index(j) /= j) then
604 write(msgstring3, *)'original index was ', new_qc_index(j), ' now ', j
605 else
606 msgstring3 = 'QC index has not changed'
607 endif
608 call error_handler(e_msg,'obs_sequence_tool',msgstring1,&
609 text2=msgstring2,text3=msgstring3)
610
611 else
612 if (new_qc_index(j) > 0) then
613 meta_data = get_qc_meta_data(seq_in, new_qc_index(j))
614 else
615 meta_data = get_qc_meta_data(seq_out, j)
616 endif
617 if (new_qc_index(j) /= j) then
618 write(msgstring1, *)'qc with meta_data: ' // trim(meta_data)
619 write(msgstring2, *)'moved from index ', new_qc_index(j), ' now ', j
620 call error_handler(e_msg,'obs_sequence_tool',msgstring1,text2=msgstring2)
621 endif
622 endif
623 call set_qc_meta_data(seq_out, j, meta_data)
624 enddo
625 first_seq = i
626 else
627 ! we have an existing output sequence already. make sure the next one
628 ! is completely compatible.
629
630 ! Compare metadata between the observation sequences.
631 ! This routine exits if they do not match.
632 call compare_metadata(seq_out, seq_in, filename_seq(first_seq), filename_seq(i))
633 endif
634
635 size_seq_out = get_num_key_range(seq_out) !current size of seq_out
636 size_seq_in = get_num_key_range(seq_in) !current size of seq_in
637
638 ! ensure the linked list times are in increasing time order
639 call validate_obs_seq_time(seq_in, filename_seq(i))
640
641 if (print_only) call print_obs_seq(seq_in, filename_seq(i))
642
643 !-------------------------------------------------------------
644 ! Start to insert obs from sequence_in into sequence_out
645 !
646 ! NOTE: insert_obs_in_seq CHANGES the obs passed in.
647 ! Must pass a copy of incoming obs to insert_obs_in_seq.
648 !--------------------------------------------------------------
649 num_inserted = 0
650 is_there_one = get_first_obs(seq_in, obs_in)
651
652 if ( is_there_one ) then
653
654 if (editing_obs) then
655 ! copy subset or reordering
656 call copy_partial_obs(obs_out, obs_in, &
657 num_copies_out, new_copy_index, &
658 num_qc_out, new_qc_index)
659
660 ! if new copies or qcs were added, set the initial data values
661 call set_new_data(obs_out, edit_copies, new_copy_index, new_copy_data, &
662 edit_qcs, new_qc_index, new_qc_data)
663 else
664 obs_out = obs_in
665 endif
666
667 ! changing metadata to output precomputed FOs based on namelist
668 call set_fo_metadata(obs_out)
669
670!#! ! see comment in subroutine code for an explanation
671!#! call change_obs(obs_out)
672
673 call insert_obs_in_seq(seq_out, obs_out) ! new_obs linked list info changes
674
675 prev_obs_out = obs_out ! records new position in seq_out
676 num_inserted = num_inserted + 1
677
678 call get_next_obs(seq_in, obs_in, next_obs_in, is_this_last)
679 obsloop : do while ( .not. is_this_last)
680
681 if (print_every > 0) then
682 if (mod(num_inserted,print_every) == 0) then
683 print*, 'inserted number ',num_inserted,' of ',size_seq_in
684 endif
685 endif
686
687 obs_in = next_obs_in ! essentially records position in seq_out
688
689 !setting flag to output precomputed FOs based on namelist
690 call set_fo_metadata(obs_in)
691
692 if (editing_obs) then
693 ! copy subset or reordering
694 call copy_partial_obs(obs_out, obs_in, &
695 num_copies_out, new_copy_index, &
696 num_qc_out, new_qc_index)
697
698 ! if new copies or qcs were added, set the initial data values
699 call set_new_data(obs_out, edit_copies, new_copy_index, new_copy_data, &
700 edit_qcs, new_qc_index, new_qc_data)
701 else
702 obs_out = obs_in
703 endif
704
705!#! ! see comment in subroutine code for an explanation
706!#! call change_obs(obs_out)
707
708 ! Since the stride through the observation sequence file is always
709 ! guaranteed to be in temporally-ascending order, we can use the
710 ! 'previous' observation as the starting point to search for the
711 ! correct insertion point. This speeds up the insert code a lot.
712
713 call insert_obs_in_seq(seq_out, obs_out, prev_obs_out)
714
715 prev_obs_out = obs_out ! update position in seq_in for next insert
716 num_inserted = num_inserted + 1
717
718 call get_next_obs(seq_in, obs_in, next_obs_in, is_this_last)
719
720 enddo obsloop
721
722 total_num_inserted = total_num_inserted + num_inserted
723
724 else
725 write(msgstring1,*)'no first observation in ',trim(filename_seq(i))
726 call error_handler(e_msg,'obs_sequence_tool', msgstring1)
727 endif
728
729 if (.not. print_only) then
730 print*, '-------------- Obs seq file # : ', i
731 print*, 'Number of obs in previous seq : ', size_seq_out
732 print*, 'Number of obs to be inserted : ', size_seq_in
733 print*, 'Number of obs really inserted : ', num_inserted
734 print*, '---------------------------------------------------------'
735 endif
736
737 call destroy_obs_sequence(seq_in)
738
739enddo
740
741
742write(msgstring1,*) 'Starting to process output sequence file ', trim(filename_out)
743call error_handler(e_msg,'obs_sequence_tool',msgstring1)
744
745if (.not. print_only) then
746 print*, 'Total number of obs inserted : ', total_num_inserted
747 print*, 'Actual number of obs in the new seq file :', get_num_key_range(seq_out)
748else
749 print*, 'Total number of selected obs in all files :', get_num_key_range(seq_out)
750endif
751
752call print_obs_seq(seq_out, filename_out)
753if (.not. print_only) then
754 call write_obs_seq(seq_out, filename_out)
755else
756 write(msgstring1,*) 'Output sequence file not created; print_only in namelist is .true.'
757 call error_handler(e_msg,'', msgstring1)
758endif
759
760! Time to clean up
761
762call destroy_obs_sequence(seq_out)
763call destroy_obs( obs_in )
764call destroy_obs(next_obs_in )
765call destroy_obs( obs_out)
766call destroy_obs(prev_obs_out)
767
768call error_handler(e_msg, 'obs_sequence_tool', 'Finished successfully.',source)
769call finalize_utilities()
770

772contains

786subroutine compare_metadata(seq1, seq2, fname1, fname2)
787
788!
789! This subroutine compares the metadata for two different observation
790! sequences and terminates the program if they are not conformable.
791! In order to be merged, the two observation sequences must have the same
792! number of qc values, the same number of copies ...
793!
794! change the error to use the 2 additional text lines rather than
795! making some warnings and the last one an error.
796!
797! FIXME: this routine uses several globals now from the namelist that
798! should be arguments to this routine. i'm being lazy (or expedient) here
799! but this should be fixed at some point soon.
800!
801! and now there is an additional restriction -- if editing the copies or
802! qc entries, seq1 has already been edited and only 2 needs the editing
803! applied. before they were completely symmetric.
804!
805! in this case, seq1 must be the new out sequence, and seq2 is one
806! of the input sequences
807
808type(obs_sequence_type), intent(in) :: seq1, seq2
809character(len=*), optional :: fname1, fname2
810
811integer :: num_copies1, num_qc1
812integer :: num_copies2, num_qc2
813integer :: num_copies , num_qc, i, j
814logical :: have_match1, have_match2
815character(len=metadatalength) :: str1, str2
816
817num_copies1 = get_num_copies(seq1)
818num_qc1 = get_num_qc( seq1)
819
820num_copies2 = get_num_copies(seq2)
821num_qc2 = get_num_qc( seq2)
822
823if (edit_copies) num_copies2 = copy_index_len
824if (edit_qcs ) num_qc2 = qc_index_len
825
826num_copies = num_copies2
827num_qc = num_qc2
828
829! get this ready in case we have to use it
830if (present(fname1) .and. present(fname2)) then
831 write(msgstring1,*)'Sequence files ', trim(fname1), ' and ', trim(fname2), &
832 ' are not compatible'
833else
834 msgstring1 = 'Sequence files cannot be merged because they are not compatible'
835endif
836
837if ( num_copies1 /= num_copies2 ) then
838 write(msgstring2,*)'Different numbers of data copies found: ', &
839 num_copies1, ' vs ', num_copies2
840 num_copies = -1
841else
842 write(msgstring2,*)'Number of data copies match'
843endif
844if ( num_qc1 /= num_qc2 ) then
845 write(msgstring3,*)'Different different numbers of QCs found: ', &
846 num_qc1, ' vs ', num_qc2
847 num_qc = -1
848else
849 write(msgstring3,*)'Number of qc copies match'
850endif
851if ( num_copies < 0 .or. num_qc < 0 ) then
852 call error_handler(e_err, 'obs_sequence_tool', msgstring1, &
853 source, text2=msgstring2, text3=msgstring3)
854endif
855
856! watch the code flow in this loop and the one below it.
857! the smoothest code order is to determine what the strings are,
858! and then try different things to match them. if a match is found,
859! cycle. if you get to the bottom of the loop, there is no match
860! and a single set of (fatal) error messages is called.
861copymetadata : do i=1, num_copies
862 str1 = get_copy_meta_data(seq1,i)
863
864 if (edit_copy_metadata) then
865 str2 = new_copy_metadata(i)
866 else
867 if (new_copy_index(i) > 0) then
868 str2 = get_copy_meta_data(seq2, new_copy_index(i))
869 else
870 str2 = get_copy_meta_data(seq1, i)
871 endif
872 endif
873
874 ! easy case - they match. cycle to next copy.
875 if( str1 == str2 ) then
876 write(msgstring2,*)'metadata ',trim(str1), ' in both.'
877 call error_handler(e_msg, 'obs_sequence_tool', msgstring2)
878 cycle copymetadata
879 endif
880
881 ! see if user provided a list of metadata strings that are
882 ! the same values and can be considered a match.
883 if (matching_copy_metadata) then
884 have_match1 = .false.
885 do j=1, matching_copy_limit
886 if (trim(str1) == trim(synonymous_copy_list(j))) then
887 have_match1 = .true.
888 exit
889 endif
890 enddo
891
892 have_match2 = .false.
893 do j=1, matching_copy_limit
894 if (trim(str2) == trim(synonymous_copy_list(j))) then
895 have_match2 = .true.
896 exit
897 endif
898 enddo
899
900 ! if both are true, you found both strings in the list and
901 ! it is ok to proceed.
902 if (have_match1 .and. have_match2) then
903 write(msgstring2,*)'one is: ', trim(str1)
904 write(msgstring3,*)'one is: ', trim(str2)
905 call error_handler(e_msg, 'obs_sequence_tool', &
906 'different copy metadata strings ok because both on synonymous list', &
907 text2=msgstring2, text3=msgstring3)
908 cycle copymetadata
909 endif
910
911 ! if no match, fall out of the if.
912 endif
913
914 ! if you get here, the metadata is not the same and the user has not
915 ! given us strings that are ok to match. fail.
916 write(msgstring2,*)'copy metadata mismatch, file 1: ', trim(str1)
917 write(msgstring3,*)'copy metadata mismatch, file 2: ', trim(str2)
918 call error_handler(e_err, 'obs_sequence_tool', msgstring1, &
919 source, &
920 text2=msgstring2, text3=msgstring3)
921
922enddo copymetadata
923
924qcmetadata : do i=1, num_qc
925 str1 = get_qc_meta_data(seq1,i)
926
927 if (edit_qc_metadata) then
928 str2 = new_qc_metadata(i)
929 else
930 if (new_qc_index(i) > 0) then
931 str2 = get_qc_meta_data(seq2, new_qc_index(i))
932 else
933 str2 = get_qc_meta_data(seq1, i)
934 endif
935 endif
936
937
938 ! easy case - they match. cycle to next copy.
939 if( str1 == str2 ) then
940 write(msgstring2,*)'metadata ',trim(str1), ' in both.'
941 call error_handler(e_msg, 'obs_sequence_tool', msgstring2)
942 cycle qcmetadata
943 endif
944
945 ! see if user provided a list of metadata strings that are
946 ! the same values and can be considered a match.
947 if (matching_qc_metadata) then
948 have_match1 = .false.
949 do j=1, matching_qc_limit
950 if (trim(str1) == trim(synonymous_qc_list(j))) then
951 have_match1 = .true.
952 exit
953 endif
954 enddo
955
956 have_match2 = .false.
957 do j=1, matching_qc_limit
958 if (trim(str2) == trim(synonymous_qc_list(j))) then
959 have_match2 = .true.
960 exit
961 endif
962 enddo
963
964 ! if both are true, you found both strings in the list and
965 ! it is ok to proceed.
966 if (have_match1 .and. have_match2) then
967 write(msgstring2,*)'one is: ', trim(str1)
968 write(msgstring3,*)'one is: ', trim(str2)
969 call error_handler(e_msg, 'obs_sequence_tool', &
970 'different qc metadata strings ok because both on synonymous list', &
971 text2=msgstring2, text3=msgstring3)
972 cycle qcmetadata
973 endif
974
975 ! if no match, fall out of the if.
976 endif
977
978 ! if you get here, the metadata is not the same and the user has not
979 ! given us strings that are ok to match. fail.
980 write(msgstring2,*)'qc metadata mismatch, file 1: ', trim(str1)
981 write(msgstring3,*)'qc metadata mismatch, file 2: ', trim(str2)
982 call error_handler(e_err, 'obs_sequence_tool', msgstring1, &
983 source, text2=msgstring2, text3=msgstring3)
984
985enddo qcmetadata
986
987end subroutine compare_metadata

989!---------------------------------------------------------------------
990! pass in an already opened sequence and a start/end time. this routine
991! really trims the observations out of the sequence, and returns a count
992! of how many remain.
993subroutine trim_seq(seq, trim_first, first_time, trim_last, last_time, &
994 seqfilename, print_msg, remaining_obs_count)
995 type(obs_sequence_type), intent(inout) :: seq
996 logical, intent(in) :: trim_first, trim_last
997 type(time_type), intent(in) :: first_time, last_time
998 character(len=*), intent(in) :: seqfilename
999 logical, intent(in) :: print_msg
1000 integer, intent(out) :: remaining_obs_count
1001
1002 integer :: i
1003 logical :: found
1004
1005 ! Need to find first obs with appropriate time, delete all earlier ones
1006 if(trim_first) then
1007 call delete_seq_head(first_time, seq, all_gone)
1008 if(all_gone) then
1009 if (print_msg) then
1010 msgstring1 = 'Skipping: all obs in ' // trim(seqfilename) // &
1011 ' are before first_obs time'
1012 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1013 endif
1014 remaining_obs_count = 0
1015 return
1016 endif
1017 endif
1018
1019 ! Also get rid of observations past the last_obs_time if requested
1020 if(trim_last) then
1021 call delete_seq_tail(last_time, seq, all_gone)
1022 if(all_gone) then
1023 if (print_msg) then
1024 msgstring1 = 'Skipping: all obs in ' // trim(seqfilename) // &
1025 ' are after last_obs time'
1026 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1027 endif
1028 remaining_obs_count = 0
1029 return
1030 endif
1031 endif
1032
1033 ! optionally select only a list of obs types
1034 if (restrict_by_obs_type) then
1035 call delete_obs_by_typelist(num_obs_input_types, obs_types, &
1036 keep_types, seq, all_gone)
1037 if(all_gone) then
1038 if (print_msg) then
1039 if (keep_types) then
1040 msgstring1 = 'Skipping: no obs in ' // trim(seqfilename) // &
1041 ' are on the keep obs_types list'
1042 else
1043 msgstring1 = 'Skipping: all obs in ' // trim(seqfilename) // &
1044 ' are on the discard obs_types list'
1045 endif
1046 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1047 endif
1048 remaining_obs_count = 0
1049 return
1050 endif
1051 endif
1052
1053 ! optionally select only obs inside a bounding box
1054 ! most common expected use is a lat/lon box with the 3d sphere locations
1055 ! mod, but this should work with any locations mod if the namelist uses
1056 ! the right number of corners appropriate for the dimensions of the location
1057 ! when setting min_box and max_box.
1058 if (restrict_by_location) then
1059 call select_obs_by_location(min_loc, max_loc, seq, all_gone)
1060 if(all_gone) then
1061 if (print_msg) then
1062 msgstring1 = 'Skipping: no obs in ' // trim(seqfilename) // &
1063 ' are within the given min/max location bounding box'
1064 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1065 endif
1066 remaining_obs_count = 0
1067 return
1068 endif
1069 endif
1070
1071 ! optionally select only a range of qc and copies
1072 if (restrict_by_qc) then
1073 ! validate the metadata string
1074 found = .false.
1075 do i=1, num_qc_in
1076 meta_data = get_qc_meta_data(seq_in, i)
1077 if (trim(qc_metadata) == trim(meta_data)) then
1078 call delete_obs_by_qc(i, min_qc, max_qc, seq, all_gone)
1079 found = .true.
1080 exit
1081 endif
1082 enddo
1083 if (.not. found) then
1084 msgstring1 = 'QC metadata string: ' // trim(qc_metadata) // &
1085 ' not found in obs_seq file'
1086 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1087 endif
1088 if(all_gone) then
1089 if (print_msg) then
1090 msgstring1 = 'Skipping: all obs in ' // trim(seqfilename) // &
1091 ' are outside the qc min/max range'
1092 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1093 endif
1094 remaining_obs_count = 0
1095 return
1096 endif
1097 endif
1098 if (restrict_by_copy) then
1099 ! validate the metadata string
1100 found = .false.
1101 do i=1, num_copies_in
1102 meta_data = get_copy_meta_data(seq_in, i)
1103 if (trim(copy_metadata) == trim(meta_data)) then
1104 call delete_obs_by_copy(i, min_copy, max_copy, copy_type, &
1105 seq, all_gone)
1106 found = .true.
1107 exit
1108 endif
1109 enddo
1110 if (.not. found) then
1111 msgstring1 = 'Copy metadata string: ' // trim(copy_metadata) // &
1112 ' not found in obs_seq file'
1113 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1114 endif
1115 if(all_gone) then
1116 if (print_msg) then
1117 msgstring1 = 'Skipping: all obs in ' // trim(seqfilename) // &
1118 ' are outside the copy min/max range'
1119 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1120 endif
1121 remaining_obs_count = 0
1122 return
1123 endif
1124 endif
1125
1126!%! ! SPECIAL: optionally restrict GPS obs to above a given height
1127!%! ! see comments in select_gps_by_height() for more info.
1128!%! if (restrict_by_height) then
1129!%! call select_gps_by_height(min_gps_height, seq, all_gone)
1130!%! if(all_gone) then
1131!%! if (print_msg) then
1132!%! msgstring1 = 'Skipping: no obs in ' // trim(seqfilename) // &
1133!%! ' are above the GPS height threshold'
1134!%! call error_handler(E_MSG,'obs_sequence_tool',msgstring1)
1135!%! endif
1136!%! remaining_obs_count = 0
1137!%! return
1138!%! endif
1139!%! endif
1140
1141 remaining_obs_count = get_num_key_range(seq)
1142
1143end subroutine trim_seq

1147subroutine print_obs_seq(seq_in, filename)
1148
1149! you can get more info by running the obs_diag program, but this
1150! prints out a quick table of obs types and counts, overall start and
1151! stop times, and metadata strings and counts.
1152
1153type(obs_sequence_type), intent(in) :: seq_in
1154character(len=*), intent(in) :: filename
1155
1156type(obs_type) :: obs, next_obs
1157type(obs_def_type) :: this_obs_def
1158logical :: is_there_one, is_this_last
1159integer :: size_seq_in
1160integer :: i
1161integer :: this_obs_kind
1162! max_defined_types_of_obs is a public from obs_kind_mod.f90 and really is
1163! counting the max number of types, not kinds
1164integer :: type_count(max_defined_types_of_obs), identity_count
1165
1166
1167! Initialize input obs_types
1168do i = 1, max_defined_types_of_obs
1169 type_count(i) = 0
1170enddo
1171identity_count = 0
1172
1173! make sure there are obs left to process before going on.
1174! num_obs should be ok since we just constructed this seq so it should
1175! have no unlinked obs. if it might for some reason, use this instead:
1176! size_seq_in = get_num_key_range(seq_in) !current size of seq_in
1177
1178size_seq_in = get_num_obs(seq_in)
1179if (size_seq_in == 0) then
1180 msgstring1 = 'Obs_seq file '//trim(filename)//' is empty.'
1181 call error_handler(e_msg,'obs_sequence_tool',msgstring1)
1182 return
1183endif
1184
