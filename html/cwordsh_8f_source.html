45 program cwordsh
46
47 parameter(mxbufr=150000)
48 parameter(mxbufrd4=mxbufr/4)
49
50 character*80 bfile,ufile
51 character*8 cword
52 character*1 bufr(mxbufr)
53 dimension mbay(mxbufrd4)
54 equivalence(bufr(1),mbay(1))
55
56c Read and process the input arguments.
57
58 read(5,'(a)') cword
59 if(cword.eq.'block') then
60 read(5,'(a)') ufile
61 read(5,'(a)') bfile
62 elseif(cword.eq.'unblk') then
63 read(5,'(a)') bfile
64 read(5,'(a)') ufile
65 else
66 print*,'cword must be block or unblk'
67 stop
68 endif
69
70c Open the input and output files.
71
72 if(cword.eq.'block') then
73 print*,'blocking from: ',ufile
74 print*,' to: ',bfile
75 call cobfl(ufile,'r')
76 open(51,file=bfile,form='unformatted')
77 else
78 print*,'unblocking from: ',bfile
79 print*,' to: ',ufile
80 call cobfl(bfile,'r')
81 call cobfl(ufile,'w')
82 endif
83
84c Read the next message from the input file.
85
86 call crbmg(bufr,mxbufr,nbyt,ierr)
87 do while(ierr.ge.0)
88 if(ierr.eq.0) then
89
90c Pad the end of the message with zeroed-out bytes up to the
91c next 8-byte boundary.
92
93 call padmsg(mbay,mxbufrd4,npbyt)
94 ntbyt = nbyt + npbyt
95
96c Write the message plus padding to the output file...
97
98 if(cword.eq.'block') then
99
100c using a FORTRAN write.
101
102 write(51) (bufr(i),i=1,ntbyt)
103 else
104
105c using a C write.
106
107 call cwbmg(bufr,ntbyt,ierw)
108 if(ierw.ne.0) print*,'return value from cwbmg was ',ierw
109 endif
110 else
111 print*,'return value from crbmg was ',ierr,
112 . '; message not written to output'
113 endif
114 call crbmg(bufr,mxbufr,nbyt,ierr)
115 enddo
116
117c Close the input and output files.
118
119 call ccbfl
120 if(cword.eq.'block') close(51)
121
122 stop
123 end
