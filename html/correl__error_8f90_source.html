14type (random_seq_type) :: ran_id
15real(r8) :: zero_2(2) = 0.0, cov(2, 2)

20integer i, j, k, lji, ens_size

22call init_random_seq(ran_id)
23
24! Loop through a range of ensemble sizes
25do lji = 4, 48, 4
26 ens_size = lji
27 if(lji > 32) ens_size = 32 + (lji - 32) * 4

32 !!!ens_size = 128

35 write(*, *) 'stats for ensemble size ', ens_size
36 ! Loop through real correlations every 0.05
37 !!!do j = 0, 50
38 do j = 0, 0
39 ! Loop through a large number of samples to get mean
40 correl_mean = 0.0_r8
41 correl_mean2 = 0.0_r8
42 ratio_mean = 0.0_r8
43 ratio_mean2 = 0.0_r8
44 do k = 1, sample_size
45 t_correl = j * 0.02_r8
46
47 ! Generate the covariance matrix for this correlation
48 cov(1, 1) = 1.0; cov(2, 2) = 1.0_r8
49 cov(1, 2) = t_correl; cov(2, 1) = t_correl
50
51 ! Loop to generate an ensemble size sample from this correl
52 ! Generate a random sample of size ens_size from something with this correlation
53 do i = 1, ens_size
54 call twod_gaussians(ran_id, zero_2, cov, pairs(:, i))
55 enddo
56
57 ! Compute the sample correlation
58 call comp_correl(pairs, ens_size, sample_correl)
59
60 ! Keep a sum of the sample_correl
61 correl_mean = correl_mean + abs(sample_correl)
62 correl_mean2 = correl_mean2 + sample_correl**2
63
64 !-----------------
65 ! Also interested in finding out what the spurious variance reduction factor is
66 ! First need to compute sample s.d. for the obs and unobs variable
67 do i = 1, 2
68 call sample_mean_var(pairs(i, :), ens_size, s_mean(i), s_var(i))
69 enddo
70 ! Next, compute the increments for obs variable given reduction in s.d. by a
71 a = 0.0_r8
72 do i = 1, ens_size
73 obs_inc(i) = (a - 1.0_r8) * (pairs(1, i) - s_mean(1))
74 enddo
75
76 ! Regress these onto the unobserved variable
77 do i = 1, ens_size
78 unobs_inc(i) = sample_correl * sqrt(s_var(2) / s_var(1)) * obs_inc(i)
79 new_unobs(i) = pairs(2, i) + unobs_inc(i)
80 enddo
81
82 ! Now compute the updated variance
83 call sample_mean_var(new_unobs, ens_size, new_mean, new_var)
84
85 !write(*, *) 'old and new var ', s_var(2), new_var
86 !write(*, *) 'old and new sd ', sqrt(s_var(2)), sqrt(new_var)
87 !write(*, *) 'unobs sd ratio ', sqrt(new_var) / sqrt(s_var(2))
88 ratio_mean = ratio_mean + sqrt(new_var) / sqrt(s_var(2))
89 ratio_mean2 = ratio_mean2 + (sqrt(new_var) / sqrt(s_var(2)))**2
90
91 !-----------------
92
93 enddo
94 correl_mean = correl_mean / sample_size
95 correl_sd = sqrt((correl_mean2 - sample_size * correl_mean**2) / (sample_size - 1))
96 write(*, *) t_correl, correl_mean, correl_sd
97 ratio_mean = ratio_mean / sample_size
98 ratio_sd = sqrt((ratio_mean2 - sample_size * ratio_mean**2) / (sample_size - 1))
99 write(*, *) 'ratio mean ', ratio_mean, (1 - ratio_mean) / (1.0 - a)
100 expected_ratio = 1.0 - ((1.0 - a) * t_correl)
101 write(44, 11) ens_size, t_correl, ratio_mean, ratio_sd, correl_mean, correl_sd
102 11 format(i8, 6(1x, f10.5))
103 enddo
104 deallocate(pairs, obs_inc, unobs_inc, new_unobs)
105enddo

111subroutine comp_correl(ens, n, correl)
112
113implicit none
114
115integer, intent(in) :: n
116real(r8), intent(in) :: ens(2, n)
117real(r8), intent(out) :: correl
118real(r8) :: sum_x, sum_y, sum_xy, sum_x2, sum_y2
119
120
121sum_x = sum(ens(2, :))
122sum_y = sum(ens(1, :))
123sum_xy = sum(ens(2, :) * ens(1, :))
124sum_x2 = sum(ens(2, :) * ens(2, :))
125
126! Computation of correlation
127sum_y2 = sum(ens(1, :) * ens(1, :))
128
129correl = (n * sum_xy - sum_x * sum_y) / &
130 sqrt((n * sum_x2 - sum_x**2) * (n * sum_y2 - sum_y**2))
131
132end subroutine comp_correl

149var = (s_x2 - sx**2 / n) / (n - 1)
