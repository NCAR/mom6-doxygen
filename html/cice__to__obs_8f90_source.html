16! ;============data description from CC matlab script=================================
17! ; Data are stored as flat two-byte integers representing sea ice concentration
18! ; values. The sea ice concentration data values are packed into integer
19! ; format by multiplying the original sea ice concentration values by
20! ; 10. These values range from 0 to 1,000, with land registered as -800 and
21! ; -100 being the northern hemisphere hole poleward of 85 degrees for the
22! ; SMMR data and poleward of 87 degrees for the SSM/I data, which the
23! ; satellite cannot cover due to orbit inclination.
24! ;
25! ; (nsc -- what are these? ...)
26! ; little-endian byte order no mention of header
27! ; One-byte (scaled, unsigned integer) flat binary arrays preceded by a 300-byte header;
28! ; *uint8
29! ;=====================================================================================

458!> take a character string which must contain YYYY, MM, and DD
459!> and substitute the year, month and day. it is an error to
460!> have a pattern without all 3 items.
461
462subroutine fix_filename(inpattern, outstring, year, month, day)
463
464character(len=*), intent(in) :: inpattern
465character(len=*), intent(out) :: outstring
466integer, intent(in) :: year
467integer, intent(in) :: month
468integer, intent(in) :: day
469
470character(len=32) :: buf
471
472outstring = inpattern
473
474! substitute year for YYYY, month for MM, day for DD
475!>@todo add error handling
476
477start_index = index(outstring, 'YYYY')
478if (start_index < 0) then
479 write(msgstring,*)'String "YYYY" was not found in pattern "'//trim(inpattern)//'"'
480 write(msgstring1, *)'Trying to replace YYYY with year ', year
481 call error_handler(e_err, 'cice_to_obs: ', msgstring, source, revision, revdate, &
482 text2=msgstring1)
483endif
484write(buf, '(I4.4)') year
485outstring(start_index:start_index+3) = buf(1:4)
486
487start_index = index(outstring, 'MM')
488if (start_index < 0) then
489 write(msgstring,*)'String "MM" was not found in pattern "'//trim(inpattern)//'"'
490 write(msgstring1, *)'Trying to replace MM with month ', month
491 call error_handler(e_err, 'cice_to_obs: ', msgstring, source, revision, revdate, &
492 text2=msgstring1)
493endif
494write(buf, '(I2.2)') month
495outstring(start_index:start_index+1) = buf(1:2)
496
497start_index = index(outstring, 'DD')
498if (start_index < 0) then
499 write(msgstring,*)'String "DD" was not found in pattern "'//trim(inpattern)//'"'
500 write(msgstring1, *)'Trying to replace DD with day ', day
501 call error_handler(e_err, 'cice_to_obs: ', msgstring, source, revision, revdate, &
502 text2=msgstring1)
503endif
504write(buf, '(I2.2)') day
505outstring(start_index:start_index+1) = buf(1:2)
506
507end subroutine fix_filename
508
509end program cice_to_obs
