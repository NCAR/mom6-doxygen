62!-----------------------------------------------------------------------------
63! Namelist with default values
64!
65character(len = 129) :: var_file = 'qc_radr_3dvar_2002083100.dat', &
66 obs_seq_out_file_name = 'obs_seq.out'
67integer :: calendar_type = gregorian

69namelist /rad_3dvar_to_dart_nml/ var_file, obs_seq_out_file_name, calendar_type

72
73call initialize_utilities('rad_3dvar_to_dart')
74call register_module(source, revision, revdate)

76! Read the namelist entry
77call find_namelist_in_file("input.nml", "rad_3dvar_to_dart_nml", iunit)
78read(iunit, nml = rad_3dvar_to_dart_nml, iostat = io)
79call check_namelist_read(iunit, io, "rad_3dvar_to_dart_nml")

81! Record the namelist values used for the run ...
82write(nmlfileunit, nml=rad_3dvar_to_dart_nml)
83write( * , nml=rad_3dvar_to_dart_nml)
84
85call set_calendar_type(calendar_type)

87iunit = open_file(var_file, action = 'read')

89! -------------------------------------------------------------------
90! Initialize the counters:
91
92num_radar = 0
93num_obs = 0
94key = 0
95
96!-----------------------------------------------------------------------------!
97! Read the header of a MM5 3D-VAR 2.0 Radar observation file
98!-----------------------------------------------------------------------------!
99
100READ (unit = iunit, iostat = iost, &
101 fmt = '(A19,F8.3,2X,F8.3,F10.1,2X,A19,2I6)' ) &
102 radar_name, radar_long, radar_lat, radar_elev, &
103 start_scan_date, &
104 num_prof, max_levels

107READ (unit = iunit, iostat = iost, fmt = '(A80)' ) dummy

113num_qc = 1
114

116call init_obs_sequence(seq, num_copies, num_qc, max_num_obs)
117

119call set_qc_meta_data(seq, 1, 'missing')
120
121call init_obs(obs, num_copies, num_qc)
122
123! READ FORMATS
124! ------------
125
126! LOOP OVER RECORDS
127! -----------------
128
129reports: &
130 DO
131! READ STATION GENERAL INFO
132! =============================
133
134READ (unit = iunit, iostat = iost, &
135 fmt = '(A12,3X,I4,5(A1,I2),2X,2(F12.3,2X),F8.1,2X,I6)' ) &
136 platform_name, &
137 year, dummy, month, dummy, day, dummy, hours, dummy, minutes, dummy, seconds, &
138 lat, &
139 lon, &
140 elv, &
141 levels
142
143time = set_date(year, month, day, hours, minutes, seconds)
144
145IF (iost /= 0) THEN
146 WRITE (*,'(/,A,I3,/)') ' END OF UNIT: ',iunit
147 WRITE (*,'(A,I3)') ' IOSTAT == ',iost
148 EXIT reports
149ENDIF
150
151! READ EACH LEVELS
152! ----------------
153
154loop_level: DO ii = 1, levels
155
156 READ (unit = iunit, fmt = '( 3X, F12.1, 2(F12.3,I4,F12.3,2X) )' ) &
157 height, &
158 rv_inv, &
159 rv_qc, &
160 rv_error, &
161 rf_inv, &
162 rf_qc, &
163 rf_error
164
165 if (rv_inv /= missing_r8) then
166
167 num_obs = num_obs + 1
168 key = key + 1
169
170 location = set_location(lon, lat, height, 3)
171 call set_obs_def_location(obs_def, location)
172
173 location = set_location(radar_long, radar_lat, radar_elev, 3)
174 call set_obs_def_key(obs_def, key)
175
176 obs_lat = lat*deg2rad
177 rad_lat = radar_lat*deg2rad
178 obs_lon = lon*deg2rad
179 rad_lon = radar_long*deg2rad
180
181 ae = 1000.0_r8 * earth_radius
182 x = ae * cos((obs_lat + rad_lat)/2.0_r8) * (obs_lon - rad_lon)
183 y = ae * (obs_lat - rad_lat)
184 raz = atan(x/y)
185 spath = sqrt(x*x + y*y)
186 h = height - radar_elev
187
188 ae = 4000.0_r8 * earth_radius / 3.0_r8
189 rgate = spath
190
191 do it=1,10
192 elev_rad = asin((h*h + 2.0_r8*ae*h - rgate*rgate)/(2.0_r8*ae*rgate))
193 rgate = (ae+h)*sin(spath/ae)/cos(elev_rad)
194 enddo
195
196 elev_obs = sqrt(1.5_r8 * h / (1000.0_r8 * earth_radius) + elev_rad*elev_rad)
197
198 orientation(1) = sin(raz)*cos(elev_obs)
199 orientation(2) = cos(raz)*cos(elev_obs)
200 orientation(3) = sin(elev_obs)
201
202 nyquist_velocity = missing_r8
203
204 call set_radial_vel(key, location, orientation, nyquist_velocity)
205
206 call set_obs_def_time(obs_def, time)
207
208 call set_obs_def_type_of_obs(obs_def, doppler_radial_velocity)
209
210 if (rv_error /= missing_r8) then
211 call set_obs_def_error_variance(obs_def, rv_error*rv_error)
212 else
213 call set_obs_def_error_variance(obs_def, 2.0_r8)
214 endif
215
216 call set_obs_def(obs, obs_def)
217
218 obs_value(1) = rv_inv
219 call set_obs_values(obs, obs_value, 1)
220
221 if (rv_inv == missing_r8 .or. &
222 rv_error == missing_r8 ) then
223
224 rv_qc = missing_data
225
226 end if
227
228 rstatus(1,1) = rv_qc
229 call set_qc(obs, rstatus(1,:), 1)
230
231 if(num_obs == 1) then
232 call insert_obs_in_seq(seq, obs)
233 else
234 call insert_obs_in_seq(seq, obs, prev_obs)
235 endif
236
237 prev_obs = obs
238
239 endif
240
241 if (rf_inv /= missing_r8) then
242
243 num_obs = num_obs + 1
244
245 call set_obs_def_type_of_obs(obs_def, radar_reflectivity)
246
247 if (rv_error /= missing_r8) then
248 call set_obs_def_error_variance(obs_def, rf_error*rf_error)
249 else
250 call set_obs_def_error_variance(obs_def, 2.0_r8)
251 endif
252
253 call set_obs_def(obs, obs_def)
254
255 obs_value(1) = rf_inv
256 call set_obs_values(obs, obs_value, 1)
257
258 if (rf_inv == missing_r8 .or. &
259 rf_error == missing_r8 ) then
260
261 rf_qc = missing_data
262
263 end if
264
265 rstatus(1,1) = rf_qc
266 call set_qc(obs, rstatus(1,:), 1)
267
268 if(num_obs == 1) then
269 call insert_obs_in_seq(seq, obs)
270 else
271 call insert_obs_in_seq(seq, obs, prev_obs)
272 endif
273
274 prev_obs = obs
275
276 endif
277
278ENDDO loop_level
279
280num_radar = num_radar + 1
281
282ENDDO reports
283
284call close_file(iunit)
285
286write(unit=*, fmt='(5x,a,i6,a)') 'Read: ', num_radar, ' Radar reports,'
287
288call write_obs_seq(seq, obs_seq_out_file_name)
289
290call error_handler(e_msg, 'rad_3dvar_to_dart', 'FINISHED rad_3dvar_to_dart.')
291call error_handler(e_msg, 'rad_3dvar_to_dart', 'Finished successfully.', &
292 source,revision,revdate)
293call finalize_utilities()
294
295
296END PROGRAM rad_3dvar_to_dart
