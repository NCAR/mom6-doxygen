242subroutine assim_tools_init()
243
244integer :: iunit, io, i, j
245integer :: num_special_cutoff, type_index
246logical :: cache_override = .false.
247
248
249! do this up front
250module_initialized = .true.
251
252! give these guys initial values at runtime *before* we read
253! in the namelist. this is to help detect how many items are
254! actually given in the namelist.
255special_localization_obs_types(:) = 'null'
256special_localization_cutoffs(:) = missing_r8
257
258! Read the namelist entry
259call find_namelist_in_file("input.nml", "assim_tools_nml", iunit)
260read(iunit, nml = assim_tools_nml, iostat = io)
261call check_namelist_read(iunit, io, "assim_tools_nml")
262
263! Write the namelist values to the log file
264if (do_nml_file()) write(nmlfileunit, nml=assim_tools_nml)
265if (do_nml_term()) write( * , nml=assim_tools_nml)
266
267! Forcing distributed_mean for single processor.
268! Note null_win_mod.f90 ignores distibute_mean.
269if (task_count() == 1) distribute_mean = .true.
270
271! FOR NOW, can only do spread restoration with filter option 1 (need to extend this)
272if(spread_restoration .and. .not. filter_kind == 1) then
273 write(msgstring, *) 'cannot combine spread_restoration and filter_kind ', filter_kind
274 call error_handler(e_err,'assim_tools_init:', msgstring, source)
275endif
276
277! allocate a list in all cases - even the ones where there is only
278! a single cutoff value. note that in spite of the name these
279! are specific types (e.g. RADIOSONDE_TEMPERATURE, AIRCRAFT_TEMPERATURE)
280! because that's what get_close() is passed. and because i've confused
281! myself several times -- we define generic kinds starting at 0, but
282! the specific types are autogenerated and always start at 1. so the
283! cutoff list is never (0:num_types); it is always (num_types).
284num_types = get_num_types_of_obs()
285allocate(cutoff_list(num_types))
286cutoff_list(:) = cutoff
287has_special_cutoffs = .false.
288
289! Go through special-treatment observation kinds, if any.
290num_special_cutoff = 0
291j = 0
292do i = 1, max_items
293 if(special_localization_obs_types(i) == 'null') exit
294 if(special_localization_cutoffs(i) == missing_r8) then
295 write(msgstring, *) 'cutoff value', i, ' is uninitialized.'
296 call error_handler(e_err,'assim_tools_init:', &
297 'special cutoff namelist for types and distances do not match', &
298 source, &
299 text2='kind = '//trim(special_localization_obs_types(i)), &
300 text3=trim(msgstring))
301 endif
302 j = j + 1
303enddo
304num_special_cutoff = j
305
306if (num_special_cutoff > 0) has_special_cutoffs = .true.
307
308do i = 1, num_special_cutoff
309 type_index = get_index_for_type_of_obs(special_localization_obs_types(i))
310 if (type_index < 0) then
311 write(msgstring, *) 'unrecognized TYPE_ in the special localization namelist:'
312 call error_handler(e_err,'assim_tools_init:', msgstring, source, &
313 text2=trim(special_localization_obs_types(i)))
314 endif
315 cutoff_list(type_index) = special_localization_cutoffs(i)
316end do
317
318! cannot cache previous obs location if different obs types have different
319! localization radii. change it to false, and warn user why.
320if (has_special_cutoffs .and. close_obs_caching) then
321 cache_override = .true.
322 close_obs_caching = .false.
323endif
324
325if(sampling_error_correction) then
326 sec_table_size = get_sampling_error_table_size()
327 allocate(exp_true_correl(sec_table_size), alpha(sec_table_size))
328 ! we can't read the table here because we don't have access to the ens_size
329endif
330
331is_doing_vertical_conversion = (has_vertical_choice() .and. vertical_localization_on())
332
333call log_namelist_selections(num_special_cutoff, cache_override)
334
335end subroutine assim_tools_init

3293subroutine get_correction_from_table(scorrel, mean_factor, expected_true_correl, ens_size)
3294
3295real(r8), intent(in) :: scorrel
3296real(r8), intent(out) :: mean_factor, expected_true_correl
3297integer, intent(in) :: ens_size
3298
3299! Uses interpolation to get correction factor into the table
3300
3301integer :: low_indx, high_indx
3302real(r8) :: correl, fract, low_correl, low_exp_correl, low_alpha
3303real(r8) :: high_correl, high_exp_correl, high_alpha
3304
3305logical, save :: first_time = .true.
3306
3307if (first_time) then
3308 call read_sampling_error_correction(ens_size, exp_true_correl, alpha)
3309 first_time = .false.
3310endif
3311
3312! Interpolate to get values of expected correlation and mean_factor
3313if(scorrel < -1.0_r8) then
3314 correl = -1.0_r8
3315 mean_factor = 1.0_r8
3316else if(scorrel > 1.0_r8) then
3317 correl = 1.0_r8
3318 mean_factor = 1.0_r8
3319else if(scorrel <= -0.995_r8) then
3320 fract = (scorrel + 1.0_r8) / 0.005_r8
3321 correl = (exp_true_correl(1) + 1.0_r8) * fract - 1.0_r8
3322 mean_factor = (alpha(1) - 1.0_r8) * fract + 1.0_r8
3323else if(scorrel >= 0.995_r8) then
3324 fract = (scorrel - 0.995_r8) / 0.005_r8
3325 correl = (1.0_r8 - exp_true_correl(sec_table_size)) * fract + exp_true_correl(sec_table_size)
3326 mean_factor = (1.0_r8 - alpha(sec_table_size)) * fract + alpha(sec_table_size)
3327else
3328 ! given the ifs above, the floor() computation below for low_indx
3329 ! should always result in a value in the range 1 to 199. but if this
3330 ! code is compiled with r8=r4 (single precision reals) it turns out
3331 ! to be possible to get values a few bits below 0 which results in
3332 ! a very large negative integer. the limit tests below ensure the
3333 ! index stays in a legal range.
3334 low_indx = floor((scorrel + 0.995_r8) / 0.01_r8 + 1.0_r8)

3347endif

3351! Don't want Monte Carlo interpolation problems to put us outside of a

3354if(expected_true_correl * scorrel <= 0.0_r8) then
3355 expected_true_correl = 0.0_r8
3356else if(abs(expected_true_correl) > abs(scorrel)) then
3357 ! If same sign, expected should not be bigger in absolute value
3358 expected_true_correl = scorrel
3359endif

3361end subroutine get_correction_from_table

3365subroutine obs_increment_boxcar(ens, ens_size, obs, obs_var, obs_inc, rel_weight)
3366!------------------------------------------------------------------------
3367!
3368! An observation space update that uses a set of boxcar kernels plus two
3369! half-gaussians on the wings to represent the prior distribution. If N is
3370! the ensemble size, 1/(N+1) of the mass is placed between each ensemble
3371! member. This is reminiscent of the ranked historgram approach for
3372! evaluating ensembles. The prior distribution on the wings is
3373! represented by a half gaussian with mean being the outermost ensemble
3374! member (left or right) and variance being somewhat arbitrarily chosen
3375! as half the total ensemble sample variance. A particle
3376! filter like algorithm is then used for the update. The weight associated
3377! with each prior ensemble member is computed by evaluating the likelihood.
3378! For the interior, the domain for each boxcar is divided in half and each
3379! half is associated with the nearest ensemble member. The updated mass in
3380! each half box is the product of the prior mass and the ensemble weight.
3381! In the wings, the observation likelihood gaussian is convolved with the
3382! prior gaussian to get an updated weighted gaussian that is assumed to
3383! represent the posterior outside of the outermost ensemble members. The
3384! updated ensemble members are chosen so that 1/(N+1) of the updated
3385! mass is between each member and also on the left and right wings. This
3386! algorithm is able to deal well with outliers, bimodality and other
3387! non-gaussian behavior in observation space. It could also be modified to
3388! deal with non-gaussian likelihoods in the future.
3389
3390integer, intent(in) :: ens_size
3391real(r8), intent(in) :: ens(ens_size), obs, obs_var
3392real(r8), intent(out) :: obs_inc(ens_size)
3393real(r8), intent(out) :: rel_weight(ens_size)
3394
3395integer :: i, e_ind(ens_size), lowest_box, j
3396real(r8) :: sx, prior_mean, prior_var, prior_var_d2
3397real(r8) :: var_ratio, new_var, new_sd, umass, left_weight, right_weight
3398real(r8) :: mass(2*ens_size), weight(ens_size), cumul_mass(0:2*ens_size)
3399real(r8) :: new_mean_left, new_mean_right, prod_weight_left, prod_weight_right
3400real(r8) :: new_ens(ens_size), mass_sum, const_term
3401real(r8) :: x(1:2*ens_size - 1), sort_inc(ens_size)
3402
3403! The factor a is not defined for this filter for now (could it be???)
3404
3405! The relative weights could be used for a multi-dimensional particle-type
3406! update using update_ens_from_weights. There are algorithmic challenges
3407! with outliers so this is not currently a supported option. For now,
3408! rel_weight is simply set to 0 and is unused elsewhere.
3409rel_weight = 0.0_r8
3410
3411! Do an index sort of the ensemble members; Need sorted ensemble
3412call index_sort(ens, e_ind, ens_size)
3413
3414! Prior distribution is boxcar in the central bins with 1/(n+1) density
3415! in each intermediate bin. BUT, distribution on the wings is a normal with
3416! 1/(n + 1) of the mass on each side.
3417
3418! Begin by computing a weight for each of the prior ensemble membersA
3419! This is just evaluating the gaussian likelihood
3420const_term = 1.0_r8 / (sqrt(2.0_r8 * pi) * sqrt(obs_var))
3421do i = 1, ens_size
3422 weight(i) = const_term * exp(-1.0_r8 * (ens(i) - obs)**2 / (2.0_r8 * obs_var))
3423end do
3424
3425! Compute the points that bound all the updated mass boxes; start with ensemble
3426do i = 1, ens_size
3427 x(2*i - 1) = ens(e_ind(i))
3428end do
3429! Compute the mid-point interior boundaries; these are halfway between ensembles
3430do i = 2, 2*ens_size - 2, 2
3431 x(i) = (x(i - 1) + x(i + 1)) / 2.0_r8
3432end do
3433
3434! Compute the s.d. of the ensemble for getting the gaussian wings
3435sx = sum(ens)
3436prior_mean = sx / ens_size
3437prior_var = sum((ens - prior_mean)**2) / (ens_size - 1)
3438
3439! Need to normalize the wings so they have 1/(ens_size + 1) mass outside
3440! Since 1/2 of a normal is outside, need to multiply by 2 / (ens_size + 1)
3441
3442! Need some sort of width for the boundary kernel, try 1/2 the VAR for now
3443prior_var_d2 = prior_var / 2.0_r8
3444
3445! Compute the product of the obs error gaussian with the prior gaussian (EAKF)
3446! Left wing first
3447var_ratio = obs_var / (prior_var_d2 + obs_var)
3448new_var = var_ratio * prior_var_d2
3449new_sd = sqrt(new_var)
3450new_mean_left = var_ratio * (ens(e_ind(1)) + prior_var_d2*obs / obs_var)
3451new_mean_right = var_ratio * (ens(e_ind(ens_size)) + prior_var_d2*obs / obs_var)
3452! REMEMBER, this product has an associated weight which must be taken into account
3453! See Anderson and Anderson for this weight term (or tutorial kernel filter)
3454prod_weight_left = 2.71828_r8 ** (-0.5_r8 * (ens(e_ind(1))**2 / prior_var_d2 + &
3455 obs**2 / obs_var - new_mean_left**2 / new_var)) / sqrt(2.0_r8 * pi)
3456
3457prod_weight_right = 2.71828_r8 ** (-0.5_r8 * (ens(e_ind(ens_size))**2 / prior_var_d2 + &
3458 obs**2 / obs_var - new_mean_right**2 / new_var)) / sqrt(2.0_r8 * pi)
3459
3460! Split into 2*ens_size domains; mass in each is computed
3461! Start by computing mass in the outermost (gaussian) regions
3462mass(1) = norm_cdf(ens(e_ind(1)), new_mean_left, new_sd) * &
3463 prod_weight_left * (2.0_r8 / (ens_size + 1.0_r8))
3464mass(2*ens_size) = (1.0_r8 - norm_cdf(ens(e_ind(ens_size)), new_mean_right, &
3465 new_sd)) * prod_weight_right * (2.0_r8 / (ens_size + 1.0_r8))
3466
3467! Compute mass in the inner half boxes that have ensemble point on the left
3468do i = 2, 2*ens_size - 2, 2
3469 mass(i) = (1.0_r8 / (2.0_r8 * (ens_size + 1.0_r8))) * weight(e_ind(i/2))
3470end do
3471
3472! Now right inner half boxes
3473do i = 3, 2*ens_size - 1, 2
3474 mass(i) = (1.0_r8 / (2.0_r8 * (ens_size + 1.0_r8))) * weight(e_ind(i/2 + 1))
3475end do
3476
3477! Now normalize the mass in the different bins
3478mass_sum = sum(mass)
3479mass = mass / mass_sum
3480
3481! Find cumulative mass at each box boundary and middle boundary
3482cumul_mass(0) = 0.0_r8
3483do i = 1, 2*ens_size
3484 cumul_mass(i) = cumul_mass(i - 1) + mass(i)
3485end do
3486
3487! Get resampled ensemble, Need 1/(ens_size + 1) between each
3488umass = 1.0_r8 / (ens_size + 1.0_r8)
3489
3490! Begin search at bottom of lowest box, but then update for efficiency
3491lowest_box = 1
3492
3493! Find each new ensemble members location
3494do i = 1, ens_size
3495 ! If it's in the inner or outer range have to use normal
3496 if(umass < cumul_mass(1)) then
3497 ! In the first normal box
3498 left_weight = (1.0_r8 / mass_sum) * prod_weight_left * (2.0_r8 / (ens_size + 1.0_r8))
3499 call weighted_norm_inv(left_weight, new_mean_left, new_sd, umass, new_ens(i))
3500 else if(umass > cumul_mass(2*ens_size - 1)) then
3501 ! In the last normal box; Come in from the outside
3502 right_weight = (1.0_r8 / mass_sum) * prod_weight_right * (2.0_r8 / (ens_size + 1.0_r8))
3503 call weighted_norm_inv(right_weight, new_mean_right, new_sd, 1.0_r8 - umass, new_ens(i))
3504 new_ens(i) = new_mean_right + (new_mean_right - new_ens(i))
3505 else
3506 ! In one of the inner uniform boxes.
3507 find_box:do j = lowest_box, 2 * ens_size - 2
3508 ! Find the box that this mass is in
3509 if(umass >= cumul_mass(j) .and. umass <= cumul_mass(j + 1)) then
3510 new_ens(i) = x(j) + ((umass - cumul_mass(j)) / (cumul_mass(j+1) - cumul_mass(j))) * &
3511 (x(j + 1) - x(j))
3512 ! Don't need to search lower boxes again
3513 lowest_box = j
3514 exit find_box
3515 end if
3516 end do find_box
3517 endif
3518 ! Want equally partitioned mass in update with exception that outermost boxes have half
3519 umass = umass + 1.0_r8 / (ens_size + 1.0_r8)
3520end do
3521
3522! Can now compute sorted increments
3523do i = 1, ens_size
3524 sort_inc(i) = new_ens(i) - ens(e_ind(i))
3525end do
3526
3527! Now, need to convert to increments for unsorted
3528do i = 1, ens_size
3529 obs_inc(e_ind(i)) = sort_inc(i)
3530end do
3531
3532end subroutine obs_increment_boxcar

3536subroutine obs_increment_rank_histogram(ens, ens_size, prior_var, &
3537 obs, obs_var, obs_inc)
3538!------------------------------------------------------------------------
3539!
3540! Revised 14 November 2008
3541!
3542! Does observation space update by approximating the prior distribution by
3543! a rank histogram. Prior and posterior are assumed to have 1/(n+1) probability
3544! mass between each ensemble member. The tails are assumed to be gaussian with
3545! a variance equal to sample variance of the entire ensemble and a mean
3546! selected so that 1/(n+1) of the mass is in each tail.
3547!
3548! The likelihood between the extreme ensemble members is approximated by
3549! quadrature. Two options are available and controlled by the namelist entry
3550! rectangular_quadrature. If this namelist is true than the likelihood between
3551! a pair of ensemble members is assumed to be uniform with the average of
3552! the likelihood computed at the two ensemble members. If it is false then
3553! the likelihood between two ensemble members is approximated by a line
3554! connecting the values of the likelihood computed at each of the ensemble
3555! members (trapezoidal quadrature).
3556!
3557! Two options are available for approximating the likelihood on the tails.
3558! If gaussian_likelihood_tails is true that the likelihood is assumed to
3559! be N(obs, obs_var) on the tails. If this is false, then the likelihood
3560! on the tails is taken to be uniform (to infinity) with the value at the
3561! outermost ensemble members.
3562!
3563! A product of the approximate prior and approximate posterior is taken
3564! and new ensemble members are located so that 1/(n+1) of the mass is between
3565! each member and on the tails.
3566
3567! This code is still under development. Please contact Jeff Anderson at
3568! jla@ucar.edu if you are interested in trying it.
3569
3570integer, intent(in) :: ens_size
3571real(r8), intent(in) :: ens(ens_size), prior_var, obs, obs_var
3572real(r8), intent(out) :: obs_inc(ens_size)
3573
3574integer :: i, e_ind(ens_size), lowest_box, j
3575real(r8) :: prior_sd, var_ratio, umass, left_amp, right_amp
3576real(r8) :: left_sd, left_var, right_sd, right_var, left_mean, right_mean
3577real(r8) :: mass(ens_size + 1), like(ens_size), cumul_mass(0:ens_size + 1)
3578real(r8) :: nmass(ens_size + 1)
3579real(r8) :: new_mean_left, new_mean_right, prod_weight_left, prod_weight_right
3580real(r8) :: new_var_left, new_var_right, new_sd_left, new_sd_right
3581real(r8) :: new_ens(ens_size), mass_sum
3582real(r8) :: x(ens_size)
3583real(r8) :: like_dense(2:ens_size), height(2:ens_size)
3584real(r8) :: dist_for_unit_sd
3585real(r8) :: a, b, c, hright, hleft, r1, r2, adj_r1, adj_r2
3586
3587! Do an index sort of the ensemble members; Will want to do this very efficiently
3588call index_sort(ens, e_ind, ens_size)
3589
3590do i = 1, ens_size
3591 ! The boundaries of the interior bins are just the sorted ensemble members
3592 x(i) = ens(e_ind(i))
3593 ! Compute likelihood for each ensemble member; just evaluate the gaussian
3594 ! No need to compute the constant term since relative likelihood is what matters
3595 like(i) = exp(-1.0_r8 * (x(i) - obs)**2 / (2.0_r8 * obs_var))
3596end do
3597
3598! Prior distribution is boxcar in the central bins with 1/(n+1) density
3599! in each intermediate bin. BUT, distribution on the tails is a normal with
3600! 1/(n + 1) of the mass on each side.
3601
3602! Can now compute the mean likelihood density in each interior bin
3603do i = 2, ens_size
3604 like_dense(i) = ((like(i - 1) + like(i)) / 2.0_r8)
3605end do
3606
3607! Compute the s.d. of the ensemble for getting the gaussian tails
3608prior_sd = sqrt(prior_var)
3609
3610! For unit normal, find distance from mean to where cdf is 1/(n+1)
3611! Lots of this can be done once in first call and then saved
3612call weighted_norm_inv(1.0_r8, 0.0_r8, 1.0_r8, &
3613 1.0_r8 / (ens_size + 1.0_r8), dist_for_unit_sd)
3614dist_for_unit_sd = -1.0_r8 * dist_for_unit_sd
3615
3616! Have variance of tails just be sample prior variance
3617! Mean is adjusted so that 1/(n+1) is outside
3618left_mean = x(1) + dist_for_unit_sd * prior_sd
3619left_var = prior_var
3620left_sd = prior_sd
3621! Same for right tail
3622right_mean = x(ens_size) - dist_for_unit_sd * prior_sd
3623right_var = prior_var
3624right_sd = prior_sd
3625
3626if(gaussian_likelihood_tails) then
3627 !*************** Block to do Gaussian-Gaussian on tail **************
3628 ! Compute the product of the obs likelihood gaussian with the priors
3629 ! Left tail gaussian first
3630 var_ratio = obs_var / (left_var + obs_var)
3631 new_var_left = var_ratio * left_var
3632 new_sd_left = sqrt(new_var_left)
3633 new_mean_left = var_ratio * (left_mean + left_var*obs / obs_var)
3634 ! REMEMBER, this product has an associated weight which must be taken into account
3635 ! See Anderson and Anderson for this weight term (or tutorial kernel filter)
3636 ! NOTE: The constant term has been left off the likelihood so we don't have
3637 ! to divide by sqrt(2 PI) in this expression
3638 prod_weight_left = exp(-0.5_r8 * (left_mean**2 / left_var + &
3639 obs**2 / obs_var - new_mean_left**2 / new_var_left)) / &
3640 sqrt(left_var + obs_var)
3641 ! Determine how much mass is in the updated tails by computing gaussian cdf
3642 mass(1) = norm_cdf(x(1), new_mean_left, new_sd_left) * prod_weight_left
3643
3644 ! Same for the right tail
3645 var_ratio = obs_var / (right_var + obs_var)
3646 new_var_right = var_ratio * right_var
3647 new_sd_right = sqrt(new_var_right)
3648 new_mean_right = var_ratio * (right_mean + right_var*obs / obs_var)
3649 ! NOTE: The constant term has been left off the likelihood so we don't have
3650 ! to divide by sqrt(2 PI) in this expression
3651 prod_weight_right = exp(-0.5_r8 * (right_mean**2 / right_var + &
3652 obs**2 / obs_var - new_mean_right**2 / new_var_right)) / &
3653 sqrt(right_var + obs_var)
3654 ! Determine how much mass is in the updated tails by computing gaussian cdf
3655 mass(ens_size + 1) = (1.0_r8 - norm_cdf(x(ens_size), new_mean_right, &
3656 new_sd_right)) * prod_weight_right
3657 !************ End Block to do Gaussian-Gaussian on tail **************
3658else
3659 !*************** Block to do flat tail for likelihood ****************
3660 ! Flat tails: THIS REMOVES ASSUMPTIONS ABOUT LIKELIHOOD AND CUTS COST
3661 new_var_left = left_var
3662 new_sd_left = left_sd
3663 new_mean_left = left_mean
3664 prod_weight_left = like(1)
3665 mass(1) = like(1) / (ens_size + 1.0_r8)
3666
3667 ! Same for right tail
3668 new_var_right = right_var
3669 new_sd_right = right_sd
3670 new_mean_right = right_mean
3671 prod_weight_right = like(ens_size)
3672 mass(ens_size + 1) = like(ens_size) / (ens_size + 1.0_r8)
3673 !*************** End block to do flat tail for likelihood ****************
3674endif
3675
3676! The mass in each interior box is the height times the width
3677! The height of the likelihood is like_dense
3678! For the prior, mass is 1/(n+1), and mass = height x width so...
3679! The height of the prior is 1 / ((n+1) width); multiplying by width leaves 1/(n+1)
3680
3681! In prior, have 1/(n+1) mass in each bin, multiply by mean likelihood density
3682! to get approximate mass in updated bin
3683do i = 2, ens_size
3684 mass(i) = like_dense(i) / (ens_size + 1.0_r8)
3685 ! Height of prior in this bin is mass/width; Only needed for trapezoidal
3686 ! If two ensemble members are the same, set height to -1 as flag
3687 if(x(i) == x(i - 1)) then
3688 height(i) = -1.0_r8
3689 else
3690 height(i) = 1.0_r8 / ((ens_size + 1.0_r8) * (x(i) - x(i-1)))
3691 endif
3692end do
3693
3694! Now normalize the mass in the different bins to get a pdf
3695mass_sum = sum(mass)
3696nmass = mass / mass_sum
3697
3698! Get the weight for the final normalized tail gaussians
3699! This is the same as left_amp=(ens_size + 1)*nmass(1)
3700left_amp = prod_weight_left / mass_sum
3701! This is the same as right_amp=(ens_size + 1)*nmass(ens_size + 1)
3702right_amp = prod_weight_right / mass_sum
3703
3704! Find cumulative mass at each box boundary and middle boundary
3705cumul_mass(0) = 0.0_r8
3706do i = 1, ens_size + 1
3707 cumul_mass(i) = cumul_mass(i - 1) + nmass(i)
3708end do
3709
3710! Begin intenal box search at bottom of lowest box, update for efficiency
3711lowest_box = 1
3712
3713! Find each new ensemble members location
3714do i = 1, ens_size
3715 ! Each update ensemble member has 1/(n+1) mass before it
3716 umass = (1.0_r8 * i) / (ens_size + 1.0_r8)
3717
3718 ! If it is in the inner or outer range have to use normal
3719 if(umass < cumul_mass(1)) then
3720 ! It's in the left tail
3721 ! Get position of x in weighted gaussian where the cdf has value umass
3722 call weighted_norm_inv(left_amp, new_mean_left, new_sd_left, &
3723 umass, new_ens(i))
3724 else if(umass > cumul_mass(ens_size)) then
3725 ! It's in the right tail
3726 ! Get position of x in weighted gaussian where the cdf has value umass
3727 call weighted_norm_inv(right_amp, new_mean_right, new_sd_right, &
3728 1.0_r8 - umass, new_ens(i))
3729 ! Coming in from the right, use symmetry after pretending its on left
3730 new_ens(i) = new_mean_right + (new_mean_right - new_ens(i))
3731 else
3732 ! In one of the inner uniform boxes.
3733 find_box:do j = lowest_box, ens_size - 1
3734 ! Find the box that this mass is in
3735 if(umass >= cumul_mass(j) .and. umass <= cumul_mass(j + 1)) then
3736
3737 if(rectangular_quadrature) then
3738 !********* Block for rectangular quadrature *******************
3739 ! Linearly interpolate in mass
3740 new_ens(i) = x(j) + ((umass - cumul_mass(j)) / &
3741 (cumul_mass(j+1) - cumul_mass(j))) * (x(j + 1) - x(j))
3742 !********* End block for rectangular quadrature *******************
3743
3744 else
3745
3746 !********* Block for trapezoidal interpolation *******************
3747 ! Assume that mass has linear profile, quadratic interpolation
3748 ! If two ensemble members are the same, just keep that value
3749 if(height(j + 1) < 0) then
3750 new_ens(i) = x(j)
3751 else
3752 ! Height on left side and right side
3753 hleft = height(j + 1) * like(j) / mass_sum
3754 hright = height(j + 1) * like(j + 1) / mass_sum
3755 ! Will solve a quadratic for desired x-x(j)
3756 ! a is 0.5(hright - hleft) / (x(j+1) - x(j))
3757 a = 0.5_r8 * (hright - hleft) / (x(j+1) - x(j))
3758 ! b is hleft
3759 b = hleft
3760 ! c is cumul_mass(j) - umass
3761 c = cumul_mass(j) - umass
3762 ! Use stable quadratic solver
3763 call solve_quadratic(a, b, c, r1, r2)
3764 adj_r1 = r1 + x(j)
3765 adj_r2 = r2 + x(j)
3766 if(adj_r1 >= x(j) .and. adj_r1 <= x(j+1)) then
3767 new_ens(i) = adj_r1
3768 elseif (adj_r2 >= x(j) .and. adj_r2 <= x(j+1)) then
3769 new_ens(i) = adj_r2
3770 else
3771 msgstring = 'Did not get a satisfactory quadratic root'
3772 call error_handler(e_err, 'obs_increment_rank_histogram', msgstring, &
3773 source)
3774 endif
3775 endif
3776 !********* End block for quadratic interpolation *******************
3777
3778 endif
3779
3780 ! Don't need to search lower boxes again
3781 lowest_box = j
3782 exit find_box
3783 end if
3784 end do find_box
3785 endif
3786end do
3787
3788! Convert to increments for unsorted
3789do i = 1, ens_size
3790 obs_inc(e_ind(i)) = new_ens(i) - x(i)
3791end do

3798subroutine update_ens_from_weights(ens, ens_size, rel_weight, ens_inc)
3799!------------------------------------------------------------------------
3800! Given relative weights for an ensemble, compute increments for the
3801! ensemble members. Assumes that prior distributon is equal uniform mass
3802! between each ensemble member. On the edges, have a normal with the
3803! sample mean and s.d. BUT normalized by a factor alpha so that only
3804! 1/(2*ens_size) of the total mass lies on each flank.
3805
3806integer, intent(in) :: ens_size
3807real(r8), intent(in) :: ens(ens_size), rel_weight(ens_size)
3808real(r8), intent(out) :: ens_inc(ens_size)
3809
3810integer :: i, j, lowest_box
3811integer :: e_ind(ens_size)
3812real(r8) :: x(1:2*ens_size - 1), cumul_mass(1:2*ens_size - 1), new_ens(ens_size)
3813real(r8) :: sort_inc(ens_size), updated_mass(2 * ens_size)
3814real(r8) :: sx, prior_mean, prior_var, prior_sd, mass
3815real(r8) :: total_mass_left, total_mass_right, alpha(2)
3816
3817! Initialize assim_tools_module if needed
3818if (.not. module_initialized) call assim_tools_init()
3819
3820call error_handler(e_err,'update_ens_from_weight','Routine needs testing.', &
3821 source, text2='Talk to Jeff before using.')
3822
3823! Do an index sort of the ensemble members
3824call index_sort(ens, e_ind, ens_size)
3825
3826! Have half boxes between all ensembles in the interior
3827! Total number of mass boxes is 2*ens_size
3828
3829! Compute the points that bound all the updated mass boxes; start with ensemble
3830do i = 1, ens_size
3831 x(2*i - 1) = ens(e_ind(i))
3832end do
3833! Compute the mid-point interior boundaries; these are halfway between ensembles
3834do i = 2, 2*ens_size - 2, 2
3835 x(i) = (x(i - 1) + x(i + 1)) / 2.0_r8
3836end do
3837
3838! Compute the mean and s.d. of the prior ensemble to handle wings
3839sx = sum(ens)
3840prior_mean = sx / ens_size
3841prior_var = sum((ens - prior_mean)**2) / (ens_size - 1)
3842prior_sd = sqrt(prior_var)
3843
3844! Need to normalize the wings so they have 1/(2*ens_size) mass outside
3845! Use cdf to find out how much mass is left of 1st member, right of last
3846total_mass_left = norm_cdf(ens(e_ind(1)), prior_mean, prior_sd)
3847total_mass_right = 1.0_r8 - norm_cdf(ens(e_ind(ens_size)), prior_mean, prior_sd)
3848
3849! Find the mass in each division given the initial equal partition and the weights
3850updated_mass(1) = rel_weight(e_ind(1)) / (2.0_r8 * ens_size)
3851updated_mass(2 * ens_size) = rel_weight(e_ind(ens_size)) / (2.0_r8 * ens_size)
3852do i = 2, 2*ens_size - 2, 2
3853 updated_mass(i) = rel_weight(e_ind(i / 2)) / (2.0_r8 * ens_size)
3854end do
3855do i = 3, 2*ens_size - 1, 2
3856 updated_mass(i) = rel_weight(e_ind((i+1) / 2)) / (2.0_r8 * ens_size)
3857end do
3858
3859! Normalize the mass; (COULD IT EVER BE 0 necessitating error check?)
3860updated_mass = updated_mass / sum(updated_mass)
3861
3862! Find a normalization factor to get tail mass right
3863if(total_mass_left > 0.0_r8) then
3864 alpha(1) = updated_mass(1) / total_mass_left
3865else
3866 alpha(1) = 0.0_r8
3867endif
3868if(total_mass_right > 0.0_r8) then
3869 alpha(2) = updated_mass(2 * ens_size) / total_mass_right
3870else
3871 alpha(2) = 0.0_r8
3872endif
3873
3874! Find cumulative mass at each box boundary and middle boundary
3875cumul_mass(1) = updated_mass(1)
3876do i = 2, 2*ens_size - 1
3877 cumul_mass(i) = cumul_mass(i - 1) + updated_mass(i)
3878end do
3879
3880! Get resampled position an inefficient way
3881! Need 1/ens_size between each EXCEPT for outers which get half of this
3882mass = 1.0_r8 / (2.0_r8 * ens_size)
3883
3884do i = 1, ens_size
3885 ! If it's in the inner or outer range have to use normal
3886 if(mass < cumul_mass(1)) then
3887 ! In the first normal box
3888 call weighted_norm_inv(alpha(1), prior_mean, prior_sd, mass, new_ens(i))
3889 else if(mass > cumul_mass(2*ens_size - 1)) then
3890 ! In the last normal box; Come in from the outside
3891 call weighted_norm_inv(alpha(2), prior_mean, prior_sd, 1.0_r8 - mass, new_ens(i))
3892 new_ens(i) = prior_mean + (prior_mean - new_ens(i))
3893 else
3894 ! In one of the inner uniform boxes. Make this much more efficient search?
3895 lowest_box = 1
3896 find_box:do j = lowest_box, 2 * ens_size - 2
3897 ! Find the box that this mass is in
3898 if(mass >= cumul_mass(j) .and. mass <= cumul_mass(j + 1)) then
3899 new_ens(i) = x(j) + ((mass - cumul_mass(j)) / (cumul_mass(j+1) - cumul_mass(j))) * &
3900 (x(j + 1) - x(j))
3901 ! Don't need to search lower boxes again
3902 lowest_box = j
3903 exit find_box
3904 end if
3905 end do find_box
3906 endif
3907 ! Want equally partitioned mass in update with exception that outermost boxes have half
3908 mass = mass + 1.0_r8 / ens_size
3909end do
3910
3911! Can now compute sorted increments
3912do i = 1, ens_size
3913 sort_inc(i) = new_ens(i) - ens(e_ind(i))
3914end do
3915
3916! Now, need to convert to increments for unsorted
3917do i = 1, ens_size
3918 ens_inc(e_ind(i)) = sort_inc(i)
3919end do
3920
3921end subroutine update_ens_from_weights

3926function norm_cdf(x_in, mean, sd)
3927
3928! Approximate cumulative distribution function for normal
3929! with mean and sd evaluated at point x_in
3930! Only works for x>= 0.
3931
3932real(r8) :: norm_cdf
3933real(r8), intent(in) :: x_in, mean, sd
3934
3935real(digits12) :: x, p, b1, b2, b3, b4, b5, t, density, nx
3936
3937! Convert to a standard normal
3938nx = (x_in - mean) / sd
3939
3940x = abs(nx)
3941
3942
3943! Use formula from Abramowitz and Stegun to approximate
3944p = 0.2316419_digits12
3945b1 = 0.319381530_digits12
3946b2 = -0.356563782_digits12
3947b3 = 1.781477937_digits12
3948b4 = -1.821255978_digits12
3949b5 = 1.330274429_digits12
3950
3951t = 1.0_digits12 / (1.0_digits12 + p * x)
3952
3953density = (1.0_digits12 / sqrt(2.0_digits12 * pi)) * exp(-x*x / 2.0_digits12)
3954
3955norm_cdf = 1.0_digits12 - density * &
3956 ((((b5 * t + b4) * t + b3) * t + b2) * t + b1) * t
3957
3958if(nx < 0.0_digits12) norm_cdf = 1.0_digits12 - norm_cdf
3959
3960!write(*, *) 'cdf is ', norm_cdf
3961
3962end function norm_cdf
