131!-------------------------------------------------------------------------------
132!> The forward operator for Total Water Storage.
133!> Get the soil moisture at several layers, add in the snow amounts ...
134
135subroutine get_expected_tws(state_handle, ens_size, location, &
136 velkey, expected_obs, istatus)
137
138type(ensemble_type), intent(in) :: state_handle
139integer, intent(in) :: ens_size
140type(location_type), intent(in) :: location
141type(location_type) :: location_soil
142integer, intent(in) :: velkey
143real(r8), intent(out) :: expected_obs(ens_size)
144integer, intent(out) :: istatus(ens_size)
145
146character(len=*), parameter :: routine = 'get_expected_TWS'
147
148real(r8):: obs_h2osno(ens_size)
149real(r8):: h2osoi_1(ens_size)
150real(r8):: h2osoi_2(ens_size)
151real(r8):: h2osoi_3(ens_size)
152real(r8):: h2osoi_4(ens_size)
153real(r8):: obs_h2osoi(ens_size)
154real(r8):: obs_wa(ens_size)
155
156integer :: istatus1(ens_size)
157integer :: istatus2(ens_size)
158integer :: istatus3(ens_size)
159logical :: return_now
160
161integer :: num
162real(r8):: loc(3)
163
164istatus = 0
165expected_obs = missing_r8
166
167if ( .not. module_initialized ) call initialize_module
168
169call error_handler(e_err,routine,'lots of work to do',source, revision, revdate)
170
171!>@todo Determine number of layers and their thicknesses automatically
172
173! Simple error check on key number before accessing the array
174!call velkey_out_of_range(velkey,'get_expected_TWS')
175
176loc = get_location(location)
177
178!-------------------------------------------------------------------------------
179! Get the soil moisture at 4 different levels
180
181loc(3) = 0.05_r8
182location_soil=set_location(loc(1),loc(2),loc(3),vertisheight)
183call interpolate(state_handle, ens_size, location_soil, qty_soil_moisture, &
184 h2osoi_1, istatus1)
185call track_status(ens_size, istatus1, h2osoi_1, istatus, return_now)
186if (return_now) return
187
188loc(3) = 0.25_r8
189location_soil=set_location(loc(1),loc(2),loc(3),vertisheight)
190call interpolate(state_handle, ens_size, location_soil, qty_soil_moisture, &
191 h2osoi_2, istatus1)
192call track_status(ens_size, istatus1, h2osoi_2, istatus, return_now)
193if (return_now) return
194
195loc(3) = 0.7_r8
196location_soil=set_location(loc(1),loc(2),loc(3),vertisheight)
197call interpolate(state_handle, ens_size, location_soil, qty_soil_moisture, &
198 h2osoi_3, istatus1)
199call track_status(ens_size, istatus1, h2osoi_3, istatus, return_now)
200if (return_now) return
201
202loc(3) = 1.5_r8
203location_soil=set_location(loc(1),loc(2),loc(3),vertisheight)
204call interpolate(state_handle, ens_size, location_soil, qty_soil_moisture, &
205 h2osoi_4, istatus1)
206call track_status(ens_size, istatus1, h2osoi_4, istatus, return_now)
207if (return_now) return
208
209!>@todo replace the 100, 300, 600, 1000 with the actual layer thicknesses.
210obs_h2osoi = h2osoi_1*100 + h2osoi_2*300 + h2osoi_3*600 + h2osoi_4*1000
211
212if (debug) then
213 write(string1,*)'soil moisture components:', &
214 h2osoi_1(1),h2osoi_2(1),h2osoi_3(1),h2osoi_4(1),obs_h2osoi(1)
215 call error_handler(e_msg, 'get_expected_TWS', string1, &
216 source, revision, revdate)
217endif
218
219! Get the amount of water in the snow
220
221call interpolate(state_handle, ens_size, location, qty_snow_water, &
222 obs_h2osno, istatus2)
223call track_status(ens_size, istatus2, obs_h2osno, istatus, return_now)
224if (return_now) return
225
226!-------------------------------------------------------------------------------
227! Get the amount of water in the aquifer ...
228!>@todo some models may not have aquifer ... should this be an error ...
229
230call interpolate(state_handle, ens_size, location, qty_aquifer_water, &
231 obs_wa, istatus3)
232call track_status(ens_size, istatus3, obs_wa, istatus, return_now)
233if (return_now) return
234
235!-------------------------------------------------------------------------------
236! Add up the water amounts from the different sources
237! warn if things dont make sense
238
239where (istatus == 0 ) expected_obs = obs_h2osno + obs_h2osoi + obs_wa
240
241do num = 1, ens_size
242 if (expected_obs(num) < 0.0_r8 .and. istatus(num) == 0) then
243
244 call write_location(0,location,charstring=string3)
245 write(string1,*)'TWS expected_obs for member ',num,' at ', &
246 trim(string3),' is unphysical ',expected_obs(num)
247 write(string2,*)'obs_h2osno, obs_h2osoi, obs_wa', &
248 obs_h2osno(num), obs_h2osoi(num), obs_wa(num)
249 write(string3,*)'istatus1, istatus2, istatus3', &
250 istatus1(num), istatus2(num), istatus3(num)
251 call error_handler(e_msg, 'get_expected_TWS', string1, &
252 source, revision, revdate, text2=string2, text3=string3)
253
254 expected_obs(num) = missing_r8
255 istatus = 1
256
257 endif
258enddo
259
260end subroutine get_expected_tws
